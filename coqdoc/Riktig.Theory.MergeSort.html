Lemma split_perm_bounded {A} (n : nat) :
  ∀ l : list A, length l ≤ n →
    Permutation l (fst (split l) ++ snd (split l)).
Proof.
  induction n as [|n IH]; intros l Hlen.
  - destruct l; simpl in *; [constructor | lia].
  - destruct l as [|x [|y rest]].
    + constructor.
    + apply Permutation_refl.
    + simpl in Hlen.
      assert (Hrest : Permutation rest (fst (split rest) ++ snd (split rest)))
        by (apply IH; lia).
      simpl. destruct (split rest) as [l1 l2] eqn:E.
      simpl in Hrest |- ×.
      apply perm_skip.
      apply perm_trans with (y :: l1 ++ l2).
      × apply perm_skip. exact Hrest.
      × apply Permutation_middle.
Qed.

Lemma split_perm {A} (l : list A) :
  Permutation l (fst (split l) ++ snd (split l)).
Proof. apply (split_perm_bounded (length l)); lia. Qed.


Lemma merge_perm : ∀ l1 l2 : list nat,
  Permutation (l1 ++ l2) (merge l1 l2).
Proof.
  intros l1 l2. unfold merge.
  apply perm_trans with (l2 ++ l1).
  - apply Permutation_app_comm.
  - apply (Riktig.Theory.Merge.merge_perm (O := Nat_DecTotalOrder)).
Qed.


Lemma merge_sorted : ∀ l1 l2 : list nat,
  Sorted le l1 → Sorted le l2 → Sorted le (merge l1 l2).
Proof.
  intros l1 l2 S1 S2. unfold merge.
  apply (Riktig.Theory.Merge.merge_sorted (O := Nat_DecTotalOrder)); assumption.
Qed.


Lemma msort_nil : msort [] = [].
Proof. rewrite msort_step. reflexivity. Qed.

Lemma msort_single (x : nat) : msort [x] = [x].
Proof. rewrite msort_step. reflexivity. Qed.

Lemma msort_cons2 (x y : nat) (rest : list nat) :
  msort (x :: y :: rest) =
    merge (msort (fst (split (x :: y :: rest))))
          (msort (snd (split (x :: y :: rest)))).
Proof. rewrite msort_step. reflexivity. Qed.


Theorem msort_perm : ∀ l : list nat, Permutation l (msort l).
Proof.
  induction l as [l IH]
    using (well_founded_ind (well_founded_ltof (list nat) (@length nat))).
  destruct l as [|x [|y rest]].
  - rewrite msort_nil. constructor.
  - rewrite msort_single. apply Permutation_refl.
  - pose proof (split_length_lt x y rest) as [H1 H2].
    rewrite msort_cons2.
    apply perm_trans
      with (fst (split (x :: y :: rest)) ++ snd (split (x :: y :: rest))).
    + apply split_perm.
    + apply perm_trans
        with (msort (fst (split (x :: y :: rest))) ++
              msort (snd (split (x :: y :: rest)))).
      × apply Permutation_app; apply IH; [exact H1 | exact H2].
      × apply merge_perm.
Qed.

Theorem msort_sorted : ∀ l : list nat, Sorted le (msort l).
Proof.
  induction l as [l IH]
    using (well_founded_ind (well_founded_ltof (list nat) (@length nat))).
  destruct l as [|x [|y rest]].
  - rewrite msort_nil. constructor.
  - rewrite msort_single. constructor; constructor.
  - pose proof (split_length_lt x y rest) as [H1 H2].
    rewrite msort_cons2.
    apply merge_sorted; apply IH; [exact H1 | exact H2].
Qed.

Corollary msort_correct : ∀ l : list nat,
  Permutation l (msort l) ∧ Sorted le (msort l).
Proof. intros l. split; [apply msort_perm | apply msort_sorted]. Qed.
