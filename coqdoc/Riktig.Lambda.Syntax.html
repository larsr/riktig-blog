Library Riktig.Lambda.Syntax


Inductive tm : Type :=
  | Var : nat → tm
  | Abs : tm → tm
  | App : tm → tm → tm.

Fixpoint lift (d k : nat) (t : tm) : tm :=
  match t with
  | Var n ⇒ Var (if Nat.leb k n then d + n else n)
  | Abs b ⇒ Abs (lift d (S k) b)
  | App f a ⇒ App (lift d k f) (lift d k a)
  end.

Fixpoint subst (k : nat) (u : tm) (t : tm) : tm :=
  match t with
  | Var n ⇒
      if Nat.ltb n k then Var n
      else if Nat.eqb n k then lift k 0 u
      else Var (pred n)
  | Abs b ⇒ Abs (subst (S k) u b)
  | App f a ⇒ App (subst k u f) (subst k u a)
  end.


From Coq Require Export PeanoNat.

Lemma lift_0 : ∀ t k, lift 0 k t = t.
Proof.
  induction t as [n | b IH | f IHf a IHa]; intros k; simpl.
  - destruct (Nat.leb k n); reflexivity.
  - rewrite IH. reflexivity.
  - rewrite IHf, IHa. reflexivity.
Qed.

Lemma subst_lift_le : ∀ t u d j k,
  j ≤ k → k ≤ j + d → subst k u (lift (S d) j t) = lift d j t.
Proof.
  induction t as [n | b IH | f IHf a IHa]; intros u d j k Hj Hk;
    cbn [lift subst].
  - destruct (Nat.leb j n) eqn:E; cbn [lift subst].
    + apply Nat.leb_le in E.
      assert (Hk' : k ≤ d + n).
      { eapply Nat.le_trans; [exact Hk|].
        rewrite (Nat.add_comm j d). apply Nat.add_le_mono_l. exact E. }
      assert (E1 : (S d + n <? k) = false).
      { apply Nat.ltb_ge. apply Nat.le_le_succ_r. exact Hk'. }
      assert (E2 : (S d + n =? k) = false).
      { apply Nat.eqb_neq. intros Ek.
        apply (Nat.nle_succ_diag_l (d + n)).
        rewrite <- Ek in Hk'. exact Hk'. }
      rewrite E1, E2. reflexivity.
    + apply Nat.leb_gt in E.
      assert (E1 : (n <? k) = true).
      { apply Nat.ltb_lt. eapply Nat.lt_le_trans; [exact E | exact Hj]. }
      rewrite E1. reflexivity.
  - rewrite IH.
    + reflexivity.
    + apply le_n_S. exact Hj.
    + apply le_n_S. exact Hk.
  - rewrite IHf; [| exact Hj | exact Hk].
    rewrite IHa; [| exact Hj | exact Hk].
    reflexivity.
Qed.

Lemma subst_lift : ∀ t u k, subst k u (lift 1 k t) = t.
Proof.
  intros t u k.
  rewrite (subst_lift_le t u 0 k k).
  - apply lift_0.
  - apply Nat.le_refl.
  - apply Nat.le_add_r.
Qed.

Lemma subst_lift_lower : ∀ t u, subst 1 u (lift 2 0 t) = lift 1 0 t.
Proof.
  intros t u.
  apply (subst_lift_le t u 1 0 1).
  - apply Nat.le_0_l.
  - apply Nat.le_refl.
Qed.


Inductive value : tm → Prop :=
  | value_var : ∀ n, value (Var n)
  | value_abs : ∀ b, value (Abs b).

Definition is_value (t : tm) : bool :=
  match t with
  | App _ _ ⇒ false
  | _ ⇒ true
  end.

Lemma is_value_value : ∀ t, is_value t = true → value t.
Proof.
  intros t H. destruct t.
  - constructor.
  - constructor.
  - discriminate H.
Qed.

Lemma value_is_value : ∀ t, value t → is_value t = true.
Proof.
  intros t H. destruct H; reflexivity.
Qed.


Inductive step : tm → tm → Prop :=
  | step_beta : ∀ b v,
      value v →
      step (App (Abs b) v) (subst 0 v b)
  | step_app1 : ∀ f f' a,
      step f f' →
      step (App f a) (App f' a)
  | step_app2 : ∀ v a a',
      value v →
      step a a' →
      step (App v a) (App v a').

Lemma step_not_value : ∀ t t', step t t' → is_value t = false.
Proof.
  intros t t' H. destruct H; reflexivity.
Qed.

Inductive star : tm → tm → Prop :=
  | star_refl : ∀ t, star t t
  | star_step : ∀ t u v, step t u → star u v → star t v.


Fixpoint step_fn (t : tm) : option tm :=
  match t with
  | App f a ⇒
      if is_value f then
        if is_value a then
          match f with
          | Abs b ⇒ Some (subst 0 a b)
          | _ ⇒ None
          end
        else option_map (App f) (step_fn a)
      else option_map (fun f' ⇒ App f' a) (step_fn f)
  | _ ⇒ None
  end.

Lemma step_fn_sound : ∀ t t', step_fn t = Some t' → step t t'.
Proof.
  induction t as [n | b _ | f IHf a IHa]; intros t' H; simpl in H.
  - discriminate.
  - discriminate.
  - destruct (is_value f) eqn:Hf.
    + destruct (is_value a) eqn:Ha.
      × destruct f; try discriminate.
        injection H as <-. constructor. apply is_value_value; exact Ha.
      × destruct (step_fn a) as [a'|]; [|discriminate].
        injection H as <-. constructor.
        { apply is_value_value; exact Hf. }
        { apply IHa; reflexivity. }
    + destruct (step_fn f) as [f'|]; [|discriminate].
      injection H as <-. constructor. apply IHf; reflexivity.
Qed.

Lemma step_fn_complete : ∀ t t', step t t' → step_fn t = Some t'.
Proof.
  intros t t' H. induction H; simpl.
  - rewrite (value_is_value _ H). reflexivity.
  - rewrite (step_not_value _ _ H). rewrite IHstep. reflexivity.
  - rewrite (value_is_value _ H).
    rewrite (step_not_value _ _ H0). rewrite IHstep. reflexivity.
Qed.


Fixpoint eval (fuel : nat) (t : tm) : tm :=
  match fuel with
  | O ⇒ t
  | S fuel' ⇒
      match step_fn t with
      | Some t' ⇒ eval fuel' t'
      | None ⇒ t
      end
  end.

Lemma eval_sound : ∀ fuel t, star t (eval fuel t).
Proof.
  induction fuel as [| fuel IH]; intros t; simpl.
  - constructor.
  - destruct (step_fn t) as [t'|] eqn:H.
    + econstructor.
      { apply step_fn_sound; exact H. }
      { apply IH. }
    + constructor.
Qed.


Definition idt : tm := Abs (Var 0). Definition k : tm := Abs (Abs (Var 1)).
Example eval_id_id : eval 10 (App idt idt) = idt.
Proof. reflexivity. Qed.

Example eval_k : eval 10 (App (App k idt) (Abs (Abs (Var 0)))) = idt.
Proof. reflexivity. Qed.

Example subst_avoids_capture :
  subst 0 (Abs (App (Var 0) (Var 1))) (Abs (Var 1))
  = Abs (Abs (App (Var 0) (Var 2))).
Proof. reflexivity. Qed.
