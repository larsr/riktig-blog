Section FinVecInner.
  Context {F : Type} `{Field F}.

  Definition dot {n : nat} (u v : Fin.t n → F) : F :=
    bigopF (fun i ⇒ u i × v i).

  #[global] Instance FinVec_Inner {n : nat} : HasInner F (Fin.t n → F) :=
    @dot n.

  Lemma dot_def : ∀ n (u v : Fin.t n → F),
    dot u v == bigopF (fun i ⇒ u i × v i).
  Proof.
    intros n u v. reflexivity.
  Qed.

  Lemma dot_proper (n : nat) :
    Proper (equ ==> equ ==> equ) (@inner F (Fin.t n → F) _).
  Proof.
    intros u u' Hu v v' Hv.
    exact (bigopF_ext n (fun i ⇒ u i × v i) (fun i ⇒ u' i × v' i)
             (fun i ⇒ ring_mul_proper (u i) (u' i) (Hu i)
                         (v i) (v' i) (Hv i))).
  Qed.

  Lemma dot_sym (n : nat) : ∀ u v : Fin.t n → F,
    dot u v == dot v u.
  Proof.
    intros u v.
    exact (bigopF_ext n (fun i ⇒ u i × v i) (fun i ⇒ v i × u i)
             (fun i ⇒ field_mul_comm (u i) (v i))).
  Qed.

  Lemma dot_add_l (n : nat) : ∀ u v w : Fin.t n → F,
    dot (u + v) w == dot u w + dot v w.
  Proof.
    intros u v w.
    unfold dot.
    calc (bigopF (fun i ⇒ (u + v) i × w i))
      == (bigopF (fun i ⇒ u i × w i + v i × w i))
         { (bigopF_ext n (fun i ⇒ (u + v) i × w i)
              (fun i ⇒ u i × w i + v i × w i)
              (fun i ⇒ distr (u i) (v i) (w i))) }
      == (bigopF (fun i ⇒ u i × w i) + bigopF (fun i ⇒ v i × w i))
         { (bigopF_add n (fun i ⇒ u i × w i) (fun i ⇒ v i × w i)) }.
  Qed.

  Lemma dot_scale_l (n : nat) : ∀ (a : F) (u v : Fin.t n → F),
    dot (a *: u) v == a × dot u v.
  Proof.
    intros a u v.
    unfold dot.
    calc (bigopF (fun i ⇒ (a *: u) i × v i))
      == (bigopF (fun i ⇒ a × (u i × v i)))
         { (bigopF_ext n (fun i ⇒ (a *: u) i × v i)
              (fun i ⇒ a × (u i × v i))
              (fun i ⇒ assoc a (u i) (v i))) }
      == (a × bigopF (fun i ⇒ u i × v i))
         { (bigopF_mul_l n a (fun i ⇒ u i × v i)) }.
  Qed.

  #[global] Instance FinVec_InnerProduct (n : nat) :
    InnerProduct F (Fin.t n → F) :=
    @Build_InnerProduct F (Fin.t n → F) _
      (@PW_VectorSpace (Fin.t n) F F _ Field_VectorSpace)
      (@FinVec_Inner n)
      (dot_proper n) (dot_sym n) (dot_add_l n) (dot_scale_l n).

End FinVecInner.
