Lemma eq_to_equ {T : Type} {e : EqRel T} {Eqv : Equivalence e} (a b : T) :
  a = b → a == b.
Proof. intros →. reflexivity. Qed.


Section EvalLM.
  Context {F : Type} {FF : Field F} {n : nat} (j : Fin.t n).

  Lemma eval_lm_proper : Proper (equ ==> equ) (fun f : Fin.t n → F ⇒ f j).
  Proof. intros f g Hfg. exact (Hfg j). Qed.

  Lemma eval_lm_add : ∀ f g : Fin.t n → F, (f + g) j == f j + g j.
  Proof. intros f g. reflexivity. Qed.

  Lemma eval_lm_scale : ∀ (a : F) (f : Fin.t n → F),
    (a *: f) j == a *: f j.
  Proof. intros a f. reflexivity. Qed.

  Definition eval_lm : LinearMap F (Fin.t n → F) F :=
    @Build_LinearMap F (Fin.t n → F) F FF
      (@PW_VectorSpace (Fin.t n) F F FF Field_VectorSpace) Field_VectorSpace
      (fun f ⇒ f j) eval_lm_proper eval_lm_add eval_lm_scale.

End EvalLM.


Section BasisTransport.
  Context {F W W' : Type} {FF : Field F}
          {VW : VectorSpace F W} {VW' : VectorSpace F W'}.
  Context (T : LinearMap F W W') (Tinv : LinearMap F W' W).
  Context (Hlr : ∀ w : W, Tinv (T w) == w)
          (Hrl : ∀ w' : W', T (Tinv w') == w').
  Context {k : nat} (Bk : Basis k F W).

  Definition bt_vec (i : Fin.t k) : W' := T (basis_vec Bk i).

  Definition bt_coords : LinearMap F W' (Fin.t k → F) :=
    lm_comp (basis_coords Bk) Tinv.

  Lemma bt_recon : ∀ w' : W', lincomb bt_vec (bt_coords w') == w'.
  Proof.
    intros w'.
    unfold bt_coords, lm_comp; simpl.
    calc (lincomb bt_vec (basis_coords Bk (Tinv w')))
      == (T (lincomb (basis_vec Bk) (basis_coords Bk (Tinv w'))))
      == (T (Tinv w'))
      == w'.
    - symmetry.
      exact (lm_lincomb T k (basis_vec Bk) (basis_coords Bk (Tinv w'))).
    - apply (lm_proper T).
      exact (basis_recon Bk (Tinv w')).
    - exact (Hrl w').
  Qed.

  Lemma bt_coords_lincomb : ∀ c : Fin.t k → F,
    bt_coords (lincomb bt_vec c) == c.
  Proof.
    intros c.
    unfold bt_coords, lm_comp; simpl.
    calc (basis_coords Bk (Tinv (lincomb bt_vec c)))
      == (basis_coords Bk (Tinv (T (lincomb (basis_vec Bk) c))))
      == (basis_coords Bk (lincomb (basis_vec Bk) c))
      == c.
    - apply (lm_proper (basis_coords Bk)), (lm_proper Tinv).
      symmetry.
      exact (lm_lincomb T k (basis_vec Bk) c).
    - apply (lm_proper (basis_coords Bk)).
      exact (Hlr (lincomb (basis_vec Bk) c)).
    - exact (basis_coords_lincomb Bk c).
  Qed.

  Definition basis_transport : Basis k F W' :=
    @Build_Basis k F W' FF VW' bt_vec bt_coords bt_recon bt_coords_lincomb.

End BasisTransport.


Section AndSubspace.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V}.
  Context (P1 P2 : V → Prop) {SP1 : SubspacePred F V P1}
          {SP2 : SubspacePred F V P2}.

  Definition Pand (v : V) : Prop := P1 v ∧ P2 v.

  Lemma and_zero : Pand 0.
  Proof. exact (conj sp_zero sp_zero). Qed.

  Lemma and_add : ∀ u v : V, Pand u → Pand v → Pand (u + v).
  Proof.
    intros u v [Hu1 Hu2] [Hv1 Hv2].
    exact (conj (sp_add u v Hu1 Hv1) (sp_add u v Hu2 Hv2)).
  Qed.

  Lemma and_scale : ∀ (a : F) (u : V), Pand u → Pand (a *: u).
  Proof.
    intros a u [Hu1 Hu2].
    exact (conj (sp_scale a u Hu1) (sp_scale a u Hu2)).
  Qed.

  #[global] Instance SubspacePred_and : SubspacePred F V Pand := {|
    sp_zero := and_zero;
    sp_add := and_add;
    sp_scale := and_scale
  |}.

End AndSubspace.


Section AndComm.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V}.
  Context (P1 P2 : V → Prop) {SP1 : SubspacePred F V P1}
          {SP2 : SubspacePred F V P2}.

  Definition andcomm_to (x : Sub V (Pand P1 P2)) : Sub V (Pand P2 P1) :=
    exist (Pand P2 P1) (proj1_sig x) (conj (proj2 (proj2_sig x)) (proj1 (proj2_sig x))).

  Definition andcomm_from (x : Sub V (Pand P2 P1)) : Sub V (Pand P1 P2) :=
    exist (Pand P1 P2) (proj1_sig x) (conj (proj2 (proj2_sig x)) (proj1 (proj2_sig x))).

  Lemma andcomm_to_proper : Proper (equ ==> equ) andcomm_to.
  Proof. intros x y Hxy. exact Hxy. Qed.

  Lemma andcomm_to_add : ∀ x y : Sub V (Pand P1 P2),
    andcomm_to (x + y) == andcomm_to x + andcomm_to y.
  Proof.
    intros x y.
    change (proj1_sig (andcomm_to (x + y)) == proj1_sig (andcomm_to x + andcomm_to y)).
    reflexivity.
  Qed.

  Lemma andcomm_to_scale : ∀ (a : F) (x : Sub V (Pand P1 P2)),
    andcomm_to (a *: x) == a *: andcomm_to x.
  Proof.
    intros a x.
    change (proj1_sig (andcomm_to (a *: x)) == proj1_sig (a *: andcomm_to x)).
    reflexivity.
  Qed.

  Definition andcomm_to_lm : LinearMap F (Sub V (Pand P1 P2)) (Sub V (Pand P2 P1)) :=
    @Build_LinearMap F (Sub V (Pand P1 P2)) (Sub V (Pand P2 P1)) FF
      (Sub_VectorSpace (Pand P1 P2)) (Sub_VectorSpace (Pand P2 P1))
      andcomm_to andcomm_to_proper andcomm_to_add andcomm_to_scale.

  Lemma andcomm_from_proper : Proper (equ ==> equ) andcomm_from.
  Proof. intros x y Hxy. exact Hxy. Qed.

  Lemma andcomm_from_add : ∀ x y : Sub V (Pand P2 P1),
    andcomm_from (x + y) == andcomm_from x + andcomm_from y.
  Proof.
    intros x y.
    change (proj1_sig (andcomm_from (x + y)) == proj1_sig (andcomm_from x + andcomm_from y)).
    reflexivity.
  Qed.

  Lemma andcomm_from_scale : ∀ (a : F) (x : Sub V (Pand P2 P1)),
    andcomm_from (a *: x) == a *: andcomm_from x.
  Proof.
    intros a x.
    change (proj1_sig (andcomm_from (a *: x)) == proj1_sig (a *: andcomm_from x)).
    reflexivity.
  Qed.

  Definition andcomm_from_lm : LinearMap F (Sub V (Pand P2 P1)) (Sub V (Pand P1 P2)) :=
    @Build_LinearMap F (Sub V (Pand P2 P1)) (Sub V (Pand P1 P2)) FF
      (Sub_VectorSpace (Pand P2 P1)) (Sub_VectorSpace (Pand P1 P2))
      andcomm_from andcomm_from_proper andcomm_from_add andcomm_from_scale.

  Lemma andcomm_lr : ∀ x : Sub V (Pand P1 P2), andcomm_from (andcomm_to x) == x.
  Proof.
    intros x.
    change (proj1_sig (andcomm_from (andcomm_to x)) == proj1_sig x).
    reflexivity.
  Qed.

  Lemma andcomm_rl : ∀ x : Sub V (Pand P2 P1), andcomm_to (andcomm_from x) == x.
  Proof.
    intros x.
    change (proj1_sig (andcomm_to (andcomm_from x)) == proj1_sig x).
    reflexivity.
  Qed.

  Definition basis_and_comm (k : nat) (Bk : Basis k F (Sub V (Pand P1 P2))) :
    Basis k F (Sub V (Pand P2 P1)) :=
    basis_transport andcomm_to_lm andcomm_from_lm andcomm_lr andcomm_rl Bk.

End AndComm.


Section DoubleSub.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V}.
  Context (P1 P2 : V → Prop) {SP1 : SubspacePred F V P1}
          {SP2 : SubspacePred F V P2}.

  Definition Pinner (x : Sub V P1) : Prop := P2 (proj1_sig x).

  Lemma pinner_zero : Pinner (0 : Sub V P1).
  Proof. exact (@sp_zero F V _ _ P2 SP2). Qed.

  Lemma pinner_add : ∀ x y : Sub V P1,
    Pinner x → Pinner y → Pinner (x + y).
  Proof.
    intros x y Hx Hy.
    exact (@sp_add F V _ _ P2 SP2 (proj1_sig x) (proj1_sig y) Hx Hy).
  Qed.

  Lemma pinner_scale : ∀ (a : F) (x : Sub V P1),
    Pinner x → Pinner (a *: x).
  Proof.
    intros a x Hx.
    exact (@sp_scale F V _ _ P2 SP2 a (proj1_sig x) Hx).
  Qed.

  #[global] Instance SubspacePred_pullback : SubspacePred F (Sub V P1) Pinner := {|
    sp_zero := pinner_zero;
    sp_add := pinner_add;
    sp_scale := pinner_scale
  |}.

  Definition ds_to (x : Sub (Sub V P1) Pinner) : Sub V (Pand P1 P2) :=
    exist (Pand P1 P2) (proj1_sig (proj1_sig x))
      (conj (proj2_sig (proj1_sig x)) (proj2_sig x)).

  Definition ds_from (v : Sub V (Pand P1 P2)) : Sub (Sub V P1) Pinner :=
    exist Pinner
      (exist P1 (proj1_sig v) (proj1 (proj2_sig v)))
      (proj2 (proj2_sig v)).

  Lemma ds_to_proper : Proper (equ ==> equ) ds_to.
  Proof. intros x y Hxy. exact Hxy. Qed.

  Lemma ds_to_add : ∀ x y : Sub (Sub V P1) Pinner,
    ds_to (x + y) == ds_to x + ds_to y.
  Proof.
    intros x y.
    change (proj1_sig (ds_to (x + y)) == proj1_sig (ds_to x + ds_to y)).
    reflexivity.
  Qed.

  Lemma ds_to_scale : ∀ (a : F) (x : Sub (Sub V P1) Pinner),
    ds_to (a *: x) == a *: ds_to x.
  Proof.
    intros a x.
    change (proj1_sig (ds_to (a *: x)) == proj1_sig (a *: ds_to x)).
    reflexivity.
  Qed.

  Definition ds_to_lm : LinearMap F (Sub (Sub V P1) Pinner) (Sub V (Pand P1 P2)) :=
    @Build_LinearMap F (Sub (Sub V P1) Pinner) (Sub V (Pand P1 P2)) FF
      (Sub_VectorSpace Pinner) (Sub_VectorSpace (Pand P1 P2))
      ds_to ds_to_proper ds_to_add ds_to_scale.

  Lemma ds_from_proper : Proper (equ ==> equ) ds_from.
  Proof. intros x y Hxy. exact Hxy. Qed.

  Lemma ds_from_add : ∀ x y : Sub V (Pand P1 P2),
    ds_from (x + y) == ds_from x + ds_from y.
  Proof.
    intros x y.
    change (proj1_sig (proj1_sig (ds_from (x + y)))
              == proj1_sig (proj1_sig (ds_from x + ds_from y))).
    reflexivity.
  Qed.

  Lemma ds_from_scale : ∀ (a : F) (x : Sub V (Pand P1 P2)),
    ds_from (a *: x) == a *: ds_from x.
  Proof.
    intros a x.
    change (proj1_sig (proj1_sig (ds_from (a *: x)))
              == proj1_sig (proj1_sig (a *: ds_from x))).
    reflexivity.
  Qed.

  Definition ds_from_lm : LinearMap F (Sub V (Pand P1 P2)) (Sub (Sub V P1) Pinner) :=
    @Build_LinearMap F (Sub V (Pand P1 P2)) (Sub (Sub V P1) Pinner) FF
      (Sub_VectorSpace (Pand P1 P2)) (Sub_VectorSpace Pinner)
      ds_from ds_from_proper ds_from_add ds_from_scale.

  Lemma ds_lr : ∀ x : Sub (Sub V P1) Pinner, ds_from (ds_to x) == x.
  Proof.
    intros x.
    change (proj1_sig (proj1_sig (ds_from (ds_to x))) == proj1_sig (proj1_sig x)).
    reflexivity.
  Qed.

  Lemma ds_rl : ∀ v : Sub V (Pand P1 P2), ds_to (ds_from v) == v.
  Proof.
    intros v.
    change (proj1_sig (ds_to (ds_from v)) == proj1_sig v).
    reflexivity.
  Qed.

  Definition basis_and_collapse (k : nat) (Bk : Basis k F (Sub (Sub V P1) Pinner)) :
    Basis k F (Sub V (Pand P1 P2)) :=
    basis_transport ds_to_lm ds_from_lm ds_lr ds_rl Bk.

  Definition basis_and_expand (k : nat) (Bk : Basis k F (Sub V (Pand P1 P2))) :
    Basis k F (Sub (Sub V P1) Pinner) :=
    basis_transport ds_from_lm ds_to_lm ds_rl ds_lr Bk.

End DoubleSub.


Section IffSubspace.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V}.
  Context (P Q : V → Prop) {SP : SubspacePred F V P} {SQ : SubspacePred F V Q}.
  Context (Hiff : ∀ v : V, P v ↔ Q v).

  Definition iff_to (x : Sub V P) : Sub V Q :=
    exist Q (proj1_sig x) (proj1 (Hiff (proj1_sig x)) (proj2_sig x)).

  Definition iff_from (x : Sub V Q) : Sub V P :=
    exist P (proj1_sig x) (proj2 (Hiff (proj1_sig x)) (proj2_sig x)).

  Lemma iff_to_proper : Proper (equ ==> equ) iff_to.
  Proof. intros x y Hxy. exact Hxy. Qed.

  Lemma iff_to_add : ∀ x y : Sub V P, iff_to (x + y) == iff_to x + iff_to y.
  Proof.
    intros x y.
    change (proj1_sig (iff_to (x + y)) == proj1_sig (iff_to x + iff_to y)).
    reflexivity.
  Qed.

  Lemma iff_to_scale : ∀ (a : F) (x : Sub V P), iff_to (a *: x) == a *: iff_to x.
  Proof.
    intros a x.
    change (proj1_sig (iff_to (a *: x)) == proj1_sig (a *: iff_to x)).
    reflexivity.
  Qed.

  Definition iff_to_lm : LinearMap F (Sub V P) (Sub V Q) :=
    @Build_LinearMap F (Sub V P) (Sub V Q) FF
      (Sub_VectorSpace P) (Sub_VectorSpace Q)
      iff_to iff_to_proper iff_to_add iff_to_scale.

  Lemma iff_from_proper : Proper (equ ==> equ) iff_from.
  Proof. intros x y Hxy. exact Hxy. Qed.

  Lemma iff_from_add : ∀ x y : Sub V Q, iff_from (x + y) == iff_from x + iff_from y.
  Proof.
    intros x y.
    change (proj1_sig (iff_from (x + y)) == proj1_sig (iff_from x + iff_from y)).
    reflexivity.
  Qed.

  Lemma iff_from_scale : ∀ (a : F) (x : Sub V Q), iff_from (a *: x) == a *: iff_from x.
  Proof.
    intros a x.
    change (proj1_sig (iff_from (a *: x)) == proj1_sig (a *: iff_from x)).
    reflexivity.
  Qed.

  Definition iff_from_lm : LinearMap F (Sub V Q) (Sub V P) :=
    @Build_LinearMap F (Sub V Q) (Sub V P) FF
      (Sub_VectorSpace Q) (Sub_VectorSpace P)
      iff_from iff_from_proper iff_from_add iff_from_scale.

  Lemma iff_lr : ∀ x : Sub V P, iff_from (iff_to x) == x.
  Proof.
    intros x.
    change (proj1_sig (iff_from (iff_to x)) == proj1_sig x).
    reflexivity.
  Qed.

  Lemma iff_rl : ∀ x : Sub V Q, iff_to (iff_from x) == x.
  Proof.
    intros x.
    change (proj1_sig (iff_to (iff_from x)) == proj1_sig x).
    reflexivity.
  Qed.

  Definition basis_pred_iff (k : nat) (Bk : Basis k F (Sub V P)) :
    Basis k F (Sub V Q) :=
    basis_transport iff_to_lm iff_from_lm iff_lr iff_rl Bk.

End IffSubspace.


Section DropLM.
  Context {F : Type} {FF : Field F} {m : nat}.

  Lemma drop_lm_proper :
    Proper (equ ==> equ) (fun f : Fin.t (S m) → F ⇒ fun i : Fin.t m ⇒ f (Fin.FS i)).
  Proof. intros f g Hfg i. exact (Hfg (Fin.FS i)). Qed.

  Lemma drop_lm_add : ∀ f g : Fin.t (S m) → F,
    (fun i : Fin.t m ⇒ (f + g) (Fin.FS i))
      == (fun i ⇒ f (Fin.FS i)) + (fun i ⇒ g (Fin.FS i)).
  Proof. intros f g i. reflexivity. Qed.

  Lemma drop_lm_scale : ∀ (a : F) (f : Fin.t (S m) → F),
    (fun i : Fin.t m ⇒ (a *: f) (Fin.FS i)) == a *: (fun i ⇒ f (Fin.FS i)).
  Proof. intros a f i. reflexivity. Qed.

  Definition drop_lm : LinearMap F (Fin.t (S m) → F) (Fin.t m → F) :=
    @Build_LinearMap F (Fin.t (S m) → F) (Fin.t m → F) FF
      (@PW_VectorSpace (Fin.t (S m)) F F FF Field_VectorSpace)
      (@PW_VectorSpace (Fin.t m) F F FF Field_VectorSpace)
      (fun f i ⇒ f (Fin.FS i)) drop_lm_proper drop_lm_add drop_lm_scale.

End DropLM.


Section ConsCoord.
  Context {F : Type} {m : nat}.

  Definition cons_c (c0 : F) (c : Fin.t m → F) : Fin.t (S m) → F :=
    fun j ⇒ Fin.caseS' j (fun _ ⇒ F) c0 c.

  Lemma cons_c_F1 : ∀ (c0 : F) (c : Fin.t m → F), cons_c c0 c Fin.F1 = c0.
  Proof. intros c0 c. reflexivity. Qed.

  Lemma cons_c_FS : ∀ (c0 : F) (c : Fin.t m → F) (i : Fin.t m),
    cons_c c0 c (Fin.FS i) = c i.
  Proof. intros c0 c i. reflexivity. Qed.

End ConsCoord.


Section Hyperplane.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V}.
  Context {m : nat} (Bn : Basis (S m) F V).

  Definition c1 : LinearMap F V F := lm_comp (eval_lm Fin.F1) (basis_coords Bn).

  Definition H (v : V) : Prop := ker_pred c1 v.

  #[global] Instance H_SubspacePred : SubspacePred F V H := Ker_SubspacePred c1.

  Definition hp_vec (i : Fin.t m) : V := basis_vec Bn (Fin.FS i).

  Lemma hp_vec_in_H : ∀ i : Fin.t m, H (hp_vec i).
  Proof.
    intros i.
    unfold H, ker_pred, c1, hp_vec, lm_comp; simpl.
    transitivity (e (Fin.FS i) Fin.F1).
    - exact (coords_vec Bn (Fin.FS i) Fin.F1).
    - exact (e_at_neq (S m) (Fin.FS i) Fin.F1 (fin_FS_neq_F1 m i)).
  Qed.

  Definition hp_basis_vec (i : Fin.t m) : Sub V H :=
    exist H (hp_vec i) (hp_vec_in_H i).

  Definition hp_coords_fun (x : Sub V H) : Fin.t m → F :=
    drop_lm (basis_coords Bn (proj1_sig x)).

  Lemma hp_coords_proper : Proper (equ ==> equ) hp_coords_fun.
  Proof.
    intros x y Hxy i.
    unfold hp_coords_fun.
    apply (lm_proper drop_lm).
    apply (lm_proper (basis_coords Bn)).
    exact Hxy.
  Qed.

  Lemma hp_coords_add : ∀ x y : Sub V H,
    hp_coords_fun (x + y) == hp_coords_fun x + hp_coords_fun y.
  Proof.
    intros x y i.
    unfold hp_coords_fun, drop_lm; simpl.
    exact (lm_add (basis_coords Bn) (proj1_sig x) (proj1_sig y) (Fin.FS i)).
  Qed.

  Lemma hp_coords_scale : ∀ (a : F) (x : Sub V H),
    hp_coords_fun (a *: x) == a *: hp_coords_fun x.
  Proof.
    intros a x i.
    unfold hp_coords_fun, drop_lm; simpl.
    exact (lm_scale (basis_coords Bn) a (proj1_sig x) (Fin.FS i)).
  Qed.

  Definition hp_coords : LinearMap F (Sub V H) (Fin.t m → F) :=
    @Build_LinearMap F (Sub V H) (Fin.t m → F) FF (Sub_VectorSpace H)
      (@PW_VectorSpace (Fin.t m) F F FF Field_VectorSpace)
      hp_coords_fun hp_coords_proper hp_coords_add hp_coords_scale.

  Lemma lincomb_cons_drop : ∀ (c : Fin.t m → F),
    lincomb (basis_vec Bn) (cons_c 0 c) == lincomb hp_vec c.
  Proof.
    intros c.
    unfold lincomb.
    calc (bigopF (fun j ⇒ cons_c 0 c j *: basis_vec Bn j))
      == (cons_c 0 c Fin.F1 *: basis_vec Bn Fin.F1
          + bigopF (fun i ⇒ cons_c 0 c (Fin.FS i) *: basis_vec Bn (Fin.FS i)))
      == ((0 : F) *: basis_vec Bn Fin.F1
          + bigopF (fun i ⇒ cons_c 0 c (Fin.FS i) *: basis_vec Bn (Fin.FS i)))
      == ((0 : V)
          + bigopF (fun i ⇒ cons_c 0 c (Fin.FS i) *: basis_vec Bn (Fin.FS i)))
      == (bigopF (fun i ⇒ cons_c 0 c (Fin.FS i) *: basis_vec Bn (Fin.FS i)))
      == (bigopF (fun i ⇒ c i *: hp_vec i)).
    - exact (bigopF_S m (fun j ⇒ cons_c 0 c j *: basis_vec Bn j)).
    - rewrite (cons_c_F1 0 c). reflexivity.
    - apply mon_add_proper;
        [exact (zero_scale (basis_vec Bn Fin.F1)) | reflexivity].
    - exact (mon_idl (bigopF (fun i ⇒ cons_c 0 c (Fin.FS i) *: basis_vec Bn (Fin.FS i)))).
    - apply bigopF_ext. intros i.
      rewrite (cons_c_FS 0 c i).
      reflexivity.
  Qed.

  Lemma hp_proj1_sig_lincomb : ∀ c : Fin.t m → F,
    proj1_sig (lincomb hp_basis_vec c) == lincomb hp_vec c.
  Proof.
    intros c.
    exact (lm_lincomb (sub_inject H) m hp_basis_vec c).
  Qed.

  Lemma hp_recon : ∀ x : Sub V H,
    lincomb hp_basis_vec (hp_coords x) == x.
  Proof.
    intros x.
    change (proj1_sig (lincomb hp_basis_vec (hp_coords x)) == proj1_sig x).
    rewrite (hp_proj1_sig_lincomb (hp_coords x)).
    unfold hp_coords, hp_coords_fun; simpl.
    calc (lincomb hp_vec (drop_lm (basis_coords Bn (proj1_sig x))))
      == (lincomb (basis_vec Bn) (cons_c 0 (drop_lm (basis_coords Bn (proj1_sig x)))))
      == (lincomb (basis_vec Bn) (basis_coords Bn (proj1_sig x)))
      == (proj1_sig x).
    - symmetry. exact (lincomb_cons_drop (drop_lm (basis_coords Bn (proj1_sig x)))).
    - apply lincomb_proper. intros j.
      apply (Fin.caseS' j (fun j0 ⇒ cons_c 0 (drop_lm (basis_coords Bn (proj1_sig x))) j0
                              == basis_coords Bn (proj1_sig x) j0)).
      + rewrite (cons_c_F1 0 (drop_lm (basis_coords Bn (proj1_sig x)))).
        symmetry.
        exact (proj2_sig x).
      + intros i.
        rewrite (cons_c_FS 0 (drop_lm (basis_coords Bn (proj1_sig x))) i).
        reflexivity.
    - exact (basis_recon Bn (proj1_sig x)).
  Qed.

  Lemma hp_coords_lincomb : ∀ c : Fin.t m → F,
    hp_coords (lincomb hp_basis_vec c) == c.
  Proof.
    intros c i.
    unfold hp_coords, hp_coords_fun; simpl.
    calc (basis_coords Bn (proj1_sig (lincomb hp_basis_vec c)) (Fin.FS i))
      == (basis_coords Bn (lincomb hp_vec c) (Fin.FS i))
      == (basis_coords Bn (lincomb (basis_vec Bn) (cons_c 0 c)) (Fin.FS i))
      == (cons_c 0 c (Fin.FS i))
      == (c i).
    - exact (lm_proper (basis_coords Bn) _ _ (hp_proj1_sig_lincomb c) (Fin.FS i)).
    - exact (lm_proper (basis_coords Bn) _ _
               (symmetry (lincomb_cons_drop c)) (Fin.FS i)).
    - exact (basis_coords_lincomb Bn (cons_c 0 c) (Fin.FS i)).
    - rewrite (cons_c_FS 0 c i). reflexivity.
  Qed.

  Definition hyperplane_basis : Basis m F (Sub V H) :=
    @Build_Basis m F (Sub V H) FF (Sub_VectorSpace H)
      hp_basis_vec hp_coords hp_recon hp_coords_lincomb.

End Hyperplane.


Section ExtendBasis.
  Context {F W : Type} {FF : Field F} {VW : VectorSpace F W}.
  Context (phi : LinearMap F W F) (w1 : W) (Hphi1 : phi w1 == 1).

  Context (HW : W → Prop) {HW_SubspacePred : SubspacePred F W HW}.
  Context (Hchar : ∀ w : W, HW w ↔ phi w == 0).

  Context {k : nat} (BHW : Basis k F (Sub W HW)).

  Definition eb_vec (i : Fin.t (S k)) : W :=
    Fin.caseS' i (fun _ ⇒ W) w1 (fun i' ⇒ proj1_sig (basis_vec BHW i')).

  Lemma eb_vec_F1 : eb_vec Fin.F1 = w1.
  Proof. reflexivity. Qed.

  Lemma eb_vec_FS : ∀ i' : Fin.t k, eb_vec (Fin.FS i') = proj1_sig (basis_vec BHW i').
  Proof. intros i'. reflexivity. Qed.

  Definition residual_val (w : W) : W := w + (- phi w) *: w1.

  Lemma residual_pf : ∀ w : W, HW (residual_val w).
  Proof.
    intros w.
    apply (proj2 (Hchar (residual_val w))).
    unfold residual_val.
    calc (phi (w + (- phi w) *: w1))
      == (phi w + phi ((- phi w) *: w1)) { (lm_add phi w ((- phi w) *: w1)) }
      == (phi w + (- phi w) × phi w1) { (lm_scale phi (- phi w) w1) }
      == (phi w + (- phi w) × 1) { Hphi1 }
      == (phi w + (- phi w)) { (ring_mul_1r (- phi w)) }
      == (0 : F) { (grp_invr (phi w)) }.
  Qed.

  Lemma residual_val_add : ∀ u v : W,
    residual_val (u + v) == residual_val u + residual_val v.
  Proof.
    intros u v.
    unfold residual_val.
    calc (u + v + (- phi (u + v)) *: w1)
      == (u + v + (- (phi u + phi v)) *: w1)
           { (lm_add phi u v) }
      == (u + v + ((- phi u) + (- phi v)) *: w1)
           { (neg_add (phi u) (phi v)) }
      == (u + v + ((- phi u) *: w1 + (- phi v) *: w1))
           { (sc_add_scale (- phi u) (- phi v) w1) }
      == ((u + (- phi u) *: w1) + (v + (- phi v) *: w1))
           { (add_interchange u v ((- phi u) *: w1) ((- phi v) *: w1)) }.
  Qed.

  Lemma residual_val_scale : ∀ (a : F) (u : W),
    residual_val (a *: u) == a *: residual_val u.
  Proof.
    intros a u.
    unfold residual_val.
    calc (a *: u + (- phi (a *: u)) *: w1)
      == (a *: u + (- (a × phi u)) *: w1)
           { (lm_scale phi a u) }
      == (a *: u + (a × (- phi u)) *: w1)
           { (mul_neg_r a (phi u)) }
      == (a *: u + a *: ((- phi u) *: w1))
           { (sc_scale_scale a (- phi u) w1) }
      == (a *: (u + (- phi u) *: w1))
           { (sc_scale_add a u ((- phi u) *: w1)) }.
  Qed.

  Definition residual (w : W) : Sub W HW := exist HW (residual_val w) (residual_pf w).

  Lemma residual_add : ∀ u v : W, residual (u + v) == residual u + residual v.
  Proof. intros u v. exact (residual_val_add u v). Qed.

  Lemma residual_scale : ∀ (a : F) (u : W), residual (a *: u) == a *: residual u.
  Proof. intros a u. exact (residual_val_scale a u). Qed.

  Lemma residual_proper : Proper (equ ==> equ) residual.
  Proof.
    intros u v Huv.
    unfold residual, residual_val.
    change (u + (- phi u) *: w1 == v + (- phi v) *: w1).
    rewrite Huv.
    reflexivity.
  Qed.

  Definition eb_coords_fun (w : W) : Fin.t (S k) → F :=
    cons_c (phi w) (basis_coords BHW (residual w)).

  Lemma eb_coords_proper : Proper (equ ==> equ) eb_coords_fun.
  Proof.
    intros u v Huv i.
    unfold eb_coords_fun.
    apply (Fin.caseS' i (fun i0 ⇒ cons_c (phi u) (basis_coords BHW (residual u)) i0
                            == cons_c (phi v) (basis_coords BHW (residual v)) i0)).
    - rewrite (cons_c_F1 (phi u) (basis_coords BHW (residual u))),
        (cons_c_F1 (phi v) (basis_coords BHW (residual v))).
      exact (lm_proper phi u v Huv).
    - intros i'.
      rewrite (cons_c_FS (phi u) (basis_coords BHW (residual u)) i'),
        (cons_c_FS (phi v) (basis_coords BHW (residual v)) i').
      exact (lm_proper (basis_coords BHW) (residual u) (residual v)
               (residual_proper u v Huv) i').
  Qed.

  Lemma eb_coords_add : ∀ u v : W,
    eb_coords_fun (u + v) == eb_coords_fun u + eb_coords_fun v.
  Proof.
    intros u v i.
    unfold eb_coords_fun.
    apply (Fin.caseS' i (fun i0 ⇒
      cons_c (phi (u + v)) (basis_coords BHW (residual (u + v))) i0
        == (cons_c (phi u) (basis_coords BHW (residual u))
              + cons_c (phi v) (basis_coords BHW (residual v))) i0)).
    - calc (cons_c (phi (u + v)) (basis_coords BHW (residual (u + v))) Fin.F1)
        == (phi (u + v))
        == (phi u + phi v)
        == (cons_c (phi u) (basis_coords BHW (residual u)) Fin.F1
            + cons_c (phi v) (basis_coords BHW (residual v)) Fin.F1).
      + exact (eq_to_equ _ _ (cons_c_F1 (phi (u + v)) (basis_coords BHW (residual (u + v))))).
      + exact (lm_add phi u v).
      + change (phi u + phi v ==
                  cons_c (phi u) (basis_coords BHW (residual u)) Fin.F1
                  + cons_c (phi v) (basis_coords BHW (residual v)) Fin.F1).
        rewrite (cons_c_F1 (phi u) (basis_coords BHW (residual u))),
          (cons_c_F1 (phi v) (basis_coords BHW (residual v))).
        reflexivity.
    - intros i'.
      calc (cons_c (phi (u + v)) (basis_coords BHW (residual (u + v))) (Fin.FS i'))
        == (basis_coords BHW (residual (u + v)) i')
        == (basis_coords BHW (residual u + residual v) i')
        == (basis_coords BHW (residual u) i' + basis_coords BHW (residual v) i')
        == (cons_c (phi u) (basis_coords BHW (residual u)) (Fin.FS i')
            + cons_c (phi v) (basis_coords BHW (residual v)) (Fin.FS i')).
      + exact (eq_to_equ _ _ (cons_c_FS (phi (u + v)) (basis_coords BHW (residual (u + v))) i')).
      + exact (lm_proper (basis_coords BHW) _ _ (residual_add u v) i').
      + exact (lm_add (basis_coords BHW) (residual u) (residual v) i').
      + change (basis_coords BHW (residual u) i' + basis_coords BHW (residual v) i'
                  == cons_c (phi u) (basis_coords BHW (residual u)) (Fin.FS i')
                     + cons_c (phi v) (basis_coords BHW (residual v)) (Fin.FS i')).
        rewrite (cons_c_FS (phi u) (basis_coords BHW (residual u)) i'),
          (cons_c_FS (phi v) (basis_coords BHW (residual v)) i').
        reflexivity.
  Qed.

  Lemma eb_coords_scale : ∀ (a : F) (u : W),
    eb_coords_fun (a *: u) == a *: eb_coords_fun u.
  Proof.
    intros a u i.
    unfold eb_coords_fun.
    apply (Fin.caseS' i (fun i0 ⇒
      cons_c (phi (a *: u)) (basis_coords BHW (residual (a *: u))) i0
        == (a *: cons_c (phi u) (basis_coords BHW (residual u))) i0)).
    - calc (cons_c (phi (a *: u)) (basis_coords BHW (residual (a *: u))) Fin.F1)
        == (phi (a *: u))
        == (a × phi u)
        == (a *: cons_c (phi u) (basis_coords BHW (residual u)) Fin.F1).
      + exact (eq_to_equ _ _ (cons_c_F1 (phi (a *: u)) (basis_coords BHW (residual (a *: u))))).
      + exact (lm_scale phi a u).
      + change (a × phi u == a *: cons_c (phi u) (basis_coords BHW (residual u)) Fin.F1).
        rewrite (cons_c_F1 (phi u) (basis_coords BHW (residual u))).
        reflexivity.
    - intros i'.
      calc (cons_c (phi (a *: u)) (basis_coords BHW (residual (a *: u))) (Fin.FS i'))
        == (basis_coords BHW (residual (a *: u)) i')
        == (basis_coords BHW (a *: residual u) i')
        == (a × basis_coords BHW (residual u) i')
        == (a *: cons_c (phi u) (basis_coords BHW (residual u)) (Fin.FS i')).
      + exact (eq_to_equ _ _ (cons_c_FS (phi (a *: u)) (basis_coords BHW (residual (a *: u))) i')).
      + exact (lm_proper (basis_coords BHW) _ _ (residual_scale a u) i').
      + exact (lm_scale (basis_coords BHW) a (residual u) i').
      + change (a × basis_coords BHW (residual u) i'
                  == a *: cons_c (phi u) (basis_coords BHW (residual u)) (Fin.FS i')).
        rewrite (cons_c_FS (phi u) (basis_coords BHW (residual u)) i').
        reflexivity.
  Qed.

  Definition eb_coords : LinearMap F W (Fin.t (S k) → F) :=
    @Build_LinearMap F W (Fin.t (S k) → F) FF VW
      (@PW_VectorSpace (Fin.t (S k)) F F FF Field_VectorSpace)
      eb_coords_fun eb_coords_proper eb_coords_add eb_coords_scale.

  Lemma eb_zero_tail_sum : ∀ (c : Fin.t k → F),
    phi (bigopF (fun i' ⇒ c i' *: proj1_sig (basis_vec BHW i'))) == 0.
  Proof.
    intros c.
    calc (phi (bigopF (fun i' ⇒ c i' *: proj1_sig (basis_vec BHW i'))))
      == (bigopF (fun i' ⇒ phi (c i' *: proj1_sig (basis_vec BHW i'))))
      == (bigopF (fun i' ⇒ c i' × phi (proj1_sig (basis_vec BHW i'))))
      == (bigopF (fun _ : Fin.t k ⇒ (0 : F)))
      == (0 : F).
    - exact (lm_bigopF phi k (fun i' ⇒ c i' *: proj1_sig (basis_vec BHW i'))).
    - apply bigopF_ext. intros i'.
      exact (lm_scale phi (c i') (proj1_sig (basis_vec BHW i'))).
    - apply bigopF_ext. intros i'.
      rewrite (proj1 (Hchar (proj1_sig (basis_vec BHW i'))) (proj2_sig (basis_vec BHW i'))).
      exact (mul_0_r (c i')).
    - exact (bigopF_zero k).
  Qed.

  Lemma eb_recon : ∀ w : W, lincomb eb_vec (eb_coords_fun w) == w.
  Proof.
    intros w.
    unfold lincomb.
    calc (bigopF (fun j ⇒ eb_coords_fun w j *: eb_vec j))
      == (eb_coords_fun w Fin.F1 *: eb_vec Fin.F1
          + bigopF (fun i' ⇒ eb_coords_fun w (Fin.FS i') *: eb_vec (Fin.FS i')))
      == (phi w *: w1
          + bigopF (fun i' ⇒ basis_coords BHW (residual w) i'
                                 *: proj1_sig (basis_vec BHW i')))
      == (phi w *: w1 + proj1_sig (residual w))
      == w.
    - exact (bigopF_S k (fun j ⇒ eb_coords_fun w j *: eb_vec j)).
    - unfold eb_coords_fun.
      apply mon_add_proper.
      + rewrite (cons_c_F1 (phi w) (basis_coords BHW (residual w))), eb_vec_F1.
        reflexivity.
      + apply bigopF_ext. intros i'.
        rewrite (cons_c_FS (phi w) (basis_coords BHW (residual w)) i'), eb_vec_FS.
        reflexivity.
    - apply mon_add_proper.
      + reflexivity.
      + calc (bigopF (fun i' ⇒ basis_coords BHW (residual w) i'
                                   *: proj1_sig (basis_vec BHW i')))
          == (sub_inject HW (lincomb (basis_vec BHW) (basis_coords BHW (residual w))))
          == (proj1_sig (residual w)).
        × symmetry.
          exact (lm_lincomb (sub_inject HW) k (basis_vec BHW) (basis_coords BHW (residual w))).
        × exact (basis_recon BHW (residual w)).
    - unfold residual, residual_val; simpl.
      calc (phi w *: w1 + (w + (- phi w) *: w1))
        == (w + (phi w *: w1 + (- phi w) *: w1))
             { (add_swap_front (phi w *: w1) w ((- phi w) *: w1)) }
        == (w + (phi w + (- phi w)) *: w1)
             { (sc_add_scale (phi w) (- phi w) w1) }
        == (w + (0 : F) *: w1)
             { (grp_invr (phi w)) }
        == (w + (0 : W))
             { (zero_scale w1) }
        == w
             { mon_idr }.
  Qed.

  Lemma eb_coords_lincomb : ∀ c : Fin.t (S k) → F,
    eb_coords_fun (lincomb eb_vec c) == c.
  Proof.

    unfold eb_coords_fun.
    apply (Fin.caseS' i (fun i0 ⇒
      cons_c (phi (lincomb eb_vec c)) (basis_coords BHW (residual (lincomb eb_vec c))) i0
        == c i0)).
    - transitivity (phi (lincomb eb_vec c)).
      + exact (eq_to_equ _ _ (cons_c_F1 (phi (lincomb eb_vec c)) (basis_coords BHW (residual (lincomb eb_vec c))))).
      + transitivity (c Fin.F1).
        × unfold lincomb.
          transitivity (phi (c Fin.F1 *: eb_vec Fin.F1
            + bigopF (fun i' ⇒ c (Fin.FS i') *: eb_vec (Fin.FS i')))).
          -- apply (lm_proper phi).
             exact (bigopF_S k (fun j ⇒ c j *: eb_vec j)).
          -- transitivity (phi (c Fin.F1 *: eb_vec Fin.F1)
               + phi (bigopF (fun i' ⇒ c (Fin.FS i') *: eb_vec (Fin.FS i')))).
             ++ exact (lm_add phi (c Fin.F1 *: eb_vec Fin.F1)
                          (bigopF (fun i' ⇒ c (Fin.FS i') *: eb_vec (Fin.FS i')))).
             ++ transitivity (c Fin.F1 × phi (eb_vec Fin.F1)
                  + phi (bigopF (fun i' ⇒ c (Fin.FS i') *: eb_vec (Fin.FS i')))).
                ** apply mon_add_proper.
                   --- exact (lm_scale phi (c Fin.F1) (eb_vec Fin.F1)).
                   --- reflexivity.
                ** rewrite eb_vec_F1, Hphi1.
                   transitivity (c Fin.F1 × 1
                     + phi (bigopF (fun i' ⇒ c (Fin.FS i')
                                       *: proj1_sig (basis_vec BHW i')))).
                   { apply mon_add_proper; [reflexivity | apply (lm_proper phi), bigopF_ext;
                       intros i'; rewrite eb_vec_FS; reflexivity]. }
                   { rewrite (eb_zero_tail_sum (fun i' ⇒ c (Fin.FS i'))).
                     rewrite (ring_mul_1r (c Fin.F1)).
                     exact (mon_idr (c Fin.F1)). }
        × reflexivity.
    - intros i'.
      transitivity (basis_coords BHW (residual (lincomb eb_vec c)) i').
      + exact (eq_to_equ _ _ (cons_c_FS (phi (lincomb eb_vec c)) (basis_coords BHW (residual (lincomb eb_vec c))) i')).
      + transitivity (basis_coords BHW (lincomb (basis_vec BHW) (fun j ⇒ c (Fin.FS j))) i').
        × apply (lm_proper (basis_coords BHW) _ _).
          change (proj1_sig (residual (lincomb eb_vec c))
                    == proj1_sig (lincomb (basis_vec BHW) (fun j ⇒ c (Fin.FS j)))).
          transitivity (bigopF (fun j ⇒ c (Fin.FS j) *: proj1_sig (basis_vec BHW j))).
          -- assert (Hrv : residual_val (lincomb eb_vec c)
                 == bigopF (fun j ⇒ c (Fin.FS j) *: proj1_sig (basis_vec BHW j))).
             { unfold residual_val, lincomb.
               transitivity ((c Fin.F1 *: eb_vec Fin.F1
                   + bigopF (fun i0 ⇒ c (Fin.FS i0) *: eb_vec (Fin.FS i0)))
                 + (- phi (c Fin.F1 *: eb_vec Fin.F1
                        + bigopF (fun i0 ⇒ c (Fin.FS i0) *: eb_vec (Fin.FS i0)))) *: w1).
               - apply mon_add_proper.
                 + exact (bigopF_S k (fun j ⇒ c j *: eb_vec j)).
                 + apply sc_proper; [| reflexivity].
                   apply grp_neg_proper, (lm_proper phi).
                   exact (bigopF_S k (fun j ⇒ c j *: eb_vec j)).
               - rewrite eb_vec_F1.
                 assert (Hphi_tail : phi (c Fin.F1 *: w1
                     + bigopF (fun i0 ⇒ c (Fin.FS i0) *: eb_vec (Fin.FS i0))) == c Fin.F1).
                 { transitivity (phi (c Fin.F1 *: w1)
                     + phi (bigopF (fun i0 ⇒ c (Fin.FS i0) *: eb_vec (Fin.FS i0)))).
                   - exact (lm_add phi (c Fin.F1 *: w1)
                              (bigopF (fun i0 ⇒ c (Fin.FS i0) *: eb_vec (Fin.FS i0)))).
                   - transitivity (c Fin.F1 × phi w1
                       + phi (bigopF (fun i0 ⇒ c (Fin.FS i0)
                                         *: proj1_sig (basis_vec BHW i0)))).
                     + apply mon_add_proper.
                       × exact (lm_scale phi (c Fin.F1) w1).
                       × apply (lm_proper phi), bigopF_ext.
                         intros i0. rewrite eb_vec_FS. reflexivity.
                     + rewrite Hphi1, (ring_mul_1r (c Fin.F1)),
                         (eb_zero_tail_sum (fun i0 ⇒ c (Fin.FS i0))).
                       exact (mon_idr (c Fin.F1)).

                 rewrite Hphi_tail.
                 transitivity ((c Fin.F1 *: w1 + (- c Fin.F1) *: w1)
                   + bigopF (fun i0 ⇒ c (Fin.FS i0) *: eb_vec (Fin.FS i0))).
                 × transitivity (c Fin.F1 *: w1
                     + (bigopF (fun i0 ⇒ c (Fin.FS i0) *: eb_vec (Fin.FS i0)) + (- c Fin.F1) *: w1)).
                   -- exact (mon_assoc (c Fin.F1 *: w1)
                                (bigopF (fun i0 ⇒ c (Fin.FS i0) *: eb_vec (Fin.FS i0)))
                                ((- c Fin.F1) *: w1)).
                   -- transitivity (c Fin.F1 *: w1
                        + ((- c Fin.F1) *: w1 + bigopF (fun i0 ⇒ c (Fin.FS i0) *: eb_vec (Fin.FS i0)))).
                      ++ apply mon_add_proper.
                         ** reflexivity.
                         ** exact (ab_comm (bigopF (fun i0 ⇒ c (Fin.FS i0) *: eb_vec (Fin.FS i0)))
                                     ((- c Fin.F1) *: w1)).
                      ++ symmetry.
                         exact (mon_assoc (c Fin.F1 *: w1) ((- c Fin.F1) *: w1)
                                  (bigopF (fun i0 ⇒ c (Fin.FS i0) *: eb_vec (Fin.FS i0)))).
                 × transitivity ((0 : F) *: w1
                     + bigopF (fun i0 ⇒ c (Fin.FS i0) *: eb_vec (Fin.FS i0))).
                   -- apply mon_add_proper.
                      ++ transitivity ((c Fin.F1 + (- c Fin.F1)) *: w1).
                         ** symmetry. exact (sc_add_scale (c Fin.F1) (- c Fin.F1) w1).
                         ** rewrite (grp_invr (c Fin.F1)). reflexivity.
                      ++ apply bigopF_ext. intros i0. rewrite eb_vec_FS. reflexivity.
                   -- rewrite (zero_scale w1).
                      exact (mon_idl (bigopF (fun i0 ⇒ c (Fin.FS i0)
                                                 *: proj1_sig (basis_vec BHW i0)))).
             }
             change (proj1_sig (residual (lincomb eb_vec c))
                       == bigopF (fun j ⇒ c (Fin.FS j) *: proj1_sig (basis_vec BHW j))).
             exact Hrv.
          -- symmetry.
             exact (lm_lincomb (sub_inject HW) k (basis_vec BHW) (fun j ⇒ c (Fin.FS j))).
        × exact (basis_coords_lincomb BHW (fun j ⇒ c (Fin.FS j)) i').
  Qed.

  Definition extend_basis_by_vector : Basis (S k) F W :=
    @Build_Basis (S k) F W FF VW eb_vec eb_coords eb_recon eb_coords_lincomb.

End ExtendBasis.


Section TrivialBasis.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V}.
  Context (B0 : Basis 0 F V).

  Lemma v_trivial : ∀ v : V, v == 0.
  Proof.
    intros v.
    calc v
      == (lincomb (basis_vec B0) (basis_coords B0 v))
      == (lincomb (basis_vec B0) (basis_coords B0 0))
      == (0 : V).
    - symmetry. exact (basis_recon B0 v).
    - apply lincomb_proper. intros i. exact (Fin.case0 (fun i0 ⇒ _) i).
    - exact (basis_recon B0 0).
  Qed.

  Context (P : V → Prop) {SP : SubspacePred F V P}.

  Definition triv_vec : Fin.t 0 → Sub V P := fun i ⇒ Fin.case0 _ i.

  Definition triv_coords : LinearMap F (Sub V P) (Fin.t 0 → F) := lm_const0.

  Lemma triv_recon : ∀ x : Sub V P, lincomb triv_vec (triv_coords x) == x.
  Proof.
    intros x.
    transitivity (0 : Sub V P).
    - exact (bigopF_0 (fun i ⇒ triv_coords x i *: triv_vec i)).
    - symmetry. exact (v_trivial (proj1_sig x)).
  Qed.

  Lemma triv_coords_lincomb : ∀ c : Fin.t 0 → F,
    triv_coords (lincomb triv_vec c) == c.
  Proof.
    intros c i.
    exact (Fin.case0 (fun i0 ⇒ triv_coords (lincomb triv_vec c) i0 == c i0) i).
  Qed.

  Definition trivial_basis : Basis 0 F (Sub V P) :=
    @Build_Basis 0 F (Sub V P) FF (Sub_VectorSpace P)
      triv_vec triv_coords triv_recon triv_coords_lincomb.

End TrivialBasis.


Theorem subspace_basis_exists : ∀ (n : nat) (F V : Type) (FF : Field F)
    (VV : VectorSpace F V) (Bn : Basis n F V) (P : V → Prop)
    (SP : SubspacePred F V P),
  ∃ k : nat, Nat.le k n ∧ ∃ Bk : Basis k F (Sub V P), True.
Proof.
  induction n as [| m IH]; intros F V FF VV Bn P SP.
  - ∃ (0 : nat). split.
    + exact (le_n 0).
    + ∃ (trivial_basis Bn P). exact I.
  - destruct (IH F (Sub V (H Bn)) FF _ (hyperplane_basis Bn) (Pinner (H Bn) P)
                 (SubspacePred_pullback (H Bn) P))
      as [k0 [Hk0 [Bk0 _]]].
    pose (BPw := basis_and_collapse (H Bn) P k0 Bk0).
    destruct (classic (∃ v0 : V, P v0 ∧ c1 Bn v0 =/= 0)) as [Hex | Hnex].
    + destruct (constructive_indefinite_description _ Hex) as [v0 [Hv0P Hv0ne]].
      pose (v1 := inv (c1 Bn v0) *: v0).
      pose proof (sp_scale (inv (c1 Bn v0)) v0 Hv0P) as Hv1P.
      assert (Hv1_1 : c1 Bn v1 == 1).
      { unfold v1.
        transitivity (inv (c1 Bn v0) × c1 Bn v0).
        - exact (lm_scale (c1 Bn) (inv (c1 Bn v0)) v0).
        - exact (field_invl (c1 Bn v0) Hv0ne). }
      pose (w1 := exist P v1 Hv1P : Sub V P).
      pose (phi := lm_comp (c1 Bn) (sub_inject P)).
      pose proof Hv1_1 as Hphiw1.
      pose (BPw' := basis_and_comm (H Bn) P k0 BPw).
      pose (BkPH := basis_and_expand P (H Bn) k0 BPw').
      assert (Hchar : ∀ w : Sub V P, Pinner P (H Bn) w ↔ phi w == 0)
        by (intros w; unfold Pinner, H, ker_pred, phi, lm_comp; simpl; reflexivity).
      ∃ (S k0). split.
      × exact (le_n_S k0 m Hk0).
      × ∃ (extend_basis_by_vector phi w1 Hphiw1 (Pinner P (H Bn)) Hchar BkPH). exact I.
    + assert (Hiff : ∀ v : V, Pand (H Bn) P v ↔ P v).
      { intros v. unfold Pand, H, ker_pred. split.
        - intros [_ HPv]. exact HPv.
        - intros HPv.
          split.
          + destruct (classic (c1 Bn v == 0)) as [Hz | Hnz].
            × exact Hz.
            × exfalso. apply Hnex. ∃ v. exact (conj HPv Hnz).
          + exact HPv.
      }
      ∃ k0. split.
      × exact (le_S k0 m Hk0).
      × ∃ (basis_pred_iff (Pand (H Bn) P) P Hiff k0 BPw). exact I.
Qed.


Theorem subspace_has_basis : subspace_has_basis_statement.
Proof.
  unfold subspace_has_basis_statement.
  intros F V FF VV n Bn P SP.
  exact (subspace_basis_exists n F V FF VV Bn P SP).
Qed.
