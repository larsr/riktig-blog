Library Riktig.Interface.Bilinear

From Coq Require Import Morphisms.
From Coq.Vectors Require Fin.
From Riktig Require Import Ops Props Interface.Field Interface.VectorSpace.


Record BilinearMap (F U V W : Type) {FF : Field F}
    {VU : VectorSpace F U} {VV : VectorSpace F V}
    {VW : VectorSpace F W} := {
  bm_fun :> U → V → W;
  bm_proper : Proper (equ ==> equ ==> equ) bm_fun;
  bm_add_l : ∀ (u u' : U) (v : V),
    bm_fun (u + u') v == bm_fun u v + bm_fun u' v;
  bm_scale_l : ∀ (a : F) (u : U) (v : V),
    bm_fun (a *: u) v == a *: bm_fun u v;
  bm_add_r : ∀ (u : U) (v v' : V),
    bm_fun u (v + v') == bm_fun u v + bm_fun u v';
  bm_scale_r : ∀ (a : F) (u : U) (v : V),
    bm_fun u (a *: v) == a *: bm_fun u v
}.

Arguments bm_fun {F U V W FF VU VV VW}.
Arguments bm_proper {F U V W FF VU VV VW}.
Arguments bm_add_l {F U V W FF VU VV VW}.
Arguments bm_scale_l {F U V W FF VU VV VW}.
Arguments bm_add_r {F U V W FF VU VV VW}.
Arguments bm_scale_r {F U V W FF VU VV VW}.


Definition famEq {V : Type} {EV : EqRel V} {n : nat}
    (vs ws : Fin.t n → V) : Prop :=
  ∀ i : Fin.t n, vs i == ws i.

Definition upd {V : Type} {n : nat}
    (vs : Fin.t n → V) (i : Fin.t n) (x : V) : Fin.t n → V :=
  fun j ⇒ match Fin.eq_dec i j with
           | left _ ⇒ x
           | right _ ⇒ vs j
           end.


Record MultilinearForm (n : nat) (F V : Type) {FF : Field F}
    {VV : VectorSpace F V} := {
  mf_fun :> (Fin.t n → V) → F;
  mf_proper : Proper (@famEq V _ n ==> equ) mf_fun;
  mf_add : ∀ (vs : Fin.t n → V) (i : Fin.t n) (x y : V),
    mf_fun (upd vs i (x + y)) == mf_fun (upd vs i x) + mf_fun (upd vs i y);
  mf_scale : ∀ (vs : Fin.t n → V) (i : Fin.t n) (a : F) (x : V),
    mf_fun (upd vs i (a *: x)) == a × mf_fun (upd vs i x)
}.

Arguments mf_fun {n F V FF VV}.
Arguments mf_proper {n F V FF VV}.
Arguments mf_add {n F V FF VV}.
Arguments mf_scale {n F V FF VV}.


Record Alternating {n : nat} {F V : Type} {FF : Field F}
    {VV : VectorSpace F V} (m : MultilinearForm n F V) := {
  alt_repeat : ∀ (vs : Fin.t n → V) (i j : Fin.t n),
    i ≠ j → vs i == vs j → m vs == 0
}.

Arguments alt_repeat {n F V FF VV m}.
