Section FunCompProper.
  Context {P Q R : Type} {EP : EqRel P} {EQ : EqRel Q} {ER : EqRel R}.

  Lemma fun_comp_proper : ∀ (g : P → Q) (h : Q → R),
    Proper (equ ==> equ) g → Proper (equ ==> equ) h →
    Proper (equ ==> equ) (fun p : P ⇒ h (g p)).
  Proof.
    intros g h Pg Ph u v Huv.
    apply Ph, Pg, Huv.
  Qed.

End FunCompProper.


Section Backprop3.
  Context {F U V W X : Type} `{NormedSpace F U}.
  Context {VSV : VectorSpace F V} {NSV : NormedSpace F V}.
  Context {VSW : VectorSpace F W} {NSW : NormedSpace F W}.
  Context {VSX : VectorSpace F X} {NSX : NormedSpace F X}.
  Context (n m p q : nat) (BU : Basis n F U) (BV : Basis m F V)
    (BW : Basis p F W) (BX : Basis q F X).


  Lemma jacobian_left_matmul : ∀ (L1 : LinearMap F U V)
      (L2 : LinearMap F V W) (L3 : LinearMap F W X),
    mat_of BU BX (lm_comp (lm_comp L3 L2) L1)
      == mat_mul (mat_mul (mat_of BW BX L3) (mat_of BV BW L2))
           (mat_of BU BV L1).
  Proof.
    intros L1 L2 L3.
    calc (mat_of BU BX (lm_comp (lm_comp L3 L2) L1))
      == (mat_mul (mat_of BV BX (lm_comp L3 L2)) (mat_of BU BV L1))
      == (mat_mul (mat_mul (mat_of BW BX L3) (mat_of BV BW L2))
            (mat_of BU BV L1)).
    - exact (mat_of_comp BU BV BX (lm_comp L3 L2) L1).
    - exact (mat_mul_proper q m n (mat_of BV BX (lm_comp L3 L2))
               (mat_mul (mat_of BW BX L3) (mat_of BV BW L2))
               (mat_of_comp BV BW BX L3 L2)
               (mat_of BU BV L1) (mat_of BU BV L1)
               (reflexivity (mat_of BU BV L1))).
  Qed.

  Lemma jacobian_right_matmul : ∀ (L1 : LinearMap F U V)
      (L2 : LinearMap F V W) (L3 : LinearMap F W X),
    mat_of BU BX (lm_comp L3 (lm_comp L2 L1))
      == mat_mul (mat_of BW BX L3)
           (mat_mul (mat_of BV BW L2) (mat_of BU BV L1)).
  Proof.
    intros L1 L2 L3.
    calc (mat_of BU BX (lm_comp L3 (lm_comp L2 L1)))
      == (mat_mul (mat_of BW BX L3) (mat_of BU BW (lm_comp L2 L1)))
      == (mat_mul (mat_of BW BX L3)
            (mat_mul (mat_of BV BW L2) (mat_of BU BV L1))).
    - exact (mat_of_comp BU BW BX L3 (lm_comp L2 L1)).
    - exact (mat_mul_proper q p n (mat_of BW BX L3) (mat_of BW BX L3)
               (reflexivity (mat_of BW BX L3))
               (mat_of BU BW (lm_comp L2 L1))
               (mat_mul (mat_of BV BW L2) (mat_of BU BV L1))
               (mat_of_comp BU BV BW L2 L1)).
  Qed.

  Theorem jacobian_chain3_assoc :
    ∀ (f1 : U → V) (f2 : V → W) (f3 : W → X) (a : U)
      (L1 : LinearMap F U V) (L2 : LinearMap F V W) (L3 : LinearMap F W X)
      (P2 : Proper (equ ==> equ) f2) (P3 : Proper (equ ==> equ) f3)
      (Hd1 : is_deriv f1 a L1) (Hd2 : is_deriv f2 (f1 a) L2)
      (Hd3 : is_deriv f3 (f2 (f1 a)) L3),
    jacobian n q BU BX
      (deriv_comp f1 (fun v : V ⇒ f3 (f2 v)) a L1 (lm_comp L3 L2)
         (fun_comp_proper f2 f3 P2 P3) Hd1
         (deriv_comp f2 f3 (f1 a) L2 L3 P3 Hd2 Hd3))
    == jacobian n q BU BX
      (deriv_comp (fun u : U ⇒ f2 (f1 u)) f3 a (lm_comp L2 L1) L3 P3
         (deriv_comp f1 f2 a L1 L2 P2 Hd1 Hd2) Hd3).
  Proof.
    intros f1 f2 f3 a L1 L2 L3 P2 P3 Hd1 Hd2 Hd3.
    exact (mat_of_proper BU BX (lm_comp (lm_comp L3 L2) L1)
             (lm_comp L3 (lm_comp L2 L1)) (lm_comp_assoc L3 L2 L1)).
  Qed.


  Theorem jacobian_reverse_mode :
    ∀ (f1 : U → V) (f2 : V → W) (f3 : W → X) (a : U)
      (L1 : LinearMap F U V) (L2 : LinearMap F V W) (L3 : LinearMap F W X)
      (P2 : Proper (equ ==> equ) f2) (P3 : Proper (equ ==> equ) f3)
      (Hd1 : is_deriv f1 a L1) (Hd2 : is_deriv f2 (f1 a) L2)
      (Hd3 : is_deriv f3 (f2 (f1 a)) L3),
    jacobian n q BU BX
      (deriv_comp f1 (fun v : V ⇒ f3 (f2 v)) a L1 (lm_comp L3 L2)
         (fun_comp_proper f2 f3 P2 P3) Hd1
         (deriv_comp f2 f3 (f1 a) L2 L3 P3 Hd2 Hd3))
    == mat_mul (mat_mul (jacobian p q BW BX Hd3) (jacobian m p BV BW Hd2))
         (jacobian n m BU BV Hd1).
  Proof.
    intros f1 f2 f3 a L1 L2 L3 P2 P3 Hd1 Hd2 Hd3.
    exact (jacobian_left_matmul L1 L2 L3).
  Qed.

  Theorem jacobian_forward_mode :
    ∀ (f1 : U → V) (f2 : V → W) (f3 : W → X) (a : U)
      (L1 : LinearMap F U V) (L2 : LinearMap F V W) (L3 : LinearMap F W X)
      (P2 : Proper (equ ==> equ) f2) (P3 : Proper (equ ==> equ) f3)
      (Hd1 : is_deriv f1 a L1) (Hd2 : is_deriv f2 (f1 a) L2)
      (Hd3 : is_deriv f3 (f2 (f1 a)) L3),
    jacobian n q BU BX
      (deriv_comp (fun u : U ⇒ f2 (f1 u)) f3 a (lm_comp L2 L1) L3 P3
         (deriv_comp f1 f2 a L1 L2 P2 Hd1 Hd2) Hd3)
    == mat_mul (jacobian p q BW BX Hd3)
         (mat_mul (jacobian m p BV BW Hd2) (jacobian n m BU BV Hd1)).
  Proof.
    intros f1 f2 f3 a L1 L2 L3 P2 P3 Hd1 Hd2 Hd3.
    exact (jacobian_right_matmul L1 L2 L3).
  Qed.

  Theorem backprop_eq_forward_mode :
    ∀ (f1 : U → V) (f2 : V → W) (f3 : W → X) (a : U)
      (L1 : LinearMap F U V) (L2 : LinearMap F V W) (L3 : LinearMap F W X)
      (P2 : Proper (equ ==> equ) f2) (P3 : Proper (equ ==> equ) f3)
      (Hd1 : is_deriv f1 a L1) (Hd2 : is_deriv f2 (f1 a) L2)
      (Hd3 : is_deriv f3 (f2 (f1 a)) L3),
    mat_mul (mat_mul (jacobian p q BW BX Hd3) (jacobian m p BV BW Hd2))
      (jacobian n m BU BV Hd1)
    == mat_mul (jacobian p q BW BX Hd3)
         (mat_mul (jacobian m p BV BW Hd2) (jacobian n m BU BV Hd1)).
  Proof.
    intros f1 f2 f3 a L1 L2 L3 P2 P3 Hd1 Hd2 Hd3.
    exact (mat_mul_assoc q p m n (jacobian p q BW BX Hd3)
             (jacobian m p BV BW Hd2) (jacobian n m BU BV Hd1)).
  Qed.

End Backprop3.


Section Backprop4.
  Context {F U V W X Y : Type} `{NormedSpace F U}.
  Context {VSV : VectorSpace F V} {NSV : NormedSpace F V}.
  Context {VSW : VectorSpace F W} {NSW : NormedSpace F W}.
  Context {VSX : VectorSpace F X} {NSX : NormedSpace F X}.
  Context {VSY : VectorSpace F Y} {NSY : NormedSpace F Y}.
  Context (n m p q r : nat) (BU : Basis n F U) (BV : Basis m F V)
    (BW : Basis p F W) (BX : Basis q F X) (BY : Basis r F Y).

  Theorem jacobian_chain4_assoc : ∀ (L1 : LinearMap F U V)
      (L2 : LinearMap F V W) (L3 : LinearMap F W X) (L4 : LinearMap F X Y),
    mat_of BU BY (lm_comp (lm_comp (lm_comp L4 L3) L2) L1)
      == mat_of BU BY (lm_comp L4 (lm_comp L3 (lm_comp L2 L1))).
  Proof.
    intros L1 L2 L3 L4.
    exact (mat_of_proper BU BY (lm_comp (lm_comp (lm_comp L4 L3) L2) L1)
             (lm_comp L4 (lm_comp L3 (lm_comp L2 L1))) (lm_comp_assoc4 L1 L2 L3 L4)).
  Qed.


End Backprop4.
