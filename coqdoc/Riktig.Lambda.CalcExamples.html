Example self_app_id :
  App (Abs (App (Var 0) (Var 0))) idt == idt.
Proof.
  calc (App (Abs (App (Var 0) (Var 0))) idt)
    == (App idt idt) { (conv_beta (App (Var 0) (Var 0)) idt (value_abs (Var 0))) }
    == idt { conv_id_id }.
Qed.


Example ifb_true_selects_first :
  IFB TRUE idt k == idt.
Proof.
  calc (IFB TRUE idt k)
    ==[inner] (App (Abs (Abs (Var 0))) k)
    ==[outer] idt.
  [inner]: apply conv_appl;
    exact (conv_beta (Abs (Var 1)) idt (value_abs (Var 0))).
  [outer]: exact (conv_beta (Abs (Var 0)) k (value_abs (Abs (Var 1)))).
Qed.


Example cps_idt_normal_form :
  App (cps idt) idt == Abs (Abs (App (Var 0) (Var 1))).
Proof.
  calc (App (cps idt) idt)
    == (App idt (psi idt)) { (conv_step _ _
          (cps_value_step idt idt (value_abs (Var 0)) (value_abs (Var 0)))) }
    == (Abs (Abs (App (Var 0) (Var 1)))) { (conv_step _ _
          (step_beta (Var 0) (psi idt) (value_abs (Abs (App (Var 0) (Var 1)))))) }.
Qed.


#[local] Instance cstar_Transitive : Transitive cstar := cstar_trans.

Lemma cstar_of_step : ∀ t u, cstep t u → cstar t u.
Proof. intros t u H. eapply cstar_step. { exact H. } apply cstar_refl. Qed.

Example babs_self_app_duplicates :
  cstar (babs (CApp (CVar 0) (CVar 0)) @@ CVar 1) (CVar 1 @@ CVar 1).
Proof.
  cbn [babs].
  calc (CS @@ CI @@ CI @@ CVar 1)
    == ((CI @@ CVar 1) @@ (CI @@ CVar 1))
         { (cstar_of_step _ _ (cstep_S CI CI (CVar 1))) }
    == (CVar 1 @@ (CI @@ CVar 1))
         { (cstar_of_step _ _ (cstep_app1 _ _ _ (cstep_I (CVar 1)))) }
    == (CVar 1 @@ CVar 1)
         { (cstar_of_step _ _ (cstep_app2 _ _ _ (cstep_I (CVar 1)))) }.
Qed.
