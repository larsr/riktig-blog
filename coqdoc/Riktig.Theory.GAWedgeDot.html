Section ScaleMixin.
  Context {F M : Type} `{Field F} {MG : AbGroup M} {MS : HasScale F M}
    {ML : ScaleLaws F M}.

  #[global] Instance sc_l_proper : ∀ a : F,
    Proper (equ ==> equ) (@scale F M _ a).
  Proof.
    intros a x y Hxy.
    now rewrite Hxy.
  Qed.

  Theorem sc_zero : ∀ a : F, a *: (0 : M) == 0.
  Proof.
    intros a.
    exact (hom_zero (scale a) (sc_scale_add a)).
  Qed.

  Theorem sc_neg : ∀ (a : F) (x : M), a *: (- x) == - (a *: x).
  Proof.
    intros a x.
    exact (hom_neg (scale a) (sc_scale_add a) x).
  Qed.

  Theorem sc_swap : ∀ (a b : F) (x : M),
    a *: (b *: x) == b *: (a *: x).
  Proof.
    intros a b x.
    calc (a *: (b *: x))
      == ((a × b) *: x) { (sc_scale_scale a b x) }
      == ((b × a) *: x) { (comm a b) }
      == (b *: (a *: x)) { (sc_scale_scale b a x) }.
  Qed.

End ScaleMixin.


Section FieldHalving.
  Context {F : Type} `{OrderedField F}.

  Lemma half_one_double : ∀ x : F, half 1 × (x + x) == x.
  Proof.
    intros x.
    calc (half 1 × (x + x))
      == (half 1 × x + half 1 × x) { (ring_distl (half 1) x x) }
      == ((half 1 + half 1) × x) { (ring_distr (half 1) (half 1) x) }
      == (1 × x) { (half_sum (1 : F)) }
      == x { ring_mul_1l }.
  Qed.

End FieldHalving.

Section ScaleHalving.
  Context {F M : Type} `{OrderedField F} {MG : AbGroup M}
    {MS : HasScale F M} {ML : ScaleLaws F M}.

  Lemma sc_half_double : ∀ x : M, half 1 *: (x + x) == x.
  Proof.
    intros x.
    calc (half 1 *: (x + x))
      == (half 1 *: x + half 1 *: x) { (sc_scale_add (half 1) x x) }
      == ((half 1 + half 1) *: x) { (sc_add_scale (half 1) (half 1) x) }
      == (1 *: x) { (half_sum (1 : F)) }
      == x { (sc_scale_1 x) }.
  Qed.

End ScaleHalving.


Section CleanDecomposition.
  Context {F V A : Type} `{OrderedField F} {VS : VectorSpace F V}
    {IP : InnerProduct F V} {GA : GeometricAlgebra F V A}.

  Theorem ga_dot_from_product : ∀ u v : V,
    ⟨ u , v ⟩ *: (1 : A)
    == half 1 *: (ga_vec u × ga_vec v + ga_vec v × ga_vec u).
  Proof.
    intros u v.
    calc ((⟨ u , v ⟩ *: (1 : A)) : A)
      == (((half 1 × (⟨ u , v ⟩ + ⟨ u , v ⟩)) *: (1 : A)) : A)
        { (half_one_double ⟨ u , v ⟩) }
      == ((half 1 *: ((⟨ u , v ⟩ + ⟨ u , v ⟩) *: (1 : A))) : A)
        { (sc_scale_scale (half 1) (⟨ u , v ⟩ + ⟨ u , v ⟩) (1 : A)) }
      == ((half 1 *: (ga_vec u × ga_vec v + ga_vec v × ga_vec u)) : A)
        { (ga_symmetric_part u v) }.
  Qed.

  Corollary ga_symmetric_part_clean : ∀ u v : V,
    half 1 *: (ga_vec u × ga_vec v + ga_vec v × ga_vec u)
    == ⟨ u , v ⟩ *: (1 : A).
  Proof.
    intros u v.
    symmetry.
    exact (ga_dot_from_product u v).
  Qed.

  Theorem ga_product_decompose : ∀ u v : V,
    ga_vec u × ga_vec v == ⟨ u , v ⟩ *: (1 : A) + half 1 *: wedge u v.
  Proof.
    intros u v.
    calc ((ga_vec u × ga_vec v) : A)
      == ((half 1 *: (ga_vec u × ga_vec v + ga_vec u × ga_vec v)) : A)
        { (sc_half_double (ga_vec u × ga_vec v)) }
      ==[hard] ((half 1 *: ((ga_vec u × ga_vec v + ga_vec v × ga_vec u)
            + (ga_vec u × ga_vec v + - (ga_vec v × ga_vec u)))) : A)
      == ((half 1 *: (ga_vec u × ga_vec v + ga_vec v × ga_vec u)
            + half 1 *: (ga_vec u × ga_vec v + - (ga_vec v × ga_vec u))) : A)
        { (sc_scale_add (half 1) (ga_vec u × ga_vec v + ga_vec v × ga_vec u)
             (ga_vec u × ga_vec v + - (ga_vec v × ga_vec u))) }
      == ((⟨ u , v ⟩ *: (1 : A) + half 1 *: wedge u v) : A)
        { (ga_symmetric_part_clean u v) }.
    [hard]: calcrw ((ga_vec u × ga_vec v + ga_vec u × ga_vec v) : A)
      == (((ga_vec u × ga_vec v + ga_vec u × ga_vec v) + 0) : A)
        { idr }
      == (((ga_vec u × ga_vec v + ga_vec u × ga_vec v)
            + (ga_vec v × ga_vec u + - (ga_vec v × ga_vec u))) : A)
        { (invr (ga_vec v × ga_vec u)) }
      == (((ga_vec u × ga_vec v + ga_vec v × ga_vec u)
            + (ga_vec u × ga_vec v + - (ga_vec v × ga_vec u))) : A)
        { (add_interchange (ga_vec u × ga_vec v) (ga_vec u × ga_vec v)
             (ga_vec v × ga_vec u) (- (ga_vec v × ga_vec u))) }.
  Qed.

  Corollary ga_antisymmetric_part_clean : ∀ u v : V,
    half 1 *: wedge u v == ga_vec u × ga_vec v + - (⟨ u , v ⟩ *: (1 : A)).
  Proof.
    intros u v.
    calc ((half 1 *: wedge u v) : A)
      == ((0 + half 1 *: wedge u v) : A)
        { idl }
      == (((- (⟨ u , v ⟩ *: (1 : A)) + ⟨ u , v ⟩ *: (1 : A))
            + half 1 *: wedge u v) : A)
        { (invl (⟨ u , v ⟩ *: (1 : A))) }
      == ((- (⟨ u , v ⟩ *: (1 : A))
            + (⟨ u , v ⟩ *: (1 : A) + half 1 *: wedge u v)) : A)
        { (assoc (- (⟨ u , v ⟩ *: (1 : A))) (⟨ u , v ⟩ *: (1 : A))
             (half 1 *: wedge u v)) }
      == ((- (⟨ u , v ⟩ *: (1 : A)) + ga_vec u × ga_vec v) : A)
        { (ga_product_decompose u v) }
      == ((ga_vec u × ga_vec v + - (⟨ u , v ⟩ *: (1 : A))) : A)
        { (comm (- (⟨ u , v ⟩ *: (1 : A))) (ga_vec u × ga_vec v)) }.
  Qed.

End CleanDecomposition.



Section WedgeBilinearity.
  Context {F V A : Type} `{GeometricAlgebra F V A}.

  #[global] Instance wedge_proper :
    Proper (equ ==> equ ==> equ) (@wedge F V A _ _ _ _).
  Proof.
    intros u u' Hu v v' Hv.
    unfold wedge.
    now rewrite Hu, Hv.
  Qed.

  Theorem wedge_add_l : ∀ u u' v : V,
    (wedge (u + u') v : A) == wedge u v + wedge u' v.
  Proof.
    intros u u' v.
    unfold wedge.
    calc ((ga_vec (u + u') × ga_vec v + - (ga_vec v × ga_vec (u + u'))) : A)
      == (((ga_vec u + ga_vec u') × ga_vec v
            + - (ga_vec v × (ga_vec u + ga_vec u'))) : A)
        { (ga_vec_add u u') }
      == (((ga_vec u × ga_vec v + ga_vec u' × ga_vec v)
            + - (ga_vec v × (ga_vec u + ga_vec u'))) : A)
        { (ring_distr (ga_vec u) (ga_vec u') (ga_vec v)) }
      == (((ga_vec u × ga_vec v + ga_vec u' × ga_vec v)
            + - (ga_vec v × ga_vec u + ga_vec v × ga_vec u')) : A)
        { (ring_distl (ga_vec v) (ga_vec u) (ga_vec u')) }
      == (((ga_vec u × ga_vec v + ga_vec u' × ga_vec v)
            + (- (ga_vec v × ga_vec u) + - (ga_vec v × ga_vec u'))) : A)
        { (neg_add (ga_vec v × ga_vec u) (ga_vec v × ga_vec u')) }
      == (((ga_vec u × ga_vec v + - (ga_vec v × ga_vec u))
            + (ga_vec u' × ga_vec v + - (ga_vec v × ga_vec u'))) : A)
        { (add_interchange (ga_vec u × ga_vec v) (ga_vec u' × ga_vec v)
             (- (ga_vec v × ga_vec u)) (- (ga_vec v × ga_vec u'))) }.
  Qed.

  Theorem wedge_scale_l : ∀ (a : F) (u v : V),
    (wedge (a *: u) v : A) == a *: wedge u v.
  Proof.
    intros a u v.
    unfold wedge.
    calc ((ga_vec (a *: u) × ga_vec v + - (ga_vec v × ga_vec (a *: u))) : A)
      == (((a *: ga_vec u) × ga_vec v
            + - (ga_vec v × (a *: ga_vec u))) : A)
        { (ga_vec_scale a u) }
      == ((a *: (ga_vec u × ga_vec v)
            + - (ga_vec v × (a *: ga_vec u))) : A)
        { (ga_scale_mul_l a (ga_vec u) (ga_vec v)) }
      == ((a *: (ga_vec u × ga_vec v)
            + - (a *: (ga_vec v × ga_vec u))) : A)
        { (ga_scale_mul_r a (ga_vec v) (ga_vec u)) }
      == ((a *: (ga_vec u × ga_vec v)
            + a *: (- (ga_vec v × ga_vec u))) : A)
        { (sc_neg a (ga_vec v × ga_vec u)) }
      == ((a *: (ga_vec u × ga_vec v + - (ga_vec v × ga_vec u))) : A)
        { (sc_scale_add a (ga_vec u × ga_vec v) (- (ga_vec v × ga_vec u))) }.
  Qed.

  Theorem wedge_add_r : ∀ u v v' : V,
    (wedge u (v + v') : A) == wedge u v + wedge u v'.
  Proof.
    intros u v v'.
    unfold wedge.
    calc ((ga_vec u × ga_vec (v + v') + - (ga_vec (v + v') × ga_vec u)) : A)
      == ((ga_vec u × (ga_vec v + ga_vec v')
            + - ((ga_vec v + ga_vec v') × ga_vec u)) : A)
        { (ga_vec_add v v') }
      == (((ga_vec u × ga_vec v + ga_vec u × ga_vec v')
            + - ((ga_vec v + ga_vec v') × ga_vec u)) : A)
        { (ring_distl (ga_vec u) (ga_vec v) (ga_vec v')) }
      == (((ga_vec u × ga_vec v + ga_vec u × ga_vec v')
            + - (ga_vec v × ga_vec u + ga_vec v' × ga_vec u)) : A)
        { (ring_distr (ga_vec v) (ga_vec v') (ga_vec u)) }
      == (((ga_vec u × ga_vec v + ga_vec u × ga_vec v')
            + (- (ga_vec v × ga_vec u) + - (ga_vec v' × ga_vec u))) : A)
        { (neg_add (ga_vec v × ga_vec u) (ga_vec v' × ga_vec u)) }
      == (((ga_vec u × ga_vec v + - (ga_vec v × ga_vec u))
            + (ga_vec u × ga_vec v' + - (ga_vec v' × ga_vec u))) : A)
        { (add_interchange (ga_vec u × ga_vec v) (ga_vec u × ga_vec v')
             (- (ga_vec v × ga_vec u)) (- (ga_vec v' × ga_vec u))) }.
  Qed.

  Theorem wedge_scale_r : ∀ (a : F) (u v : V),
    (wedge u (a *: v) : A) == a *: wedge u v.
  Proof.
    intros a u v.
    unfold wedge.
    calc ((ga_vec u × ga_vec (a *: v) + - (ga_vec (a *: v) × ga_vec u)) : A)
      == ((ga_vec u × (a *: ga_vec v)
            + - ((a *: ga_vec v) × ga_vec u)) : A)
        { (ga_vec_scale a v) }
      == ((a *: (ga_vec u × ga_vec v)
            + - ((a *: ga_vec v) × ga_vec u)) : A)
        { (ga_scale_mul_r a (ga_vec u) (ga_vec v)) }
      == ((a *: (ga_vec u × ga_vec v)
            + - (a *: (ga_vec v × ga_vec u))) : A)
        { (ga_scale_mul_l a (ga_vec v) (ga_vec u)) }
      == ((a *: (ga_vec u × ga_vec v)
            + a *: (- (ga_vec v × ga_vec u))) : A)
        { (sc_neg a (ga_vec v × ga_vec u)) }
      == ((a *: (ga_vec u × ga_vec v + - (ga_vec v × ga_vec u))) : A)
        { (sc_scale_add a (ga_vec u × ga_vec v) (- (ga_vec v × ga_vec u))) }.
  Qed.

End WedgeBilinearity.

Class WedgeDotPresentation (F V A : Type) `{InnerProduct F V} := {
  #[global] wd_ring :: Ring A;
  #[global] wd_scale :: HasScale F A;
  #[global] wd_scale_laws :: ScaleLaws F A;
  wd_scale_mul_l : ∀ (a : F) (x y : A), (a *: x) × y == a *: (x × y);
  wd_scale_mul_r : ∀ (a : F) (x y : A), x × (a *: y) == a *: (x × y);
  wd_vec : V → A;
  #[global] wd_vec_proper :: Proper (equ ==> equ) wd_vec;
  wd_vec_add : ∀ u v : V, wd_vec (u + v) == wd_vec u + wd_vec v;
  wd_vec_scale : ∀ (a : F) (u : V), wd_vec (a *: u) == a *: wd_vec u;
  wd_w : V → V → A;
  #[global] wd_w_proper :: Proper (equ ==> equ ==> equ) wd_w;
  wd_w_add_l : ∀ u u' v : V, wd_w (u + u') v == wd_w u v + wd_w u' v;
  wd_w_scale_l : ∀ (a : F) (u v : V), wd_w (a *: u) v == a *: wd_w u v;
  wd_w_add_r : ∀ u v v' : V, wd_w u (v + v') == wd_w u v + wd_w u v';
  wd_w_scale_r : ∀ (a : F) (u v : V), wd_w u (a *: v) == a *: wd_w u v;
  wd_w_alt : ∀ u : V, wd_w u u == 0;
  wd_decomp : ∀ u v : V,
                     wd_vec u × wd_vec v == ⟨ u , v ⟩ *: 1 + wd_w u v
}.

#[global] Hint Mode WedgeDotPresentation - - ! - - - : typeclass_instances.


Section WedgeDotTheory.
  Context {F V A : Type} `{WedgeDotPresentation F V A}.

  Theorem wd_w_anti : ∀ u v : V, (wd_w u v : A) == - wd_w v u.
  Proof.
    intros u v.
    apply neg_unique_l.
    calc ((wd_w u v + wd_w v u) : A)
      ==[pad] (((wd_w u u + wd_w u v) + (wd_w v u + wd_w v v)) : A)
      ==[gather] ((wd_w (u + v) (u + v)) : A)
      ==[alt] (0 : A).
    [pad]: now rewrite (wd_w_alt u), (wd_w_alt v), idl, idr.
    [gather]: now rewrite (wd_w_add_l u v (u + v)), (wd_w_add_r u u v),
                (wd_w_add_r v u v).
    [alt]: exact (wd_w_alt (u + v)).
  Qed.

  Theorem wedgedot_to_contract : ∀ u : V,
    wd_vec u × wd_vec u == norm2 u *: (1 : A).
  Proof.
    intros u.
    calc ((wd_vec u × wd_vec u) : A)
      == ((⟨ u , u ⟩ *: (1 : A) + wd_w u u) : A) { (wd_decomp u u) }
      == ((⟨ u , u ⟩ *: (1 : A) + 0) : A) { (wd_w_alt u) }
      == ((norm2 u *: (1 : A)) : A) { idr }.
  Qed.

  Definition wedgedot_to_ga : GeometricAlgebra F V A :=
    @Build_GeometricAlgebra F V A _ _ _
      wd_ring wd_scale wd_scale_laws wd_scale_mul_l wd_scale_mul_r
      wd_vec wd_vec_proper wd_vec_add wd_vec_scale
      wedgedot_to_contract.

End WedgeDotTheory.


Section ContractToWedgeDot.
  Context {F V A : Type} `{OrderedField F} {VS : VectorSpace F V}
    {IP : InnerProduct F V} {GA : GeometricAlgebra F V A}.

  Definition half_wedge (u v : V) : A := half 1 *: wedge u v.

  #[global] Instance half_wedge_proper :
    Proper (equ ==> equ ==> equ) half_wedge.
  Proof.
    intros u u' Hu v v' Hv.
    unfold half_wedge.
    now rewrite (wedge_proper u u' Hu v v' Hv).
  Qed.

  Lemma half_wedge_add_l : ∀ u u' v : V,
    half_wedge (u + u') v == half_wedge u v + half_wedge u' v.
  Proof.
    intros u u' v.
    unfold half_wedge.
    calc ((half 1 *: wedge (u + u') v) : A)
      == ((half 1 *: (wedge u v + wedge u' v)) : A)
        { (wedge_add_l u u' v) }
      == ((half 1 *: wedge u v + half 1 *: wedge u' v) : A)
        { (sc_scale_add (half 1) (wedge u v) (wedge u' v)) }.
  Qed.

  Lemma half_wedge_scale_l : ∀ (a : F) (u v : V),
    half_wedge (a *: u) v == a *: half_wedge u v.
  Proof.
    intros a u v.
    unfold half_wedge.
    calc ((half 1 *: wedge (a *: u) v) : A)
      == ((half 1 *: (a *: wedge u v)) : A)
        { (wedge_scale_l a u v) }
      == ((a *: (half 1 *: wedge u v)) : A)
        { (sc_swap (half 1) a (wedge u v)) }.
  Qed.

  Lemma half_wedge_add_r : ∀ u v v' : V,
    half_wedge u (v + v') == half_wedge u v + half_wedge u v'.
  Proof.
    intros u v v'.
    unfold half_wedge.
    calc ((half 1 *: wedge u (v + v')) : A)
      == ((half 1 *: (wedge u v + wedge u v')) : A)
        { (wedge_add_r u v v') }
      == ((half 1 *: wedge u v + half 1 *: wedge u v') : A)
        { (sc_scale_add (half 1) (wedge u v) (wedge u v')) }.
  Qed.

  Lemma half_wedge_scale_r : ∀ (a : F) (u v : V),
    half_wedge u (a *: v) == a *: half_wedge u v.
  Proof.
    intros a u v.
    unfold half_wedge.
    calc ((half 1 *: wedge u (a *: v)) : A)
      == ((half 1 *: (a *: wedge u v)) : A)
        { (wedge_scale_r a u v) }
      == ((a *: (half 1 *: wedge u v)) : A)
        { (sc_swap (half 1) a (wedge u v)) }.
  Qed.

  Lemma half_wedge_alt : ∀ u : V, half_wedge u u == 0.
  Proof.
    intros u.
    unfold half_wedge.
    calc ((half 1 *: wedge u u) : A)
      == ((half 1 *: (0 : A)) : A) { (wedge_self_zero u) }
      == (0 : A) { (sc_zero (half 1)) }.
  Qed.

  Lemma half_wedge_decomp : ∀ u v : V,
    ga_vec u × ga_vec v == ⟨ u , v ⟩ *: (1 : A) + half_wedge u v.
  Proof.
    intros u v.
    exact (ga_product_decompose u v).
  Qed.

  Definition contract_to_wedgedot : WedgeDotPresentation F V A :=
    @Build_WedgeDotPresentation F V A _ VS IP
      ga_ring ga_scale ga_scale_laws ga_scale_mul_l ga_scale_mul_r
      ga_vec ga_vec_proper ga_vec_add ga_vec_scale
      half_wedge half_wedge_proper half_wedge_add_l half_wedge_scale_l
      half_wedge_add_r half_wedge_scale_r half_wedge_alt
      half_wedge_decomp.

End ContractToWedgeDot.
