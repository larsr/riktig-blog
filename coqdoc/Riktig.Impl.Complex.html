Record C (F : Type) := mkC { re : F; im : F }.

Arguments mkC {F} re im.
Arguments re {F} c.
Arguments im {F} c.


Section ComplexOps.
  Context {F : Type} `{Field F}.

  #[global] Instance C_EqRel : EqRel (C F) :=
    fun z w ⇒ re z == re w ∧ im z == im w.

  #[global] Instance C_Zero : Zero (C F) := mkC 0 0.

  #[global] Instance C_One : One (C F) := mkC 1 0.

  #[global] Instance C_Add : HasAdd (C F) :=
    fun z w ⇒ mkC (re z + re w) (im z + im w).

  #[global] Instance C_Neg : HasNeg (C F) :=
    fun z ⇒ mkC (- re z) (- im z).

  #[global] Instance C_Mult : HasMult (C F) :=
    fun z w ⇒ mkC (re z × re w + - (im z × im w))
                   (re z × im w + im z × re w).

  Definition norm2 (z : C F) : F := re z × re z + im z × im z.

  #[global] Instance C_Inv : HasInv (C F) :=
    fun z ⇒ mkC (re z × inv (norm2 z)) (- im z × inv (norm2 z)).

  Lemma norm2_eq : ∀ z : C F, norm2 z = re z × re z + im z × im z.
  Proof.
    intros z.
    reflexivity.
  Qed.

  Lemma norm2_proper : ∀ z w : C F, z == w → norm2 z == norm2 w.
  Proof.
    intros z w [Hr Hi].
    rewrite norm2_eq, norm2_eq, Hr, Hi.
    reflexivity.
  Qed.

  #[global] Instance C_Equivalence : Equivalence C_EqRel.
  Proof.
    apply Build_Equivalence.
    - intros z. split.
      + reflexivity.
      + reflexivity.
    - intros z w [Hr Hi]. split.
      + symmetry. exact Hr.
      + symmetry. exact Hi.
    - intros z w v [Hr1 Hi1] [Hr2 Hi2]. split.
      + rewrite Hr1. exact Hr2.
      + rewrite Hi1. exact Hi2.
  Qed.

End ComplexOps.


Section ComplexRing.
  Context {F : Type} `{Field F}.


  Lemma c_add_assoc : ∀ z w v : C F, (z + w) + v == z + (w + v).
  Proof.
    intros z w v. split.
    - exact (mon_assoc (re z) (re w) (re v)).
    - exact (mon_assoc (im z) (im w) (im v)).
  Qed.

  Lemma c_add_idl : ∀ z : C F, 0 + z == z.
  Proof.
    intros z. split.
    - exact (mon_idl (re z)).
    - exact (mon_idl (im z)).
  Qed.

  Lemma c_add_idr : ∀ z : C F, z + 0 == z.
  Proof.
    intros z. split.
    - exact (mon_idr (re z)).
    - exact (mon_idr (im z)).
  Qed.

  Lemma c_add_proper : Proper (equ ==> equ ==> equ) (@add (C F) _).
  Proof.
    intros z z' [Hr Hi] w w' [Hr' Hi']. split.
    - exact (mon_add_proper (re z) (re z') Hr (re w) (re w') Hr').
    - exact (mon_add_proper (im z) (im z') Hi (im w) (im w') Hi').
  Qed.

  Lemma c_add_invl : ∀ z : C F, (- z) + z == 0.
  Proof.
    intros z. split.
    - exact (grp_invl (re z)).
    - exact (grp_invl (im z)).
  Qed.

  Lemma c_add_invr : ∀ z : C F, z + (- z) == 0.
  Proof.
    intros z. split.
    - exact (grp_invr (re z)).
    - exact (grp_invr (im z)).
  Qed.

  Lemma c_neg_proper : Proper (equ ==> equ) (@neg (C F) _).
  Proof.
    intros z w [Hr Hi]. split.
    - exact (grp_neg_proper (re z) (re w) Hr).
    - exact (grp_neg_proper (im z) (im w) Hi).
  Qed.

  Lemma c_add_comm : ∀ z w : C F, z + w == w + z.
  Proof.
    intros z w. split.
    - exact (ab_comm (re z) (re w)).
    - exact (ab_comm (im z) (im w)).
  Qed.

  #[global] Instance C_Monoid : Monoid (C F) := {|
    mon_equ := C_EqRel;
    mon_equiv := C_Equivalence;
    mon_zero := C_Zero;
    mon_add := C_Add;
    mon_assoc := c_add_assoc;
    mon_idl := c_add_idl;
    mon_idr := c_add_idr;
    mon_add_proper := c_add_proper
  |}.

  #[global] Instance C_Group : Group (C F) := {|
    grp_monoid := C_Monoid;
    grp_neg := C_Neg;
    grp_invl := c_add_invl;
    grp_invr := c_add_invr;
    grp_neg_proper := c_neg_proper
  |}.

  #[global] Instance C_AbGroup : AbGroup (C F) := {|
    ab_group := C_Group;
    ab_comm := c_add_comm
  |}.


  Lemma cmul_1l_re : ∀ a b : F, 1 × a + - (0 × b) == a.
  Proof.
    intros a b.
    rewrite ring_mul_1l, mul_0_l, neg_zero, idr.
    reflexivity.
  Qed.

  Lemma cmul_1l_im : ∀ a b : F, 1 × b + 0 × a == b.
  Proof.
    intros a b.
    rewrite ring_mul_1l, mul_0_l, idr.
    reflexivity.
  Qed.

  Lemma cmul_1r_re : ∀ a b : F, a × 1 + - (b × 0) == a.
  Proof.
    intros a b.
    rewrite ring_mul_1r, mul_0_r, neg_zero, idr.
    reflexivity.
  Qed.

  Lemma cmul_1r_im : ∀ a b : F, a × 0 + b × 1 == b.
  Proof.
    intros a b.
    rewrite mul_0_r, ring_mul_1r, idl.
    reflexivity.
  Qed.

  Lemma cmul_assoc_re : ∀ a b c d e f : F,
    (a × c + - (b × d)) × e + - ((a × d + b × c) × f)
    == a × (c × e + - (d × f)) + - (b × (c × f + d × e)).
  Proof.
    intros a b c d e f.
    rewrite ring_distr, ring_distr, ring_distl, ring_distl.
    rewrite mul_neg_l, mul_neg_r, neg_add, neg_add.
    rewrite ring_mul_assoc, ring_mul_assoc, ring_mul_assoc, ring_mul_assoc.
    rewrite add_interchange, (comm (- (b × (d × e))) (- (b × (c × f)))).
    reflexivity.
  Qed.

  Lemma cmul_assoc_im : ∀ a b c d e f : F,
    (a × c + - (b × d)) × f + (a × d + b × c) × e
    == a × (c × f + d × e) + b × (c × e + - (d × f)).
  Proof.
    intros a b c d e f.
    rewrite ring_distr, ring_distr, ring_distl, ring_distl.
    rewrite mul_neg_l, mul_neg_r.
    rewrite ring_mul_assoc, ring_mul_assoc, ring_mul_assoc, ring_mul_assoc.
    rewrite add_interchange, (comm (- (b × (d × f))) (b × (c × e))).
    reflexivity.
  Qed.

  Lemma cmul_distl_re : ∀ a b c d e f : F,
    a × (c + e) + - (b × (d + f)) == (a × c + - (b × d)) + (a × e + - (b × f)).
  Proof.
    intros a b c d e f.
    rewrite ring_distl, ring_distl, neg_add, add_interchange.
    reflexivity.
  Qed.

  Lemma cmul_distl_im : ∀ a b c d e f : F,
    a × (d + f) + b × (c + e) == (a × d + b × c) + (a × f + b × e).
  Proof.
    intros a b c d e f.
    rewrite ring_distl, ring_distl, add_interchange.
    reflexivity.
  Qed.

  Lemma cmul_distr_re : ∀ a b c d e f : F,
    (a + c) × e + - ((b + d) × f) == (a × e + - (b × f)) + (c × e + - (d × f)).
  Proof.
    intros a b c d e f.
    rewrite ring_distr, ring_distr, neg_add, add_interchange.
    reflexivity.
  Qed.

  Lemma cmul_distr_im : ∀ a b c d e f : F,
    (a + c) × f + (b + d) × e == (a × f + b × e) + (c × f + d × e).
  Proof.
    intros a b c d e f.
    rewrite ring_distr, ring_distr, add_interchange.
    reflexivity.
  Qed.

  Lemma cmul_re_proper : ∀ a a' b b' c c' d d' : F,
    a == a' → b == b' → c == c' → d == d' →
    a × c + - (b × d) == a' × c' + - (b' × d').
  Proof.
    intros a a' b b' c c' d d' Ha Hb Hc Hd.
    rewrite Ha, Hb, Hc, Hd.
    reflexivity.
  Qed.

  Lemma cmul_im_proper : ∀ a a' b b' c c' d d' : F,
    a == a' → b == b' → c == c' → d == d' →
    a × d + b × c == a' × d' + b' × c'.
  Proof.
    intros a a' b b' c c' d d' Ha Hb Hc Hd.
    rewrite Ha, Hb, Hc, Hd.
    reflexivity.
  Qed.


  Lemma c_mul_assoc : ∀ z w v : C F, (z × w) × v == z × (w × v).
  Proof.
    intros z w v. split.
    - exact (cmul_assoc_re (re z) (im z) (re w) (im w) (re v) (im v)).
    - exact (cmul_assoc_im (re z) (im z) (re w) (im w) (re v) (im v)).
  Qed.

  Lemma c_mul_1l : ∀ z : C F, 1 × z == z.
  Proof.
    intros z. split.
    - exact (cmul_1l_re (re z) (im z)).
    - exact (cmul_1l_im (re z) (im z)).
  Qed.

  Lemma c_mul_1r : ∀ z : C F, z × 1 == z.
  Proof.
    intros z. split.
    - exact (cmul_1r_re (re z) (im z)).
    - exact (cmul_1r_im (re z) (im z)).
  Qed.

  Lemma c_distl : ∀ z w v : C F, z × (w + v) == z × w + z × v.
  Proof.
    intros z w v. split.
    - exact (cmul_distl_re (re z) (im z) (re w) (im w) (re v) (im v)).
    - exact (cmul_distl_im (re z) (im z) (re w) (im w) (re v) (im v)).
  Qed.

  Lemma c_distr : ∀ z w v : C F, (z + w) × v == z × v + w × v.
  Proof.
    intros z w v. split.
    - exact (cmul_distr_re (re z) (im z) (re w) (im w) (re v) (im v)).
    - exact (cmul_distr_im (re z) (im z) (re w) (im w) (re v) (im v)).
  Qed.

  Lemma c_mul_proper : Proper (equ ==> equ ==> equ) (@mult (C F) _).
  Proof.
    intros z z' [Hr Hi] w w' [Hr' Hi']. split.
    - exact (cmul_re_proper (re z) (re z') (im z) (im z')
               (re w) (re w') (im w) (im w') Hr Hi Hr' Hi').
    - exact (cmul_im_proper (re z) (re z') (im z) (im z')
               (re w) (re w') (im w) (im w') Hr Hi Hr' Hi').
  Qed.

  #[global] Instance C_Ring : Ring (C F) := {|
    ring_ab := C_AbGroup;
    ring_one := C_One;
    ring_mul := C_Mult;
    ring_mul_assoc := c_mul_assoc;
    ring_mul_1l := c_mul_1l;
    ring_mul_1r := c_mul_1r;
    ring_distl := c_distl;
    ring_distr := c_distr;
    ring_mul_proper := c_mul_proper
  |}.

  Example c_mul_0_l : ∀ z : C F, 0 × z == 0.
  Proof.
    intros z.
    exact (mul_0_l z).
  Qed.


  Lemma cmul_comm_re : ∀ a b c d : F,
    a × c + - (b × d) == c × a + - (d × b).
  Proof.
    intros a b c d.
    rewrite (field_mul_comm a c), (field_mul_comm b d).
    reflexivity.
  Qed.

  Lemma cmul_comm_im : ∀ a b c d : F,
    a × d + b × c == c × b + d × a.
  Proof.
    intros a b c d.
    rewrite (field_mul_comm a d), (field_mul_comm b c).
    apply comm.
  Qed.

  Lemma c_mul_comm : ∀ z w : C F, z × w == w × z.
  Proof.
    intros z w. split.
    - exact (cmul_comm_re (re z) (im z) (re w) (im w)).
    - exact (cmul_comm_im (re z) (im z) (re w) (im w)).
  Qed.

  Lemma mul_inv_shuffle : ∀ x n : F, (x × inv n) × x == inv n × (x × x).
  Proof.
    intros x n.
    rewrite (field_mul_comm x (inv n)), ring_mul_assoc.
    reflexivity.
  Qed.

  Lemma mul_inv_swap : ∀ x y n : F, (x × inv n) × y == (y × inv n) × x.
  Proof.
    intros x y n.
    rewrite (field_mul_comm x (inv n)), (field_mul_comm y (inv n)).
    rewrite ring_mul_assoc, ring_mul_assoc, (field_mul_comm x y).
    reflexivity.
  Qed.

  Lemma mul_inv_proper2 : ∀ x y n m : F,
    x == y → n == m → x × inv n == y × inv m.
  Proof.
    intros x y n m Hxy Hnm.
    rewrite Hxy, Hnm.
    reflexivity.
  Qed.


  Lemma cinv_re : ∀ a b n : F,
    n =/= 0 → a × a + b × b == n →
    (a × inv n) × a + - ((- b × inv n) × b) == 1.
  Proof.
    intros a b n Hn Heq.
    rewrite (mul_neg_l b (inv n)), (mul_neg_l (b × inv n) b), neg_involutive.
    rewrite (mul_inv_shuffle a n), (mul_inv_shuffle b n), <- ring_distl, Heq.
    exact (field_invl n Hn).
  Qed.

  Lemma cinv_im : ∀ a b n : F,
    (a × inv n) × b + (- b × inv n) × a == 0.
  Proof.
    intros a b n.
    rewrite (mul_neg_l b (inv n)), (mul_neg_l (b × inv n) a).
    rewrite (mul_inv_swap a b n).
    apply invr.
  Qed.

  Lemma c_inv_proper : Proper (equ ==> equ) (@inv (C F) _).
  Proof.
    intros z w Hzw.
    destruct Hzw as [Hr Hi]. split.
    - exact (mul_inv_proper2 (re z) (re w) (norm2 z) (norm2 w)
               Hr (norm2_proper z w (conj Hr Hi))).
    - exact (mul_inv_proper2 (- im z) (- im w) (norm2 z) (norm2 w)
               (grp_neg_proper (im z) (im w) Hi)
               (norm2_proper z w (conj Hr Hi))).
  Qed.

  Lemma c_nontrivial : (1 : C F) =/= 0.
  Proof.
    intros [Hr Hi].
    exact (field_nontrivial Hr).
  Qed.

End ComplexRing.


Section ComplexField.
  Context {F : Type} `{FormallyReal F}.

  Lemma norm2_nonzero : ∀ z : C F, z =/= 0 → norm2 z =/= 0.
  Proof.
    intros z Hz Hn.
    apply Hz. split.
    - exact (fr_sos (re z) (im z) Hn).
    - exact (fr_sos_r (re z) (im z) Hn).
  Qed.

  Lemma c_invl : ∀ z : C F, z =/= 0 → inv z × z == 1.
  Proof.
    intros z Hz. split.
    - apply (cinv_re (re z) (im z) (norm2 z)).
      + exact (norm2_nonzero z Hz).
      + reflexivity.
    - exact (cinv_im (re z) (im z) (norm2 z)).
  Qed.

  #[global] Instance C_Field : Field (C F) := {|
    field_ring := C_Ring;
    field_mul_comm := c_mul_comm;
    field_inv := C_Inv;
    field_invl := c_invl;
    field_inv_proper := c_inv_proper;
    field_nontrivial := c_nontrivial
  |}.

End ComplexField.
