Library Riktig.Impl.Zr

From Coq Require Import ZArith Morphisms.
From Riktig Require Import Ops Interface.Monoid Interface.Group
  Interface.AbGroup Interface.Ring Theory.Ring.

#[global] Instance Z_EqRel : EqRel Z := eq.
#[global] Instance Z_Zero : Zero Z := 0%Z.
#[global] Instance Z_One : One Z := 1%Z.
#[global] Instance Z_Add : HasAdd Z := Z.add.
#[global] Instance Z_Mult : HasMult Z := Z.mul.
#[global] Instance Z_Neg : HasNeg Z := Z.opp.

#[global] Instance Z_Monoid : Monoid Z := {|
  mon_equ := Z_EqRel;
  mon_zero := Z_Zero;
  mon_add := Z_Add;
  mon_equiv := eq_equivalence;
  mon_assoc := fun a b c ⇒ eq_sym (Z.add_assoc a b c);
  mon_idl := Z.add_0_l;
  mon_idr := Z.add_0_r;
  mon_add_proper := ltac:(solve_proper)
|}.

#[global] Instance Z_Group : Group Z := {|
  grp_monoid := Z_Monoid;
  grp_neg := Z_Neg;
  grp_invl := Z.add_opp_diag_l;
  grp_invr := Z.add_opp_diag_r;
  grp_neg_proper := ltac:(solve_proper)
|}.

#[global] Instance Z_AbGroup : AbGroup Z := {|
  ab_group := Z_Group;
  ab_comm := Z.add_comm
|}.

#[global] Instance Z_Ring : Ring Z := {|
  ring_ab := Z_AbGroup;
  ring_one := Z_One;
  ring_mul := Z_Mult;
  ring_mul_assoc := fun a b c ⇒ eq_sym (Z.mul_assoc a b c);
  ring_mul_1l := Z.mul_1_l;
  ring_mul_1r := Z.mul_1_r;
  ring_distl := Z.mul_add_distr_l;
  ring_distr := Z.mul_add_distr_r;
  ring_mul_proper := ltac:(solve_proper)
|}.

Example z_mul_0_l : (0 × 7)%Z == 0.
Proof. exact (mul_0_l 7%Z). Qed.
