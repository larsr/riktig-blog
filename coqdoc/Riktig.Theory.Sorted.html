Library Riktig.Theory.Sorted


From Coq Require Export RelationClasses.
Require Export List Permutation Sorted. Import ListNotations.
From Riktig Require Export Interface.Decidable Interface.DecTotalOrder Theory.DecTotalOrder.

Section SortedTheory.
  Context {A : Type} `{O : DecTotalOrder A}.

  Lemma sorted_head_le : ∀ (x : A) l y, Sorted ole (x::l) → In y (x::l) → ole x y.
    intros x l y S [<-|I]; [apply ole_refl|].
    assert (F : Forall (ole x) l) by (eapply Sorted_extends; [exact ole_trans | exact S]).
    exact (proj1 (Forall_forall (ole x) l) F y I).
  Qed.

  Lemma sorted_perm_unique: ∀ a b : list A,
      Sorted ole a → Sorted ole b → Permutation a b → a = b.
    induction a as [|x a IH]; intros [|y b] SA SB P; auto.
    - now apply Permutation_nil in P.
    - now apply Permutation_sym, Permutation_nil in P.
    - assert (x = y); [|subst y; f_equal].
      + apply ole_antisym.
        × apply (sorted_head_le x a y SA),
                (Permutation_in y (Permutation_sym P)), in_eq.
        × apply (sorted_head_le y b x SB),
                (Permutation_in x P), in_eq.
      + inversion_clear SA; inversion_clear SB;
        eauto using Permutation_cons_inv.
   Qed.

End SortedTheory.
