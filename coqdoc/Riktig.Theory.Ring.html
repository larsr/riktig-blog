Library Riktig.Theory.Ring

From Coq Require Export Morphisms.
From Riktig Require Export Ops Props Interface.Ring Theory.Group Lib.Calc.

Section RingTheory.
  Context {R : Type} `{Ring R}.

  Lemma mul_l_proper : ∀ a : R, Proper (equ ==> equ) (mult a).
  Proof.
    intros a x y Hxy.
    rewrite Hxy.
    reflexivity.
  Qed.

  Lemma mul_r_proper : ∀ a : R, Proper (equ ==> equ) (fun x ⇒ x × a).
  Proof.
    intros a x y Hxy.
    rewrite Hxy.
    reflexivity.
  Qed.

  Theorem mul_0_l : ∀ a : R, 0 × a == 0.
  Proof.
    intros a.
    exact (hom_zero (fun x ⇒ x × a) (mul_r_proper a) (fun x y ⇒ distr x y a)).
  Qed.

  Theorem mul_0_r : ∀ a : R, a × 0 == 0.
  Proof.
    intros a.
    exact (hom_zero (mult a) (mul_l_proper a) (distl a)).
  Qed.

  Theorem mul_neg_l : ∀ a b : R, (- a) × b == - (a × b).
  Proof.
    intros a b.
    exact (hom_neg (fun x ⇒ x × b) (mul_r_proper b) (fun x y ⇒ distr x y b) a).
  Qed.

  Theorem mul_neg_r : ∀ a b : R, a × (- b) == - (a × b).
  Proof.
    intros a b.
    exact (hom_neg (mult a) (mul_l_proper a) (distl a) b).
  Qed.

  Theorem mul_neg_neg : ∀ a b : R, (- a) × (- b) == a × b.
  Proof.
    intros a b.
    calc ((- a) × (- b))
      == (- (a × (- b))) { (mul_neg_l a (- b)) }
      == (- - (a × b)) { (mul_neg_r a b) }
      == (a × b) { (neg_involutive (a × b)) }.
  Qed.

  Theorem mul_neg1_l : ∀ a : R, (- (1)) × a == - a.
  Proof.
    intros a.
    calc ((- (1)) × a)
      == (- (1 × a)) { (mul_neg_l 1 a) }
      == (- a) { idl }.
  Qed.

  Theorem mul_neg1_r : ∀ a : R, a × (- (1)) == - a.
  Proof.
    intros a.
    calc (a × (- (1)))
      == (- (a × 1)) { (mul_neg_r a 1) }
      == (- a) { idr }.
  Qed.

  Theorem mul_add_expand : ∀ a b c d : R,
    (a + b) × (c + d) == (a × c + b × c) + (a × d + b × d).
  Proof.
    intros a b c d.
    calc ((a + b) × (c + d))
      == ((a + b) × c + (a + b) × d) { (distl (a + b) c d) }
      == ((a × c + b × c) + (a + b) × d) { (distr a b c) }
      == ((a × c + b × c) + (a × d + b × d)) { (distr a b d) }.
  Qed.

End RingTheory.
