Library Riktig.Theory.BinarySearch


Require Export Arith Lia.
Require Export List.
From Riktig Require Export Interface.Decidable Impl.Decidable.
From Riktig Require Export Lib.MeasureInd.
Require Export Riktig.Programs.BinarySearch.

Section BinarySearchProps.

  Variable V : Type.
  Variable leb : V → V → bool.

  Variable mem : Type.
  Variable read : mem → nat → V.

  Variable m : mem.
  Variable target : V.

  Hypothesis leb_trans :
    ∀ x y z, leb x y = true → leb y z = true → leb x z = true.

  Notation bsearch := (bsearch V leb mem read m target).
  Notation bsearch_step := (bsearch_step V leb mem read m target).


  Theorem bsearch_sound :
    ∀ lo hi i,
      bsearch lo hi = Some i →
      lo ≤ i ≤ hi
      ∧ leb (read m i) target = true
      ∧ leb target (read m i) = true.
  Proof.
    apply (measure_ind2 (fun lo hi ⇒ S hi - lo)
             (fun lo hi ⇒ ∀ i,
                  bsearch lo hi = Some i →
                  lo ≤ i ≤ hi
                  ∧ leb (read m i) target = true
                  ∧ leb target (read m i) = true)).
    intros lo hi IH i.
    rewrite bsearch_step.
    destruct (dec (hi < lo)) as [L|L]; [intros E; discriminate E|].
    cbv zeta.
    pose proof (half_le (hi - lo)) as Hq.
    destruct (dec (leb (read m (lo + (hi - lo) / 2)) target = true)) as [L1|L1].
    - destruct (dec (leb target (read m (lo + (hi - lo) / 2)) = true)) as [L2|L2].
      +
        intros E.
        assert (Ei : i = lo + (hi - lo) / 2) by congruence.
        subst i.
        split; [lia|split; assumption].
      +
        intros E.
        assert (Hd : S hi - S (lo + (hi - lo) / 2) < S hi - lo) by lia.
        destruct (IH _ hi Hd _ E) as (Hr & H1 & H2).
        split; [lia|split; assumption].
    - destruct (dec (lo + (hi - lo) / 2 = lo)) as [Eeq|Eeq];
        [intros E; discriminate E|].
      intros E.
      assert (Hd : S (lo + (hi - lo) / 2 - 1) - lo < S hi - lo) by lia.
      destruct (IH lo _ Hd _ E) as (Hr & H1 & H2).
      split; [lia|split; assumption].
  Qed.

  Corollary bsearch_sound_eq :
    (∀ x y, leb x y = true → leb y x = true → x = y) →
    ∀ lo hi i,
      bsearch lo hi = Some i →
      lo ≤ i ≤ hi ∧ read m i = target.
  Proof.
    intros leb_antisym lo hi i E.
    apply bsearch_sound in E.
    destruct E as (Hr & H1 & H2).
    split; [assumption|apply leb_antisym; assumption].
  Qed.


  Theorem bsearch_complete :
    ∀ lo hi,
      (∀ i j, lo ≤ i → i ≤ j → j ≤ hi →
         leb (read m i) (read m j) = true) →
      ∀ j, lo ≤ j ≤ hi →
        leb (read m j) target = true →
        leb target (read m j) = true →
        ∃ i, bsearch lo hi = Some i.
  Proof.
    apply (measure_ind2 (fun lo hi ⇒ S hi - lo)
             (fun lo hi ⇒
                (∀ i j, lo ≤ i → i ≤ j → j ≤ hi →
                   leb (read m i) (read m j) = true) →
                ∀ j, lo ≤ j ≤ hi →
                  leb (read m j) target = true →
                  leb target (read m j) = true →
                  ∃ i, bsearch lo hi = Some i)).
    intros lo hi IH Hsort j Hj Hjt Htj.
    rewrite bsearch_step.
    destruct (dec (hi < lo)) as [L|L]; [exfalso; lia|].
    cbv zeta.
    pose proof (half_le (hi - lo)) as Hq.
    destruct (dec (leb (read m (lo + (hi - lo) / 2)) target = true)) as [L1|L1].
    - destruct (dec (leb target (read m (lo + (hi - lo) / 2)) = true)) as [L2|L2].
      +
        eexists; reflexivity.
      +
        assert (Hmid : lo + (hi - lo) / 2 < j).
        { destruct (le_lt_dec j (lo + (hi - lo) / 2)) as [Hle|Hgt];
            [|exact Hgt].
          exfalso; apply L2.
          apply leb_trans with (read m j); [assumption|].
          apply Hsort; lia. }
        assert (Hd : S hi - S (lo + (hi - lo) / 2) < S hi - lo) by lia.
        apply (IH (S (lo + (hi - lo) / 2)) hi Hd) with (j := j);
          [ intros a b Ha Hab Hb; apply Hsort; lia
          | lia | assumption | assumption ].
    -
      assert (Hmid : j < lo + (hi - lo) / 2).
      { destruct (le_lt_dec (lo + (hi - lo) / 2) j) as [Hle|Hlt];
          [|exact Hlt].
        exfalso; apply L1.
        apply leb_trans with (read m j); [|assumption].
        apply Hsort; lia. }
      destruct (dec (lo + (hi - lo) / 2 = lo)) as [Eeq|Eeq]; [exfalso; lia|].
      assert (Hd : S (lo + (hi - lo) / 2 - 1) - lo < S hi - lo) by lia.
      apply (IH lo (lo + (hi - lo) / 2 - 1) Hd) with (j := j);
        [ intros a b Ha Hab Hb; apply Hsort; lia
        | lia | assumption | assumption ].
  Qed.

End BinarySearchProps.


Corollary bsearch_sound_nat (l : list nat) (target lo hi i : nat) :
  bsearch nat Nat.leb (list nat) (nth_default 0) l target lo hi = Some i →
  lo ≤ i ≤ hi ∧ nth_default 0 l i = target.
Proof.
  apply bsearch_sound_eq.
  intros x y Hxy Hyx.
  apply Nat.le_antisymm; apply Nat.leb_le; assumption.
Qed.

Corollary bsearch_complete_nat (l : list nat) (target lo hi : nat) :
  (∀ i j, lo ≤ i → i ≤ j → j ≤ hi →
     nth_default 0 l i ≤ nth_default 0 l j) →
  ∀ j, lo ≤ j ≤ hi → nth_default 0 l j = target →
  ∃ i,
    bsearch nat Nat.leb (list nat) (nth_default 0) l target lo hi = Some i.
Proof.
  intros Hsort j Hj Hjt.
  apply (bsearch_complete nat Nat.leb (list nat) (nth_default 0) l target)
    with (j := j).
  -
    intros x y z Hxy Hyz.
    apply Nat.leb_le; apply Nat.leb_le in Hxy; apply Nat.leb_le in Hyz; lia.
  -
    intros a b Ha Hab Hb; apply Nat.leb_le; auto.
  - exact Hj.
  - rewrite Hjt; apply Nat.leb_refl.
  - rewrite Hjt; apply Nat.leb_refl.
Qed.
