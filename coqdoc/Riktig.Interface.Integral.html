Library Riktig.Interface.Integral

From Coq Require Import Morphisms.
From Riktig Require Import Ops Props Interface.Order Interface.VectorSpace
  Interface.Norm.

Class Integral (F V : Type) `{OrderedField F}
    {VS : VectorSpace F V} {NS : NormedSpace F V} := {
  
  int : (F → V) → F → F → V;
  #[global] int_proper ::
    Proper (pointwise_relation F equ ==> equ ==> equ ==> equ) int;
  int_linear_add : ∀ (f g : F → V) (a b : F),
    int (fun t : F ⇒ f t + g t) a b == int f a b + int g a b;
  int_linear_scale : ∀ (c : F) (f : F → V) (a b : F),
    int (fun t : F ⇒ c *: f t) a b == c *: int f a b;
  int_additivity : ∀ (f : F → V) (a b c : F),
    int f a b + int f b c == int f a c;
  int_const : ∀ (v : V) (a b : F),
    int (fun _ : F ⇒ v) a b == (b + - a) *: v;
  int_bound : ∀ (f : F → V) (a b M : F),
    (∀ t : F, a ≤ t → t ≤ b → ‖ f t ‖ ≤ M) →
    a ≤ b →
    ‖ int f a b ‖ ≤ M × (b + - a)
}.

#[global] Hint Mode Integral - ! - - - - : typeclass_instances.
