Lemma swap_fin_hit : ∀ (n : nat) (i j : Fin.t n), swap_fin i j i = j.
Proof.
  intros n i j.
  unfold swap_fin.
  destruct (Fin.eq_dec i i) as [_ | Hne].
  - reflexivity.
  - destruct (Hne eq_refl).
Qed.

Lemma swap_fin_FSFS_F1 : ∀ (n : nat) (i j : Fin.t n),
  swap_fin (Fin.FS i) (Fin.FS j) Fin.F1 = Fin.F1.
Proof.
  intros n i j.
  unfold swap_fin.
  destruct (Fin.eq_dec (Fin.F1 : Fin.t (S n)) (Fin.FS i)) as [He1 | _].
  - destruct (fin_F1_neq_FS n i He1).
  - destruct (Fin.eq_dec (Fin.F1 : Fin.t (S n)) (Fin.FS j)) as [He2 | _].
    + destruct (fin_F1_neq_FS n j He2).
    + reflexivity.
Qed.

Lemma swap_fin_FSFS_FS : ∀ (n : nat) (i j s : Fin.t n),
  swap_fin (Fin.FS i) (Fin.FS j) (Fin.FS s) = Fin.FS (swap_fin i j s).
Proof.
  intros n i j s.
  unfold swap_fin.
  destruct (Fin.eq_dec (Fin.FS s) (Fin.FS i)) as [HeqI | HneI].
  - destruct (Fin.eq_dec s i) as [_ | Hni].
    + reflexivity.
    + destruct (Hni (Fin.FS_inj s i HeqI)).
  - destruct (Fin.eq_dec s i) as [Heqi | _].
    + destruct (HneI (f_equal Fin.FS Heqi)).
    + destruct (Fin.eq_dec (Fin.FS s) (Fin.FS j)) as [HeqJ | HneJ].
      × destruct (Fin.eq_dec s j) as [_ | Hnj].
        -- reflexivity.
        -- destruct (Hnj (Fin.FS_inj s j HeqJ)).
      × destruct (Fin.eq_dec s j) as [Heqj | _].
        -- destruct (HneJ (f_equal Fin.FS Heqj)).
        -- reflexivity.
Qed.

  Lemma mul_interchange : ∀ a b c d : F,
    (a × b) × (c × d) == (a × c) × (b × d).
  Proof.
    intros a b c d.
    transitivity (((a × b) × c) × d).
    - symmetry. exact (ring_mul_assoc (a × b) c d).
    - transitivity (((a × c) × b) × d).
      + exact (ring_mul_proper (a × b × c) (a × c × b) (mul_assoc_comm_r a b c)
                 d d (reflexivity d)).
      + exact (ring_mul_assoc (a × c) b d).
  Qed.

  Lemma cancel_unit : ∀ f g x : F, f × g == 1 → f × x == 0 → x == 0.
  Proof.
    intros f g x Hfg Hfx.
    transitivity ((1 : F) × x).
    - symmetry. exact (ring_mul_1l x).
    - transitivity ((f × g) × x).
      + symmetry. exact (ring_mul_proper (f × g) 1 Hfg x x (reflexivity x)).
      + transitivity ((g × f) × x).
        × exact (ring_mul_proper (f × g) (g × f) (field_mul_comm f g)
                   x x (reflexivity x)).
        × transitivity (g × (f × x)).
          -- exact (ring_mul_assoc g f x).
          -- transitivity (g × (0 : F)).
             ++ exact (ring_mul_proper g g (reflexivity g) (f × x) 0 Hfx).
             ++ exact (mul_0_r g).
  Qed.


  Theorem det_zero_row : ∀ (n : nat) (A : Matrix n n F) (i : Fin.t n),
    (∀ j : Fin.t n, A i j == 0) → detM A == 0.
  Proof.
    intros n A i Hz.
    destruct n as [| p].
    - exact (Fin.case0 (fun i0 : Fin.t 0 ⇒ detM A == 0) i).
    - transitivity (detM (upd_row A i (fun k ⇒ 0 × A i k))).
      + apply (detM_proper (S p)).
        intros r c.
        destruct (Fin.eq_dec r i) as [Heq | Hne].
        × rewrite Heq.
          transitivity (0 : F).
          -- exact (Hz c).
          -- symmetry.
             transitivity ((0 : F) × A i c).
             ++ exact (upd_row_at (S p) A i (fun k ⇒ 0 × A i k) c).
             ++ exact (mul_0_l (A i c)).
        × symmetry.
          exact (upd_row_ne (S p) A i r (fun k ⇒ 0 × A i k) c Hne).
      + transitivity ((0 : F) × detM (upd_row A i (A i))).
        × exact (det_rowI_scale p i A 0 (A i)).
        × exact (mul_0_l (detM (upd_row A i (A i)))).
  Qed.

  Lemma mat_mul_zero_row : ∀ (n p : nat) (A : Matrix n n F)
      (B : Matrix n p F) (i : Fin.t n),
    (∀ j : Fin.t n, A i j == 0) →
    ∀ c : Fin.t p, mat_mul A B i c == 0.
  Proof.
    intros n p A B i Hz c.
    transitivity (bigopF (fun k : Fin.t n ⇒ (0 : F))).
    - apply bigopF_ext. intros k.
      transitivity ((0 : F) × B k c).
      + exact (ring_mul_proper (A i k) 0 (Hz k) (B k c) (B k c)
                 (reflexivity (B k c))).
      + exact (mul_0_l (B k c)).
    - exact (bigopF_zero n).
  Qed.


  Theorem det_upper_triangular : ∀ (n : nat) (A : Matrix n n F),
    (∀ i j : Fin.t n, (fin_to_nat j < fin_to_nat i)%nat → A i j == 0) →
    detM A == prodF (fun i ⇒ A i i).
  Proof.
    intros n; induction n as [| p IH]; intros A Hup.
    - reflexivity.
    - assert (Hcol : ∀ s : Fin.t p, A (Fin.FS s) Fin.F1 == 0).
      { intros s. apply (Hup (Fin.FS s) Fin.F1).
        rewrite (fin_to_nat_F1 p), (fin_to_nat_FS p s). lia. }
      rewrite (detM_S p A),
        (bigopF_S p (fun i ⇒ sign_at i × (A i Fin.F1 × detM (minor A i)))),
        (bigopF_ext p
           (fun s : Fin.t p ⇒ sign_at (Fin.FS s)
              × (A (Fin.FS s) Fin.F1 × detM (minor A (Fin.FS s))))
           (fun _ : Fin.t p ⇒ (0 : F))
           (fun s ⇒ det_term_col0 p A (Fin.FS s) (Hcol s))),
        (bigopF_zero p),
        (sign_at_F1 p),
        (ring_mul_1l (A Fin.F1 Fin.F1 × detM (minor A Fin.F1))),
        (mon_idr (A Fin.F1 Fin.F1 × detM (minor A Fin.F1))).
      assert (Hmin : ∀ s c : Fin.t p,
        (fin_to_nat c < fin_to_nat s)%nat → minor A Fin.F1 s c == 0).
      { intros s c Hlt.
        rewrite (minor_def p A Fin.F1 s c), (skip_F1 p s).
        apply (Hup (Fin.FS s) (Fin.FS c)).
        rewrite (fin_to_nat_FS p s), (fin_to_nat_FS p c). lia. }
      rewrite (IH (minor A Fin.F1) Hmin).
      assert (Hpe : ∀ s : Fin.t p,
        minor A Fin.F1 s s == A (Fin.FS s) (Fin.FS s)).
      { intros s.
        rewrite (minor_def p A Fin.F1 s s), (skip_F1 p s).
        reflexivity. }
      rewrite (prodF_ext p (fun s ⇒ minor A Fin.F1 s s)
                 (fun s ⇒ A (Fin.FS s) (Fin.FS s)) Hpe).
      symmetry.
      exact (prodF_S p (fun i ⇒ A i i)).
  Qed.


  Inductive ElemOp (n : nat) : Type :=
  | eop_swap (i j : Fin.t n) (Hij : i ≠ j)
  | eop_scale (i : Fin.t n) (c d : F) (Hcd : c × d == 1)
  | eop_add (i j : Fin.t n) (c : F) (Hij : i ≠ j).

  Definition eop_mat {n : nat} (op : ElemOp n) : Matrix n n F :=
    match op with
    | eop_swap _ i j _ ⇒ elem_swap i j
    | eop_scale _ i c _ _ ⇒ elem_scale i c
    | eop_add _ i j c _ ⇒ elem_add i j c
    end.

  Definition eop_factor {n : nat} (op : ElemOp n) : F :=
    match op with
    | eop_swap _ _ _ _ ⇒ - (1)
    | eop_scale _ _ c _ _ ⇒ c
    | eop_add _ _ _ _ _ ⇒ 1
    end.

  Definition eop_cofactor {n : nat} (op : ElemOp n) : F :=
    match op with
    | eop_swap _ _ _ _ ⇒ - (1)
    | eop_scale _ _ _ d _ ⇒ d
    | eop_add _ _ _ _ _ ⇒ 1
    end.

  Definition eop_apply {n : nat} (op : ElemOp n) (M : Matrix n n F)
      : Matrix n n F :=
    mat_mul (eop_mat op) M.

  Lemma eop_factor_cofactor : ∀ (n : nat) (op : ElemOp n),
    eop_factor op × eop_cofactor op == 1.
  Proof.
    intros n op; destruct op as [i j Hij | i c d Hcd | i j c Hij].
    - transitivity ((1 : F) × 1).
      + exact (mul_neg_neg (1 : F) 1).
      + exact (ring_mul_1l (1 : F)).
    - exact Hcd.
    - exact (ring_mul_1l (1 : F)).
  Qed.

  Lemma eop_add_at : ∀ (n : nat) (i j : Fin.t n) (c : F) (Hij : i ≠ j)
      (M : Matrix n n F) (k : Fin.t n),
    eop_apply (eop_add n i j c Hij) M i k == M i k + c × M j k.
  Proof.
    intros n i j c Hij M k.
    exact (elem_add_mul_eq n n i j c M k).
  Qed.

  Lemma eop_add_ne : ∀ (n : nat) (i j : Fin.t n) (c : F) (Hij : i ≠ j)
      (M : Matrix n n F) (r k : Fin.t n),
    r ≠ i → eop_apply (eop_add n i j c Hij) M r k == M r k.
  Proof.
    intros n i j c Hij M r k Hr.
    exact (elem_add_mul_ne n n i j r c M k Hr).
  Qed.

  Lemma elem_scale_mul_eq_upd_row : ∀ (n : nat) (i : Fin.t n) (c : F)
      (M : Matrix n n F),
    mat_mul (elem_scale i c) M == upd_row M i (fun k ⇒ c × M i k).
  Proof.
    intros n i c M r0 k.
    destruct (Fin.eq_dec r0 i) as [Heq | Hne].
    - rewrite Heq.
      transitivity (c × M i k).
      + exact (elem_scale_mul_eq n n i c M k).
      + symmetry. exact (upd_row_at n M i (fun k0 ⇒ c × M i k0) k).
    - transitivity (M r0 k).
      + exact (elem_scale_mul_ne n n i r0 c M k Hne).
      + symmetry. exact (upd_row_ne n M i r0 (fun k0 ⇒ c × M i k0) k Hne).
  Qed.

  Lemma det_eop_scale : ∀ (n : nat) (i : Fin.t n) (c : F)
      (M : Matrix n n F),
    detM (mat_mul (elem_scale i c) M) == c × detM M.
  Proof.
    intros n i c M.
    destruct n as [| p].
    - exact (Fin.case0
        (fun i0 : Fin.t 0 ⇒ detM (mat_mul (elem_scale i0 c) M) == c × detM M)
        i).
    - rewrite (detM_proper (S p) (mat_mul (elem_scale i c) M)
                 (upd_row M i (fun k ⇒ c × M i k))
                 (elem_scale_mul_eq_upd_row (S p) i c M)).
      rewrite (det_rowI_scale p i M c (M i)).
      rewrite (detM_proper (S p) (upd_row M i (M i)) M (upd_row_self (S p) M i)).
      reflexivity.
  Qed.

  Theorem det_eop_apply : ∀ (n : nat) (op : ElemOp n) (M : Matrix n n F),
    detM (eop_apply op M) == eop_factor op × detM M.
  Proof.
    intros n op M; destruct op as [i j Hij | i c d Hcd | i j c Hij].
    - transitivity (- detM M).
      + exact (det_elem_swap n i j Hij M).
      + symmetry. exact (mul_neg1_l (detM M)).
    - exact (det_eop_scale n i c M).
    - transitivity (detM M).
      + exact (det_elem_add n i j c M Hij).
      + symmetry. exact (ring_mul_1l (detM M)).
  Qed.


  Fixpoint eops_apply {n : nat} (l : list (ElemOp n)) (M : Matrix n n F)
      {struct l} : Matrix n n F :=
    match l with
    | nil ⇒ M
    | op :: l' ⇒ eops_apply l' (eop_apply op M)
    end.

  Fixpoint eops_factor {n : nat} (l : list (ElemOp n)) {struct l} : F :=
    match l with
    | nil ⇒ 1
    | op :: l' ⇒ eop_factor op × eops_factor l'
    end.

  Fixpoint eops_cofactor {n : nat} (l : list (ElemOp n)) {struct l} : F :=
    match l with
    | nil ⇒ 1
    | op :: l' ⇒ eop_cofactor op × eops_cofactor l'
    end.

  Lemma eop_apply_proper : ∀ (n : nat) (op : ElemOp n)
      (M M' : Matrix n n F),
    M == M' → eop_apply op M == eop_apply op M'.
  Proof.
    intros n op M M' HM.
    exact (mat_mul_proper n n n (eop_mat op) (eop_mat op)
             (fun i j ⇒ reflexivity (eop_mat op i j)) M M' HM).
  Qed.

  Lemma eops_apply_proper : ∀ (n : nat) (l : list (ElemOp n))
      (M M' : Matrix n n F),
    M == M' → eops_apply l M == eops_apply l M'.
  Proof.
    intros n l; induction l as [| op l' IH]; intros M M' HM.
    - exact HM.
    - exact (IH (eop_apply op M) (eop_apply op M')
              (eop_apply_proper n op M M' HM)).
  Qed.

  Lemma eops_apply_app : ∀ (n : nat) (l1 l2 : list (ElemOp n))
      (M : Matrix n n F),
    eops_apply (l1 ++ l2) M = eops_apply l2 (eops_apply l1 M).
  Proof.
    intros n l1; induction l1 as [| op l1' IH]; intros l2 M.
    - reflexivity.
    - exact (IH l2 (eop_apply op M)).
  Qed.

  Theorem det_eops_apply : ∀ (n : nat) (l : list (ElemOp n))
      (M : Matrix n n F),
    detM (eops_apply l M) == eops_factor l × detM M.
  Proof.
    intros n l; induction l as [| op l' IH]; intros M.
    - symmetry. exact (ring_mul_1l (detM M)).
    - transitivity (eops_factor l' × detM (eop_apply op M)).
      + exact (IH (eop_apply op M)).
      + transitivity (eops_factor l' × (eop_factor op × detM M)).
        × exact (ring_mul_proper (eops_factor l') (eops_factor l')
                   (reflexivity (eops_factor l'))
                   (detM (eop_apply op M)) (eop_factor op × detM M)
                   (det_eop_apply n op M)).
        × transitivity (eop_factor op × (eops_factor l' × detM M)).
          -- exact (mul_comm_swap (eops_factor l') (eop_factor op) (detM M)).
          -- symmetry.
             exact (ring_mul_assoc (eop_factor op) (eops_factor l') (detM M)).
  Qed.

  Lemma eops_factor_cofactor : ∀ (n : nat) (l : list (ElemOp n)),
    eops_factor l × eops_cofactor l == 1.
  Proof.
    intros n l; induction l as [| op l' IH].
    - exact (ring_mul_1l (1 : F)).
    - transitivity ((eop_factor op × eop_cofactor op)
                    × (eops_factor l' × eops_cofactor l')).
      + exact (mul_interchange (eop_factor op) (eops_factor l')
                 (eop_cofactor op) (eops_cofactor l')).
      + transitivity ((1 : F) × 1).
        × exact (ring_mul_proper (eop_factor op × eop_cofactor op) 1
                   (eop_factor_cofactor n op)
                   (eops_factor l' × eops_cofactor l') 1 IH).
        × exact (ring_mul_1l (1 : F)).
  Qed.

  Lemma eops_apply_mul : ∀ (n : nat) (l : list (ElemOp n))
      (A B : Matrix n n F),
    eops_apply l (mat_mul A B) == mat_mul (eops_apply l A) B.
  Proof.
    intros n l; induction l as [| op l' IH]; intros A B.
    - reflexivity.
    - transitivity (eops_apply l' (mat_mul (eop_apply op A) B)).
      + apply (eops_apply_proper n l').
        intros i k.
        symmetry.
        exact (mat_mul_assoc n n n n (eop_mat op) A B i k).
      + exact (IH (eop_apply op A) B).
  Qed.


  Definition subm {p : nat} (M : Matrix (S p) (S p) F) : Matrix p p F :=
    fun s c ⇒ M (Fin.FS s) (Fin.FS c).

  Definition lift_eop {p : nat} (op : ElemOp p) : ElemOp (S p) :=
    match op with
    | eop_swap _ i j Hij ⇒
        eop_swap (S p) (Fin.FS i) (Fin.FS j)
          (fun Heq ⇒ Hij (Fin.FS_inj i j Heq))
    | eop_scale _ i c d Hcd ⇒ eop_scale (S p) (Fin.FS i) c d Hcd
    | eop_add _ i j c Hij ⇒
        eop_add (S p) (Fin.FS i) (Fin.FS j) c
          (fun Heq ⇒ Hij (Fin.FS_inj i j Heq))
    end.

  Lemma lift_eop_row_F1 : ∀ (p : nat) (op : ElemOp p)
      (M : Matrix (S p) (S p) F) (j : Fin.t (S p)),
    eop_apply (lift_eop op) M Fin.F1 j == M Fin.F1 j.
  Proof.
    intros p op M j; destruct op as [i1 i2 Hij | i1 c d Hcd | i1 i2 c Hij].
    - transitivity (M (swap_fin (Fin.FS i1) (Fin.FS i2) Fin.F1) j).
      + exact (elem_swap_mul (S p) (S p) (Fin.FS i1) (Fin.FS i2) M Fin.F1 j).
      + rewrite (swap_fin_FSFS_F1 p i1 i2). reflexivity.
    - exact (elem_scale_mul_ne (S p) (S p) (Fin.FS i1) Fin.F1 c M j
               (fin_F1_neq_FS p i1)).
    - exact (elem_add_mul_ne (S p) (S p) (Fin.FS i1) (Fin.FS i2) Fin.F1 c M j
               (fin_F1_neq_FS p i1)).
  Qed.

  Lemma lift_eop_sub : ∀ (p : nat) (op : ElemOp p)
      (M : Matrix (S p) (S p) F) (s c : Fin.t p),
    eop_apply (lift_eop op) M (Fin.FS s) (Fin.FS c) == eop_apply op (subm M) s c.
  Proof.
    intros p op M s c; destruct op as [i1 i2 Hij | i1 c0 d Hcd | i1 i2 c0 Hij].
    - transitivity (M (swap_fin (Fin.FS i1) (Fin.FS i2) (Fin.FS s)) (Fin.FS c)).
      + exact (elem_swap_mul (S p) (S p) (Fin.FS i1) (Fin.FS i2) M
                 (Fin.FS s) (Fin.FS c)).
      + rewrite (swap_fin_FSFS_FS p i1 i2 s).
        symmetry.
        exact (elem_swap_mul p p i1 i2 (subm M) s c).
    - destruct (Fin.eq_dec s i1) as [Heq | Hne].
      + rewrite Heq.
        transitivity (c0 × M (Fin.FS i1) (Fin.FS c)).
        × exact (elem_scale_mul_eq (S p) (S p) (Fin.FS i1) c0 M (Fin.FS c)).
        × symmetry. exact (elem_scale_mul_eq p p i1 c0 (subm M) c).
      + transitivity (M (Fin.FS s) (Fin.FS c)).
        × exact (elem_scale_mul_ne (S p) (S p) (Fin.FS i1) (Fin.FS s) c0 M
                   (Fin.FS c) (fun Heq ⇒ Hne (Fin.FS_inj s i1 Heq))).
        × symmetry. exact (elem_scale_mul_ne p p i1 s c0 (subm M) c Hne).
    - destruct (Fin.eq_dec s i1) as [Heq | Hne].
      + rewrite Heq.
        transitivity (M (Fin.FS i1) (Fin.FS c) + c0 × M (Fin.FS i2) (Fin.FS c)).
        × exact (elem_add_mul_eq (S p) (S p) (Fin.FS i1) (Fin.FS i2) c0 M
                   (Fin.FS c)).
        × symmetry. exact (elem_add_mul_eq p p i1 i2 c0 (subm M) c).
      + transitivity (M (Fin.FS s) (Fin.FS c)).
        × exact (elem_add_mul_ne (S p) (S p) (Fin.FS i1) (Fin.FS i2)
                   (Fin.FS s) c0 M (Fin.FS c)
                   (fun Heq ⇒ Hne (Fin.FS_inj s i1 Heq))).
        × symmetry. exact (elem_add_mul_ne p p i1 i2 s c0 (subm M) c Hne).
  Qed.

  Lemma lift_eop_col0 : ∀ (p : nat) (op : ElemOp p)
      (M : Matrix (S p) (S p) F),
    (∀ t : Fin.t p, M (Fin.FS t) Fin.F1 == 0) →
    ∀ s : Fin.t p, eop_apply (lift_eop op) M (Fin.FS s) Fin.F1 == 0.
  Proof.
    intros p op M Hz s; destruct op as [i1 i2 Hij | i1 c d Hcd | i1 i2 c Hij].
    - transitivity (M (swap_fin (Fin.FS i1) (Fin.FS i2) (Fin.FS s)) Fin.F1).
      + exact (elem_swap_mul (S p) (S p) (Fin.FS i1) (Fin.FS i2) M
                 (Fin.FS s) Fin.F1).
      + rewrite (swap_fin_FSFS_FS p i1 i2 s).
        exact (Hz (swap_fin i1 i2 s)).
    - destruct (Fin.eq_dec s i1) as [Heq | Hne].
      + rewrite Heq.
        transitivity (c × M (Fin.FS i1) Fin.F1).
        × exact (elem_scale_mul_eq (S p) (S p) (Fin.FS i1) c M Fin.F1).
        × transitivity (c × (0 : F)).
          -- exact (ring_mul_proper c c (reflexivity c)
                      (M (Fin.FS i1) Fin.F1) 0 (Hz i1)).
          -- exact (mul_0_r c).
      + transitivity (M (Fin.FS s) Fin.F1).
        × exact (elem_scale_mul_ne (S p) (S p) (Fin.FS i1) (Fin.FS s) c M Fin.F1
                   (fun Heq ⇒ Hne (Fin.FS_inj s i1 Heq))).
        × exact (Hz s).
    - destruct (Fin.eq_dec s i1) as [Heq | Hne].
      + rewrite Heq.
        transitivity (M (Fin.FS i1) Fin.F1 + c × M (Fin.FS i2) Fin.F1).
        × exact (elem_add_mul_eq (S p) (S p) (Fin.FS i1) (Fin.FS i2) c M Fin.F1).
        × transitivity ((0 : F) + c × (0 : F)).
          -- exact (mon_add_proper (M (Fin.FS i1) Fin.F1) 0 (Hz i1)
                      (c × M (Fin.FS i2) Fin.F1) (c × 0)
                      (ring_mul_proper c c (reflexivity c)
                         (M (Fin.FS i2) Fin.F1) 0 (Hz i2))).
          -- transitivity ((0 : F) + 0).
             ++ exact (mon_add_proper (0 : F) 0 (reflexivity (0 : F))
                         (c × 0) 0 (mul_0_r c)).
             ++ exact (mon_idl (0 : F)).
      + transitivity (M (Fin.FS s) Fin.F1).
        × exact (elem_add_mul_ne (S p) (S p) (Fin.FS i1) (Fin.FS i2)
                   (Fin.FS s) c M Fin.F1
                   (fun Heq ⇒ Hne (Fin.FS_inj s i1 Heq))).
        × exact (Hz s).
  Qed.


  Lemma lift_eops_row_F1 : ∀ (p : nat) (l : list (ElemOp p))
      (M : Matrix (S p) (S p) F) (j : Fin.t (S p)),
    eops_apply (map lift_eop l) M Fin.F1 j == M Fin.F1 j.
  Proof.
    intros p l; induction l as [| op l' IH]; intros M j.
    - reflexivity.
    - transitivity (eop_apply (lift_eop op) M Fin.F1 j).
      + exact (IH (eop_apply (lift_eop op) M) j).
      + exact (lift_eop_row_F1 p op M j).
  Qed.

  Lemma lift_eops_sub : ∀ (p : nat) (l : list (ElemOp p))
      (M : Matrix (S p) (S p) F) (s c : Fin.t p),
    eops_apply (map lift_eop l) M (Fin.FS s) (Fin.FS c)
      == eops_apply l (subm M) s c.
  Proof.
    intros p l; induction l as [| op l' IH]; intros M s c.
    - reflexivity.
    - transitivity (eops_apply l' (subm (eop_apply (lift_eop op) M)) s c).
      + exact (IH (eop_apply (lift_eop op) M) s c).
      + exact (eops_apply_proper p l' (subm (eop_apply (lift_eop op) M))
                 (eop_apply op (subm M))
                 (fun s0 c0 ⇒ lift_eop_sub p op M s0 c0) s c).
  Qed.

  Lemma lift_eops_col0 : ∀ (p : nat) (l : list (ElemOp p))
      (M : Matrix (S p) (S p) F),
    (∀ t : Fin.t p, M (Fin.FS t) Fin.F1 == 0) →
    ∀ s : Fin.t p, eops_apply (map lift_eop l) M (Fin.FS s) Fin.F1 == 0.
  Proof.
    intros p l; induction l as [| op l' IH]; intros M Hz s.
    - exact (Hz s).
    - exact (IH (eop_apply (lift_eop op) M) (lift_eop_col0 p op M Hz) s).
  Qed.


  Fixpoint clear_col_ops {p : nat} (M0 : Matrix (S p) (S p) F)
      (l : list (Fin.t p)) {struct l} : list (ElemOp (S p)) :=
    match l with
    | nil ⇒ nil
    | s :: l' ⇒
        eop_add (S p) (Fin.FS s) Fin.F1 (- M0 (Fin.FS s) Fin.F1)
          (fin_FS_neq_F1 p s)
        :: clear_col_ops M0 l'
    end.

  Lemma clear_col_spec : ∀ (p : nat) (l : list (Fin.t p))
      (M0 M : Matrix (S p) (S p) F),
    NoDup l →
    M Fin.F1 Fin.F1 == 1 →
    (∀ s : Fin.t p, In s l → M (Fin.FS s) Fin.F1 == M0 (Fin.FS s) Fin.F1) →
    (∀ j : Fin.t (S p),
       eops_apply (clear_col_ops M0 l) M Fin.F1 j == M Fin.F1 j)
    ∧ (∀ s : Fin.t p, In s l →
          eops_apply (clear_col_ops M0 l) M (Fin.FS s) Fin.F1 == 0)
    ∧ (∀ (s : Fin.t p) (j : Fin.t (S p)), ¬ In s l →
          eops_apply (clear_col_ops M0 l) M (Fin.FS s) j == M (Fin.FS s) j).
  Proof.
    intros p l; induction l as [| s0 l' IH]; intros M0 M Hnd H11 Horig.
    - split; [intros j; reflexivity |
        split; [intros s Hin; destruct Hin | intros s j _; reflexivity]].
    - inversion Hnd as [| x xs Hnotin Hnd' Hxs]; subst.
      set (M1 := eop_apply
        (eop_add (S p) (Fin.FS s0) Fin.F1 (- M0 (Fin.FS s0) Fin.F1)
           (fin_FS_neq_F1 p s0)) M).
      assert (HM1F1 : ∀ j : Fin.t (S p), M1 Fin.F1 j == M Fin.F1 j).
      { intros j.
        exact (eop_add_ne (S p) (Fin.FS s0) Fin.F1 (- M0 (Fin.FS s0) Fin.F1)
                 (fin_FS_neq_F1 p s0) M Fin.F1 j (fin_F1_neq_FS p s0)). }
      assert (HM1ne : ∀ (s : Fin.t p) (j : Fin.t (S p)),
        s ≠ s0 → M1 (Fin.FS s) j == M (Fin.FS s) j).
      { intros s j Hs.
        exact (eop_add_ne (S p) (Fin.FS s0) Fin.F1 (- M0 (Fin.FS s0) Fin.F1)
                 (fin_FS_neq_F1 p s0) M (Fin.FS s) j
                 (fun Heq ⇒ Hs (Fin.FS_inj s s0 Heq))). }
      assert (HM1s0 : M1 (Fin.FS s0) Fin.F1 == 0).
      { transitivity (M (Fin.FS s0) Fin.F1
                      + (- M0 (Fin.FS s0) Fin.F1) × M Fin.F1 Fin.F1).
        - exact (eop_add_at (S p) (Fin.FS s0) Fin.F1 (- M0 (Fin.FS s0) Fin.F1)
                   (fin_FS_neq_F1 p s0) M Fin.F1).
        - rewrite H11, (ring_mul_1r (- M0 (Fin.FS s0) Fin.F1)),
            (Horig s0 (in_eq s0 l')).
          exact (grp_invr (M0 (Fin.FS s0) Fin.F1)). }
      assert (H11' : M1 Fin.F1 Fin.F1 == 1).
      { transitivity (M Fin.F1 Fin.F1); [exact (HM1F1 Fin.F1) | exact H11]. }
      assert (Horig' : ∀ s : Fin.t p, In s l' →
        M1 (Fin.FS s) Fin.F1 == M0 (Fin.FS s) Fin.F1).
      { intros s Hin.
        assert (Hs : s ≠ s0)
          by (intros He; apply Hnotin; rewrite <- He; exact Hin).
        transitivity (M (Fin.FS s) Fin.F1).
        - exact (HM1ne s Fin.F1 Hs).
        - exact (Horig s (in_cons s0 s l' Hin)). }
      destruct (IH M0 M1 Hnd' H11' Horig') as [IH1 [IH2 IH3]].
      split; [| split].
      + intros j.
        transitivity (M1 Fin.F1 j); [exact (IH1 j) | exact (HM1F1 j)].
      + intros s Hin. destruct Hin as [Heq | Hin'].
        × rewrite <- Heq.
          transitivity (M1 (Fin.FS s0) Fin.F1);
            [exact (IH3 s0 Fin.F1 Hnotin) | exact HM1s0].
        × exact (IH2 s Hin').
      + intros s j Hnin.
        assert (Hs : s ≠ s0)
          by (intros He; apply Hnin; rewrite He; exact (in_eq s0 l')).
        assert (Hnin' : ¬ In s l')
          by (intros Hi; apply Hnin; exact (in_cons s0 s l' Hi)).
        transitivity (M1 (Fin.FS s) j);
          [exact (IH3 s j Hnin') | exact (HM1ne s j Hs)].
  Qed.


  Fixpoint clear_row_ops {p : nat} (M0 : Matrix (S p) (S p) F)
      (l : list (Fin.t p)) {struct l} : list (ElemOp (S p)) :=
    match l with
    | nil ⇒ nil
    | c :: l' ⇒
        eop_add (S p) Fin.F1 (Fin.FS c) (- M0 Fin.F1 (Fin.FS c))
          (fin_F1_neq_FS p c)
        :: clear_row_ops M0 l'
    end.

  Lemma clear_row_spec : ∀ (p : nat) (l : list (Fin.t p))
      (M0 M : Matrix (S p) (S p) F),
    NoDup l →
    (∀ (s : Fin.t p) (j : Fin.t (S p)),
       M (Fin.FS s) j == @mat_id F FF (S p) (Fin.FS s) j) →
    (∀ c : Fin.t p, In c l → M Fin.F1 (Fin.FS c) == M0 Fin.F1 (Fin.FS c)) →
    (∀ (s : Fin.t p) (j : Fin.t (S p)),
       eops_apply (clear_row_ops M0 l) M (Fin.FS s) j == M (Fin.FS s) j)
    ∧ (eops_apply (clear_row_ops M0 l) M Fin.F1 Fin.F1 == M Fin.F1 Fin.F1)
    ∧ (∀ c : Fin.t p, In c l →
          eops_apply (clear_row_ops M0 l) M Fin.F1 (Fin.FS c) == 0)
    ∧ (∀ c : Fin.t p, ¬ In c l →
          eops_apply (clear_row_ops M0 l) M Fin.F1 (Fin.FS c)
            == M Fin.F1 (Fin.FS c)).
  Proof.
    intros p l; induction l as [| c0 l' IH]; intros M0 M Hnd Hid Horig.
    - split; [intros s j; reflexivity |
        split; [reflexivity |
          split; [intros c Hin; destruct Hin | intros c _; reflexivity]]].
    - inversion Hnd as [| x xs Hnotin Hnd' Hxs]; subst.
      set (M1 := eop_apply
        (eop_add (S p) Fin.F1 (Fin.FS c0) (- M0 Fin.F1 (Fin.FS c0))
           (fin_F1_neq_FS p c0)) M).
      assert (HM1FS : ∀ (s : Fin.t p) (j : Fin.t (S p)),
        M1 (Fin.FS s) j == M (Fin.FS s) j).
      { intros s j.
        exact (eop_add_ne (S p) Fin.F1 (Fin.FS c0) (- M0 Fin.F1 (Fin.FS c0))
                 (fin_F1_neq_FS p c0) M (Fin.FS s) j (fin_FS_neq_F1 p s)). }
      assert (HM1F1 : ∀ j : Fin.t (S p),
        M1 Fin.F1 j == M Fin.F1 j + (- M0 Fin.F1 (Fin.FS c0)) × M (Fin.FS c0) j).
      { intros j.
        exact (eop_add_at (S p) Fin.F1 (Fin.FS c0) (- M0 Fin.F1 (Fin.FS c0))
                 (fin_F1_neq_FS p c0) M j). }
      assert (HM1_11 : M1 Fin.F1 Fin.F1 == M Fin.F1 Fin.F1).
      { transitivity (M Fin.F1 Fin.F1
                      + (- M0 Fin.F1 (Fin.FS c0)) × M (Fin.FS c0) Fin.F1).
        - exact (HM1F1 Fin.F1).
        - transitivity (M Fin.F1 Fin.F1 + (- M0 Fin.F1 (Fin.FS c0)) × (0 : F)).
          + apply mon_add_proper; [reflexivity |].
            apply ring_mul_proper; [reflexivity |].
            transitivity (@mat_id F FF (S p) (Fin.FS c0) Fin.F1).
            × exact (Hid c0 Fin.F1).
            × exact (mat_id_col_0 p c0).
          + transitivity (M Fin.F1 Fin.F1 + (0 : F)).
            × apply mon_add_proper;
                [reflexivity | exact (mul_0_r (- M0 Fin.F1 (Fin.FS c0)))].
            × exact (mon_idr (M Fin.F1 Fin.F1)). }
      assert (HM1_at : M1 Fin.F1 (Fin.FS c0) == 0).
      { transitivity (M Fin.F1 (Fin.FS c0)
                      + (- M0 Fin.F1 (Fin.FS c0)) × M (Fin.FS c0) (Fin.FS c0)).
        - exact (HM1F1 (Fin.FS c0)).
        - transitivity (M Fin.F1 (Fin.FS c0)
                        + (- M0 Fin.F1 (Fin.FS c0)) × (1 : F)).
          + apply mon_add_proper; [reflexivity |].
            apply ring_mul_proper; [reflexivity |].
            transitivity (@mat_id F FF (S p) (Fin.FS c0) (Fin.FS c0)).
            × exact (Hid c0 (Fin.FS c0)).
            × exact (mat_id_diag (S p) (Fin.FS c0)).
          + transitivity (M Fin.F1 (Fin.FS c0) + - M0 Fin.F1 (Fin.FS c0)).
            × apply mon_add_proper;
                [reflexivity | exact (ring_mul_1r (- M0 Fin.F1 (Fin.FS c0)))].
            × transitivity (M0 Fin.F1 (Fin.FS c0) + - M0 Fin.F1 (Fin.FS c0)).
              -- apply mon_add_proper;
                   [exact (Horig c0 (in_eq c0 l')) | reflexivity].
              -- exact (grp_invr (M0 Fin.F1 (Fin.FS c0))). }
      assert (HM1_ne : ∀ c : Fin.t p, c ≠ c0 →
        M1 Fin.F1 (Fin.FS c) == M Fin.F1 (Fin.FS c)).
      { intros c Hc.
        transitivity (M Fin.F1 (Fin.FS c)
                      + (- M0 Fin.F1 (Fin.FS c0)) × M (Fin.FS c0) (Fin.FS c)).
        - exact (HM1F1 (Fin.FS c)).
        - transitivity (M Fin.F1 (Fin.FS c)
                        + (- M0 Fin.F1 (Fin.FS c0)) × (0 : F)).
          + apply mon_add_proper; [reflexivity |].
            apply ring_mul_proper; [reflexivity |].
            transitivity (@mat_id F FF (S p) (Fin.FS c0) (Fin.FS c)).
            × exact (Hid c0 (Fin.FS c)).
            × exact (e_at_neq (S p) (Fin.FS c0) (Fin.FS c)
                       (fun Heq ⇒ Hc (eq_sym (Fin.FS_inj c0 c Heq)))).
          + transitivity (M Fin.F1 (Fin.FS c) + (0 : F)).
            × apply mon_add_proper;
                [reflexivity | exact (mul_0_r (- M0 Fin.F1 (Fin.FS c0)))].
            × exact (mon_idr (M Fin.F1 (Fin.FS c))). }
      assert (Hid' : ∀ (s : Fin.t p) (j : Fin.t (S p)),
        M1 (Fin.FS s) j == @mat_id F FF (S p) (Fin.FS s) j).
      { intros s j.
        transitivity (M (Fin.FS s) j); [exact (HM1FS s j) | exact (Hid s j)]. }
      assert (Horig' : ∀ c : Fin.t p, In c l' →
        M1 Fin.F1 (Fin.FS c) == M0 Fin.F1 (Fin.FS c)).
      { intros c Hin.
        assert (Hc : c ≠ c0)
          by (intros He; apply Hnotin; rewrite <- He; exact Hin).
        transitivity (M Fin.F1 (Fin.FS c)).
        - exact (HM1_ne c Hc).
        - exact (Horig c (in_cons c0 c l' Hin)). }
      destruct (IH M0 M1 Hnd' Hid' Horig') as [IH1 [IH2 [IH3 IH4]]].
      split; [| split; [| split]].
      + intros s j.
        transitivity (M1 (Fin.FS s) j); [exact (IH1 s j) | exact (HM1FS s j)].
      + transitivity (M1 Fin.F1 Fin.F1); [exact IH2 | exact HM1_11].
      + intros c Hin. destruct Hin as [Heq | Hin'].
        × rewrite <- Heq.
          transitivity (M1 Fin.F1 (Fin.FS c0));
            [exact (IH4 c0 Hnotin) | exact HM1_at].
        × exact (IH3 c Hin').
      + intros c Hnin.
        assert (Hc : c ≠ c0)
          by (intros He; apply Hnin; rewrite He; exact (in_eq c0 l')).
        assert (Hnin' : ¬ In c l')
          by (intros Hi; apply Hnin; exact (in_cons c0 c l' Hi)).
        transitivity (M1 Fin.F1 (Fin.FS c));
          [exact (IH4 c Hnin') | exact (HM1_ne c Hc)].
  Qed.


  Definition find_nonzero (dec : DecEq F) :
    ∀ (n : nat) (v : Fin.t n → F),
      {i : Fin.t n | v i =/= 0} + {∀ i : Fin.t n, v i == 0}.
  Proof.
    intros n; induction n as [| m IH]; intros v.
    - right. intros i. exact (Fin.case0 (fun i0 : Fin.t 0 ⇒ v i0 == 0) i).
    - destruct (dec (v Fin.F1) 0) as [Hz | Hnz].
      + destruct (IH (fun s ⇒ v (Fin.FS s))) as [[i Hi] | Hall].
        × left. ∃ (Fin.FS i). exact Hi.
        × right. intros i.
          apply (Fin.caseS' i (fun i0 ⇒ v i0 == 0)).
          -- exact Hz.
          -- intros s. exact (Hall s).
      + left. ∃ Fin.F1. exact Hnz.
  Defined.


  Theorem gauss_reduce : ∀ (dec : DecEq F) (n : nat) (A : Matrix n n F),
    ∃ l : list (ElemOp n),
      (eops_apply l A == @mat_id F FF n)
      ∨ (∃ i : Fin.t n, ∀ j : Fin.t n, eops_apply l A i j == 0).
  Proof.
    intros dec n; induction n as [| p IH]; intros A.
    - ∃ (@nil (ElemOp 0)). left.
      intros i.
      exact (Fin.case0
        (fun i0 : Fin.t 0 ⇒ equ (eops_apply nil A i0) (@mat_id F FF 0 i0)) i).
    - destruct (find_nonzero dec (S p) (fun i ⇒ A i Fin.F1)) as
        [[i0 Hi0] | Hall].
      +
        assert (Hstep1 : ∃ l1 : list (ElemOp (S p)),
          eops_apply l1 A Fin.F1 Fin.F1 =/= 0).
        { destruct (Fin.eq_dec i0 Fin.F1) as [He | Hne].
          - ∃ (@nil (ElemOp (S p))).
            rewrite He in Hi0. exact Hi0.
          - assert (Hne' : Fin.F1 ≠ i0)
              by (intros He'; apply Hne; exact (eq_sym He')).
            ∃ (eop_swap (S p) Fin.F1 i0 Hne' :: nil).
            intros Hzero. apply Hi0.
            transitivity (eop_apply (eop_swap (S p) Fin.F1 i0 Hne') A
                            Fin.F1 Fin.F1).
            + symmetry.
              transitivity (A (swap_fin Fin.F1 i0 Fin.F1) Fin.F1).
              × exact (elem_swap_mul (S p) (S p) Fin.F1 i0 A Fin.F1 Fin.F1).
              × rewrite (swap_fin_hit (S p) Fin.F1 i0). reflexivity.
            + exact Hzero. }
        destruct Hstep1 as [l1 Hpiv].
        set (A1 := eops_apply l1 A).
        set (op2 := eop_scale (S p) Fin.F1 (inv (A1 Fin.F1 Fin.F1))
                      (A1 Fin.F1 Fin.F1)
                      (field_invl (A1 Fin.F1 Fin.F1) Hpiv)).
        set (A2 := eop_apply op2 A1).
        assert (H2 : A2 Fin.F1 Fin.F1 == 1).
        { transitivity (inv (A1 Fin.F1 Fin.F1) × A1 Fin.F1 Fin.F1).
          - exact (elem_scale_mul_eq (S p) (S p) Fin.F1
                     (inv (A1 Fin.F1 Fin.F1)) A1 Fin.F1).
          - exact (field_invl (A1 Fin.F1 Fin.F1) Hpiv). }
        destruct (clear_col_spec p (fin_all p) A2 A2 (fin_all_NoDup p) H2
                    (fun s _ ⇒ reflexivity (A2 (Fin.FS s) Fin.F1)))
          as [HC1 [HC2 HC3]].
        set (A3 := eops_apply (clear_col_ops A2 (fin_all p)) A2).
        assert (H3_11 : A3 Fin.F1 Fin.F1 == 1).
        { transitivity (A2 Fin.F1 Fin.F1); [exact (HC1 Fin.F1) | exact H2]. }
        assert (H3col : ∀ s : Fin.t p, A3 (Fin.FS s) Fin.F1 == 0).
        { intros s. exact (HC2 s (fin_all_complete p s)). }
        destruct (IH (subm A3)) as [lsub [Hsid | [isub Hszero]]].
        ×
          set (A4 := eops_apply (map lift_eop lsub) A3).
          assert (H4F1 : ∀ j : Fin.t (S p), A4 Fin.F1 j == A3 Fin.F1 j).
          { intros j. exact (lift_eops_row_F1 p lsub A3 j). }
          assert (H4id : ∀ (s : Fin.t p) (j : Fin.t (S p)),
            A4 (Fin.FS s) j == @mat_id F FF (S p) (Fin.FS s) j).
          { intros s j.
            apply (Fin.caseS' j (fun j0 ⇒
              A4 (Fin.FS s) j0 == @mat_id F FF (S p) (Fin.FS s) j0)).
            - transitivity (0 : F).
              + exact (lift_eops_col0 p lsub A3 H3col s).
              + symmetry. exact (mat_id_col_0 p s).
            - intros c.
              transitivity (eops_apply lsub (subm A3) s c).
              + exact (lift_eops_sub p lsub A3 s c).
              + transitivity (@mat_id F FF p s c).
                × exact (Hsid s c).
                × symmetry. exact (e_FS p s c). }
          destruct (clear_row_spec p (fin_all p) A4 A4 (fin_all_NoDup p) H4id
                      (fun c _ ⇒ reflexivity (A4 Fin.F1 (Fin.FS c))))
            as [HR1 [HR2 [HR3 HR4]]].
          ∃ (l1 ++ (op2 :: nil) ++ clear_col_ops A2 (fin_all p)
                  ++ map lift_eop lsub ++ clear_row_ops A4 (fin_all p)).
          left.
          assert (Etotal :
            eops_apply (l1 ++ (op2 :: nil) ++ clear_col_ops A2 (fin_all p)
                        ++ map lift_eop lsub ++ clear_row_ops A4 (fin_all p)) A
            = eops_apply (clear_row_ops A4 (fin_all p)) A4).
          { repeat rewrite (eops_apply_app (S p)). reflexivity. }
          rewrite Etotal.
          intros r.
          apply (Fin.caseS' r (fun r0 ⇒
            equ (eops_apply (clear_row_ops A4 (fin_all p)) A4 r0)
                (@mat_id F FF (S p) r0))).
          -- intros j.
             apply (Fin.caseS' j (fun j0 ⇒

  Theorem detM_mul_dec : ∀ (dec : DecEq F) (n : nat)
      (A B : Matrix n n F),
    detM (mat_mul A B) == detM A × detM B.
  Proof.
    intros dec n A B.
    destruct (gauss_reduce dec n A) as [l [Hid | [i Hz]]].
    -
      assert (HfA : eops_factor l × detM A == 1).
      { transitivity (detM (eops_apply l A)).
        - symmetry. exact (det_eops_apply n l A).
        - transitivity (detM (@mat_id F FF n)).
          + exact (detM_proper n (eops_apply l A) mat_id Hid).
          + exact (Impl.Det.det_id n). }
      assert (HfAB : eops_factor l × detM (mat_mul A B) == detM B).
      { transitivity (detM (eops_apply l (mat_mul A B))).
        - symmetry. exact (det_eops_apply n l (mat_mul A B)).
        - transitivity (detM (mat_mul (eops_apply l A) B)).
          + exact (detM_proper n (eops_apply l (mat_mul A B))
                     (mat_mul (eops_apply l A) B) (eops_apply_mul n l A B)).
          + transitivity (detM (mat_mul (@mat_id F FF n) B)).
            × exact (detM_proper n (mat_mul (eops_apply l A) B)
                       (mat_mul mat_id B)
                       (mat_mul_proper n n n (eops_apply l A) mat_id Hid
                          B B (fun i0 k ⇒ reflexivity (B i0 k)))).
            × exact (detM_proper n (mat_mul mat_id B) B (mat_mul_1l n n B)). }
      transitivity ((1 : F) × detM (mat_mul A B)).
      + symmetry. exact (ring_mul_1l (detM (mat_mul A B))).
      + transitivity ((eops_factor l × detM A) × detM (mat_mul A B)).
        × symmetry.
          exact (ring_mul_proper (eops_factor l × detM A) 1 HfA
                   (detM (mat_mul A B)) (detM (mat_mul A B))
                   (reflexivity (detM (mat_mul A B)))).
        × transitivity (detM A × (eops_factor l × detM (mat_mul A B))).
          -- transitivity (eops_factor l × (detM A × detM (mat_mul A B))).
             ++ exact (ring_mul_assoc (eops_factor l) (detM A)
                         (detM (mat_mul A B))).
             ++ exact (mul_comm_swap (eops_factor l) (detM A)
                         (detM (mat_mul A B))).
          -- exact (ring_mul_proper (detM A) (detM A) (reflexivity (detM A))
                      (eops_factor l × detM (mat_mul A B)) (detM B) HfAB).
    -
      assert (HfA : eops_factor l × detM A == 0).
      { transitivity (detM (eops_apply l A)).
        - symmetry. exact (det_eops_apply n l A).
        - exact (det_zero_row n (eops_apply l A) i Hz). }
      assert (HfAB : eops_factor l × detM (mat_mul A B) == 0).
      { transitivity (detM (eops_apply l (mat_mul A B))).
        - symmetry. exact (det_eops_apply n l (mat_mul A B)).
        - transitivity (detM (mat_mul (eops_apply l A) B)).
          + exact (detM_proper n (eops_apply l (mat_mul A B))
                     (mat_mul (eops_apply l A) B) (eops_apply_mul n l A B)).
          + exact (det_zero_row n (mat_mul (eops_apply l A) B) i
                     (mat_mul_zero_row n n (eops_apply l A) B i Hz)). }
      assert (HA0 : detM A == 0).
      { exact (cancel_unit (eops_factor l) (eops_cofactor l) (detM A)
                 (eops_factor_cofactor n l) HfA). }
      assert (HAB0 : detM (mat_mul A B) == 0).
      { exact (cancel_unit (eops_factor l) (eops_cofactor l)
                 (detM (mat_mul A B)) (eops_factor_cofactor n l) HfAB). }
      transitivity (0 : F); [exact HAB0 |].
      symmetry.
      transitivity ((0 : F) × detM B).
      + exact (ring_mul_proper (detM A) 0 HA0 (detM B) (detM B)
                 (reflexivity (detM B))).
      + exact (mul_0_l (detM B)).
  Qed.

Corollary detM_mul_Q : ∀ (n : nat) (A B : Matrix n n Q),
