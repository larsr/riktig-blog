Library Riktig.Interface.Monad

From Coq Require Import Morphisms.
From Riktig Require Import Ops.

Class HasBind (M : Type → Type) := bind : ∀ {A B}, M A → (A → M B) → M B.
Class HasReturn (M : Type → Type) := ret : ∀ {A}, A → M A.

Arguments bind {M _ A B} p f.
Arguments ret {M _ A} x.

Infix ">>=" := bind (at level 58, left associativity).
Notation "f >=> g" := (fun x ⇒ bind (f x) g) (at level 55, right associativity).

Class Monad (M : Type → Type) `{∀ A, EqRel (M A)} `{HasBind M} `{HasReturn M} := {
  
  #[global] monad_equiv :: ∀ A, Equivalence (@equ (M A) _);
  
  #[global] bind_proper :: ∀ A B,
    Proper (equ ==> (fun f g ⇒ ∀ x, f x == g x) ==> equ) (@bind M _ A B);
  bind_ret_l : ∀ A B (x : A) (f : A → M B), bind (ret x) f == f x;
  bind_ret_r : ∀ A (m : M A), bind m ret == m;
  bind_assoc : ∀ A B C (m : M A) (f : A → M B) (g : B → M C),
    bind (bind m f) g == bind m (fun x ⇒ bind (f x) g)
}.

#[global] Hint Mode HasBind ! : typeclass_instances.
#[global] Hint Mode HasReturn ! : typeclass_instances.
#[global] Hint Mode Monad ! - - - : typeclass_instances.
