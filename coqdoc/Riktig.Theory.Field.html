From Riktig Require Import Ops Props Interface.Ring Interface.Field
  Interface.FormallyReal Theory.Ring.

Section FieldTheory.
  Context {F : Type} `{Field F}.

  Theorem field_invr : ∀ a : F, a =/= 0 → a × inv a == 1.
  Proof.
    intros a Ha.
    rewrite field_mul_comm.
    exact (field_invl a Ha).
  Qed.

  Theorem mul_cancel_l : ∀ a b c : F,
    a =/= 0 → a × b == a × c → b == c.
  Proof.
    intros a b c Ha Heq.
    rewrite <- (ring_mul_1l b), <- (field_invl a Ha), ring_mul_assoc, Heq,
      <- ring_mul_assoc, (field_invl a Ha), ring_mul_1l.
    reflexivity.
  Qed.

  Theorem field_mul_neq_0 : ∀ a b : F,
    a =/= 0 → b =/= 0 → a × b =/= 0.
  Proof.
    intros a b Ha Hb Heq.
    apply Hb.
    rewrite <- (ring_mul_1l b), <- (field_invl a Ha), ring_mul_assoc, Heq,
      mul_0_r.
    reflexivity.
  Qed.

  Theorem field_mul_div_cancel : ∀ a b : F,
    b =/= 0 → b × (a × inv b) == a.
  Proof.
    intros a b Hb.
    rewrite (field_mul_comm a (inv b)), <- ring_mul_assoc, (field_invr b Hb),
      ring_mul_1l.
    reflexivity.
  Qed.

  Theorem inv_mul_cancel : ∀ a b : F, a =/= 0 → inv a × (a × b) == b.
  Proof.
    intros a b Ha.
    rewrite <- ring_mul_assoc, (field_invl a Ha), ring_mul_1l.
    reflexivity.
  Qed.

  Theorem zero_neq_one : (0 : F) =/= 1.
  Proof.
    intros Heq. apply field_nontrivial. symmetry. exact Heq.
  Qed.

  Theorem mul_comm_swap : ∀ a b c : F, a × (b × c) == b × (a × c).
  Proof.
    intros a b c.
    rewrite <- (ring_mul_assoc a b c), (field_mul_comm a b),
      (ring_mul_assoc b a c).
    reflexivity.
  Qed.

  Theorem mul_assoc_comm_r : ∀ a b c : F, a × b × c == a × c × b.
  Proof.
    intros a b c.
    rewrite ring_mul_assoc, (field_mul_comm b c), <- ring_mul_assoc.
    reflexivity.
  Qed.

End FieldTheory.


Section FormallyRealTheory.
  Context {F : Type} `{FormallyReal F}.

  Lemma fr_sos_r : ∀ a b : F, a × a + b × b == 0 → b == 0.
  Proof.
    intros a b Hab.
    apply (fr_sos b a).
    rewrite (comm (b × b) (a × a)).
    exact Hab.
  Qed.

  Lemma fr_nonzero : ∀ a b : F, a =/= 0 → a × a + b × b =/= 0.
  Proof.
    intros a b Ha Hn.
    apply Ha.
    exact (fr_sos a b Hn).
  Qed.

End FormallyRealTheory.
