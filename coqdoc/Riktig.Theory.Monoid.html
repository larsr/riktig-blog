Library Riktig.Theory.Monoid

From Riktig Require Import Ops Props Interface.Monoid Lib.Calc.

Section MonoidTheory.
  Context {M : Type} `{Monoid M}.

  Theorem id_unique : ∀ e f : M,
    (∀ a, e + a == a) → (∀ a, a + f == a) → e == f.
  Proof.
    intros e f He Hf.
    calc e
      == (e + f) { (Hf e) }
      == f { (He f) }.
  Qed.

  Theorem zero_unique_l : ∀ e : M, (∀ a, e + a == a) → e == 0.
  Proof.
    intros e He.
    exact (id_unique e 0 He idr).
  Qed.

  Theorem zero_unique_r : ∀ e : M, (∀ a, a + e == a) → e == 0.
  Proof.
    intros e He.
    symmetry.
    exact (id_unique 0 e idl He).
  Qed.

End MonoidTheory.

#[global] Instance Monoid_LeftIdUnique M (HM : Monoid M) :
  Unique M equ (fun e : M ⇒ ∀ a, e + a == a).
Proof.
  intros e f He Hf.
  calc e
    == (0 : M) { (zero_unique_l e He) }
    == f { (zero_unique_l f Hf) }.
Qed.
