Library Riktig.Lambda.FreeVars


Require Export Riktig.Lambda.Syntax.
From Coq Require Import Bool Lia Setoid.


Fixpoint free_in (k : nat) (t : tm) {struct t} : Prop :=
  match t with
  | Var n ⇒ k = n
  | Abs b ⇒ free_in (S k) b
  | App f a ⇒ free_in k f ∨ free_in k a
  end.

Fixpoint free_inb (k : nat) (t : tm) {struct t} : bool :=
  match t with
  | Var n ⇒ Nat.eqb k n
  | Abs b ⇒ free_inb (S k) b
  | App f a ⇒ free_inb k f || free_inb k a
  end.

Lemma free_inb_spec : ∀ (t : tm) (k : nat),
  free_inb k t = true ↔ free_in k t.
Proof.
  induction t as [n | b IH | f IHf a IHa]; intros k; simpl.
  - apply Nat.eqb_eq.
  - apply IH.
  - rewrite orb_true_iff, IHf, IHa. reflexivity.
Qed.

Lemma free_in_reflect : ∀ (k : nat) (t : tm),
  reflect (free_in k t) (free_inb k t).
Proof.
  intros k t. apply iff_reflect. symmetry. apply free_inb_spec.
Qed.


Definition closed_above (c : nat) (t : tm) : Prop :=
  ∀ k : nat, free_in k t → k < c.

Definition closed (t : tm) : Prop := closed_above 0 t.

Lemma closed_above_spec : ∀ (c : nat) (t : tm),
  closed_above c t ↔ (∀ k : nat, c ≤ k → ¬ free_in k t).
Proof.
  intros c t. split.
  - intros H k Hck Hfree. specialize (H k Hfree). lia.
  - intros H k Hfree.
    destruct (Nat.lt_ge_cases k c) as [Hlt | Hge].
    + exact Hlt.
    + exfalso. exact (H k Hge Hfree).
Qed.


Lemma closed_above_var : ∀ (c n : nat),
  closed_above c (Var n) ↔ n < c.
Proof.
  intros c n. unfold closed_above. simpl. split.
  - intros H. exact (H n eq_refl).
  - intros H j Hj. subst j. exact H.
Qed.

Lemma closed_above_abs : ∀ (c : nat) (b : tm),
  closed_above c (Abs b) ↔ closed_above (S c) b.
Proof.
  intros c b. unfold closed_above. simpl. split.
  - intros H j Hj. destruct j as [| j'].
    + lia.
    + specialize (H j' Hj). lia.
  - intros H j Hj. specialize (H (S j) Hj). lia.
Qed.

Lemma closed_above_app : ∀ (c : nat) (f a : tm),
  closed_above c (App f a) ↔ closed_above c f ∧ closed_above c a.
Proof.
  intros c f a. unfold closed_above. simpl. split.
  - intros H. split; intros j Hj; apply H; [left | right]; exact Hj.
  - intros [Hf Ha] j [Hj | Hj]; [apply Hf | apply Ha]; exact Hj.
Qed.


Fixpoint closed_aboveb (c : nat) (t : tm) {struct t} : bool :=
  match t with
  | Var n ⇒ Nat.ltb n c
  | Abs b ⇒ closed_aboveb (S c) b
  | App f a ⇒ closed_aboveb c f && closed_aboveb c a
  end.

Definition closedb (t : tm) : bool := closed_aboveb 0 t.

Lemma closed_aboveb_spec : ∀ (t : tm) (c : nat),
  closed_aboveb c t = true ↔ closed_above c t.
Proof.
  induction t as [n | b IH | f IHf a IHa]; intros c; simpl.
  - rewrite closed_above_var. apply Nat.ltb_lt.
  - rewrite closed_above_abs. apply IH.
  - rewrite closed_above_app, andb_true_iff, IHf, IHa. reflexivity.
Qed.

Lemma closed_above_reflect : ∀ (c : nat) (t : tm),
  reflect (closed_above c t) (closed_aboveb c t).
Proof.
  intros c t. apply iff_reflect. symmetry. apply closed_aboveb_spec.
Qed.

Lemma closedb_spec : ∀ t : tm, closedb t = true ↔ closed t.
Proof.
  intros t. apply closed_aboveb_spec.
Qed.


Lemma free_in_lift : ∀ (t : tm) (d c j : nat),
  free_in j (lift d c t)
  ↔ (j < c ∧ free_in j t) ∨ (c + d ≤ j ∧ free_in (j - d) t).
Proof.
  induction t as [n | b IH | f IHf a IHa]; intros d c j; simpl.
  - destruct (Nat.leb c n) eqn:E;
      [apply Nat.leb_le in E | apply Nat.leb_gt in E]; lia.
  - rewrite (IH d (S c) (S j)).
    split; intros [[H1 H2] | [H1 H2]].
    + left. split; [lia | exact H2].
    + right. split; [lia | ].
      replace (S (j - d)) with (S j - d) by lia. exact H2.
    + left. split; [lia | exact H2].
    + right. split; [lia | ].
      replace (S j - d) with (S (j - d)) by lia. exact H2.
  - rewrite IHf, IHa. tauto.
Qed.


Lemma free_in_lift_lo : ∀ (t : tm) (d c j : nat),
  j < c → (free_in j (lift d c t) ↔ free_in j t).
Proof.
  intros t d c j Hj. rewrite free_in_lift. split.
  - intros [[H1 H2] | [H1 H2]]; [exact H2 | lia].
  - intros H. left. split; [exact Hj | exact H].
Qed.

Lemma free_in_lift_hi : ∀ (t : tm) (d c j : nat),
  c ≤ j → (free_in (d + j) (lift d c t) ↔ free_in j t).
Proof.
  intros t d c j Hj. rewrite free_in_lift. split.
  - intros [[H1 H2] | [H1 H2]]; [lia | ].
    replace (d + j - d) with j in H2 by lia. exact H2.
  - intros H. right. split; [lia | ].
    replace (d + j - d) with j by lia. exact H.
Qed.

Lemma free_in_lift_gap : ∀ (t : tm) (d c j : nat),
  c ≤ j → j < c + d → ¬ free_in j (lift d c t).
Proof.
  intros t d c j H1 H2 Hj. rewrite free_in_lift in Hj.
  destruct Hj as [[G1 G2] | [G1 G2]]; lia.
Qed.


Lemma free_in_subst : ∀ (t u : tm) (c j : nat),
  free_in j (subst c u t)
  ↔ (j < c ∧ free_in j t)
   ∨ (c ≤ j ∧ free_in (S j) t)
   ∨ (free_in c t ∧ c ≤ j ∧ free_in (j - c) u).
Proof.
  induction t as [n | b IH | f IHf a IHa]; intros u c j; simpl.
  - destruct (Nat.ltb n c) eqn:E1.
    + apply Nat.ltb_lt in E1. simpl. split.
      × intros →. left. split; [lia | reflexivity].
      × intros [[H1 H2] | [[H1 H2] | [H1 [H2 H3]]]]; lia.
    + apply Nat.ltb_ge in E1.
      destruct (Nat.eqb n c) eqn:E2.
      × apply Nat.eqb_eq in E2. subst n.
        rewrite free_in_lift. split.
        -- intros [[H1 H2] | [H1 H2]]; [lia | ].
           right. right.
           split; [reflexivity | split; [lia | exact H2]].
        -- intros [[H1 H2] | [[H1 H2] | [H1 [H2 H3]]]]; [lia | lia | ].
           right. split; [lia | exact H3].
      × apply Nat.eqb_neq in E2. simpl. split.
        -- intros →. right. left. split; lia.
        -- intros [[H1 H2] | [[H1 H2] | [H1 [H2 H3]]]]; lia.
  - rewrite (IH u (S c) (S j)).
    split; intros [[H1 H2] | [[H1 H2] | [H1 [H2 H3]]]].
    + left. split; [lia | exact H2].
    + right. left. split; [lia | exact H2].
    + right. right. split; [exact H1 | split; [lia | exact H3]].
    + left. split; [lia | exact H2].
    + right. left. split; [lia | exact H2].
    + right. right. split; [exact H1 | split; [lia | exact H3]].
  - rewrite IHf, IHa. tauto.
Qed.


Lemma subst_not_free : ∀ (t u : tm) (k : nat),
  ¬ free_in k t → lift 1 k (subst k u t) = t.
Proof.
  induction t as [n | b IH | f IHf a IHa]; intros u k H; simpl in H |- ×.
  - destruct (Nat.ltb n k) eqn:E1.
    + apply Nat.ltb_lt in E1. simpl.
      destruct (Nat.leb k n) eqn:E2.
      × apply Nat.leb_le in E2. lia.
      × reflexivity.
    + apply Nat.ltb_ge in E1.
      destruct (Nat.eqb n k) eqn:E2.
      × apply Nat.eqb_eq in E2. exfalso. apply H. lia.
      × apply Nat.eqb_neq in E2. simpl.
        destruct (Nat.leb k (Nat.pred n)) eqn:E3.
        -- f_equal. lia.
        -- apply Nat.leb_gt in E3. lia.
  - f_equal. apply IH. exact H.
  - f_equal.
    + apply IHf. intros Hf. apply H. left. exact Hf.
    + apply IHa. intros Ha. apply H. right. exact Ha.
Qed.

Lemma subst_not_free_irrel : ∀ (t u v : tm) (k : nat),
  ¬ free_in k t → subst k u t = subst k v t.
Proof.
  induction t as [n | b IH | f IHf a IHa]; intros u v k H; simpl in H |- ×.
  - destruct (Nat.ltb n k); [reflexivity | ].
    destruct (Nat.eqb n k) eqn:E2; [ | reflexivity].
    apply Nat.eqb_eq in E2. exfalso. apply H. lia.
  - f_equal. apply IH. exact H.
  - f_equal.
    + apply IHf. intros Hf. apply H. left. exact Hf.
    + apply IHa. intros Ha. apply H. right. exact Ha.
Qed.


Lemma closed_above_mono : ∀ (t : tm) (c c' : nat),
  c ≤ c' → closed_above c t → closed_above c' t.
Proof.
  intros t c c' Hle H j Hj. specialize (H j Hj). lia.
Qed.

Lemma lift_closed_above : ∀ (t : tm) (d c : nat),
  closed_above c t → lift d c t = t.
Proof.
  induction t as [n | b IH | f IHf a IHa]; intros d c H; simpl.
  - apply closed_above_var in H.
    destruct (Nat.leb c n) eqn:E.
    + apply Nat.leb_le in E. lia.
    + reflexivity.
  - f_equal. apply IH. apply closed_above_abs. exact H.
  - apply closed_above_app in H. destruct H as [Hf Ha].
    f_equal; [apply IHf; exact Hf | apply IHa; exact Ha].
Qed.

Lemma subst_closed_above : ∀ (t u : tm) (c : nat),
  closed_above c t → subst c u t = t.
Proof.
  induction t as [n | b IH | f IHf a IHa]; intros u c H; simpl.
  - apply closed_above_var in H.
    destruct (Nat.ltb n c) eqn:E.
    + reflexivity.
    + apply Nat.ltb_ge in E. lia.
  - f_equal. apply IH. apply closed_above_abs. exact H.
  - apply closed_above_app in H. destruct H as [Hf Ha].
    f_equal; [apply IHf; exact Hf | apply IHa; exact Ha].
Qed.


Lemma lift_closed : ∀ (t : tm) (d c : nat),
  closed t → lift d c t = t.
Proof.
  intros t d c H. apply lift_closed_above.
  apply (closed_above_mono t 0 c).
  - apply Nat.le_0_l.
  - exact H.
Qed.

Lemma subst_closed : ∀ (t u : tm) (k : nat),
  closed t → subst k u t = t.
Proof.
  intros t u k H. apply subst_closed_above.
  apply (closed_above_mono t 0 k).
  - apply Nat.le_0_l.
  - exact H.
Qed.

Lemma closed_above_lift : ∀ (t : tm) (d k c : nat),
  closed_above c t → closed_above (c + d) (lift d k t).
Proof.
  intros t d k c H j Hj. apply free_in_lift in Hj.
  destruct Hj as [[H1 H2] | [H1 H2]]; specialize (H _ H2); lia.
Qed.

Lemma closed_above_subst_0 : ∀ (t u : tm) (c : nat),
  closed_above (S c) t → closed_above c u → closed_above c (subst 0 u t).
Proof.
  intros t u c Ht Hu j Hj. apply free_in_subst in Hj.
  destruct Hj as [[H1 H2] | [[H1 H2] | [H1 [H2 H3]]]].
  - lia.
  - specialize (Ht _ H2). lia.
  - specialize (Hu _ H3). lia.
Qed.


Fixpoint fv_bound (t : tm) {struct t} : nat :=
  match t with
  | Var n ⇒ S n
  | Abs b ⇒ Nat.pred (fv_bound b)
  | App f a ⇒ Nat.max (fv_bound f) (fv_bound a)
  end.

Lemma fv_bound_spec : ∀ (t : tm) (k : nat),
  free_in k t → k < fv_bound t.
Proof.
  induction t as [n | b IH | f IHf a IHa]; intros j Hj; simpl in Hj |- ×.
  - lia.
  - specialize (IH _ Hj). lia.
  - destruct Hj as [Hj | Hj];
      [specialize (IHf _ Hj) | specialize (IHa _ Hj)]; lia.
Qed.

Lemma fv_bound_closed_above : ∀ t : tm, closed_above (fv_bound t) t.
Proof.
  intros t j Hj. apply fv_bound_spec. exact Hj.
Qed.

Lemma fv_bound_tight : ∀ (t : tm) (m : nat),
  fv_bound t = S m → free_in m t.
Proof.
  induction t as [n | b IH | f IHf a IHa]; intros m Hm; simpl in Hm |- ×.
  - lia.
  - apply IH. lia.
  - destruct (Nat.max_spec (fv_bound f) (fv_bound a)) as [[H1 H2] | [H1 H2]].
    + right. apply IHa. lia.
    + left. apply IHf. lia.
Qed.
