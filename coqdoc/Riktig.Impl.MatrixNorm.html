Library Riktig.Impl.MatrixNorm

From Coq Require Import Morphisms.
From Coq.Vectors Require Fin.
From Riktig Require Import Ops Interface.Monoid Interface.AbGroup
  Interface.Ring Interface.Field Interface.VectorSpace Interface.Order
  Interface.Norm Theory.Order Theory.BigOp Impl.Matrix.


Section MatrixNorm.
  Context {F : Type} `{OrderedField F}.
  Context (m n : nat).

  Definition mat_norm (A : Matrix m n F) : F :=
    bigopF (fun i : Fin.t m ⇒ bigopF (fun j : Fin.t n ⇒ abs (A i j))).

  #[global] Instance Matrix_HasNorm : HasNorm F (Matrix m n F) := mat_norm.

  Lemma mat_norm_proper :
    Proper (equ ==> equ) (@vnorm F (Matrix m n F) Matrix_HasNorm).
  Proof.
    intros A A' HA.
    unfold vnorm, Matrix_HasNorm, mat_norm.
    apply bigopF_ext. intros i.
    apply bigopF_ext. intros j.
    apply abs_proper. exact (HA i j).
  Qed.

  Lemma mat_norm_nonneg : ∀ A : Matrix m n F, 0 ≤ ‖ A ‖.
  Proof.
    intros A.
    unfold vnorm, Matrix_HasNorm, mat_norm.
    apply bigopF_nonneg. intros i.
    apply bigopF_nonneg. intros j.
    apply abs_nonneg.
  Qed.

  Lemma mat_norm_def : ∀ A : Matrix m n F, ‖ A ‖ == 0 → A == 0.
  Proof.
    intros A Heq i j.
    unfold vnorm, Matrix_HasNorm, mat_norm in Heq.
    apply (proj1 (abs_zero_iff (A i j))).
    exact (bigopF_nonneg_zero n (fun j0 ⇒ abs (A i j0))
      (fun j0 ⇒ abs_nonneg (A i j0))
      (bigopF_nonneg_zero m
        (fun i0 ⇒ bigopF (fun j0 ⇒ abs (A i0 j0)))
        (fun i0 ⇒ bigopF_nonneg n (fun j0 ⇒ abs (A i0 j0))
                     (fun j0 ⇒ abs_nonneg (A i0 j0)))
        Heq i)
      j).
  Qed.

  Lemma mat_norm_scale : ∀ (a : F) (A : Matrix m n F),
    ‖ a *: A ‖ == abs a × ‖ A ‖.
  Proof.
    intros a A.
    unfold vnorm, Matrix_HasNorm, mat_norm.
    calc (bigopF (fun i ⇒ bigopF (fun j ⇒ abs ((a *: A) i j))))
      == (bigopF (fun i ⇒ bigopF (fun j ⇒ abs a × abs (A i j))))
      == (bigopF (fun i ⇒ abs a × bigopF (fun j ⇒ abs (A i j))))
      == (abs a × bigopF (fun i ⇒ bigopF (fun j ⇒ abs (A i j)))).
    - apply bigopF_ext. intros i.
      apply bigopF_ext. intros j.
      exact (abs_mul a (A i j)).
    - apply bigopF_ext. intros i.
      exact (bigopF_mul_l n (abs a) (fun j ⇒ abs (A i j))).
    - exact (bigopF_mul_l m (abs a) (fun i ⇒ bigopF (fun j ⇒ abs (A i j)))).
  Qed.

  Lemma mat_norm_triangle : ∀ A B : Matrix m n F,
    ‖ A + B ‖ ≤ ‖ A ‖ + ‖ B ‖.
  Proof.
    intros A B.
    unfold vnorm, Matrix_HasNorm, mat_norm.
    apply (le_trans
      (bigopF (fun i ⇒ bigopF (fun j ⇒ abs ((A + B) i j))))
      (bigopF (fun i ⇒ bigopF (fun j ⇒ abs (A i j) + abs (B i j))))
      (bigopF (fun i ⇒ bigopF (fun j ⇒ abs (A i j)))
         + bigopF (fun i ⇒ bigopF (fun j ⇒ abs (B i j))))).
    - apply bigopF_le. intros i.
      apply bigopF_le. intros j.
      exact (abs_triangle (A i j) (B i j)).
    - apply eq_le.
      transitivity (bigopF (fun i ⇒ bigopF (fun j ⇒ abs (A i j))
                                    + bigopF (fun j ⇒ abs (B i j)))).
      + apply bigopF_ext. intros i.
        exact (bigopF_add n (fun j ⇒ abs (A i j)) (fun j ⇒ abs (B i j))).
      + exact (bigopF_add m (fun i ⇒ bigopF (fun j ⇒ abs (A i j)))
                 (fun i ⇒ bigopF (fun j ⇒ abs (B i j)))).
  Qed.

  #[global] Instance Matrix_NormedSpace : NormedSpace F (Matrix m n F) := {|
    ns_norm := Matrix_HasNorm;
    norm_proper := mat_norm_proper;
    norm_nonneg := mat_norm_nonneg;
    norm_def := mat_norm_def;
    norm_scale := mat_norm_scale;
    norm_triangle := mat_norm_triangle
  |}.

End MatrixNorm.
