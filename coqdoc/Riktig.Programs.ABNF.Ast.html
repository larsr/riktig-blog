Library Riktig.Programs.ABNF.Ast

Require Import String.
Require Import List.

Open Scope string_scope.


Inductive NumBase := BinBase | DecBase | HexBase.

Inductive NumSpec :=
  | NumConcat : NumBase → list nat → NumSpec
  | NumRange : NumBase → nat → nat → NumSpec.


Record RepeatSpec := mkRepeat {
  repeat_min : option nat;
  repeat_max : option nat
}.


Inductive ABNFElement :=
  | ElemRuleName : string → ABNFElement
  | ElemCharVal : string → ABNFElement
  | ElemNumVal : NumSpec → ABNFElement
  | ElemProseVal : string → ABNFElement
  | ElemGroup : ABNFAlternation → ABNFElement
  | ElemOption : ABNFAlternation → ABNFElement

with ABNFAlternation :=
  | Alt : list ABNFConcatenation → ABNFAlternation

with ABNFConcatenation :=
  | Concat : list ABNFRepetition → ABNFConcatenation

with ABNFRepetition :=
  | Rep : option RepeatSpec → ABNFElement → ABNFRepetition.


Inductive DefinedAs := DefEqual | DefIncremental.

Record ABNFRule := mkRule {
  rule_name : string;
  rule_defined : DefinedAs;
  rule_body : ABNFAlternation
}.

Definition ABNFRulelist := list ABNFRule.
