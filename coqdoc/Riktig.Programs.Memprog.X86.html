Definition byte_at (x : Z) (i : nat) : Z :=
  Z.land (Z.shiftr x (8 × Z.of_nat i)) 255.

Definition le16 (x : Z) : list Z := [byte_at x 0; byte_at x 1].

Definition le32 (x : Z) : list Z :=
  [byte_at x 0; byte_at x 1; byte_at x 2; byte_at x 3].

Definition le64 (x : Z) : list Z :=
  [ byte_at x 0; byte_at x 1; byte_at x 2; byte_at x 3
  ; byte_at x 4; byte_at x 5; byte_at x 6; byte_at x 7 ].

Definition rex_w (r x b : Z) : Z :=
  Z.lor 72
    (Z.lor (if 8 <=? r then 4 else 0)
       (Z.lor (if 8 <=? x then 2 else 0)
              (if 8 <=? b then 1 else 0))).

Definition mod_rm (m reg rm : Z) : Z :=
  Z.lor (Z.shiftl m 6) (Z.lor (Z.shiftl (Z.land reg 7) 3) (Z.land rm 7)).

Definition sib (scale idx base : Z) : Z :=
  Z.lor (Z.shiftl scale 6) (Z.lor (Z.shiftl (Z.land idx 7) 3) (Z.land base 7)).

Definition xreg (r : Reg) : Z :=
  match r with
  | R0 ⇒ 0 | R1 ⇒ 1 | R2 ⇒ 2 | R3 ⇒ 3
  | R4 ⇒ 6 | R5 ⇒ 7 | R6 ⇒ 8 | R7 ⇒ 9
  end.

Definition r10_code : Z := 10.


Definition emit_mov (rd rs : Z) : list Z :=
  [rex_w rd 0 rs; 139; mod_rm 3 rd rs].

Definition emit_movabs (rd imm : Z) : list Z :=
  [rex_w 0 0 rd; 184 + Z.land rd 7] ++ le64 imm.

Definition emit_add_rr (rd rs : Z) : list Z :=
  [rex_w rd 0 rs; 3; mod_rm 3 rd rs].

Definition emit_sub_rr (rd rs : Z) : list Z :=
  [rex_w rd 0 rs; 43; mod_rm 3 rd rs].

Definition emit_neg (rd : Z) : list Z :=
  [rex_w 0 0 rd; 247; mod_rm 3 3 rd].

Definition emit_add_imm (rd imm : Z) : list Z :=
  [rex_w 0 0 rd; 129; mod_rm 3 0 rd] ++ le32 imm.

Definition emit_sub_imm (rd imm : Z) : list Z :=
  [rex_w 0 0 rd; 129; mod_rm 3 5 rd] ++ le32 imm.

Definition emit_cmp_rr (r1 r2 : Z) : list Z :=
  [rex_w r1 0 r2; 59; mod_rm 3 r1 r2].

Definition emit_cmp_imm (rs imm : Z) : list Z :=
  [rex_w 0 0 rs; 129; mod_rm 3 7 rs] ++ le32 imm.

Definition emit_load (rd idx off : Z) : list Z :=
  [rex_w rd idx r10_code; 139; mod_rm 2 rd 4; sib 3 idx r10_code]
    ++ le32 (off × 8).

Definition emit_store (rs idx off : Z) : list Z :=
  [rex_w rs idx r10_code; 137; mod_rm 2 rs 4; sib 3 idx r10_code]
    ++ le32 (off × 8).

Definition emit_push (rs : Z) : list Z :=
  (if 8 <=? rs then [65] else []) ++ [80 + Z.land rs 7].

Definition emit_pop (rd : Z) : list Z :=
  (if 8 <=? rd then [65] else []) ++ [88 + Z.land rd 7].

Definition emit_jmp (rel : Z) : list Z := 233 :: le32 rel.
Definition emit_jcc (cc rel : Z) : list Z := [15; 128 + cc] ++ le32 rel.
Definition emit_call (rel : Z) : list Z := 232 :: le32 rel.

Definition emit_jmp_ind (rs : Z) : list Z :=
  (if 8 <=? rs then [rex_w 0 0 rs] else []) ++ [255; mod_rm 3 4 rs].

Definition emit_ret : list Z := [195].
Definition emit_nop : list Z := [144].
Definition emit_syscall : list Z := [15; 5].
Definition cc_e : Z := 4.
Definition cc_ne : Z := 5.
Definition cc_l : Z := 12.
Definition cc_le : Z := 14.
Definition cc_g : Z := 15.
Definition cc_ge : Z := 13.

Definition emit_add3 (rd rs1 rs2 : Z) : list Z :=
  if rd =? rs1 then emit_add_rr rd rs2
  else if rd =? rs2 then emit_add_rr rd rs1
  else emit_mov rd rs1 ++ emit_add_rr rd rs2.

Definition emit_sub3 (rd rs1 rs2 : Z) : list Z :=
  if rd =? rs1 then emit_sub_rr rd rs2
  else if rd =? rs2 then emit_neg rd ++ emit_add_rr rd rs1
  else emit_mov rd rs1 ++ emit_sub_rr rd rs2.

Definition emit_addi2 (rd rs imm : Z) : list Z :=
  if rd =? rs then emit_add_imm rd imm
  else emit_mov rd rs ++ emit_add_imm rd imm.

Definition emit_subi2 (rd rs imm : Z) : list Z :=
  if rd =? rs then emit_sub_imm rd imm
  else emit_mov rd rs ++ emit_sub_imm rd imm.

Definition encode_instr (i : Instr Z) (disp : Z) : list Z :=
  match i with
  | MOV rd rs ⇒ emit_mov (xreg rd) (xreg rs)
  | LI rd v ⇒ emit_movabs (xreg rd) v
  | ADD rd rs1 rs2 ⇒ emit_add3 (xreg rd) (xreg rs1) (xreg rs2)
  | ADDI rd rs v ⇒ emit_addi2 (xreg rd) (xreg rs) v
  | SUB rd rs1 rs2 ⇒ emit_sub3 (xreg rd) (xreg rs1) (xreg rs2)
  | SUBI rd rs v ⇒ emit_subi2 (xreg rd) (xreg rs) v
  | LW rd base off ⇒ emit_load (xreg rd) (xreg base) off
  | SW src base off ⇒ emit_store (xreg src) (xreg base) off
  | CMP rs1 rs2 ⇒ emit_cmp_rr (xreg rs1) (xreg rs2)
  | CMPI rs v ⇒ emit_cmp_imm (xreg rs) v
  | B _ ⇒ emit_jmp disp
  | BEQ _ ⇒ emit_jcc cc_e disp
  | BNE _ ⇒ emit_jcc cc_ne disp
  | BLT _ ⇒ emit_jcc cc_l disp
  | BLE _ ⇒ emit_jcc cc_le disp
  | BGT _ ⇒ emit_jcc cc_g disp
  | BGE _ ⇒ emit_jcc cc_ge disp
  | BR rs ⇒ emit_jmp_ind (xreg rs)
  | PUSH rs ⇒ emit_push (xreg rs)
  | POP rd ⇒ emit_pop (xreg rd)
  | CALL _ ⇒ emit_call disp
  | RET ⇒ emit_ret
  | HALT ⇒ emit_syscall
  | NOP ⇒ emit_nop
  end.

Definition branch_off (i : Instr Z) : option Z :=
  match i with
  | B off ⇒ Some off
  | BEQ off ⇒ Some off
  | BNE off ⇒ Some off
  | BLT off ⇒ Some off
  | BLE off ⇒ Some off
  | BGT off ⇒ Some off
  | BGE off ⇒ Some off
  | CALL off ⇒ Some off
  | _ ⇒ None
  end.


Definition instr_size (i : Instr Z) : Z :=
  Z.of_nat (length (encode_instr i 0)).

Fixpoint offsets_from (acc : Z) (is : list (Instr Z)) : list Z :=
  match is with
  | [] ⇒ [acc]
  | i :: rest ⇒ acc :: offsets_from (acc + instr_size i) rest
  end.

Definition znth (l : list Z) (i : Z) : Z := nth (Z.to_nat i) l 0.

Fixpoint assemble_go (all_offs : list Z) (n idx : Z) (is : list (Instr Z))
  : list Z :=
  match is with
  | [] ⇒ []
  | i :: rest ⇒
      let disp :=
        match branch_off i with
        | None ⇒ 0
        | Some off ⇒
            let target := idx + off in
            if andb (0 <=? target) (target <=? n)
            then znth all_offs target - znth all_offs idx - instr_size i
            else 0
        end in
      encode_instr i disp ++ assemble_go all_offs n (idx + 1) rest
  end.

Definition assemble (instrs : list (Instr Z)) : list Z :=
  assemble_go (offsets_from 0 instrs) (Z.of_nat (length instrs)) 0 instrs.


Inductive X86Op : Type :=
  | XMov : Z → Z → X86Op
  | XMovabs : Z → Z → X86Op
  | XAddRR : Z → Z → X86Op
  | XSubRR : Z → Z → X86Op
  | XNeg : Z → X86Op
  | XAddImm : Z → Z → X86Op
  | XSubImm : Z → Z → X86Op
  | XCmpRR : Z → Z → X86Op
  | XCmpImm : Z → Z → X86Op
  | XLoad : Z → Z → Z → X86Op
  | XStore : Z → Z → Z → X86Op
  | XPush : Z → X86Op
  | XPop : Z → X86Op
  | XJmp : Z → X86Op
  | XJcc : Z → Z → X86Op
  | XJmpInd : Z → X86Op
  | XCall : Z → X86Op
  | XRet : X86Op
  | XNop : X86Op
  | XSyscall : X86Op.

Fixpoint le_to_z (n : nat) (bs : list Z) : Z :=
  match n with
  | O ⇒ 0
  | S n' ⇒ match bs with
            | [] ⇒ 0
            | b :: rest ⇒ b + 256 × le_to_z n' rest
            end
  end.

Definition u32 (bs : list Z) : Z := le_to_z 4 bs.
Definition s32 (bs : list Z) : Z :=
  let u := u32 bs in if u <? 2147483648 then u else u - 4294967296.
Definition u64 (bs : list Z) : Z := le_to_z 8 bs.

Definition hi (bit : bool) : Z := if bit then 8 else 0.

Definition decode (bs : list Z) : option (X86Op × nat) :=
  match bs with
  | [] ⇒ None
  | b0 :: r0 ⇒
    if b0 =? 15 then
      
      match r0 with
      | b1 :: r1 ⇒
          if b1 =? 5 then Some (XSyscall, 2%nat)
          else Some (XJcc (b1 - 128) (s32 r1), 6%nat)
      | [] ⇒ None
      end
    else if b0 =? 233 then Some (XJmp (s32 r0), 5%nat)
    else if b0 =? 232 then Some (XCall (s32 r0), 5%nat)
    else if b0 =? 195 then Some (XRet, 1%nat)
    else if b0 =? 144 then Some (XNop, 1%nat)
    else if b0 =? 255 then
      
      match r0 with
      | m :: _ ⇒ Some (XJmpInd (Z.land m 7), 2%nat)
      | [] ⇒ None
      end
    else if b0 =? 65 then
      
      match r0 with
      | b1 :: r1 ⇒
          if andb (80 <=? b1) (b1 <=? 87) then Some (XPush (8 + (b1 - 80)), 2%nat)
          else if andb (88 <=? b1) (b1 <=? 95) then Some (XPop (8 + (b1 - 88)), 2%nat)
          else if b1 =? 255 then
            match r1 with
            | m :: _ ⇒ Some (XJmpInd (8 + Z.land m 7), 3%nat)
            | [] ⇒ None
            end
          else None
      | [] ⇒ None
      end
    else if andb (80 <=? b0) (b0 <=? 87) then Some (XPush (b0 - 80), 1%nat)
    else if andb (88 <=? b0) (b0 <=? 95) then Some (XPop (b0 - 88), 1%nat)
    else if andb (72 <=? b0) (b0 <=? 79) then
      
      let rb := Z.testbit b0 0 in
      let rx := Z.testbit b0 1 in
      let rr := Z.testbit b0 2 in
      match r0 with
      | op :: r1 ⇒
          if andb (184 <=? op) (op <=? 191) then
            Some (XMovabs (hi rb + (op - 184)) (u64 r1), 10%nat)
          else
            match r1 with
            | modrm :: r2 ⇒
                let md := Z.shiftr modrm 6 in
                let regf := Z.land (Z.shiftr modrm 3) 7 in
                let rmf := Z.land modrm 7 in
                let regF := hi rr + regf in
                let rmF := hi rb + rmf in
                if op =? 139 then
                  if md =? 3 then Some (XMov regF rmF, 3%nat)
                  else
                    match r2 with
                    | sibb :: r3 ⇒
                        let idxF := hi rx + Z.land (Z.shiftr sibb 3) 7 in
                        Some (XLoad regF idxF (s32 r3 / 8), 8%nat)
                    | [] ⇒ None
                    end
                else if op =? 137 then
                  match r2 with
                  | sibb :: r3 ⇒
                      let idxF := hi rx + Z.land (Z.shiftr sibb 3) 7 in
                      Some (XStore regF idxF (s32 r3 / 8), 8%nat)
                  | [] ⇒ None
                  end
                else if op =? 3 then Some (XAddRR regF rmF, 3%nat)
                else if op =? 43 then Some (XSubRR regF rmF, 3%nat)
                else if op =? 59 then Some (XCmpRR regF rmF, 3%nat)
                else if op =? 247 then Some (XNeg rmF, 3%nat)
                else if op =? 129 then
                  if regf =? 0 then Some (XAddImm rmF (s32 r2), 7%nat)
                  else if regf =? 5 then Some (XSubImm rmF (s32 r2), 7%nat)
                  else if regf =? 7 then Some (XCmpImm rmF (s32 r2), 7%nat)
                  else None
                else if op =? 255 then Some (XJmpInd rmF, 3%nat)
                else None
            | [] ⇒ None
            end
      | [] ⇒ None
      end
    else None
  end.


Record X86State : Type := MkX86 {
  x_regs : Z → Z;
  x_rip : Z;
  x_flags : Flag;
  x_data : Z → Z;
  x_stack : Z → Z
}.

Definition zupd (f : Z → Z) (k v : Z) : Z → Z :=
  fun x ⇒ if Z.eqb k x then v else f x.

Definition xcmp (a b : Z) : Flag :=
  if a <? b then FLT else if a =? b then FEQ else FGT.

Definition jcc_taken (cc : Z) (f : Flag) : bool :=
  if cc =? cc_e then flag_eq f
  else if cc =? cc_ne then negb (flag_eq f)
  else if cc =? cc_l then flag_lt f
  else if cc =? cc_le then orb (flag_lt f) (flag_eq f)
  else if cc =? cc_g then flag_gt f
  else if cc =? cc_ge then orb (flag_gt f) (flag_eq f)
  else false.

Definition st_reg (st : X86State) (r v nrip : Z) : X86State :=
  MkX86 (zupd (x_regs st) r v) nrip (x_flags st) (x_data st) (x_stack st).
Definition st_flag (st : X86State) (f : Flag) (nrip : Z) : X86State :=
  MkX86 (x_regs st) nrip f (x_data st) (x_stack st).
Definition st_data (st : X86State) (a v nrip : Z) : X86State :=
  MkX86 (x_regs st) nrip (x_flags st) (zupd (x_data st) a v) (x_stack st).
Definition st_rip (st : X86State) (nrip : Z) : X86State :=
  MkX86 (x_regs st) nrip (x_flags st) (x_data st) (x_stack st).

Definition x86_step (code : list Z) (st : X86State)
  : option (option X86State) :=
  match decode (skipn (Z.to_nat (x_rip st)) code) with
  | None ⇒ None
  | Some (op, len) ⇒
    let nrip := x_rip st + Z.of_nat len in
    match op with
    | XSyscall ⇒ Some None
    | XNop ⇒ Some (Some (st_rip st nrip))
    | XMov rd rs ⇒ Some (Some (st_reg st rd (x_regs st rs) nrip))
    | XMovabs rd imm ⇒ Some (Some (st_reg st rd imm nrip))
    | XAddRR rd rs ⇒ Some (Some (st_reg st rd (x_regs st rd + x_regs st rs) nrip))
    | XSubRR rd rs ⇒ Some (Some (st_reg st rd (x_regs st rd - x_regs st rs) nrip))
    | XNeg rd ⇒ Some (Some (st_reg st rd (- x_regs st rd) nrip))
    | XAddImm rd imm ⇒ Some (Some (st_reg st rd (x_regs st rd + imm) nrip))
    | XSubImm rd imm ⇒ Some (Some (st_reg st rd (x_regs st rd - imm) nrip))
    | XCmpRR r1 r2 ⇒ Some (Some (st_flag st (xcmp (x_regs st r1) (x_regs st r2)) nrip))
    | XCmpImm rs imm ⇒ Some (Some (st_flag st (xcmp (x_regs st rs) imm) nrip))
    | XLoad rd idx dw ⇒
        Some (Some (st_reg st rd (x_data st (x_regs st idx + dw)) nrip))
    | XStore rs idx dw ⇒
        Some (Some (st_data st (x_regs st idx + dw) (x_regs st rs) nrip))
    | XJmp rel ⇒ Some (Some (st_rip st (nrip + rel)))
    | XJcc cc rel ⇒
        Some (Some (st_rip st (if jcc_taken cc (x_flags st) then nrip + rel else nrip)))
    | XJmpInd rs ⇒ Some (Some (st_rip st (x_regs st rs)))
    | XPush rs ⇒
        let sp := x_regs st 4 - 1 in
        Some (Some (MkX86 (zupd (x_regs st) 4 sp) nrip (x_flags st)
                          (x_data st) (zupd (x_stack st) sp (x_regs st rs))))
    | XPop rd ⇒
        let sp := x_regs st 4 in
        Some (Some (MkX86 (zupd (zupd (x_regs st) rd (x_stack st sp)) 4 (sp + 1))
                          nrip (x_flags st) (x_data st) (x_stack st)))
    | XCall rel ⇒
        let sp := x_regs st 4 - 1 in
        Some (Some (MkX86 (zupd (x_regs st) 4 sp) (nrip + rel) (x_flags st)
                          (x_data st) (zupd (x_stack st) sp nrip)))
    | XRet ⇒
        let sp := x_regs st 4 in
        Some (Some (MkX86 (zupd (x_regs st) 4 (sp + 1)) (x_stack st sp)
                          (x_flags st) (x_data st) (x_stack st)))
    end
  end.

Fixpoint x86_run (code : list Z) (fuel : nat) (st : X86State)
  : option X86State :=
  match fuel with
  | O ⇒ None
  | S fuel' ⇒
      match x86_step code st with
      | None ⇒ None
      | Some None ⇒ Some st
      | Some (Some st') ⇒ x86_run code fuel' st'
      end
  end.

Fixpoint decode_all (fuel : nat) (bs : list Z) : option (list X86Op) :=
  match fuel with
  | O ⇒ match bs with [] ⇒ Some [] | _ ⇒ None end
  | S fuel' ⇒
      match bs with
      | [] ⇒ Some []
      | _ ⇒ match decode bs with
             | Some (op, len) ⇒
                 option_map (cons op) (decode_all fuel' (skipn len bs))
             | None ⇒ None
             end
      end
  end.

Definition data_of (l : list Z) : Z → Z := fun i ⇒ nth (Z.to_nat i) l 0.

Fixpoint read_words (st : X86State) (base : Z) (n : nat) : list Z :=
  match n with
  | O ⇒ []
  | S n' ⇒ x_data st base :: read_words st (base + 1) n'
  end.

Definition swap_init : X86State :=
  MkX86 (zupd (zupd (fun _ ⇒ 0) 1 1) 2 3) 0 FEQ
        (data_of [10; 20; 30; 40; 50]) (fun _ ⇒ 0).

Definition x86_swap_result : option (list Z) :=
  match x86_run (assemble swap_mc) 1000 swap_init with
  | Some st ⇒ Some (read_words st 0 5)
  | None ⇒ None
  end.

Example x86_swap_matches_mc : x86_swap_result = Some [10; 40; 30; 20; 50].
Proof. vm_compute. reflexivity. Qed.
