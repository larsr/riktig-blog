Library Riktig.Lib.NatFacts


Require Import Arith Lia.

Lemma half_le d : d / 2 ≤ d.
Proof. apply Nat.Div0.div_le_upper_bound. lia. Qed.

Lemma max_le_l : ∀ a b n : nat,
  (Nat.max a b ≤ n)%nat → (a ≤ n)%nat.
Proof.
  intros a b n Hn.
  exact (Nat.le_trans a (Nat.max a b) n (Nat.le_max_l a b) Hn).
Qed.

Lemma max_le_r : ∀ a b n : nat,
  (Nat.max a b ≤ n)%nat → (b ≤ n)%nat.
Proof.
  intros a b n Hn.
  exact (Nat.le_trans b (Nat.max a b) n (Nat.le_max_r a b) Hn).
Qed.
