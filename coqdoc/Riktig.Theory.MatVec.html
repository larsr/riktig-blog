Section MatVecOps.
  Context {F : Type} {FF : Field F}.

  Definition mat_vec {m n : nat} (A : Matrix m n F) (v : Fin.t n → F)
      : Fin.t m → F :=
    fun i ⇒ bigopF (fun j ⇒ A i j × v j).

  Lemma mat_vec_proper : ∀ (m n : nat),
    Proper (equ ==> equ ==> equ) (@mat_vec m n).
  Proof.
    intros m n A A' HA v v' Hv i.
    exact (bigopF_ext n (fun j ⇒ A i j × v j) (fun j ⇒ A' i j × v' j)
             (fun j ⇒ ring_mul_proper (A i j) (A' i j) (HA i j)
                         (v j) (v' j) (Hv j))).
  Qed.

  Lemma mat_vec_proper_v : ∀ (m n : nat) (A : Matrix m n F),
    Proper (equ ==> equ) (mat_vec A).
  Proof.
    intros m n A v v' Hv i.
    exact (bigopF_ext n (fun j ⇒ A i j × v j) (fun j ⇒ A i j × v' j)
             (fun j ⇒ ring_mul_proper (A i j) (A i j) (reflexivity (A i j))
                         (v j) (v' j) (Hv j))).
  Qed.


  Lemma mat_vec_add : ∀ (m n : nat) (A : Matrix m n F)
      (v w : Fin.t n → F),
    mat_vec A (v + w) == mat_vec A v + mat_vec A w.
  Proof.
    intros m n A v w i.
    transitivity (bigopF (fun j ⇒ A i j × v j + A i j × w j)).
    - apply bigopF_ext. intros j.
      exact (ring_distl (A i j) (v j) (w j)).
    - exact (bigopF_add n (fun j ⇒ A i j × v j) (fun j ⇒ A i j × w j)).
  Qed.

  Lemma mat_vec_scale : ∀ (m n : nat) (A : Matrix m n F) (a : F)
      (v : Fin.t n → F),
    mat_vec A (a *: v) == a *: mat_vec A v.
  Proof.
    intros m n A a v i.
    transitivity (bigopF (fun j ⇒ a × (A i j × v j))).
    - apply bigopF_ext. intros j.
      exact (mul_comm_swap (A i j) a (v j)).
    - exact (bigopF_mul_l n a (fun j ⇒ A i j × v j)).
  Qed.

  Definition mat_vec_lm {m n : nat} (A : Matrix m n F) :
      LinearMap F (Fin.t n → F) (Fin.t m → F) :=
    @Build_LinearMap F (Fin.t n → F) (Fin.t m → F) FF
      (@PW_VectorSpace (Fin.t n) F F FF (@Field_VectorSpace F FF))
      (@PW_VectorSpace (Fin.t m) F F FF (@Field_VectorSpace F FF))
      (mat_vec A)
      (mat_vec_proper_v m n A)
      (mat_vec_add m n A)
      (mat_vec_scale m n A).

  Theorem mat_vec_mul : ∀ (m n p : nat) (A : Matrix m n F)
      (B : Matrix n p F) (v : Fin.t p → F),
    mat_vec (mat_mul A B) v == mat_vec A (mat_vec B v).
  Proof.
    intros m n p A B v i.
    exact (mat_mul_assoc m n p 1 A B
             (fun (k : Fin.t p) (_ : Fin.t 1) ⇒ v k) i Fin.F1).
  Qed.

  Theorem mat_vec_lm_mul : ∀ (m n p : nat) (A : Matrix m n F)
      (B : Matrix n p F),
    mat_vec_lm (mat_mul A B) == lm_comp (mat_vec_lm A) (mat_vec_lm B).
  Proof.
    intros m n p A B v.
    exact (mat_vec_mul m n p A B v).
  Qed.

  Theorem mat_vec_id : ∀ (n : nat) (v : Fin.t n → F),
    mat_vec mat_id v == v.
  Proof.
    intros n v i.
    exact (mat_mul_1l n 1 (fun (k : Fin.t n) (_ : Fin.t 1) ⇒ v k) i Fin.F1).
  Qed.


  Definition mat_tr {m n : nat} (A : Matrix m n F) : Matrix n m F :=
    fun i j ⇒ A j i.

  Lemma mat_tr_proper : ∀ (m n : nat),
    Proper (equ ==> equ) (@mat_tr m n).
  Proof.
    intros m n A A' HA i j.
    exact (HA j i).
  Qed.

End MatVecOps.
