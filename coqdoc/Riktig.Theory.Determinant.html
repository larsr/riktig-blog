Section Pullback.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V} {n : nat}.
  Context (f : LinearMap F V V) (omega : MultilinearForm n F V).

  Definition fstar_fun (vs : Fin.t n → V) : F := omega (fun i ⇒ f (vs i)).

  Lemma fstar_proper : Proper (@famEq V _ n ==> equ) fstar_fun.
  Proof.
    intros vs ws Hvw.
    apply (mf_proper omega).
    intros i.
    exact (lm_proper f (vs i) (ws i) (Hvw i)).
  Qed.

  Lemma fstar_upd_comm : ∀ (vs : Fin.t n → V) (i : Fin.t n) (x : V)
      (j : Fin.t n),
    f (upd vs i x j) == upd (fun k ⇒ f (vs k)) i (f x) j.
  Proof.
    intros vs i x j.
    unfold upd; destruct (Fin.eq_dec i j).
    - reflexivity.
    - reflexivity.
  Qed.

  Lemma upd_val_proper : ∀ (ys : Fin.t n → V) (i : Fin.t n) (x y : V),
    x == y → famEq (upd ys i x) (upd ys i y).
  Proof.
    intros ys i x y Hxy j.
    unfold upd; destruct (Fin.eq_dec i j).
    - exact Hxy.
    - reflexivity.
  Qed.

  Lemma fstar_upd_eq : ∀ (vs : Fin.t n → V) (i : Fin.t n) (x : V),
    omega (upd (fun k ⇒ f (vs k)) i (f x)) == fstar_fun (upd vs i x).
  Proof.
    intros vs i x.
    unfold fstar_fun.
    apply (mf_proper omega).
    intros j.
    symmetry.
    exact (fstar_upd_comm vs i x j).
  Qed.

  Lemma fstar_add : ∀ (vs : Fin.t n → V) (i : Fin.t n) (x y : V),
    fstar_fun (upd vs i (x + y))
      == fstar_fun (upd vs i x) + fstar_fun (upd vs i y).
  Proof.
    intros vs i x y.
    unfold fstar_fun.
    transitivity (omega (upd (fun k ⇒ f (vs k)) i (f (x + y)))).
    - apply (mf_proper omega). intros j. exact (fstar_upd_comm vs i (x + y) j).
    - transitivity (omega (upd (fun k ⇒ f (vs k)) i (f x + f y))).
      + apply (mf_proper omega).
        exact (upd_val_proper (fun k ⇒ f (vs k)) i (f (x + y)) (f x + f y)
                 (lm_add f x y)).
      + transitivity (omega (upd (fun k ⇒ f (vs k)) i (f x))
                       + omega (upd (fun k ⇒ f (vs k)) i (f y))).
        × exact (mf_add omega (fun k ⇒ f (vs k)) i (f x) (f y)).
        × exact (mon_add_proper
                   (omega (upd (fun k ⇒ f (vs k)) i (f x)))
                   (fstar_fun (upd vs i x)) (fstar_upd_eq vs i x)
                   (omega (upd (fun k ⇒ f (vs k)) i (f y)))
                   (fstar_fun (upd vs i y)) (fstar_upd_eq vs i y)).
  Qed.

  Lemma fstar_scale : ∀ (vs : Fin.t n → V) (i : Fin.t n) (a : F) (x : V),
    fstar_fun (upd vs i (a *: x)) == a × fstar_fun (upd vs i x).
  Proof.
    intros vs i a x.
    unfold fstar_fun.
    transitivity (omega (upd (fun k ⇒ f (vs k)) i (f (a *: x)))).
    - apply (mf_proper omega). intros j. exact (fstar_upd_comm vs i (a *: x) j).
    - transitivity (omega (upd (fun k ⇒ f (vs k)) i (a *: f x))).
      + apply (mf_proper omega).
        exact (upd_val_proper (fun k ⇒ f (vs k)) i (f (a *: x)) (a *: f x)
                 (lm_scale f a x)).
      + transitivity (a × omega (upd (fun k ⇒ f (vs k)) i (f x))).
        × exact (mf_scale omega (fun k ⇒ f (vs k)) i a (f x)).
        × exact (ring_mul_proper a a (reflexivity a)
                   (omega (upd (fun k ⇒ f (vs k)) i (f x)))
                   (fstar_fun (upd vs i x)) (fstar_upd_eq vs i x)).
  Qed.

  Definition fstar : MultilinearForm n F V :=
    @Build_MultilinearForm n F V FF VV fstar_fun fstar_proper fstar_add fstar_scale.

  Theorem fstar_alternating : Alternating omega → Alternating fstar.
  Proof.
    intros Halt.
    apply Build_Alternating.
    intros vs i j Hij Heq.
    exact (alt_repeat Halt (fun k ⇒ f (vs k)) i j Hij
             (lm_proper f (vs i) (vs j) Heq)).
  Qed.

End Pullback.


Section CanonicalForm.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V} {p : nat}.
  Context (B : Basis (S p) F V).

  Definition mat_of_fam (vs : Fin.t (S p) → V) : Matrix (S p) (S p) F :=
    fun i j ⇒ basis_coords B (vs j) i.

  Definition omega_B (vs : Fin.t (S p) → V) : F := detM (mat_of_fam vs).

  Lemma omega_B_proper : Proper (@famEq V _ (S p) ==> equ) omega_B.
  Proof.
    intros vs ws Hvw.
    apply detM_proper.
    intros i j.
    exact (lm_proper (basis_coords B) (vs j) (ws j) (Hvw j) i).
  Qed.

  Lemma upd_F1_eq : ∀ (vs : Fin.t (S p) → V) (x : V),
    upd vs Fin.F1 x Fin.F1 == x.
  Proof.
    intros vs x.
    unfold upd; destruct (Fin.eq_dec Fin.F1 Fin.F1) as [_ | Hne].
    - reflexivity.
    - destruct (Hne eq_refl).
  Qed.

  Lemma upd_F1_off : ∀ (vs : Fin.t (S p) → V) (x : V) (j' : Fin.t p),
    upd vs Fin.F1 x (Fin.FS j') == vs (Fin.FS j').
  Proof.
    intros vs x j'.
    unfold upd; destruct (Fin.eq_dec Fin.F1 (Fin.FS j')) as [Heq | _].
    - discriminate Heq.
    - reflexivity.
  Qed.

  Lemma mat_of_fam_upd_F1 : ∀ (vs : Fin.t (S p) → V) (x : V)
      (i j : Fin.t (S p)),
    mat_of_fam (upd vs Fin.F1 x) i j
      == upd_col1 (mat_of_fam vs) (fun i0 ⇒ basis_coords B x i0) i j.
  Proof.
    intros vs x i j.
    apply (Fin.caseS' j
             (fun j0 ⇒ mat_of_fam (upd vs Fin.F1 x) i j0
               == upd_col1 (mat_of_fam vs) (fun i0 ⇒ basis_coords B x i0) i j0)).
    - unfold mat_of_fam.
      transitivity (basis_coords B x i).
      + exact (lm_proper (basis_coords B) (upd vs Fin.F1 x Fin.F1) x
                 (upd_F1_eq vs x) i).
      + symmetry.
        exact (upd_col1_F1 (S p) p (mat_of_fam vs)
                 (fun i0 ⇒ basis_coords B x i0) i).
    - intros j'.
      unfold mat_of_fam.
      transitivity (basis_coords B (vs (Fin.FS j')) i).
      + exact (lm_proper (basis_coords B) (upd vs Fin.F1 x (Fin.FS j'))
                 (vs (Fin.FS j')) (upd_F1_off vs x j') i).
      + symmetry.
        exact (upd_col1_FS (S p) p (mat_of_fam vs)
                 (fun i0 ⇒ basis_coords B x i0) i j').
  Qed.

  Lemma mat_of_fam_upd_F1_sym : ∀ (vs : Fin.t (S p) → V) (x : V)
      (i j : Fin.t (S p)),
    upd_col1 (mat_of_fam vs) (fun i0 ⇒ basis_coords B x i0) i j
      == mat_of_fam (upd vs Fin.F1 x) i j.
  Proof.
    intros vs x i j.
    symmetry.
    exact (mat_of_fam_upd_F1 vs x i j).
  Qed.

  Lemma upd_col1_ext : ∀ (A : Matrix (S p) (S p) F) (c c' : Fin.t (S p) → F),
    (∀ i, c i == c' i) →
    ∀ i j : Fin.t (S p), upd_col1 A c i j == upd_col1 A c' i j.
  Proof.
    intros A c c' Hc i j.
    apply (Fin.caseS' j (fun j0 ⇒ upd_col1 A c i j0 == upd_col1 A c' i j0)).
    - rewrite (upd_col1_F1 (S p) p A c i), (upd_col1_F1 (S p) p A c' i).
      exact (Hc i).
    - intros j'.
      rewrite (upd_col1_FS (S p) p A c i j'), (upd_col1_FS (S p) p A c' i j').
      reflexivity.
  Qed.

  Theorem omega_B_add_F1 : ∀ (vs : Fin.t (S p) → V) (x y : V),
    omega_B (upd vs Fin.F1 (x + y))
      == omega_B (upd vs Fin.F1 x) + omega_B (upd vs Fin.F1 y).
  Proof.
    intros vs x y.
    unfold omega_B.
    transitivity
      (detM (upd_col1 (mat_of_fam vs) (fun i ⇒ basis_coords B (x + y) i))).
    - exact (detM_proper (S p) (mat_of_fam (upd vs Fin.F1 (x + y)))
               (upd_col1 (mat_of_fam vs) (fun i ⇒ basis_coords B (x + y) i))
               (mat_of_fam_upd_F1 vs (x + y))).
    - transitivity
        (detM (upd_col1 (mat_of_fam vs)
           (fun i ⇒ basis_coords B x i + basis_coords B y i))).
      + exact (detM_proper (S p) _ _
                 (upd_col1_ext (mat_of_fam vs) (fun i ⇒ basis_coords B (x + y) i)
                    (fun i ⇒ basis_coords B x i + basis_coords B y i)
                    (fun i ⇒ lm_add (basis_coords B) x y i))).
      + transitivity
          (detM (upd_col1 (mat_of_fam vs) (fun i ⇒ basis_coords B x i))
           + detM (upd_col1 (mat_of_fam vs) (fun i ⇒ basis_coords B y i))).
        × exact (det_col1_add p (mat_of_fam vs)
                   (fun i ⇒ basis_coords B x i) (fun i ⇒ basis_coords B y i)).
        × exact (mon_add_proper
                   (detM (upd_col1 (mat_of_fam vs) (fun i ⇒ basis_coords B x i)))
                   (omega_B (upd vs Fin.F1 x))
                   (detM_proper (S p) _ _
                      (fun i j ⇒ mat_of_fam_upd_F1_sym vs x i j))
                   (detM (upd_col1 (mat_of_fam vs) (fun i ⇒ basis_coords B y i)))
                   (omega_B (upd vs Fin.F1 y))
                   (detM_proper (S p) _ _
                      (fun i j ⇒ mat_of_fam_upd_F1_sym vs y i j))).
  Qed.

  Theorem omega_B_scale_F1 : ∀ (vs : Fin.t (S p) → V) (a : F) (x : V),
    omega_B (upd vs Fin.F1 (a *: x)) == a × omega_B (upd vs Fin.F1 x).
  Proof.
    intros vs a x.
    unfold omega_B.
    transitivity
      (detM (upd_col1 (mat_of_fam vs) (fun i ⇒ basis_coords B (a *: x) i))).
    - exact (detM_proper (S p) (mat_of_fam (upd vs Fin.F1 (a *: x)))
               (upd_col1 (mat_of_fam vs) (fun i ⇒ basis_coords B (a *: x) i))
               (mat_of_fam_upd_F1 vs (a *: x))).
    - transitivity
        (detM (upd_col1 (mat_of_fam vs) (fun i ⇒ a × basis_coords B x i))).
      + exact (detM_proper (S p) _ _
                 (upd_col1_ext (mat_of_fam vs) (fun i ⇒ basis_coords B (a *: x) i)
                    (fun i ⇒ a × basis_coords B x i)
                    (fun i ⇒ lm_scale (basis_coords B) a x i))).
      + transitivity (a × detM (upd_col1 (mat_of_fam vs) (fun i ⇒ basis_coords B x i))).
        × exact (det_col1_scale p (mat_of_fam vs) a (fun i ⇒ basis_coords B x i)).
        × exact (ring_mul_proper a a (reflexivity a)
                   (detM (upd_col1 (mat_of_fam vs) (fun i ⇒ basis_coords B x i)))
                   (omega_B (upd vs Fin.F1 x))
                   (detM_proper (S p) _ _
                      (fun i j ⇒ mat_of_fam_upd_F1_sym vs x i j))).
  Qed.


End CanonicalForm.


Section EndoDet.
  Context {F U : Type} {FF : Field F} {VU : VectorSpace F U} {n : nat}.

  Definition det (B : Basis n F U) (f : LinearMap F U U) : F :=
    detM (mat_of B B f).

End EndoDet.


Section EndoDetConditional.
  Context {F U : Type} {FF : Field F} {VU : VectorSpace F U} {n : nat}.
  Context (detM_mul : ∀ (m : nat) (A A' : Matrix m m F),
    detM (mat_mul A A') == detM A × detM A').

  Lemma cob_det_inv : ∀ B B' : Basis n F U,
    detM (cob B' B) × detM (cob B B') == 1.
  Proof.
    intros B B'.
    transitivity (detM (mat_mul (cob B' B) (cob B B'))).
    - symmetry. exact (detM_mul n (cob B' B) (cob B B')).
    - transitivity (detM (@mat_id F FF n)).
      + exact (detM_proper n (mat_mul (cob B' B) (cob B B')) mat_id
                 (cob_inv B B')).
      + exact (det_id n).
  Qed.

  Theorem det_basis_indep : ∀ (B B' : Basis n F U) (f : LinearMap F U U),
    det B' f == det B f.
  Proof.
    intros B B' f.
    unfold det.
    transitivity
      (detM (mat_mul (cob B' B) (mat_mul (mat_of B B f) (cob B B')))).
    - exact (detM_proper n (mat_of B' B' f)
               (mat_mul (cob B' B) (mat_mul (mat_of B B f) (cob B B')))
               (mat_of_conj B B' f)).
    - transitivity
        (detM (cob B' B) × detM (mat_mul (mat_of B B f) (cob B B'))).
      + exact (detM_mul n (cob B' B) (mat_mul (mat_of B B f) (cob B B'))).
      + transitivity
          (detM (cob B' B) × (detM (mat_of B B f) × detM (cob B B'))).
        × exact (ring_mul_proper (detM (cob B' B)) (detM (cob B' B))
                   (reflexivity (detM (cob B' B)))
                   (detM (mat_mul (mat_of B B f) (cob B B')))
                   (detM (mat_of B B f) × detM (cob B B'))
                   (detM_mul n (mat_of B B f) (cob B B'))).
        × transitivity
            (detM (mat_of B B f) × (detM (cob B' B) × detM (cob B B'))).
          -- exact (mul_comm_swap (detM (cob B' B)) (detM (mat_of B B f))
                       (detM (cob B B'))).
          -- transitivity (detM (mat_of B B f) × 1).
             ++ exact (ring_mul_proper (detM (mat_of B B f)) (detM (mat_of B B f))
                          (reflexivity (detM (mat_of B B f)))
                          (detM (cob B' B) × detM (cob B B')) 1
                          (cob_det_inv B B')).
             ++ exact (ring_mul_1r (detM (mat_of B B f))).
  Qed.

  Theorem det_comp : ∀ (B : Basis n F U) (f g : LinearMap F U U),
    det B (lm_comp f g) == det B f × det B g.
  Proof.
    intros B f g.
    unfold det.
    transitivity (detM (mat_mul (mat_of B B f) (mat_of B B g))).
    - exact (detM_proper n (mat_of B B (lm_comp f g))
               (mat_mul (mat_of B B f) (mat_of B B g)) (mat_of_comp B B B f g)).
    - exact (detM_mul n (mat_of B B f) (mat_of B B g)).
  Qed.

End EndoDetConditional.


Section EndoDetId.
  Context {F U : Type} {FF : Field F} {VU : VectorSpace F U} {n : nat}.

  Theorem det_id : ∀ B : Basis n F U, det B (@lm_id F U FF VU) == 1.
  Proof.
    intros B.
    unfold det.
    transitivity (detM (@mat_id F FF n)).
    - exact (detM_proper n (mat_of B B (@lm_id F U FF VU)) mat_id (mat_of_id B)).
    - exact (det_id n).
  Qed.

End EndoDetId.
