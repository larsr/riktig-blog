From Coq Require Import List ZArith.
Import ListNotations.
From Riktig Require Import Programs.Memprog.Syntax.
From Riktig Require Import Programs.Memprog.Machine.



Definition with_frame {addr val a : Type}
  (body : MemT addr val a) : MemT addr val a :=
  do! base <- get_sp_m;
  push_addr base >>
  do! result <- body;
  do! old_sp <- read_addr_m base;
  set_sp_m old_sp >>
  ret_t result.


Definition call_frame {addr val a : Type}
  (push_args : MemT addr val unit)
  (callee : addr → MemT addr val a) : MemT addr val a :=
  push_args >>
  do! sp <- get_sp_m;
  callee sp.


Section Slots.
  Context {addr val : Type}.
  Context (val_sub : val → val → val).
  Context (val0 : val).
  Context (val_of_nat : nat → val).


  Definition slot_offset (n : nat) : val := val_sub val0 (val_of_nat n).

  Definition slot_addr (fp : addr) (n : nat) : MemT addr val addr :=
    offset_m fp (slot_offset n).

  Definition read_slot (fp : addr) (n : nat) : MemT addr val val :=
    do! a <- slot_addr fp n; read_m a.

  Definition read_slot_addr (fp : addr) (n : nat) : MemT addr val addr :=
    do! a <- slot_addr fp n; read_addr_m a.

  Definition write_slot (fp : addr) (n : nat) (v : val) : MemT addr val unit :=
    do! a <- slot_addr fp n; write_m a v.

End Slots.


Local Open Scope Z_scope.

Definition slot_addr_z : Z → nat → MemT Z Z Z :=
  slot_addr Z.sub 0 Z.of_nat.
Definition read_slot_z : Z → nat → MemT Z Z Z :=
  read_slot Z.sub 0 Z.of_nat.
Definition read_slot_addr_z : Z → nat → MemT Z Z Z :=
  read_slot_addr Z.sub 0 Z.of_nat.
Definition write_slot_z : Z → nat → Z → MemT Z Z unit :=
  write_slot Z.sub 0 Z.of_nat.



Definition add_seven : MemT Z Z Z :=
  call_frame (push_val 35)
    (fun _ ⇒ with_frame (
       stack_alloc_n 1 >>
       do! fp <- get_sp_m;
       write_slot_z fp 1 7 >>
       do! x <- read_slot_z fp 1;
       do! y <- read_slot_z fp 3;
       ret_t (x + y))).


Definition frame_demo : MemT Z Z (Z × Z) :=
  do! r <- add_seven;
  do! sp <- get_sp_m;
  ret_t (r, sp).

Example frame_demo_result :
  fst (run_z 1000 frame_demo 32) = Some (42, 1).
Proof. vm_compute. reflexivity. Qed.
