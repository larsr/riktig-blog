Library Riktig.Theory.Order

From Coq Require Import Morphisms.
From Riktig Require Import Ops Props Interface.AbGroup Interface.Ring
  Interface.Field Interface.Order Theory.Group Theory.AbGroup Theory.Ring
  Theory.Field.

#[global] Instance lt_proper {F : Type} `{OrderedField F} :
  Proper (equ ==> equ ==> iff) (@lt F _ _).
Proof.
  intros a b Hab c d Hcd.
  split; intros [Hle Hne]; split.
  - rewrite <- Hab, <- Hcd. exact Hle.
  - intros Heq. apply Hne. rewrite Hab, Hcd. exact Heq.
  - rewrite Hab, Hcd. exact Hle.
  - intros Heq. apply Hne. rewrite <- Hab, <- Hcd. exact Heq.
Qed.

Section OrderedFieldLemmas.
  Context {F : Type} `{OrderedField F}.

  Lemma abs_zero : abs (0 : F) == 0.
  Proof.
    apply (proj2 (abs_zero_iff 0)).
    reflexivity.
  Qed.

  Lemma abs_one : abs (1 : F) == 1.
  Proof.
    apply (mul_cancel_l (abs 1)).
    - intros Hz. exact (field_nontrivial (proj1 (abs_zero_iff 1) Hz)).
    - rewrite <- abs_mul, ring_mul_1r, ring_mul_1r. reflexivity.
  Qed.

  Lemma zero_le_one : (0 : F) ≤ 1.
  Proof.
    rewrite <- abs_one.
    apply abs_nonneg.
  Qed.

  Lemma eq_le : ∀ a b : F, a == b → a ≤ b.
  Proof.
    intros a b Hab.
    rewrite Hab.
    apply le_refl.
  Qed.

  Lemma le_rewrite_l : ∀ a b c : F, a == b → a ≤ c → b ≤ c.
  Proof.
    intros a b c Hab Hac.
    rewrite <- Hab.
    exact Hac.
  Qed.

  Lemma lt_le : ∀ a b : F, a < b → a ≤ b.
  Proof.
    intros a b Hab.
    destruct Hab as [Hle Hne].
    exact Hle.
  Qed.

  Lemma lt_irrefl : ∀ a : F, ¬ a < a.
  Proof.
    intros a [Hle Hne].
    apply Hne.
    reflexivity.
  Qed.

  Lemma pos_neq_0 : ∀ a : F, 0 < a → a =/= 0.
  Proof.
    intros a [Hle Hne] Heq.
    apply Hne.
    symmetry.
    exact Heq.
  Qed.

  Lemma le_lt_trans : ∀ a b c : F, a ≤ b → b < c → a < c.
  Proof.
    intros a b c Hab [Hbc Hne].
    split.
    - exact (le_trans a b c Hab Hbc).
    - intros Heq.
      apply Hne.
      apply (le_antisym b c Hbc).
      rewrite <- Heq.
      exact Hab.
  Qed.

  Lemma lt_le_trans : ∀ a b c : F, a < b → b ≤ c → a < c.
  Proof.
    intros a b c [Hab Hne] Hbc.
    split.
    - exact (le_trans a b c Hab Hbc).
    - intros Heq.
      apply Hne.
      apply (le_antisym a b Hab).
      rewrite Heq.
      exact Hbc.
  Qed.

  Lemma lt_trans : ∀ a b c : F, a < b → b < c → a < c.
  Proof.
    intros a b c Hab Hbc.
    exact (lt_le_trans a b c Hab (lt_le b c Hbc)).
  Qed.

  Lemma add_le_mono_r : ∀ a b c : F, a ≤ b → a + c ≤ b + c.
  Proof.
    intros a b c Hab.
    rewrite (ab_comm a c), (ab_comm b c).
    apply add_le_mono_l.
    exact Hab.
  Qed.

  Lemma add_le_mono : ∀ a b c d : F, a ≤ b → c ≤ d → a + c ≤ b + d.
  Proof.
    intros a b c d Hab Hcd.
    apply (le_trans (a + c) (b + c) (b + d)).
    - apply add_le_mono_r. exact Hab.
    - apply add_le_mono_l. exact Hcd.
  Qed.

  Lemma add_lt_mono_l : ∀ a b c : F, a < b → c + a < c + b.
  Proof.
    intros a b c [Hab Hne].
    split.
    - apply add_le_mono_l. exact Hab.
    - intros Heq.
      apply Hne.
      exact (cancel_l c a b Heq).
  Qed.

  Lemma add_lt_mono_r : ∀ a b c : F, a < b → a + c < b + c.
  Proof.
    intros a b c Hab.
    rewrite (ab_comm a c), (ab_comm b c).
    apply add_lt_mono_l.
    exact Hab.
  Qed.

  Lemma add_le_lt_mono : ∀ a b c d : F, a ≤ b → c < d → a + c < b + d.
  Proof.
    intros a b c d Hab Hcd.
    apply (le_lt_trans (a + c) (b + c) (b + d)).
    - apply add_le_mono_r. exact Hab.
    - apply add_lt_mono_l. exact Hcd.
  Qed.

  Lemma add_lt_le_mono : ∀ a b c d : F, a < b → c ≤ d → a + c < b + d.
  Proof.
    intros a b c d Hab Hcd.
    apply (lt_le_trans (a + c) (b + c) (b + d)).
    - apply add_lt_mono_r. exact Hab.
    - apply add_le_mono_l. exact Hcd.
  Qed.

  Lemma add_lt_mono : ∀ a b c d : F, a < b → c < d → a + c < b + d.
  Proof.
    intros a b c d Hab Hcd.
    exact (add_lt_le_mono a b c d Hab (lt_le c d Hcd)).
  Qed.

  Lemma lt_add_pos_r : ∀ a b : F, 0 < b → a < a + b.
  Proof.
    intros a b Hb.
    rewrite <- (idr a), assoc, idl.
    apply add_lt_mono_l.
    exact Hb.
  Qed.

  Lemma sub_nonneg_of_le : ∀ a b : F, a ≤ b → 0 ≤ b + - a.
  Proof.
    intros a b Hab.
    rewrite <- (invr a).
    apply add_le_mono_r.
    exact Hab.
  Qed.

  Lemma le_of_sub_nonneg : ∀ a b : F, 0 ≤ b + - a → a ≤ b.
  Proof.
    intros a b Hab.
    rewrite <- (add_sub_cancel a b).
    apply (le_trans a (a + 0) (a + (b + - a))).
    - apply eq_le. symmetry. apply idr.
    - apply add_le_mono_l. exact Hab.
  Qed.

  Lemma neg_nonneg : ∀ a : F, a ≤ 0 → 0 ≤ - a.
  Proof.
    intros a Ha.
    rewrite <- (idl (- a)).
    apply sub_nonneg_of_le.
    exact Ha.
  Qed.

  Lemma mul_nonneg_nonpos : ∀ a b : F, 0 ≤ a → b ≤ 0 → a × b ≤ 0.
  Proof.
    intros a b Ha Hb.
    apply (le_of_sub_nonneg (a × b) 0).
    rewrite idl, <- mul_neg_r.
    apply mul_nonneg.
    - exact Ha.
    - apply neg_nonneg. exact Hb.
  Qed.

  Lemma mul_le_mono_l : ∀ a b c : F, 0 ≤ c → a ≤ b → c × a ≤ c × b.
  Proof.
    intros a b c Hc Hab.
    apply le_of_sub_nonneg.
    rewrite <- mul_neg_r, <- ring_distl.
    apply mul_nonneg.
    - exact Hc.
    - apply sub_nonneg_of_le. exact Hab.
  Qed.

  Lemma mul_le_mono_r : ∀ a b c : F, 0 ≤ c → a ≤ b → a × c ≤ b × c.
  Proof.
    intros a b c Hc Hab.
    rewrite (field_mul_comm a c), (field_mul_comm b c).
    apply mul_le_mono_l.
    - exact Hc.
    - exact Hab.
  Qed.

  Lemma mul_pos : ∀ a b : F, 0 < a → 0 < b → 0 < a × b.
  Proof.
    intros a b Ha Hb.
    split.
    - apply mul_nonneg.
      + apply lt_le. exact Ha.
      + apply lt_le. exact Hb.
    - intros Heq.
      apply (field_mul_neq_0 a b (pos_neq_0 a Ha) (pos_neq_0 b Hb)).
      symmetry.
      exact Heq.
  Qed.

  Lemma lt_0_1 : (0 : F) < 1.
  Proof.
    split.
    - apply zero_le_one.
    - intros Heq.
      apply field_nontrivial.
      symmetry.
      exact Heq.
  Qed.

  Lemma not_le_1_0 : ¬ (1 : F) ≤ 0.
  Proof.
    intros Hle.
    apply field_nontrivial.
    apply (le_antisym 1 0).
    - exact Hle.
    - exact zero_le_one.
  Qed.

  Definition two : F := 1 + 1.

  Lemma two_eq : two == 1 + 1.
  Proof. reflexivity. Qed.

  Lemma two_pos : 0 < two.
  Proof.
    rewrite two_eq, <- (idl (0 : F)).
    apply add_lt_mono.
    - exact lt_0_1.
    - exact lt_0_1.
  Qed.

  Lemma two_neq_0 : two =/= 0.
  Proof.
    apply pos_neq_0.
    exact two_pos.
  Qed.

  Lemma pos_inv : ∀ a : F, 0 < a → 0 < inv a.
  Proof.
    intros a Ha.
    split.
    - destruct (le_total 0 (inv a)) as [Hok | Hbad].
      + exact Hok.
      + destruct (not_le_1_0 (le_rewrite_l (a × inv a) 1 0
          (field_invr a (pos_neq_0 a Ha))
          (mul_nonneg_nonpos a (inv a) (lt_le 0 a Ha) Hbad))).
    - intros Heq.
      apply field_nontrivial.
      rewrite <- (field_invr a (pos_neq_0 a Ha)), <- Heq, mul_0_r.
      reflexivity.
  Qed.

  Definition half (a : F) : F := a × inv two.

  Lemma half_eq : ∀ a : F, half a == a × inv two.
  Proof. intros a. reflexivity. Qed.

  Lemma mul_two : ∀ a : F, a × two == a + a.
  Proof.
    intros a.
    rewrite two_eq, ring_distl, ring_mul_1r.
    reflexivity.
  Qed.

  Lemma half_sum : ∀ a : F, half a + half a == a.
  Proof.
    intros a.
    rewrite half_eq, <- ring_distr, <- mul_two, ring_mul_assoc,
      (field_invr two two_neq_0), ring_mul_1r.
    reflexivity.
  Qed.

  Lemma half_pos : ∀ a : F, 0 < a → 0 < half a.
  Proof.
    intros a Ha.
    exact (mul_pos a (inv two) Ha (pos_inv two two_pos)).
  Qed.

  Lemma half_lt : ∀ a : F, 0 < a → half a < a.
  Proof.
    intros a Ha.
    apply (lt_le_trans (half a) (half a + half a) a).
    - apply lt_add_pos_r. apply half_pos. exact Ha.
    - apply eq_le. apply half_sum.
  Qed.

  Lemma pos_min_exists : ∀ a b : F, 0 < a → 0 < b →
    ∃ c : F, 0 < c ∧ c ≤ a ∧ c ≤ b.
  Proof.
    intros a b Ha Hb.
    destruct (le_total a b) as [Hab | Hba].
    - ∃ a.
      split.
      + exact Ha.
      + split.
        × apply le_refl.
        × exact Hab.
    - ∃ b.
      split.
      + exact Hb.
      + split.
        × exact Hba.
        × apply le_refl.
  Qed.

  Lemma one_plus_nonneg_pos : ∀ a : F, 0 ≤ a → 0 < 1 + a.
  Proof.
    intros a Ha.
    rewrite <- (idl (0 : F)).
    apply add_lt_le_mono.
    - exact lt_0_1.
    - exact Ha.
  Qed.

  Lemma mul_lt_pad : ∀ a t d : F,
    0 ≤ a → 0 ≤ t → t < d → a × t < (1 + a) × d.
  Proof.
    intros a t d Ha Ht Htd.
    apply (le_lt_trans (a × t) (a × d) ((1 + a) × d)).
    - apply mul_le_mono_l.
      + exact Ha.
      + apply lt_le. exact Htd.
    - rewrite ring_distr, ring_mul_1l, (ab_comm d (a × d)).
      apply lt_add_pos_r.
      exact (le_lt_trans 0 t d Ht Htd).
  Qed.

  Lemma all_pos_bound_not_pos : ∀ x : F,
    (∀ eps : F, 0 < eps → x ≤ eps) → ¬ 0 < x.
  Proof.
    intros x Hall Hpos.
    apply (lt_irrefl x).
    exact (le_lt_trans x (half x) x (Hall (half x) (half_pos x Hpos))
             (half_lt x Hpos)).
  Qed.

  Lemma nonneg_below_all_pos_is_zero : ∀ x : F,
    0 ≤ x → (∀ eps : F, 0 < eps → x ≤ eps) → x == 0.
  Proof.
    intros x Hx Hall.
    apply (le_antisym x 0).
    - apply le_stable.
      intros Hnot.
      apply (all_pos_bound_not_pos x Hall).
      split.
      + exact Hx.
      + intros Heq.
        apply Hnot.
        rewrite <- Heq.
        apply le_refl.
    - exact Hx.
  Qed.

  Lemma lt_one_add : ∀ a : F, a < 1 + a.
  Proof.
    intros a.
    rewrite (ab_comm (1 : F) a).
    apply lt_add_pos_r.
    exact lt_0_1.
  Qed.

  Lemma le_one_add : ∀ a : F, a ≤ 1 + a.
  Proof.
    intros a.
    apply lt_le.
    apply lt_one_add.
  Qed.

  Lemma mul_lt_mono_l : ∀ a b c : F, 0 < c → a < b → c × a < c × b.
  Proof.
    intros a b c Hc [Hab Hne].
    split.
    - apply mul_le_mono_l.
      + apply lt_le. exact Hc.
      + exact Hab.
    - intros Heq.
      apply Hne.
      exact (mul_cancel_l c a b (pos_neq_0 c Hc) Heq).
  Qed.

  Lemma mul_le_cancel_l : ∀ a b c : F, 0 < c → c × a ≤ c × b → a ≤ b.
  Proof.
    intros a b c Hc Hab.
    rewrite <- (inv_mul_cancel c a (pos_neq_0 c Hc)),
      <- (inv_mul_cancel c b (pos_neq_0 c Hc)).
    apply mul_le_mono_l.
    - apply lt_le. apply pos_inv. exact Hc.
    - exact Hab.
  Qed.

  Lemma mul_pad_div_le : ∀ a c : F,
    0 ≤ a → 0 ≤ c → c × (a × inv (1 + c)) ≤ a.
  Proof.
    intros a c Ha Hc.
    apply (le_trans (c × (a × inv (1 + c))) ((1 + c) × (a × inv (1 + c))) a).
    - apply mul_le_mono_r.
      + apply mul_nonneg.
        × exact Ha.
        × apply lt_le. apply pos_inv. apply one_plus_nonneg_pos. exact Hc.
      + apply le_one_add.
    - apply eq_le.
      apply field_mul_div_cancel.
      apply pos_neq_0.
      apply one_plus_nonneg_pos.
      exact Hc.
  Qed.

  Lemma small_scale_exists : ∀ d c : F, 0 < d → 0 ≤ c →
    ∃ t : F, 0 < t ∧ t × c < d.
  Proof.
    intros d c Hd Hc.
    ∃ (half d × inv (1 + c)).
    split.
    - apply mul_pos.
      + apply half_pos. exact Hd.
      + apply pos_inv. apply one_plus_nonneg_pos. exact Hc.
    - apply (lt_le_trans (half d × inv (1 + c) × c)
        (half d × inv (1 + c) × (1 + c)) d).
      + apply mul_lt_mono_l.
        × apply mul_pos.
          -- apply half_pos. exact Hd.
          -- apply pos_inv. apply one_plus_nonneg_pos. exact Hc.
        × apply lt_one_add.
      + apply (le_trans (half d × inv (1 + c) × (1 + c)) (half d) d).
        × apply (le_rewrite_l ((1 + c) × (half d × inv (1 + c)))).
          -- apply field_mul_comm.
          -- apply eq_le.
             apply field_mul_div_cancel.
             apply pos_neq_0.
             apply one_plus_nonneg_pos.
             exact Hc.
        × apply lt_le. apply half_lt. exact Hd.
  Qed.

  Lemma half_mul_sum : ∀ a b : F, half a × b + half a × b == a × b.
  Proof.
    intros a b.
    rewrite <- (ring_distr (half a) (half a) b), half_sum.
    reflexivity.
  Qed.

  Lemma add_nonneg_pos : ∀ a b : F, 0 ≤ a → 0 < b → 0 < a + b.
  Proof.
    intros a b Ha Hb.
    rewrite <- (idl (0 : F)).
    apply add_le_lt_mono.
    - exact Ha.
    - exact Hb.
  Qed.

  Lemma le_add_nonneg_r : ∀ a b : F, 0 ≤ b → a ≤ a + b.
  Proof.
    intros a b Hb.
    rewrite <- (idr a), assoc, idl.
    apply add_le_mono_l.
    exact Hb.
  Qed.

  Lemma le_add_nonneg_l : ∀ a b : F, 0 ≤ a → b ≤ a + b.
  Proof.
    intros a b Ha.
    apply (le_trans b (b + a) (a + b)).
    - apply le_add_nonneg_r. exact Ha.
    - apply eq_le. apply ab_comm.
  Qed.

  Lemma nonneg_add_eq_zero : ∀ a b : F,
    0 ≤ a → 0 ≤ b → a + b == 0 → a == 0 ∧ b == 0.
  Proof.
    intros a b Ha Hb Hab.
    split.
    - apply le_antisym.
      + rewrite <- Hab. apply le_add_nonneg_r. exact Hb.
      + exact Ha.
    - apply le_antisym.
      + rewrite <- Hab. apply le_add_nonneg_l. exact Ha.
      + exact Hb.
  Qed.

  Lemma sq_nonneg : ∀ a : F, 0 ≤ a × a.
  Proof.
    intros a.
    destruct (le_total 0 a) as [Ha | Ha].
    - apply mul_nonneg.
      + exact Ha.
      + exact Ha.
    - rewrite <- (mul_neg_neg a a).
      apply mul_nonneg.
      + apply neg_nonneg. exact Ha.
      + apply neg_nonneg. exact Ha.
  Qed.

  Lemma add_le_cancel_r : ∀ a b c : F, a + c ≤ b + c → a ≤ b.
  Proof.
    intros a b c Habc.
    rewrite <- (idr a), <- (idr b), <- (invr c),
      <- (assoc a c (- c)), <- (assoc b c (- c)).
    apply add_le_mono_r.
    exact Habc.
  Qed.

  Lemma le_of_le_add_all_eps : ∀ a b : F,
    (∀ eps : F, 0 < eps → a ≤ b + eps) → a ≤ b.
  Proof.
    intros a b Hall.
    destruct (le_total a b) as [Hab | Hba].
    - exact Hab.
    - apply eq_le.
      apply (sub_eq_zero a b).
      apply nonneg_below_all_pos_is_zero.
      + apply sub_nonneg_of_le. exact Hba.
      + intros eps Heps.
        apply (le_trans (a + - b) ((b + eps) + - b) eps).
        × apply add_le_mono_r. exact (Hall eps Heps).
        × apply eq_le. apply (add_neg_cancel_l b eps).
  Qed.

  Lemma le_of_le_add_all_eps_mul : ∀ a b c : F, 0 ≤ c →
    (∀ eps : F, 0 < eps → a ≤ b + eps × c) → a ≤ b.
  Proof.
    intros a b c Hc Hall.
    apply le_of_le_add_all_eps.
    intros eps Heps.
    destruct (small_scale_exists eps c Heps Hc) as [d [Hd Hdc]].
    apply (le_trans a (b + d × c) (b + eps)).
    - exact (Hall d Hd).
    - apply add_le_mono_l. apply lt_le. exact Hdc.
  Qed.

End OrderedFieldLemmas.
