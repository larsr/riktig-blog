  Theory.InnerProduct.

Definition orthogonal {F V : Type} `{InnerProduct F V} (u v : V) : Prop :=
  ⟨ u , v ⟩ == 0.

Definition proj {F V : Type} `{InnerProduct F V} (v u : V) : V :=
  (⟨ u , v ⟩ × inv (norm2 v)) *: v.


Section InnerKit.
  Context {F V : Type} `{InnerProduct F V}.

  Lemma orthogonal_eq : ∀ u v : V, orthogonal u v → ⟨ u , v ⟩ == 0.
  Proof.
    intros u v Huv.
    exact Huv.
  Qed.

  Lemma orthogonal_def : ∀ u v : V, ⟨ u , v ⟩ == 0 → orthogonal u v.
  Proof.
    intros u v Huv.
    exact Huv.
  Qed.

  Lemma proj_eq : ∀ u v : V,
    proj v u == (⟨ u , v ⟩ × inv (norm2 v)) *: v.
  Proof.
    intros u v.
    reflexivity.
  Qed.

  Theorem orthogonal_sym : ∀ u v : V, orthogonal u v → orthogonal v u.
  Proof.
    intros u v Huv.
    apply orthogonal_def.
    rewrite (ip_sym v u).
    exact (orthogonal_eq u v Huv).
  Qed.

  Theorem orthogonal_scale_r : ∀ (u v : V) (a : F),
    orthogonal u v → orthogonal u (a *: v).
  Proof.
    intros u v a Huv.
    apply orthogonal_def.
    rewrite (ip_scale_r a u v), (orthogonal_eq u v Huv), (mul_0_r a).
    reflexivity.
  Qed.

  Theorem norm2_neg : ∀ u : V, norm2 (- u) == norm2 u.
  Proof.
    intros u.
    rewrite (norm2_spec (- u)), (ip_neg_l u (- u)), (ip_neg_r u u),
      (neg_involutive (⟨ u , u ⟩)), (norm2_spec u).
    reflexivity.
  Qed.

  Theorem pythagoras : ∀ u v : V,
    orthogonal u v → norm2 (u + v) == norm2 u + norm2 v.
  Proof.
    intros u v Huv.
    rewrite (norm2_add u v), (ip_sym v u), (orthogonal_eq u v Huv),
      (idl (0 : F)), idr.
    reflexivity.
  Qed.

  Theorem norm2_scale_sub_expand : ∀ (a b : F) (u v : V),
    norm2 (a *: u + - (b *: v)) ==
      ((a × a) × norm2 u + (b × b) × norm2 v)
      + (- (a × (b × ⟨ u , v ⟩)) + - (a × (b × ⟨ u , v ⟩))).
  Proof.
    intros a b u v.
    rewrite (norm2_add (a *: u) (- (b *: v))), (norm2_scale a u),
      (norm2_neg (b *: v)), (norm2_scale b v),
      (ip_neg_r (a *: u) (b *: v)), (ip_neg_l (b *: v) (a *: u)),
      (ip_scale_l a u (b *: v)), (ip_scale_r b u v),
      (ip_scale_l b v (a *: u)), (ip_scale_r a v u), (ip_sym v u),
      (mul_comm_swap b a (⟨ u , v ⟩)).
    reflexivity.
  Qed.

  Lemma proj_sub_scale : ∀ (u v : V) (c : F),
    proj v u + - (c *: v) == (⟨ u , v ⟩ × inv (norm2 v) + - c) *: v.
  Proof.
    intros u v c.
    rewrite (sub_scale (⟨ u , v ⟩ × inv (norm2 v)) c v), (proj_eq u v).
    reflexivity.
  Qed.

  Theorem proj_orthogonal : ∀ u v : V,
    norm2 v =/= 0 → orthogonal (u + - proj v u) v.
  Proof.
    intros u v Hv.
    apply orthogonal_def.
    rewrite (ip_add_l u (- proj v u) v), (ip_neg_l (proj v u) v),
      (proj_eq u v), (ip_scale_l (⟨ u , v ⟩ × inv (norm2 v)) v v),
      <- (norm2_spec v),
      (field_mul_comm (⟨ u , v ⟩ × inv (norm2 v)) (norm2 v)),
      (field_mul_div_cancel (⟨ u , v ⟩) (norm2 v) Hv).
    apply invr.
  Qed.

End InnerKit.


Section PosInnerGeometry.
  Context {F V : Type} `{PosInner F V}.

  Theorem norm2_nonneg : ∀ u : V, 0 ≤ norm2 u.
  Proof.
    intros u.
    exact (pi_nonneg u).
  Qed.

  Theorem norm2_neq_0_iff : ∀ v : V, norm2 v =/= 0 ↔ v =/= 0.
  Proof.
    intros v.
    split.
    - intros Hn Hv.
      apply Hn.
      rewrite (norm2_spec v), Hv.
      apply ip_zero_l.
    - intros Hv Hn.
      apply Hv.
      apply pi_definite.
      exact Hn.
  Qed.

  Theorem cs_padded : ∀ (u v : V) (e : F), 0 < e →
    ⟨ u , v ⟩ × ⟨ u , v ⟩ ≤ (norm2 v + e) × norm2 u.
  Proof.
    intros u v e He.
    apply (mul_le_cancel_l (⟨ u , v ⟩ × ⟨ u , v ⟩)
      ((norm2 v + e) × norm2 u) (norm2 v + e)
      (add_nonneg_pos (norm2 v) e (norm2_nonneg v) He)).
    apply (add_le_cancel_r
      ((norm2 v + e) × (⟨ u , v ⟩ × ⟨ u , v ⟩))
      ((norm2 v + e) × ((norm2 v + e) × norm2 u))
      ((norm2 v + e) × (⟨ u , v ⟩ × ⟨ u , v ⟩))).
    apply (le_trans
      ((norm2 v + e) × (⟨ u , v ⟩ × ⟨ u , v ⟩)
        + (norm2 v + e) × (⟨ u , v ⟩ × ⟨ u , v ⟩))
      (((norm2 v + e) × (norm2 v + e)) × norm2 u
        + (⟨ u , v ⟩ × ⟨ u , v ⟩) × norm2 v)
      ((norm2 v + e) × ((norm2 v + e) × norm2 u)
        + (norm2 v + e) × (⟨ u , v ⟩ × ⟨ u , v ⟩))).
    - apply le_of_sub_nonneg.
      rewrite (neg_add ((norm2 v + e) × (⟨ u , v ⟩ × ⟨ u , v ⟩))
                       ((norm2 v + e) × (⟨ u , v ⟩ × ⟨ u , v ⟩))),
        <- (norm2_scale_sub_expand (norm2 v + e) (⟨ u , v ⟩) u v).
      apply norm2_nonneg.
    - apply add_le_mono.
      + apply eq_le.
        apply (ring_mul_assoc (norm2 v + e) (norm2 v + e) (norm2 u)).
      + rewrite (field_mul_comm (norm2 v + e) (⟨ u , v ⟩ × ⟨ u , v ⟩)).
        apply mul_le_mono_l.
        × apply sq_nonneg.
        × apply le_add_nonneg_r. apply lt_le. exact He.
  Qed.

  Theorem cauchy_schwarz : ∀ u v : V,
    ⟨ u , v ⟩ × ⟨ u , v ⟩ ≤ norm2 u × norm2 v.
  Proof.
    intros u v.
    apply (le_of_le_add_all_eps_mul (⟨ u , v ⟩ × ⟨ u , v ⟩)
      (norm2 u × norm2 v) (norm2 u) (norm2_nonneg u)).
    intros e He.
    rewrite (field_mul_comm (norm2 u) (norm2 v)),
      <- (ring_distr (norm2 v) e (norm2 u)).
    exact (cs_padded u v e He).
  Qed.

  Theorem best_approx : ∀ (u v : V) (c : F), norm2 v =/= 0 →
    norm2 (u + - proj v u) ≤ norm2 (u + - (c *: v)).
  Proof.
    intros u v c Hv.
    rewrite (norm2_spec (u + - (c *: v))),
      <- (add_neg_telescope u (proj v u) (c *: v)),
      (proj_sub_scale u v c),
      <- (norm2_spec
            ((u + - proj v u) + ((⟨ u , v ⟩ × inv (norm2 v) + - c) *: v))),
      (pythagoras (u + - proj v u)
        ((⟨ u , v ⟩ × inv (norm2 v) + - c) *: v)
        (orthogonal_scale_r (u + - proj v u) v
          (⟨ u , v ⟩ × inv (norm2 v) + - c) (proj_orthogonal u v Hv))).
    apply le_add_nonneg_r.
    apply norm2_nonneg.
  Qed.

  Theorem inner_nondegenerate : ∀ x : V,
    (∀ w : V, ⟨ w , x ⟩ == 0) → x == 0.
  Proof.
    intros x Hx.
    apply pi_definite.
    exact (Hx x).
  Qed.

  Theorem adjoint_unique : ∀ f g1 g2 : LinearMap F V V,
    (∀ u w : V, ⟨ lm_fun g1 u , w ⟩ == ⟨ u , lm_fun f w ⟩) →
    (∀ u w : V, ⟨ lm_fun g2 u , w ⟩ == ⟨ u , lm_fun f w ⟩) →
    ∀ u : V, lm_fun g1 u == lm_fun g2 u.
  Proof.
    intros f g1 g2 Hg1 Hg2 u.
    apply (sub_eq_zero (lm_fun g1 u) (lm_fun g2 u)).
    apply inner_nondegenerate.
    intros w.
    rewrite (ip_add_r w (lm_fun g1 u) (- lm_fun g2 u)),
      (ip_neg_r w (lm_fun g2 u)), (ip_sym w (lm_fun g1 u)),
      (ip_sym w (lm_fun g2 u)), (Hg1 u w), (Hg2 u w).
    apply invr.
  Qed.

End PosInnerGeometry.
