From Coq Require Export Morphisms.
From Coq.Vectors Require Fin.
From Riktig Require Export Ops Interface.Monoid Interface.Ring
  Interface.Field Interface.VectorSpace Interface.Order Interface.Norm
  Interface.LinearMap Interface.Bilinear Theory.Order Theory.BigOp

Section MatMulBilinear.
  Context {F : Type} `{OrderedField F}.
  Context (m n p : nat).


  Definition mat_mul_bilinear :
      BilinearMap F (Matrix m n F) (Matrix n p F) (Matrix m p F) :=
    @Build_BilinearMap F (Matrix m n F) (Matrix n p F) (Matrix m p F) _ _ _ _
      (@mat_mul F _ m n p)
      (mat_mul_proper m n p)
      (mat_mul_add_l m n p)
      (mat_mul_scale_l m n p)
      (mat_mul_add_r m n p)
      (mat_mul_scale_r m n p).


  Lemma mat_mul_entry_bound : ∀ (A : Matrix m n F) (B : Matrix n p F)
      (i : Fin.t m) (k : Fin.t p),
    abs (mat_mul A B i k) ≤ bigopF (fun j ⇒ abs (A i j) × abs (B j k)).
  Proof.
    intros A B i k.
    calc (abs (mat_mul A B i k))
      == (bigopF (fun j ⇒ abs (A i j × B j k)))
      == (bigopF (fun j ⇒ abs (A i j) × abs (B j k))).
    - exact (abs_bigopF_le n (fun j ⇒ A i j × B j k)).
    - apply eq_le.
      apply bigopF_ext. intros j.
      exact (abs_mul (A i j) (B j k)).
  Qed.

  Lemma mat_mul_norm_bound_step1 : ∀ (A : Matrix m n F) (B : Matrix n p F),
    ‖ mat_mul A B ‖
      ≤ bigopF (fun i ⇒ bigopF (fun k ⇒
           bigopF (fun j ⇒ abs (A i j) × abs (B j k)))).
  Proof.
    intros A B.
    unfold vnorm, Matrix_HasNorm, mat_norm.
    apply bigopF_le. intros i.
    apply bigopF_le. intros k.
    exact (mat_mul_entry_bound A B i k).
  Qed.

  Lemma mat_mul_norm_bound_step2 : ∀ (A : Matrix m n F) (B : Matrix n p F),
    bigopF (fun i ⇒ bigopF (fun k ⇒
        bigopF (fun j ⇒ abs (A i j) × abs (B j k))))
      == bigopF (fun i ⇒ bigopF (fun j ⇒
           bigopF (fun k ⇒ abs (A i j) × abs (B j k)))).
  Proof.
    intros A B.
    apply bigopF_ext. intros i.
    exact (bigopF_swap p n (fun k j ⇒ abs (A i j) × abs (B j k))).
  Qed.

  Lemma mat_mul_norm_bound_step3 : ∀ (A : Matrix m n F) (B : Matrix n p F),
    bigopF (fun i ⇒ bigopF (fun j ⇒
        bigopF (fun k ⇒ abs (A i j) × abs (B j k))))
      == bigopF (fun i ⇒ bigopF (fun j ⇒
           abs (A i j) × bigopF (fun k ⇒ abs (B j k)))).
  Proof.
    intros A B.
    apply bigopF_ext. intros i.
    apply bigopF_ext. intros j.
    exact (bigopF_mul_l p (abs (A i j)) (fun k ⇒ abs (B j k))).
  Qed.

  Lemma mat_mul_norm_bound_step4 : ∀ (B : Matrix n p F) (j : Fin.t n),
    bigopF (fun k ⇒ abs (B j k)) ≤ ‖ B ‖.
  Proof.
    intros B j.
    unfold vnorm, Matrix_HasNorm, mat_norm.
    exact (bigopF_le_one_term n (fun j' ⇒ bigopF (fun k ⇒ abs (B j' k))) j
      (fun j' ⇒ bigopF_nonneg p (fun k ⇒ abs (B j' k))
                   (fun k ⇒ abs_nonneg (B j' k)))).
  Qed.

  Lemma mat_mul_norm_bound_step5 : ∀ (A : Matrix m n F) (B : Matrix n p F),
    bigopF (fun i ⇒ bigopF (fun j ⇒
        abs (A i j) × bigopF (fun k ⇒ abs (B j k))))
      ≤ bigopF (fun i ⇒ bigopF (fun j ⇒ abs (A i j) × ‖ B ‖)).
  Proof.
    intros A B.
    apply bigopF_le. intros i.
    apply bigopF_le. intros j.
    apply mul_le_mono_l.
    - apply abs_nonneg.
    - exact (mat_mul_norm_bound_step4 B j).
  Qed.

  Lemma mat_mul_norm_bound_step6 : ∀ (A : Matrix m n F) (B : Matrix n p F),
    bigopF (fun i ⇒ bigopF (fun j ⇒ abs (A i j) × ‖ B ‖)) == ‖ A ‖ × ‖ B ‖.
  Proof.
    intros A B.
    calc (bigopF (fun i ⇒ bigopF (fun j ⇒ abs (A i j) × ‖ B ‖)))
      == (bigopF (fun i ⇒ bigopF (fun j ⇒ abs (A i j)) × ‖ B ‖))
      == (‖ A ‖ × ‖ B ‖).
    - apply bigopF_ext. intros i.
      exact (bigopF_mul_r n (‖ B ‖) (fun j ⇒ abs (A i j))).
    - exact (bigopF_mul_r m (‖ B ‖) (fun i ⇒ bigopF (fun j ⇒ abs (A i j)))).
  Qed.

  Theorem mat_mul_norm_submultiplicative : ∀ (A : Matrix m n F)
      (B : Matrix n p F), ‖ mat_mul A B ‖ ≤ ‖ A ‖ × ‖ B ‖.
  Proof.
    intros A B.
    calc (‖ mat_mul A B ‖)
      == (bigopF (fun i ⇒ bigopF (fun k ⇒
            bigopF (fun j ⇒ abs (A i j) × abs (B j k)))))
         { (mat_mul_norm_bound_step1 A B) }
      == (bigopF (fun i ⇒ bigopF (fun j ⇒
            bigopF (fun k ⇒ abs (A i j) × abs (B j k)))))
         { (mat_mul_norm_bound_step2 A B) }
      == (bigopF (fun i ⇒ bigopF (fun j ⇒
            abs (A i j) × bigopF (fun k ⇒ abs (B j k)))))
         { (mat_mul_norm_bound_step3 A B) }
      == (bigopF (fun i ⇒ bigopF (fun j ⇒ abs (A i j) × ‖ B ‖)))
         { (mat_mul_norm_bound_step5 A B) }
      == (‖ A ‖ × ‖ B ‖)
         { (mat_mul_norm_bound_step6 A B) }.
  Qed.


  Theorem mat_mul_bm_bounded : bm_bounded mat_mul_bilinear.
  Proof.
    ∃ 1.
    split.
    - exact zero_le_one.
    - intros A B.
      rewrite ring_mul_1l.
      exact (mat_mul_norm_submultiplicative A B).
  Qed.


  Definition mat_mul_deriv_lm (A0 : Matrix m n F) (B0 : Matrix n p F) :
      LinearMap F (Matrix m n F × Matrix n p F) (Matrix m p F) :=
    bilinear_deriv_lm mat_mul_bilinear A0 B0.

  Theorem mat_mul_is_deriv : ∀ (A0 : Matrix m n F) (B0 : Matrix n p F),
    is_deriv (fun AB : Matrix m n F × Matrix n p F ⇒ mat_mul (fst AB) (snd AB))
      (A0, B0) (mat_mul_deriv_lm A0 B0).
  Proof.
    intros A0 B0.
    exact (bilinear_is_deriv mat_mul_bilinear A0 B0 mat_mul_bm_bounded).
  Qed.

End MatMulBilinear.
