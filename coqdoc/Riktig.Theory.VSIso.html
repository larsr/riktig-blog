Record VSIso (F U V : Type) {FF : Field F}
    {VU : VectorSpace F U} {VV : VectorSpace F V} := {
  vsi_to : LinearMap F U V;
  vsi_from : LinearMap F V U;
  vsi_fromto : ∀ u : U, vsi_from (vsi_to u) == u;
  vsi_tofrom : ∀ v : V, vsi_to (vsi_from v) == v
}.

Arguments vsi_to {F U V FF VU VV}.
Arguments vsi_from {F U V FF VU VV}.
Arguments vsi_fromto {F U V FF VU VV}.
Arguments vsi_tofrom {F U V FF VU VV}.


Section Unitor.
  Context {F A : Type} {FF : Field F} {VA : VectorSpace F A}.

  Let VAU : VectorSpace F (A × unit) :=
    @Prod_VectorSpace F A unit FF VA (@Unit_VectorSpace F FF).

  #[global] Instance unitor_to_proper : Proper (equ ==> equ) (fun p : A × unit ⇒ fst p).
  Proof.
    intros p q [H1 _].
    exact H1.
  Qed.

  Lemma unitor_to_add : ∀ p q : A × unit, fst (p + q) == fst p + fst q.
  Proof.
    intros p q.
    reflexivity.
  Qed.

  Lemma unitor_to_scale : ∀ (a : F) (p : A × unit),
    fst (a *: p) == a *: fst p.
  Proof.
    intros a p.
    reflexivity.
  Qed.

  Definition unitor_to : LinearMap F (A × unit) A :=
    @Build_LinearMap F (A × unit) A FF VAU VA (fun p : A × unit ⇒ fst p)
      unitor_to_proper unitor_to_add unitor_to_scale.

  #[global] Instance unitor_from_proper : Proper (equ ==> equ) (fun a : A ⇒ (a, tt)).
  Proof.
    intros a b Hab.
    split.
    - exact Hab.
    - exact I.
  Qed.

  Lemma unitor_from_add : ∀ a b : A, (a + b, tt) == (a, tt) + (b, tt).
  Proof.
    intros a b.
    split.
    - reflexivity.
    - exact I.
  Qed.

  Lemma unitor_from_scale : ∀ (a : F) (u : A),
    (a *: u, tt) == a *: (u, tt).
  Proof.
    intros a u.
    split.
    - reflexivity.
    - exact I.
  Qed.

  Definition unitor_from : LinearMap F A (A × unit) :=
    @Build_LinearMap F A (A × unit) FF VA VAU (fun a : A ⇒ (a, tt))
      unitor_from_proper unitor_from_add unitor_from_scale.

  Lemma unitor_fromto : ∀ p : A × unit, (fst p, tt) == p.
  Proof.
    intros p.
    split.
    - reflexivity.
    - exact I.
  Qed.

  Lemma unitor_tofrom : ∀ a : A, fst (a, tt) == a.
  Proof.
    intros a.
    reflexivity.
  Qed.

  Definition vsiso_unitor : @VSIso F (A × unit) A FF VAU VA :=
    @Build_VSIso F (A × unit) A FF VAU VA
      unitor_to unitor_from unitor_fromto unitor_tofrom.

End Unitor.


Section Swap.
  Context {F A B : Type} {FF : Field F}
          {VA : VectorSpace F A} {VB : VectorSpace F B}.

  Let VAB : VectorSpace F (A × B) := @Prod_VectorSpace F A B FF VA VB.
  Let VBA : VectorSpace F (B × A) := @Prod_VectorSpace F B A FF VB VA.

  #[global] Instance swap_proper : Proper (equ ==> equ) (fun p : A × B ⇒ (snd p, fst p)).
  Proof.
    intros p q [H1 H2].
    split.
    - exact H2.
    - exact H1.
  Qed.

  Lemma swap_add : ∀ p q : A × B,
    (snd (p + q), fst (p + q)) == (snd p, fst p) + (snd q, fst q).
  Proof.
    intros p q.
    split.
    - reflexivity.
    - reflexivity.
  Qed.

  Lemma swap_scale : ∀ (a : F) (p : A × B),
    (snd (a *: p), fst (a *: p)) == a *: (snd p, fst p).
  Proof.
    intros a p.
    split.
    - reflexivity.
    - reflexivity.
  Qed.

  Definition swap_lm : LinearMap F (A × B) (B × A) :=
    @Build_LinearMap F (A × B) (B × A) FF VAB VBA
      (fun p : A × B ⇒ (snd p, fst p))
      swap_proper swap_add swap_scale.

  Lemma prod_eta : ∀ p : A × B, (fst p, snd p) == p.
  Proof.
    intros p.
    split.
    - reflexivity.
    - reflexivity.
  Qed.

End Swap.

Section Braiding.
  Context {F A B : Type} {FF : Field F}
          {VA : VectorSpace F A} {VB : VectorSpace F B}.

  Let VAB : VectorSpace F (A × B) := @Prod_VectorSpace F A B FF VA VB.
  Let VBA : VectorSpace F (B × A) := @Prod_VectorSpace F B A FF VB VA.

  Definition vsiso_braiding : @VSIso F (A × B) (B × A) FF VAB VBA :=
    @Build_VSIso F (A × B) (B × A) FF VAB VBA
      (@swap_lm F A B FF VA VB) (@swap_lm F B A FF VB VA)
      (@prod_eta F A B FF VA VB) (@prod_eta F B A FF VB VA).

End Braiding.


Section Associator.
  Context {F A B C : Type} {FF : Field F}
          {VA : VectorSpace F A} {VB : VectorSpace F B}
          {VC : VectorSpace F C}.

  Let VAB : VectorSpace F (A × B) := @Prod_VectorSpace F A B FF VA VB.
  Let VBC : VectorSpace F (B × C) := @Prod_VectorSpace F B C FF VB VC.
  Let VAB_C : VectorSpace F ((A × B) × C) :=
    @Prod_VectorSpace F (A × B) C FF VAB VC.
  Let VA_BC : VectorSpace F (A × (B × C)) :=
    @Prod_VectorSpace F A (B × C) FF VA VBC.

  #[global] Instance assoc_to_proper : Proper (equ ==> equ)
    (fun p : (A × B) × C ⇒ (fst (fst p), (snd (fst p), snd p))).
  Proof.
    intros p q [[H1 H2] H3].
    split.
    - exact H1.
    - split.
      + exact H2.
      + exact H3.
  Qed.

  Lemma assoc_to_add : ∀ p q : (A × B) × C,
    (fst (fst (p + q)), (snd (fst (p + q)), snd (p + q)))
      == (fst (fst p), (snd (fst p), snd p))
         + (fst (fst q), (snd (fst q), snd q)).
  Proof.
    intros p q.
    split.
    - reflexivity.
    - split.
      + reflexivity.
      + reflexivity.
  Qed.

  Lemma assoc_to_scale : ∀ (a : F) (p : (A × B) × C),
    (fst (fst (a *: p)), (snd (fst (a *: p)), snd (a *: p)))
      == a *: (fst (fst p), (snd (fst p), snd p)).
  Proof.
    intros a p.
    split.
    - reflexivity.
    - split.
      + reflexivity.
      + reflexivity.
  Qed.

  Definition assoc_to : LinearMap F ((A × B) × C) (A × (B × C)) :=
    @Build_LinearMap F ((A × B) × C) (A × (B × C)) FF VAB_C VA_BC
      (fun p : (A × B) × C ⇒ (fst (fst p), (snd (fst p), snd p)))
      assoc_to_proper assoc_to_add assoc_to_scale.

  #[global] Instance assoc_from_proper : Proper (equ ==> equ)
    (fun q : A × (B × C) ⇒ ((fst q, fst (snd q)), snd (snd q))).
  Proof.
    intros p q [H1 [H2 H3]].
    split.
    - split.
      + exact H1.
      + exact H2.
    - exact H3.
  Qed.

  Lemma assoc_from_add : ∀ p q : A × (B × C),
    ((fst (p + q), fst (snd (p + q))), snd (snd (p + q)))
      == ((fst p, fst (snd p)), snd (snd p))
         + ((fst q, fst (snd q)), snd (snd q)).
  Proof.
    intros p q.
    split.
    - split.
      + reflexivity.
      + reflexivity.
    - reflexivity.
  Qed.

  Lemma assoc_from_scale : ∀ (a : F) (q : A × (B × C)),
    ((fst (a *: q), fst (snd (a *: q))), snd (snd (a *: q)))
      == a *: ((fst q, fst (snd q)), snd (snd q)).
  Proof.
    intros a q.
    split.
    - split.
      + reflexivity.
      + reflexivity.
    - reflexivity.
  Qed.

  Definition assoc_from : LinearMap F (A × (B × C)) ((A × B) × C) :=
    @Build_LinearMap F (A × (B × C)) ((A × B) × C) FF VA_BC VAB_C
      (fun q : A × (B × C) ⇒ ((fst q, fst (snd q)), snd (snd q)))
      assoc_from_proper assoc_from_add assoc_from_scale.

  Lemma assoc_fromto : ∀ p : (A × B) × C,
    ((fst (fst p), snd (fst p)), snd p) == p.
  Proof.
    intros p.
    split.
    - split.
      + reflexivity.
      + reflexivity.
    - reflexivity.
  Qed.

  Lemma assoc_tofrom : ∀ q : A × (B × C),
    (fst q, (fst (snd q), snd (snd q))) == q.
  Proof.
    intros q.
    split.
    - reflexivity.
    - split.
      + reflexivity.
      + reflexivity.
  Qed.

  Definition vsiso_associator :
    @VSIso F ((A × B) × C) (A × (B × C)) FF VAB_C VA_BC :=
    @Build_VSIso F ((A × B) × C) (A × (B × C)) FF VAB_C VA_BC
      assoc_to assoc_from assoc_fromto assoc_tofrom.

End Associator.


Section SoA.
  Context {F X A B : Type} {FF : Field F}
          {VA : VectorSpace F A} {VB : VectorSpace F B}.

  Let VAB : VectorSpace F (A × B) := @Prod_VectorSpace F A B FF VA VB.
  Let VXAB : VectorSpace F (X → A × B) :=
    @PW_VectorSpace X F (A × B) FF VAB.
  Let VXA : VectorSpace F (X → A) := @PW_VectorSpace X F A FF VA.
  Let VXB : VectorSpace F (X → B) := @PW_VectorSpace X F B FF VB.
  Let VCols : VectorSpace F ((X → A) × (X → B)) :=
    @Prod_VectorSpace F (X → A) (X → B) FF VXA VXB.

  #[global] Instance soa_split_proper : Proper (equ ==> equ)
    (fun f : X → A × B ⇒ (fun x : X ⇒ fst (f x), fun x : X ⇒ snd (f x))).
  Proof.
    intros f g Hfg.
    split.
    - intros x.
      destruct (Hfg x) as [H1 _].
      exact H1.
    - intros x.
      destruct (Hfg x) as [_ H2].
      exact H2.
  Qed.

  Lemma soa_split_add : ∀ f g : X → A × B,
    (fun x : X ⇒ fst ((f + g) x), fun x : X ⇒ snd ((f + g) x))
      == (fun x : X ⇒ fst (f x), fun x : X ⇒ snd (f x))
         + (fun x : X ⇒ fst (g x), fun x : X ⇒ snd (g x)).
  Proof.
    intros f g.
    split.
    - intros x.
      reflexivity.
    - intros x.
      reflexivity.
  Qed.

  Lemma soa_split_scale : ∀ (a : F) (f : X → A × B),
    (fun x : X ⇒ fst ((a *: f) x), fun x : X ⇒ snd ((a *: f) x))
      == a *: (fun x : X ⇒ fst (f x), fun x : X ⇒ snd (f x)).
  Proof.
    intros a f.
    split.
    - intros x.
      reflexivity.
    - intros x.
      reflexivity.
  Qed.

  Definition soa_split : LinearMap F (X → A × B) ((X → A) × (X → B)) :=
    @Build_LinearMap F (X → A × B) ((X → A) × (X → B)) FF VXAB VCols
      (fun f : X → A × B ⇒ (fun x : X ⇒ fst (f x), fun x : X ⇒ snd (f x)))
      soa_split_proper soa_split_add soa_split_scale.

  #[global] Instance soa_merge_proper : Proper (equ ==> equ)
    (fun (p : (X → A) × (X → B)) (x : X) ⇒ (fst p x, snd p x)).
  Proof.
    intros p q [H1 H2] x.
    split.
    - exact (H1 x).
    - exact (H2 x).
  Qed.

  Lemma soa_merge_add : ∀ p q : (X → A) × (X → B),
    (fun x : X ⇒ (fst (p + q) x, snd (p + q) x))
      == (fun x : X ⇒ (fst p x, snd p x)) + (fun x : X ⇒ (fst q x, snd q x)).
  Proof.
    intros p q x.
    split.
    - reflexivity.
    - reflexivity.
  Qed.

  Lemma soa_merge_scale : ∀ (a : F) (p : (X → A) × (X → B)),
    (fun x : X ⇒ (fst (a *: p) x, snd (a *: p) x))
      == a *: (fun x : X ⇒ (fst p x, snd p x)).
  Proof.
    intros a p x.
    split.
    - reflexivity.
    - reflexivity.
  Qed.

  Definition soa_merge : LinearMap F ((X → A) × (X → B)) (X → A × B) :=
    @Build_LinearMap F ((X → A) × (X → B)) (X → A × B) FF VCols VXAB
      (fun (p : (X → A) × (X → B)) (x : X) ⇒ (fst p x, snd p x))
      soa_merge_proper soa_merge_add soa_merge_scale.

  Lemma soa_fromto : ∀ f : X → A × B,
    (fun x : X ⇒ (fst (f x), snd (f x))) == f.
  Proof.
    intros f x.
    split.
    - reflexivity.
    - reflexivity.
  Qed.

  Lemma soa_tofrom : ∀ p : (X → A) × (X → B),
    ((fun x : X ⇒ fst p x), (fun x : X ⇒ snd p x)) == p.
  Proof.
    intros p.
    split.
    - intros x.
      reflexivity.
    - intros x.
      reflexivity.
  Qed.

  Definition vsiso_soa :
    @VSIso F (X → A × B) ((X → A) × (X → B)) FF VXAB VCols :=
    @Build_VSIso F (X → A × B) ((X → A) × (X → B)) FF VXAB VCols
      soa_split soa_merge soa_fromto soa_tofrom.

End SoA.


Section Currying.
  Context {F U V W : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}
          {VW : VectorSpace F W}.

  Local Notation HomVW := (LinearMap F V W).
  Local Notation Bilin := (BilinearMap F U V W).

  Let HVW : VectorSpace F HomVW := @Hom_VectorSpace F V W FF VV VW.
  Let HomUVW : VectorSpace F (LinearMap F U HomVW) :=
    @Hom_VectorSpace F U HomVW FF VU HVW.
  Let VBilin : VectorSpace F Bilin :=
    @Bilin_VectorSpace F U V W FF VU VV VW.

  #[global] Instance curry_lm_proper : Proper (equ ==> equ) (@curry F U V W FF VU VV VW).
  Proof.
    intros b b' Hbb' u v.
    exact (Hbb' u v).
  Qed.

  Lemma curry_lm_add : ∀ b b' : Bilin, curry (b + b') == curry b + curry b'.
  Proof.
    intros b b' u v.
    reflexivity.
  Qed.

  Lemma curry_lm_scale : ∀ (a : F) (b : Bilin), curry (a *: b) == a *: curry b.
  Proof.
    intros a b u v.
    reflexivity.
  Qed.

  Definition curry_lm : LinearMap F Bilin (LinearMap F U HomVW) :=
    @Build_LinearMap F Bilin (LinearMap F U HomVW) FF VBilin HomUVW
      curry curry_lm_proper curry_lm_add curry_lm_scale.

  #[global] Instance uncurry_lm_proper : Proper (equ ==> equ) (@uncurry F U V W FF VU VV VW).
  Proof.
    intros g g' Hgg' u v.
    exact (Hgg' u v).
  Qed.

  Lemma uncurry_lm_add : ∀ g g' : LinearMap F U HomVW,
    uncurry (g + g') == uncurry g + uncurry g'.
  Proof.
    intros g g' u v.
    reflexivity.
  Qed.

  Lemma uncurry_lm_scale : ∀ (a : F) (g : LinearMap F U HomVW),
    uncurry (a *: g) == a *: uncurry g.
  Proof.
    intros a g u v.
    reflexivity.
  Qed.

  Definition uncurry_lm : LinearMap F (LinearMap F U HomVW) Bilin :=
    @Build_LinearMap F (LinearMap F U HomVW) Bilin FF HomUVW VBilin
      uncurry uncurry_lm_proper uncurry_lm_add uncurry_lm_scale.

  Definition vsiso_curry : @VSIso F Bilin (LinearMap F U HomVW) FF VBilin HomUVW :=
    @Build_VSIso F Bilin (LinearMap F U HomVW) FF VBilin HomUVW
      curry_lm uncurry_lm uncurry_curry curry_uncurry.

End Currying.

Example vsiso_curry_check (F U V W : Type) (FF : Field F)
  (VU : VectorSpace F U) (VV : VectorSpace F V) (VW : VectorSpace F W) :=
  @vsiso_curry F U V W FF VU VV VW.


Section Sym.
  Context {F U V : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}.

  Definition vsiso_sym (i : VSIso F U V) : VSIso F V U :=
    @Build_VSIso F V U FF VV VU (vsi_from i) (vsi_to i) (vsi_tofrom i) (vsi_fromto i).

End Sym.

Section Trans.
  Context {F U V W : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V} {VW : VectorSpace F W}.

  Lemma vsiso_trans_fromto : ∀ (i1 : VSIso F U V) (i2 : VSIso F V W) (u : U),
    vsi_from i1 (vsi_from i2 (vsi_to i2 (vsi_to i1 u))) == u.
  Proof.
    intros i1 i2 u.
    calc (vsi_from i1 (vsi_from i2 (vsi_to i2 (vsi_to i1 u))))
      == (vsi_from i1 (vsi_to i1 u))
           { (lm_proper (vsi_from i1) _ _ (vsi_fromto i2 (vsi_to i1 u))) }
      == u { (vsi_fromto i1 u) }.
  Qed.

  Lemma vsiso_trans_tofrom : ∀ (i1 : VSIso F U V) (i2 : VSIso F V W) (w : W),
    vsi_to i2 (vsi_to i1 (vsi_from i1 (vsi_from i2 w))) == w.
  Proof.
    intros i1 i2 w.
    calc (vsi_to i2 (vsi_to i1 (vsi_from i1 (vsi_from i2 w))))
      == (vsi_to i2 (vsi_from i2 w))
           { (lm_proper (vsi_to i2) _ _ (vsi_tofrom i1 (vsi_from i2 w))) }
      == w { (vsi_tofrom i2 w) }.
  Qed.

  Definition vsiso_trans (i1 : VSIso F U V) (i2 : VSIso F V W) : VSIso F U W :=
    @Build_VSIso F U W FF VU VW
      (lm_comp (vsi_to i2) (vsi_to i1)) (lm_comp (vsi_from i1) (vsi_from i2))
      (vsiso_trans_fromto i1 i2) (vsiso_trans_tofrom i1 i2).

End Trans.

Example vsiso_chain_check (F A B : Type) (FF : Field F)
    (VA : VectorSpace F A) (VB : VectorSpace F B) :
  @VSIso F (A × (B × unit)) (B × A) FF
    (@Prod_VectorSpace F A (B × unit) FF VA
       (@Prod_VectorSpace F B unit FF VB (@Unit_VectorSpace F FF)))
    (@Prod_VectorSpace F B A FF VB VA) :=
  vsiso_trans
    (vsiso_trans (vsiso_sym vsiso_associator) vsiso_unitor)
    vsiso_braiding.

Example vsiso_soa_check (F X A B : Type) (FF : Field F)
  (VA : VectorSpace F A) (VB : VectorSpace F B) :=
  @vsiso_soa F X A B FF VA VB.

Section VSIsoCategorical.
  Context {F : Type} {FF : Field F}.


  Definition vsiso_to_iso {U V : Type} {VU : VectorSpace F U} {VV : VectorSpace F V}
      (i : VSIso F U V) :
      Iso (Build_VectObj F FF U VU) (Build_VectObj F FF V VV) :=
    @Build_Iso (VectObj F) (@Vect_Category F FF)
      (Build_VectObj F FF U VU) (Build_VectObj F FF V VV)
      (vsi_to i) (vsi_from i) (vsi_fromto i) (vsi_tofrom i).

  Definition iso_to_vsiso {U V : Type} {VU : VectorSpace F U} {VV : VectorSpace F V}
      (i : Iso (Build_VectObj F FF U VU) (Build_VectObj F FF V VV)) :
      VSIso F U V :=
    {| vsi_to := iso_to i;
       vsi_from := iso_from i;
       vsi_fromto := iso_fromto i;
       vsi_tofrom := iso_tofrom i |}.

  Definition VS_Isomorphic (X Y : VectObj F) : Prop := X ≅ Y.

  #[global] Instance VS_Isomorphic_Equivalence : Equivalence VS_Isomorphic :=
    Isomorphic_Equivalence.

  Corollary vsiso_isomorphic {U V : Type} {VU : VectorSpace F U} {VV : VectorSpace F V}
      (i : VSIso F U V) :
    Build_VectObj F FF U VU ≅ Build_VectObj F FF V VV.
  Proof.
    exact (inhabits (vsiso_to_iso i)).
  Qed.

End VSIsoCategorical.

Example vsiso_curry_isomorphic (F U V W : Type) (FF : Field F)
    (VU : VectorSpace F U) (VV : VectorSpace F V) (VW : VectorSpace F W) :
  Build_VectObj F FF (BilinearMap F U V W) (@Bilin_VectorSpace F U V W FF VU VV VW)
    ≅ Build_VectObj F FF (LinearMap F U (LinearMap F V W))
        (@Hom_VectorSpace F U (LinearMap F V W) FF VU (@Hom_VectorSpace F V W FF VV VW)) :=
  vsiso_isomorphic (@vsiso_curry F U V W FF VU VV VW).
