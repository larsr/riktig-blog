Section AdjointKit.
  Context {F V : Type} `{InnerProduct F V}.

  Definition is_adjoint (f g : LinearMap F V V) : Prop :=
    ∀ u w : V, ⟨ f u , w ⟩ == ⟨ u , g w ⟩.

  Definition self_adjoint (f : LinearMap F V V) : Prop := is_adjoint f f.

  Definition is_isometry (f : LinearMap F V V) : Prop :=
    ∀ u : V, norm2 (f u) == norm2 u.

  Lemma is_adjoint_eq : ∀ f g : LinearMap F V V,
    is_adjoint f g → ∀ u w : V, ⟨ f u , w ⟩ == ⟨ u , g w ⟩.
  Proof.
    intros f g Hfg u w.
    exact (Hfg u w).
  Qed.

  Lemma is_adjoint_def : ∀ f g : LinearMap F V V,
    (∀ u w : V, ⟨ f u , w ⟩ == ⟨ u , g w ⟩) → is_adjoint f g.
  Proof.
    intros f g Hfg u w.
    exact (Hfg u w).
  Qed.

  Theorem is_adjoint_sym : ∀ f g : LinearMap F V V,
    is_adjoint f g → is_adjoint g f.
  Proof.
    intros f g Hfg u w.
    calc (⟨ g u , w ⟩)
      == (⟨ w , g u ⟩) { (ip_sym (g u) w) }
      == (⟨ f w , u ⟩) { (Hfg w u) }
      == (⟨ u , f w ⟩) { ip_sym }.
  Qed.

  Theorem is_adjoint_id : is_adjoint lm_id lm_id.
  Proof.
    intros u w.
    reflexivity.
  Qed.

  Theorem self_adjoint_id : self_adjoint lm_id.
  Proof.
    exact is_adjoint_id.
  Qed.

  Theorem is_adjoint_comp : ∀ f1 g1 f2 g2 : LinearMap F V V,
    is_adjoint f1 g1 → is_adjoint f2 g2 →
    is_adjoint (lm_comp f2 f1) (lm_comp g1 g2).
  Proof.
    intros f1 g1 f2 g2 Hfg1 Hfg2 u w.
    calc (⟨ f2 (f1 u) , w ⟩)
      == (⟨ f1 u , g2 w ⟩) { (Hfg2 (f1 u) w) }
      == (⟨ u , g1 (g2 w) ⟩) { (Hfg1 u (g2 w)) }.
  Qed.

  Theorem adjoint_inverse_preserves_inner : ∀ f f' : LinearMap F V V,
    is_adjoint f f' → (∀ u : V, f' (f u) == u) →
    ∀ u w : V, ⟨ f u , f w ⟩ == ⟨ u , w ⟩.
  Proof.
    intros f f' Hadj Hinv u w.
    calc (⟨ f u , f w ⟩)
      == (⟨ u , f' (f w) ⟩) { (Hadj u (f w)) }
      == (⟨ u , w ⟩) { (Hinv w) }.
  Qed.

End AdjointKit.


Section AdjointUniqueness.
  Context {F V : Type} `{PosInner F V}.

  Theorem is_adjoint_uniqueness : ∀ f g1 g2 : LinearMap F V V,
    is_adjoint g1 f → is_adjoint g2 f → ∀ u : V, g1 u == g2 u.
  Proof.
    intros f g1 g2 Hg1 Hg2 u.
    exact (adjoint_unique f g1 g2 Hg1 Hg2 u).
  Qed.

  Lemma double_inj : ∀ a b : F, a + a == b + b → a == b.
  Proof.
    intros a b Hab.
    apply (mul_cancel_l two a b two_neq_0).
    rewrite (field_mul_comm two a), (field_mul_comm two b),
      (mul_two a), (mul_two b).
    exact Hab.
  Qed.

  Theorem isometry_preserves_inner : ∀ f : LinearMap F V V,
    is_isometry f → ∀ u v : V, ⟨ f u , f v ⟩ == ⟨ u , v ⟩.
  Proof.
    intros f Hiso u v.
    assert (Hfuv : norm2 (f u + f v) == norm2 (u + v)).
    { rewrite <- (lm_add f u v). exact (Hiso (u + v)). }
    rewrite (norm2_add (f u) (f v)), (norm2_add u v),
      (Hiso u), (Hiso v) in Hfuv.
    apply (cancel_l (norm2 u + norm2 v)) in Hfuv.
    rewrite (ip_sym (f v) (f u)), (ip_sym v u) in Hfuv.
    exact (double_inj (⟨ f u , f v ⟩) (⟨ u , v ⟩) Hfuv).
  Qed.

End AdjointUniqueness.
