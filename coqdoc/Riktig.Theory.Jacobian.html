Section Jacobian.
  Context {F U V : Type} `{NormedSpace F U}.
  Context {VSV : VectorSpace F V} {NSV : NormedSpace F V}.
  Context (n m : nat) (BU : Basis n F U) (BV : Basis m F V).

  Definition jacobian {f : U → V} {a : U} {L : LinearMap F U V}
      (Hd : is_deriv f a L) : Matrix m n F :=
    mat_of BU BV L.

  Theorem jacobian_col : ∀ {f : U → V} {a : U} {L : LinearMap F U V}
      (Hd : is_deriv f a L) (i : Fin.t m) (j : Fin.t n),
    jacobian Hd i j == basis_coords BV (partial_at L BU j) i.
  Proof.
    intros f a L Hd i j.
    reflexivity.
  Qed.

  Example jacobian_of_linear : ∀ (g : LinearMap F U V)
      (Hb : lm_bounded g) (a : U),
    jacobian (deriv_lm g a Hb) == mat_of BU BV g.
  Proof.
    intros g Hb a.
    reflexivity.
  Qed.

End Jacobian.


Section JacobianComp.
  Context {F U V W : Type} `{NormedSpace F U}.
  Context {VSV : VectorSpace F V} {NSV : NormedSpace F V}.
  Context {VSW : VectorSpace F W} {NSW : NormedSpace F W}.
  Context (n m p : nat) (BU : Basis n F U) (BV : Basis m F V)
    (BW : Basis p F W).

  Theorem jacobian_comp : ∀ (f : U → V) (g : V → W) (a : U)
      (Lf : LinearMap F U V) (Lg : LinearMap F V W),
    is_deriv f a Lf → is_deriv g (f a) Lg →
    is_deriv (fun u : U ⇒ g (f u)) a (lm_comp Lg Lf) →
    mat_of BU BW (lm_comp Lg Lf)
      == mat_mul (mat_of BV BW Lg) (mat_of BU BV Lf).
  Proof.
    intros f g a Lf Lg Hf Hg Hgf.
    exact (mat_of_comp BU BV BW Lg Lf).
  Qed.

  Theorem jacobian_chain_rule : ∀ (f : U → V) (g : V → W) (a : U)
      (Lf : LinearMap F U V) (Lg : LinearMap F V W)
      (Pg : Proper (equ ==> equ) g)
      (Hf : is_deriv f a Lf) (Hg : is_deriv g (f a) Lg),
    jacobian n p BU BW (deriv_comp f g a Lf Lg Pg Hf Hg)
      == mat_mul (jacobian m p BV BW Hg) (jacobian n m BU BV Hf).
  Proof.
    intros f g a Lf Lg Pg Hf Hg.
    exact (mat_of_comp BU BV BW Lg Lf).
  Qed.

End JacobianComp.

From Coq Require Import QArith.
From Riktig Require Import Theory.MatVec Impl.Qf.

Definition rowQ2 (a b : Q) : Fin.t 2 → Q :=
  fun j ⇒ match j with Fin.F1 ⇒ a | _ ⇒ b end.

Definition matQ2 (a11 a12 a21 a22 : Q) : Matrix 2 2 Q :=
  fun i ⇒ match i with
           | Fin.F1 ⇒ rowQ2 a11 a12
           | _ ⇒ rowQ2 a21 a22
           end.

Definition Af : Matrix 2 2 Q := matQ2 (2#1) (3#1) (5#1) (7#1).
Definition Bg : Matrix 2 2 Q := matQ2 (1#1) (1#1) (0#1) (1#1).

Example partial_x_of_f :
     partial_at (mat_vec_lm Af) (FinVec_Basis 2) Fin.F1 Fin.F1 == (2#1)%Q
  ∧ partial_at (mat_vec_lm Af) (FinVec_Basis 2) Fin.F1 (Fin.FS Fin.F1)
       == (5#1)%Q.
Proof. split; vm_compute; reflexivity. Qed.

Example partial_y_of_f :
     partial_at (mat_vec_lm Af) (FinVec_Basis 2) (Fin.FS Fin.F1) Fin.F1
       == (3#1)%Q
  ∧ partial_at (mat_vec_lm Af) (FinVec_Basis 2) (Fin.FS Fin.F1)
       (Fin.FS Fin.F1) == (7#1)%Q.
Proof. split; vm_compute; reflexivity. Qed.

Example jacobian_matrix_of_f :
     mat_of (FinVec_Basis 2) (FinVec_Basis 2) (mat_vec_lm Af) Fin.F1 Fin.F1
       == (2#1)%Q
  ∧ mat_of (FinVec_Basis 2) (FinVec_Basis 2) (mat_vec_lm Af) Fin.F1
       (Fin.FS Fin.F1) == (3#1)%Q
  ∧ mat_of (FinVec_Basis 2) (FinVec_Basis 2) (mat_vec_lm Af)
       (Fin.FS Fin.F1) Fin.F1 == (5#1)%Q
  ∧ mat_of (FinVec_Basis 2) (FinVec_Basis 2) (mat_vec_lm Af)
       (Fin.FS Fin.F1) (Fin.FS Fin.F1) == (7#1)%Q.
Proof. repeat split; vm_compute; reflexivity. Qed.

Example chain_rule_concrete :
  mat_of (FinVec_Basis 2) (FinVec_Basis 2)
      (lm_comp (mat_vec_lm Bg) (mat_vec_lm Af))
    == mat_mul (mat_of (FinVec_Basis 2) (FinVec_Basis 2) (mat_vec_lm Bg))
         (mat_of (FinVec_Basis 2) (FinVec_Basis 2) (mat_vec_lm Af)).
Proof.
  exact (mat_of_comp (FinVec_Basis 2) (FinVec_Basis 2) (FinVec_Basis 2)
           (mat_vec_lm Bg) (mat_vec_lm Af)).
Qed.

Example partial_x_of_g_after_f :
     partial_at (lm_comp (mat_vec_lm Bg) (mat_vec_lm Af)) (FinVec_Basis 2)
       Fin.F1 Fin.F1 == (7#1)%Q
  ∧ partial_at (lm_comp (mat_vec_lm Bg) (mat_vec_lm Af)) (FinVec_Basis 2)
       Fin.F1 (Fin.FS Fin.F1) == (5#1)%Q
  ∧ mat_mul Bg Af Fin.F1 Fin.F1 == (7#1)%Q
  ∧ mat_mul Bg Af (Fin.FS Fin.F1) Fin.F1 == (5#1)%Q.
Proof. repeat split; vm_compute; reflexivity. Qed.
