Library Riktig.Impl.VectCat

From Coq Require Export Morphisms.
From Riktig Require Export Ops Interface.Field Interface.VectorSpace
  Interface.Category Interface.LinearMap.

Record VectObj (F : Type) {FF : Field F} := {
  vo_car : Type;
  vo_vs : VectorSpace F vo_car
}.

Arguments vo_car {F FF}.
Arguments vo_vs {F FF}.

#[global] Existing Instance vo_vs.

Section VectCat.
  Context {F : Type} {FF : Field F}.


  Definition VHom (X Y : VectObj F) : Type :=
    LinearMap F (vo_car X) (vo_car Y).

  Definition vhom_equ (X Y : VectObj F) : EqRel (VHom X Y) :=
    @lm_equ F (vo_car X) (vo_car Y) FF (vo_vs X) (vo_vs Y).

  Lemma vhom_equ_refl (X Y : VectObj F) (f : VHom X Y) : vhom_equ X Y f f.
  Proof.
    intros u.
    reflexivity.
  Qed.

  Lemma vhom_equ_sym (X Y : VectObj F) (f g : VHom X Y) :
    vhom_equ X Y f g → vhom_equ X Y g f.
  Proof.
    intros Hfg u.
    symmetry.
    exact (Hfg u).
  Qed.

  Lemma vhom_equ_trans (X Y : VectObj F) (f g h : VHom X Y) :
    vhom_equ X Y f g → vhom_equ X Y g h → vhom_equ X Y f h.
  Proof.
    intros Hfg Hgh u.
    rewrite (Hfg u).
    exact (Hgh u).
  Qed.

  #[global] Instance vhom_equiv (X Y : VectObj F) :
    Equivalence (vhom_equ X Y).
  Proof.
    apply Build_Equivalence.
    - exact (vhom_equ_refl X Y).
    - exact (vhom_equ_sym X Y).
    - exact (vhom_equ_trans X Y).
  Qed.


  Definition vid (X : VectObj F) : VHom X X := lm_id.

  Definition vcomp (X Y Z : VectObj F)
    (g : VHom Y Z) (f : VHom X Y) : VHom X Z := lm_comp g f.

  Lemma vcomp_ext (X Y Z : VectObj F)
      (g g' : VHom Y Z) (f f' : VHom X Y) :
    vhom_equ Y Z g g' → vhom_equ X Y f f' →
    ∀ u : vo_car X, g (f u) == g' (f' u).
  Proof.
    intros Hg Hf u.
    rewrite (lm_proper g _ _ (Hf u)).
    exact (Hg (f' u)).
  Qed.

  #[global] Instance vcomp_proper (X Y Z : VectObj F) :
    Proper (vhom_equ Y Z ==> vhom_equ X Y ==> vhom_equ X Z)
           (vcomp X Y Z).
  Proof.
    intros g g' Hg f f' Hf u.
    exact (vcomp_ext X Y Z g g' f f' Hg Hf u).
  Qed.


  Lemma vcat_idl (X Y : VectObj F) (f : VHom X Y) :
    vhom_equ X Y (vcomp X Y Y (vid Y) f) f.
  Proof.
    intros u.
    reflexivity.
  Qed.

  Lemma vcat_idr (X Y : VectObj F) (f : VHom X Y) :
    vhom_equ X Y (vcomp X X Y f (vid X)) f.
  Proof.
    intros u.
    reflexivity.
  Qed.

  Lemma vcat_assoc (W X Y Z : VectObj F)
      (f : VHom Y Z) (g : VHom X Y) (h : VHom W X) :
    vhom_equ W Z (vcomp W Y Z f (vcomp W X Y g h))
                 (vcomp W X Z (vcomp X Y Z f g) h).
  Proof.
    intros u.
    reflexivity.
  Qed.


  #[global] Instance Vect_Category : Category (VectObj F) := {|
    hom := VHom;
    hom_equ := vhom_equ;
    hom_equiv := vhom_equiv;
    cat_id := vid;
    comp := vcomp;
    comp_proper := vcomp_proper;
    cat_idl := vcat_idl;
    cat_idr := vcat_idr;
    cat_assoc := vcat_assoc
  |}.

End VectCat.

Example vect_cat_check (F : Type) (FF : Field F) : Category (VectObj F) := _.
