Definition Sub (V : Type) (P : V → Prop) := { v : V | P v }.


#[global] Instance Sub_EqRel {V : Type} `{EqRel V} (P : V → Prop) :
  EqRel (Sub V P) :=
  fun x y ⇒ proj1_sig x == proj1_sig y.

#[global] Instance Sub_Zero {F V : Type} `{VectorSpace F V}
    (P : V → Prop) `{SubspacePred F V P} : Zero (Sub V P) :=
  exist P 0 sp_zero.

#[global] Instance Sub_Add {F V : Type} `{VectorSpace F V}
    (P : V → Prop) `{SubspacePred F V P} : HasAdd (Sub V P) :=
  fun x y ⇒
    exist P (proj1_sig x + proj1_sig y)
      (sp_add (proj1_sig x) (proj1_sig y) (proj2_sig x) (proj2_sig y)).

#[global] Instance Sub_Neg {F V : Type} `{VectorSpace F V}
    (P : V → Prop) `{SubspacePred F V P} : HasNeg (Sub V P) :=
  fun x ⇒
    exist P ((- (1 : F)) *: proj1_sig x)
      (sp_scale (- (1 : F)) (proj1_sig x) (proj2_sig x)).

#[global] Instance Sub_Scale {F V : Type} `{VectorSpace F V}
    (P : V → Prop) `{SubspacePred F V P} : HasScale F (Sub V P) :=
  fun a x ⇒ exist P (a *: proj1_sig x) (sp_scale a (proj1_sig x) (proj2_sig x)).

#[global] Instance Sub_Equivalence {V : Type} {e : EqRel V} {E : Equivalence e}
    (P : V → Prop) : Equivalence (@Sub_EqRel V e P).
Proof.
  apply Build_Equivalence.
  - intros x. unfold Sub_EqRel. reflexivity.
  - intros x y Hxy. unfold Sub_EqRel in ×. symmetry. exact Hxy.
  - intros x y z Hxy Hyz. unfold Sub_EqRel in ×. rewrite Hxy. exact Hyz.
Qed.

  Proof. intros x. apply mon_idr. Qed.

Section SubGroup.
  Context {F V : Type} `{VectorSpace F V}.
  Context (P : V → Prop) `{SubspacePred F V P}.

  Lemma sub_invl : ∀ x : Sub V P, (- x) + x == 0.
  Proof.
    intros x.
    unfold equ, Sub_EqRel, add, Sub_Add, neg, Sub_Neg, zero, Sub_Zero.
    simpl.
    rewrite (neg1_scale (proj1_sig x)).
    apply grp_invl.
  Qed.

  Lemma sub_invr : ∀ x : Sub V P, x + (- x) == 0.
  Proof.
    intros x.
    unfold equ, Sub_EqRel, add, Sub_Add, neg, Sub_Neg, zero, Sub_Zero.
    simpl.
    rewrite (neg1_scale (proj1_sig x)).
    apply grp_invr.
  Qed.

  #[global] Instance sub_neg_proper : Proper (equ ==> equ) (@neg (Sub V P) _).
  Proof.
    intros x x' Hx.
    apply sc_proper; [reflexivity | exact Hx].
  Qed.

  #[global] Instance Sub_Group : Group (Sub V P) := {|
    grp_monoid := Sub_Monoid P;
    grp_neg := Sub_Neg P;
    grp_invl := sub_invl;
    grp_invr := sub_invr;
    grp_neg_proper := sub_neg_proper
  |}.

End SubGroup.

Section SubVectorSpace.
  Context {F V : Type} `{VectorSpace F V}.
  Context (P : V → Prop) `{SubspacePred F V P}.

  #[global] Instance sub_sc_proper : Proper (equ ==> equ ==> equ) (@scale F (Sub V P) _).
  Proof.
    intros a a' Ha x x' Hx.
    apply sc_proper; [exact Ha | exact Hx].
  Qed.

  Lemma sub_scale_add : ∀ (a : F) (x y : Sub V P),
    a *: (x + y) == a *: x + a *: y.
  Proof. intros a x y. apply sc_scale_add. Qed.

  Lemma sub_add_scale : ∀ (a b : F) (x : Sub V P),
    (a + b) *: x == a *: x + b *: x.
  Proof. intros a b x. apply sc_add_scale. Qed.

  Lemma sub_scale_scale : ∀ (a b : F) (x : Sub V P),
    (a × b) *: x == a *: (b *: x).
  Proof. intros a b x. apply sc_scale_scale. Qed.

  Lemma sub_scale_1 : ∀ x : Sub V P, 1 *: x == x.
  Proof. intros x. apply sc_scale_1. Qed.

  #[global] Instance Sub_ScaleLaws : ScaleLaws F (Sub V P) :=
    @Build_ScaleLaws F (Sub V P) _ (Sub_AbGroup P) (Sub_Scale P)
      sub_sc_proper sub_scale_add sub_add_scale sub_scale_scale sub_scale_1.

  #[global] Instance Sub_VectorSpace : VectorSpace F (Sub V P) :=
    @Build_VectorSpace F (Sub V P) _ (Sub_AbGroup P) (Sub_Scale P) Sub_ScaleLaws.

End SubVectorSpace.


Section SubInject.
  Context {F V : Type} `{VectorSpace F V}.
  Context (P : V → Prop) `{SubspacePred F V P}.

  #[global] Instance sub_inject_proper : Proper (equ ==> equ) (fun x : Sub V P ⇒ proj1_sig x).
  Proof. intros x y Hxy. exact Hxy. Qed.

  Lemma sub_inject_add : ∀ x y : Sub V P,
    proj1_sig (x + y) == proj1_sig x + proj1_sig y.
  Proof. intros x y. reflexivity. Qed.

  Lemma sub_inject_scale : ∀ (a : F) (x : Sub V P),
    proj1_sig (a *: x) == a *: proj1_sig x.
  Proof. intros a x. reflexivity. Qed.

  Definition sub_inject : LinearMap F (Sub V P) V :=
    @Build_LinearMap F (Sub V P) V _ (Sub_VectorSpace P) _
      (fun x ⇒ proj1_sig x)
      sub_inject_proper sub_inject_add sub_inject_scale.

End SubInject.
