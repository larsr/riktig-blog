Library Riktig.Theory.Exponential

From Coq Require Import Morphisms.
From Riktig Require Import Ops Interface.Category Theory.Category Theory.Product.

Section ExponentialTheory.
  Context {O : Type} {C : Category O} {P : HasProducts C}
          {E : @HasExponentials O C P}.

  #[global] Instance par_mor_proper (a b c d : O) :
    Proper (equ ==> equ ==> equ) (@par_mor O C P a b c d).
  Proof.
    intros f f' Hf g g' Hg.
    apply pair_unique.
    - rewrite par_pi1, Hf. reflexivity.
    - rewrite par_pi2, Hg. reflexivity.
  Qed.

  Lemma par_id : ∀ a b : O,
      (cat_id ×× cat_id) == (cat_id : prod_obj a b ~> prod_obj a b).
  Proof.
    intros a b.
    symmetry.
    apply pair_unique.
    - rewrite cat_idr, cat_idl. reflexivity.
    - rewrite cat_idr, cat_idl. reflexivity.
  Qed.

  Lemma par_comp : ∀ (a b c d x y : O)
      (f : c ~> x) (g : d ~> y) (h : a ~> c) (k : b ~> d),
      (f ×× g) ∘ (h ×× k) == (f ∘ h) ×× (g ∘ k).
  Proof.
    intros a b c d x y f g h k.
    apply pair_unique.
    - rewrite cat_assoc, par_pi1, <- (cat_assoc _ _ _ _ f pi1), par_pi1,
        cat_assoc.
      reflexivity.
    - rewrite cat_assoc, par_pi2, <- (cat_assoc _ _ _ _ g pi2), par_pi2,
        cat_assoc.
      reflexivity.
  Qed.

  Lemma par_comp_id : ∀ (x y z a : O) (f : y ~> z) (g : x ~> y),
      ((f ∘ g) ×× (cat_id : a ~> a)) == (f ×× cat_id) ∘ (g ×× cat_id).
  Proof.
    intros x y z a f g.
    rewrite par_comp, cat_idl.
    reflexivity.
  Qed.

  Definition uncurry {x a b : O} (k : x ~> exp_obj a b) : prod_obj x a ~> b :=
    eval_mor ∘ (k ×× cat_id).

  Theorem curry_uncurry : ∀ (x a b : O) (k : x ~> exp_obj a b),
      curry (uncurry k) == k.
  Proof.
    intros x a b k.
    symmetry.
    apply curry_unique.
    reflexivity.
  Qed.

  Theorem uncurry_curry : ∀ (x a b : O) (h : prod_obj x a ~> b),
      uncurry (curry h) == h.
  Proof.
    intros x a b h.
    apply curry_beta.
  Qed.

  Lemma curry_eval (E' : @HasExponentials O C P) : ∀ a b : O,
      @curry O C P E' (@exp_obj O C P E' a b) a b (@eval_mor O C P E' a b)
      == cat_id.
  Proof.
    intros a b.
    symmetry.
    apply (@curry_unique O C P E').
    rewrite par_id, cat_idr.
    reflexivity.
  Qed.

  Lemma exp_roundtrip (E1 E2 : @HasExponentials O C P) : ∀ a b : O,
      @curry O C P E1 (@exp_obj O C P E2 a b) a b (@eval_mor O C P E2 a b)
        ∘ @curry O C P E2 (@exp_obj O C P E1 a b) a b (@eval_mor O C P E1 a b)
      == cat_id.
  Proof.
    intros a b.
    rewrite <- (curry_eval E1 a b).
    apply (@curry_unique O C P E1).
    rewrite par_comp_id, cat_assoc,
      (@curry_beta O C P E1), (@curry_beta O C P E2).
    reflexivity.
  Qed.

  Definition exp_unique_up_to (E1 E2 : @HasExponentials O C P) (a b : O) :
    Iso (@exp_obj O C P E1 a b) (@exp_obj O C P E2 a b) := {|
      iso_to := @curry O C P E2 (@exp_obj O C P E1 a b) a b
                  (@eval_mor O C P E1 a b);
      iso_from := @curry O C P E1 (@exp_obj O C P E2 a b) a b
                  (@eval_mor O C P E2 a b);
      iso_fromto := exp_roundtrip E1 E2 a b;
      iso_tofrom := exp_roundtrip E2 E1 a b
    |}.

End ExponentialTheory.
