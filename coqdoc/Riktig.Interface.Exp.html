Library Riktig.Interface.Exp

From Coq Require Import Morphisms.
From Riktig Require Import Ops Interface.Field Interface.Order.

Class Exp (F : Type) `{OrderedField F} := {
  exp_fn : F → F;
  #[global] exp_fn_proper :: Proper (equ ==> equ) exp_fn;

  
  exp_zero : exp_fn 0 == 1;
  exp_add : ∀ a b : F, exp_fn (a + b) == exp_fn a × exp_fn b;

  
  exp_pos : ∀ a : F, 0 ≤ exp_fn a;

  
  exp_mono : ∀ a b : F, a < b → exp_fn a < exp_fn b
}.

#[global] Hint Mode Exp ! - - : typeclass_instances.
