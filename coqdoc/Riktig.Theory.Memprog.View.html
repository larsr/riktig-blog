Section StoreLemmas.
  Context {addr val : Type}.
  Context (eqb : addr → addr → bool).
  Hypothesis eqb_eq : ∀ a b : addr, eqb a b = true ↔ a = b.

  Lemma eqb_same (a : addr) : eqb a a = true.
  Proof. apply eqb_eq; reflexivity. Qed.

  Lemma eqb_diff (a b : addr) : a ≠ b → eqb a b = false.
  Proof.
    intros N; destruct (eqb a b) eqn:E; [|reflexivity].
    exfalso; apply N, eqb_eq, E.
  Qed.

  Lemma store_lookup_insert_eq (a : addr) (c : Cell addr val) (s : AStore addr val) :
    store_lookup eqb a (store_insert a c s) = Some c.
  Proof. cbn; now rewrite eqb_same. Qed.

  Lemma store_lookup_insert_neq (a a' : addr) (c : Cell addr val) (s : AStore addr val) :
    a' ≠ a → store_lookup eqb a' (store_insert a c s) = store_lookup eqb a' s.
  Proof. intros N; cbn; now rewrite (eqb_diff a' a N). Qed.

  Lemma store_member_insert_eq (a : addr) (c : Cell addr val) (s : AStore addr val) :
    store_member eqb a (store_insert a c s) = true.
  Proof. cbn; now rewrite eqb_same. Qed.

  Lemma store_member_insert_neq (a a' : addr) (c : Cell addr val) (s : AStore addr val) :
    a' ≠ a → store_member eqb a' (store_insert a c s) = store_member eqb a' s.
  Proof. intros N; cbn; now rewrite (eqb_diff a' a N). Qed.


  Lemma store_lookup_member (a : addr) (s : AStore addr val) (c : Cell addr val) :
    store_lookup eqb a s = Some c → store_member eqb a s = true.
  Proof.
    induction s as [|[k c'] s IH]; cbn; [discriminate|].
    destruct (eqb a k); [reflexivity | exact IH].
  Qed.

  Lemma store_member_lookup (a : addr) (s : AStore addr val) :
    store_member eqb a s = true →
    ∃ c : Cell addr val, store_lookup eqb a s = Some c.
  Proof.
    induction s as [|[k c'] s IH]; cbn; [discriminate|].
    destruct (eqb a k); [intros _; now ∃ c' | exact IH].
  Qed.

End StoreLemmas.



Fixpoint upd_nth {A : Type} (i : nat) (x : A) (l : list A) : list A :=
  match i, l with
  | _, [] ⇒ []
  | O, _ :: t ⇒ x :: t
  | S i', h :: t ⇒ h :: upd_nth i' x t
  end.

Lemma upd_nth_length {A : Type} : ∀ (l : list A) (i : nat) (x : A),
  length (upd_nth i x l) = length l.
Proof.
  induction l as [|h t IH]; intros [|i] x; cbn; try reflexivity.
  now rewrite IH.
Qed.

Lemma nth_error_upd_nth_eq {A : Type} : ∀ (l : list A) (i : nat) (x : A),
  (i < length l)%nat → nth_error (upd_nth i x l) i = Some x.
Proof.
  induction l as [|h t IH]; intros [|i] x H; cbn in *; try lia.
  - reflexivity.
  - apply IH; lia.
Qed.

Lemma nth_error_upd_nth_neq {A : Type} : ∀ (l : list A) (i j : nat) (x : A),
  j ≠ i → nth_error (upd_nth i x l) j = nth_error l j.
Proof.
  induction l as [|h t IH]; intros [|i] [|j] x N; cbn; try reflexivity.
  - congruence.
  - apply IH; congruence.
Qed.


Lemma perm_by_count {A : Type} (eq_dec : ∀ x y : A, {x = y} + {x ≠ y})
    (l1 l2 : list A) :
  forallb (fun x ⇒ Nat.eqb (count_occ eq_dec l1 x) (count_occ eq_dec l2 x))
          (l1 ++ l2) = true →
  Permutation l1 l2.
Proof.
  intros H. apply (Permutation_count_occ eq_dec). intros x.
  destruct (in_dec eq_dec x (l1 ++ l2)) as [I|NI].
  - apply Nat.eqb_eq. exact (proj1 (forallb_forall _ (l1 ++ l2)) H x I).
  - assert (N1 : ¬ In x l1) by (intro; apply NI, in_or_app; auto).
    assert (N2 : ¬ In x l2) by (intro; apply NI, in_or_app; auto).
    apply (count_occ_not_In eq_dec) in N1, N2.
    now rewrite N1, N2.
Qed.


Local Open Scope Z_scope.


Definition mach_write (a : Z) (c : Cell Z Z) (m : Machine Z Z) : Machine Z Z :=
  MkMachine (store_insert a c (mach_store m)) (mach_sp m) (mach_hp m).

Lemma mach_write_store (a : Z) (c : Cell Z Z) (m : Machine Z Z) :
  mach_store (mach_write a c m) = store_insert a c (mach_store m).
Proof. reflexivity. Qed.


Lemma mem_read_cells_0 (base : Z) (m : Machine Z Z) :
  mem_read_cells 0 base m = Some [].
Proof. reflexivity. Qed.

Lemma mem_read_cells_S (n : nat) (base : Z) (m : Machine Z Z) :
  mem_read_cells (S n) base m =
  match store_lookup Z.eqb base (mach_store m) with
  | Some (CellVal v) ⇒ option_map (cons v) (mem_read_cells n (base + 1) m)
  | _ ⇒ None
  end.
Proof. reflexivity. Qed.


Lemma mem_to_list_some (m : Machine Z Z) (base : Z) (n : nat) :
  mem_to_list m (Some base) n = mem_read_cells n base m.
Proof. reflexivity. Qed.


Lemma mem_read_cells_length : ∀ (n : nat) (base : Z) (m : Machine Z Z) (l : list Z),
  mem_read_cells n base m = Some l → length l = n.
Proof.
  induction n as [|n IH]; intros base m l.
  - cbn. now intros E; injection E as <-.
  - rewrite mem_read_cells_S.
    destruct (store_lookup Z.eqb base (mach_store m)) as [[v|p|q]|];
      try (intros E; discriminate E).
    destruct (mem_read_cells n (base + 1) m) as [l'|] eqn:Etl; cbn;
      [|intros E; discriminate E].
    intros E; injection E as <-.
    cbn [length]; now rewrite (IH _ _ _ Etl).
Qed.


Lemma mem_read_cells_ext : ∀ (n : nat) (base : Z) (m1 m2 : Machine Z Z),
  (∀ a : Z, base ≤ a < base + Z.of_nat n →
     store_lookup Z.eqb a (mach_store m1) = store_lookup Z.eqb a (mach_store m2)) →
  mem_read_cells n base m1 = mem_read_cells n base m2.
Proof.
  induction n as [|n IH]; intros base m1 m2 H; [reflexivity|].
  rewrite !mem_read_cells_S.
  rewrite (H base) by lia.
  destruct (store_lookup Z.eqb base (mach_store m2)) as [[v|p|q]|]; try reflexivity.
  rewrite (IH (base + 1) m1 m2); [reflexivity|].
  intros a Ha; apply H; lia.
Qed.


Lemma mem_read_cells_store_eq (n : nat) (base : Z) (m1 m2 : Machine Z Z) :
  mach_store m1 = mach_store m2 →
  mem_read_cells n base m1 = mem_read_cells n base m2.
Proof. intros E; apply mem_read_cells_ext; intros a _; now rewrite E. Qed.


Lemma mem_read_cells_insert_outside (n : nat) (base a : Z) (c : Cell Z Z)
    (m : Machine Z Z) :
  a < base ∨ base + Z.of_nat n ≤ a →
  mem_read_cells n base (mach_write a c m) = mem_read_cells n base m.
Proof.
  intros Hout. apply mem_read_cells_ext. intros a' Ha'.
  rewrite mach_write_store.
  apply (store_lookup_insert_neq Z.eqb Z.eqb_eq); lia.
Qed.


Lemma mem_read_cells_insert_inside :
  ∀ (n i : nat) (base v : Z) (m : Machine Z Z) (l : list Z),
    (i < n)%nat →
    mem_read_cells n base m = Some l →
    mem_read_cells n base (mach_write (base + Z.of_nat i) (CellVal v) m)
      = Some (upd_nth i v l).
Proof.
  induction n as [|n IH]; intros i base v m l Hi; [lia|].
  rewrite !mem_read_cells_S, mach_write_store.
  destruct i as [|i].
  -
    replace (base + Z.of_nat 0) with base by lia.
    rewrite (store_lookup_insert_eq Z.eqb Z.eqb_eq).
    destruct (store_lookup Z.eqb base (mach_store m)) as [[w|p|q]|];
      try (intros E; discriminate E).
    destruct (mem_read_cells n (base + 1) m) as [l'|] eqn:Etl; cbn;
      [|intros E; discriminate E].
    intros E; injection E as <-.
    cbn [upd_nth].
    rewrite mem_read_cells_insert_outside by lia.
    now rewrite Etl.
  -
    rewrite (store_lookup_insert_neq Z.eqb Z.eqb_eq) by lia.
    destruct (store_lookup Z.eqb base (mach_store m)) as [[w|p|q]|];
      try (intros E; discriminate E).
    destruct (mem_read_cells n (base + 1) m) as [l'|] eqn:Etl; cbn [option_map];
      [|intros E; discriminate E].
    intros E; injection E as <-.
    cbn [upd_nth].
    replace (base + Z.of_nat (S i)) with (base + 1 + Z.of_nat i) by lia.
    rewrite (IH i (base + 1) v m l'); [reflexivity | lia | exact Etl].
Qed.


Lemma mem_read_cells_nth_error :
  ∀ (n i : nat) (base : Z) (m : Machine Z Z) (l : list Z),
    mem_read_cells n base m = Some l → (i < n)%nat →
    ∃ v : Z,
      nth_error l i = Some v ∧
      store_lookup Z.eqb (base + Z.of_nat i) (mach_store m) = Some (CellVal v).
Proof.
  induction n as [|n IH]; intros i base m l E Hi; [lia|].
  revert E. rewrite mem_read_cells_S.
  destruct (store_lookup Z.eqb base (mach_store m)) as [[w|p|q]|] eqn:Elk;
    try (intros E; discriminate E).
  destruct (mem_read_cells n (base + 1) m) as [l'|] eqn:Etl; cbn;
    [|intros E; discriminate E].
  intros E; injection E as <-.
  destruct i as [|i].
  - ∃ w. split; [reflexivity|].
    now replace (base + Z.of_nat 0) with base by lia.
  - destruct (IH i (base + 1) m l' Etl ltac:(lia)) as (v & Hnth & Hlk).
    ∃ v. split; [exact Hnth|].
    now replace (base + Z.of_nat (S i)) with (base + 1 + Z.of_nat i) by lia.
Qed.


Corollary mem_read_cells_member (n i : nat) (base : Z) (m : Machine Z Z)
    (l : list Z) :
  mem_read_cells n base m = Some l → (i < n)%nat →
  store_member Z.eqb (base + Z.of_nat i) (mach_store m) = true.
Proof.
  intros E Hi.
  destruct (mem_read_cells_nth_error n i base m l E Hi) as (v & _ & Hlk).
  exact (store_lookup_member Z.eqb _ _ _ Hlk).
Qed.


Lemma mem_read_cells_split :
  ∀ (n m' : nat) (base : Z) (mach : Machine Z Z),
    mem_read_cells (n + m')%nat base mach =
    match mem_read_cells n base mach with
    | Some l1 ⇒
        match mem_read_cells m' (base + Z.of_nat n) mach with
        | Some l2 ⇒ Some (l1 ++ l2)
        | None ⇒ None
        end
    | None ⇒ None
    end.
Proof.
  induction n as [|n IH]; intros m' base mach.
  - cbn. rewrite Z.add_0_r.
    destruct (mem_read_cells m' base mach); reflexivity.
  - cbn [Nat.add]. rewrite !mem_read_cells_S.
    destruct (store_lookup Z.eqb base (mach_store mach)) as [[v|p|q]|];
      try reflexivity.
    rewrite IH.
    replace (base + Z.of_nat (S n)) with (base + 1 + Z.of_nat n) by lia.
    destruct (mem_read_cells n (base + 1) mach) as [l1|]; cbn; [|reflexivity].
    destruct (mem_read_cells m' (base + 1 + Z.of_nat n) mach) as [l2|]; cbn;
      reflexivity.
Qed.

Corollary mem_read_cells_app (n m' : nat) (base : Z) (mach : Machine Z Z)
    (l1 l2 : list Z) :
  mem_read_cells n base mach = Some l1 →
  mem_read_cells m' (base + Z.of_nat n) mach = Some l2 →
  mem_read_cells (n + m')%nat base mach = Some (l1 ++ l2).
Proof. intros E1 E2. now rewrite mem_read_cells_split, E1, E2. Qed.

Corollary mem_read_cells_split_inv (n m' : nat) (base : Z) (mach : Machine Z Z)
    (l : list Z) :
  mem_read_cells (n + m')%nat base mach = Some l →
  ∃ l1 l2 : list Z,
    l = l1 ++ l2 ∧
    mem_read_cells n base mach = Some l1 ∧
    mem_read_cells m' (base + Z.of_nat n) mach = Some l2.
Proof.
  rewrite mem_read_cells_split.
  destruct (mem_read_cells n base mach) as [l1|] eqn:E1; [|discriminate].
  destruct (mem_read_cells m' (base + Z.of_nat n) mach) as [l2|] eqn:E2;
    [|discriminate].
  intros E; injection E as <-. now ∃ l1, l2.
Qed.



Definition sorted_output (input output : list Z) : Prop :=
  Permutation input output ∧ Sorted ole output.


Lemma sorted_output_Zle (input output : list Z) :
  sorted_output input output ↔ Permutation input output ∧ Sorted Z.le output.
Proof. reflexivity. Qed.


Lemma sorted_output_unique (input o1 o2 : list Z) :
  sorted_output input o1 → sorted_output input o2 → o1 = o2.
Proof.
  intros [P1 S1] [P2 S2].
  apply sorted_perm_unique; [exact S1 | exact S2 |].
  exact (perm_trans (Permutation_sym P1) P2).
Qed.


Example sorted_output_example :
  sorted_output [5; 3; 8; 1; 9; 2; 7; 4; 6] [1; 2; 3; 4; 5; 6; 7; 8; 9].
Proof.
  split.
  - apply (perm_by_count Z.eq_dec). vm_compute. reflexivity.
  - repeat constructor; apply (proj1 (Z.leb_le _ _)); reflexivity.
Qed.
