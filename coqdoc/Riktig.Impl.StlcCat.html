Library Riktig.Impl.StlcCat


From Coq Require Export Morphisms.
From Riktig Require Export Interface.Category.


Inductive ty : Type :=
  | Base : ty
  | Arr : ty → ty → ty.

Inductive var : list ty → ty → Type :=
  | here : ∀ G T, var (T :: G) T
  | there : ∀ G U T, var G T → var (U :: G) T.
Arguments here {G T}.
Arguments there {G U T} _.

Inductive tm : list ty → ty → Type :=
  | TVar : ∀ G T, var G T → tm G T
  | TAbs : ∀ G A B, tm (A :: G) B → tm G (Arr A B)
  | TApp : ∀ G A B, tm G (Arr A B) → tm G A → tm G B.
Arguments TVar {G T} _.
Arguments TAbs {G A B} _.
Arguments TApp {G A B} _ _.


Fixpoint wkvar {G D T} (x : var G T) : var (G ++ D) T :=
  match x in var G0 T0 return var (G0 ++ D) T0 with
  | here ⇒ here
  | there x' ⇒ there (wkvar x')
  end.

Fixpoint wkt {G D T} (t : tm G T) : tm (G ++ D) T :=
  match t in tm G0 T0 return tm (G0 ++ D) T0 with
  | TVar x ⇒ TVar (wkvar x)
  | TAbs b ⇒ TAbs (wkt b)
  | TApp f a ⇒ TApp (wkt f) (wkt a)
  end.

Definition wk {D T} (t : tm nil T) : tm D T := wkt t.


Definition Hom (A B : ty) : Type := tm nil (Arr A B).

Definition tid (A : ty) : Hom A A := TAbs (TVar here).

Definition tcomp {A B C} (g : Hom B C) (f : Hom A B) : Hom A C :=
  TAbs (TApp (wk g) (TApp (wk f) (TVar here))).


Inductive conv : ∀ (G : list ty) (T : ty), tm G T → tm G T → Prop :=
  | conv_refl : ∀ G T (t : tm G T), conv G T t t
  | conv_sym : ∀ G T (s t : tm G T),
      conv G T s t → conv G T t s
  | conv_trans : ∀ G T (s t u : tm G T),
      conv G T s t → conv G T t u → conv G T s u
  | conv_app : ∀ G A B (f f' : tm G (Arr A B)) (a a' : tm G A),
      conv G (Arr A B) f f' → conv G A a a' →
      conv G B (TApp f a) (TApp f' a')
  | conv_abs : ∀ G A B (b b' : tm (A :: G) B),
      conv (A :: G) B b b' → conv G (Arr A B) (TAbs b) (TAbs b')
  
  | conv_wk : ∀ D T (s t : tm nil T),
      conv nil T s t → conv D T (wk s) (wk t)
  
  | conv_beta_id : ∀ G A (v : tm G A),
      conv G A (TApp (TAbs (TVar here)) v) v
  
  | conv_beta_comp : ∀ G A B C (g : Hom B C) (f : Hom A B) (v : tm G A),
      conv G C (TApp (wk (tcomp g f)) v) (TApp (wk g) (TApp (wk f) v))
  
  | conv_eta : ∀ A B (f : Hom A B),
      conv nil (Arr A B) (TAbs (TApp (wk f) (TVar here))) f.

Arguments conv {G T} _ _.
Arguments conv_refl {G T} t.
Arguments conv_sym {G T s t} _.
Arguments conv_trans {G T s t u} _ _.
Arguments conv_app {G A B f f' a a'} _ _.
Arguments conv_abs {G A B b b'} _.
Arguments conv_wk {D T s t} _.
Arguments conv_beta_id {G A} v.
Arguments conv_beta_comp {G A B C} g f v.
Arguments conv_eta {A B} f.

#[global] Instance conv_equivalence {G : list ty} {T : ty} :
  Equivalence (@conv G T).
Proof.
  split.
  - intros t. exact (conv_refl t).
  - intros s t H. exact (conv_sym H).
  - intros s t u H1 H2. exact (conv_trans H1 H2).
Qed.


Lemma conv_idl : ∀ A B (f : Hom A B), conv (tcomp (tid B) f) f.
Proof.
  intros A B f.
  apply (conv_trans (t := TAbs (TApp (wk f) (TVar here)))).
  - apply conv_abs. apply conv_beta_id.
  - apply conv_eta.
Qed.

Lemma conv_idr : ∀ A B (f : Hom A B), conv (tcomp f (tid A)) f.
Proof.
  intros A B f.
  apply (conv_trans (t := TAbs (TApp (wk f) (TVar here)))).
  - apply conv_abs. apply conv_app.
    + apply conv_refl.
    + apply conv_beta_id.
  - apply conv_eta.
Qed.

Lemma conv_assoc :
  ∀ W X Y Z (f : Hom Y Z) (g : Hom X Y) (h : Hom W X),
    conv (tcomp f (tcomp g h)) (tcomp (tcomp f g) h).
Proof.
  intros W X Y Z f g h.
  apply (conv_trans
    (t := TAbs (TApp (wk f) (TApp (wk g) (TApp (wk h) (TVar here)))))).
  - apply conv_abs. apply conv_app.
    + apply conv_refl.
    + apply conv_beta_comp.
  - apply conv_sym. apply conv_abs. apply conv_beta_comp.
Qed.

#[global] Instance tcomp_proper (A B C : ty) :
  Proper (@conv nil (Arr B C) ==> @conv nil (Arr A B) ==> @conv nil (Arr A C))
         (@tcomp A B C).
Proof.
  intros g g' Hg f f' Hf.
  apply conv_abs. apply conv_app.
  - apply conv_wk. exact Hg.
  - apply conv_app.
    + apply conv_wk. exact Hf.
    + apply conv_refl.
Qed.


#[global] Instance Stlc_Category : Category ty := {|
  hom := Hom;
  hom_equ := fun A B ⇒ @conv nil (Arr A B);
  hom_equiv := fun A B ⇒ @conv_equivalence nil (Arr A B);
  cat_id := tid;
  comp := @tcomp;
  comp_proper := tcomp_proper;
  cat_idl := conv_idl;
  cat_idr := conv_idr;
  cat_assoc := conv_assoc
|}.


Example wk_tid_computes :
  wk (D := Base :: nil) (tid Base) = TAbs (TVar here).
Proof. reflexivity. Qed.

Example tcomp_computes :
  tcomp (tid Base) (tid Base)
  = TAbs (TApp (TAbs (TVar here)) (TApp (TAbs (TVar here)) (TVar here))).
Proof. reflexivity. Qed.

Example conv_idl_base : conv (tcomp (tid Base) (tid Base)) (tid Base).
Proof. exact (conv_idl Base Base (tid Base)). Qed.
