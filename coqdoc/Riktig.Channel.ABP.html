Library Riktig.Channel.ABP


From Coq Require Export List Bool.
From Riktig.Theory Require Export Subseq.
From Riktig.Channel Require Export Trace.
Import ListNotations.

Section ABP.
  Context {M : Type}.


  Definition bit := bool.

  Record Packet := { pkt_bit : bit; pkt_msg : M }.


  Fixpoint sender_burst (b : bit) (m : M) (k : nat) : list Packet :=
    match k with
    | O ⇒ []
    | S k' ⇒ {| pkt_bit := b; pkt_msg := m |} :: sender_burst b m k'
    end.

  Fixpoint sender_stream (b : bit) (ms : list M) (k : nat) : list Packet :=
    match ms with
    | [] ⇒ []
    | m :: ms' ⇒ sender_burst b m k ++ sender_stream (negb b) ms' k
    end.


  Fixpoint deliver (expect : bit) (ps : list Packet) : list M :=
    match ps with
    | [] ⇒ []
    | p :: ps' ⇒
        if Bool.eqb (pkt_bit p) expect
        then pkt_msg p :: deliver (negb expect) ps'
        else deliver expect ps'
    end.


  Fixpoint next_bit (expect : bit) (ps : list Packet) : bit :=
    match ps with
    | [] ⇒ expect
    | p :: ps' ⇒
        if Bool.eqb (pkt_bit p) expect
        then next_bit (negb expect) ps'
        else next_bit expect ps'
    end.


  Lemma negb_neq : ∀ b : bit, negb b ≠ b.
  Proof.
    intros b H.
    destruct b; discriminate H.
  Qed.

  Lemma bit_dec : ∀ b1 b2 : bit, {b1 = b2} + {b1 ≠ b2}.
  Proof.
    intros b1 b2.
    destruct b1; destruct b2.
    - left. reflexivity.
    - right. intros H. discriminate H.
    - right. intros H. discriminate H.
    - left. reflexivity.
  Defined.

  Lemma eqb_bit_true : ∀ b1 b2 : bit, b1 = b2 → Bool.eqb b1 b2 = true.
  Proof.
    intros b1 b2 H.
    rewrite H.
    destruct b2; reflexivity.
  Qed.

  Lemma eqb_bit_false : ∀ b1 b2 : bit, b1 ≠ b2 → Bool.eqb b1 b2 = false.
  Proof.
    intros b1 b2 H.
    destruct b1; destruct b2; simpl.
    - destruct (H eq_refl).
    - reflexivity.
    - reflexivity.
    - destruct (H eq_refl).
  Qed.


  Lemma deliver_app : ∀ (w1 w2 : list Packet) (expect : bit),
    deliver expect (w1 ++ w2) = deliver expect w1 ++ deliver (next_bit expect w1) w2.
  Proof.
    induction w1 as [| p w1' IH]; intros w2 expect.
    - simpl. reflexivity.
    - simpl. destruct (Bool.eqb (pkt_bit p) expect) eqn:Hcmp.
      + simpl. rewrite (IH w2 (negb expect)). reflexivity.
      + apply IH.
  Qed.


  Lemma sender_burst_all_eq : ∀ (b : bit) (m : M) (k : nat) (p : Packet),
    In p (sender_burst b m k) → p = {| pkt_bit := b; pkt_msg := m |}.
  Proof.
    intros b m k.
    induction k as [| k' IH]; intros p Hp.
    - simpl in Hp. destruct Hp.
    - simpl in Hp. destruct Hp as [Hp | Hp].
      + symmetry. exact Hp.
      + apply IH. exact Hp.
  Qed.



  Lemma deliver_homog_neq : ∀ (x0 : Packet) (w : list Packet) (expect : bit),
    (∀ p, In p w → p = x0) → pkt_bit x0 ≠ expect →
    deliver expect w = [] ∧ next_bit expect w = expect.
  Proof.
    intros x0 w.
    induction w as [| p w' IH]; intros expect Hall Hneq.
    - split; reflexivity.
    - assert (Hp : p = x0).
      { apply Hall. simpl. left. reflexivity. }
      rewrite Hp. simpl.
      rewrite (eqb_bit_false (pkt_bit x0) expect Hneq).
      apply IH.
      + intros q Hq. apply Hall. simpl. right. exact Hq.
      + exact Hneq.
  Qed.

  Lemma deliver_homog_eq : ∀ (x0 : Packet) (w : list Packet) (expect : bit),
    (∀ p, In p w → p = x0) → pkt_bit x0 = expect →
    (deliver expect w = [] ∧ next_bit expect w = expect) ∨
    (deliver expect w = [pkt_msg x0] ∧ next_bit expect w = negb expect).
  Proof.
    intros x0 w expect Hall Heq.
    destruct w as [| p w'].
    - left. split; reflexivity.
    - assert (Hp : p = x0).
      { apply Hall. simpl. left. reflexivity. }
      right.
      rewrite Hp. simpl.
      rewrite (eqb_bit_true (pkt_bit x0) expect Heq). simpl.
      assert (Hnw : ∀ q, In q w' → q = x0).
      { intros q Hq. apply Hall. simpl. right. exact Hq. }
      assert (Hneq2 : pkt_bit x0 ≠ negb expect).
      { intros Hcontra.
        rewrite Heq in Hcontra.
        apply (negb_neq expect).
        symmetry. exact Hcontra. }
      destruct (deliver_homog_neq x0 w' (negb expect) Hnw Hneq2) as [Hd Hn].
      rewrite Hd. rewrite Hn.
      split; reflexivity.
  Qed.


  Lemma burst_delivery : ∀ (b : bit) (m : M) (k : nat) (w : list Packet) (expect : bit),
    lossy_dup (sender_burst b m k) w →
    (deliver expect w = [] ∧ next_bit expect w = expect) ∨
    (deliver expect w = [m] ∧ next_bit expect w = negb expect).
  Proof.
    intros b m k w expect H.
    assert (Hall : ∀ p, In p w → p = {| pkt_bit := b; pkt_msg := m |}).
    { apply (lossy_dup_preserves_all_eq {| pkt_bit := b; pkt_msg := m |}
               (sender_burst b m k) w H).
      intros p Hp. apply (sender_burst_all_eq b m k p Hp). }
    destruct (bit_dec (pkt_bit ({| pkt_bit := b; pkt_msg := m |})) expect) as [Heq | Hneq].
    - destruct (deliver_homog_eq {| pkt_bit := b; pkt_msg := m |} w expect Hall Heq)
        as [[Hd Hn] | [Hd Hn]].
      + left. split.
        × exact Hd.
        × exact Hn.
      + right. split.
        × exact Hd.
        × exact Hn.
    - left. apply (deliver_homog_neq {| pkt_bit := b; pkt_msg := m |} w expect Hall Hneq).
  Qed.



  Lemma abp_deliver_subseq : ∀ (k : nat) (ms : list M)
      (b_stream expect : bit) (wire : list Packet),
    lossy_dup (sender_stream b_stream ms k) wire → subseq (deliver expect wire) ms.
  Proof.
    intros k ms.
    induction ms as [| m ms' IH]; intros b_stream expect wire H.
    - simpl in H.
      assert (Hnil : wire = []).
      { apply (lossy_dup_nil_l [] wire H). reflexivity. }
      rewrite Hnil. simpl. apply subseq_nil.
    - simpl in H.
      destruct (lossy_dup_app_split _ _ H
                  (sender_burst b_stream m k) (sender_stream (negb b_stream) ms' k)
                  eq_refl) as [w1 [w2 [Hwire [H1 H2]]]].
      rewrite Hwire.
      rewrite (deliver_app w1 w2 expect).
      destruct (burst_delivery b_stream m k w1 expect H1) as [[Hd1 Hn1] | [Hd1 Hn1]].
      + rewrite Hd1. rewrite Hn1. simpl.
        apply subseq_drop.
        apply (IH (negb b_stream) expect w2 H2).
      + rewrite Hd1. rewrite Hn1. simpl.
        apply subseq_keep.
        apply (IH (negb b_stream) (negb expect) w2 H2).
  Qed.


  Theorem abp_safety : ∀ (k : nat) (b0 : bit) (ms : list M) (wire : list Packet),
    lossy_dup (sender_stream b0 ms k) wire → subseq (deliver b0 wire) ms.
  Proof.
    intros k b0 ms wire H.
    apply (abp_deliver_subseq k ms b0 b0 wire H).
  Qed.

End ABP.


Example abp_gap_wire :
  lossy_dup (sender_stream false [0; 1; 2; 3] 1)
    [{| pkt_bit := false; pkt_msg := 0 |};
     {| pkt_bit := false; pkt_msg := 2 |};
     {| pkt_bit := true; pkt_msg := 3 |}].
Proof.
  simpl.
  apply lossy_dup_keep.
  apply lossy_dup_drop.
  apply lossy_dup_keep.
  apply lossy_dup_keep.
  apply lossy_dup_nil.
Qed.


Example abp_gap_not_prefix :
  ¬ prefix (deliver false
    [{| pkt_bit := false; pkt_msg := 0 |};
     {| pkt_bit := false; pkt_msg := 2 |};
     {| pkt_bit := true; pkt_msg := 3 |}])
    [0; 1; 2; 3].
Proof.
  intros [s2 Heq].
  simpl in Heq.
  discriminate Heq.
Qed.


Example abp_reorder_scrambles_wire :
  reorder (sender_stream false [0; 1; 2] 1) (rev (sender_stream false [0; 1; 2] 1)).
Proof.
  apply Permutation_rev.
Qed.

Example abp_reorder_scrambles_not_subseq :
  ¬ subseq (deliver false (rev (sender_stream false [0; 1; 2] 1))) [0; 1; 2].
Proof.
  intros H.
  assert (Heq : deliver false (rev (sender_stream false [0; 1; 2] 1)) = [0; 1; 2]).
  { apply subseq_length_eq.
    - exact H.
    - reflexivity. }
  simpl in Heq.
  discriminate Heq.
Qed.
