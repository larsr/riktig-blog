Section FieldScaleBridge.
  Context {F : Type} `{FF : Field F}.

  Lemma field_scale_eq : ∀ a u : F, a *: u == a × u.
  Proof.
    intros a u.
    reflexivity.
  Qed.

End FieldScaleBridge.


Section TrivialGA.
  Context {F : Type} `{FF : Field F}.

  Definition self_inner (u v : F) : F := u × v.

  #[global] Instance self_inner_proper : Proper (equ ==> equ ==> equ) self_inner.
  Proof.
    intros u u' Hu v v' Hv.
    exact (ring_mul_proper u u' Hu v v' Hv).
  Qed.

  Lemma self_inner_sym : ∀ u v : F, self_inner u v == self_inner v u.
  Proof.
    intros u v.
    exact (field_mul_comm u v).
  Qed.

  Lemma self_inner_add_l : ∀ u v w : F,
    self_inner (u + v) w == self_inner u w + self_inner v w.
  Proof.
    intros u v w.
    exact (ring_distr u v w).
  Qed.

  Lemma self_inner_scale_l : ∀ a u v : F,
    self_inner (a *: u) v == a × self_inner u v.
  Proof.
    intros a u v.
    rewrite field_scale_eq.
    exact (ring_mul_assoc a u v).
  Qed.

  Definition SelfDot_InnerProduct : InnerProduct F F :=
    @Build_InnerProduct F F _ _ self_inner
      self_inner_proper self_inner_sym self_inner_add_l self_inner_scale_l.

  Lemma self_inner_norm2 : ∀ u : F,
    @InnerProduct.norm2 F F _ _ SelfDot_InnerProduct u == self_inner u u.
  Proof.
    intros u.
    reflexivity.
  Qed.

  Definition trivial_vec (u : F) : F := u.

  #[global] Instance trivial_vec_proper : Proper (equ ==> equ) trivial_vec.
  Proof.
    intros u v Huv.
    exact Huv.
  Qed.

  Lemma trivial_vec_add : ∀ u v : F,
    trivial_vec (u + v) == trivial_vec u + trivial_vec v.
  Proof.
    intros u v.
    reflexivity.
  Qed.

  Lemma trivial_vec_scale : ∀ a u : F,
    trivial_vec (a *: u) == a *: trivial_vec u.
  Proof.
    intros a u.
    reflexivity.
  Qed.

  Lemma trivial_scale_mul_l : ∀ a x y : F,
    (a *: x) × y == a *: (x × y).
  Proof.
    intros a x y.
    rewrite field_scale_eq, (field_scale_eq a (x × y)).
    exact (ring_mul_assoc a x y).
  Qed.

  Lemma trivial_scale_mul_r : ∀ a x y : F,
    x × (a *: y) == a *: (x × y).
  Proof.
    intros a x y.
    rewrite field_scale_eq, (field_scale_eq a (x × y)).
    exact (mul_comm_swap x a y).
  Qed.

  Lemma trivial_contract : ∀ u : F,
    trivial_vec u × trivial_vec u
    == @InnerProduct.norm2 F F _ _ SelfDot_InnerProduct u *: 1.
  Proof.
    intros u.
    unfold trivial_vec.
    rewrite self_inner_norm2, (field_scale_eq (self_inner u u) 1).
    unfold self_inner.
    symmetry.
    apply ring_mul_1r.
  Qed.

  Definition GA_trivial : GeometricAlgebra F F F :=
    @Build_GeometricAlgebra F F F _ _ SelfDot_InnerProduct field_ring
      Field_Scale Field_ScaleLaws trivial_scale_mul_l trivial_scale_mul_r
      trivial_vec trivial_vec_proper trivial_vec_add trivial_vec_scale
      trivial_contract.

End TrivialGA.

Example ga_trivial_check (F : Type) (FF : Field F) : GeometricAlgebra F F F :=
  GA_trivial.


Section Cliff1GA.
  Context {F : Type} `{FF : Field F}.

  Definition cliff1_neg_inner (u v : F) : F := - (u × v).

  #[global] Instance cliff1_ip_proper : Proper (equ ==> equ ==> equ) cliff1_neg_inner.
  Proof.
    intros u u' Hu v v' Hv.
    exact (group_neg_proper (u × v) (u' × v') (ring_mul_proper u u' Hu v v' Hv)).
  Qed.

  Lemma cliff1_ip_sym : ∀ u v : F,
    cliff1_neg_inner u v == cliff1_neg_inner v u.
  Proof.
    intros u v.
    exact (group_neg_proper (u × v) (v × u) (field_mul_comm u v)).
  Qed.

  Lemma cliff1_ip_add_l : ∀ u v w : F,
    cliff1_neg_inner (u + v) w
    == cliff1_neg_inner u w + cliff1_neg_inner v w.
  Proof.
    intros u v w.
    unfold cliff1_neg_inner.
    rewrite ring_distr.
    apply neg_add.
  Qed.

  Lemma cliff1_ip_scale_l : ∀ a u v : F,
    cliff1_neg_inner (a *: u) v == a × cliff1_neg_inner u v.
  Proof.
    intros a u v.
    unfold cliff1_neg_inner.
    rewrite field_scale_eq, ring_mul_assoc.
    symmetry.
    apply mul_neg_r.
  Qed.

  Definition Cliff1_NegInner : InnerProduct F F :=
    @Build_InnerProduct F F _ _ cliff1_neg_inner
      cliff1_ip_proper cliff1_ip_sym cliff1_ip_add_l cliff1_ip_scale_l.

  Definition Cliff1_Scale (a : F) (z : C F) : C F :=
    mkC (a × re z) (a × im z).

  #[global] Instance cliff1_sc_proper : Proper (equ ==> equ ==> equ) Cliff1_Scale.
  Proof.
    intros a a' Ha z z' [Hr Hi].
    split.
    - exact (ring_mul_proper a a' Ha (re z) (re z') Hr).
    - exact (ring_mul_proper a a' Ha (im z) (im z') Hi).
  Qed.

  Lemma cliff1_scale_add : ∀ (a : F) (z w : C F),
    Cliff1_Scale a (z + w) == Cliff1_Scale a z + Cliff1_Scale a w.
  Proof.
    intros a z w.
    split.
    - exact (ring_distl a (re z) (re w)).
    - exact (ring_distl a (im z) (im w)).
  Qed.

  Lemma cliff1_add_scale : ∀ (a b : F) (u : C F),
    Cliff1_Scale (a + b) u == Cliff1_Scale a u + Cliff1_Scale b u.
  Proof.
    intros a b u.
    split.
    - exact (ring_distr a b (re u)).
    - exact (ring_distr a b (im u)).
  Qed.

  Lemma cliff1_scale_scale : ∀ (a b : F) (u : C F),
    Cliff1_Scale (a × b) u == Cliff1_Scale a (Cliff1_Scale b u).
  Proof.
    intros a b u.
    split.
    - exact (ring_mul_assoc a b (re u)).
    - exact (ring_mul_assoc a b (im u)).
  Qed.

  Lemma cliff1_scale_1 : ∀ u : C F, Cliff1_Scale 1 u == u.
  Proof.
    intros u.
    split.
    - exact (ring_mul_1l (re u)).
    - exact (ring_mul_1l (im u)).
  Qed.

  Definition Cliff1_ScaleLaws : ScaleLaws F (C F) :=
    @Build_ScaleLaws F (C F) _ _ Cliff1_Scale cliff1_sc_proper
      cliff1_scale_add cliff1_add_scale cliff1_scale_scale cliff1_scale_1.

  Definition cliff1_ga_vec (u : F) : C F := mkC 0 u.

  #[global] Instance cliff1_ga_vec_proper : Proper (equ ==> equ) cliff1_ga_vec.
  Proof.
    intros u v Huv.
    split.
    - reflexivity.
    - exact Huv.
  Qed.

  Lemma cliff1_ga_vec_add : ∀ u v : F,
    cliff1_ga_vec (u + v) == cliff1_ga_vec u + cliff1_ga_vec v.
  Proof.
    intros u v.
    split.
    - symmetry. exact (idl (0 : F)).
    - reflexivity.
  Qed.

  Lemma cliff1_ga_vec_scale : ∀ a u : F,
    cliff1_ga_vec (a *: u) == Cliff1_Scale a (cliff1_ga_vec u).
  Proof.
    intros a u.
    split.
    - symmetry. exact (mul_0_r a).
    - exact (field_scale_eq a u).
  Qed.

  Lemma cliff1_contract_re : ∀ u : F,
    0 × 0 + - (u × u) == (- (u × u)) × 1.
  Proof.
    intros u.
    calc (0 × 0 + - (u × u))
      == (0 + - (u × u)) { mul_0_l }
      == (- (u × u)) { idl }
      == ((- (u × u)) × 1) { ring_mul_1r }.
  Qed.

  Lemma cliff1_contract_im : ∀ u : F,
    0 × u + u × 0 == (- (u × u)) × 0.
  Proof.
    intros u.
    calc (0 × u + u × 0)
      == (0 + u × 0) { mul_0_l }
      == ((0 : F) + 0) { mul_0_r }
      == (0 : F) { idl }
      == ((- (u × u)) × 0) { mul_0_r }.
  Qed.

  Lemma cliff1_ga_contract : ∀ u : F,
    cliff1_ga_vec u × cliff1_ga_vec u
    == Cliff1_Scale (@InnerProduct.norm2 F F _ _ Cliff1_NegInner u) 1.
  Proof.
    intros u.
    split.
    - exact (cliff1_contract_re u).
    - exact (cliff1_contract_im u).
  Qed.

  Lemma cliff1_ga_scale_mul_re : ∀ a p q r s : F,
    (a × p) × r + - ((a × q) × s) == a × (p × r + - (q × s)).
  Proof.
    intros a p q r s.
    calc ((a × p) × r + - ((a × q) × s))
      == (a × (p × r) + - ((a × q) × s)) { assoc }
      == (a × (p × r) + - (a × (q × s))) { assoc }
      == (a × (p × r) + a × (- (q × s))) { mul_neg_r }
      == (a × (p × r + - (q × s))) { distl }.
  Qed.

  Lemma cliff1_ga_scale_mul_im : ∀ a p q r s : F,
    (a × p) × s + (a × q) × r == a × (p × s + q × r).
  Proof.
    intros a p q r s.
    calc ((a × p) × s + (a × q) × r)
      == (a × (p × s) + (a × q) × r) { assoc }
      == (a × (p × s) + a × (q × r)) { assoc }
      == (a × (p × s + q × r)) { distl }.
  Qed.

  Lemma cliff1_ga_scale_mul_l : ∀ (a : F) (x y : C F),
    Cliff1_Scale a x × y == Cliff1_Scale a (x × y).
  Proof.
    intros a x y.
    split.
    - exact (cliff1_ga_scale_mul_re a (re x) (im x) (re y) (im y)).
    - exact (cliff1_ga_scale_mul_im a (re x) (im x) (re y) (im y)).
  Qed.

  Lemma cliff1_ga_mul_scale_re : ∀ a p q r s : F,
    p × (a × r) + - (q × (a × s)) == a × (p × r + - (q × s)).
  Proof.
    intros a p q r s.
    calc (p × (a × r) + - (q × (a × s)))
      == (a × (p × r) + - (q × (a × s))) { (mul_comm_swap p a r) }
      == (a × (p × r) + - (a × (q × s))) { (mul_comm_swap q a s) }
      == (a × (p × r) + a × (- (q × s))) { mul_neg_r }
      == (a × (p × r + - (q × s))) { distl }.
  Qed.

  Lemma cliff1_ga_mul_scale_im : ∀ a p q r s : F,
    p × (a × s) + q × (a × r) == a × (p × s + q × r).
  Proof.
    intros a p q r s.
    calc (p × (a × s) + q × (a × r))
      == (a × (p × s) + q × (a × r)) { (mul_comm_swap p a s) }
      == (a × (p × s) + a × (q × r)) { (mul_comm_swap q a r) }
      == (a × (p × s + q × r)) { distl }.
  Qed.

  Lemma cliff1_ga_scale_mul_r : ∀ (a : F) (x y : C F),
    x × Cliff1_Scale a y == Cliff1_Scale a (x × y).
  Proof.
    intros a x y.
    split.
    - exact (cliff1_ga_mul_scale_re a (re x) (im x) (re y) (im y)).
    - exact (cliff1_ga_mul_scale_im a (re x) (im x) (re y) (im y)).
  Qed.

  Definition GA_Cliff1 : GeometricAlgebra F F (C F) :=
    @Build_GeometricAlgebra F F (C F) _ _ Cliff1_NegInner C_Ring Cliff1_Scale
      Cliff1_ScaleLaws cliff1_ga_scale_mul_l cliff1_ga_scale_mul_r
      cliff1_ga_vec cliff1_ga_vec_proper cliff1_ga_vec_add
      cliff1_ga_vec_scale cliff1_ga_contract.

End Cliff1GA.

Example ga_cliff1_check (F : Type) (FF : Field F) :
  GeometricAlgebra F F (C F) := GA_Cliff1.
