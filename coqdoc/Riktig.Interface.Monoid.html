Library Riktig.Interface.Monoid

From Coq Require Export Morphisms.
From Riktig Require Export Ops Props.

Class Monoid T := {
  #[global] mon_equ :: EqRel T;
  #[global] mon_equiv :: Equivalence mon_equ;
  #[global] mon_zero :: Zero T;
  #[global] mon_add :: HasAdd T;
  mon_assoc : ∀ a b c : T, (a + b) + c == a + (b + c);
  mon_idl : ∀ a : T, 0 + a == a;
  mon_idr : ∀ a : T, a + 0 == a;
  #[global] mon_add_proper :: Proper (equ ==> equ ==> equ) (@add T _)
}.

#[global] Instance Monoid_Assoc T (M : Monoid T) : Assoc T equ add := mon_assoc.
#[global] Instance Monoid_Idl T (M : Monoid T) : Idl T equ add zero := mon_idl.
#[global] Instance Monoid_Idr T (M : Monoid T) : Idr T equ add zero := mon_idr.

#[global] Hint Mode Monoid ! : typeclass_instances.
