Section KerPred.
  Context {F U V : Type} `{VectorSpace F U} `{VV : !VectorSpace F V}.
  Context (f : LinearMap F U V).

  Definition ker_pred (u : U) : Prop := f u == 0.

  Lemma ker_zero : ker_pred 0.
  Proof. exact (lm_zero f). Qed.

  Lemma ker_add : ∀ u v : U, ker_pred u → ker_pred v → ker_pred (u + v).
  Proof.
    intros u v Hu Hv.
    unfold ker_pred in Hu, Hv |- ×.
    rewrite (lm_add f u v), Hu, Hv.
    apply mon_idl.
  Qed.

  Lemma ker_scale : ∀ (a : F) (u : U), ker_pred u → ker_pred (a *: u).
  Proof.
    intros a u Hu.
    unfold ker_pred in Hu |- ×.
    rewrite (lm_scale f a u), Hu.
    apply scale_zero.
  Qed.

  #[global] Instance Ker_SubspacePred : SubspacePred F U ker_pred := {|
    sp_zero := ker_zero;
    sp_add := ker_add;
    sp_scale := ker_scale
  |}.

End KerPred.

Definition Ker {F U V : Type} `{VectorSpace F U} `{VV : !VectorSpace F V}
  (f : LinearMap F U V) := Sub U (ker_pred f).


Section ImPred.
  Context {F U V : Type} `{VectorSpace F U} `{VV : !VectorSpace F V}.
  Context (f : LinearMap F U V).

  Definition im_pred (v : V) : Prop := ∃ u : U, f u == v.

  Lemma im_zero : im_pred 0.
  Proof. ∃ 0. exact (lm_zero f). Qed.

  Lemma im_add : ∀ v1 v2 : V, im_pred v1 → im_pred v2 → im_pred (v1 + v2).
  Proof.
    intros v1 v2 Hv1 Hv2.
    destruct Hv1 as [u1 Hu1].
    destruct Hv2 as [u2 Hu2].
    ∃ (u1 + u2).
    rewrite (lm_add f u1 u2), Hu1, Hu2.
    reflexivity.
  Qed.

  Lemma im_scale : ∀ (a : F) (v : V), im_pred v → im_pred (a *: v).
  Proof.
    intros a v Hv.
    destruct Hv as [u Hu].
    ∃ (a *: u).
    rewrite (lm_scale f a u), Hu.
    reflexivity.
  Qed.

  #[global] Instance Im_SubspacePred : SubspacePred F V im_pred := {|
    sp_zero := im_zero;
    sp_add := im_add;
    sp_scale := im_scale
  |}.

End ImPred.

Definition Im {F U V : Type} `{VectorSpace F U} `{VV : !VectorSpace F V}
  (f : LinearMap F U V) := Sub V (im_pred f).

Definition subspace_has_basis_statement : Prop :=
  ∀ (F V : Type) (FF : Field F) (VV : VectorSpace F V) (n : nat)
    (Bn : Basis n F V) (P : V → Prop) (SP : SubspacePred F V P),
    ∃ k : nat, k ≤ n ∧ ∃ Bk : Basis k F (Sub V P), True.

Definition rank_nullity_statement : Prop :=
  ∀ (F U V : Type) (FF : Field F)
    (VU : VectorSpace F U) (VV : VectorSpace F V)
    (f : LinearMap F U V) (n : nat) (Bn : Basis n F U)
    (k : nat) (Bk : Basis k F (Ker f)) (r : nat) (Br : Basis r F (Im f)),
    n = Nat.add k r.
