Library Riktig.Interface.VectorSpace

From Coq Require Import Morphisms.
From Riktig Require Import Ops Interface.AbGroup Interface.Field.

Class ScaleLaws (F V : Type) `{Field F} `{AbGroup V} `{HasScale F V} := {
  #[global] sc_proper :: Proper (equ ==> equ ==> equ) (@scale F V _);
  sc_scale_add : ∀ (a : F) (u v : V), a *: (u + v) == a *: u + a *: v;
  sc_add_scale : ∀ (a b : F) (u : V), (a + b) *: u == a *: u + b *: u;
  sc_scale_scale : ∀ (a b : F) (u : V), (a × b) *: u == a *: (b *: u);
  sc_scale_1 : ∀ u : V, 1 *: u == u
}.

Class VectorSpace (F V : Type) `{Field F} := {
  #[global] vs_ab :: AbGroup V;
  #[global] vs_scale :: HasScale F V;
  #[global] vs_laws :: ScaleLaws F V
}.

#[global] Hint Mode ScaleLaws - ! - - - : typeclass_instances.
#[global] Hint Mode VectorSpace - ! - : typeclass_instances.
