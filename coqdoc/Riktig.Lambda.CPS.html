Library Riktig.Lambda.CPS


Require Import Riktig.Lambda.Syntax.



Fixpoint cps (t : tm) : tm :=
  match t with
  | Var n ⇒ Abs (App (Var 0) (Var (S n)))
  | Abs b ⇒ Abs (App (Var 0) (Abs (lift 1 1 (cps b))))
  | App f a ⇒
      Abs (App (lift 1 0 (cps f))
               (Abs (App (lift 2 0 (cps a))
                         (Abs (App (App (Var 1) (Var 0)) (Var 2))))))
  end.

Definition psi (v : tm) : tm :=
  match v with
  | Abs b ⇒ Abs (cps b)
  | other ⇒ other
  end.

Lemma psi_value : ∀ v, value v → value (psi v).
Proof.
  intros v Hv. destruct Hv; constructor.
Qed.


Lemma cps_var_beta : ∀ c n,
  subst 0 c (App (Var 0) (Var (S n))) = App c (Var n).
Proof.
  intros c n. simpl. rewrite lift_0. reflexivity.
Qed.

Lemma cps_abs_beta : ∀ c b,
  subst 0 c (App (Var 0) (Abs (lift 1 1 (cps b)))) = App c (Abs (cps b)).
Proof.
  intros c b. simpl. rewrite lift_0. rewrite subst_lift. reflexivity.
Qed.

Lemma cps_value_step : ∀ v c, value v → value c →
  step (App (cps v) c) (App c (psi v)).
Proof.
  intros v c Hv Hc. destruct Hv as [n | b]; simpl.
  - rewrite <- (cps_var_beta c n). apply step_beta. exact Hc.
  - rewrite <- (cps_abs_beta c b). apply step_beta. exact Hc.
Qed.

Lemma cps_value_id : ∀ v, value v →
  star (App (cps v) idt) (psi v).
Proof.
  intros v Hv.
  apply star_step with (App idt (psi v)).
  - apply cps_value_step.
    + exact Hv.
    + constructor.
  - apply star_step with (subst 0 (psi v) (Var 0)).
    + apply step_beta. apply psi_value. exact Hv.
    + simpl. rewrite lift_0. constructor.
Qed.


Example cps_id_id :
  eval 100 (App (cps (App idt idt)) idt) = psi (eval 10 (App idt idt)).
Proof. reflexivity. Qed.

Example cps_id_id_shape :
  eval 100 (App (cps (App idt idt)) idt) = Abs (Abs (App (Var 0) (Var 1))).
Proof. reflexivity. Qed.

Example cps_two_apps :
  eval 100 (App (cps (App idt (App idt k))) idt)
  = psi (eval 10 (App idt (App idt k))).
Proof. reflexivity. Qed.

Example cps_two_apps_shape :
  eval 100 (App (cps (App idt (App idt k))) idt) = psi k.
Proof. reflexivity. Qed.
