Library Riktig.Impl.Attention

From Coq Require Import Arith NArith QArith.
From Coq.Vectors Require Fin.
From Riktig Require Import Ops Interface.Field Impl.Matrix
  Theory.MatVec Impl.Qf.


Section LinearAttention.
  Context {F : Type} {FF : Field F}.
  Context (n d : nat).
  Context (Q K V : Matrix n d F).

  Definition attn_naive : Matrix n d F := mat_mul (mat_mul Q (mat_tr K)) V.

  Definition attn_fast : Matrix n d F := mat_mul Q (mat_mul (mat_tr K) V).

  Theorem attn_assoc : attn_naive == attn_fast.
  Proof.
    exact (mat_mul_assoc n d n d Q (mat_tr K) V).
  Qed.

  Theorem attn_pipeline : ∀ w : Fin.t d → F,
    mat_vec attn_naive w == mat_vec Q (mat_vec (mat_tr K) (mat_vec V w)).
  Proof.
    intros w i.
    transitivity (mat_vec (mat_mul Q (mat_tr K)) (mat_vec V w) i).
    - exact (mat_vec_mul n n d (mat_mul Q (mat_tr K)) V w i).
    - exact (mat_vec_mul n d n Q (mat_tr K) (mat_vec V w) i).
  Qed.

End LinearAttention.


Definition mulcost (m n p : nat) : nat := (m × n × p)%nat.

Definition cost_naive (n d : nat) : nat :=
  (mulcost n d n + mulcost n n d)%nat.

Definition cost_fast (n d : nat) : nat :=
  (mulcost d n d + mulcost n d d)%nat.

Lemma cost_fast_le : ∀ n d : nat,
  (d ≤ n)%nat → (cost_fast n d ≤ cost_naive n d)%nat.
Proof.
  intros n d Hdn.
  unfold cost_fast, cost_naive, mulcost.
  apply Nat.add_le_mono.
  -
    rewrite (Nat.mul_comm d n).
    apply Nat.mul_le_mono_l.
    exact Hdn.
  -
    apply Nat.mul_le_mono_r.
    apply Nat.mul_le_mono_l.
    exact Hdn.
Qed.

Lemma intermediate_smaller : ∀ n d : nat,
  (d ≤ n)%nat → (d × d ≤ n × n)%nat.
Proof.
  intros n d Hdn.
  exact (Nat.mul_le_mono d n d n Hdn Hdn).
Qed.


Lemma cost_naive_toN : ∀ n d : nat,
  N.of_nat (cost_naive n d)
    = (N.of_nat n × N.of_nat d × N.of_nat n
       + N.of_nat n × N.of_nat n × N.of_nat d)%N.
Proof.
  intros n d.
  unfold cost_naive, mulcost.
  rewrite Nat2N.inj_add,
    (Nat2N.inj_mul (n × d)%nat n), (Nat2N.inj_mul n d),
    (Nat2N.inj_mul (n × n)%nat d), (Nat2N.inj_mul n n).
  reflexivity.
Qed.

Lemma cost_fast_toN : ∀ n d : nat,
  N.of_nat (cost_fast n d)
    = (N.of_nat d × N.of_nat n × N.of_nat d
       + N.of_nat n × N.of_nat d × N.of_nat d)%N.
Proof.
  intros n d.
  unfold cost_fast, mulcost.
  rewrite Nat2N.inj_add,
    (Nat2N.inj_mul (d × n)%nat d), (Nat2N.inj_mul d n),
    (Nat2N.inj_mul (n × d)%nat d), (Nat2N.inj_mul n d).
  reflexivity.
Qed.

Example cost_naive_1024_64 : N.of_nat (cost_naive 1024 64) = 134217728%N.
Proof.
  rewrite (cost_naive_toN 1024 64).
  reflexivity.
Qed.

Example cost_fast_1024_64 : N.of_nat (cost_fast 1024 64) = 8388608%N.
Proof.
  rewrite (cost_fast_toN 1024 64).
  reflexivity.
Qed.

Example cost_naive_16_4 : cost_naive 16 4 = 2048%nat.
Proof. reflexivity. Qed.

Example cost_fast_16_4 : cost_fast 16 4 = 512%nat.
Proof. reflexivity. Qed.


Definition row2 {A : Type} (a b : A) : Fin.t 2 → A :=
  fun j ⇒ match j with
           | Fin.F1 ⇒ a
           | Fin.FS _ ⇒ b
           end.

Definition mat32 {A : Type} (a11 a12 a21 a22 a31 a32 : A) : Matrix 3 2 A :=
  fun i ⇒ match i with
           | Fin.F1 ⇒ row2 a11 a12
           | Fin.FS Fin.F1 ⇒ row2 a21 a22
           | Fin.FS (Fin.FS _) ⇒ row2 a31 a32
           end.

Definition Qm3 : Matrix 3 2 Q :=
  mat32 (1#1)%Q (2#1)%Q (0#1)%Q (1#1)%Q (1#1)%Q ((-1)#1)%Q.

Definition Km3 : Matrix 3 2 Q :=
  mat32 (1#1)%Q (0#1)%Q (2#1)%Q (1#1)%Q (0#1)%Q (3#1)%Q.

Definition Vm3 : Matrix 3 2 Q :=
  mat32 (1#1)%Q (1#1)%Q (2#1)%Q (0#1)%Q (0#1)%Q (1#1)%Q.

Example attn_naive_Q_compute :
     attn_naive 3 2 Qm3 Km3 Vm3 Fin.F1 Fin.F1 == (9#1)%Q
  ∧ attn_naive 3 2 Qm3 Km3 Vm3 Fin.F1 (Fin.FS Fin.F1) == (7#1)%Q
  ∧ attn_naive 3 2 Qm3 Km3 Vm3 (Fin.FS Fin.F1) Fin.F1 == (2#1)%Q
  ∧ attn_naive 3 2 Qm3 Km3 Vm3 (Fin.FS Fin.F1) (Fin.FS Fin.F1) == (3#1)%Q
  ∧ attn_naive 3 2 Qm3 Km3 Vm3 (Fin.FS (Fin.FS Fin.F1)) Fin.F1 == (3#1)%Q
  ∧ attn_naive 3 2 Qm3 Km3 Vm3 (Fin.FS (Fin.FS Fin.F1)) (Fin.FS Fin.F1)
       == ((-2)#1)%Q.
Proof.
  repeat split; reflexivity.
Qed.

Example attn_fast_Q_compute :
     attn_fast 3 2 Qm3 Km3 Vm3 Fin.F1 Fin.F1 == (9#1)%Q
  ∧ attn_fast 3 2 Qm3 Km3 Vm3 Fin.F1 (Fin.FS Fin.F1) == (7#1)%Q
  ∧ attn_fast 3 2 Qm3 Km3 Vm3 (Fin.FS Fin.F1) Fin.F1 == (2#1)%Q
  ∧ attn_fast 3 2 Qm3 Km3 Vm3 (Fin.FS Fin.F1) (Fin.FS Fin.F1) == (3#1)%Q
  ∧ attn_fast 3 2 Qm3 Km3 Vm3 (Fin.FS (Fin.FS Fin.F1)) Fin.F1 == (3#1)%Q
  ∧ attn_fast 3 2 Qm3 Km3 Vm3 (Fin.FS (Fin.FS Fin.F1)) (Fin.FS Fin.F1)
       == ((-2)#1)%Q.
Proof.
  repeat split; reflexivity.
Qed.

Example attn_assoc_Q : attn_naive 3 2 Qm3 Km3 Vm3 == attn_fast 3 2 Qm3 Km3 Vm3.
Proof.
  exact (attn_assoc 3 2 Qm3 Km3 Vm3).
Qed.

Definition w2 : Fin.t 2 → Q := row2 (1#1)%Q (2#1)%Q.

Example attn_pipeline_Q_compute :
     mat_vec Qm3 (mat_vec (mat_tr Km3) (mat_vec Vm3 w2)) Fin.F1 == (23#1)%Q
  ∧ mat_vec Qm3 (mat_vec (mat_tr Km3) (mat_vec Vm3 w2)) (Fin.FS Fin.F1)
       == (8#1)%Q
  ∧ mat_vec Qm3 (mat_vec (mat_tr Km3) (mat_vec Vm3 w2))
       (Fin.FS (Fin.FS Fin.F1)) == ((-1)#1)%Q
  ∧ mat_vec (attn_naive 3 2 Qm3 Km3 Vm3) w2 Fin.F1 == (23#1)%Q
  ∧ mat_vec (attn_naive 3 2 Qm3 Km3 Vm3) w2 (Fin.FS Fin.F1) == (8#1)%Q
  ∧ mat_vec (attn_naive 3 2 Qm3 Km3 Vm3) w2 (Fin.FS (Fin.FS Fin.F1))
       == ((-1)#1)%Q.
Proof.
  repeat split; reflexivity.
Qed.

Example attn_pipeline_Q :
  mat_vec (attn_naive 3 2 Qm3 Km3 Vm3) w2
    == mat_vec Qm3 (mat_vec (mat_tr Km3) (mat_vec Vm3 w2)).
Proof.
  exact (attn_pipeline 3 2 Qm3 Km3 Vm3 w2).
Qed.
