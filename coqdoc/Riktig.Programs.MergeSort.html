Fixpoint split {A} (l : list A) : list A × list A :=
  match l with
  | [] ⇒ ([], [])
  | [x] ⇒ ([x], [])
  | x :: y :: rest ⇒
      let '(l1, l2) := split rest in
      (x :: l1, y :: l2)
  end.

Lemma split_length_le_bounded {A} (n : nat) :
  ∀ l : list A, length l ≤ n →
    length (fst (split l)) ≤ length l ∧ length (snd (split l)) ≤ length l.
Proof.
  induction n as [|n IH]; intros l Hlen.
  - destruct l; simpl in *; lia.
  - destruct l as [|x [|y rest]].
    + simpl; lia.
    + simpl; lia.
    + simpl in Hlen |- ×.
      destruct (split rest) as [l1 l2] eqn:E.
      simpl in ×.
      assert (Hle : length rest ≤ n) by lia.
      specialize (IH rest Hle). rewrite E in IH. simpl in IH.
      lia.
Qed.

Lemma split_length_le {A} (l : list A) :
  length (fst (split l)) ≤ length l ∧ length (snd (split l)) ≤ length l.
Proof. apply (split_length_le_bounded (length l)); lia. Qed.

Lemma split_length_lt {A} (x y : A) (rest : list A) :
  length (fst (split (x :: y :: rest))) < length (x :: y :: rest) ∧
  length (snd (split (x :: y :: rest))) < length (x :: y :: rest).
Proof.
  simpl. destruct (split rest) as [l1 l2] eqn:E.
  pose proof (split_length_le rest) as [H1 H2].
  rewrite E in H1, H2. simpl in ×. lia.
Qed.


Fixpoint merge (l1 l2 : list nat) : list nat :=
  match l1 with
  | [] ⇒ l2
  | x :: xs ⇒
      let fix merge_r (l2 : list nat) : list nat :=
        match l2 with
        | [] ⇒ l1
        | y :: ys ⇒
            if dec (Nat.leb x y = true)
            then x :: merge xs l2
            else y :: merge_r ys
        end
      in merge_r l2
  end.


Definition msort : list nat → list nat.
apply (Fix (well_founded_ltof (list nat) (@length nat)) (fun _ ⇒ list nat)).
intros l rec.
destruct l as [|x [|y rest]].
- exact [].
- exact [x].
- refine (merge (rec (fst (split (x :: y :: rest))) _)
                (rec (snd (split (x :: y :: rest))) _));
  abstract (unfold ltof; apply (split_length_lt x y rest)).
Defined.

Lemma msort_step l :
  msort l =
    match l with
    | [] ⇒ l
    | [_] ⇒ l
    | x :: y :: rest ⇒
        merge (msort (fst (split (x :: y :: rest))))
              (msort (snd (split (x :: y :: rest))))
    end.
Proof.
  unfold msort at 1. rewrite Fix_eq.
  - destruct l as [|x [|y rest]]; reflexivity.
  - intros u f g Hfg.
    destruct u as [|x [|y rest]]; try reflexivity.
    rewrite !Hfg. reflexivity.
Qed.


Section TestMsort.
Import ListNotations.

Check Logic.eq_refl : split [1;2;3;4;5;6] = ([1;3;5],[2;4;6]).
Check Logic.eq_refl : merge [1;3;5] [2;4;6] = [1;2;3;4;5;6].
Check Logic.eq_refl : msort [5;3;8;1;9;2;7] = [1;2;3;5;7;8;9].
Check Logic.eq_refl : msort ([] : list nat) = [].
Check Logic.eq_refl : msort [1] = [1].
Check Logic.eq_refl : msort [2;1] = [1;2].

End TestMsort.
