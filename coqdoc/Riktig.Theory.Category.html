From Coq Require Import RelationClasses.
From Riktig Require Import Ops Interface.Category.

Section CategoryTheory.
  Context {O : Type} `{Category O}.

  Record Iso (x y : O) := {
    iso_to : x ~> y;
    iso_from : y ~> x;
    iso_fromto : iso_from ∘ iso_to == cat_id;
    iso_tofrom : iso_to ∘ iso_from == cat_id
  }.

  Global Arguments iso_to {x y}.
  Global Arguments iso_from {x y}.
  Global Arguments iso_fromto {x y}.
  Global Arguments iso_tofrom {x y}.

  Definition iso_refl (x : O) : Iso x x := {|
    iso_to := cat_id;
    iso_from := cat_id;
    iso_fromto := cat_idl x x cat_id;
    iso_tofrom := cat_idl x x cat_id
  |}.

  Definition iso_sym {x y : O} (f : Iso x y) : Iso y x := {|
    iso_to := iso_from f;
    iso_from := iso_to f;
    iso_fromto := iso_tofrom f;
    iso_tofrom := iso_fromto f
  |}.

  Lemma comp_cancel :
    ∀ (x y z : O) (a : y ~> z) (b : x ~> y) (b' : y ~> x) (a' : z ~> y),
      b ∘ b' == cat_id → a ∘ a' == cat_id → (a ∘ b) ∘ (b' ∘ a') == cat_id.
  Proof.
    intros x y z a b b' a' Hb Ha.
    rewrite <- cat_assoc, (cat_assoc _ _ _ _ b b' a'), Hb, cat_idl.
    exact Ha.
  Qed.

  Definition iso_trans {x y z : O} (f : Iso x y) (g : Iso y z) : Iso x z := {|
    iso_to := iso_to g ∘ iso_to f;
    iso_from := iso_from f ∘ iso_from g;
    iso_fromto := comp_cancel _ _ _ _ _ _ _ (iso_fromto g) (iso_fromto f);
    iso_tofrom := comp_cancel _ _ _ _ _ _ _ (iso_tofrom f) (iso_tofrom g)
  |}.

  Definition Isomorphic (x y : O) : Prop := inhabited (Iso x y).

  Global Instance Isomorphic_Equivalence : Equivalence Isomorphic.
  Proof.
    split.
    - intros x; exact (inhabits (iso_refl x)).
    - intros x y [f]; exact (inhabits (iso_sym f)).
    - intros x y z [f] [g]; exact (inhabits (iso_trans f g)).
  Qed.

  Theorem id_unique_l :
    ∀ (x : O) (e : x ~> x), (∀ y (f : x ~> y), f ∘ e == f) → e == cat_id.
  Proof.
    intros x e Hu.
    rewrite <- (cat_idl x x e).
    exact (Hu x cat_id).
  Qed.

End CategoryTheory.
