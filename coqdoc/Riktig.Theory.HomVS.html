Section HomVS.
  Context {F U V : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}.

  Local Notation Hom := (LinearMap F U V).


  #[global] Instance Hom_EqRel : EqRel Hom := lm_equ.

  #[global] Instance Hom_Equivalence : Equivalence (@equ Hom Hom_EqRel).
  Proof.
    apply Build_Equivalence.
    - intros f u. reflexivity.
    - intros f g Hfg u. symmetry. apply Hfg.
    - intros f g h Hfg Hgh u. rewrite (Hfg u). apply Hgh.
  Qed.


  #[global] Instance Hom_Zero : Zero Hom := @lm_const0 F U V FF VU VV.


  Section HomPlus.
    Context (f g : Hom).

    #[global] Instance hom_plus_proper : Proper (equ ==> equ) (fun u : U ⇒ f u + g u).
    Proof.
      intros u v Huv.
      rewrite (lm_proper f _ _ Huv), (lm_proper g _ _ Huv).
      reflexivity.
    Qed.

    Lemma hom_plus_add : ∀ u v : U,
      f (u + v) + g (u + v) == (f u + g u) + (f v + g v).
    Proof.
      intros u v.
      rewrite (lm_add f u v), (lm_add g u v).
      apply add_interchange.
    Qed.

    Lemma hom_plus_scale : ∀ (a : F) (u : U),
      f (a *: u) + g (a *: u) == a *: (f u + g u).
    Proof.
      intros a u.
      rewrite (lm_scale f a u), (lm_scale g a u).
      symmetry.
      apply sc_scale_add.
    Qed.

    Definition hom_plus : Hom :=
      @Build_LinearMap F U V FF VU VV (fun u : U ⇒ f u + g u)
        hom_plus_proper hom_plus_add hom_plus_scale.

  End HomPlus.

  #[global] Instance Hom_Add : HasAdd Hom := hom_plus.


  Section HomOpp.
    Context (f : Hom).

    #[global] Instance hom_opp_proper : Proper (equ ==> equ) (fun u : U ⇒ - f u).
    Proof.
      intros u v Huv.
      rewrite (lm_proper f _ _ Huv).
      reflexivity.
    Qed.

    Lemma hom_opp_add : ∀ u v : U, - f (u + v) == - f u + - f v.
    Proof.
      intros u v.
      rewrite (lm_add f u v).
      apply neg_add.
    Qed.

    Lemma hom_opp_scale : ∀ (a : F) (u : U), - f (a *: u) == a *: - f u.
    Proof.
      intros a u.
      rewrite (lm_scale f a u).
      symmetry.
      apply scale_neg.
    Qed.

    Definition hom_opp : Hom :=
      @Build_LinearMap F U V FF VU VV (fun u : U ⇒ - f u)
        hom_opp_proper hom_opp_add hom_opp_scale.

  End HomOpp.

  #[global] Instance Hom_Neg : HasNeg Hom := hom_opp.


  Section HomSmul.
    Context (a : F) (f : Hom).

    #[global] Instance hom_smul_proper : Proper (equ ==> equ) (fun u : U ⇒ a *: f u).
    Proof.
      intros u v Huv.
      rewrite (lm_proper f _ _ Huv).
      reflexivity.
    Qed.

    Lemma hom_smul_add : ∀ u v : U,
      a *: f (u + v) == a *: f u + a *: f v.
    Proof.
      intros u v.
      rewrite (lm_add f u v).
      apply sc_scale_add.
    Qed.

    Lemma hom_smul_scale : ∀ (b : F) (u : U),
      a *: f (b *: u) == b *: (a *: f u).
    Proof.
      intros b u.
      rewrite (lm_scale f b u).
      apply scale_swap.
    Qed.

    Definition hom_smul : Hom :=
      @Build_LinearMap F U V FF VU VV (fun u : U ⇒ a *: f u)
        hom_smul_proper hom_smul_add hom_smul_scale.

  End HomSmul.

  #[global] Instance Hom_Scale : HasScale F Hom := hom_smul.



  Lemma hom_assoc : ∀ f g h : Hom, (f + g) + h == f + (g + h).
  Proof.
    intros f g h u.
    apply monoid_associative.
  Qed.

  Lemma hom_idl : ∀ f : Hom, 0 + f == f.
  Proof.
    intros f u.
    apply monoid_left_identity.
  Qed.

  Lemma hom_idr : ∀ f : Hom, f + 0 == f.
  Proof.
    intros f u.
    apply monoid_right_identity.
  Qed.

  #[global] Instance hom_add_proper : Proper (equ ==> equ ==> equ) (@add Hom Hom_Add).
  Proof.
    intros f f' Hf g g' Hg u.
    apply monoid_add_proper.
    - apply Hf.
    - apply Hg.
  Qed.

  #[global] Instance Hom_Monoid : Monoid Hom := {|
    monoid_eq := Hom_EqRel;
    monoid_equivalence := Hom_Equivalence;
    monoid_zero := Hom_Zero;
    monoid_add := Hom_Add;
    monoid_associative := hom_assoc;
    monoid_left_identity := hom_idl;
    monoid_right_identity := hom_idr;
    monoid_add_proper := hom_add_proper
  |}.


  Lemma hom_invl : ∀ f : Hom, (- f) + f == 0.
  Proof.
    intros f u.
    apply group_left_inverse.
  Qed.

  Lemma hom_invr : ∀ f : Hom, f + (- f) == 0.
  Proof.
    intros f u.
    apply group_right_inverse.
  Qed.

  #[global] Instance hom_neg_proper : Proper (equ ==> equ) (@neg Hom Hom_Neg).
  Proof.
    intros f f' Hf u.
    apply group_neg_proper, Hf.
  Qed.

  #[global] Instance Hom_Group : Group Hom := {|
    group_monoid := Hom_Monoid;
    group_neg := Hom_Neg;
    group_left_inverse := hom_invl;
    group_right_inverse := hom_invr;
    group_neg_proper := hom_neg_proper
  |}.


  Lemma hom_comm : ∀ f g : Hom, f + g == g + f.
  Proof.
    intros f g u.
    apply ab_comm.
  Qed.

  #[global] Instance Hom_AbGroup : AbGroup Hom := {|
    ab_group := Hom_Group;
    ab_comm := hom_comm
  |}.


  #[global] Instance hom_sc_proper : Proper (equ ==> equ ==> equ) (@scale F Hom Hom_Scale).
  Proof.
    intros a a' Ha f f' Hf u.
    apply sc_proper.
    - exact Ha.
    - apply Hf.
  Qed.

  Lemma hom_scale_add : ∀ (a : F) (f g : Hom),
    a *: (f + g) == a *: f + a *: g.
  Proof.
    intros a f g u.
    apply sc_scale_add.
  Qed.

  Lemma hom_add_scale : ∀ (a b : F) (f : Hom),
    (a + b) *: f == a *: f + b *: f.
  Proof.
    intros a b f u.
    apply sc_add_scale.
  Qed.

  Lemma hom_scale_scale : ∀ (a b : F) (f : Hom),
    (a × b) *: f == a *: (b *: f).
  Proof.
    intros a b f u.
    apply sc_scale_scale.
  Qed.

  Lemma hom_scale_1 : ∀ f : Hom, 1 *: f == f.
  Proof.
    intros f u.
    apply sc_scale_1.
  Qed.

  #[global] Instance Hom_ScaleLaws : ScaleLaws F Hom :=
    @Build_ScaleLaws F Hom FF Hom_AbGroup Hom_Scale
      hom_sc_proper hom_scale_add hom_add_scale hom_scale_scale hom_scale_1.

  #[global] Instance Hom_VectorSpace : VectorSpace F Hom :=
    @Build_VectorSpace F Hom FF Hom_AbGroup Hom_Scale Hom_ScaleLaws.

End HomVS.


Section LmCompAssoc.
  Context {F P Q R S : Type} {FF : Field F}.
  Context {VP : VectorSpace F P} {VQ : VectorSpace F Q}
    {VR : VectorSpace F R} {VSp : VectorSpace F S}.

  Theorem lm_comp_assoc : ∀ (g : LinearMap F R S) (h : LinearMap F Q R)
      (k : LinearMap F P Q),
    lm_comp (lm_comp g h) k == lm_comp g (lm_comp h k).
  Proof.
    intros g h k u.
    reflexivity.
  Qed.

End LmCompAssoc.

Section LmCompAssoc4.
  Context {F U V W X Y : Type} {FF : Field F}.
  Context {VU : VectorSpace F U} {VV : VectorSpace F V}
    {VW : VectorSpace F W} {VX : VectorSpace F X} {VY : VectorSpace F Y}.

  Theorem lm_comp_assoc4 : ∀ (L1 : LinearMap F U V)
      (L2 : LinearMap F V W) (L3 : LinearMap F W X) (L4 : LinearMap F X Y),
    lm_comp (lm_comp (lm_comp L4 L3) L2) L1
      == lm_comp L4 (lm_comp L3 (lm_comp L2 L1)).
  Proof.
    intros L1 L2 L3 L4.
    calc (lm_comp (lm_comp (lm_comp L4 L3) L2) L1)
      == (lm_comp (lm_comp L4 L3) (lm_comp L2 L1))
      == (lm_comp L4 (lm_comp L3 (lm_comp L2 L1))).
    - exact (lm_comp_assoc (lm_comp L4 L3) L2 L1).
    - exact (lm_comp_assoc L4 L3 (lm_comp L2 L1)).
  Qed.

End LmCompAssoc4.


Section EndRing.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V}.

  Local Notation EndV := (LinearMap F V V).

  #[global] Instance End_One : One EndV := lm_id.

  #[global] Instance End_Mult : HasMult EndV :=
    fun f g : EndV ⇒ lm_comp f g.


  Lemma end_mul_assoc : ∀ f g h : EndV, (f × g) × h == f × (g × h).
  Proof.
    intros f g h u.
    reflexivity.
  Qed.

  Lemma end_mul_1l : ∀ f : EndV, 1 × f == f.
  Proof.
    intros f u.
    reflexivity.
  Qed.

  Lemma end_mul_1r : ∀ f : EndV, f × 1 == f.
  Proof.
    intros f u.
    reflexivity.
  Qed.


  Lemma end_distl : ∀ f g h : EndV, f × (g + h) == f × g + f × h.
  Proof.
    intros f g h u.
    apply (lm_add f).
  Qed.

  Lemma end_distr : ∀ f g h : EndV, (f + g) × h == f × h + g × h.
  Proof.
    intros f g h u.
    reflexivity.
  Qed.

  #[global] Instance end_mul_proper : Proper (equ ==> equ ==> equ) (@mult EndV End_Mult).
  Proof.
    intros f f' Hf g g' Hg u.
    rewrite (lm_proper f _ _ (Hg u)).
    apply Hf.
  Qed.

  #[global] Instance End_Ring : Ring EndV := {|
    ring_ab := @Hom_AbGroup F V V FF VV VV;
    ring_one := End_One;
    ring_mul := End_Mult;
    ring_mul_assoc := end_mul_assoc;
    ring_mul_1l := end_mul_1l;
    ring_mul_1r := end_mul_1r;
    ring_distl := end_distl;
    ring_distr := end_distr;
    ring_mul_proper := end_mul_proper
  |}.

End EndRing.
