Library Riktig.Theory.Maxwell

From Coq Require Import Ring.
From Riktig Require Import Ops Props Interface.Field Theory.Ring Theory.Field
  Impl.Cliff13.

Section Maxwell.
  Context {F : Type} `{FF : Field F}.

  Add Ring c13_F_ring_maxwell : (ring_theory_of_Ring (R := F) field_mul_comm).

  Context (E1 E2 E3 B1 B2 B3 : F).
  Context (d0 d1 d2 d3 : F).

  Definition F_field : C13 F :=
    mkC13 0 0 0 0 0 E1 E2 E3 (- B3) B2 (- B1) 0 0 0 0 0.

  Definition Nabla : C13 F := c13_vec d0 d1 d2 d3.

  Theorem nabla_F_grade_decomposition :
    c13_s (Nabla × F_field) == 0 ∧
    c13_01 (Nabla × F_field) == 0 ∧
    c13_02 (Nabla × F_field) == 0 ∧
    c13_03 (Nabla × F_field) == 0 ∧
    c13_12 (Nabla × F_field) == 0 ∧
    c13_13 (Nabla × F_field) == 0 ∧
    c13_23 (Nabla × F_field) == 0 ∧
    c13_0123 (Nabla × F_field) == 0.
  Proof.
    unfold Nabla, F_field, c13_vec, C13_Mult, c13_mul.
    simpl. repeat split; ring.
  Qed.

  Theorem nabla_F_0 : c13_0 (Nabla × F_field) == d1 × E1 + (d2 × E2 + d3 × E3).
  Proof.
    unfold Nabla, F_field, c13_vec, C13_Mult, c13_mul. simpl. ring.
  Qed.

  Theorem nabla_F_1 : c13_1 (Nabla × F_field) == d0 × E1 + (- (d2 × B3) + d3 × B2).
  Proof.
    unfold Nabla, F_field, c13_vec, C13_Mult, c13_mul. simpl. ring.
  Qed.

  Theorem nabla_F_2 : c13_2 (Nabla × F_field) == d0 × E2 + (d1 × B3 + - (d3 × B1)).
  Proof.
    unfold Nabla, F_field, c13_vec, C13_Mult, c13_mul. simpl. ring.
  Qed.

  Theorem nabla_F_3 : c13_3 (Nabla × F_field) == d0 × E3 + (- (d1 × B2) + d2 × B1).
  Proof.
    unfold Nabla, F_field, c13_vec, C13_Mult, c13_mul. simpl. ring.
  Qed.

  Theorem nabla_F_123 : c13_123 (Nabla × F_field) == - (d1 × B1) + (- (d2 × B2) + - (d3 × B3)).
  Proof.
    unfold Nabla, F_field, c13_vec, C13_Mult, c13_mul. simpl. ring.
  Qed.

  Theorem nabla_F_012 : c13_012 (Nabla × F_field) == - (d0 × B3) + (- (d1 × E2) + d2 × E1).
  Proof.
    unfold Nabla, F_field, c13_vec, C13_Mult, c13_mul. simpl. ring.
  Qed.

  Theorem nabla_F_013 : c13_013 (Nabla × F_field) == d0 × B2 + (- (d1 × E3) + d3 × E1).
  Proof.
    unfold Nabla, F_field, c13_vec, C13_Mult, c13_mul. simpl. ring.
  Qed.

  Theorem nabla_F_023 : c13_023 (Nabla × F_field) == - (d0 × B1) + (- (d2 × E3) + d3 × E2).
  Proof.
    unfold Nabla, F_field, c13_vec, C13_Mult, c13_mul. simpl. ring.
  Qed.

  Definition J : C13 F :=
    c13_vec
      (d1 × E1 + (d2 × E2 + d3 × E3))
      (d0 × E1 + (- (d2 × B3) + d3 × B2))
      (d0 × E2 + (d1 × B3 + - (d3 × B1)))
      (d0 × E3 + (- (d1 × B2) + d2 × B1)).

  Theorem nabla_F_eq_J :
    c13_123 (Nabla × F_field) == 0 →
    c13_012 (Nabla × F_field) == 0 →
    c13_013 (Nabla × F_field) == 0 →
    c13_023 (Nabla × F_field) == 0 →
    Nabla × F_field == J.
  Proof.
    intros H123 H012 H013 H023.
    destruct nabla_F_grade_decomposition
      as (Hs & H01 & H02 & H03 & H12 & H13 & H23 & H0123).
    unfold J, c13_vec.
    repeat split; simpl.
    - exact Hs.
    - exact nabla_F_0.
    - exact nabla_F_1.
    - exact nabla_F_2.
    - exact nabla_F_3.
    - exact H01.
    - exact H02.
    - exact H03.
    - exact H12.
    - exact H13.
    - exact H23.
    - exact H012.
    - exact H013.
    - exact H023.
    - exact H123.
    - exact H0123.
  Qed.

End Maxwell.
