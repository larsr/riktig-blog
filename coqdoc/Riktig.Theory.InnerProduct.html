Library Riktig.Theory.InnerProduct

From Coq Require Import Morphisms.
From Riktig Require Import Ops Props Interface.Ring Interface.InnerProduct
  Theory.Group Theory.AbGroup.

Section InnerProductTheory.
  Context {F V : Type} `{InnerProduct F V}.

  Theorem norm2_spec : ∀ u : V, norm2 u == ⟨ u , u ⟩.
  Proof.
    intros u. reflexivity.
  Qed.

  Theorem ip_add_r : ∀ u v w : V,
    ⟨ u , v + w ⟩ == ⟨ u , v ⟩ + ⟨ u , w ⟩.
  Proof.
    intros u v w.
    rewrite (ip_sym u (v + w)), ip_add_l, (ip_sym v u), (ip_sym w u).
    reflexivity.
  Qed.

  Theorem ip_scale_r : ∀ (a : F) (u v : V),
    ⟨ u , a *: v ⟩ == a × ⟨ u , v ⟩.
  Proof.
    intros a u v.
    rewrite (ip_sym u (a *: v)), ip_scale_l, (ip_sym v u).
    reflexivity.
  Qed.

  Lemma inner_l_proper : ∀ v : V, Proper (equ ==> equ) (fun w : V ⇒ ⟨ w , v ⟩).
  Proof.
    intros v x y Hxy.
    rewrite Hxy.
    reflexivity.
  Qed.

  Theorem ip_zero_l : ∀ u : V, ⟨ (0 : V) , u ⟩ == 0.
  Proof.
    intros u.
    exact (hom_zero (fun w : V ⇒ ⟨ w , u ⟩) (inner_l_proper u)
             (fun x y ⇒ ip_add_l x y u)).
  Qed.

  Theorem ip_zero_r : ∀ u : V, ⟨ u , (0 : V) ⟩ == 0.
  Proof.
    intros u.
    rewrite (ip_sym u (0 : V)).
    apply ip_zero_l.
  Qed.

  Theorem ip_neg_l : ∀ u v : V, ⟨ - u , v ⟩ == - ⟨ u , v ⟩.
  Proof.
    intros u v.
    exact (hom_neg (fun w : V ⇒ ⟨ w , v ⟩) (inner_l_proper v)
             (fun x y ⇒ ip_add_l x y v) u).
  Qed.

  Theorem ip_neg_r : ∀ u v : V, ⟨ u , - v ⟩ == - ⟨ u , v ⟩.
  Proof.
    intros u v.
    rewrite (ip_sym u (- v)), ip_neg_l, (ip_sym v u).
    reflexivity.
  Qed.

  Theorem norm2_zero : norm2 (0 : V) == 0.
  Proof.
    rewrite norm2_spec, ip_zero_l.
    reflexivity.
  Qed.

  Theorem inner_expand : ∀ u v : V,
    ⟨ u + v , u + v ⟩ == (⟨ u , u ⟩ + ⟨ v , v ⟩) + (⟨ u , v ⟩ + ⟨ v , u ⟩).
  Proof.
    intros u v.
    rewrite ip_add_l, (ip_add_r u u v), (ip_add_r v u v), add_rotate.
    reflexivity.
  Qed.

  Theorem norm2_add : ∀ u v : V,
    norm2 (u + v) == (norm2 u + norm2 v) + (⟨ u , v ⟩ + ⟨ v , u ⟩).
  Proof.
    intros u v.
    rewrite norm2_spec, inner_expand, (norm2_spec u), (norm2_spec v).
    reflexivity.
  Qed.

  Theorem norm2_scale : ∀ (a : F) (u : V),
    norm2 (a *: u) == (a × a) × norm2 u.
  Proof.
    intros a u.
    rewrite norm2_spec, ip_scale_l, ip_scale_r, <- ring_mul_assoc,
      (norm2_spec u).
    reflexivity.
  Qed.

End InnerProductTheory.
