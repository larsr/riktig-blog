#[global] Instance PW_EqRel {X M : Type} `{EqRel M} : EqRel (X → M) :=
  fun f g ⇒ ∀ x, f x == g x.

#[global] Instance PW_Zero {X M : Type} `{Zero M} : Zero (X → M) :=
  fun _ ⇒ 0.

#[global] Instance PW_Add {X M : Type} `{HasAdd M} : HasAdd (X → M) :=
  fun f g x ⇒ f x + g x.

#[global] Instance PW_Neg {X M : Type} `{HasNeg M} : HasNeg (X → M) :=
  fun f x ⇒ - f x.

#[global] Instance PW_Scale {X F V : Type} `{HasScale F V} : HasScale F (X → V) :=
  fun a f x ⇒ a *: f x.

#[global] Instance PW_Equivalence {X M : Type} {e : EqRel M} {E : Equivalence e} :
  Equivalence (@PW_EqRel X M e).
Proof.
  apply Build_Equivalence.
  - intros f x. reflexivity.
  - intros f g Hfg x. symmetry. apply Hfg.
  - intros f g h Hfg Hgh x. rewrite (Hfg x). apply Hgh.
Qed.


Section PointwiseMonoid.
  Context {X M : Type} `{Monoid M}.

  Lemma pw_assoc : ∀ f g h : X → M, (f + g) + h == f + (g + h).
  Proof. intros f g h x. apply mon_assoc. Qed.

  Lemma pw_idl : ∀ f : X → M, 0 + f == f.
  Proof. intros f x. apply mon_idl. Qed.

  Lemma pw_idr : ∀ f : X → M, f + 0 == f.
  Proof. intros f x. apply mon_idr. Qed.

  Lemma pw_add_proper : Proper (equ ==> equ ==> equ) (@add (X → M) _).
  Proof.
    intros f f' Hf g g' Hg x.
    apply mon_add_proper; [apply Hf | apply Hg].
  Qed.

  #[global] Instance PW_Monoid : Monoid (X → M) := {|
    mon_equ := PW_EqRel;
    mon_zero := PW_Zero;
    mon_add := PW_Add;
    mon_equiv := PW_Equivalence;
    mon_assoc := pw_assoc;
    mon_idl := pw_idl;
    mon_idr := pw_idr;
    mon_add_proper := pw_add_proper
  |}.

End PointwiseMonoid.


Section PointwiseGroup.
  Context {X M : Type} `{Group M}.

  Lemma pw_invl : ∀ f : X → M, (- f) + f == 0.
  Proof. intros f x. apply grp_invl. Qed.

  Lemma pw_invr : ∀ f : X → M, f + (- f) == 0.
  Proof. intros f x. apply grp_invr. Qed.

  Lemma pw_neg_proper : Proper (equ ==> equ) (@neg (X → M) _).
  Proof. intros f f' Hf x. apply grp_neg_proper, Hf. Qed.

  #[global] Instance PW_Group : Group (X → M) := {|
    grp_monoid := PW_Monoid;
    grp_neg := PW_Neg;
    grp_invl := pw_invl;
    grp_invr := pw_invr;
    grp_neg_proper := pw_neg_proper
  |}.

End PointwiseGroup.


Section PointwiseAbGroup.
  Context {X M : Type} `{AbGroup M}.

  Lemma pw_comm : ∀ f g : X → M, f + g == g + f.
  Proof. intros f g x. apply ab_comm. Qed.

  #[global] Instance PW_AbGroup : AbGroup (X → M) := {|
    ab_group := PW_Group;
    ab_comm := pw_comm
  |}.

End PointwiseAbGroup.


Section PointwiseVectorSpace.
  Context {X F V : Type} `{VectorSpace F V}.

  Lemma pw_sc_proper : Proper (equ ==> equ ==> equ) (@scale F (X → V) _).
  Proof.
    intros a a' Ha f f' Hf x.
    apply sc_proper; [exact Ha | apply Hf].
  Qed.

  Lemma pw_scale_add : ∀ (a : F) (u v : X → V),
    a *: (u + v) == a *: u + a *: v.
  Proof. intros a u v x. apply sc_scale_add. Qed.

  Lemma pw_add_scale : ∀ (a b : F) (u : X → V),
    (a + b) *: u == a *: u + b *: u.
  Proof. intros a b u x. apply sc_add_scale. Qed.

  Lemma pw_scale_scale : ∀ (a b : F) (u : X → V),
    (a × b) *: u == a *: (b *: u).
  Proof. intros a b u x. apply sc_scale_scale. Qed.

  Lemma pw_scale_1 : ∀ u : X → V, 1 *: u == u.
  Proof. intros u x. apply sc_scale_1. Qed.

  #[global] Instance PW_ScaleLaws : ScaleLaws F (X → V) :=
    @Build_ScaleLaws F (X → V) _ PW_AbGroup PW_Scale
      pw_sc_proper pw_scale_add pw_add_scale pw_scale_scale pw_scale_1.

  #[global] Instance PW_VectorSpace : VectorSpace F (X → V) :=
    @Build_VectorSpace F (X → V) _ PW_AbGroup PW_Scale PW_ScaleLaws.

End PointwiseVectorSpace.
