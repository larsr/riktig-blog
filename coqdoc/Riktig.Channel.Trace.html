Library Riktig.Channel.Trace


From Coq Require Export List Permutation.
From Riktig.Theory Require Export Subseq.
Import ListNotations.

Section FaultClasses.
  Context {M : Type}.
  Implicit Types s r : list M.


  Definition perfect s r : Prop := r = s.

  Definition lossy s r : Prop := subseq r s.


  Inductive dup : list M → list M → Prop :=
    | dup_nil : dup [] []
    | dup_keep : ∀ x s r, dup s r → dup (x :: s) (x :: r)
    | dup_stutter : ∀ x s r, dup s (x :: r) → dup s (x :: x :: r).


  Inductive lossy_dup : list M → list M → Prop :=
    | lossy_dup_nil : lossy_dup [] []
    | lossy_dup_keep : ∀ x s r,
        lossy_dup s r → lossy_dup (x :: s) (x :: r)
    | lossy_dup_drop : ∀ x s r,
        lossy_dup s r → lossy_dup (x :: s) r
    | lossy_dup_stutter : ∀ x s r,
        lossy_dup s (x :: r) → lossy_dup s (x :: x :: r).


  Definition reorder s r : Prop := Permutation s r.


  Lemma dup_refl : ∀ s, dup s s.
  Proof.
    induction s as [| x s IH].
    - apply dup_nil.
    - apply dup_keep. exact IH.
  Qed.

  Lemma lossy_dup_refl : ∀ s, lossy_dup s s.
  Proof.
    induction s as [| x s IH].
    - apply lossy_dup_nil.
    - apply lossy_dup_keep. exact IH.
  Qed.


  Lemma dup_length : ∀ s r, dup s r → length s ≤ length r.
  Proof.
    intros s r H.
    induction H as [| x s r H IH | x s r H IH]; simpl.
    - apply le_n.
    - apply le_n_S. exact IH.
    - apply le_S. exact IH.
  Qed.


  Lemma perfect_lossy : ∀ s r, perfect s r → lossy s r.
  Proof.
    intros s r H. destruct H. exact (subseq_refl r).
  Qed.

  Lemma perfect_reorder : ∀ s r, perfect s r → reorder s r.
  Proof.
    intros s r H. destruct H. exact (Permutation_refl r).
  Qed.

  Lemma subseq_lossy_dup : ∀ s r, subseq r s → lossy_dup s r.
  Proof.
    intros s r H.
    induction H as [| x r s H IH | x r s H IH].
    - apply lossy_dup_nil.
    - apply lossy_dup_keep. exact IH.
    - apply lossy_dup_drop. exact IH.
  Qed.

  Lemma lossy_lossy_dup : ∀ s r, lossy s r → lossy_dup s r.
  Proof.
    intros s r H. exact (subseq_lossy_dup s r H).
  Qed.

  Lemma dup_lossy_dup : ∀ s r, dup s r → lossy_dup s r.
  Proof.
    intros s r H.
    induction H as [| x s r H IH | x s r H IH].
    - apply lossy_dup_nil.
    - apply lossy_dup_keep. exact IH.
    - apply lossy_dup_stutter. exact IH.
  Qed.


  Lemma cons_eq_head : ∀ (a1 a2 : M) (l1 l2 : list M),
    a1 :: l1 = a2 :: l2 → a1 = a2.
  Proof.
    intros a1 a2 l1 l2 H.
    exact (f_equal (fun l ⇒ match l with [] ⇒ a1 | a :: _ ⇒ a end) H).
  Qed.

  Lemma cons_eq_tail : ∀ (a1 a2 : M) (l1 l2 : list M),
    a1 :: l1 = a2 :: l2 → l1 = l2.
  Proof.
    intros a1 a2 l1 l2 H.
    exact (f_equal (fun l ⇒ match l with [] ⇒ l1 | _ :: t ⇒ t end) H).
  Qed.

  Lemma lossy_dup_nil_l : ∀ s r, lossy_dup s r → s = [] → r = [].
  Proof.
    intros s r H.
    induction H as [| x s0 r0 H0 IH0 | x s0 r0 H0 IH0 | x s0 r0 H0 IH0]; intros Hs.
    - reflexivity.
    - discriminate Hs.
    - discriminate Hs.
    - assert (Hcontra : x :: r0 = []). { apply IH0. exact Hs. }
      discriminate Hcontra.
  Qed.


  Lemma lossy_dup_preserves_all_eq : ∀ (x0 : M) (s r : list M),
    lossy_dup s r → (∀ p, In p s → p = x0) → ∀ p, In p r → p = x0.
  Proof.
    intros x0 s r H.
    induction H as [| x s0 r0 H0 IH0 | x s0 r0 H0 IH0 | x s0 r0 H0 IH0];
      intros Hs p Hp.
    - simpl in Hp. destruct Hp.
    - simpl in Hp. destruct Hp as [Hp | Hp].
      + rewrite <- Hp. apply Hs. simpl. left. reflexivity.
      + apply IH0.
        × intros q Hq. apply Hs. simpl. right. exact Hq.
        × exact Hp.
    - apply IH0.
      + intros q Hq. apply Hs. simpl. right. exact Hq.
      + exact Hp.
    - simpl in Hp. destruct Hp as [Hp | Hp].
      + rewrite <- Hp. apply IH0.
        × exact Hs.
        × simpl. left. reflexivity.
      + apply IH0.
        × exact Hs.
        × exact Hp.
  Qed.


  Lemma lossy_dup_app : ∀ s1 r1 s2 r2 : list M,
    lossy_dup s1 r1 → lossy_dup s2 r2 → lossy_dup (s1 ++ s2) (r1 ++ r2).
  Proof.
    intros s1 r1 s2 r2 H1.
    induction H1 as [| x s1' r1' H1' IH | x s1' r1' H1' IH | x s1' r1' H1' IH]; intros H2.
    - simpl. exact H2.
    - simpl. apply lossy_dup_keep. apply IH. exact H2.
    - simpl. apply lossy_dup_drop. apply IH. exact H2.
    - simpl. apply lossy_dup_stutter. apply IH. exact H2.
  Qed.


  Lemma lossy_dup_app_split : ∀ (s r : list M),
    lossy_dup s r → ∀ l1 l2 : list M, s = l1 ++ l2 →
    ∃ r1 r2, r = r1 ++ r2 ∧ lossy_dup l1 r1 ∧ lossy_dup l2 r2.
  Proof.
    intros s r H.
    induction H as [| x s0 r0 H0 IH0 | x s0 r0 H0 IH0 | x s0 r0 H0 IH0];
      intros l1 l2 Heq.
    - destruct l1 as [| y l1'].
      + destruct l2 as [| z l2'].
        × ∃ [], []. split.
          -- reflexivity.
          -- split; apply lossy_dup_nil.
        × simpl in Heq. discriminate Heq.
      + simpl in Heq. discriminate Heq.
    - destruct l1 as [| y l1'].
      + simpl in Heq.
        ∃ [], (x :: r0). split.
        × reflexivity.
        × split.
          -- apply lossy_dup_nil.
          -- rewrite <- Heq. apply lossy_dup_keep. exact H0.
      + simpl in Heq.
        assert (Heqx : x = y). { apply (cons_eq_head x y s0 (l1' ++ l2)). exact Heq. }
        assert (Heqs0 : s0 = l1' ++ l2). { apply (cons_eq_tail x y s0 (l1' ++ l2)). exact Heq. }
        destruct (IH0 l1' l2 Heqs0) as [r1' [r2' [Hr [Hl1 Hl2]]]].
        ∃ (x :: r1'), r2'. split.
        × simpl. rewrite Hr. reflexivity.
        × split.
          -- rewrite Heqx. apply lossy_dup_keep. exact Hl1.
          -- exact Hl2.
    - destruct l1 as [| y l1'].
      + simpl in Heq.
        ∃ [], r0. split.
        × reflexivity.
        × split.
          -- apply lossy_dup_nil.
          -- rewrite <- Heq. apply lossy_dup_drop. exact H0.
      + simpl in Heq.
        assert (Heqs0 : s0 = l1' ++ l2). { apply (cons_eq_tail x y s0 (l1' ++ l2)). exact Heq. }
        destruct (IH0 l1' l2 Heqs0) as [r1' [r2' [Hr [Hl1 Hl2]]]].
        ∃ r1', r2'. split.
        × exact Hr.
        × split.
          -- apply lossy_dup_drop. exact Hl1.
          -- exact Hl2.
    - destruct (IH0 l1 l2 Heq) as [r1' [r2' [Hr [Hl1 Hl2]]]].
      destruct r1' as [| y r1''].
      + simpl in Hr.
        ∃ [], (x :: x :: r0). split.
        × reflexivity.
        × split.
          -- exact Hl1.
          -- rewrite <- Hr in Hl2.
             apply lossy_dup_stutter. exact Hl2.
      + simpl in Hr.
        assert (Heqxy : x = y). { apply (cons_eq_head x y r0 (r1'' ++ r2')). exact Hr. }
        assert (Heqr0 : r0 = r1'' ++ r2'). { apply (cons_eq_tail x y r0 (r1'' ++ r2')). exact Hr. }
        ∃ (x :: x :: r1''), r2'. split.
        × simpl. rewrite Heqr0. reflexivity.
        × split.
          -- assert (Hl1' : lossy_dup l1 (x :: r1'')).
             { rewrite Heqxy. exact Hl1. }
             apply lossy_dup_stutter. exact Hl1'.
          -- exact Hl2.
  Qed.

End FaultClasses.


Example reorder_swaps : reorder [0; 1] [1; 0].
Proof. apply perm_swap. Qed.

Example reorder_not_lossy_dup : ¬ lossy_dup [0; 1] [1; 0].
Proof.
  intros H.
  repeat match goal with
  | Hc : lossy_dup _ _ |- False ⇒ inversion Hc; clear Hc; subst
  | Heq : _ = _ |- False ⇒ discriminate Heq
  end.
Qed.

Example lossy_drops : lossy [0; 1] [0].
Proof.
  apply subseq_keep. apply subseq_drop. apply subseq_nil.
Qed.

Example lossy_not_dup : ¬ dup [0; 1] [0].
Proof.
  intros H.
  repeat match goal with
  | Hc : dup _ _ |- False ⇒ inversion Hc; clear Hc; subst
  | Heq : _ = _ |- False ⇒ discriminate Heq
  end.
Qed.
