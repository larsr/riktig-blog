Require Export List.
From Riktig Require Export Interface.Decidable.


#[global] Instance and_dec (P Q : Prop) `{Decidable P} `{Decidable Q} : Decidable (P ∧ Q).
destruct H; [destruct H0; [left; auto | right; tauto] | right; tauto].
Defined.

#[global] Instance or_dec (P Q : Prop) `{Decidable P} `{Decidable Q} : Decidable (P ∨ Q).
destruct H; [left; auto | destruct H0; [left; auto | right; tauto]].
Defined.

#[global] Instance not_dec (P : Prop) `{Decidable P} : Decidable (¬ P).
destruct H; [right; tauto | left; tauto].
Defined.

#[global] Instance impl_dec (P Q : Prop) `{Decidable P} `{Decidable Q} : Decidable (P → Q).
destruct H0; [left; tauto | destruct H; [right; tauto | left; tauto]].
Defined.

#[global] Instance True_dec : Decidable True := left I.
#[global] Instance False_dec : Decidable False := right (fun H ⇒ H).


#[global] Instance list_eq_dec {A} `{EqDec A} : EqDec (list A).
intros lx; induction lx as [|x xs IH]; intros [|y ys].
- now left.
- now right.
- now right.
- destruct (dec (x = y)); destruct (IH ys); subst;
  [now left | right; congruence | right; congruence | right; congruence].
Defined.

#[global] Instance prod_eq_dec {A B} `{EqDec A} `{EqDec B} : EqDec (A × B).
intros [a1 b1] [a2 b2].
destruct (dec (a1 = a2)); destruct (dec (b1 = b2)); subst;
  [now left | right; congruence | right; congruence | right; congruence].
Defined.


#[global] Instance In_dec {A} `{EqDec A} (x : A) (l : list A) : Decidable (In x l) :=
  List.in_dec eq_dec x l.

#[global] Instance nil_dec {A} (l : list A) : Decidable (l = nil).
destruct l; [left; reflexivity | right; discriminate].
Defined.


Lemma dec_if_bool (P : Prop) {DP : Decidable P} (b : bool) (A : Type) (a1 a2 : A) :
  (P ↔ b = true) →
  (if dec P then a1 else a2) = (if b then a1 else a2).
Proof.
  intros Hb.
  destruct (dec P) as [p | np]; destruct b; try reflexivity.
  - apply Hb in p; discriminate.
  - exfalso; apply np, Hb; reflexivity.
Qed.
