Record Seq (C S : Type) := mkSeq { uncons : S → option (C × S) }.
Arguments uncons {C S}.
Arguments mkSeq {C S}.

Definition stringSeq : Seq ascii string := mkSeq (fun s ⇒
  match s with
  | EmptyString ⇒ None
  | String c rest ⇒ Some (c, rest)
  end).


Definition Parser (S T : Type) := S → option (T × S).

#[global] Instance Parser_EqRel (S T : Type) : EqRel (Parser S T) :=
  fun p q ⇒ ∀ s, p s = q s.

#[global] Instance Parser_HasBind (S : Type) : HasBind (Parser S) :=
  fun A B p f ⇒ fun input ⇒
    match p input with
    | None ⇒ None
    | Some (v, rest) ⇒ f v rest
    end.

#[global] Instance Parser_HasReturn (S : Type) : HasReturn (Parser S) :=
  fun A x ⇒ fun input ⇒ Some (x, input).

#[global] Instance Parser_HasFail (S : Type) : HasFail (Parser S) :=
  fun A ⇒ fun _ : S ⇒ None.

#[global] Instance Parser_HasAlt (S : Type) : HasAlt (Parser S) :=
  fun A p q ⇒ fun input ⇒
    match p input with
    | None ⇒ q input
    | r ⇒ r
    end.


Section ParserMonad.
  Context {S : Type}.

  Lemma parser_eqrel_refl : ∀ T (p : Parser S T), p == p.
  Proof. intros T p s. reflexivity. Qed.

  Lemma parser_eqrel_sym : ∀ T (p q : Parser S T), p == q → q == p.
  Proof. intros T p q H s. symmetry. exact (H s). Qed.

  Lemma parser_eqrel_trans : ∀ T (p q r : Parser S T),
    p == q → q == r → p == r.
  Proof. intros T p q r Hpq Hqr s. rewrite (Hpq s). exact (Hqr s). Qed.

  #[global] Instance Parser_EqRel_Equivalence (T : Type) :
    Equivalence (@equ (Parser S T) (Parser_EqRel S T)).
  Proof.
    apply Build_Equivalence.
    - exact (parser_eqrel_refl T).
    - exact (parser_eqrel_sym T).
    - exact (parser_eqrel_trans T).
  Qed.

  Lemma parser_bind_proper : ∀ A B,
    Proper (equ ==> (fun f g ⇒ ∀ x, f x == g x) ==> equ)
      (@bind (Parser S) (Parser_HasBind S) A B).
  Proof.
    intros A B p1 p2 Hp f g Hfg input.
    unfold bind, Parser_HasBind.
    rewrite (Hp input).
    destruct (p2 input) as [[v rest] |].
    - exact (Hfg v rest).
    - reflexivity.
  Qed.

  Lemma parser_bind_ret_l : ∀ A B (x : A) (f : A → Parser S B),
    bind (ret x) f == f x.
  Proof. intros A B x f input. reflexivity. Qed.

  Lemma parser_bind_ret_r : ∀ A (m : Parser S A), bind m ret == m.
  Proof.
    intros A m input.
    unfold bind, ret, Parser_HasBind, Parser_HasReturn.
    destruct (m input) as [[v rest] |]; reflexivity.
  Qed.

  Lemma parser_bind_assoc : ∀ A B C
      (m : Parser S A) (f : A → Parser S B) (g : B → Parser S C),
    bind (bind m f) g == bind m (fun x ⇒ bind (f x) g).
  Proof.
    intros A B C m f g input.
    unfold bind, Parser_HasBind.
    destruct (m input) as [[v rest] |]; reflexivity.
  Qed.

  #[global] Instance Parser_Monad : Monad (Parser S) :=
    @Build_Monad (Parser S) (Parser_EqRel S) (Parser_HasBind S) (Parser_HasReturn S)
      Parser_EqRel_Equivalence
      parser_bind_proper
      parser_bind_ret_l
      parser_bind_ret_r
      parser_bind_assoc.

End ParserMonad.


Section ParserAlternative.
  Context {S : Type}.

  Lemma parser_alt_proper : ∀ A,
    Proper (equ ==> equ ==> equ) (@alt (Parser S) (Parser_HasAlt S) A).
  Proof.
    intros A p1 p2 Hp q1 q2 Hq input.
    unfold alt, Parser_HasAlt.
    rewrite (Hp input).
    destruct (p2 input) as [pr |].
    - reflexivity.
    - exact (Hq input).
  Qed.

  Lemma parser_alt_assoc : ∀ A (p q r : Parser S A),
    alt (alt p q) r == alt p (alt q r).
  Proof.
    intros A p q r input.
    unfold alt, Parser_HasAlt.
    destruct (p input) as [pr |]; reflexivity.
  Qed.

  Lemma parser_fail_alt_l : ∀ A (q : Parser S A), alt fail q == q.
  Proof. intros A q input. reflexivity. Qed.

  Lemma parser_alt_fail_r : ∀ A (p : Parser S A), alt p fail == p.
  Proof.
    intros A p input.
    unfold alt, fail, Parser_HasAlt, Parser_HasFail.
    destruct (p input) as [pr |]; reflexivity.
  Qed.

  #[global] Instance Parser_Alternative : Alternative (Parser S) :=
    @Build_Alternative (Parser S) (Parser_EqRel S) (Parser_HasFail S) (Parser_HasAlt S)
      Parser_EqRel_Equivalence
      parser_alt_proper
      parser_alt_assoc
      parser_fail_alt_l
      parser_alt_fail_r.

End ParserAlternative.

Lemma parser_bind_fail_l : ∀ S A B (f : A → Parser S B),
  @bind (Parser S) _ A B fail f == fail.
Proof. intros S A B f input. reflexivity. Qed.

Definition counter_p : Parser nat nat := ret O.
Definition counter_q : Parser nat nat := ret (S O).
Definition counter_f (n : nat) : Parser nat nat :=
  match n with
  | O ⇒ fail
  | S _ ⇒ ret n
  end.

Lemma bind_alt_not_distributive :
  ¬ (bind (alt counter_p counter_q) counter_f
      == alt (bind counter_p counter_f) (bind counter_q counter_f)).
Proof.
  intros H.
  specialize (H O).
  unfold bind, alt, ret, fail, counter_p, counter_q, counter_f,
    Parser_HasBind, Parser_HasAlt, Parser_HasReturn, Parser_HasFail
    in H.
  simpl in H.
  discriminate H.
Qed.

Lemma bind_alt_distr_when_no_backtrack :
  ∀ S T U (p q : Parser S T) (f : T → Parser S U),
    (∀ input : S, bind p f input = None → p input = None) →
    bind (alt p q) f == alt (bind p f) (bind q f).
Proof.
  intros S T U p q f Hf input.
  unfold bind, alt, Parser_HasBind, Parser_HasAlt in ×.
  destruct (p input) as [[v rest] |] eqn:Hp.
  - destruct (f v rest) as [[v' rest'] |] eqn:Hfv.
    + reflexivity.
    + assert (Hcontra : match p input with
                         | None ⇒ None
                         | Some (v0, rest0) ⇒ f v0 rest0
                         end = None).
      { rewrite Hp. exact Hfv. }
      specialize (Hf input Hcontra).
      rewrite Hf in Hp.
      discriminate Hp.
  - reflexivity.
Qed.


Section ParserPrimitives.
  Context {C S : Type}.

  Definition satisfy (seq : Seq C S) (pred : C → bool) : Parser S C :=
    fun input ⇒
      match uncons seq input with
      | Some (c, rest) ⇒ if pred c then Some (c, rest) else None
      | None ⇒ None
      end.

  Definition any (seq : Seq C S) : Parser S C := satisfy seq (fun _ ⇒ true).

  Definition exact (seq : Seq C S) (eq_dec : C → C → bool) (c : C) : Parser S C :=
    satisfy seq (eq_dec c).

  Definition eos (seq : Seq C S) : Parser S unit :=
    fun input ⇒
      match uncons seq input with
      | None ⇒ Some (tt, input)
      | Some _ ⇒ None
      end.

End ParserPrimitives.


Definition seqL {S T U : Type} (p : Parser S T) (q : Parser S U) : Parser S T :=
  bind p (fun x ⇒ bind q (fun _ ⇒ ret x)).

Definition seqR {S T U : Type} (p : Parser S T) (q : Parser S U) : Parser S U :=
  bind p (fun _ ⇒ q).

Notation "p <* q" := (seqL p q) (at level 40, left associativity).
Notation "p *> q" := (seqR p q) (at level 40, left associativity).

Definition optional {S T : Type} (p : Parser S T) : Parser S (option T) :=
  alt (map (@Some T) p) (ret None).

Definition between {S A B T : Type}
    (open : Parser S A) (close : Parser S B) (p : Parser S T) : Parser S T :=
  bind open (fun _ ⇒ bind p (fun x ⇒ bind close (fun _ ⇒ ret x))).


Fixpoint many {T : Type} (p : Parser string T) (input : string)
    : Parser string (list T) :=
  match input with
  | EmptyString ⇒ ret []
  | String _ input' ⇒
    alt (bind p (fun x ⇒ bind (many p input') (fun xs ⇒ ret (x :: xs))))
        (ret [])
  end.

Definition many1 {T : Type} (p : Parser string T) (input : string)
    : Parser string (list T) :=
  bind p (fun x ⇒ bind (many p input) (fun xs ⇒ ret (x :: xs))).

Fixpoint sepBy1 {T U : Type} (p : Parser string T) (sep : Parser string U)
    (input : string) : Parser string (list T) :=
  match input with
  | EmptyString ⇒ bind p (fun x ⇒ ret [x])
  | String _ input' ⇒
    bind p (fun x ⇒
      alt (bind sep (fun _ ⇒ bind (sepBy1 p sep input') (fun xs ⇒ ret (x :: xs))))
          (ret [x]))
  end.

Fixpoint skipMany {T : Type} (p : Parser string T) (input : string)
    : Parser string unit :=
  match input with
  | EmptyString ⇒ ret tt
  | String _ input' ⇒ alt (bind p (fun _ ⇒ skipMany p input')) (ret tt)
  end.


Definition char (c : ascii) : Parser string ascii :=
  exact stringSeq Ascii.eqb c.

Definition run {S T : Type} (p : Parser S T) (input : S) : option T :=
  match p input with
  | Some (v, _) ⇒ Some v
  | None ⇒ None
  end.

Definition runExact {S C T : Type} (seq : Seq C S) (p : Parser S T) (input : S)
    : option T :=
  match p input with
  | Some (v, rest) ⇒
    match uncons seq rest with
    | None ⇒ Some v
    | Some _ ⇒ None
    end
  | None ⇒ None
  end.


Example test_char : run (char "a"%char) "abc" = Some "a"%char.
Proof. reflexivity. Qed.

Example test_bind :
  run (bind (char "a"%char) (fun x ⇒ bind (char "b"%char) (fun y ⇒ ret (x, y)))) "abc"
    = Some ("a"%char, "b"%char).
Proof. reflexivity. Qed.

Example test_alt : run (alt (char "x"%char) (char "a"%char)) "abc" = Some "a"%char.
Proof. reflexivity. Qed.
