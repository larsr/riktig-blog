Section GramSchmidtKit.
  Context {F V : Type} `{PosInner F V}.


  Fixpoint gs_aux (done : list V) (v : V) : V :=
    match done with
    | nil ⇒ v
    | cons d done' ⇒ gs_aux done' (v + - proj d v)
    end.

  Fixpoint gs_step (acc vs : list V) : list V :=
    match vs with
    | nil ⇒ acc
    | cons v vs' ⇒ gs_step (acc ++ (gs_aux acc v :: nil)) vs'
    end.

  Definition gs (vs : list V) : list V := gs_step nil vs.

  Lemma gs_aux_nil : ∀ v : V, gs_aux nil v == v.
  Proof.
    intros v.
    reflexivity.
  Qed.

  Lemma gs_aux_cons : ∀ (d : V) (done' : list V) (v : V),
    gs_aux (d :: done') v == gs_aux done' (v + - proj d v).
  Proof.
    intros d done' v.
    reflexivity.
  Qed.


  Definition PairwiseOrth (l : list V) : Prop :=
    ∀ d1 d2 : V, In d1 l → In d2 l → orthogonal d1 d2 ∨ d1 = d2.

  Lemma pairwise_orth_nil : PairwiseOrth (nil : list V).
  Proof.
    intros d1 d2 Hd1 Hd2.
    destruct Hd1.
  Qed.

  Lemma nonzero_nil : ∀ d : V, In d (nil : list V) → norm2 d =/= 0.
  Proof.
    intros d Hd.
    destruct Hd.
  Qed.


  Lemma in_singleton : ∀ x y : V, In x (y :: nil) → x = y.
  Proof.
    intros x y Hx.
    destruct Hx as [Heq | Hf].
    - symmetry.
      exact Heq.
    - destruct Hf.
  Qed.

  Lemma orthogonal_sub : ∀ u v w : V,
    orthogonal u w → orthogonal v w → orthogonal (u + - v) w.
  Proof.
    intros u v w Hu Hv.
    apply orthogonal_def.
    rewrite (ip_add_l u (- v) w), (ip_neg_l v w),
      (orthogonal_eq u w Hu), (orthogonal_eq v w Hv), neg_zero.
    apply idr.
  Qed.

  Lemma proj_orthogonal_to : ∀ (d0 d1 w : V),
    orthogonal d0 d1 → orthogonal (proj d1 w) d0.
  Proof.
    intros d0 d1 w Hd0d1.
    apply orthogonal_def.
    rewrite (proj_eq w d1), (ip_scale_l (⟨ w , d1 ⟩ × inv (norm2 d1)) d1 d0),
      (orthogonal_eq d1 d0 (orthogonal_sym d0 d1 Hd0d1)).
    apply mul_0_r.
  Qed.

  Lemma proj_self_zero : ∀ d0 w : V, orthogonal w d0 → proj d0 w == 0.
  Proof.
    intros d0 w Hw.
    rewrite (proj_eq w d0), (orthogonal_eq w d0 Hw), (mul_0_l (inv (norm2 d0))).
    apply zero_scale.
  Qed.


  Lemma gs_aux_preserves_orth : ∀ (done : list V) (d0 : V),
    (∀ d : V, In d done → orthogonal d0 d ∨ d = d0) →
    ∀ w : V, orthogonal w d0 → orthogonal (gs_aux done w) d0.
  Proof.
    intros done; induction done as [| d1 done' IH]; intros d0 Hdone w Hw.
    - exact Hw.
    - apply IH.
      + intros d Hd.
        apply Hdone.
        right.
        exact Hd.
      + destruct (Hdone d1 (or_introl eq_refl)) as [Hortho | Heq].
        × apply orthogonal_sub.
          -- exact Hw.
          -- exact (proj_orthogonal_to d0 d1 w Hortho).
        × apply orthogonal_def.
          rewrite Heq, (proj_self_zero d0 w Hw), neg_zero.
          rewrite idr.
          exact (orthogonal_eq w d0 Hw).
  Qed.

  Lemma gs_aux_orth : ∀ done : list V,
    PairwiseOrth done →
    (∀ d : V, In d done → norm2 d =/= 0) →
    ∀ v d : V, In d done → orthogonal (gs_aux done v) d.
  Proof.
    intros done; induction done as [| d0 done' IH]; intros Hpw Hnz v d Hd.
    - destruct Hd.
    - destruct Hd as [Heq | Hin].
      + rewrite <- Heq.
        apply (gs_aux_preserves_orth done' d0).
        × intros d' Hd'.
          destruct (Hpw d0 d' (or_introl eq_refl) (or_intror Hd')) as [Ho | He].
          -- left. exact Ho.
          -- right. symmetry. exact He.
        × exact (proj_orthogonal v d0 (Hnz d0 (or_introl eq_refl))).
      + apply IH.
        × intros e1 e2 He1 He2.
          apply Hpw.
          -- right. exact He1.
          -- right. exact He2.
        × intros e He.
          apply Hnz.
          right.
          exact He.
        × exact Hin.
  Qed.


  Lemma gs_step_extends : ∀ (vs acc : list V) (d : V),
    In d acc → In d (gs_step acc vs).
  Proof.
    intros vs; induction vs as [| v vs' IH]; intros acc d Hd.
    - exact Hd.
    - apply IH.
      apply in_or_app.
      left.
      exact Hd.
  Qed.

  Lemma gs_step_pairwise_extend : ∀ (acc : list V) (v : V),
    PairwiseOrth acc →
    (∀ d : V, In d acc → norm2 d =/= 0) →
    PairwiseOrth (acc ++ (gs_aux acc v :: nil)).
  Proof.
    intros acc v Hpw Hnz d1 d2 Hd1 Hd2.
    destruct (in_app_or acc (gs_aux acc v :: nil) d1 Hd1) as [Hd1a | Hd1w].
    - destruct (in_app_or acc (gs_aux acc v :: nil) d2 Hd2) as [Hd2a | Hd2w].
      + exact (Hpw d1 d2 Hd1a Hd2a).
      + left.
        rewrite (in_singleton d2 (gs_aux acc v) Hd2w).
        apply orthogonal_sym.
        exact (gs_aux_orth acc Hpw Hnz v d1 Hd1a).
    - destruct (in_app_or acc (gs_aux acc v :: nil) d2 Hd2) as [Hd2a | Hd2w].
      + left.
        rewrite (in_singleton d1 (gs_aux acc v) Hd1w).
        exact (gs_aux_orth acc Hpw Hnz v d2 Hd2a).
      + right.
        rewrite (in_singleton d1 (gs_aux acc v) Hd1w),
          (in_singleton d2 (gs_aux acc v) Hd2w).
        reflexivity.
  Qed.

  Lemma gs_step_new_nonzero : ∀ (vs acc : list V) (v : V),
    (∀ d : V, In d (gs_step acc (v :: vs)) → norm2 d =/= 0) →
    norm2 (gs_aux acc v) =/= 0.
  Proof.
    intros vs acc v Hnzf.
    apply Hnzf.
    apply (gs_step_extends vs (acc ++ (gs_aux acc v :: nil))).
    apply in_or_app.
    right.
    left.
    reflexivity.
  Qed.

  Theorem gs_step_pairwise : ∀ (vs acc : list V),
    PairwiseOrth acc →
    (∀ d : V, In d acc → norm2 d =/= 0) →
    (∀ d : V, In d (gs_step acc vs) → norm2 d =/= 0) →
    PairwiseOrth (gs_step acc vs).
  Proof.
    intros vs; induction vs as [| v vs' IH]; intros acc Hpw Hnz Hnzf.
    - exact Hpw.
    - apply IH.
      + exact (gs_step_pairwise_extend acc v Hpw Hnz).
      + intros d Hd.
        destruct (in_app_or acc (gs_aux acc v :: nil) d Hd) as [Hda | Hdw].
        × exact (Hnz d Hda).
        × rewrite (in_singleton d (gs_aux acc v) Hdw).
          exact (gs_step_new_nonzero vs' acc v Hnzf).
      + exact Hnzf.
  Qed.


  Theorem gs_pairwise_orthogonal : ∀ vs : list V,
    (∀ d : V, In d (gs vs) → norm2 d =/= 0) →
    ∀ d1 d2 : V,
      In d1 (gs vs) → In d2 (gs vs) → d1 ≠ d2 → orthogonal d1 d2.
  Proof.
    intros vs Hnz d1 d2 Hd1 Hd2 Hneq.
    destruct (gs_step_pairwise vs nil pairwise_orth_nil nonzero_nil Hnz
                d1 d2 Hd1 Hd2) as [Ho | Heq].
    - exact Ho.
    - destruct (Hneq Heq).
  Qed.


  Fixpoint list_lincomb (coeffs : list F) (vecs : list V) : V :=
    match coeffs with
    | nil ⇒ 0
    | c :: coeffs' ⇒
      match vecs with
      | nil ⇒ 0
      | v :: vecs' ⇒ c *: v + list_lincomb coeffs' vecs'
      end
    end.

  Lemma list_lincomb_nil_l : ∀ vecs : list V, list_lincomb nil vecs == 0.
  Proof.
    intros vecs.
    reflexivity.
  Qed.

  Lemma list_lincomb_cons : ∀ (c : F) (coeffs' : list F) (v : V)
      (vecs' : list V),
    list_lincomb (c :: coeffs') (v :: vecs') == c *: v + list_lincomb coeffs' vecs'.
  Proof.
    intros c coeffs' v vecs'.
    reflexivity.
  Qed.

  Lemma gs_aux_combination : ∀ (done : list V) (v : V),
    ∃ coeffs : list F, gs_aux done v == v + list_lincomb coeffs done.
  Proof.
    intros done; induction done as [| d done' IH]; intros v.
    - ∃ nil.
      rewrite gs_aux_nil, list_lincomb_nil_l.
      symmetry.
      apply idr.
    - destruct (IH (v + - proj d v)) as [coeffs' Heq].
      ∃ ((- (⟨ v , d ⟩ × inv (norm2 d))) :: coeffs').
      rewrite gs_aux_cons, Heq, list_lincomb_cons, proj_eq, <- neg_scale.
      apply assoc.
  Qed.

  Lemma gs_step_new_combination : ∀ (vs acc : list V) (w : V),
    In w (gs_step acc vs) →
    In w acc ∨
    ∃ (v : V) (pre : list V) (coeffs : list F),
      In v vs ∧ (∀ d : V, In d pre → In d (gs_step acc vs)) ∧
      w == v + list_lincomb coeffs pre.
  Proof.
    intros vs; induction vs as [| v' vs' IH]; intros acc w Hw.
    - left. exact Hw.
    - destruct (IH (acc ++ (gs_aux acc v' :: nil)) w Hw)
        as [Hacc' | [v [pre [coeffs [Hv [Hpre Heqw]]]]]].
      + destruct (in_app_or acc (gs_aux acc v' :: nil) w Hacc')
          as [Hina | Hinw].
        × left. exact Hina.
        × right.
          destruct (gs_aux_combination acc v') as [coeffs0 Heq0].
          ∃ v', acc, coeffs0.
          split.
          -- left. reflexivity.
          -- split.
             ++ intros d Hd.
                apply (gs_step_extends vs' (acc ++ (gs_aux acc v' :: nil)) d).
                apply in_or_app.
                left. exact Hd.
             ++ rewrite (in_singleton w (gs_aux acc v') Hinw).
                exact Heq0.
      + right.
        ∃ v, pre, coeffs.
        split.
        × right. exact Hv.
        × split.
          -- exact Hpre.
          -- exact Heqw.
  Qed.

  Theorem gs_element_combination : ∀ (vs : list V) (w : V),
    In w (gs vs) →
    ∃ (v : V) (pre : list V) (coeffs : list F),
      In v vs ∧ (∀ d : V, In d pre → In d (gs vs)) ∧
      w == v + list_lincomb coeffs pre.
  Proof.
    intros vs w Hw.
    destruct (gs_step_new_combination vs nil w Hw) as [Hnil | Hex].
    - destruct Hnil.
    - exact Hex.
  Qed.
