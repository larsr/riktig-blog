Section BilinearVS.
  Context {F U V W : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}
          {VW : VectorSpace F W}.

  Local Notation Bilin := (BilinearMap F U V W).



  #[global] Instance Bilin_Zero : Zero Bilin :=
    @bm_zero_map F U V W FF VU VV VW.


  Section BilinPlus.
    Context (b b' : Bilin).

    #[global] Instance bilin_plus_proper :
      Proper (equ ==> equ ==> equ) (fun (u : U) (v : V) ⇒ b u v + b' u v).
    Proof.
      intros u u' Hu v v' Hv.
      rewrite (bm_proper b u u' Hu v v' Hv), (bm_proper b' u u' Hu v v' Hv).
      reflexivity.
    Qed.

    Lemma bilin_plus_add_l : ∀ (u u' : U) (v : V),
      b (u + u') v + b' (u + u') v == (b u v + b' u v) + (b u' v + b' u' v).
    Proof.
      intros u u' v.
      rewrite (bm_add_l b u u' v), (bm_add_l b' u u' v).
      apply add_interchange.
    Qed.

    Lemma bilin_plus_scale_l : ∀ (a : F) (u : U) (v : V),
      b (a *: u) v + b' (a *: u) v == a *: (b u v + b' u v).
    Proof.
      intros a u v.
      rewrite (bm_scale_l b a u v), (bm_scale_l b' a u v).
      symmetry.
      apply sc_scale_add.
    Qed.

    Lemma bilin_plus_add_r : ∀ (u : U) (v v' : V),
      b u (v + v') + b' u (v + v') == (b u v + b' u v) + (b u v' + b' u v').
    Proof.
      intros u v v'.
      rewrite (bm_add_r b u v v'), (bm_add_r b' u v v').
      apply add_interchange.
    Qed.

    Lemma bilin_plus_scale_r : ∀ (a : F) (u : U) (v : V),
      b u (a *: v) + b' u (a *: v) == a *: (b u v + b' u v).
    Proof.
      intros a u v.
      rewrite (bm_scale_r b a u v), (bm_scale_r b' a u v).
      symmetry.
      apply sc_scale_add.
    Qed.

    Definition bilin_plus : Bilin :=
      @Build_BilinearMap F U V W FF VU VV VW (fun (u : U) (v : V) ⇒ b u v + b' u v)
        bilin_plus_proper bilin_plus_add_l bilin_plus_scale_l
        bilin_plus_add_r bilin_plus_scale_r.

  End BilinPlus.

  #[global] Instance Bilin_Add : HasAdd Bilin := bilin_plus.


  Section BilinOpp.
    Context (b : Bilin).

    #[global] Instance bilin_opp_proper :
      Proper (equ ==> equ ==> equ) (fun (u : U) (v : V) ⇒ - b u v).
    Proof.
      intros u u' Hu v v' Hv.
      rewrite (bm_proper b u u' Hu v v' Hv).
      reflexivity.
    Qed.

    Lemma bilin_opp_add_l : ∀ (u u' : U) (v : V),
      - b (u + u') v == - b u v + - b u' v.
    Proof.
      intros u u' v.
      rewrite (bm_add_l b u u' v).
      apply neg_add.
    Qed.

    Lemma bilin_opp_scale_l : ∀ (a : F) (u : U) (v : V),
      - b (a *: u) v == a *: - b u v.
    Proof.
      intros a u v.
      rewrite (bm_scale_l b a u v).
      symmetry.
      apply scale_neg.
    Qed.

    Lemma bilin_opp_add_r : ∀ (u : U) (v v' : V),
      - b u (v + v') == - b u v + - b u v'.
    Proof.
      intros u v v'.
      rewrite (bm_add_r b u v v').
      apply neg_add.
    Qed.

    Lemma bilin_opp_scale_r : ∀ (a : F) (u : U) (v : V),
      - b u (a *: v) == a *: - b u v.
    Proof.
      intros a u v.
      rewrite (bm_scale_r b a u v).
      symmetry.
      apply scale_neg.
    Qed.

    Definition bilin_opp : Bilin :=
      @Build_BilinearMap F U V W FF VU VV VW (fun (u : U) (v : V) ⇒ - b u v)
        bilin_opp_proper bilin_opp_add_l bilin_opp_scale_l
        bilin_opp_add_r bilin_opp_scale_r.

  End BilinOpp.

  #[global] Instance Bilin_Neg : HasNeg Bilin := bilin_opp.


  Section BilinSmul.
    Context (a : F) (b : Bilin).

    #[global] Instance bilin_smul_proper :
      Proper (equ ==> equ ==> equ) (fun (u : U) (v : V) ⇒ a *: b u v).
    Proof.
      intros u u' Hu v v' Hv.
      rewrite (bm_proper b u u' Hu v v' Hv).
      reflexivity.
    Qed.

    Lemma bilin_smul_add_l : ∀ (u u' : U) (v : V),
      a *: b (u + u') v == a *: b u v + a *: b u' v.
    Proof.
      intros u u' v.
      rewrite (bm_add_l b u u' v).
      apply sc_scale_add.
    Qed.

    Lemma bilin_smul_scale_l : ∀ (c : F) (u : U) (v : V),
      a *: b (c *: u) v == c *: (a *: b u v).
    Proof.
      intros c u v.
      rewrite (bm_scale_l b c u v).
      apply scale_swap.
    Qed.

    Lemma bilin_smul_add_r : ∀ (u : U) (v v' : V),
      a *: b u (v + v') == a *: b u v + a *: b u v'.
    Proof.
      intros u v v'.
      rewrite (bm_add_r b u v v').
      apply sc_scale_add.
    Qed.

    Lemma bilin_smul_scale_r : ∀ (c : F) (u : U) (v : V),
      a *: b u (c *: v) == c *: (a *: b u v).
    Proof.
      intros c u v.
      rewrite (bm_scale_r b c u v).
      apply scale_swap.
    Qed.

    Definition bilin_smul : Bilin :=
      @Build_BilinearMap F U V W FF VU VV VW (fun (u : U) (v : V) ⇒ a *: b u v)
        bilin_smul_proper bilin_smul_add_l bilin_smul_scale_l
        bilin_smul_add_r bilin_smul_scale_r.

  End BilinSmul.

  #[global] Instance Bilin_Scale : HasScale F Bilin := bilin_smul.



  Lemma bilin_assoc : ∀ b c d : Bilin, (b + c) + d == b + (c + d).
  Proof.
    intros b c d u v.
    apply mon_assoc.
  Qed.

  Lemma bilin_idl : ∀ b : Bilin, 0 + b == b.
  Proof.
    intros b u v.
    apply mon_idl.
  Qed.

  Lemma bilin_idr : ∀ b : Bilin, b + 0 == b.
  Proof.
    intros b u v.
    apply mon_idr.
  Qed.

  #[global] Instance bilin_add_proper : Proper (equ ==> equ ==> equ) (@add Bilin Bilin_Add).
  Proof.
    intros b b' Hb c c' Hc u v.
    apply mon_add_proper.
    - apply Hb.
    - apply Hc.
  Qed.

  #[global] Instance Bilin_Monoid : Monoid Bilin := {|
    mon_equ := Bilin_EqRel;
    mon_equiv := Bilin_Equivalence;
    mon_zero := Bilin_Zero;
    mon_add := Bilin_Add;
    mon_assoc := bilin_assoc;
    mon_idl := bilin_idl;
    mon_idr := bilin_idr;
    mon_add_proper := bilin_add_proper
  |}.


  Lemma bilin_invl : ∀ b : Bilin, (- b) + b == 0.
  Proof.
    intros b u v.
    apply grp_invl.
  Qed.

  Lemma bilin_invr : ∀ b : Bilin, b + (- b) == 0.
  Proof.
    intros b u v.
    apply grp_invr.
  Qed.

  #[global] Instance bilin_neg_proper : Proper (equ ==> equ) (@neg Bilin Bilin_Neg).
  Proof.
    intros b b' Hb u v.
    apply grp_neg_proper, Hb.
  Qed.

  #[global] Instance Bilin_Group : Group Bilin := {|
    grp_monoid := Bilin_Monoid;
    grp_neg := Bilin_Neg;
    grp_invl := bilin_invl;
    grp_invr := bilin_invr;
    grp_neg_proper := bilin_neg_proper
  |}.


  Lemma bilin_comm : ∀ b c : Bilin, b + c == c + b.
  Proof.
    intros b c u v.
    apply ab_comm.
  Qed.

  #[global] Instance Bilin_AbGroup : AbGroup Bilin := {|
    ab_group := Bilin_Group;
    ab_comm := bilin_comm
  |}.


  #[global] Instance bilin_sc_proper : Proper (equ ==> equ ==> equ) (@scale F Bilin Bilin_Scale).
  Proof.
    intros a a' Ha b b' Hb u v.
    exact (sc_proper a a' Ha (b u v) (b' u v) (Hb u v)).
  Qed.

  Lemma bilin_scale_add : ∀ (a : F) (b c : Bilin),
    a *: (b + c) == a *: b + a *: c.
  Proof.
    intros a b c u v.
    exact (sc_scale_add a (b u v) (c u v)).
  Qed.

  Lemma bilin_add_scale : ∀ (a c : F) (b : Bilin),
    (a + c) *: b == a *: b + c *: b.
  Proof.
    intros a c b u v.
    exact (sc_add_scale a c (b u v)).
  Qed.

  Lemma bilin_scale_scale : ∀ (a c : F) (b : Bilin),
    (a × c) *: b == a *: (c *: b).
  Proof.
    intros a c b u v.
    exact (sc_scale_scale a c (b u v)).
  Qed.

  Lemma bilin_scale_1 : ∀ b : Bilin, 1 *: b == b.
  Proof.
    intros b u v.
    exact (sc_scale_1 (b u v)).
  Qed.

  #[global] Instance Bilin_ScaleLaws : ScaleLaws F Bilin :=
    @Build_ScaleLaws F Bilin FF Bilin_AbGroup Bilin_Scale
      bilin_sc_proper bilin_scale_add bilin_add_scale bilin_scale_scale bilin_scale_1.

  #[global] Instance Bilin_VectorSpace : VectorSpace F Bilin :=
    @Build_VectorSpace F Bilin FF Bilin_AbGroup Bilin_Scale Bilin_ScaleLaws.

End BilinearVS.
