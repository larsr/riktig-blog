Section FieldNormedSpace.
  Context {F : Type} `{OrderedField F}.

  #[global] Instance Field_HasNorm : HasNorm F F := abs.

  Lemma field_vnorm_eq_abs : ∀ u : F, ‖ u ‖ == abs u.
  Proof. intros u. reflexivity. Qed.

  #[global] Instance field_norm_proper :
    Proper (equ ==> equ) (@vnorm F F Field_HasNorm).
  Proof. exact abs_proper. Qed.

  Lemma field_norm_nonneg : ∀ u : F, 0 ≤ ‖ u ‖.
  Proof. intros u. exact (abs_nonneg u). Qed.

  Lemma field_norm_def : ∀ u : F, ‖ u ‖ == 0 → u == 0.
  Proof. intros u Heq. exact (proj1 (abs_zero_iff u) Heq). Qed.

  Lemma field_norm_scale : ∀ a u : F, ‖ a *: u ‖ == abs a × ‖ u ‖.
  Proof. intros a u. exact (abs_mul a u). Qed.

  Lemma field_norm_triangle : ∀ u v : F, ‖ u + v ‖ ≤ ‖ u ‖ + ‖ v ‖.
  Proof. intros u v. exact (abs_triangle u v). Qed.

  Lemma field_norm_neg : ∀ h : F, ‖ - h ‖ == ‖ h ‖.
  Proof. intros h. exact (abs_neg h). Qed.

  #[global] Instance Field_NormedSpace : NormedSpace F F := {|
    ns_norm := Field_HasNorm;
    norm_proper := field_norm_proper;
    norm_nonneg := field_norm_nonneg;
    norm_def := field_norm_def;
    norm_scale := field_norm_scale;
    norm_triangle := field_norm_triangle
  |}.

End FieldNormedSpace.

Example field_ns_check (F : Type) (HFF : Field F) (HF : OrderedField F) :
  NormedSpace F F := _.


Section ScaleLM.
  Context {F : Type} `{Field F}.
  Context (c : F).

  Definition scale_map (h : F) : F := c × h.

  Lemma scale_map_proper : Proper (equ ==> equ) scale_map.
  Proof.
    intros h h' Hh.
    unfold scale_map.
    apply ring_mul_proper; [reflexivity | exact Hh].
  Qed.

  Lemma scale_map_add : ∀ h1 h2 : F,
    scale_map (h1 + h2) == scale_map h1 + scale_map h2.
  Proof. intros h1 h2. exact (ring_distl c h1 h2). Qed.

  Lemma scale_map_scale : ∀ (a : F) (h : F),
    scale_map (a *: h) == a *: scale_map h.
  Proof.
    intros a h.
    unfold scale_map, scale, Field_Scale.
    calc (c × (a × h))
      == ((c × a) × h) { (ring_mul_assoc c a h) }
      == ((a × c) × h) { (comm c a) }
      == (a × (c × h)) { (ring_mul_assoc a c h) }.
  Qed.

  Definition lm_scale_map : LinearMap F F F :=
    @Build_LinearMap F F F _ Field_VectorSpace Field_VectorSpace scale_map
      scale_map_proper scale_map_add scale_map_scale.

End ScaleLM.

Section NegLM.
  Context {F : Type} `{Field F}.

  Definition neg_map (h : F) : F := - h.

  Lemma neg_map_proper : Proper (equ ==> equ) neg_map.
  Proof. intros h h' Hh. unfold neg_map. rewrite Hh. reflexivity. Qed.

  Lemma neg_map_add : ∀ h1 h2 : F,
    neg_map (h1 + h2) == neg_map h1 + neg_map h2.
  Proof. intros h1 h2. exact (neg_add h1 h2). Qed.

  Lemma neg_map_scale : ∀ (a h : F), neg_map (a *: h) == a *: neg_map h.
  Proof.
    intros a h.
    unfold neg_map, scale, Field_Scale.
    symmetry.
    exact (mul_neg_r a h).
  Qed.

  Definition lm_neg_map : LinearMap F F F :=
    @Build_LinearMap F F F _ Field_VectorSpace Field_VectorSpace neg_map
      neg_map_proper neg_map_add neg_map_scale.

End NegLM.

Section ScaleLMOrder.
  Context {F : Type} `{OrderedField F}.

  Lemma lm_scale_map_bounded : ∀ c : F, lm_bounded (lm_scale_map c).
  Proof.
    intros c.
    ∃ (abs c).
    split.
    - exact (abs_nonneg c).
    - intros h.
      apply eq_le.
      exact (abs_mul c h).
  Qed.

  Lemma lm_neg_map_bounded : lm_bounded lm_neg_map.
  Proof.
    ∃ 1.
    split.
    - exact zero_le_one.
    - intros h.
      apply eq_le.
      rewrite (ring_mul_1l (‖ h ‖)).
      unfold vnorm, Field_HasNorm.
      exact (abs_neg h).
  Qed.

  Lemma is_deriv_scale_map : ∀ (c a : F),
    is_deriv (scale_map c) a (lm_scale_map c).
  Proof.
    intros c a.
    exact (deriv_lm (lm_scale_map c) a (lm_scale_map_bounded c)).
  Qed.

  Lemma is_deriv_neg_map : ∀ a : F, is_deriv neg_map a lm_neg_map.
  Proof.
    intros a.
    exact (deriv_lm lm_neg_map a lm_neg_map_bounded).
  Qed.

  Lemma lm_comp_scale_scale : ∀ c d : F,
    lm_comp (lm_scale_map d) (lm_scale_map c) == lm_scale_map (d × c).
  Proof.
    intros c d h.
    unfold lm_comp, lm_scale_map, scale_map; simpl.
    symmetry.
    exact (ring_mul_assoc d c h).
  Qed.

  Lemma lm_comp_scale_neg : ∀ d : F,
    lm_comp (lm_scale_map d) lm_neg_map == lm_scale_map (- d).
  Proof.
    intros d h.
    unfold lm_comp, lm_scale_map, lm_neg_map, scale_map, neg_map; simpl.
    transitivity (- (d × h)).
    - exact (mul_neg_r d h).
    - symmetry. exact (mul_neg_l d h).
  Qed.

  Lemma is_deriv_L_equ : ∀ (f : F → F) (a : F) (L1 L2 : LinearMap F F F),
    L1 == L2 → is_deriv f a L1 → is_deriv f a L2.
  Proof.
    intros f a L1 L2 HL [[C [HC Hbnd]] HD].
    split.
    - ∃ C.
      split.
      + exact HC.
      + intros u. rewrite <- (HL u). exact (Hbnd u).
    - intros eps Heps.
      destruct (HD eps Heps) as [delta [Hd HR]].
      ∃ delta.
      split.
      + exact Hd.
      + intros h Hh. rewrite <- (HL h). exact (HR h Hh).
  Qed.

End ScaleLMOrder.


Section ExpDerivChain.
  Context {F : Type} `{Exp F}.

  Context (exp_is_deriv : ∀ a : F, is_deriv exp_fn a (lm_scale_map (exp_fn a))).

  Theorem is_deriv_exp_scale : ∀ (c a : F),
    is_deriv (fun x : F ⇒ exp_fn (c × x)) a (lm_scale_map (exp_fn (c × a) × c)).
  Proof.
    intros c a.
    apply (is_deriv_L_equ (fun x : F ⇒ exp_fn (c × x)) a
      (lm_comp (lm_scale_map (exp_fn (c × a))) (lm_scale_map c))).
    - exact (lm_comp_scale_scale c (exp_fn (c × a))).
    - exact (deriv_comp (scale_map c) exp_fn a (lm_scale_map c)
        (lm_scale_map (exp_fn (c × a))) exp_fn_proper
        (is_deriv_scale_map c a) (exp_is_deriv (c × a))).
  Qed.

  Theorem is_deriv_exp_neg : ∀ a : F,
    is_deriv (fun x : F ⇒ exp_fn (- x)) a (lm_scale_map (- exp_fn (- a))).
  Proof.
    intros a.
    apply (is_deriv_L_equ (fun x : F ⇒ exp_fn (- x)) a
      (lm_comp (lm_scale_map (exp_fn (- a))) lm_neg_map)).
    - exact (lm_comp_scale_neg (exp_fn (- a))).
    - exact (deriv_comp neg_map exp_fn a lm_neg_map
        (lm_scale_map (exp_fn (- a))) exp_fn_proper
        (is_deriv_neg_map a) (exp_is_deriv (- a))).
  Qed.

End ExpDerivChain.


Section LocalAgreement.
  Context {F : Type} `{OrderedField F}.

  Lemma is_deriv_of_local_eq : ∀ (f g : F → F),
    Proper (equ ==> equ) f → Proper (equ ==> equ) g →
    ∀ (a : F) (L : LinearMap F F F) (d : F),
    0 < d → (∀ h : F, ‖ h ‖ < d → f (a + h) == g (a + h)) →
    is_deriv g a L → is_deriv f a L.
  Proof.
    intros f g Pf Pg a L d Hd Hfg [Hb HD].
    assert (H0d : ‖ (0 : F) ‖ < d).
    { rewrite field_vnorm_eq_abs, abs_zero. exact Hd. }
    pose proof (Hfg 0 H0d) as Hfa0.
    assert (Hfa0' : f a == g a).
    { transitivity (f (a + 0)).
      - symmetry. apply Pf. exact (idr a).
      - transitivity (g (a + 0)).
        + exact Hfa0.
        + apply Pg. exact (idr a). }
    clear Hfa0.
    rename Hfa0' into Hfa0.
    split.
    - exact Hb.
    - intros eps Heps.
      destruct (HD eps Heps) as [d2 [Hd2 HR]].
      destruct (pos_min_exists d d2 Hd Hd2) as [d0 [Hd0 [Hdd Hdd2]]].
      ∃ d0.
      split.
      + exact Hd0.
      + intros h Hh.
        rewrite (Hfg h (lt_le_trans (‖ h ‖) d0 d Hh Hdd)), Hfa0.
        exact (HR h (lt_le_trans (‖ h ‖) d0 d2 Hh Hdd2)).
  Qed.

  Lemma pos_add_of_norm_lt : ∀ a h : F, 0 < a → ‖ h ‖ < a → 0 < a + h.
  Proof.
    intros a h Ha Hh.
    destruct (le_total 0 h) as [H0h | Hh0].
    -
      assert (Hle : a + 0 ≤ a + h).
      { apply add_le_mono_l. exact H0h. }
      rewrite (idr a) in Hle.
      exact (lt_le_trans 0 a (a + h) Ha Hle).
    -
      assert (Hah : abs h == - h).
      { rewrite <- (abs_neg h). apply abs_nonneg_id. exact (neg_nonneg h Hh0). }
      rewrite field_vnorm_eq_abs, Hah in Hh.
      pose proof (add_lt_mono_l (- h) a h Hh) as Hstep.
      rewrite (invr h), (comm h a) in Hstep.
      exact Hstep.
  Qed.

  Lemma pos_neg_of_neg : ∀ a : F, a < 0 → 0 < - a.
  Proof.
    intros a [Hle Hne].
    split.
    - exact (neg_nonneg a Hle).
    - intros Heq.
      apply Hne.
      rewrite <- (neg_involutive a), <- Heq.
      apply neg_zero.
  Qed.

  Lemma neg_of_pos_neg : ∀ x : F, 0 < - x → x < 0.
  Proof.
    intros x [Hle Hne].
    split.
    - apply (le_of_sub_nonneg x 0).
      rewrite (idl (- x)).
      exact Hle.
    - intros Heq.
      apply Hne.
      rewrite Heq, neg_zero.
      reflexivity.
  Qed.

  Lemma neg_add_of_norm_lt : ∀ a h : F, a < 0 → ‖ h ‖ < - a → a + h < 0.
  Proof.
    intros a h Ha Hh.
    apply (neg_of_pos_neg (a + h)).
    rewrite (neg_add a h).
    apply (pos_add_of_norm_lt (- a) (- h)).
    - exact (pos_neg_of_neg a Ha).
    - rewrite (field_norm_neg h). exact Hh.
  Qed.

  Lemma lm_id_bounded_F : lm_bounded (@lm_id F F _ _).
  Proof.
    ∃ 1.
    split.
    - exact zero_le_one.
    - intros u.
      apply eq_le.
      symmetry.
      exact (ring_mul_1l (‖ u ‖)).
  Qed.

  Theorem is_deriv_relu_pos : ∀ (f : F → F),
    Proper (equ ==> equ) f →
    ∀ a : F, 0 < a → (∀ x : F, 0 < x → f x == x) →
    is_deriv f a lm_id.
  Proof.
    intros f Pf a Ha Hf.
    apply (is_deriv_of_local_eq f (fun x : F ⇒ x) Pf (fun x y Hxy ⇒ Hxy)
      a lm_id a Ha).
    - intros h Hh.
      apply Hf.
      exact (pos_add_of_norm_lt a h Ha Hh).
    - exact (deriv_lm lm_id a lm_id_bounded_F).
  Qed.

  Theorem is_deriv_relu_neg : ∀ (f : F → F),
    Proper (equ ==> equ) f →
    ∀ a : F, a < 0 → (∀ x : F, x < 0 → f x == 0) →
    is_deriv f a lm_const0.
  Proof.
    intros f Pf a Ha Hf.
    assert (Pconst0 : Proper (equ ==> equ) (fun _ : F ⇒ (0 : F))).
    { intros x y _. reflexivity. }
    apply (is_deriv_of_local_eq f (fun _ : F ⇒ 0) Pf Pconst0
      a lm_const0 (- a) (pos_neg_of_neg a Ha)).
    - intros h Hh.
      apply Hf.
      exact (neg_add_of_norm_lt a h Ha Hh).
    - exact (deriv_const 0 a).
  Qed.


End LocalAgreement.
