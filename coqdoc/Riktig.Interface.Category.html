From Riktig Require Import Ops.

Class Category (obj : Type) := {
  hom : obj → obj → Type;
  #[global] hom_equ :: ∀ x y, EqRel (hom x y);
  #[global] hom_equiv :: ∀ x y, Equivalence (hom_equ x y);
  cat_id : ∀ {x : obj}, hom x x;
  comp : ∀ {x y z : obj}, hom y z → hom x y → hom x z;
  #[global] comp_proper :: ∀ x y z, Proper (equ ==> equ ==> equ) (@comp x y z);
  cat_idl : ∀ (x y : obj) (f : hom x y), comp cat_id f == f;
  cat_idr : ∀ (x y : obj) (f : hom x y), comp f cat_id == f;
  cat_assoc : ∀ (w x y z : obj) (f : hom y z) (g : hom x y) (h : hom w x),
      comp f (comp g h) == comp (comp f g) h
}.

Notation "x ~> y" := (hom x y) (at level 70, right associativity).
Infix "∘" := comp (at level 40, left associativity).

Class Functor {C D : Type} (cC : Category C) (cD : Category D) := {
  fobj : C → D;
  fmap : ∀ {x y : C}, (x ~> y) → (fobj x ~> fobj y);
  #[global] fmap_proper :: ∀ x y : C, Proper (equ ==> equ) (@fmap x y);
  fmap_id : ∀ x : C, fmap (@cat_id C cC x) == cat_id;
  fmap_comp : ∀ (x y z : C) (f : y ~> z) (g : x ~> y),
      fmap (f ∘ g) == fmap f ∘ fmap g
}.

Class NaturalTransformation {C D : Type} {cC : Category C} {cD : Category D}
    (F G : Functor cC cD) := {
  eta : ∀ x : C, fobj (Functor:=F) x ~> fobj (Functor:=G) x;
  naturality : ∀ (x y : C) (f : x ~> y),
      fmap (Functor:=G) f ∘ eta x == eta y ∘ fmap (Functor:=F) f
}.

Class Initial {O : Type} (C : Category O) := {
  init : O;
  emit : ∀ x : O, init ~> x;
  emit_unique : ∀ (x : O) (f : init ~> x), f == emit x
}.

Class Terminal {O : Type} (C : Category O) := {
  term : O;
  bang : ∀ x : O, x ~> term;
  bang_unique : ∀ (x : O) (f : x ~> term), f == bang x
}.

Class HasProducts {O : Type} (C : Category O) := {
  prod_obj : O → O → O;
  pi1 : ∀ {a b : O}, prod_obj a b ~> a;
  pi2 : ∀ {a b : O}, prod_obj a b ~> b;
  pair_mor : ∀ {x a b : O}, (x ~> a) → (x ~> b) → (x ~> prod_obj a b);
  #[global] pair_proper :: ∀ x a b : O,
      Proper (equ ==> equ ==> equ) (@pair_mor x a b);
  pair_pi1 : ∀ (x a b : O) (f : x ~> a) (g : x ~> b),
      pi1 ∘ pair_mor f g == f;
  pair_pi2 : ∀ (x a b : O) (f : x ~> a) (g : x ~> b),
      pi2 ∘ pair_mor f g == g;
  pair_unique : ∀ (x a b : O) (f : x ~> a) (g : x ~> b)
                       (h : x ~> prod_obj a b),
      pi1 ∘ h == f → pi2 ∘ h == g → h == pair_mor f g
}.

Definition par_mor {O : Type} {C : Category O} `{@HasProducts O C}
  {a b c d : O} (f : a ~> c) (g : b ~> d) : prod_obj a b ~> prod_obj c d :=
  pair_mor (f ∘ pi1) (g ∘ pi2).

Notation "f ×× g" := (par_mor f g) (at level 41).

Class HasCoproducts {O : Type} (C : Category O) := {
  sum_obj : O → O → O;
  inj1 : ∀ {a b : O}, a ~> sum_obj a b;
  inj2 : ∀ {a b : O}, b ~> sum_obj a b;
  case_mor : ∀ {x a b : O}, (a ~> x) → (b ~> x) → (sum_obj a b ~> x);
  #[global] case_proper :: ∀ x a b : O,
      Proper (equ ==> equ ==> equ) (@case_mor x a b);
  case_inj1 : ∀ (x a b : O) (f : a ~> x) (g : b ~> x),
      case_mor f g ∘ inj1 == f;
  case_inj2 : ∀ (x a b : O) (f : a ~> x) (g : b ~> x),
      case_mor f g ∘ inj2 == g;
  case_unique : ∀ (x a b : O) (f : a ~> x) (g : b ~> x)
                       (h : sum_obj a b ~> x),
      h ∘ inj1 == f → h ∘ inj2 == g → h == case_mor f g
}.

Class HasExponentials {O : Type} (C : Category O) `{!HasProducts C} := {
  exp_obj : O → O → O;
  eval_mor : ∀ {a b : O}, prod_obj (exp_obj a b) a ~> b;
  curry : ∀ {x a b : O}, (prod_obj x a ~> b) → (x ~> exp_obj a b);
  #[global] curry_proper :: ∀ x a b : O,
      Proper (equ ==> equ) (@curry x a b);
  curry_beta : ∀ (x a b : O) (h : prod_obj x a ~> b),
      eval_mor ∘ (curry h ×× cat_id) == h;
  curry_unique : ∀ (x a b : O) (h : prod_obj x a ~> b)
                        (k : x ~> exp_obj a b),
      eval_mor ∘ (k ×× cat_id) == h → k == curry h
}.

Class CCC {O : Type} (C : Category O) := {
  #[global] ccc_term :: Terminal C;
  #[global] ccc_prod :: HasProducts C;
  #[global] ccc_exp :: @HasExponentials O C ccc_prod
}.
