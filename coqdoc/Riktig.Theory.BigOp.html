Fixpoint bigop {M : Type} `{Monoid M} (n : nat) (f : nat → M) : M :=
  match n with
  | O ⇒ 0
  | S m ⇒ f O + bigop m (fun i ⇒ f (S i))
  end.

Fixpoint bigopF {M : Type} `{Monoid M} {n : nat} (f : Fin.t n → M)
    {struct n} : M :=
  match n as n0 return ((Fin.t n0 → M) → M) with
  | O ⇒ fun _ ⇒ 0
  | S m ⇒ fun g ⇒ g Fin.F1 + bigopF (fun i ⇒ g (Fin.FS i))
  end f.


Section BigOpMonoid.
  Context {M : Type} `{Monoid M}.

  Theorem bigop_0 : ∀ f : nat → M, bigop O f == 0.
  Proof.
    intros f. reflexivity.
  Qed.

  Theorem bigop_S : ∀ n (f : nat → M),
    bigop (S n) f == f O + bigop n (fun i ⇒ f (S i)).
  Proof.
    intros n f. reflexivity.
  Qed.

  Theorem bigop_ext : ∀ n (f g : nat → M),
    (∀ i, f i == g i) → bigop n f == bigop n g.
  Proof.
    intros n; induction n as [| n IH]; intros f g Hfg.
    - reflexivity.
    - rewrite (bigop_S n f), (bigop_S n g), (Hfg O),
        (IH (fun i ⇒ f (S i)) (fun i ⇒ g (S i)) (fun i ⇒ Hfg (S i))).
      reflexivity.
  Qed.

  Theorem bigopF_0 : ∀ f : Fin.t O → M, bigopF f == 0.
  Proof.
    intros f. reflexivity.
  Qed.

  Theorem bigopF_S : ∀ n (f : Fin.t (S n) → M),
    bigopF f == f Fin.F1 + bigopF (fun i ⇒ f (Fin.FS i)).
  Proof.
    intros n f. reflexivity.
  Qed.

  Theorem bigopF_ext : ∀ n (f g : Fin.t n → M),
    (∀ i, f i == g i) → bigopF f == bigopF g.
  Proof.
    intros n; induction n as [| n IH]; intros f g Hfg.
    - reflexivity.
    - rewrite (bigopF_S n f), (bigopF_S n g), (Hfg Fin.F1),
        (IH (fun i ⇒ f (Fin.FS i)) (fun i ⇒ g (Fin.FS i))
            (fun i ⇒ Hfg (Fin.FS i))).
      reflexivity.
  Qed.

  Theorem bigop_zero : ∀ n : nat,
    bigop n (fun _ : nat ⇒ (0 : M)) == 0.
  Proof.
    intros n; induction n as [| n IH].
    - reflexivity.
    - rewrite (bigop_S n (fun _ : nat ⇒ (0 : M))), IH.
      exact (mon_idr 0).
  Qed.

  Theorem bigopF_zero : ∀ n : nat,
    bigopF (fun _ : Fin.t n ⇒ (0 : M)) == 0.
  Proof.
    intros n; induction n as [| n IH].
    - reflexivity.
    - rewrite (bigopF_S n (fun _ : Fin.t (S n) ⇒ (0 : M))), IH.
      exact (mon_idr 0).
  Qed.

  Theorem bigop_bigopF : ∀ n (f : nat → M),
    bigopF (fun i : Fin.t n ⇒ f (fin_to_nat i)) == bigop n f.
  Proof.
    intros n; induction n as [| n IH]; intros f.
    - reflexivity.
    - rewrite (bigopF_S n (fun i ⇒ f (fin_to_nat i))), (bigop_S n f),
        (IH (fun j ⇒ f (S j))).
      reflexivity.
  Qed.

End BigOpMonoid.

#[global] Instance bigop_proper {M : Type} `{Monoid M} (n : nat) :
  Proper (pointwise_relation nat equ ==> equ) (@bigop M _ n).
Proof.
  intros f g Hfg. apply bigop_ext. exact Hfg.
Qed.

#[global] Instance bigopF_proper {M : Type} `{Monoid M} (n : nat) :
  Proper (pointwise_relation (Fin.t n) equ ==> equ) (@bigopF M _ n).
Proof.
  intros f g Hfg. apply bigopF_ext. exact Hfg.
Qed.


Section BigOpCommMonoid.
  Context {M : Type} {MM : Monoid M} {MC : Commutes M equ add}.

  Lemma add_swap_front : ∀ a b c : M, a + (b + c) == b + (a + c).
  Proof.
    intros a b c.
    rewrite <- (mon_assoc a b c), (comm a b), (mon_assoc b a c).
    reflexivity.
  Qed.

  Theorem bigopF_extract : ∀ (m : nat) (p : Fin.t (S m)) (g : Fin.t (S m) → M),
    bigopF g == g p + bigopF (fun s : Fin.t m ⇒ g (skip p s)).
  Proof.
    intros m; induction m as [| m IH]; intros p g.
    - revert g.
      apply (Fin.caseS' p (fun p0 ⇒ ∀ g : Fin.t 1 → M,
        bigopF g == g p0 + bigopF (fun s : Fin.t 0 ⇒ g (skip p0 s)))).
      + intros g.
        rewrite (bigopF_S 0 g), (bigopF_0 (fun i : Fin.t 0 ⇒ g (Fin.FS i))),
          (bigopF_0 (fun s : Fin.t 0 ⇒ g (skip Fin.F1 s))).
        reflexivity.
      + intros p'.
        exact (Fin.case0 (fun p0 ⇒ ∀ g : Fin.t 1 → M,
          bigopF g == g (Fin.FS p0) + bigopF (fun s : Fin.t 0 ⇒ g (skip (Fin.FS p0) s)))
          p').
    - revert g.
      apply (Fin.caseS' p (fun p0 ⇒ ∀ g : Fin.t (S (S m)) → M,
        bigopF g == g p0 + bigopF (fun s : Fin.t (S m) ⇒ g (skip p0 s)))).
      + intros g.
        rewrite (bigopF_S (S m) g).
        apply mon_add_proper.
        × reflexivity.
        × apply bigopF_ext. intros s.
          rewrite (skip_F1 (S m) s). reflexivity.
      + intros p' g.
        rewrite (bigopF_S (S m) g), (IH p' (fun s ⇒ g (Fin.FS s))),
          (bigopF_S m (fun s : Fin.t (S m) ⇒ g (skip (Fin.FS p') s))),
          (skip_FS_F1 m p').
        exact (add_swap_front (g Fin.F1) (g (Fin.FS p'))
                 (bigopF (fun s' : Fin.t m ⇒ g (Fin.FS (skip p' s'))))).
  Qed.

  Theorem bigopF_extract2 : ∀ (n0 : nat) (p1 p2 : Fin.t (S (S n0)))
      (g : Fin.t (S (S n0)) → M),
    p1 ≠ p2 →
    (∀ c : Fin.t (S (S n0)), c ≠ p1 → c ≠ p2 → g c == 0) →
    bigopF g == g p1 + g p2.
  Proof.
    intros n0 p1 p2 g Hne Hzero.
    rewrite (bigopF_extract (S n0) p1 g).
    apply mon_add_proper.
    - reflexivity.
    - destruct (unskip_some (S n0) p1 p2 Hne) as [t [Hu Hsk]].
      rewrite (bigopF_extract n0 t (fun s ⇒ g (skip p1 s))).
      rewrite Hsk.
      transitivity (g p2 + (0 : M)).
      + apply mon_add_proper.
        × reflexivity.
        × transitivity (bigopF (fun _ : Fin.t n0 ⇒ (0 : M))).
          -- apply bigopF_ext. intros s'.
             apply Hzero.
             ++ exact (skip_ne (S n0) p1 (skip t s')).
             ++ intros Heq.
                apply (skip_ne n0 t s').
                apply (skip_inj (S n0) p1 (skip t s') t).
                rewrite Heq.
                exact (eq_sym Hsk).
          -- exact (bigopF_zero n0).
      + exact (mon_idr (g p2)).
  Qed.

  Theorem bigopF_reindex : ∀ (n : nat) (g : Fin.t n → M)
      (tau tau' : Fin.t n → Fin.t n),
    (∀ x, tau' (tau x) = x) → (∀ y, tau (tau' y) = y) →
    bigopF g == bigopF (fun i ⇒ g (tau i)).
  Proof.
    intros n; induction n as [| m IH]; intros g tau tau' H1 H2.
    - reflexivity.
    - transitivity (g (tau Fin.F1) + bigopF (fun s : Fin.t m ⇒ g (skip (tau Fin.F1) s))).
      + exact (bigopF_extract m (tau Fin.F1) g).
      + transitivity (g (tau Fin.F1) + bigopF (fun s : Fin.t m ⇒ g (tau (Fin.FS s)))).
        × apply mon_add_proper.
          -- reflexivity.
          -- transitivity (bigopF (fun s : Fin.t m ⇒
                 g (skip (tau Fin.F1) (bstep2 Fin.F1 tau tau' s)))).
             ++ exact (IH (fun t ⇒ g (skip (tau Fin.F1) t))
                         (bstep2 Fin.F1 tau tau') (bstep2' Fin.F1 tau tau')
                         (bstep2_bstep2' m Fin.F1 tau tau' H1 H2)
                         (bstep2'_bstep2 m Fin.F1 tau tau' H1 H2)).
             ++ apply bigopF_ext. intros s.
                rewrite (bstep2_spec m Fin.F1 tau tau' H1 H2 s), (skip_F1 m s).
                reflexivity.
        × symmetry. exact (bigopF_S m (fun i ⇒ g (tau i))).
  Qed.

End BigOpCommMonoid.


Section BigOpPointwise.
  Context {X M : Type} `{Monoid M}.

  Theorem bigopF_app : ∀ (n : nat) (f : Fin.t n → X → M) (x : X),
    bigopF f x == bigopF (fun i ⇒ f i x).
  Proof.
    intros n; induction n as [| n IH]; intros f x.
    - reflexivity.
    - exact (mon_add_proper
               (f Fin.F1 x) (f Fin.F1 x) (reflexivity (f Fin.F1 x))
               (bigopF (fun i ⇒ f (Fin.FS i)) x)
               (bigopF (fun i ⇒ f (Fin.FS i) x))
               (IH (fun i ⇒ f (Fin.FS i)) x)).
  Qed.

End BigOpPointwise.

Section BigOpAbGroup.
  Context {G : Type} `{AbGroup G}.

  Theorem bigop_add : ∀ n (f g : nat → G),
    bigop n (fun i ⇒ f i + g i) == bigop n f + bigop n g.
  Proof.
    intros n; induction n as [| n IH]; intros f g.
    - rewrite (bigop_0 (fun i ⇒ f i + g i)), (bigop_0 f), (bigop_0 g), idl.
      reflexivity.
    - rewrite (bigop_S n (fun i ⇒ f i + g i)), (bigop_S n f), (bigop_S n g),
        (IH (fun i ⇒ f (S i)) (fun i ⇒ g (S i))),
        (add_interchange (f O) (bigop n (fun i ⇒ f (S i)))
           (g O) (bigop n (fun i ⇒ g (S i)))).
      reflexivity.
  Qed.

  Theorem bigopF_add : ∀ n (f g : Fin.t n → G),
    bigopF (fun i ⇒ f i + g i) == bigopF f + bigopF g.
  Proof.
    intros n; induction n as [| n IH]; intros f g.
    - rewrite (bigopF_0 (fun i ⇒ f i + g i)), (bigopF_0 f), (bigopF_0 g), idl.
      reflexivity.
    - rewrite (bigopF_S n (fun i ⇒ f i + g i)), (bigopF_S n f),
        (bigopF_S n g),
        (IH (fun i ⇒ f (Fin.FS i)) (fun i ⇒ g (Fin.FS i))),
        (add_interchange (f Fin.F1) (bigopF (fun i ⇒ f (Fin.FS i)))
           (g Fin.F1) (bigopF (fun i ⇒ g (Fin.FS i)))).
      reflexivity.
  Qed.

  Theorem bigopF_neg : ∀ (n : nat) (f : Fin.t n → G),
    bigopF (fun i ⇒ - f i) == - bigopF f.
  Proof.
    intros n; induction n as [| n IH]; intros f.
    - rewrite (bigopF_0 (fun i ⇒ - f i)), (bigopF_0 f).
      symmetry. exact neg_zero.
    - rewrite (bigopF_S n (fun i ⇒ - f i)), (bigopF_S n f),
        (IH (fun i ⇒ f (Fin.FS i))).
      symmetry.
      exact (neg_add (f Fin.F1) (bigopF (fun i ⇒ f (Fin.FS i)))).
  Qed.

  Theorem bigop_swap : ∀ (m n : nat) (f : nat → nat → G),
    bigop m (fun i ⇒ bigop n (fun j ⇒ f i j))
      == bigop n (fun j ⇒ bigop m (fun i ⇒ f i j)).
  Proof.
    intros m; induction m as [| m IH]; intros n f.
    - symmetry.
      exact (bigop_zero n).
    - rewrite (bigop_S m (fun i ⇒ bigop n (fun j ⇒ f i j))),
        (IH n (fun i j ⇒ f (S i) j)).
      symmetry.
      exact (bigop_add n (fun j ⇒ f O j)
               (fun j ⇒ bigop m (fun i ⇒ f (S i) j))).
  Qed.

  Theorem bigopF_swap : ∀ (m n : nat) (f : Fin.t m → Fin.t n → G),
    bigopF (fun i ⇒ bigopF (fun j ⇒ f i j))
      == bigopF (fun j ⇒ bigopF (fun i ⇒ f i j)).
  Proof.
    intros m; induction m as [| m IH]; intros n f.
    - symmetry.
      exact (bigopF_zero n).
    - rewrite (bigopF_S m (fun i ⇒ bigopF (fun j ⇒ f i j))),
        (IH n (fun i j ⇒ f (Fin.FS i) j)).
      symmetry.
      exact (bigopF_add n (fun j ⇒ f Fin.F1 j)
               (fun j ⇒ bigopF (fun i ⇒ f (Fin.FS i) j))).
  Qed.

End BigOpAbGroup.

Section BigOpRing.
  Context {R : Type} `{Ring R}.

  Theorem bigop_mul_l : ∀ n (a : R) (f : nat → R),
    bigop n (fun i ⇒ a × f i) == a × bigop n f.
  Proof.
    intros n; induction n as [| n IH]; intros a f.
    - rewrite (bigop_0 (fun i ⇒ a × f i)), (bigop_0 f), (mul_0_r a).
      reflexivity.
    - rewrite (bigop_S n (fun i ⇒ a × f i)), (bigop_S n f),
        (IH a (fun i ⇒ f (S i))),
        (ring_distl a (f O) (bigop n (fun i ⇒ f (S i)))).
      reflexivity.
  Qed.

  Theorem bigop_mul_r : ∀ n (a : R) (f : nat → R),
    bigop n (fun i ⇒ f i × a) == bigop n f × a.
  Proof.
    intros n; induction n as [| n IH]; intros a f.
    - rewrite (bigop_0 (fun i ⇒ f i × a)), (bigop_0 f), (mul_0_l a).
      reflexivity.
    - rewrite (bigop_S n (fun i ⇒ f i × a)), (bigop_S n f),
        (IH a (fun i ⇒ f (S i))),
        (ring_distr (f O) (bigop n (fun i ⇒ f (S i))) a).
      reflexivity.
  Qed.

  Theorem bigopF_mul_l : ∀ n (a : R) (f : Fin.t n → R),
    bigopF (fun i ⇒ a × f i) == a × bigopF f.
  Proof.
    intros n; induction n as [| n IH]; intros a f.
    - rewrite (bigopF_0 (fun i ⇒ a × f i)), (bigopF_0 f), (mul_0_r a).
      reflexivity.
    - rewrite (bigopF_S n (fun i ⇒ a × f i)), (bigopF_S n f),
        (IH a (fun i ⇒ f (Fin.FS i))),
        (ring_distl a (f Fin.F1) (bigopF (fun i ⇒ f (Fin.FS i)))).
      reflexivity.
  Qed.

  Theorem bigopF_mul_r : ∀ n (a : R) (f : Fin.t n → R),
    bigopF (fun i ⇒ f i × a) == bigopF f × a.
  Proof.
    intros n; induction n as [| n IH]; intros a f.
    - rewrite (bigopF_0 (fun i ⇒ f i × a)), (bigopF_0 f), (mul_0_l a).
      reflexivity.
    - rewrite (bigopF_S n (fun i ⇒ f i × a)), (bigopF_S n f),
        (IH a (fun i ⇒ f (Fin.FS i))),
        (ring_distr (f Fin.F1) (bigopF (fun i ⇒ f (Fin.FS i))) a).
      reflexivity.
  Qed.


  Fixpoint prodF {n : nat} (f : Fin.t n → R) {struct n} : R :=
    match n as n0 return (Fin.t n0 → R) → R with
    | O ⇒ fun _ ⇒ 1
    | S m ⇒ fun g ⇒ g Fin.F1 × prodF (fun i ⇒ g (Fin.FS i))
    end f.

  Lemma prodF_0 : ∀ f : Fin.t O → R, prodF f == 1.
  Proof. intros f. reflexivity. Qed.

  Lemma prodF_S : ∀ (m : nat) (f : Fin.t (S m) → R),
    prodF f == f Fin.F1 × prodF (fun i ⇒ f (Fin.FS i)).
  Proof. intros m f. reflexivity. Qed.

  Lemma prodF_ext : ∀ (n : nat) (f g : Fin.t n → R),
    (∀ i, f i == g i) → prodF f == prodF g.
  Proof.
    intros n; induction n as [| n IH]; intros f g Hfg.
    - reflexivity.
    - rewrite (prodF_S n f), (prodF_S n g), (Hfg Fin.F1),
        (IH (fun i ⇒ f (Fin.FS i)) (fun i ⇒ g (Fin.FS i))
            (fun i ⇒ Hfg (Fin.FS i))).
      reflexivity.
  Qed.

End BigOpRing.

Section BigOpVectorSpace.
  Context {F V : Type} `{VectorSpace F V}.

  Theorem bigop_scale : ∀ n (a : F) (f : nat → V),
    bigop n (fun i ⇒ a *: f i) == a *: bigop n f.
  Proof.
    intros n; induction n as [| n IH]; intros a f.
    - rewrite (bigop_0 (fun i ⇒ a *: f i)), (bigop_0 f), (scale_zero a).
      reflexivity.
    - rewrite (bigop_S n (fun i ⇒ a *: f i)), (bigop_S n f),
        (IH a (fun i ⇒ f (S i))),
        (sc_scale_add a (f O) (bigop n (fun i ⇒ f (S i)))).
      reflexivity.
  Qed.

  Theorem bigopF_scale : ∀ n (a : F) (f : Fin.t n → V),
    bigopF (fun i ⇒ a *: f i) == a *: bigopF f.
  Proof.
    intros n; induction n as [| n IH]; intros a f.
    - rewrite (bigopF_0 (fun i ⇒ a *: f i)), (bigopF_0 f), (scale_zero a).
      reflexivity.
    - rewrite (bigopF_S n (fun i ⇒ a *: f i)), (bigopF_S n f),
        (IH a (fun i ⇒ f (Fin.FS i))),
        (sc_scale_add a (f Fin.F1) (bigopF (fun i ⇒ f (Fin.FS i)))).
      reflexivity.
  Qed.

  Theorem bigop_scale_r : ∀ (n : nat) (f : nat → F) (v : V),
    bigop n (fun i ⇒ f i *: v) == bigop n f *: v.
  Proof.
    intros n; induction n as [| n IH]; intros f v.
    - symmetry.
      exact (zero_scale v).
    - rewrite (bigop_S n (fun i ⇒ f i *: v)), (IH (fun i ⇒ f (S i)) v).
      symmetry.
      exact (sc_add_scale (f O) (bigop n (fun i ⇒ f (S i))) v).
  Qed.

  Theorem bigopF_scale_r : ∀ (n : nat) (f : Fin.t n → F) (v : V),
    bigopF (fun i ⇒ f i *: v) == bigopF f *: v.
  Proof.
    intros n; induction n as [| n IH]; intros f v.
    - symmetry.
      exact (zero_scale v).
    - rewrite (bigopF_S n (fun i ⇒ f i *: v)),
        (IH (fun i ⇒ f (Fin.FS i)) v).
      symmetry.
      exact (sc_add_scale (f Fin.F1) (bigopF (fun i ⇒ f (Fin.FS i))) v).
  Qed.

End BigOpVectorSpace.


Section BigOpOrder.
  Context {F : Type} `{OrderedField F}.

  Lemma bigopF_nonneg : ∀ (n : nat) (f : Fin.t n → F),
    (∀ i, 0 ≤ f i) → 0 ≤ bigopF f.
  Proof.
    intros n; induction n as [| n IH]; intros f Hf.
    - apply eq_le. symmetry. exact (bigopF_0 f).
    - rewrite (bigopF_S n f).
      apply (le_trans 0 (0 + 0) (f Fin.F1 + bigopF (fun i ⇒ f (Fin.FS i)))).
      + apply eq_le. symmetry. apply mon_idl.
      + apply add_le_mono.
        × exact (Hf Fin.F1).
        × apply IH. intros i. exact (Hf (Fin.FS i)).
  Qed.

  Lemma bigopF_le : ∀ (n : nat) (f g : Fin.t n → F),
    (∀ i, f i ≤ g i) → bigopF f ≤ bigopF g.
  Proof.
    intros n; induction n as [| n IH]; intros f g Hfg.
    - rewrite (bigopF_0 f), (bigopF_0 g). apply le_refl.
    - rewrite (bigopF_S n f), (bigopF_S n g).
      apply add_le_mono.
      + exact (Hfg Fin.F1).
      + apply IH. intros i. exact (Hfg (Fin.FS i)).
  Qed.

  Lemma bigopF_nonneg_zero : ∀ (n : nat) (f : Fin.t n → F),
    (∀ i, 0 ≤ f i) → bigopF f == 0 → ∀ i, f i == 0.
  Proof.
    intros n; induction n as [| n IH]; intros f Hf Hsum i.
    - exact (Fin.case0 (fun i0 ⇒ f i0 == 0) i).
    - rewrite (bigopF_S n f) in Hsum.
      destruct (nonneg_add_eq_zero (f Fin.F1) (bigopF (fun j ⇒ f (Fin.FS j)))
        (Hf Fin.F1)
        (bigopF_nonneg n (fun j ⇒ f (Fin.FS j)) (fun j ⇒ Hf (Fin.FS j)))
        Hsum) as [H1 H2].
      apply (Fin.caseS' i (fun i0 ⇒ f i0 == 0)).
      + exact H1.
      + intros i'. exact (IH (fun j ⇒ f (Fin.FS j)) (fun j ⇒ Hf (Fin.FS j))
          H2 i').
  Qed.

  Lemma bigopF_le_one_term : ∀ (n : nat) (f : Fin.t n → F)
      (j : Fin.t n), (∀ i, 0 ≤ f i) → f j ≤ bigopF f.
  Proof.
    intros n; induction n as [| n IH]; intros f j Hf.
    - exact (Fin.case0 (fun j0 ⇒ f j0 ≤ bigopF f) j).
    - rewrite (bigopF_S n f).
      apply (Fin.caseS' j
        (fun j0 ⇒ f j0 ≤ f Fin.F1 + bigopF (fun i ⇒ f (Fin.FS i)))).
      + apply le_add_nonneg_r.
        apply bigopF_nonneg. intros i. exact (Hf (Fin.FS i)).
      + intros j'.
        apply (le_trans (f (Fin.FS j')) (bigopF (fun i ⇒ f (Fin.FS i)))
          (f Fin.F1 + bigopF (fun i ⇒ f (Fin.FS i)))).
        × exact (IH (fun i ⇒ f (Fin.FS i)) j'
            (fun i ⇒ Hf (Fin.FS i))).
        × apply le_add_nonneg_l. exact (Hf Fin.F1).
  Qed.

  Lemma abs_bigopF_le : ∀ (n : nat) (f : Fin.t n → F),
    abs (bigopF f) ≤ bigopF (fun i ⇒ abs (f i)).
  Proof.
    intros n; induction n as [| n IH]; intros f.
    - rewrite (bigopF_0 f), (bigopF_0 (fun i ⇒ abs (f i))), abs_zero.
      apply le_refl.
    - rewrite (bigopF_S n f), (bigopF_S n (fun i ⇒ abs (f i))).
      apply (le_trans (abs (f Fin.F1 + bigopF (fun i ⇒ f (Fin.FS i))))
        (abs (f Fin.F1) + abs (bigopF (fun i ⇒ f (Fin.FS i))))
        (abs (f Fin.F1) + bigopF (fun i ⇒ abs (f (Fin.FS i))))).
      + apply abs_triangle.
      + apply add_le_mono.
        × apply le_refl.
        × apply IH.
  Qed.

End BigOpOrder.
