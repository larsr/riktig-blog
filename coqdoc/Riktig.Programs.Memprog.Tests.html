Local Open Scope Z_scope.


Definition test_input : list Z := [5; 3; 8; 1; 9; 2; 7; 4; 6].
Definition test_sorted : list Z := [1; 2; 3; 4; 5; 6; 7; 8; 9].


Definition sort_l (sorter : Z → Z → MemT Z Z unit) (fuel : nat)
  (xs : list Z) : option (list Z) :=
  let prog :=
    do! base <- list_to_mem Z.of_nat xs;
    sorter base (Z.of_nat (length xs)) >>
    ret_t base in
  let '(r, m) := run_z fuel prog 256 in
  mem_to_list m r (length xs).


Definition sort_mc (xs : list Z) : option (list Z) :=
  let n := Z.of_nat (length xs) in
  let prog :=
    do! (data_base, _) <-
      setup_and_run_mc 128 xs qsort_mc
        (fun base rf ⇒ set_reg R0 base (set_reg R1 0 (set_reg R2 (n - 1) rf)))
        (20 × 1000);
    ret_t data_base in
  let '(r, m) := run_z (400 × 1000) prog 1024 in
  mem_to_list m r (length xs).


Example qsort_l1_sorts :
  sort_l call_qsort_l1 (100 × 1000) test_input = Some test_sorted.
Proof. vm_compute. reflexivity. Qed.
