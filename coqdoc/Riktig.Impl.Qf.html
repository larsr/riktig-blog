Lemma Qplus_opp_l : ∀ q : Q, (- q + q == 0#1)%Q.
Proof.
  intros q.
  calc (- q + q)%Q
    == (q + - q)%Q { Qplus_comm }
    == (0#1)%Q { Qplus_opp_r }.
Qed.

Lemma Qmult_inv_l : ∀ q : Q, ¬ (q == 0#1)%Q → (/ q × q == 1#1)%Q.
Proof.
  intros q Hq.
  calc (/ q × q)%Q
    == (q × / q)%Q { Qmult_comm }
    == (1#1)%Q { (Qmult_inv_r q Hq) }.
Qed.

  ((1#1)%Q :: (0#1)%Q :: (0#1)%Q :: nil) == ((1#1)%Q :: nil).

  intros i.
  destruct i as [| [| [| j]]]; reflexivity.
