From Coq Require Import List Lia.
From Riktig.Channel Require Import Trace ABP.
Import ListNotations.

Section Fairness.
  Context {M : Type}.



  Lemma burst_delivery_exact : ∀ (b : bit) (m : M) (k : nat), k > 0 →
    deliver b (sender_burst b m k) = [m] ∧ next_bit b (sender_burst b m k) = negb b.
  Proof.
    intros b m k Hk.
    destruct k as [| k'].
    - exfalso. lia.
    - simpl.
      assert (Hall : ∀ p, In p (sender_burst b m k') → p = {| pkt_bit := b; pkt_msg := m |}).
      { intros p Hp. apply (sender_burst_all_eq b m k' p Hp). }
      assert (Hneq : pkt_bit ({| pkt_bit := b; pkt_msg := m |}) ≠ negb b).
      { simpl. intros Heq. apply (negb_neq b). symmetry. exact Heq. }
      rewrite (eqb_bit_true b b eq_refl). simpl.
      destruct (deliver_homog_neq {| pkt_bit := b; pkt_msg := m |} (sender_burst b m k') (negb b) Hall Hneq)
        as [Hd Hn].
      split.
      + rewrite Hd. reflexivity.
      + rewrite Hn. reflexivity.
  Qed.


  Lemma abp_collapse_perfect_gen : ∀ (k : nat) (ms : list M) (b : bit),
    k > 0 → deliver b (sender_stream b ms k) = ms.
  Proof.
    intros k ms.
    induction ms as [| m ms' IH]; intros b Hk.
    - simpl. reflexivity.
    - simpl.
      rewrite (deliver_app (sender_burst b m k) (sender_stream (negb b) ms' k) b).
      destruct (burst_delivery_exact b m k Hk) as [Hd Hn].
      rewrite Hd. rewrite Hn. simpl.
      rewrite (IH (negb b) Hk).
      reflexivity.
  Qed.

  Theorem abp_collapse_perfect : ∀ (k : nat) (b0 : bit) (ms : list M) (wire : list Packet),
    k > 0 → perfect (sender_stream b0 ms k) wire → deliver b0 wire = ms.
  Proof.
    intros k b0 ms wire Hk Hp.
    unfold perfect in Hp.
    rewrite Hp.
    apply (abp_collapse_perfect_gen k ms b0 Hk).
  Qed.



  Fixpoint fair_wire (b : bit) (ms : list M) (k : nat) (wire : list Packet) : Prop :=
    match ms with
    | [] ⇒ wire = []
    | m :: ms' ⇒
        ∃ w1 w2,
          wire = w1 ++ w2 ∧
          lossy_dup (sender_burst b m k) w1 ∧
          w1 ≠ [] ∧
          fair_wire (negb b) ms' k w2
    end.


  Lemma burst_delivery_fair : ∀ (b : bit) (m : M) (k : nat) (w : list Packet),
    lossy_dup (sender_burst b m k) w → w ≠ [] →
    deliver b w = [m] ∧ next_bit b w = negb b.
  Proof.
    intros b m k w H Hne.
    assert (Hall : ∀ p, In p w → p = {| pkt_bit := b; pkt_msg := m |}).
    { apply (lossy_dup_preserves_all_eq {| pkt_bit := b; pkt_msg := m |}
               (sender_burst b m k) w H).
      intros p Hp. apply (sender_burst_all_eq b m k p Hp). }
    destruct w as [| p w'].
    - destruct (Hne eq_refl).
    - assert (Hp : p = {| pkt_bit := b; pkt_msg := m |}).
      { apply Hall. simpl. left. reflexivity. }
      rewrite Hp. simpl.
      rewrite (eqb_bit_true b b eq_refl). simpl.
      assert (Hall' : ∀ q, In q w' → q = {| pkt_bit := b; pkt_msg := m |}).
      { intros q Hq. apply Hall. simpl. right. exact Hq. }
      assert (Hneq : pkt_bit ({| pkt_bit := b; pkt_msg := m |}) ≠ negb b).
      { simpl. intros Heq. apply (negb_neq b). symmetry. exact Heq. }
      destruct (deliver_homog_neq {| pkt_bit := b; pkt_msg := m |} w' (negb b) Hall' Hneq)
        as [Hd Hn].
      rewrite Hd. rewrite Hn.
      split; reflexivity.
  Qed.


  Lemma fair_wire_lossy_dup : ∀ (k : nat) (ms : list M) (b : bit) (wire : list Packet),
    fair_wire b ms k wire → lossy_dup (sender_stream b ms k) wire.
  Proof.
    intros k ms.
    induction ms as [| m ms' IH]; intros b wire H.
    - simpl in H. rewrite H. simpl. apply lossy_dup_nil.
    - simpl in H. destruct H as [w1 [w2 [Hwire [H1 [Hne H2]]]]].
      rewrite Hwire. simpl.
      apply lossy_dup_app.
      + exact H1.
      + apply (IH (negb b) w2 H2).
  Qed.


  Lemma abp_liveness_gen : ∀ (k : nat) (ms : list M) (b : bit) (wire : list Packet),
    fair_wire b ms k wire → deliver b wire = ms.
  Proof.
    intros k ms.
    induction ms as [| m ms' IH]; intros b wire H.
    - simpl in H. rewrite H. reflexivity.
    - simpl in H. destruct H as [w1 [w2 [Hwire [H1 [Hne H2]]]]].
      rewrite Hwire.
      rewrite (deliver_app w1 w2 b).
      destruct (burst_delivery_fair b m k w1 H1 Hne) as [Hd Hn].
      rewrite Hd. rewrite Hn. simpl.
      rewrite (IH (negb b) w2 H2).
      reflexivity.
  Qed.

  Theorem abp_liveness : ∀ (k : nat) (b0 : bit) (ms : list M) (wire : list Packet),
    fair_wire b0 ms k wire → deliver b0 wire = ms.
  Proof.
    intros k b0 ms wire H.
    apply (abp_liveness_gen k ms b0 wire H).
  Qed.

End Fairness.
