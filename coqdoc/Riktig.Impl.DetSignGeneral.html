Section PermPostcomposeCompose.
  Context {F : Type} {FF : Field F}.

  Theorem perm_postcompose_compose : ∀ (n : nat) (sigma : Perm n)
      (tau1 tau1' tau2 tau2' : Fin.t n → Fin.t n),
    (∀ x, tau1' (tau1 x) = x) → (∀ y, tau1 (tau1' y) = y) →
    (∀ x, tau2' (tau2 x) = x) → (∀ y, tau2 (tau2' y) = y) →
    perm_sign (perm_postcompose (perm_postcompose sigma tau1 tau1') tau2 tau2')
      == perm_sign (perm_postcompose sigma (fun x ⇒ tau2 (tau1 x)) (fun x ⇒ tau1' (tau2' x))).
  Proof.
    intros n sigma; induction sigma as [| m c p IH];
      intros tau1 tau1' tau2 tau2' Ha1 Ha2 Hb1 Hb2.
    - reflexivity.
    - rewrite (perm_postcompose_S m c p tau1 tau1'),
        (perm_postcompose_S m (tau1 c)
           (perm_postcompose p (bstep2 c tau1 tau1') (bstep2' c tau1 tau1'))
           tau2 tau2'),
        (perm_sign_S m (tau2 (tau1 c))
           (perm_postcompose
              (perm_postcompose p (bstep2 c tau1 tau1') (bstep2' c tau1 tau1'))
              (bstep2 (tau1 c) tau2 tau2') (bstep2' (tau1 c) tau2 tau2'))),
        (perm_sign_S m ((fun x ⇒ tau2 (tau1 x)) c) (perm_postcompose p
           (bstep2 c (fun x ⇒ tau2 (tau1 x)) (fun x ⇒ tau1' (tau2' x)))
           (bstep2' c (fun x ⇒ tau2 (tau1 x)) (fun x ⇒ tau1' (tau2' x))))).
      apply ring_mul_proper.
      + reflexivity.
      + transitivity (perm_sign (perm_postcompose p
          (fun x ⇒ bstep2 (tau1 c) tau2 tau2' (bstep2 c tau1 tau1' x))
          (fun x ⇒ bstep2' c tau1 tau1' (bstep2' (tau1 c) tau2 tau2' x)))).
        × exact (IH (bstep2 c tau1 tau1') (bstep2' c tau1 tau1')
                   (bstep2 (tau1 c) tau2 tau2') (bstep2' (tau1 c) tau2 tau2')
                   (bstep2_bstep2' m c tau1 tau1' Ha1 Ha2)
                   (bstep2'_bstep2 m c tau1 tau1' Ha1 Ha2)
                   (bstep2_bstep2' m (tau1 c) tau2 tau2' Hb1 Hb2)
                   (bstep2'_bstep2 m (tau1 c) tau2 tau2' Hb1 Hb2)).
        × exact (perm_sign_postcompose_ext m p
            (fun x ⇒ bstep2 (tau1 c) tau2 tau2' (bstep2 c tau1 tau1' x))
            (bstep2 c (fun x ⇒ tau2 (tau1 x)) (fun x ⇒ tau1' (tau2' x)))
            (fun x ⇒ bstep2' c tau1 tau1' (bstep2' (tau1 c) tau2 tau2' x))
            (bstep2' c (fun x ⇒ tau2 (tau1 x)) (fun x ⇒ tau1' (tau2' x)))
            (fun x ⇒ eq_sym (bstep2_compose m c tau1 tau1' tau2 tau2' Ha1 Ha2 Hb1 Hb2 x))).
  Qed.

End PermPostcomposeCompose.



Section SwapSignFlipAdjacentGen.
  Context {F : Type} {FF : Field F}.

  Theorem swap_sign_flip_adjacent_gen : ∀ (n0 : nat) (i j : Fin.t (S n0)),
    fin_to_nat j = S (fin_to_nat i) →
    ∀ sigma : Perm (S n0),
    perm_sign (perm_postcompose sigma (swap_fin i j) (swap_fin i j)) == - perm_sign sigma.
  Proof.
    intros n0 i j Hgap sigma.
    pose proof (fin_to_nat_bound (S n0) j) as Hbound.
    assert (Hlt : (fin_to_nat i < n0)%nat) by lia.
    destruct (fweaken_surj n0 i Hlt) as [k Hik].
    assert (Hjk : j = Fin.FS k).
    { apply (fin_to_nat_inj (S n0)).
      rewrite (fin_to_nat_FS n0 k), <- (fin_to_nat_fweaken n0 k), <- Hik.
      exact Hgap. }
    rewrite Hik, Hjk.
    exact (perm_sign_postcompose_adj n0 k sigma).
  Qed.

End SwapSignFlipAdjacentGen.


Section PermPostcomposeCompose3.
  Context {F : Type} {FF : Field F}.

  Theorem perm_postcompose_compose3 : ∀ (n : nat) (sigma : Perm n)
      (tau1 tau1' tau2 tau2' tau3 tau3' : Fin.t n → Fin.t n),
    (∀ x, tau1' (tau1 x) = x) → (∀ y, tau1 (tau1' y) = y) →
    (∀ x, tau2' (tau2 x) = x) → (∀ y, tau2 (tau2' y) = y) →
    (∀ x, tau3' (tau3 x) = x) → (∀ y, tau3 (tau3' y) = y) →
    perm_sign (perm_postcompose
                 (perm_postcompose (perm_postcompose sigma tau1 tau1') tau2 tau2')
                 tau3 tau3')
      == perm_sign (perm_postcompose sigma
                      (fun x ⇒ tau3 (tau2 (tau1 x)))
                      (fun x ⇒ tau1' (tau2' (tau3' x)))).
  Proof.
    intros n sigma tau1 tau1' tau2 tau2' tau3 tau3' Ha1 Ha2 Hb1 Hb2 Hc1 Hc2.
    assert (Hcomp1 : ∀ x : Fin.t n, tau2' (tau3' (tau3 (tau2 x))) = x).
    { intros x. rewrite (Hc1 (tau2 x)). exact (Hb1 x). }
    assert (Hcomp2 : ∀ y : Fin.t n, tau3 (tau2 (tau2' (tau3' y))) = y).
    { intros y. rewrite (Hb2 (tau3' y)). exact (Hc2 y). }
    transitivity (perm_sign (perm_postcompose (perm_postcompose sigma tau1 tau1')
                    (fun x ⇒ tau3 (tau2 x)) (fun x ⇒ tau2' (tau3' x)))).
    - exact (perm_postcompose_compose n (perm_postcompose sigma tau1 tau1')
               tau2 tau2' tau3 tau3' Hb1 Hb2 Hc1 Hc2).
    - exact (perm_postcompose_compose n sigma tau1 tau1'
               (fun x ⇒ tau3 (tau2 x)) (fun x ⇒ tau2' (tau3' x))
               Ha1 Ha2 Hcomp1 Hcomp2).
  Qed.

End PermPostcomposeCompose3.


Section SwapSignFlipGap.
  Context {F : Type} {FF : Field F}.

  Theorem swap_sign_flip_gap : ∀ (g N : nat) (i j : Fin.t N),
    fin_to_nat j = (fin_to_nat i + S g)%nat →
    ∀ sigma : Perm N,
    perm_sign (perm_postcompose sigma (swap_fin i j) (swap_fin i j)) == - perm_sign sigma.
  Proof.
    induction g as [| g IH]; intros N i j Hgap sigma.
    - assert (Hgap' : fin_to_nat j = S (fin_to_nat i)) by lia.
      destruct N as [| n0].
      + exact (Fin.case0
          (fun i0 : Fin.t 0 ⇒
             perm_sign (perm_postcompose sigma (swap_fin i0 j) (swap_fin i0 j))
               == - perm_sign sigma) i).
      + exact (swap_sign_flip_adjacent_gen n0 i j Hgap' sigma).
    - destruct N as [| n0].
      + exact (Fin.case0
          (fun i0 : Fin.t 0 ⇒
             perm_sign (perm_postcompose sigma (swap_fin i0 j) (swap_fin i0 j))
               == - perm_sign sigma) i).
      + pose proof (fin_to_nat_bound (S n0) j) as Hbound.
        assert (Hlt : (fin_to_nat i < n0)%nat) by lia.
        destruct (fweaken_surj n0 i Hlt) as [i' Hi'].
        set (k := Fin.FS i').
        assert (Hik_adj : fin_to_nat k = S (fin_to_nat i)).
        { unfold k. rewrite (fin_to_nat_FS n0 i'), Hi', (fin_to_nat_fweaken n0 i').
          reflexivity. }
        assert (Hkj : fin_to_nat j = (fin_to_nat k + S g)%nat) by lia.
        assert (Hik : i ≠ k).
        { intros Heq. rewrite Heq in Hik_adj. lia. }
        assert (Hjk : j ≠ k).
        { intros Heq. rewrite Heq, Hik_adj in Hgap. lia. }
        assert (Hij : i ≠ j).
        { intros Heq. rewrite Heq in Hgap. lia. }
        set (tauA := swap_fin i k).
        set (tauB := swap_fin k j).
        assert (Hconj : perm_sign (perm_postcompose sigma (swap_fin i j) (swap_fin i j))
          == perm_sign (perm_postcompose sigma
               (fun x ⇒ tauA (tauB (tauA x))) (fun x ⇒ tauA (tauB (tauA x))))).
        { apply (perm_sign_postcompose_ext (S n0) sigma (swap_fin i j)
            (fun x ⇒ tauA (tauB (tauA x))) (swap_fin i j) (fun x ⇒ tauA (tauB (tauA x)))).
          intros x. exact (swap_fin_conj (S n0) i j k Hij Hik Hjk x). }
        assert (Hexpand : perm_sign (perm_postcompose sigma
              (fun x ⇒ tauA (tauB (tauA x))) (fun x ⇒ tauA (tauB (tauA x))))
          == perm_sign (perm_postcompose
                (perm_postcompose (perm_postcompose sigma tauA tauA) tauB tauB) tauA tauA)).
        { symmetry.
          exact (perm_postcompose_compose3 (S n0) sigma tauA tauA tauB tauB tauA tauA
                   (swap_fin_invol (S n0) i k) (swap_fin_invol (S n0) i k)
                   (swap_fin_invol (S n0) k j) (swap_fin_invol (S n0) k j)
                   (swap_fin_invol (S n0) i k) (swap_fin_invol (S n0) i k)). }
        assert (HstepA : perm_sign (perm_postcompose
                (perm_postcompose (perm_postcompose sigma tauA tauA) tauB tauB) tauA tauA)
          == - perm_sign (perm_postcompose (perm_postcompose sigma tauA tauA) tauB tauB)).
        { exact (swap_sign_flip_adjacent_gen n0 i k Hik_adj
                   (perm_postcompose (perm_postcompose sigma tauA tauA) tauB tauB)). }
        assert (HstepB : perm_sign (perm_postcompose (perm_postcompose sigma tauA tauA) tauB tauB)
          == - perm_sign (perm_postcompose sigma tauA tauA)).
        { exact (IH (S n0) k j Hkj (perm_postcompose sigma tauA tauA)). }
        assert (HstepC : perm_sign (perm_postcompose sigma tauA tauA) == - perm_sign sigma).
        { exact (swap_sign_flip_adjacent_gen n0 i k Hik_adj sigma). }
        rewrite Hconj, Hexpand, HstepA, HstepB,
          (neg_involutive (perm_sign (perm_postcompose sigma tauA tauA))).
        exact HstepC.
  Qed.

End SwapSignFlipGap.


Section PermSignPostcomposeSwap.
  Context {F : Type} {FF : Field F}.

  Theorem perm_sign_postcompose_swap : ∀ (N : nat) (i j : Fin.t N),
    i ≠ j → ∀ sigma : Perm N,
    perm_sign (perm_postcompose sigma (swap_fin i j) (swap_fin i j)) == - perm_sign sigma.
  Proof.
    intros N i j Hij sigma.
    destruct (Nat.lt_trichotomy (fin_to_nat i) (fin_to_nat j)) as [Hlt | [Heq | Hgt]].
    - assert (Hgap : fin_to_nat j = (fin_to_nat i + S (fin_to_nat j - fin_to_nat i - 1))%nat)
        by lia.
      exact (swap_sign_flip_gap (fin_to_nat j - fin_to_nat i - 1) N i j Hgap sigma).
    - destruct (Hij (fin_to_nat_inj N i j Heq)).
    - assert (Hgap : fin_to_nat i = (fin_to_nat j + S (fin_to_nat i - fin_to_nat j - 1))%nat)
        by lia.
      assert (Hflip := swap_sign_flip_gap (fin_to_nat i - fin_to_nat j - 1) N j i Hgap sigma).
      transitivity (perm_sign (perm_postcompose sigma (swap_fin j i) (swap_fin j i))).
      + apply (perm_sign_postcompose_ext N sigma (swap_fin i j) (swap_fin j i)
                 (swap_fin i j) (swap_fin j i)).
        intros x. exact (swap_fin_comm N i j x).
      + exact Hflip.
  Qed.

End PermSignPostcomposeSwap.


Section DetSwapGeneral.
  Context {F : Type} {FF : Field F}.

  Theorem detM_leibniz_swap_general : ∀ (N : nat) (i j : Fin.t N),
    i ≠ j → ∀ A : Matrix N N F,
    detM_leibniz (row_swap A i j) == - detM_leibniz A.
  Proof.
    intros N i j Hij A.
    unfold detM_leibniz at 1.
    transitivity (psum (fun sigma : Perm N ⇒ perm_sign sigma ×
      prodF (fun c ⇒ A (swap_fin i j (perm_apply sigma c)) c))).
    - apply psum_ext. intros sigma.
      apply ring_mul_proper.
      + reflexivity.
      + exact (perm_prod_row_swap N A i j sigma).
    - rewrite (psum_reindex N
        (fun sigma : Perm N ⇒ perm_sign sigma ×
          prodF (fun c ⇒ A (swap_fin i j (perm_apply sigma c)) c))
        (swap_fin i j) (swap_fin i j)
        (swap_fin_invol N i j) (swap_fin_invol N i j)).
      transitivity (psum (fun sigma : Perm N ⇒ - (perm_sign sigma × perm_prod A sigma))).
      + apply psum_ext. intros sigma.
        transitivity ((- perm_sign sigma) ×
          prodF (fun c ⇒ A (swap_fin i j
            (perm_apply (perm_postcompose sigma (swap_fin i j) (swap_fin i j)) c)) c)).
        × apply ring_mul_proper.
          -- exact (perm_sign_postcompose_swap N i j Hij sigma).
          -- reflexivity.
        × rewrite (mul_neg_l (perm_sign sigma)
            (prodF (fun c ⇒ A (swap_fin i j
              (perm_apply (perm_postcompose sigma (swap_fin i j) (swap_fin i j)) c)) c))).
          apply grp_neg_proper.
          apply ring_mul_proper.
          -- reflexivity.
          -- unfold perm_prod.
             apply prodF_ext. intros c.
             rewrite (perm_apply_postcompose N sigma (swap_fin i j) (swap_fin i j)
                        (swap_fin_invol N i j) (swap_fin_invol N i j) c),
                (swap_fin_invol N i j (perm_apply sigma c)).
             reflexivity.
      + exact (psum_neg N (fun sigma ⇒ perm_sign sigma × perm_prod A sigma)).
  Qed.

  Theorem detM_swap_general : ∀ (N : nat) (i j : Fin.t N),
    i ≠ j → ∀ A : Matrix N N F,
    detM (row_swap A i j) == - detM A.
  Proof.
    intros N i j Hij A.
    rewrite <- (detM_leibniz_eq_detM N (row_swap A i j)).
    rewrite (detM_leibniz_swap_general N i j Hij A).
    apply grp_neg_proper.
    exact (detM_leibniz_eq_detM N A).
  Qed.

End DetSwapGeneral.

Section DetElemSwap.
  Context {F : Type} {FF : Field F}.

  Theorem det_elem_swap : ∀ (n : nat) (i j : Fin.t n),
    i ≠ j → ∀ A : Matrix n n F,
    detM (mat_mul (elem_swap i j) A) == - detM A.
  Proof.
    intros n i j Hij A.
    transitivity (detM (row_swap A i j)).
    - apply detM_proper. intros r c.
      transitivity (A (swap_fin i j r) c).
      + exact (elem_swap_mul n n i j A r c).
      + symmetry. exact (row_swap_eq_swap_fin n A i j r c).
    - exact (detM_swap_general n i j Hij A).
  Qed.

End DetElemSwap.
