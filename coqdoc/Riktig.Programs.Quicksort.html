Require Export Arith Lia.
Require Export Coq.Arith.Wf_nat.
Require Export List. Import ListNotations.
Require Export Coq.Logic.FunctionalExtensionality.
Require Export Riktig.Interface.Decidable Riktig.Impl.Decidable.

Section Quicksort.

  Variable V : Type.
  Variable leb : V → V → bool.

  Variable mem : Type.
  Variable read : mem → nat → V.
  Variable swap : mem → nat → nat → mem.

  Definition part (m0 : mem) (store0 i0 hi : nat) : mem × nat.
  refine (Fix (well_founded_ltof _ (fun i ⇒ hi - i)) (fun _ ⇒ mem → nat → mem × nat)
    (fun i rec m store ⇒
      if dec (i < hi)
      then if dec (leb (read m i) (read m hi) = true)
           then rec (S i) _ (swap m i store) (S store)
           else rec (S i) _ m store
      else (swap m store hi, store))
    i0 m0 store0).
  all: unfold ltof; abstract lia.
  Defined.

  Lemma part_step m store i hi :
    part m store i hi =
      if dec (i < hi)
      then if dec (leb (read m i) (read m hi) = true)
           then part (swap m i store) (S store) (S i) hi
           else part m store (S i) hi
      else (swap m store hi, store).
  Proof.
    unfold part at 1. rewrite Fix_eq.
    - reflexivity.
    - intros x f g Hfg.
      do 2 (apply functional_extensionality; intro).
      destruct (dec (x < hi)); [|reflexivity].
      destruct (dec (_ = true)); rewrite Hfg; reflexivity.
  Qed.

  Definition qsort : mem → nat → nat → mem :=
    fun m0 lo0 hi0 ⇒
    Fix (well_founded_ltof (mem × (nat × nat)) (fun q ⇒ S (snd (snd q)) - fst (snd q)))
        (fun _ ⇒ mem)
        (fun q qsort ⇒
           match q as x return q = x → mem with
           | (m, (lo, hi)) ⇒
               fun e ⇒
               if dec (S hi - lo < 2)
               then m
               else
                 let '(m', piv) := part m lo lo hi in
                 if dec (S (piv - 1) - lo < S hi - lo)
                 then if dec (S hi - S piv < S hi - lo)
                      then qsort (qsort (m', (lo, piv - 1))
                                    ltac:(subst q; unfold ltof; cbn [fst snd]; abstract lia),
                                  (S piv, hi))
                                 ltac:(subst q; unfold ltof; cbn [fst snd]; abstract lia)
                      else m
                 else m
           end eq_refl)
        (m0, (lo0, hi0)).

  Lemma qsort_step m lo hi :
    qsort m lo hi =
      if dec (S hi - lo < 2)
      then m
      else
        let '(m', piv) := part m lo lo hi in
        if dec (S (piv - 1) - lo < S hi - lo)
        then if dec (S hi - S piv < S hi - lo)
             then qsort (qsort m' lo (piv - 1)) (S piv) hi
             else m
        else m.
  Proof.
    unfold qsort at 1. rewrite Fix_eq.
    - destruct (dec (S hi - lo < 2)); [reflexivity|].
      destruct (part m lo lo hi) as [m' piv].
      destruct (dec (S (piv - 1) - lo < S hi - lo)); [|reflexivity].
      destruct (dec (S hi - S piv < S hi - lo)); [|reflexivity].
      reflexivity.
    - intros [m1 [lo1 hi1]] f g H.
      destruct (dec (S hi1 - lo1 < 2)); [reflexivity|].
      destruct (part m1 lo1 lo1 hi1) as [m1' piv1].
      destruct (dec (S (piv1 - 1) - lo1 < S hi1 - lo1)); [|reflexivity].
      destruct (dec (S hi1 - S piv1 < S hi1 - lo1)); [|reflexivity].
      rewrite H, H. reflexivity.
  Qed.

End Quicksort.


Section TestQsort.
Import ListNotations.

Fixpoint list_write {A} (l : list A) (i : nat) (v : A) : list A :=
  match l, i with
  | [], _ ⇒ []
  | _ :: t, 0 ⇒ v :: t
  | h :: t, S i' ⇒ h :: list_write t i' v
  end.

Definition list_swap {A} (def : A) (l : list A) (i j : nat) : list A :=
  list_write (list_write l i (nth_default def l j)) j (nth_default def l i).

Check Logic.eq_refl : qsort nat Nat.leb (list nat) (nth_default 0) (list_swap 0) [3;4;2;0] 0 3 = [0;2;3;4].
Check Logic.eq_refl : qsort nat Nat.leb (list nat) (nth_default 0) (list_swap 0) [3;4;2;0] 1 2 = [3;2;4;0].

End TestQsort.
