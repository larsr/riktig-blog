Class Decidable (P : Prop) := dec : {P} + {¬ P}.
Arguments dec _ {_}.

Class EqDec (A : Type) := eq_dec : ∀ (x y : A), Decidable (x = y).
#[global] Instance EqDec_Decidable {A} `{EqDec A} : ∀ x y : A, Decidable (x = y) := eq_dec.
