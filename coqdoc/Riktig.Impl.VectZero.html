From Coq Require Import Morphisms.
From Riktig Require Import Ops Interface.Field Interface.LinearMap
  Interface.Category Theory.UnitVS Impl.VectCat.

Section VectZero.
  Context {F : Type} {FF : Field F}.

  Definition VUnit : VectObj F :=
    @Build_VectObj F FF unit (@Unit_VectorSpace F FF).


  Definition vect_emit (V : VectObj F) : VHom VUnit V := lm_const0.

  Lemma vect_emit_unique (V : VectObj F) (f : VHom VUnit V) :
    vhom_equ VUnit V f (vect_emit V).
  Proof.
    intros u.
    rewrite (lm_proper f u 0 I).
    exact (lm_zero f).
  Qed.

  #[global] Instance Vect_Initial : Initial (@Vect_Category F FF) := {|
    init := VUnit;
    emit := vect_emit;
    emit_unique := vect_emit_unique
  |}.


  Section Bang.
    Context (V : VectObj F).

    Lemma bang_proper : Proper (equ ==> equ) (fun _ : vo_car V ⇒ tt).
    Proof.
      intros u v Huv.
      exact I.
    Qed.

    Lemma bang_add : ∀ u v : vo_car V, (tt : unit) == tt + tt.
    Proof.
      intros u v.
      exact I.
    Qed.

    Lemma bang_scale : ∀ (a : F) (u : vo_car V), (tt : unit) == a *: tt.
    Proof.
      intros a u.
      exact I.
    Qed.

    Definition vect_bang : VHom V VUnit :=
      @Build_LinearMap F (vo_car V) (vo_car VUnit) FF (vo_vs V) (vo_vs VUnit)
        (fun _ : vo_car V ⇒ tt) bang_proper bang_add bang_scale.

  End Bang.

  Lemma vect_bang_unique (V : VectObj F) (f : VHom V VUnit) :
    vhom_equ V VUnit f (vect_bang V).
  Proof.
    intros u.
    exact I.
  Qed.

  #[global] Instance Vect_Terminal : Terminal (@Vect_Category F FF) := {|
    term := VUnit;
    bang := vect_bang;
    bang_unique := vect_bang_unique
  |}.


  Definition zero_map (U V : VectObj F) : VHom U V :=
    vcomp U VUnit V (vect_emit V) (vect_bang U).

  Example zero_map_direct (U V : VectObj F) :
    vhom_equ U V (zero_map U V) lm_const0.
  Proof.
    intros u.
    reflexivity.
  Qed.

End VectZero.
