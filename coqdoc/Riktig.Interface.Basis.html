Library Riktig.Interface.Basis

From Coq Require Import RelationClasses.
From Coq.Vectors Require Fin.
From Riktig Require Import Ops Interface.Field Interface.VectorSpace
  Interface.LinearMap Theory.Basis.


Record Basis (n : nat) (F V : Type) {FF : Field F}
    {VV : VectorSpace F V} := {
  basis_vec : Fin.t n → V;
  basis_coords : LinearMap F V (Fin.t n → F);
  basis_recon : ∀ v : V,
    lincomb basis_vec (basis_coords v) == v;
  basis_coords_lincomb : ∀ c : Fin.t n → F,
    basis_coords (lincomb basis_vec c) == c
}.

Arguments basis_vec {n F V FF VV}.
Arguments basis_coords {n F V FF VV}.
Arguments basis_recon {n F V FF VV}.
Arguments basis_coords_lincomb {n F V FF VV}.
