Library Riktig.Impl.ListMonoid

From Coq Require Import List Morphisms.
From Riktig Require Import Ops Interface.Monoid.

#[global] Instance List_EqRel A : EqRel (list A) := eq.
#[global] Instance List_Zero A : Zero (list A) := nil.
#[global] Instance List_Add A : HasAdd (list A) := @app A.

#[global] Instance app_proper A : Proper (@eq (list A) ==> eq ==> eq) (@app A).
Proof.
  intros x y Hxy u v Huv.
  rewrite Hxy, Huv.
  reflexivity.
Qed.

#[global] Instance List_Monoid A : Monoid (list A) := {|
  monoid_equivalence := eq_equivalence;
  monoid_associative := fun l m n ⇒ eq_sym (app_assoc l m n);
  monoid_left_identity := @app_nil_l A;
  monoid_right_identity := @app_nil_r A;
  monoid_add_proper := app_proper A
|}.
