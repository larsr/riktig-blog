Library Riktig.Channel.GoBackN


From Coq Require Export List Arith Lia.
From Riktig.Theory Require Export Subseq.
From Riktig.Channel Require Export Trace.
Import ListNotations.

Section GoBackN.
  Context {M : Type}.


  Record Packet := { pkt_seq : nat; pkt_msg : M }.


  Fixpoint sender_burst_gbn (seqn : nat) (m : M) (k : nat) : list Packet :=
    match k with
    | O ⇒ []
    | S k' ⇒ {| pkt_seq := seqn; pkt_msg := m |} :: sender_burst_gbn seqn m k'
    end.

  Fixpoint gbn_stream_from (seqn : nat) (ms : list M) (k : nat) : list Packet :=
    match ms with
    | [] ⇒ []
    | m :: ms' ⇒ sender_burst_gbn seqn m k ++ gbn_stream_from (S seqn) ms' k
    end.

  Definition gbn_stream (ms : list M) (k : nat) : list Packet :=
    gbn_stream_from 0 ms k.


  Fixpoint gbn_deliver (expect : nat) (ps : list Packet) : list M :=
    match ps with
    | [] ⇒ []
    | p :: ps' ⇒
        if Nat.eqb (pkt_seq p) expect
        then pkt_msg p :: gbn_deliver (S expect) ps'
        else gbn_deliver expect ps'
    end.


  Fixpoint gbn_next (expect : nat) (ps : list Packet) : nat :=
    match ps with
    | [] ⇒ expect
    | p :: ps' ⇒
        if Nat.eqb (pkt_seq p) expect
        then gbn_next (S expect) ps'
        else gbn_next expect ps'
    end.


  Lemma eqb_seq_true : ∀ s1 s2 : nat, s1 = s2 → Nat.eqb s1 s2 = true.
  Proof.
    intros s1 s2 H. apply Nat.eqb_eq. exact H.
  Qed.

  Lemma eqb_seq_false : ∀ s1 s2 : nat, s1 ≠ s2 → Nat.eqb s1 s2 = false.
  Proof.
    intros s1 s2 H. apply Nat.eqb_neq. exact H.
  Qed.


  Lemma gbn_deliver_app : ∀ (w1 w2 : list Packet) (expect : nat),
    gbn_deliver expect (w1 ++ w2) =
      gbn_deliver expect w1 ++ gbn_deliver (gbn_next expect w1) w2.
  Proof.
    induction w1 as [| p w1' IH]; intros w2 expect.
    - simpl. reflexivity.
    - simpl. destruct (Nat.eqb (pkt_seq p) expect) eqn:Hcmp.
      + simpl. rewrite (IH w2 (S expect)). reflexivity.
      + apply IH.
  Qed.


  Lemma sender_burst_gbn_all_eq : ∀ (seqn : nat) (m : M) (k : nat) (p : Packet),
    In p (sender_burst_gbn seqn m k) → p = {| pkt_seq := seqn; pkt_msg := m |}.
  Proof.
    intros seqn m k.
    induction k as [| k' IH]; intros p Hp.
    - simpl in Hp. destruct Hp.
    - simpl in Hp. destruct Hp as [Hp | Hp].
      + symmetry. exact Hp.
      + apply IH. exact Hp.
  Qed.

  Lemma lossy_dup_preserves_all_eq_gbn : ∀ (x0 : Packet) (s r : list Packet),
    lossy_dup s r → (∀ p, In p s → p = x0) → ∀ p, In p r → p = x0.
  Proof.
    exact lossy_dup_preserves_all_eq.
  Qed.


  Lemma gbn_deliver_homog_neq : ∀ (x0 : Packet) (w : list Packet) (expect : nat),
    (∀ p, In p w → p = x0) → pkt_seq x0 ≠ expect →
    gbn_deliver expect w = [] ∧ gbn_next expect w = expect.
  Proof.
    intros x0 w.
    induction w as [| p w' IH]; intros expect Hall Hneq.
    - split; reflexivity.
    - assert (Hp : p = x0).
      { apply Hall. simpl. left. reflexivity. }
      rewrite Hp. simpl.
      rewrite (eqb_seq_false (pkt_seq x0) expect Hneq).
      apply IH.
      + intros q Hq. apply Hall. simpl. right. exact Hq.
      + exact Hneq.
  Qed.

  Lemma gbn_deliver_homog_eq : ∀ (x0 : Packet) (w : list Packet) (expect : nat),
    (∀ p, In p w → p = x0) → pkt_seq x0 = expect →
    (gbn_deliver expect w = [] ∧ gbn_next expect w = expect) ∨
    (gbn_deliver expect w = [pkt_msg x0] ∧ gbn_next expect w = S expect).
  Proof.
    intros x0 w expect Hall Heq.
    destruct w as [| p w'].
    - left. split; reflexivity.
    - assert (Hp : p = x0).
      { apply Hall. simpl. left. reflexivity. }
      right.
      rewrite Hp. simpl.
      rewrite (eqb_seq_true (pkt_seq x0) expect Heq). simpl.
      assert (Hnw : ∀ q, In q w' → q = x0).
      { intros q Hq. apply Hall. simpl. right. exact Hq. }
      assert (Hneq2 : pkt_seq x0 ≠ S expect).
      { rewrite Heq. lia. }
      destruct (gbn_deliver_homog_neq x0 w' (S expect) Hnw Hneq2) as [Hd Hn].
      rewrite Hd. rewrite Hn.
      split; reflexivity.
  Qed.


  Lemma burst_delivery_gbn : ∀ (seqn : nat) (m : M) (k : nat) (w : list Packet) (expect : nat),
    lossy_dup (sender_burst_gbn seqn m k) w →
    (gbn_deliver expect w = [] ∧ gbn_next expect w = expect) ∨
    (gbn_deliver expect w = [m] ∧ gbn_next expect w = S expect).
  Proof.
    intros seqn m k w expect H.
    assert (Hall : ∀ p, In p w → p = {| pkt_seq := seqn; pkt_msg := m |}).
    { apply (lossy_dup_preserves_all_eq_gbn {| pkt_seq := seqn; pkt_msg := m |}
               (sender_burst_gbn seqn m k) w H).
      intros p Hp. apply (sender_burst_gbn_all_eq seqn m k p Hp). }
    destruct (Nat.eq_dec (pkt_seq ({| pkt_seq := seqn; pkt_msg := m |})) expect) as [Heq | Hneq].
    - destruct (gbn_deliver_homog_eq {| pkt_seq := seqn; pkt_msg := m |} w expect Hall Heq)
        as [[Hd Hn] | [Hd Hn]].
      + left. split.
        × exact Hd.
        × exact Hn.
      + right. split.
        × exact Hd.
        × exact Hn.
    - left. apply (gbn_deliver_homog_neq {| pkt_seq := seqn; pkt_msg := m |} w expect Hall Hneq).
  Qed.



  Lemma lossy_dup_app_split_gbn : ∀ (s r : list Packet),
    lossy_dup s r → ∀ l1 l2 : list Packet, s = l1 ++ l2 →
    ∃ r1 r2, r = r1 ++ r2 ∧ lossy_dup l1 r1 ∧ lossy_dup l2 r2.
  Proof.
    exact lossy_dup_app_split.
  Qed.


  Lemma lossy_dup_nil_l_gbn : ∀ (s r : list Packet), lossy_dup s r → s = [] → r = [].
  Proof.
    exact lossy_dup_nil_l.
  Qed.


  Lemma gbn_deliver_subseq : ∀ (k : nat) (ms : list M)
      (seq0 expect : nat) (wire : list Packet),
    lossy_dup (gbn_stream_from seq0 ms k) wire → subseq (gbn_deliver expect wire) ms.
  Proof.
    intros k ms.
    induction ms as [| m ms' IH]; intros seq0 expect wire H.
    - simpl in H.
      assert (Hnil : wire = []).
      { apply (lossy_dup_nil_l_gbn [] wire H). reflexivity. }
      rewrite Hnil. simpl. apply subseq_nil.
    - simpl in H.
      destruct (lossy_dup_app_split_gbn _ _ H
                  (sender_burst_gbn seq0 m k) (gbn_stream_from (S seq0) ms' k)
                  eq_refl) as [w1 [w2 [Hwire [H1 H2]]]].
      rewrite Hwire.
      rewrite (gbn_deliver_app w1 w2 expect).
      destruct (burst_delivery_gbn seq0 m k w1 expect H1) as [[Hd1 Hn1] | [Hd1 Hn1]].
      + rewrite Hd1. rewrite Hn1. simpl.
        apply subseq_drop.
        apply (IH (S seq0) expect w2 H2).
      + rewrite Hd1. rewrite Hn1. simpl.
        apply subseq_keep.
        apply (IH (S seq0) (S expect) w2 H2).
  Qed.


  Theorem gbn_safety : ∀ (k : nat) (ms : list M) (wire : list Packet),
    lossy_dup (gbn_stream ms k) wire → subseq (gbn_deliver 0 wire) ms.
  Proof.
    intros k ms wire H.
    apply (gbn_deliver_subseq k ms 0 0 wire H).
  Qed.



  Lemma gbn_burst_miss : ∀ (seqn : nat) (m : M) (k : nat) (w : list Packet) (expect : nat),
    lossy_dup (sender_burst_gbn seqn m k) w → seqn ≠ expect →
    gbn_deliver expect w = [] ∧ gbn_next expect w = expect.
  Proof.
    intros seqn m k w expect H Hneq.
    assert (Hall : ∀ p, In p w → p = {| pkt_seq := seqn; pkt_msg := m |}).
    { apply (lossy_dup_preserves_all_eq_gbn {| pkt_seq := seqn; pkt_msg := m |}
               (sender_burst_gbn seqn m k) w H).
      intros p Hp. apply (sender_burst_gbn_all_eq seqn m k p Hp). }
    apply (gbn_deliver_homog_neq {| pkt_seq := seqn; pkt_msg := m |} w expect Hall Hneq).
  Qed.

  Lemma gbn_deliver_nomatch : ∀ (k : nat) (ms : list M) (seqn expect : nat) (wire : list Packet),
    expect < seqn → lossy_dup (gbn_stream_from seqn ms k) wire → gbn_deliver expect wire = [].
  Proof.
    intros k ms.
    induction ms as [| m ms' IH]; intros seqn expect wire Hlt H.
    - simpl in H.
      assert (Hnil : wire = []).
      { apply (lossy_dup_nil_l_gbn [] wire H). reflexivity. }
      rewrite Hnil. reflexivity.
    - simpl in H.
      destruct (lossy_dup_app_split_gbn _ _ H
                  (sender_burst_gbn seqn m k) (gbn_stream_from (S seqn) ms' k)
                  eq_refl) as [w1 [w2 [Hwire [H1 H2]]]].
      rewrite Hwire.
      rewrite (gbn_deliver_app w1 w2 expect).
      assert (Hneq : seqn ≠ expect) by lia.
      destruct (gbn_burst_miss seqn m k w1 expect H1 Hneq) as [Hd1 Hn1].
      rewrite Hd1. rewrite Hn1. simpl.
      apply (IH (S seqn) expect w2).
      + lia.
      + exact H2.
  Qed.


  Lemma gbn_deliver_prefix_gen : ∀ (k : nat) (ms : list M) (seqn : nat) (wire : list Packet),
    lossy_dup (gbn_stream_from seqn ms k) wire → prefix (gbn_deliver seqn wire) ms.
  Proof.
    intros k ms.
    induction ms as [| m ms' IH]; intros seqn wire H.
    - simpl in H.
      assert (Hnil : wire = []).
      { apply (lossy_dup_nil_l_gbn [] wire H). reflexivity. }
      rewrite Hnil. simpl. apply prefix_refl.
    - simpl in H.
      destruct (lossy_dup_app_split_gbn _ _ H
                  (sender_burst_gbn seqn m k) (gbn_stream_from (S seqn) ms' k)
                  eq_refl) as [w1 [w2 [Hwire [H1 H2]]]].
      rewrite Hwire.
      rewrite (gbn_deliver_app w1 w2 seqn).
      destruct (burst_delivery_gbn seqn m k w1 seqn H1) as [[Hd1 Hn1] | [Hd1 Hn1]].
      +
        rewrite Hd1. rewrite Hn1. simpl.
        unfold prefix.
        assert (Hno : gbn_deliver seqn w2 = []).
        { apply (gbn_deliver_nomatch k ms' (S seqn) seqn w2).
          - lia.
          - exact H2. }
        rewrite Hno.
        ∃ (m :: ms'). reflexivity.
      +
        rewrite Hd1. rewrite Hn1. simpl.
        unfold prefix.
        destruct (IH (S seqn) w2 H2) as [ms2 Heq].
        ∃ ms2. rewrite Heq. reflexivity.
  Qed.

  Theorem gbn_prefix : ∀ (k : nat) (ms : list M) (wire : list Packet),
    lossy_dup (gbn_stream ms k) wire → prefix (gbn_deliver 0 wire) ms.
  Proof.
    intros k ms wire H.
    apply (gbn_deliver_prefix_gen k ms 0 wire H).
  Qed.

  Corollary gbn_safety_from_prefix : ∀ (k : nat) (ms : list M) (wire : list Packet),
    lossy_dup (gbn_stream ms k) wire → subseq (gbn_deliver 0 wire) ms.
  Proof.
    intros k ms wire H.
    apply prefix_subseq.
    apply (gbn_prefix k ms wire H).
  Qed.

End GoBackN.
