Library Riktig.Theory.Subseq


From Coq Require Import List.
Import ListNotations.

Section SubseqTheory.
  Context {A : Type}.
  Implicit Types r s : list A.

  Inductive subseq : list A → list A → Prop :=
    | subseq_nil : subseq [] []
    | subseq_keep : ∀ x r s, subseq r s → subseq (x :: r) (x :: s)
    | subseq_drop : ∀ x r s, subseq r s → subseq r (x :: s).


  Lemma subseq_refl : ∀ s, subseq s s.
  Proof.
    induction s as [| x s IH].
    - apply subseq_nil.
    - apply subseq_keep. exact IH.
  Qed.

  Lemma subseq_nil_l : ∀ s, subseq [] s.
  Proof.
    induction s as [| x s IH].
    - apply subseq_nil.
    - apply subseq_drop. exact IH.
  Qed.


  Lemma subseq_cons_r_inv : ∀ r x s,
    subseq r (x :: s) →
    (∃ r', r = x :: r' ∧ subseq r' s) ∨ subseq r s.
  Proof.
    intros r x s H.
    inversion H; subst.
    - left. eexists. split.
      + reflexivity.
      + eassumption.
    - right. assumption.
  Qed.


  Lemma subseq_trans : ∀ u v w,
    subseq u v → subseq v w → subseq u w.
  Proof.
    intros u v w Huv Hvw.
    induction Hvw as [| x v' w' Hvw IH | x v' w' Hvw IH] in u, Huv |- ×.
    - exact Huv.
    - destruct (subseq_cons_r_inv u x v' Huv) as [[u' [Hequ Hu']] | Hu].
      + rewrite Hequ. apply subseq_keep. apply IH. exact Hu'.
      + apply subseq_drop. apply IH. exact Hu.
    - apply subseq_drop. apply IH. exact Huv.
  Qed.


  Lemma subseq_length : ∀ r s, subseq r s → length r ≤ length s.
  Proof.
    intros r s H.
    induction H as [| x r s H IH | x r s H IH]; simpl.
    - apply le_n.
    - apply le_n_S. exact IH.
    - apply le_S. exact IH.
  Qed.

End SubseqTheory.
