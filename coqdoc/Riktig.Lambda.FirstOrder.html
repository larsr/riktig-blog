Library Riktig.Lambda.FirstOrder


From Coq Require Import List ZArith.
Import ListNotations.


Record var : Type := MkVar { var_ix : nat }.

Record fname : Type := MkFname { fname_ix : nat }.


Inductive prim : Type :=
  | PReadW : prim
  | PWriteW : prim
  | PAddW : prim
  | PSubW : prim.

Inductive rhs : Type :=
  | FConst : Z → rhs
  | FMove : var → rhs
  | FPrim : prim → list var → rhs
  | FClose : fname → list var → rhs
  | FCall : var → var → rhs.
Inductive body : Type :=
  | FLet : rhs → body → body
  | FRet : var → body.

Record program : Type := MkProgram {
  p_funs : list body;
  p_main : body
}.


Inductive val : Type :=
  | VWord : Z → val
  | VClos : fname → list val → val.

Definition mem : Type := Z → Z.

Definition mem_zero : mem := fun (_ : Z) ⇒ 0%Z.

Definition mem_upd (m : mem) (a : Z) (w : Z) : mem :=
  fun (a' : Z) ⇒ if Z.eqb a' a then w else m a'.

Definition locals : Type := list val.

Definition lookup (rho : locals) (x : var) : option val :=
  nth_error rho (var_ix x).

Fixpoint lookup_all (rho : locals) (xs : list var) {struct xs}
  : option (list val) :=
  match xs with
  | [] ⇒ Some []
  | x :: xs' ⇒
      match lookup rho x, lookup_all rho xs' with
      | Some v, Some vs ⇒ Some (v :: vs)
      | _, _ ⇒ None
      end
  end.

Definition eval_prim (p : prim) (vs : list val) (m : mem)
  : option (mem × val) :=
  match p, vs with
  | PReadW, [VWord a] ⇒ Some (m, VWord (m a))
  | PWriteW, [VWord a; VWord w] ⇒ Some (mem_upd m a w, VWord w)
  | PAddW, [VWord x; VWord y] ⇒ Some (m, VWord (Z.add x y))
  | PSubW, [VWord x; VWord y] ⇒ Some (m, VWord (Z.sub x y))
  | _, _ ⇒ None
  end.


Fixpoint eval_body (fuel : nat) (fns : list body) (rho : locals)
                   (m : mem) (b : body) {struct fuel}
  : option (mem × val) :=
  match fuel with
  | O ⇒ None
  | S fuel' ⇒
      match b with
      | FRet x ⇒
          match lookup rho x with
          | Some v ⇒ Some (m, v)
          | None ⇒ None
          end
      | FLet r b' ⇒
          match eval_rhs fuel' fns rho m r with
          | Some (m', v) ⇒ eval_body fuel' fns (v :: rho) m' b'
          | None ⇒ None
          end
      end
  end
with eval_rhs (fuel : nat) (fns : list body) (rho : locals)
              (m : mem) (r : rhs) {struct fuel}
  : option (mem × val) :=
  match fuel with
  | O ⇒ None
  | S fuel' ⇒
      match r with
      | FConst z ⇒ Some (m, VWord z)
      | FMove x ⇒
          match lookup rho x with
          | Some v ⇒ Some (m, v)
          | None ⇒ None
          end
      | FPrim p xs ⇒
          match lookup_all rho xs with
          | Some vs ⇒ eval_prim p vs m
          | None ⇒ None
          end
      | FClose f xs ⇒
          match lookup_all rho xs with
          | Some vs ⇒ Some (m, VClos f vs)
          | None ⇒ None
          end
      | FCall f x ⇒
          match lookup rho f, lookup rho x with
          | Some (VClos g cvs), Some a ⇒
              match nth_error fns (fname_ix g) with
              | Some fb ⇒ eval_body fuel' fns (a :: cvs) m fb
              | None ⇒ None
              end
          | _, _ ⇒ None
          end
      end
  end.

Definition run (fuel : nat) (p : program) : option val :=
  match eval_body fuel (p_funs p) [] mem_zero (p_main p) with
  | Some (_, v) ⇒ Some v
  | None ⇒ None
  end.


Definition prog_add : program :=
  MkProgram []
    (FLet (FConst 2)
    (FLet (FConst 40)
    (FLet (FPrim PAddW [MkVar 0; MkVar 1])
    (FRet (MkVar 0))))).

Example run_prog_add : run 10 prog_add = Some (VWord 42).
Proof. vm_compute. reflexivity. Qed.

Definition prog_store : program :=

    (FLet (FConst 7)
    (FLet (FConst 5)
    (FLet (FPrim PWriteW [MkVar 1; MkVar 0])
    (FLet (FPrim PReadW [MkVar 2])
    (FRet (MkVar 0)))))).

Definition f_add : body :=
  FLet (FPrim PAddW [MkVar 0; MkVar 1])
  (FRet (MkVar 0)).

Definition f_apply : body :=
  FLet (FConst 2)
  (FLet (FCall (MkVar 1) (MkVar 0))
  (FRet (MkVar 0))).

Definition prog_close : program :=
  MkProgram [f_add; f_apply]
    (FLet (FConst 40)
    (FLet (FClose (MkFname 0) [MkVar 0])
    (FLet (FClose (MkFname 1) [])
    (FLet (FCall (MkVar 0) (MkVar 1))
    (FRet (MkVar 0)))))).

Example run_prog_close : run 100 prog_close = Some (VWord 42).
Proof. vm_compute. reflexivity. Qed.


Definition f_self : body :=
  FLet (FCall (MkVar 0) (MkVar 0))
  (FRet (MkVar 0)).

Definition prog_loop : program :=
  MkProgram [f_self]
    (FLet (FClose (MkFname 0) [])
    (FLet (FCall (MkVar 0) (MkVar 0))
    (FRet (MkVar 0)))).

Example run_prog_loop : run 1000 prog_loop = None.
Proof. vm_compute. reflexivity. Qed.
