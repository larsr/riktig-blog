Inductive Reg : Type := R0 | R1 | R2 | R3 | R4 | R5 | R6 | R7.

Definition reg_eqb (r1 r2 : Reg) : bool :=
  match r1, r2 with
  | R0, R0 | R1, R1 | R2, R2 | R3, R3
  | R4, R4 | R5, R5 | R6, R6 | R7, R7 ⇒ true
  | _, _ ⇒ false
  end.

Inductive Mem (addr val : Type) : Type → Type :=
  | Ret : ∀ a, a → Mem addr val a
  | ReadVal : ∀ a, addr → (val → Mem addr val a) → Mem addr val a
  | WriteVal : ∀ a, addr → val → Mem addr val a → Mem addr val a
  | ReadAddr : ∀ a, addr → (addr → Mem addr val a) → Mem addr val a
  | WriteAddr : ∀ a, addr → addr → Mem addr val a → Mem addr val a
  | ReadInstr : ∀ a, addr → (Instr val → Mem addr val a) → Mem addr val a
  | WriteInstr : ∀ a, addr → Instr val → Mem addr val a → Mem addr val a
  | StackAlloc : ∀ a, (addr → Mem addr val a) → Mem addr val a
  | Alloc : ∀ a, nat → (addr → Mem addr val a) → Mem addr val a
  | Offset : ∀ a, addr → val → (addr → Mem addr val a) → Mem addr val a
  | GetSp : ∀ a, (addr → Mem addr val a) → Mem addr val a
  | SetSp : ∀ a, addr → Mem addr val a → Mem addr val a
  | While : ∀ a, Mem addr val (option bool) →
                           Mem addr val (option unit) →
                           Mem addr val a → Mem addr val a.

Fixpoint bind_mem {addr val b a : Type} (m : Mem addr val a)
  : (a → Mem addr val b) → Mem addr val b :=
  match m in Mem _ _ t return (t → Mem addr val b) → Mem addr val b with
  | Ret x ⇒ fun k ⇒ k x
  | ReadVal ad f ⇒ fun k ⇒ ReadVal ad (fun v ⇒ bind_mem (f v) k)
  | WriteVal ad v m' ⇒ fun k ⇒ WriteVal ad v (bind_mem m' k)
  | ReadAddr ad f ⇒ fun k ⇒ ReadAddr ad (fun p ⇒ bind_mem (f p) k)
  | WriteAddr ad p m' ⇒ fun k ⇒ WriteAddr ad p (bind_mem m' k)
  | ReadInstr ad f ⇒ fun k ⇒ ReadInstr ad (fun i ⇒ bind_mem (f i) k)
  | WriteInstr ad i m' ⇒ fun k ⇒ WriteInstr ad i (bind_mem m' k)
  | StackAlloc f ⇒ fun k ⇒ StackAlloc (fun p ⇒ bind_mem (f p) k)
  | Alloc n f ⇒ fun k ⇒ Alloc n (fun p ⇒ bind_mem (f p) k)
  | Offset base off f ⇒ fun k ⇒ Offset base off (fun p ⇒ bind_mem (f p) k)
  | GetSp f ⇒ fun k ⇒ GetSp (fun p ⇒ bind_mem (f p) k)
  | SetSp sp m' ⇒ fun k ⇒ SetSp sp (bind_mem m' k)
  | While c body m' ⇒ fun k ⇒ While c body (bind_mem m' k)
  end.



Definition MemT (addr val a : Type) := Mem addr val (option a).

Definition ret_t {addr val a : Type} (x : a) : MemT addr val a :=
  Ret (Some x).

Definition fail_t {addr val a : Type} : MemT addr val a :=
  Ret None.

Definition bind_t {addr val a b : Type}
  (m : MemT addr val a) (k : a → MemT addr val b) : MemT addr val b :=
  bind_mem m (fun oa ⇒ match oa with
                        | Some x ⇒ k x
                        | None ⇒ fail_t
                        end).

Definition then_t {addr val a b : Type}
  (m : MemT addr val a) (k : MemT addr val b) : MemT addr val b :=
  bind_t m (fun _ ⇒ k).


Declare Scope memprog_scope.
Delimit Scope memprog_scope with memprog.

Notation "'do!' x <- m ; rest" := (bind_t m (fun x ⇒ rest))
  (at level 61, x pattern, m at level 100, right associativity)
  : memprog_scope.
Notation "m >>= k" := (bind_t m k)
  (at level 58, left associativity) : memprog_scope.
Notation "m >> n" := (then_t m n)
  (at level 61, right associativity) : memprog_scope.

Open Scope memprog_scope.


Definition read_m {addr val : Type} (a : addr) : MemT addr val val :=
  ReadVal a (fun v ⇒ Ret (Some v)).

Definition write_m {addr val : Type} (a : addr) (v : val) : MemT addr val unit :=
  WriteVal a v (Ret (Some tt)).

Definition read_addr_m {addr val : Type} (a : addr) : MemT addr val addr :=
  ReadAddr a (fun p ⇒ Ret (Some p)).

Definition write_addr_m {addr val : Type} (a : addr) (p : addr) : MemT addr val unit :=
  WriteAddr a p (Ret (Some tt)).

Definition read_instr_m {addr val : Type} (a : addr) : MemT addr val (Instr val) :=
  ReadInstr a (fun i ⇒ Ret (Some i)).

Definition write_instr_m {addr val : Type} (a : addr) (i : Instr val) : MemT addr val unit :=
  WriteInstr a i (Ret (Some tt)).

Definition stack_alloc_m {addr val : Type} : MemT addr val addr :=
  StackAlloc (fun a ⇒ Ret (Some a)).

Definition alloc_m {addr val : Type} (n : nat) : MemT addr val addr :=
  Alloc n (fun a ⇒ Ret (Some a)).

Definition offset_m {addr val : Type} (base : addr) (off : val) : MemT addr val addr :=
  Offset base off (fun a ⇒ Ret (Some a)).

Definition get_sp_m {addr val : Type} : MemT addr val addr :=
  GetSp (fun a ⇒ Ret (Some a)).

Definition set_sp_m {addr val : Type} (sp : addr) : MemT addr val unit :=
  SetSp sp (Ret (Some tt)).

Definition while_m {addr val : Type}
  (c : MemT addr val bool) (b : MemT addr val unit) : MemT addr val unit :=
  While c b (Ret (Some tt)).


Definition push_val {addr val : Type} (v : val) : MemT addr val unit :=
  do! a <- stack_alloc_m; write_m a v.

Definition push_addr {addr val : Type} (p : addr) : MemT addr val unit :=
  do! a <- stack_alloc_m; write_addr_m a p.

Fixpoint stack_alloc_n {addr val : Type} (n : nat) : MemT addr val unit :=
  match n with
  | O ⇒ ret_t tt
  | S n' ⇒ stack_alloc_m >> stack_alloc_n n'
  end.
