From Riktig Require Import Ops Interface.Category Theory.Category.

Section InitialTerminalTheory.
  Context {O : Type} `{C : Category O}.

  Lemma bang_comp {T : Terminal C} :
    ∀ (x y : O) (f : x ~> y), bang y ∘ f == bang x.
  Proof.
    intros x y f.
    apply bang_unique.
  Qed.

  Lemma emit_comp {I : Initial C} :
    ∀ (x y : O) (f : x ~> y), f ∘ emit x == emit y.
  Proof.
    intros x y f.
    apply emit_unique.
  Qed.

  Lemma term_endo_id (T : Terminal C) :
    ∀ f : term (Terminal:=T) ~> term (Terminal:=T), f == cat_id.
  Proof.
    intros f.
    rewrite (bang_unique _ f).
    symmetry.
    apply bang_unique.
  Qed.

  Lemma init_endo_id (I : Initial C) :
    ∀ f : init (Initial:=I) ~> init (Initial:=I), f == cat_id.
  Proof.
    intros f.
    rewrite (emit_unique _ f).
    symmetry.
    apply emit_unique.
  Qed.

  Definition terminal_iso (T1 T2 : Terminal C) :
    Iso (term (Terminal:=T1)) (term (Terminal:=T2)) := {|
    iso_to := bang (Terminal:=T2) (term (Terminal:=T1));
    iso_from := bang (Terminal:=T1) (term (Terminal:=T2));
    iso_fromto := term_endo_id T1 _;
    iso_tofrom := term_endo_id T2 _
  |}.

  Definition initial_iso (I1 I2 : Initial C) :
    Iso (init (Initial:=I1)) (init (Initial:=I2)) := {|
    iso_to := emit (Initial:=I1) (init (Initial:=I2));
    iso_from := emit (Initial:=I2) (init (Initial:=I1));
    iso_fromto := init_endo_id I1 _;
    iso_tofrom := init_endo_id I2 _
  |}.

  Corollary terminal_isomorphic (T1 T2 : Terminal C) :
    term (Terminal:=T1) ≅ term (Terminal:=T2).
  Proof. exact (inhabits (terminal_iso T1 T2)). Qed.

  Corollary initial_isomorphic (I1 I2 : Initial C) :
    init (Initial:=I1) ≅ init (Initial:=I2).
  Proof. exact (inhabits (initial_iso I1 I2)). Qed.

End InitialTerminalTheory.
