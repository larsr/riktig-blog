Open Scope string_scope.


Inductive ParseTree :=
  | Leaf : string → ParseTree
  | Node : string → list ParseTree → ParseTree.


Fixpoint pretty_print (t : ParseTree) : string :=
  match t with
  | Leaf s ⇒ s
  | Node _ children ⇒
    (fix go (ts : list ParseTree) : string :=
       match ts with
       | [] ⇒ EmptyString
       | t :: rest ⇒ (pretty_print t ++ go rest)%string
       end) children
  end.


Definition find_child_in (label : string) (t : ParseTree) : option ParseTree :=
  match t with
  | Leaf _ ⇒ None
  | Node n _ ⇒ if String.eqb n label then Some t else None
  end.

Definition find_child (label : string) (children : list ParseTree)
  : option ParseTree :=
  (fix go (xs : list ParseTree) : option ParseTree :=
     match xs with
     | [] ⇒ None
     | x :: rest ⇒
       match find_child_in label x with
       | Some r ⇒ Some r
       | None ⇒ go rest
       end
     end) children.

Definition filter_in (label : string) (t : ParseTree) : list ParseTree :=
  match t with
  | Leaf _ ⇒ []
  | Node n _ ⇒ if String.eqb n label then [t] else []
  end.

Definition filter_children (label : string) (children : list ParseTree)
  : list ParseTree :=
  (fix go (xs : list ParseTree) : list ParseTree :=
     match xs with
     | [] ⇒ []
     | x :: rest ⇒ List.app (filter_in label x) (go rest)
     end) children.


Definition rule_env := list (string × ABNFAlternation).

Definition ascii_tolower (c : ascii) : ascii :=
  let n := Ascii.nat_of_ascii c in
  if andb (Nat.leb 65 n) (Nat.leb n 90)
  then Ascii.ascii_of_nat (n + 32)
  else c.

Fixpoint string_eqb_nocase (s1 s2 : string) : bool :=
  match s1, s2 with
  | EmptyString, EmptyString ⇒ true
  | String c1 r1, String c2 r2 ⇒
    andb (Ascii.eqb (ascii_tolower c1) (ascii_tolower c2))
         (string_eqb_nocase r1 r2)
  | _, _ ⇒ false
  end.

Definition lookup_rule (env : rule_env) (name : string) : option ABNFAlternation :=
  (fix go (l : list (string × ABNFAlternation)) :=
     match l with
     | [] ⇒ None
     | (n, body) :: rest ⇒
       if string_eqb_nocase n name then Some body
       else go rest
     end) env.

Definition build_env (rules : ABNFRulelist) : rule_env :=
  List.map (fun r ⇒ (rule_name r, rule_body r)) rules.


Definition as_leaf {S : Type} (p : Parser S ascii) : Parser S ParseTree :=
  c <- p ;;
  ret (Leaf (String c EmptyString)).

Definition as_leaf_str {S : Type} (p : Parser S string) : Parser S ParseTree :=
  s <- p ;;
  ret (Leaf s).


Definition interp_core_rule (name : string) : option (string → Parser string ParseTree) :=
  if string_eqb_nocase name "ALPHA" then
    Some (fun _ ⇒ as_leaf p_alpha)
  else if string_eqb_nocase name "BIT" then
    Some (fun _ ⇒ as_leaf p_bit)
  else if string_eqb_nocase name "CHAR" then
    Some (fun _ ⇒ as_leaf (satisfy stringSeq (fun c ⇒
      let n := Ascii.nat_of_ascii c in
      andb (Nat.leb 1 n) (Nat.leb n 127))))
  else if string_eqb_nocase name "CR" then
    Some (fun _ ⇒ as_leaf (satisfy stringSeq is_cr))
  else if string_eqb_nocase name "CRLF" then
    Some (fun _ ⇒
      (cr <- satisfy stringSeq is_cr ;; lf <- satisfy stringSeq is_lf ;;
       ret (Leaf (String cr (String lf EmptyString))))
      <|>
      (lf <- satisfy stringSeq is_lf ;; ret (Leaf (String lf EmptyString)))
      <|>
      (cr <- satisfy stringSeq is_cr ;; ret (Leaf (String cr EmptyString))))
  else if string_eqb_nocase name "CTL" then
    Some (fun _ ⇒ as_leaf (satisfy stringSeq (fun c ⇒
      let n := Ascii.nat_of_ascii c in
      orb (Nat.leb n 31) (Nat.eqb n 127))))
  else if string_eqb_nocase name "DIGIT" then
    Some (fun _ ⇒ as_leaf p_digit)
  else if string_eqb_nocase name "DQUOTE" then
    Some (fun _ ⇒ as_leaf (char dquote))
  else if string_eqb_nocase name "HEXDIG" then
    Some (fun _ ⇒ as_leaf p_hexdig)
  else if string_eqb_nocase name "HTAB" then
    Some (fun _ ⇒ as_leaf (satisfy stringSeq (fun c ⇒
      Nat.eqb (Ascii.nat_of_ascii c) 9)))
  else if string_eqb_nocase name "LF" then
    Some (fun _ ⇒ as_leaf (satisfy stringSeq is_lf))
  else if string_eqb_nocase name "LWSP" then
    Some (fun input ⇒
      cs <- many (
        (c <- p_wsp ;; ret (String c EmptyString))
        <|>
        (cr <- satisfy stringSeq is_cr ;; lf <- satisfy stringSeq is_lf ;;
         c <- p_wsp ;; ret (String cr (String lf (String c EmptyString))))
        <|>
        (lf <- satisfy stringSeq is_lf ;;
         c <- p_wsp ;; ret (String lf (String c EmptyString)))
        <|>
        (cr <- satisfy stringSeq is_cr ;;
         c <- p_wsp ;; ret (String cr (String c EmptyString)))
      ) input ;;
      ret (Leaf (fold_right append EmptyString cs)))
  else if string_eqb_nocase name "OCTET" then
    Some (fun _ ⇒ as_leaf (any stringSeq))
  else if string_eqb_nocase name "SP" then
    Some (fun _ ⇒ as_leaf (satisfy stringSeq (fun c ⇒
      Nat.eqb (Ascii.nat_of_ascii c) 32)))
  else if string_eqb_nocase name "VCHAR" then
    Some (fun _ ⇒ as_leaf p_vchar)
  else if string_eqb_nocase name "WSP" then
    Some (fun _ ⇒ as_leaf p_wsp)
  else
    None.


Definition match_byte (n : nat) : Parser string ParseTree :=
  c <- satisfy stringSeq (fun c ⇒ Nat.eqb (Ascii.nat_of_ascii c) n) ;;
  ret (Leaf (String c EmptyString)).

Definition match_range (lo hi : nat) : Parser string ParseTree :=
  c <- satisfy stringSeq (fun c ⇒
    let n := Ascii.nat_of_ascii c in
    andb (Nat.leb lo n) (Nat.leb n hi)) ;;
  ret (Leaf (String c EmptyString)).

Fixpoint match_ci_string (s : string) : Parser string ParseTree :=
  match s with
  | EmptyString ⇒ ret (Leaf EmptyString)
  | String c rest ⇒
    let lo := ascii_tolower c in
    c' <- satisfy stringSeq (fun c' ⇒ Ascii.eqb (ascii_tolower c') lo) ;;
    t <- match_ci_string rest ;;
    match t with
    | Leaf s' ⇒ ret (Leaf (String c' s'))
    | _ ⇒ ret (Leaf (String c' (pretty_print t)))
    end
  end.

Fixpoint repeat_parser (lo : nat) (hi : option nat) (p : Parser string (list ParseTree))
  (input : string) : Parser string (list ParseTree) :=
  match input with
  | EmptyString ⇒ if Nat.eqb lo 0 then ret [] else fail
  | String _ input' ⇒
    match hi with
    | Some 0 ⇒ if Nat.eqb lo 0 then ret [] else fail
    | _ ⇒
      let hi' := match hi with
                 | Some (Datatypes.S h) ⇒ Some h
                 | None ⇒ None
                 | Some 0 ⇒ Some 0
                 end in
      (ts <- p ;;
       rest <- repeat_parser (Nat.pred lo) hi' p input' ;;
       ret (List.app ts rest))
      <|>
      (if Nat.eqb lo 0 then ret [] else fail)
    end
  end.

Definition interp_numspec (spec : NumSpec) : Parser string ParseTree :=
  match spec with
  | NumConcat _ vals ⇒
    
    (fix go (vs : list nat) : Parser string ParseTree :=
       match vs with
       | [] ⇒ ret (Leaf EmptyString)
       | [v] ⇒ match_byte v
       | v :: rest ⇒
         t1 <- match_byte v ;;
         t2 <- go rest ;;
         match t1, t2 with
         | Leaf s1, Leaf s2 ⇒ ret (Leaf (s1 ++ s2)%string)
         | _, _ ⇒ ret (Leaf (pretty_print t1 ++ pretty_print t2)%string)
         end
       end) vals
  | NumRange _ lo hi ⇒ match_range lo hi
  end.


Section InterpAST.

Variable resolve : string → string → Parser string (list ParseTree).

Fixpoint interp_alternation (alt : ABNFAlternation) (input : string)
  {struct alt} : Parser string (list ParseTree) :=
  match alt with
  | Alt concats ⇒
    (fix try_alts (cs : list ABNFConcatenation) : Parser string (list ParseTree) :=
       match cs with
       | [] ⇒ fail
       | [c] ⇒ interp_concatenation c input
       | c :: rest ⇒
         (interp_concatenation c input) <|> (try_alts rest)
       end) concats
  end

with interp_concatenation (cat : ABNFConcatenation) (input : string)
  {struct cat} : Parser string (list ParseTree) :=
  match cat with
  | Concat reps ⇒
    (fix go (rs : list ABNFRepetition) : Parser string (list ParseTree) :=
       match rs with
       | [] ⇒ ret []
       | r :: rest ⇒
         ts <- interp_repetition r input ;;
         rest_ts <- go rest ;;
         ret (List.app ts rest_ts)
       end) reps
  end

with interp_repetition (rep : ABNFRepetition) (input : string)
  {struct rep} : Parser string (list ParseTree) :=
  match rep with
  | Rep None elem ⇒ interp_element elem input
  | Rep (Some spec) elem ⇒
    let lo := match repeat_min spec with
              | Some n ⇒ n
              | None ⇒ 0
              end in
    let hi := repeat_max spec in
    repeat_parser lo hi (interp_element elem input) input
  end

with interp_element (elem : ABNFElement) (input : string)
  {struct elem} : Parser string (list ParseTree) :=
  match elem with
  | ElemCharVal s ⇒ t <- match_ci_string s ;; ret [t]
  | ElemNumVal spec ⇒ t <- interp_numspec spec ;; ret [t]
  | ElemProseVal _ ⇒ fail
  | ElemGroup alt ⇒ interp_alternation alt input
  | ElemOption alt ⇒
    (interp_alternation alt input) <|> ret []
  | ElemRuleName name ⇒ resolve name input
  end.

End InterpAST.


Fixpoint resolve_rule (env : rule_env) (depth : nat) (name : string) (input : string)
  {struct depth} : Parser string (list ParseTree) :=
  match depth with
  | O ⇒ fail
  | Datatypes.S depth' ⇒
    match lookup_rule env name with
    | Some body ⇒
      ts <- interp_alternation (resolve_rule env depth') body input ;;
      ret [Node name ts]
    | None ⇒
      match interp_core_rule name with
      | Some p ⇒ t <- p input ;; ret [t]
      | None ⇒ fail
      end
    end
  end.


Definition make_parser (grammar : ABNFRulelist) (start_rule : string)
  (depth : nat) (input : string) : Parser string ParseTree :=
  let env := build_env grammar in
  match lookup_rule env start_rule with
  | None ⇒ fail
  | Some body ⇒
    ts <- interp_alternation (resolve_rule env depth) body input ;;
    ret (Node start_rule ts)
  end.

Definition parse_with_abnf (grammar_text : string) (start_rule : string)
  (input : string) (depth : nat) : option ParseTree :=
  match parse_abnf grammar_text with
  | None ⇒ None
  | Some rules ⇒
    run (make_parser rules start_rule depth input) input
  end.

Definition parse_and_print (grammar_text : string) (start_rule : string)
  (input : string) (depth : nat) : option string :=
  match parse_abnf grammar_text with
  | None ⇒ None
  | Some rules ⇒
    match make_parser rules start_rule depth input input with
    | Some (tree, _) ⇒ Some (pretty_print tree)
    | None ⇒ None
    end
  end.


Definition auto_depth (env : rule_env) (input : string) : nat :=
  List.length env × (1 + String.length input).

Definition parse_with_grammar (rules : ABNFRulelist) (start_rule : string)
  (input : string) : option ParseTree :=
  let env := build_env rules in
  run (make_parser rules start_rule (auto_depth env input) input) input.

Definition print_with_grammar (rules : ABNFRulelist) (start_rule : string)
  (input : string) : option string :=
  let env := build_env rules in
  let depth := auto_depth env input in
  match make_parser rules start_rule depth input input with
  | Some (tree, _) ⇒ Some (pretty_print tree)
  | None ⇒ None
  end.


Definition crlf_s : string := String (Ascii.ascii_of_nat 10) EmptyString.

Definition test_grammar_1 :=
  ("digit = %x30-39" ++ crlf_s)%string.

Example test_interp_digit :
  parse_with_abnf test_grammar_1 "digit" "5" 10 = Some (Node "digit" [Leaf "5"]).
Proof. reflexivity. Qed.

Definition test_grammar_2 :=
  ("ab = %x61 / %x62" ++ crlf_s)%string.

Example test_interp_alt :
  parse_with_abnf test_grammar_2 "ab" "b" 10 = Some (Node "ab" [Leaf "b"]).
Proof. reflexivity. Qed.

  ("hi = %x68 %x69" ++ crlf_s)%string.

Example test_interp_concat :
  parse_with_abnf test_grammar_3 "hi" "hi" 10 =
    Some (Node "hi" [Leaf "h"; Leaf "i"]).
