From Coq Require Import Morphisms.
From Riktig Require Import Ops Interface.Monoid Interface.Field
  Interface.VectorSpace Theory.Group Theory.VectorSpace.

Record LinearMap (F U V : Type) {FF : Field F}
    {VU : VectorSpace F U} {VV : VectorSpace F V} := {
  lm_fun :> U → V;
  lm_proper : Proper (equ ==> equ) lm_fun;
  lm_add : ∀ u v : U, lm_fun (u + v) == lm_fun u + lm_fun v;
  lm_scale : ∀ (a : F) (u : U), lm_fun (a *: u) == a *: lm_fun u
}.

Arguments lm_fun {F U V FF VU VV}.
Arguments lm_proper {F U V FF VU VV}.
Arguments lm_add {F U V FF VU VV}.
Arguments lm_scale {F U V FF VU VV}.


Section LinearMapEqu.
  Context {F U V : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}.

  #[global] Instance lm_equ : EqRel (LinearMap F U V) :=
    fun f g ⇒ ∀ u : U, f u == g u.

  #[global] Instance lm_equ_Equivalence : Equivalence (@equ (LinearMap F U V) lm_equ).
  Proof.
    apply Build_Equivalence.
    - intros f u. reflexivity.
    - intros f g Hfg u. symmetry. apply Hfg.
    - intros f g h Hfg Hgh u. rewrite (Hfg u). apply Hgh.
  Qed.

End LinearMapEqu.


Section LinearMapTheory.
  Context {F U V : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}.
  Context (f : LinearMap F U V).

  Theorem lm_zero : f 0 == 0.
  Proof.
    exact (hom_zero (lm_fun f) (lm_proper f) (lm_add f)).
  Qed.

  Theorem lm_neg : ∀ u : U, f (- u) == - f u.
  Proof.
    intros u.
    exact (hom_neg (lm_fun f) (lm_proper f) (lm_add f) u).
  Qed.

End LinearMapTheory.


Section LinearMapId.
  Context {F U : Type} {FF : Field F} {VU : VectorSpace F U}.

  Lemma lm_id_proper : Proper (equ ==> equ) (fun u : U ⇒ u).
  Proof.
    intros u v Huv.
    exact Huv.
  Qed.

  Lemma lm_id_add : ∀ u v : U, u + v == u + v.
  Proof.
    intros u v.
    reflexivity.
  Qed.

  Lemma lm_id_scale : ∀ (a : F) (u : U), a *: u == a *: u.
  Proof.
    intros a u.
    reflexivity.
  Qed.

  Definition lm_id : LinearMap F U U :=
    @Build_LinearMap F U U FF VU VU (fun u : U ⇒ u)
      lm_id_proper lm_id_add lm_id_scale.

End LinearMapId.


Section LinearMapComp.
  Context {F U V W : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}
          {VW : VectorSpace F W}.
  Context (g : LinearMap F V W) (f : LinearMap F U V).

  Lemma lm_comp_proper : Proper (equ ==> equ) (fun u : U ⇒ g (f u)).
  Proof.
    intros u v Huv.
    apply (lm_proper g), (lm_proper f).
    exact Huv.
  Qed.

  Lemma lm_comp_add : ∀ u v : U, g (f (u + v)) == g (f u) + g (f v).
  Proof.
    intros u v.
    rewrite (lm_proper g _ _ (lm_add f u v)).
    exact (lm_add g (f u) (f v)).
  Qed.

  Lemma lm_comp_scale : ∀ (a : F) (u : U), g (f (a *: u)) == a *: g (f u).
  Proof.
    intros a u.
    rewrite (lm_proper g _ _ (lm_scale f a u)).
    exact (lm_scale g a (f u)).
  Qed.

  Definition lm_comp : LinearMap F U W :=
    @Build_LinearMap F U W FF VU VW (fun u : U ⇒ g (f u))
      lm_comp_proper lm_comp_add lm_comp_scale.

End LinearMapComp.


Section LinearMapConst0.
  Context {F U V : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}.

  Lemma lm_const0_proper : Proper (equ ==> equ) (fun _ : U ⇒ (0 : V)).
  Proof.
    intros u v Huv.
    reflexivity.
  Qed.

  Lemma lm_const0_add : ∀ u v : U, (0 : V) == 0 + 0.
  Proof.
    intros u v.
    symmetry.
    exact (mon_idl 0).
  Qed.

  Lemma lm_const0_scale : ∀ (a : F) (u : U), (0 : V) == a *: 0.
  Proof.
    intros a u.
    symmetry.
    exact (scale_zero a).
  Qed.

  Definition lm_const0 : LinearMap F U V :=
    @Build_LinearMap F U V FF VU VV (fun _ : U ⇒ 0)
      lm_const0_proper lm_const0_add lm_const0_scale.

End LinearMapConst0.

Example lm_check (F U : Type) (FF : Field F) (VU : VectorSpace F U) :
  LinearMap F U U := lm_comp lm_id lm_id.
