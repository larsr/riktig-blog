Library Riktig.Lib.MeasureInd


Require Import Arith Lia.

Lemma measure_ind {A : Type} (f : A → nat) (P : A → Prop)
    (step : ∀ a, (∀ a', f a' < f a → P a') → P a) :
  ∀ a, P a.
Proof.
  intros a. remember (f a) as n eqn:E. revert a E.
  induction n as [n IH] using lt_wf_ind; intros a E.
  apply step; intros a' L. apply (IH (f a')); [lia | reflexivity].
Qed.

Lemma measure_ind2 {A B : Type} (f : A → B → nat) (P : A → B → Prop)
    (step : ∀ a b, (∀ a' b', f a' b' < f a b → P a' b') → P a b) :
  ∀ a b, P a b.
Proof.
  intros a b. remember (f a b) as n eqn:E. revert a b E.
  induction n as [n IH] using lt_wf_ind; intros a b E.
  apply step; intros a' b' L. apply (IH (f a' b')); [lia | reflexivity].
Qed.
