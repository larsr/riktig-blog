Library Riktig.Props

From Riktig Require Import Ops.

Class Assoc T (e : EqRel T) (op : BinOp T) :=
  assoc : ∀ a b c, op (op a b) c == op a (op b c).

Class Commutes T (e : EqRel T) (op : BinOp T) :=
  comm : ∀ a b, op a b == op b a.

Class Idl T (e : EqRel T) (op : BinOp T) unit :=
  idl : ∀ a, op unit a == a.

Class Idr T (e : EqRel T) (op : BinOp T) unit :=
  idr : ∀ a, op a unit == a.

Class Invl T (e : EqRel T) (op : BinOp T) (inv : T → T) unit :=
  invl : ∀ a, op (inv a) a == unit.

Class Invr T (e : EqRel T) (op : BinOp T) (inv : T → T) unit :=
  invr : ∀ a, op a (inv a) == unit.

Class Unique T (e : EqRel T) (P : Pred T) :=
  unique : ∀ a b, P a → P b → a == b.

Class Injective A B (e : EqRel A) (e' : EqRel B) (f : A → B) :=
  inj : ∀ a b, f a == f b → a == b.
