Lemma max_le_l : ∀ a b n : nat,
  (Nat.max a b ≤ n)%nat → (a ≤ n)%nat.
Proof.
  intros a b n Hn.
  exact (Nat.le_trans a (Nat.max a b) n (Nat.le_max_l a b) Hn).
Qed.

Lemma max_le_r : ∀ a b n : nat,
  (Nat.max a b ≤ n)%nat → (b ≤ n)%nat.
Proof.
  intros a b n Hn.
  exact (Nat.le_trans b (Nat.max a b) n (Nat.le_max_r a b) Hn).
Qed.

Section Limits.
  Context {F V : Type} `{NormedSpace F V}.

  Definition seq_lim (x : nat → V) (L : V) : Prop :=
    ∀ eps : F, 0 < eps →
      ∃ N : nat, ∀ n : nat, (N ≤ n)%nat → ‖ x n + - L ‖ < eps.

  Theorem lim_const : ∀ v : V, seq_lim (fun _ ⇒ v) v.
  Proof.
    intros v eps Heps.
    ∃ 0%nat.
    intros n Hn.
    apply (le_lt_trans _ 0 eps).
    - apply eq_le. apply norm_sub_self.
    - exact Heps.
  Qed.

  Theorem lim_unique : ∀ (x : nat → V) (L1 L2 : V),
    seq_lim x L1 → seq_lim x L2 → L1 == L2.
  Proof.
    intros x L1 L2 Hx1 Hx2.
    apply sub_eq_zero.
    apply norm_def.
    apply nonneg_below_all_pos_is_zero.
    - apply norm_nonneg.
    - intros eps Heps.
      destruct (Hx1 (half eps) (half_pos eps Heps)) as [N1 HN1].
      destruct (Hx2 (half eps) (half_pos eps Heps)) as [N2 HN2].
      apply lt_le.
      apply (le_lt_trans (‖ L1 + - L2 ‖)
        (‖ L1 + - x (Nat.max N1 N2) ‖ + ‖ x (Nat.max N1 N2) + - L2 ‖) eps).
      + apply norm_sub_triangle.
      + apply (lt_le_trans _ (half eps + half eps) eps).
        × apply add_lt_mono.
          -- rewrite (norm_sub_swap L1 (x (Nat.max N1 N2))).
             apply HN1.
             apply Nat.le_max_l.
          -- apply HN2.
             apply Nat.le_max_r.
        × apply eq_le. apply half_sum.
  Qed.

  Theorem lim_add : ∀ (x y : nat → V) (L M : V),
    seq_lim x L → seq_lim y M → seq_lim (fun n ⇒ x n + y n) (L + M).
  Proof.
    intros x y L M Hx Hy eps Heps.
    destruct (Hx (half eps) (half_pos eps Heps)) as [N1 HN1].
    destruct (Hy (half eps) (half_pos eps Heps)) as [N2 HN2].
    ∃ (Nat.max N1 N2).
    intros n Hn.
    apply (le_lt_trans _ (‖ x n + - L ‖ + ‖ y n + - M ‖) eps).
    - apply norm_sub_pair.
    - apply (lt_le_trans _ (half eps + half eps) eps).
      + apply add_lt_mono.
        × apply HN1. apply (max_le_l N1 N2 n). exact Hn.
        × apply HN2. apply (max_le_r N1 N2 n). exact Hn.
      + apply eq_le. apply half_sum.
  Qed.

  Theorem lim_scale : ∀ (a : F) (x : nat → V) (L : V),
    seq_lim x L → seq_lim (fun n ⇒ a *: x n) (a *: L).
  Proof.
    intros a x L Hx eps Heps.
    destruct (Hx (eps × inv (1 + abs a))
        (mul_pos eps (inv (1 + abs a)) Heps
           (pos_inv (1 + abs a) (one_plus_nonneg_pos (abs a) (abs_nonneg a)))))
      as [N HN].
    ∃ N.
    intros n Hn.
    apply (le_lt_trans _ (abs a × ‖ x n + - L ‖) eps).
    - apply eq_le. apply norm_scale_sub.
    - apply (lt_le_trans _ ((1 + abs a) × (eps × inv (1 + abs a))) eps).
      + apply mul_lt_pad.
        × apply abs_nonneg.
        × apply norm_nonneg.
        × apply HN. exact Hn.
      + apply eq_le.
        apply field_mul_div_cancel.
        apply pos_neq_0.
        apply one_plus_nonneg_pos.
        apply abs_nonneg.
  Qed.

  Theorem lim_neg : ∀ (x : nat → V) (L : V),
    seq_lim x L → seq_lim (fun n ⇒ - x n) (- L).
  Proof.
    intros x L Hx eps Heps.
    destruct (Hx eps Heps) as [N HN].
    ∃ N.
    intros n Hn.
    apply (le_lt_trans _ (‖ x n + - L ‖) eps).
    - apply eq_le. apply norm_neg_add.
    - apply HN. exact Hn.
  Qed.

End Limits.

#[global] Instance seq_lim_proper {F V : Type} `{NormedSpace F V} :
  Proper (pointwise_relation nat equ ==> equ ==> iff) (@seq_lim F V _ _ _ _).
Proof.
  intros x y Hxy L M HLM.
  split.
  - intros Hx eps Heps.
    destruct (Hx eps Heps) as [N HN].
    ∃ N.
    intros n Hn.
    rewrite <- (Hxy n), <- HLM.
    apply HN.
    exact Hn.
  - intros Hy eps Heps.
    destruct (Hy eps Heps) as [N HN].
    ∃ N.
    intros n Hn.
    rewrite (Hxy n), HLM.
    apply HN.
    exact Hn.
Qed.
