From Coq Require Import RelationClasses.
From Coq.Vectors Require Fin.
From Riktig Require Import Ops Interface.Field Interface.VectorSpace
  Theory.FieldVS Theory.BigOp.


Section Lincomb.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V}.

  Definition lincomb {n : nat} (b : Fin.t n → V) (c : Fin.t n → F) : V :=
    bigopF (fun i ⇒ c i *: b i).

  Lemma lincomb_def : ∀ (n : nat) (b : Fin.t n → V) (c : Fin.t n → F),
    lincomb b c == bigopF (fun i ⇒ c i *: b i).
  Proof.
    intros n b c. reflexivity.
  Qed.

  Lemma lincomb_proper : ∀ (n : nat) (b : Fin.t n → V)
      (c c' : Fin.t n → F),
    (∀ i : Fin.t n, c i == c' i) → lincomb b c == lincomb b c'.
  Proof.
    intros n b c c' Hc.
    exact (bigopF_ext n (fun i ⇒ c i *: b i) (fun i ⇒ c' i *: b i)
             (fun i ⇒ sc_proper (c i) (c' i) (Hc i)
                         (b i) (b i) (reflexivity (b i)))).
  Qed.

  Lemma lincomb_add : ∀ (n : nat) (b : Fin.t n → V)
      (c d : Fin.t n → F),
    lincomb b (c + d) == lincomb b c + lincomb b d.
  Proof.
    intros n b c d.
    rewrite (lincomb_def n b (c + d)), (lincomb_def n b c),
      (lincomb_def n b d),
      (bigopF_ext n (fun i ⇒ (c + d) i *: b i)
         (fun i ⇒ c i *: b i + d i *: b i)
         (fun i ⇒ sc_add_scale (c i) (d i) (b i))),
      (bigopF_add n (fun i ⇒ c i *: b i) (fun i ⇒ d i *: b i)).
    reflexivity.
  Qed.

  Lemma lincomb_scale : ∀ (n : nat) (a : F) (b : Fin.t n → V)
      (c : Fin.t n → F),
    lincomb b (a *: c) == a *: lincomb b c.
  Proof.
    intros n a b c.
    rewrite (lincomb_def n b (a *: c)), (lincomb_def n b c),
      (bigopF_ext n (fun i ⇒ (a *: c) i *: b i)
         (fun i ⇒ a *: (c i *: b i))
         (fun i ⇒ sc_scale_scale a (c i) (b i))),
      (bigopF_scale n a (fun i ⇒ c i *: b i)).
    reflexivity.
  Qed.

End Lincomb.
