Section Continuity.
  Context {F U V : Type} `{NormedSpace F U}.
  Context {VSV : VectorSpace F V} {NSV : NormedSpace F V}.

  Definition limit_at (f : U → V) (a : U) (L : V) : Prop :=
    ∀ eps : F, 0 < eps →
      ∃ delta : F, 0 < delta ∧
        ∀ u : U, ‖ u + - a ‖ < delta → ‖ f u + - L ‖ < eps.

  Theorem limit_at_unique : ∀ (f : U → V) (a : U) (L1 L2 : V),
    limit_at f a L1 → limit_at f a L2 → L1 == L2.
  Proof.
    intros f a L1 L2 Hl1 Hl2.
    apply sub_eq_zero.
    apply norm_def.
    apply nonneg_below_all_pos_is_zero.
    - apply norm_nonneg.
    - intros eps Heps.
      destruct (Hl1 (half eps) (half_pos eps Heps)) as [d1 [Hd1 Hf1]].
      destruct (Hl2 (half eps) (half_pos eps Heps)) as [d2 [Hd2 Hf2]].
      apply lt_le.
      apply (le_lt_trans (‖ L1 + - L2 ‖)
        (‖ L1 + - f a ‖ + ‖ f a + - L2 ‖) eps).
      + apply norm_sub_triangle.
      + apply (lt_le_trans _ (half eps + half eps) eps).
        × apply add_lt_mono.
          -- rewrite (norm_sub_swap L1 (f a)).
             apply Hf1.
             apply norm_sub_self_lt.
             exact Hd1.
          -- apply Hf2.
             apply norm_sub_self_lt.
             exact Hd2.
        × apply eq_le. apply half_sum.
  Qed.

  Definition continuous_at (f : U → V) (a : U) : Prop :=
    limit_at f a (f a).

  Definition continuous (f : U → V) : Prop :=
    ∀ a : U, continuous_at f a.

  Theorem limit_at_add : ∀ (f g : U → V) (a : U) (L M : V),
    limit_at f a L → limit_at g a M →
    limit_at (fun u : U ⇒ f u + g u) a (L + M).
  Proof.
    intros f g a L M Hf Hg eps Heps.
    destruct (Hf (half eps) (half_pos eps Heps)) as [d1 [Hd1 Hf1]].
    destruct (Hg (half eps) (half_pos eps Heps)) as [d2 [Hd2 Hg2]].
    destruct (pos_min_exists d1 d2 Hd1 Hd2) as [d [Hd [Hdd1 Hdd2]]].
    ∃ d.
    split.
    - exact Hd.
    - intros u Hu.
      apply (le_lt_trans _ (‖ f u + - L ‖ + ‖ g u + - M ‖) eps).
      + apply norm_sub_pair.
      + apply (lt_le_trans _ (half eps + half eps) eps).
        × apply add_lt_mono.
          -- apply Hf1. exact (lt_le_trans (‖ u + - a ‖) d d1 Hu Hdd1).
          -- apply Hg2. exact (lt_le_trans (‖ u + - a ‖) d d2 Hu Hdd2).
        × apply eq_le. apply half_sum.
  Qed.

  Theorem limit_at_scale : ∀ (c : F) (f : U → V) (a : U) (L : V),
    limit_at f a L → limit_at (fun u : U ⇒ c *: f u) a (c *: L).
  Proof.
    intros c f a L Hf eps Heps.
    destruct (Hf (eps × inv (1 + abs c))
        (mul_pos eps (inv (1 + abs c)) Heps
           (pos_inv (1 + abs c) (one_plus_nonneg_pos (abs c) (abs_nonneg c)))))
      as [d [Hd Hf1]].
    ∃ d.
    split.
    - exact Hd.
    - intros u Hu.
      apply (le_lt_trans _ (abs c × ‖ f u + - L ‖) eps).
      + apply eq_le. apply norm_scale_sub.
      + apply (lt_le_trans _ ((1 + abs c) × (eps × inv (1 + abs c))) eps).
        × apply mul_lt_pad.
          -- apply abs_nonneg.
          -- apply norm_nonneg.
          -- apply Hf1. exact Hu.
        × apply eq_le.
          apply field_mul_div_cancel.
          apply pos_neq_0.
          apply one_plus_nonneg_pos.
          apply abs_nonneg.
  Qed.

  Theorem limit_at_neg : ∀ (f : U → V) (a : U) (L : V),
    limit_at f a L → limit_at (fun u : U ⇒ - f u) a (- L).
  Proof.
    intros f a L Hf eps Heps.
    destruct (Hf eps Heps) as [d [Hd Hf1]].
    ∃ d.
    split.
    - exact Hd.
    - intros u Hu.
      apply (le_lt_trans _ (‖ f u + - L ‖) eps).
      + apply eq_le. apply norm_neg_add.
      + apply Hf1. exact Hu.
  Qed.


  Lemma lm_norm_sub : ∀ (f : LinearMap F U V) (u v : U),
    ‖ f u + - f v ‖ == ‖ f (u + - v) ‖.
  Proof.
    intros f u v.
    rewrite (lm_add f u (- v)), (lm_neg f v).
    reflexivity.
  Qed.

  Definition lm_bounded (f : LinearMap F U V) : Prop :=
    ∃ C : F, 0 ≤ C ∧ ∀ u : U, ‖ f u ‖ ≤ C × ‖ u ‖.

  Theorem bounded_continuous : ∀ f : LinearMap F U V,
    lm_bounded f → continuous (lm_fun f).
  Proof.
    intros f [C [HC Hb]] a eps Heps.
    ∃ (eps × inv (1 + C)).
    split.
    - exact (mul_pos eps (inv (1 + C)) Heps
        (pos_inv (1 + C) (one_plus_nonneg_pos C HC))).
    - intros u Hu.
      rewrite (lm_norm_sub f u a).
      apply (le_lt_trans _ (C × ‖ u + - a ‖) eps).
      + exact (Hb (u + - a)).
      + apply (lt_le_trans _ ((1 + C) × (eps × inv (1 + C))) eps).
        × apply mul_lt_pad.
          -- exact HC.
          -- apply norm_nonneg.
          -- exact Hu.
        × apply eq_le.
          apply field_mul_div_cancel.
          apply pos_neq_0.
          apply one_plus_nonneg_pos.
          exact HC.
  Qed.


  Theorem continuous_at_seq_lim : ∀ (f : U → V) (a : U) (x : nat → U),
    continuous_at f a → seq_lim x a →
    seq_lim (fun n : nat ⇒ f (x n)) (f a).
  Proof.
    intros f a x Hf Hx eps Heps.
    destruct (Hf eps Heps) as [delta [Hdelta Hclose]].
    destruct (Hx delta Hdelta) as [N HN].
    ∃ N.
    intros n Hn.
    apply Hclose.
    apply HN.
    exact Hn.
  Qed.

End Continuity.

#[global] Instance limit_at_proper {F U V : Type} `{NormedSpace F U}
    {VSV : VectorSpace F V} {NSV : NormedSpace F V} :
  Proper (pointwise_relation U equ ==> equ ==> equ ==> iff)
    (@limit_at F U V _ _ _ _ _ _).
Proof.
  intros f g Hfg a b Hab L M HLM.
  split.
  - intros Hf eps Heps.
    destruct (Hf eps Heps) as [delta [Hdelta Hclose]].
    ∃ delta.
    split.
    + exact Hdelta.
    + intros u Hu.
      rewrite <- (Hfg u), <- HLM.
      apply Hclose.
      rewrite Hab.
      exact Hu.
  - intros Hg eps Heps.
    destruct (Hg eps Heps) as [delta [Hdelta Hclose]].
    ∃ delta.
    split.
    + exact Hdelta.
    + intros u Hu.
      rewrite (Hfg u), HLM.
      apply Hclose.
      rewrite <- Hab.
      exact Hu.
Qed.


Section ContinuityComp.
  Context {F U V W : Type} `{NormedSpace F U}.
  Context {VSV : VectorSpace F V} {NSV : NormedSpace F V}.
  Context {VSW : VectorSpace F W} {NSW : NormedSpace F W}.

  Theorem continuous_comp : ∀ (f : U → V) (g : V → W) (a : U),
    continuous_at f a → continuous_at g (f a) →
    continuous_at (fun u : U ⇒ g (f u)) a.
  Proof.
    intros f g a Hf Hg eps Heps.
    destruct (Hg eps Heps) as [d1 [Hd1 Hg1]].
    destruct (Hf d1 Hd1) as [d2 [Hd2 Hf2]].
    ∃ d2.
    split.
    - exact Hd2.
    - intros u Hu.
      apply Hg1.
      apply Hf2.
      exact Hu.
  Qed.

End ContinuityComp.
