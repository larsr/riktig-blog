Library Riktig.Interface.Reals

From Coq Require Import Morphisms.
From Riktig Require Import Ops Interface.Field.

Class Reals (R : Type) `{Field R} := {
  
  r_le : Rel R;
  r_le_proper : Proper (equ ==> equ ==> iff) r_le;
  r_le_refl : ∀ x : R, r_le x x;
  r_le_trans : ∀ x y z : R, r_le x y → r_le y z → r_le x z;
  r_le_antisym : ∀ x y : R, r_le x y → r_le y x → x == y;
  r_le_total : ∀ x y : R, r_le x y ∨ r_le y x;

  
  r_le_add_compat : ∀ x y z : R, r_le x y → r_le (x + z) (y + z);
  r_le_mul_compat : ∀ x y : R, r_le 0 x → r_le 0 y → r_le 0 (x × y);

  
  r_complete : ∀ P : R → Prop,
    (∃ x : R, P x) →
    (∃ b : R, ∀ x : R, P x → r_le x b) →
    ∃ s : R,
      (∀ x : R, P x → r_le x s) ∧
      (∀ b : R, (∀ x : R, P x → r_le x b) → r_le s b)
}.

#[global] Hint Mode Reals ! - : typeclass_instances.
