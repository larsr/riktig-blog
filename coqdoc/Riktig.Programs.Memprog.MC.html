Library Riktig.Programs.Memprog.MC


From Coq Require Import List ZArith.
Import ListNotations.
From Riktig Require Import Programs.Memprog.Syntax.
From Riktig Require Import Programs.Memprog.Machine.



Inductive Flag : Type := FLT | FEQ | FGT.

Definition flag_lt (f : Flag) : bool := match f with FLT ⇒ true | _ ⇒ false end.
Definition flag_eq (f : Flag) : bool := match f with FEQ ⇒ true | _ ⇒ false end.
Definition flag_gt (f : Flag) : bool := match f with FGT ⇒ true | _ ⇒ false end.

Definition RegFile (val : Type) := Reg → val.

Definition get_reg {val : Type} (rf : RegFile val) (r : Reg) : val := rf r.

Definition set_reg {val : Type} (r : Reg) (v : val) (rf : RegFile val)
  : RegFile val :=
  fun r' ⇒ if reg_eqb r r' then v else rf r'.

Definition init_regs {val : Type} (v0 : val) : RegFile val := fun _ ⇒ v0.


Record MachState (val : Type) : Type := MkMachState {
  ms_regs : RegFile val;
  ms_pc : val;
  ms_sp : val;
  ms_flags : Flag
}.

Arguments MkMachState {val}.
Arguments ms_regs {val}.
Arguments ms_pc {val}.
Arguments ms_sp {val}.
Arguments ms_flags {val}.

Definition init_mach_state {val : Type} (v0 pc0 sp0 : val) : MachState val :=
  MkMachState (init_regs v0) pc0 sp0 FEQ.


Section MCInterp.
  Context {addr val : Type}.
  Context (val_add val_sub : val → val → val).
  Context (val_ltb val_eqb : val → val → bool).
  Context (val1 : val).
  Context (to_addr : val → addr).

  Definition cmp_flag (a b : val) : Flag :=
    if val_ltb a b then FLT
    else if val_eqb a b then FEQ
    else FGT.

  Definition adv_pc (ms : MachState val) : MachState val :=
    MkMachState (ms_regs ms) (val_add (ms_pc ms) val1) (ms_sp ms) (ms_flags ms).

  Definition branch_if (ms : MachState val) (cond : bool) (off : val)
    : MachState val :=
    if cond
    then MkMachState (ms_regs ms) (val_add (ms_pc ms) off) (ms_sp ms) (ms_flags ms)
    else adv_pc ms.


  Definition mc_step (ms : MachState val)
    : MemT addr val (option (MachState val)) :=
    do! instr <- read_instr_m (to_addr (ms_pc ms));
    let pc := ms_pc ms in
    let sp := ms_sp ms in
    let rf := ms_regs ms in
    let reg := get_reg rf in
    let fl := ms_flags ms in
    match instr with
    | HALT ⇒ ret_t None
    | NOP ⇒ ret_t (Some (adv_pc ms))

    | MOV rd rs ⇒
        ret_t (Some (adv_pc (MkMachState (set_reg rd (reg rs) rf) pc sp fl)))
    | LI rd v ⇒
        ret_t (Some (adv_pc (MkMachState (set_reg rd v rf) pc sp fl)))

    | ADD rd rs1 rs2 ⇒
        ret_t (Some (adv_pc (MkMachState (set_reg rd (val_add (reg rs1) (reg rs2)) rf) pc sp fl)))
    | ADDI rd rs v ⇒
        ret_t (Some (adv_pc (MkMachState (set_reg rd (val_add (reg rs) v) rf) pc sp fl)))
    | SUB rd rs1 rs2 ⇒
        ret_t (Some (adv_pc (MkMachState (set_reg rd (val_sub (reg rs1) (reg rs2)) rf) pc sp fl)))
    | SUBI rd rs v ⇒
        ret_t (Some (adv_pc (MkMachState (set_reg rd (val_sub (reg rs) v) rf) pc sp fl)))

    | LW rd rbase off ⇒
        do! v <- read_m (to_addr (val_add (reg rbase) off));
        ret_t (Some (adv_pc (MkMachState (set_reg rd v rf) pc sp fl)))
    | SW rs rbase off ⇒
        write_m (to_addr (val_add (reg rbase) off)) (reg rs) >>
        ret_t (Some (adv_pc ms))

    | CMP rs1 rs2 ⇒
        ret_t (Some (adv_pc (MkMachState rf pc sp (cmp_flag (reg rs1) (reg rs2)))))
    | CMPI rs v ⇒
        ret_t (Some (adv_pc (MkMachState rf pc sp (cmp_flag (reg rs) v))))

    | B off ⇒ ret_t (Some (MkMachState rf (val_add pc off) sp fl))
    | BEQ off ⇒ ret_t (Some (branch_if ms (flag_eq fl) off))
    | BNE off ⇒ ret_t (Some (branch_if ms (negb (flag_eq fl)) off))
    | BLT off ⇒ ret_t (Some (branch_if ms (flag_lt fl) off))
    | BLE off ⇒ ret_t (Some (branch_if ms (orb (flag_lt fl) (flag_eq fl)) off))
    | BGT off ⇒ ret_t (Some (branch_if ms (flag_gt fl) off))
    | BGE off ⇒ ret_t (Some (branch_if ms (orb (flag_gt fl) (flag_eq fl)) off))
    | BR rs ⇒ ret_t (Some (MkMachState rf (reg rs) sp fl))

    | PUSH rs ⇒
        write_m (to_addr sp) (reg rs) >>
        ret_t (Some (adv_pc (MkMachState rf pc (val_add sp val1) fl)))
    | POP rd ⇒
        let sp' := val_sub sp val1 in
        do! v <- read_m (to_addr sp');
        ret_t (Some (adv_pc (MkMachState (set_reg rd v rf) pc sp' fl)))

    | CALL off ⇒
        write_m (to_addr sp) (val_add pc val1) >>
        ret_t (Some (MkMachState rf (val_add pc off) (val_add sp val1) fl))
    | RET ⇒
        let sp' := val_sub sp val1 in
        do! ra <- read_m (to_addr sp');
        ret_t (Some (MkMachState rf ra sp' fl))
    end.


  Fixpoint mc_run (fuel : nat) (ms : MachState val)
    : MemT addr val (MachState val) :=
    match fuel with
    | O ⇒ fail_t
    | S fuel' ⇒
        do! step <- mc_step ms;
        match step with
        | None ⇒ ret_t ms
        | Some ms' ⇒ mc_run fuel' ms'
        end
    end.


  Fixpoint load_program (base : val) (instrs : list (Instr val))
    : MemT addr val unit :=
    match instrs with
    | [] ⇒ ret_t tt
    | i :: rest ⇒
        write_instr_m (to_addr base) i >>
        load_program (val_add base val1) rest
    end.

  Fixpoint load_data (base : val) (vals : list val)
    : MemT addr val unit :=
    match vals with
    | [] ⇒ ret_t tt
    | v :: rest ⇒
        write_m (to_addr base) v >>
        load_data (val_add base val1) rest
    end.

End MCInterp.


Local Open Scope Z_scope.

Definition mc_run_z : nat → MachState Z → MemT Z Z (MachState Z) :=
  mc_run Z.add Z.sub Z.ltb Z.eqb 1 (fun v : Z ⇒ v).

Definition load_program_z : Z → list (Instr Z) → MemT Z Z unit :=
  load_program Z.add 1 (fun v : Z ⇒ v).

Definition load_data_z : Z → list Z → MemT Z Z unit :=
  load_data Z.add 1 (fun v : Z ⇒ v).


Definition rel (from to : Z) : Z := to - from.


Definition setup_and_run_mc
  (stack_size : nat) (data : list Z) (code : list (Instr Z))
  (seed : Z → RegFile Z → RegFile Z) (fuel : nat)
  : MemT Z Z (Z × MachState Z) :=
  do! base <- alloc_m (stack_size + length data + length code)%nat;
  let data_base := base + Z.of_nat stack_size in
  let code_base := data_base + Z.of_nat (length data) in
  load_data_z data_base data >>
  load_program_z code_base code >>
  do! ms <- mc_run_z fuel
              (MkMachState (seed data_base (init_regs 0)) code_base base FEQ);
  ret_t (data_base, ms).



Definition swap_mc : list (Instr Z) :=
  [ ADD R3 R0 R1
  ; LW R4 R3 0
  ; ADD R3 R0 R2
  ; LW R5 R3 0
  ; ADD R3 R0 R1
  ; SW R5 R3 0
  ; ADD R3 R0 R2
  ; SW R4 R3 0
  ; HALT
  ].


Definition sum_mc : list (Instr Z) :=
  [ LI R2 0
  ; LI R3 0
  ; CMP R3 R1
  ; BGE (rel 3 9)
  ; ADD R4 R0 R3
  ; LW R5 R4 0
  ; ADD R2 R2 R5
  ; ADDI R3 R3 1
  ; B (rel 8 2)
  ; HALT
  ].


Definition swap_mc_demo : MemT Z Z Z :=
  do! (data_base, _) <-
    setup_and_run_mc 8 [10; 20; 30; 40; 50] swap_mc
      (fun base rf ⇒ set_reg R0 base (set_reg R1 1 (set_reg R2 3 rf)))
      100;
  ret_t data_base.

Example swap_mc_cells :
  (let '(r, m) := run_z 5000 swap_mc_demo 128 in mem_to_list m r 5)
  = Some [10; 40; 30; 20; 50].
Proof. vm_compute. reflexivity. Qed.


Definition sum_mc_demo : MemT Z Z Z :=
  do! (_, ms) <-
    setup_and_run_mc 16 [1; 2; 3; 4; 5] sum_mc
      (fun base rf ⇒ set_reg R0 base (set_reg R1 5 rf))
      1000;
  ret_t (get_reg (ms_regs ms) R2).

Example sum_mc_result :
  fst (run_z 4000 sum_mc_demo 128) = Some 15.
Proof. vm_compute. reflexivity. Qed.
