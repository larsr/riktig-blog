Require Import Arith Lia.
Require Import Coq.Arith.Wf_nat.
Require Import List. Import ListNotations.
Require Import Riktig.Interface.Decidable Riktig.Theory.Decidable.
Require Import Riktig.Lib.Fun.
Require Import Riktig.Interface.DecTotalOrder.
Require Import Riktig.Impl.Orders.

Section Merge.
  Context {A : Type} `{O : DecTotalOrder A}.
  Local Open Scope order_scope.

  Definition merge : list A → list A → list A.
   apply (uncurry ∘ (Fix (well_founded_ltof (list A × list A)
                (fun p ⇒ length (fst p) + length (snd p)))
             (fun _ ⇒ list A))).
    refine
    (
      fun p merge ⇒
        match p as x return p=x → list A with
        | (a, []) ⇒ fun _ ⇒ a
        | ([], b) ⇒ fun _ ⇒ b
        | (a0::a', b0::b') ⇒
            fun e ⇒
            if dec (a0 < b0)
              then a0 :: merge (a', b0 :: b') _
              else b0 :: merge (a0 :: a', b') _
        end eq_refl
    );
    (subst p; unfold ltof; simpl; abstract lia).
   Defined.

  Lemma merge_nil_l: ∀ l : list A, merge [] l = l.
    now destruct l. Qed.
  Lemma merge_nil_r: ∀ l : list A, merge l [] = l.
    now destruct l. Qed.
  Lemma merge_cons: ∀ (a : A) (b : list A) (c : A) (d : list A),
      merge (a::b) (c::d) = if dec (a < c) then a::merge b (c::d) else c::merge (a::b) d.
    intros; unfold merge at 1, compose, uncurry.
    rewrite Fix_eq.
    - destruct dec; f_equal.
    - intros [[ | x x'] [ | y y']] f g H; auto.
      destruct dec; rewrite !H; f_equal.
   Qed.

End Merge.

Lemma merge_cons_ltb: ∀ (a : nat) (b : list nat) (c : nat) (d : list nat),
    merge (a::b) (c::d) = if a<?c then a::merge b (c::d) else c::merge (a::b) d.
  intros; rewrite merge_cons; apply dec_if_bool; symmetry; apply Nat.ltb_lt. Qed.

Check Logic.eq_refl : merge [1;3;5] [2;4;6] = [1;2;3;4;5;6].
Check Logic.eq_refl : merge [] [1;2] = [1;2].
Check Logic.eq_refl : merge [1;2] [] = [1;2].
Check Logic.eq_refl : merge [2;4;6;8] [1;3] = [1;2;3;4;6;8].
Check Logic.eq_refl : merge [1;1;2] [1;3] = [1;1;1;2;3].
