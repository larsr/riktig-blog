Library Riktig.Impl.DetInterface

From Coq Require Import Morphisms.
From Coq.Vectors Require Fin.
From Riktig Require Export Interface.Determinant Impl.DetAlternating
  Impl.FinVecBasis.

Section DetMInstance.
  Context {F : Type} {FF : Field F}.


  Lemma upd_eq_upd_row : ∀ (n : nat) (A : Matrix n n F) (i : Fin.t n)
      (r : Fin.t n → F) (i' j : Fin.t n),
    upd A i r i' j == upd_row A i r i' j.
  Proof.
    intros n A i r i' j.
    unfold upd, upd_row.
    destruct (dec (i = i')) as [Heq | Hne].
    - destruct (dec (i' = i)) as [_ | Hne'].
      + reflexivity.
      + destruct (Hne' (eq_sym Heq)).
    - destruct (dec (i' = i)) as [Heq' | _].
      + destruct (Hne (eq_sym Heq')).
      + reflexivity.
  Qed.

  Lemma detM_upd_eq_upd_row : ∀ (n : nat) (A : Matrix n n F)
      (i : Fin.t n) (r : Fin.t n → F),
    detM (upd A i r) == detM (upd_row A i r).
  Proof.
    intros n A i r.
    apply detM_proper.
    intros i' j.
    exact (upd_eq_upd_row n A i r i' j).
  Qed.


  #[global] Instance detM_mf_proper (n : nat) :
    Proper (@famEq (Fin.t n → F) _ n ==> equ) (@detM F FF n).
  Proof.
    exact (detM_proper n).
  Qed.

  Lemma detM_mf_add : ∀ (n : nat) (vs : Fin.t n → Fin.t n → F)
      (i : Fin.t n) (x y : Fin.t n → F),
    detM (upd vs i (x + y)) == detM (upd vs i x) + detM (upd vs i y).
  Proof.
    intros n; destruct n as [| p]; intros vs i x y.
    - exact (Fin.case0
               (fun i0 : Fin.t 0 ⇒
                  detM (upd vs i0 (x + y)) == detM (upd vs i0 x) + detM (upd vs i0 y))
               i).
    - rewrite (detM_upd_eq_upd_row (S p) vs i (x + y)),
        (detM_upd_eq_upd_row (S p) vs i x), (detM_upd_eq_upd_row (S p) vs i y).
      exact (det_rowI_add p i vs x y).
  Qed.

  Lemma detM_mf_scale : ∀ (n : nat) (vs : Fin.t n → Fin.t n → F)
      (i : Fin.t n) (a : F) (x : Fin.t n → F),
    detM (upd vs i (a *: x)) == a × detM (upd vs i x).
  Proof.
    intros n; destruct n as [| p]; intros vs i a x.
    - exact (Fin.case0
               (fun i0 : Fin.t 0 ⇒
                  detM (upd vs i0 (a *: x)) == a × detM (upd vs i0 x))
               i).
    - rewrite (detM_upd_eq_upd_row (S p) vs i (a *: x)),
        (detM_upd_eq_upd_row (S p) vs i x).
      exact (det_rowI_scale p i vs a x).
  Qed.

  Definition detM_mf (n : nat) : MultilinearForm n F (Fin.t n → F) :=
    @Build_MultilinearForm n F (Fin.t n → F) FF _
      (@detM F FF n) (detM_mf_proper n) (detM_mf_add n) (detM_mf_scale n).


  Definition detM_alt (n : nat) : Alternating (detM_mf n) :=
    @Build_Alternating n F (Fin.t n → F) FF _ (detM_mf n)
      (@detM_alternating F FF n).


  Lemma detM_basis_1 : ∀ n : nat,
    detM_mf n (basis_vec (FinVec_Basis n)) == 1.
  Proof.
    intros n.
    exact (det_id n).
  Qed.

  Definition detM_Determinant (n : nat) :
      Determinant n F (Fin.t n → F) (FinVec_Basis n) :=
    @Build_Determinant n F (Fin.t n → F) FF _ (FinVec_Basis n)
      (detM_mf n) (detM_alt n) (detM_basis_1 n).

End DetMInstance.

Example detM_determinant_check (F : Type) (FF : Field F) (n : nat) :
  Determinant n F (Fin.t n → F) (FinVec_Basis n) := detM_Determinant n.
