Section EvenIso.
  Context {F : Type} `{FF : Field F}.

  Lemma c2_even_add_closed : ∀ x y : C2 F,
    c2_even x → c2_even y → c2_even (x + y).
  Proof.
    intros x y [Hx1 Hx2] [Hy1 Hy2].
    split; simpl.
    - rewrite Hx1, Hy1. exact (idl (0 : F)).
    - rewrite Hx2, Hy2. exact (idl (0 : F)).
  Qed.

  Lemma c2_even_neg_closed : ∀ x : C2 F, c2_even x → c2_even (- x).
  Proof.
    intros x [Hx1 Hx2].
    split; simpl.
    - rewrite Hx1. exact neg_zero.
    - rewrite Hx2. exact neg_zero.
  Qed.


  Definition c2_to_c (x : C2 F) : C F := mkC (c2_s x) (c2_12 x).
  Definition c_to_c2 (z : C F) : C2 F := mkC2 (re z) 0 0 (im z).

  #[global] Instance c2_to_c_proper : Proper (equ ==> equ) c2_to_c.
  Proof.
    intros x y [Hs [H1 [H2 H12]]].
    split; [exact Hs | exact H12].
  Qed.

  #[global] Instance c_to_c2_proper : Proper (equ ==> equ) c_to_c2.
  Proof.
    intros z w [Hr Hi].
    repeat split; simpl; [exact Hr | reflexivity | reflexivity | exact Hi].
  Qed.

  Lemma c_to_c2_to_c : ∀ x : C2 F, c2_even x → c_to_c2 (c2_to_c x) == x.
  Proof.
    intros x [Hx1 Hx2].
    repeat split; simpl.
    - reflexivity.
    - symmetry. exact Hx1.
    - symmetry. exact Hx2.
    - reflexivity.
  Qed.

  Lemma c2_to_c_to_c2 : ∀ z : C F, c2_to_c (c_to_c2 z) == z.
  Proof.
    intros z.
    split; reflexivity.
  Qed.

  Lemma c_to_c2_even : ∀ z : C F, c2_even (c_to_c2 z).
  Proof.
    intros z.
    split; reflexivity.
  Qed.


  Lemma c2_to_c_zero : c2_to_c (0 : C2 F) == 0.
  Proof. reflexivity. Qed.

  Lemma c2_to_c_one : c2_to_c (1 : C2 F) == 1.
  Proof. reflexivity. Qed.

  Lemma c2_to_c_add : ∀ x y : C2 F,
    c2_to_c (x + y) == c2_to_c x + c2_to_c y.
  Proof.
    intros x y.
    split; reflexivity.
  Qed.

  Lemma c2_to_c_neg : ∀ x : C2 F, c2_to_c (- x) == - c2_to_c x.
  Proof.
    intros x.
    split; reflexivity.
  Qed.

  Lemma c2_to_c_mul_even : ∀ x y : C2 F,
    c2_even x → c2_even y → c2_to_c (x × y) == c2_to_c x × c2_to_c y.
  Proof.
    intros x y [Hx1 Hx2] [Hy1 Hy2].
    split; simpl.
    - rewrite Hx1, Hx2, Hy1, Hy2.
      repeat rewrite mul_0_l.
      autorewrite with grp.
      reflexivity.
    - rewrite Hx1, Hx2.
      repeat rewrite mul_0_l.
      autorewrite with grp.
      reflexivity.
  Qed.

End EvenIso.


Section Rotor.
  Context {F : Type} `{OF : OrderedField F}.

  Definition c2_reverse (x : C2 F) : C2 F :=
    mkC2 (c2_s x) (c2_1 x) (c2_2 x) (- c2_12 x).

  #[global] Instance c2_reverse_proper : Proper (equ ==> equ) c2_reverse.
  Proof.
    intros x y [Hs [H1 [H2 H12]]].
    repeat split; simpl; [exact Hs | exact H1 | exact H2 |].
    exact (group_neg_proper (c2_12 x) (c2_12 y) H12).
  Qed.

  Lemma cliff2_reorder4 : ∀ a b c d : F,
    a + (b + (c + d)) == b + (a + (d + c)).
  Proof.
    intros a b c d.
    rewrite (add_swap_l a b (c + d)).
    rewrite (comm c d).
    reflexivity.
  Qed.

  Lemma c2_reverse_anti : ∀ x y : C2 F,
    c2_reverse (x × y) == c2_reverse y × c2_reverse x.
  Proof.
    intros x y.
    unfold c2_reverse.
    repeat split; simpl.
    - rewrite (mul_neg_neg (c2_12 y) (c2_12 x)).
      rewrite (field_mul_comm (c2_s y) (c2_s x)),
              (field_mul_comm (c2_1 y) (c2_1 x)),
              (field_mul_comm (c2_2 y) (c2_2 x)),
              (field_mul_comm (c2_12 y) (c2_12 x)).
      reflexivity.
    - rewrite (mul_neg_l (c2_12 y) (c2_2 x)), (mul_neg_r (c2_2 y) (c2_12 x)),
              (neg_involutive (c2_2 y × c2_12 x)).
      rewrite (field_mul_comm (c2_s y) (c2_1 x)),
              (field_mul_comm (c2_1 y) (c2_s x)),
              (field_mul_comm (c2_12 y) (c2_2 x)),
              (field_mul_comm (c2_2 y) (c2_12 x)).
      exact (cliff2_reorder4 (c2_s x × c2_1 y) (c2_1 x × c2_s y)
               (c2_12 x × c2_2 y) (- (c2_2 x × c2_12 y))).
    - rewrite (mul_neg_r (c2_1 y) (c2_12 x)), (mul_neg_l (c2_12 y) (c2_1 x)),
              (neg_involutive (c2_12 y × c2_1 x)).
      rewrite (field_mul_comm (c2_s y) (c2_2 x)),
              (field_mul_comm (c2_2 y) (c2_s x)),
              (field_mul_comm (c2_1 y) (c2_12 x)),
              (field_mul_comm (c2_12 y) (c2_1 x)).
      exact (cliff2_reorder4 (c2_s x × c2_2 y) (c2_2 x × c2_s y)
               (c2_1 x × c2_12 y) (- (c2_12 x × c2_1 y))).
    - rewrite (neg_add (c2_s x × c2_12 y)
                 (c2_12 x × c2_s y + (c2_1 x × c2_2 y + - (c2_2 x × c2_1 y)))).
      rewrite (neg_add (c2_12 x × c2_s y) (c2_1 x × c2_2 y + - (c2_2 x × c2_1 y))).
      rewrite (neg_add (c2_1 x × c2_2 y) (- (c2_2 x × c2_1 y))).
      rewrite (neg_involutive (c2_2 x × c2_1 y)).
      rewrite (mul_neg_r (c2_s y) (c2_12 x)), (mul_neg_l (c2_12 y) (c2_s x)).
      rewrite (field_mul_comm (c2_s y) (c2_12 x)),
              (field_mul_comm (c2_12 y) (c2_s x)),
              (field_mul_comm (c2_1 y) (c2_2 x)),
              (field_mul_comm (c2_2 y) (c2_1 x)).
      exact (cliff2_reorder4 (- (c2_s x × c2_12 y)) (- (c2_12 x × c2_s y))
               (- (c2_1 x × c2_2 y)) (c2_2 x × c2_1 y)).
  Qed.

  Definition rotor (c s : F) : C2 F := mkC2 c 0 0 s.

  Lemma rotor_even : ∀ c s : F, c2_even (rotor c s).
  Proof.
    intros c s.
    split; reflexivity.
  Qed.

  Lemma rotor_reverse : ∀ c s : F,
    c2_reverse (rotor c s) == mkC2 c 0 0 (- s).
  Proof.
    intros c s.
    unfold c2_reverse, rotor.
    repeat split; reflexivity.
  Qed.

  Lemma rotor_reverse_mul : ∀ c s : F,
    rotor c s × c2_reverse (rotor c s) == rotor (c×c + s×s) 0.
  Proof.
    intros c s.
    unfold rotor, c2_reverse.
    repeat split; simpl.
    - rewrite mul_0_l, mul_neg_r.
      autorewrite with grp.
      reflexivity.
    - repeat rewrite mul_0_l.
      repeat rewrite mul_0_r.
      autorewrite with grp.
      reflexivity.
    - repeat rewrite mul_0_l.
      repeat rewrite mul_0_r.
      autorewrite with grp.
      reflexivity.
    - repeat rewrite mul_0_l.
      autorewrite with grp.
      rewrite mul_neg_r, (field_mul_comm s c).
      autorewrite with grp.
      reflexivity.
  Qed.

  Lemma rotor_unit : ∀ c s : F,
    c×c + s×s == 1 → rotor c s × c2_reverse (rotor c s) == 1.
  Proof.
    intros c s Hcs.
    rewrite (rotor_reverse_mul c s).
    unfold rotor.
    repeat split; simpl; [exact Hcs | reflexivity | reflexivity | reflexivity].
  Qed.

  Lemma rotor_unit_necessary : ∀ c s : F,
    rotor c s × c2_reverse (rotor c s) == 1 → c×c + s×s == 1.
  Proof.
    intros c s Hru.
    rewrite (rotor_reverse_mul c s) in Hru.
    unfold rotor in Hru.
    destruct Hru as [Hs _].
    exact Hs.
  Qed.

  Lemma rotor_unit_iff : ∀ c s : F,
    rotor c s × c2_reverse (rotor c s) == 1 ↔ c×c + s×s == 1.
  Proof.
    intros c s.
    split; [exact (rotor_unit_necessary c s) | exact (rotor_unit c s)].
  Qed.


  Definition row2 {A : Type} (a b : A) : Fin.t 2 → A :=
    fun j ⇒ match j with
             | Fin.F1 ⇒ a
             | Fin.FS _ ⇒ b
             end.

  Definition c2_sandwich (R x : C2 F) : C2 F := R × x × c2_reverse R.

  Lemma rotor_mul_vec : ∀ c s v1 v2 : F,
    rotor c s × c2_vec (row2 v1 v2)
    == c2_vec (row2 (c×v1 + s×v2) (c×v2 + - (s×v1))).
  Proof.
    intros c s v1 v2.
    unfold rotor, c2_vec, row2.
    repeat split; simpl.
    - repeat rewrite mul_0_l.
      repeat rewrite mul_0_r.
      autorewrite with grp.
      reflexivity.
    - repeat rewrite mul_0_l.
      autorewrite with grp.
      reflexivity.
    - repeat rewrite mul_0_l.
      autorewrite with grp.
      reflexivity.
    - repeat rewrite mul_0_l.
      repeat rewrite mul_0_r.
      autorewrite with grp.
      reflexivity.
  Qed.

  Lemma vec_mul_rotor : ∀ w1 w2 c s : F,
    c2_vec (row2 w1 w2) × rotor c s
    == c2_vec (row2 (w1×c + - (w2×s)) (w2×c + w1×s)).
  Proof.
    intros w1 w2 c s.
    unfold rotor, c2_vec, row2.
    repeat split; simpl.
    - repeat rewrite mul_0_l.
      repeat rewrite mul_0_r.
      autorewrite with grp.
      reflexivity.
    - repeat rewrite mul_0_l.
      autorewrite with grp.
      reflexivity.
    - repeat rewrite mul_0_l.
      autorewrite with grp.
      reflexivity.
    - repeat rewrite mul_0_l.
      repeat rewrite mul_0_r.
      autorewrite with grp.
      reflexivity.
  Qed.

  Lemma rotor_reverse_eq_rotor : ∀ c s : F,
    c2_reverse (rotor c s) == rotor c (- s).
  Proof.
    intros c s.
    reflexivity.
  Qed.

  Lemma cliff2_sandwich_coeff1 : ∀ c s v1 v2 : F,
    (c×v1 + s×v2)*c + - ((c×v2 + - (s×v1)) × (- s))
    == (c×c + - (s×s)) × v1 + (c×s + c×s) × v2.
  Proof.
    intros c s v1 v2.
    rewrite (distr (c×v2) (- (s×v1)) (- s)).
    rewrite (mul_neg_r (c×v2) s).
    rewrite (mul_neg_neg (s×v1) s).
    rewrite (neg_add (- (c×v2×s)) (s×v1×s)).
    rewrite (neg_involutive (c×v2×s)).
    rewrite (distr (c×v1) (s×v2) c).
    rewrite (mul_assoc_comm_r c v1 c).
    rewrite (mul_assoc_comm_r s v2 c).
    rewrite (field_mul_comm s c).
    rewrite (mul_assoc_comm_r c v2 s).
    rewrite (mul_assoc_comm_r s v1 s).
    rewrite (distr (c×c) (- (s×s)) v1).
    rewrite (mul_neg_l (s×s) v1).
    rewrite (distr (c×s) (c×s) v2).
    exact (add_rotate (c×c×v1) (c×s×v2) (c×s×v2) (- (s×s×v1))).
  Qed.

  Lemma cliff2_sandwich_coeff2 : ∀ c s v1 v2 : F,
    (c×v2 + - (s×v1))*c + (c×v1 + s×v2) × (- s)
    == - ((c×s + c×s) × v1) + (c×c + - (s×s)) × v2.
  Proof.
    intros c s v1 v2.
    rewrite (distr (c×v2) (- (s×v1)) c).
    rewrite (mul_neg_l (s×v1) c).
    rewrite (distr (c×v1) (s×v2) (- s)).
    rewrite (mul_neg_r (c×v1) s), (mul_neg_r (s×v2) s).
    rewrite (mul_assoc_comm_r c v2 c).
    rewrite (mul_assoc_comm_r s v1 c), (field_mul_comm s c).
    rewrite (mul_assoc_comm_r c v1 s).
    rewrite (mul_assoc_comm_r s v2 s).
    rewrite (distr (c×s) (c×s) v1).
    rewrite (neg_add (c×s×v1) (c×s×v1)).
    rewrite (distr (c×c) (- (s×s)) v2).
    rewrite (mul_neg_l (s×s) v2).
    rewrite (add_rotate (c×c×v2) (- (c×s×v1)) (- (c×s×v1)) (- (s×s×v2))).
    exact (comm (c×c×v2 + - (s×s×v2)) (- (c×s×v1) + - (c×s×v1))).
  Qed.

  Lemma c2_sandwich_rotor_vec : ∀ c s v1 v2 : F,
    c2_sandwich (rotor c s) (c2_vec (row2 v1 v2))
    == c2_vec (row2
         ((c×c + - (s×s)) × v1 + (c×s + c×s) × v2)
         (- ((c×s + c×s) × v1) + (c×c + - (s×s)) × v2)).
  Proof.
    intros c s v1 v2.
    unfold c2_sandwich.
    rewrite (rotor_mul_vec c s v1 v2), (rotor_reverse_eq_rotor c s),
            (vec_mul_rotor (c×v1+s×v2) (c×v2 + - (s×v1)) c (- s)).
    repeat split; simpl.
    - reflexivity.
    - exact (cliff2_sandwich_coeff1 c s v1 v2).
    - exact (cliff2_sandwich_coeff2 c s v1 v2).
    - reflexivity.
  Qed.


  Lemma dot_row2 : ∀ a b : F, dot (row2 a b) (row2 a b) == a×a + b×b.
  Proof.
    intros a b.
    exact (dot2_eq (row2 a b)).
  Qed.

  Lemma cliff2_sq_mix : ∀ a b c : F, a×b*(a×c) == a×a*(b×c).
  Proof.
    intros a b c.
    rewrite (assoc a b (a×c)).
    rewrite (mul_comm_swap b a c).
    rewrite <- (assoc a a (b×c)).
    reflexivity.
  Qed.

  Lemma cliff2_cross_mix : ∀ a b c d : F, a×b*(c×d) == a×c*(b×d).
  Proof.
    intros a b c d.
    rewrite (assoc a b (c×d)).
    rewrite (mul_comm_swap b c d).
    rewrite <- (assoc a c (b×d)).
    reflexivity.
  Qed.

  Lemma cliff2_eight_cancel : ∀ a1 a2 b1 b2 p : F,
    a1 + p + (p + b2) + (b1 + - p + (- p + a2)) == a1 + a2 + (b1 + b2).
  Proof.
    intros a1 a2 b1 b2 p.
    repeat rewrite assoc.
    rewrite (add_swap_l p b2 (b1 + (- p + (- p + a2)))).
    rewrite (add_swap_l p b1 (- p + (- p + a2))).
    rewrite <- (assoc p (- p) (- p + a2)).
    rewrite (invr p).
    rewrite (idl (- p + a2)).
    rewrite (add_swap_l p b2 (b1 + (- p + a2))).
    rewrite (add_swap_l p b1 (- p + a2)).
    rewrite <- (assoc p (- p) a2).
    rewrite (invr p).
    rewrite (idl a2).
    rewrite (add_swap_l b2 b1 a2).
    rewrite (comm b2 a2).
    rewrite (add_swap_l b1 a2 b2).
    reflexivity.
  Qed.

  Lemma cliff2_normsq_scale : ∀ A B v1 v2 : F,
    (A×v1 + B×v2)*(A×v1 + B×v2) + (- (B×v1) + A×v2)*(- (B×v1) + A×v2)
    == (A×A + B×B) × (v1×v1 + v2×v2).
  Proof.
    intros A B v1 v2.
    rewrite (distr (A×v1) (B×v2) (A×v1 + B×v2)).
    rewrite (distl (A×v1) (A×v1) (B×v2)).
    rewrite (distl (B×v2) (A×v1) (B×v2)).
    rewrite (distr (- (B×v1)) (A×v2) (- (B×v1) + A×v2)).
    rewrite (distl (- (B×v1)) (- (B×v1)) (A×v2)).
    rewrite (distl (A×v2) (- (B×v1)) (A×v2)).
    rewrite (mul_neg_neg (B×v1) (B×v1)).
    rewrite (mul_neg_l (B×v1) (A×v2)).
    rewrite (mul_neg_r (A×v2) (B×v1)).
    rewrite (cliff2_sq_mix A v1 v1), (cliff2_sq_mix B v2 v2),
            (cliff2_sq_mix B v1 v1), (cliff2_sq_mix A v2 v2).
    rewrite (cliff2_cross_mix A v1 B v2).
    rewrite (cliff2_cross_mix B v2 A v1), (field_mul_comm B A), (field_mul_comm v2 v1).
    rewrite (cliff2_cross_mix B v1 A v2), (field_mul_comm B A), (field_mul_comm v1 v2).
    rewrite (cliff2_cross_mix A v2 B v1), (field_mul_comm v2 v1).
    rewrite (distr (A×A) (B×B) (v1×v1 + v2×v2)).
    rewrite (distl (A×A) (v1×v1) (v2×v2)).
    rewrite (distl (B×B) (v1×v1) (v2×v2)).
    exact (cliff2_eight_cancel (A×A*(v1×v1)) (A×A*(v2×v2)) (B×B*(v1×v1)) (B×B*(v2×v2))
             (A×B*(v1×v2))).
  Qed.

  Lemma cliff2_AB_sq : ∀ c s : F,
    (c×c + - (s×s)) × (c×c + - (s×s)) + (c×s + c×s) × (c×s + c×s)
    == (c×c + s×s) × (c×c + s×s).
  Proof.
    intros c s.
    rewrite (distr (c×c) (- (s×s)) (c×c + - (s×s))).
    rewrite (distl (c×c) (c×c) (- (s×s))).
    rewrite (distl (- (s×s)) (c×c) (- (s×s))).
    rewrite (distr (c×s) (c×s) (c×s + c×s)).
    rewrite (distl (c×s) (c×s) (c×s)).
    rewrite (distr (c×c) (s×s) (c×c + s×s)).
    rewrite (distl (c×c) (c×c) (s×s)).
    rewrite (distl (s×s) (c×c) (s×s)).
    rewrite (mul_neg_r (c×c) (s×s)).
    rewrite (mul_neg_l (s×s) (c×c)).
    rewrite (mul_neg_neg (s×s) (s×s)).
    rewrite (field_mul_comm (s×s) (c×c)).
    rewrite (cliff2_sq_mix c s s).
    set (a1 := c×c*(c×c)).
    set (p := c×c*(s×s)).
    set (b1 := s×s*(s×s)).
    repeat rewrite assoc.
    rewrite (add_swap_l (- p) b1 (p + (p + (p + p)))).
    rewrite <- (assoc (- p) p (p + (p + p))).
    rewrite (invl p).
    rewrite (idl (p + (p + p))).
    rewrite (add_swap_l (- p) b1 (p + (p + p))).
    rewrite <- (assoc (- p) p (p + p)).
    rewrite (invl p).
    rewrite (idl (p + p)).
    rewrite (add_swap_l b1 p p).
    rewrite (comm b1 p).
    reflexivity.
  Qed.

  Lemma c2_sandwich_rotor_preserves_dot : ∀ c s v1 v2 : F,
    c×c + s×s == 1 →
    dot (row2 ((c×c + - (s×s))*v1 + (c×s + c×s)*v2)
              (- ((c×s + c×s)*v1) + (c×c + - (s×s))*v2))
        (row2 ((c×c + - (s×s))*v1 + (c×s + c×s)*v2)
              (- ((c×s + c×s)*v1) + (c×c + - (s×s))*v2))
    == dot (row2 v1 v2) (row2 v1 v2).
  Proof.
    intros c s v1 v2 Hcs.
    rewrite (dot_row2 ((c×c + - (s×s))*v1 + (c×s + c×s)*v2)
               (- ((c×s + c×s)*v1) + (c×c + - (s×s))*v2)).
    rewrite (dot_row2 v1 v2).
    rewrite (cliff2_normsq_scale (c×c + - (s×s)) (c×s + c×s) v1 v2).
    rewrite (cliff2_AB_sq c s).
    rewrite Hcs.
    rewrite (ring_mul_1l (1 : F)).
    exact (ring_mul_1l (v1×v1 + v2×v2)).
  Qed.

  Lemma rotor_mul : ∀ c1 s1 c2 s2 : F,
    rotor c1 s1 × rotor c2 s2 == rotor (c1×c2 + - (s1×s2)) (c1×s2 + s1×c2).
  Proof.
    intros c1 s1 c2 s2.
    unfold rotor.
    repeat split; simpl.
    - rewrite mul_0_l.
      autorewrite with grp.
      reflexivity.
    - repeat rewrite mul_0_l.
      repeat rewrite mul_0_r.
      autorewrite with grp.
      reflexivity.
    - repeat rewrite mul_0_l.
      repeat rewrite mul_0_r.
      autorewrite with grp.
      reflexivity.
    - rewrite mul_0_l.
      autorewrite with grp.
      reflexivity.
  Qed.

  Lemma rotor_mul_to_c : ∀ c1 s1 c2 s2 : F,
    c2_to_c (rotor c1 s1 × rotor c2 s2)
    == c2_to_c (rotor c1 s1) × c2_to_c (rotor c2 s2).
  Proof.
    intros c1 s1 c2 s2.
    exact (c2_to_c_mul_even (rotor c1 s1) (rotor c2 s2)
             (rotor_even c1 s1) (rotor_even c2 s2)).
  Qed.

  Theorem rotor_composition_is_u1 : ∀ c1 s1 c2 s2 : F,
    c2_to_c (rotor c1 s1) × c2_to_c (rotor c2 s2)
    == c2_to_c (rotor (c1×c2 + - (s1×s2)) (c1×s2 + s1×c2)).
  Proof.
    intros c1 s1 c2 s2.
    rewrite <- (rotor_mul_to_c c1 s1 c2 s2).
    rewrite (rotor_mul c1 s1 c2 s2).
    reflexivity.
  Qed.

End Rotor.
