From Coq Require Import Morphisms.
From Riktig Require Import Ops Interface.Ring Interface.Field
  Interface.VectorSpace.


Section FieldAsVectorSpace.
  Context {F : Type} `{Field F}.

  #[global] Instance Field_Scale : HasScale F F := mult.

  Lemma field_sc_proper : Proper (equ ==> equ ==> equ) (@scale F F _).
  Proof.
    intros a a' Ha u u' Hu.
    exact (ring_mul_proper a a' Ha u u' Hu).
  Qed.

  #[global] Instance Field_ScaleLaws : ScaleLaws F F :=
    @Build_ScaleLaws F F _ _ Field_Scale
      field_sc_proper ring_distl ring_distr ring_mul_assoc ring_mul_1l.

  #[global] Instance Field_VectorSpace : VectorSpace F F :=
    @Build_VectorSpace F F _ _ Field_Scale Field_ScaleLaws.

End FieldAsVectorSpace.
