  Lemma sc_antisym : ∀ a b c : nat, sc a b c == - sc b a c.
  Proof.
    intros a b c.
    destruct a as [|[|[|[|[|[|[|[|[|a]]]]]]]]];
    destruct b as [|[|[|[|[|[|[|[|[|b]]]]]]]]];
    destruct c as [|[|[|[|[|[|[|[|[|c]]]]]]]]];
    cbn; autorewrite with grp; reflexivity.
  Qed.

  Definition gnum (i : Fin.t 8) : nat := S (fin_to_nat i).

  Definition f (a b c : Fin.t 8) : F := sc (gnum a) (gnum b) (gnum c).

  Lemma f_antisym : ∀ a b c : Fin.t 8, f a b c == - f b a c.
  Proof.
    intros a b c. unfold f. exact (sc_antisym (gnum a) (gnum b) (gnum c)).
  Qed.


  Definition bracket (x y : Fin.t 8 → F) : Fin.t 8 → F :=
    fun c ⇒ bigopF (fun a ⇒ bigopF (fun b ⇒ x a × y b × f a b c)).


  Lemma distr3_l : ∀ p q r s : F, (p + q) × r × s == p × r × s + q × r × s.
  Proof. intros p q r s. rewrite ring_distr, ring_distr. reflexivity. Qed.

  Lemma distr3_r : ∀ p q r s : F, p × (q + r) × s == p × q × s + p × r × s.
  Proof. intros p q r s. rewrite ring_distl, ring_distr. reflexivity. Qed.

  Lemma scale3_l : ∀ a p q r : F, (a × p) × q × r == a × (p × q × r).
  Proof.
    intros a p q r.
    rewrite (ring_mul_assoc a p q), (ring_mul_assoc a (p × q) r).
    reflexivity.
  Qed.

  Lemma scale3_r : ∀ p a q r : F, p × (a × q) × r == a × (p × q × r).
  Proof.
    intros p a q r.
    rewrite <- (ring_mul_assoc p a q).
    rewrite (field_mul_comm p a).
    rewrite (ring_mul_assoc a p q), (ring_mul_assoc a (p × q) r).
    reflexivity.
  Qed.

  Lemma bracket_add_l : ∀ x1 x2 y : Fin.t 8 → F, ∀ c : Fin.t 8,
    bracket (x1 + x2) y c == bracket x1 y c + bracket x2 y c.
  Proof.
    intros x1 x2 y c. unfold bracket.
    rewrite (bigopF_ext 8 _ _
               (fun a ⇒ bigopF_ext 8 _ _
                           (fun b ⇒ distr3_l (x1 a) (x2 a) (y b) (f a b c)))).
    rewrite (bigopF_ext 8 _ _
               (fun a ⇒ bigopF_add 8 (fun b ⇒ x1 a × y b × f a b c)
                           (fun b ⇒ x2 a × y b × f a b c))).
    exact (bigopF_add 8 (fun a ⇒ bigopF (fun b ⇒ x1 a × y b × f a b c))
             (fun a ⇒ bigopF (fun b ⇒ x2 a × y b × f a b c))).
  Qed.

  Lemma bracket_add_r : ∀ x y1 y2 : Fin.t 8 → F, ∀ c : Fin.t 8,
    bracket x (y1 + y2) c == bracket x y1 c + bracket x y2 c.
  Proof.
    intros x y1 y2 c. unfold bracket.
    rewrite (bigopF_ext 8 _ _
               (fun a ⇒ bigopF_ext 8 _ _
                           (fun b ⇒ distr3_r (x a) (y1 b) (y2 b) (f a b c)))).
    rewrite (bigopF_ext 8 _ _
               (fun a ⇒ bigopF_add 8 (fun b ⇒ x a × y1 b × f a b c)
                           (fun b ⇒ x a × y2 b × f a b c))).
    exact (bigopF_add 8 (fun a ⇒ bigopF (fun b ⇒ x a × y1 b × f a b c))
             (fun a ⇒ bigopF (fun b ⇒ x a × y2 b × f a b c))).
  Qed.

  Lemma bracket_scale_l : ∀ (a : F) (x y : Fin.t 8 → F) (c : Fin.t 8),
    bracket (a *: x) y c == a × bracket x y c.
  Proof.
    intros a x y c. unfold bracket.
    rewrite (bigopF_ext 8 _ _
               (fun p ⇒ bigopF_ext 8 _ _
                           (fun q ⇒ scale3_l a (x p) (y q) (f p q c)))).
    rewrite (bigopF_ext 8 _ _ (fun p ⇒ bigopF_mul_l 8 a (fun q ⇒ x p × y q × f p q c))).
    exact (bigopF_mul_l 8 a (fun p ⇒ bigopF (fun q ⇒ x p × y q × f p q c))).
  Qed.

  Lemma bracket_scale_r : ∀ (a : F) (x y : Fin.t 8 → F) (c : Fin.t 8),
    bracket x (a *: y) c == a × bracket x y c.
  Proof.
    intros a x y c. unfold bracket.
    rewrite (bigopF_ext 8 _ _
               (fun p ⇒ bigopF_ext 8 _ _
                           (fun q ⇒ scale3_r (x p) a (y q) (f p q c)))).
    rewrite (bigopF_ext 8 _ _ (fun p ⇒ bigopF_mul_l 8 a (fun q ⇒ x p × y q × f p q c))).
    exact (bigopF_mul_l 8 a (fun p ⇒ bigopF (fun q ⇒ x p × y q × f p q c))).
  Qed.


  Lemma mul3_comm12 : ∀ p q r : F, p × q × r == q × p × r.
  Proof. intros p q r. rewrite (field_mul_comm p q). reflexivity. Qed.

  Lemma mul_f_antisym_term : ∀ (x y : Fin.t 8 → F) (a b c : Fin.t 8),
    x a × y b × f b a c == - (x a × y b × f a b c).
  Proof.
    intros x y a b c.
    rewrite (f_antisym b a c).
    exact (mul_neg_r (x a × y b) (f a b c)).
  Qed.

  Lemma bracket_antisym : ∀ x y : Fin.t 8 → F, ∀ c : Fin.t 8,
    bracket y x c == - bracket x y c.
  Proof.
    intros x y c. unfold bracket.
    rewrite (bigopF_ext 8 _ _
      (fun p ⇒ bigopF_ext 8 _ _ (fun q ⇒ mul3_comm12 (y p) (x q) (f p q c)))).
    rewrite (bigopF_swap 8 8 (fun p q ⇒ x q × y p × f p q c)).
    rewrite (bigopF_ext 8 _ _
      (fun a ⇒ bigopF_ext 8 _ _ (fun b ⇒ mul_f_antisym_term x y a b c))).
    rewrite (bigopF_ext 8 _ _ (fun a ⇒ bigopF_neg 8 (fun b ⇒ x a × y b × f a b c))).
    exact (bigopF_neg 8 (fun a ⇒ bigopF (fun b ⇒ x a × y b × f a b c))).
  Qed.

  Lemma field_self_neg_zero : ∀ a : F, a == - a → a == 0.
  Proof.
    intros a Ha.
    assert (Hsum : a + a == 0).
    { transitivity (a + - a).
      - apply monoid_add_proper; [reflexivity | exact Ha].
      - exact (group_right_inverse a). }
    assert (Hd : a × two == 0).
    { rewrite (mul_two a). exact Hsum. }
    assert (Hz : a == a × (two × inv two)).
    { rewrite (field_invr two two_neq_0).
      symmetry. exact (ring_mul_1r a). }
    rewrite Hz.
    rewrite <- (ring_mul_assoc a two (inv two)).
    rewrite Hd.
    exact (mul_0_l (inv two)).
  Qed.

  Lemma bracket_self_zero : ∀ x : Fin.t 8 → F, ∀ c : Fin.t 8,
    bracket x x c == 0.
  Proof.
    intros x c.
    apply field_self_neg_zero.
    exact (bracket_antisym x x c).
  Qed.


  Lemma e_mul_delta_term : ∀ (K : Fin.t 8 → F) (j b : Fin.t 8),
    e j b × K b == K b × e b j.
  Proof.
    intros K j b.
    rewrite (field_mul_comm (e j b) (K b)).
    rewrite (e_sym 8 j b).
    reflexivity.
  Qed.

  Lemma e_mul_delta : ∀ (K : Fin.t 8 → F) (j : Fin.t 8),
    bigopF (fun b ⇒ e j b × K b) == K j.
  Proof.
    intros K j.
    rewrite (bigopF_ext 8 _ _ (fun b ⇒ e_mul_delta_term K j b)).
    exact (lincomb_std_delta 8 K j).
  Qed.

  Lemma bracket_inner_collapse : ∀ i j d a : Fin.t 8,
    bigopF (fun b ⇒ e i a × e j b × f a b d) == e i a × f a j d.
  Proof.
    intros i j d a.
    rewrite (bigopF_ext 8 _ _ (fun b ⇒ ring_mul_assoc (e i a) (e j b) (f a b d))).
    rewrite (bigopF_mul_l 8 (e i a) (fun b ⇒ e j b × f a b d)).
    rewrite (e_mul_delta (fun b ⇒ f a b d) j).
    reflexivity.
  Qed.

  Lemma bracket_e_e : ∀ i j d : Fin.t 8, bracket (e i) (e j) d == f i j d.
  Proof.
    intros i j d. unfold bracket.
    rewrite (bigopF_ext 8 _ _ (fun a ⇒ bracket_inner_collapse i j d a)).
    exact (e_mul_delta (fun a ⇒ f a j d) i).
  Qed.

  Lemma bracket_e_l_inner : ∀ (i : Fin.t 8) (y : Fin.t 8 → F) (c a : Fin.t 8),
    bigopF (fun b ⇒ e i a × y b × f a b c) == e i a × bigopF (fun b ⇒ y b × f a b c).
  Proof.
    intros i y c a.
    rewrite (bigopF_ext 8 _ _ (fun b ⇒ ring_mul_assoc (e i a) (y b) (f a b c))).
    exact (bigopF_mul_l 8 (e i a) (fun b ⇒ y b × f a b c)).
  Qed.

  Lemma bracket_e_l : ∀ (i : Fin.t 8) (y : Fin.t 8 → F) (c : Fin.t 8),
    bracket (e i) y c == bigopF (fun b ⇒ y b × f i b c).
  Proof.
    intros i y c. unfold bracket.
    rewrite (bigopF_ext 8 _ _ (fun a ⇒ bracket_e_l_inner i y c a)).
    exact (e_mul_delta (fun a ⇒ bigopF (fun b ⇒ y b × f a b c)) i).
  Qed.


  Definition jacobiator (x y z : Fin.t 8 → F) : Fin.t 8 → F :=
    bracket x (bracket y z) + bracket y (bracket z x) + bracket z (bracket x y).

  Lemma jacobiator_def : ∀ (x y z : Fin.t 8 → F) (d : Fin.t 8),
    jacobiator x y z d
      == bracket x (bracket y z) d + bracket y (bracket z x) d + bracket z (bracket x y) d.
  Proof. intros x y z d. reflexivity. Qed.

  Lemma jacobi_term : ∀ i j k d : Fin.t 8,
    bracket (e i) (bracket (e j) (e k)) d == bigopF (fun b ⇒ f j k b × f i b d).
  Proof.
    intros i j k d.
    rewrite (bracket_e_l i (bracket (e j) (e k)) d).
    exact (bigopF_ext 8 _ _ (fun b ⇒ ring_mul_proper (bracket (e j) (e k) b) (f j k b)
             (bracket_e_e j k b) (f i b d) (f i b d) (reflexivity (f i b d)))).
  Qed.

  Lemma jacobi_basis_expand : ∀ i j k d : Fin.t 8,
    jacobiator (e i) (e j) (e k) d ==
      bigopF (fun b ⇒ f j k b × f i b d) + bigopF (fun b ⇒ f k i b × f j b d)
      + bigopF (fun b ⇒ f i j b × f k b d).
  Proof.
    intros i j k d.
    rewrite (jacobiator_def (e i) (e j) (e k) d).
    rewrite (jacobi_term i j k d), (jacobi_term j k i d), (jacobi_term k i j d).
    reflexivity.
  Qed.

  Lemma half_mul_half : ∀ a b : F, half a × half b == (a × b) × (inv two × inv two).
  Proof.
    intros a b. unfold half.
    rewrite (ring_mul_assoc a (inv two) (b × inv two)).
    rewrite (mul_comm_swap (inv two) b (inv two)).
    rewrite <- (ring_mul_assoc a b (inv two × inv two)).
    reflexivity.
  Qed.

  Lemma half_sqrt3_sq : half sqrt3 × half sqrt3
    == half 1 × half 1 + half 1 × half 1 + half 1 × half 1.
  Proof.
    rewrite (half_mul_half sqrt3 sqrt3).
    rewrite sqrt3_sq.
    rewrite (half_mul_half 1 1).
    rewrite (ring_mul_1l 1).
    rewrite (ring_distr (1 + 1) 1 (inv two × inv two)).
    rewrite (ring_distr 1 1 (inv two × inv two)).
    reflexivity.
  Qed.

  Lemma A_plus_neg3A : ∀ A : F, A + - (A + A + A) == - (A + A).
  Proof.
    intros A.
    rewrite (neg_add (A + A) A).
    rewrite (ab_comm (- (A + A)) (- A)).
    rewrite <- (monoid_associative A (- A) (- (A + A))).
    rewrite (group_right_inverse A).
    rewrite (monoid_left_identity (- (A + A))).
    reflexivity.
  Qed.

  Lemma negA_plus_3A : ∀ A : F, - A + (A + A + A) == A + A.
  Proof.
    intros A.
    rewrite <- (monoid_associative (- A) (A + A) A).
    rewrite <- (monoid_associative (- A) A A).
    rewrite (group_left_inverse A).
    rewrite (monoid_left_identity A).
    reflexivity.
  Qed.

  Lemma negAA_plus_A_plus_A : ∀ A : F, - (A + A) + A + A == 0.
  Proof.
    intros A.
    rewrite (neg_add A A).
    rewrite (monoid_associative (- A) (- A) A).
    rewrite (group_left_inverse A).
    rewrite (monoid_right_identity (- A)).
    exact (group_left_inverse A).
  Qed.

  Lemma AA_plus_negA_plus_negA : ∀ A : F, (A + A) + - A + - A == 0.
  Proof.
    intros A.
    rewrite (monoid_associative A A (- A)).
    rewrite (group_right_inverse A).
    rewrite (monoid_right_identity A).
    exact (group_right_inverse A).
  Qed.

  Lemma jacobi_quarter_456 : half 1 + (half 1 × half 1 + - (half sqrt3 × half sqrt3)) == 0.
  Proof.
    rewrite half_sqrt3_sq.
    rewrite (A_plus_neg3A (half 1 × half 1)).
    rewrite (half_mul_sum 1 (half 1)).
    rewrite (ring_mul_1l (half 1)).
    exact (group_right_inverse (half 1)).
  Qed.

  Lemma jacobi_quarter_457 : - half 1 + (- (half 1 × half 1) + half sqrt3 × half sqrt3) == 0.
  Proof.
    rewrite half_sqrt3_sq.
    rewrite (negA_plus_3A (half 1 × half 1)).
    rewrite (half_mul_sum 1 (half 1)).
    rewrite (ring_mul_1l (half 1)).
    exact (group_left_inverse (half 1)).
  Qed.

  Lemma jacobi_quarter_467 :
    half 1 × half 1 + - (half sqrt3 × half sqrt3) + half 1 × half 1 + half 1 × half 1 == 0.

    rewrite (A_plus_neg3A (half 1 × half 1)).

  Lemma jacobi_quarter_567 :

    rewrite half_sqrt3_sq.
    rewrite (negA_plus_3A (half 1 × half 1)).
    exact (AA_plus_negA_plus_negA (half 1 × half 1)).
  Qed.

  Lemma fin8_rect : ∀ P : Fin.t 8 → Prop,
    P g1 → P g2 → P g3 → P g4 → P g5 → P g6 → P g7 → P g8 →
    ∀ i : Fin.t 8, P i.
  Proof.
    intros P H1 H2 H3 H4 H5 H6 H7 H8 i.
    apply (Fin.caseS' i); [exact H1 |]; intros i1.
    apply (Fin.caseS' i1); [exact H2 |]; intros i2.
    apply (Fin.caseS' i2); [exact H3 |]; intros i3.
    apply (Fin.caseS' i3); [exact H4 |]; intros i4.
    apply (Fin.caseS' i4); [exact H5 |]; intros i5.
    apply (Fin.caseS' i5); [exact H6 |]; intros i6.
    apply (Fin.caseS' i6); [exact H7 |]; intros i7.
    apply (Fin.caseS' i7); [exact H8 |]; intros i8.
    exact (Fin.case0
      (fun i8' ⇒ P (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i8')))))))))
      i8).
  Qed.

  Ltac su3_step :=
    (autorewrite with su3 su3fin grp;
     first
       [ (first [reflexivity | assumption])
       | (rewrite monoid_associative; autorewrite with su3 su3fin grp;
          (first [reflexivity | assumption]))
       | (rewrite (field_mul_comm (half sqrt3) (half 1));
          autorewrite with su3 su3fin grp; (first [reflexivity | assumption]))
       | (rewrite monoid_associative; rewrite (field_mul_comm (half sqrt3) (half 1));
          autorewrite with su3 su3fin grp; (first [reflexivity | assumption]))
       | exact jacobi_quarter_456
       | exact jacobi_quarter_457
       | exact jacobi_quarter_467
       | exact jacobi_quarter_567
       | idtac ]).

  Ltac su3_jacobi i j k :=
    (intros d; revert d;
     apply fin8_rect;
       (rewrite (jacobi_basis_expand i j k);
        unfold bigopF, f, gnum, fin_to_nat, sc, PW_Zero; cbn;
        su3_step)).
