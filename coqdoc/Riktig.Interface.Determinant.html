Library Riktig.Interface.Determinant

From Coq Require Import Morphisms.
From Coq.Vectors Require Fin.
From Riktig Require Export Ops Interface.Field Interface.VectorSpace
  Interface.Bilinear Interface.Basis.


Record Determinant (n : nat) (F V : Type) {FF : Field F}
    {VV : VectorSpace F V} (B : Basis n F V) := {
  det_mf :> MultilinearForm n F V;
  det_alt : Alternating det_mf;
  det_basis_1 : det_mf (basis_vec B) == 1
}.

Arguments det_mf {n F V FF VV B}.
Arguments det_alt {n F V FF VV B}.
Arguments det_basis_1 {n F V FF VV B}.
