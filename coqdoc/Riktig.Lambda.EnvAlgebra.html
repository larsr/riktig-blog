Fixpoint fin_add (k : nat) {n : nat} (x : Fin.t n) : Fin.t (k + n) :=
  match k with
  | O ⇒ x
  | S k' ⇒ Fin.FS (fin_add k' x)
  end.

Lemma fin_add_O : ∀ (n : nat) (x : Fin.t n), fin_add 0 x = x.
Proof. intros n x. reflexivity. Qed.

Lemma fin_add_S : ∀ (k n : nat) (x : Fin.t n),
  fin_add (S k) x = Fin.FS (fin_add k x).
Proof. intros k n x. reflexivity. Qed.


Inductive DEnv : nat → nat → Type :=
  | DShift : ∀ (k n : nat), DEnv n (k + n)
  | DCons : ∀ (m n : nat), Exp n → DEnv m n → DEnv (S m) n
  | DComp : ∀ (m n p : nat), DEnv m n → DEnv n p → DEnv m p.

Arguments DShift k n : clear implicits.
Arguments DCons {m n} t r.
Arguments DComp {m n p} r s.


Fixpoint denote {m n : nat} (e : DEnv m n) {struct e} : Sub m n :=
  match e in DEnv m0 n0 return Sub m0 n0 with
  | DShift k n0 ⇒ fun x ⇒ EVar (fin_add k x)
  | DCons t r ⇒ cons_sub t (denote r)
  | DComp r s ⇒ comp_sub (denote r) (denote s)
  end.

Lemma denote_DShift : ∀ (k n : nat) (x : Fin.t n),
  denote (DShift k n) x = EVar (fin_add k x).
Proof. intros k n x. reflexivity. Qed.

Lemma denote_DCons : ∀ (m n : nat) (t : Exp n) (r : DEnv m n),
  sub_eq (denote (DCons t r)) (cons_sub t (denote r)).
Proof. intros m n t r x. reflexivity. Qed.

Lemma denote_DComp : ∀ (m n p : nat) (r : DEnv m n) (s : DEnv n p),
  sub_eq (denote (DComp r s)) (comp_sub (denote r) (denote s)).
Proof. intros m n p r s x. reflexivity. Qed.


Fixpoint dapply {m n : nat} (e : DEnv m n) {struct e} : Fin.t m → Exp n :=
  match e in DEnv m0 n0 return Fin.t m0 → Exp n0 with
  | DShift k n0 ⇒ fun x ⇒ EVar (fin_add k x)
  | DCons t r ⇒
      fun x ⇒
        Fin.caseS' x (fun _ ⇒ Exp _)
          t
          (fun y ⇒ dapply r y)
  | DComp r s ⇒ fun x ⇒ subst (denote s) (dapply r x)
  end.

Lemma dapply_denote : ∀ (m n : nat) (r : DEnv m n) (x : Fin.t m),
  dapply r x = denote r x.
Proof.
  intros m n r; induction r as [k n0 | m0 n0 t r IH | m0 n0 p0 r IHr s IHs];
    intros x.
  - reflexivity.
  - apply (Fin.caseS' x
      (fun x0 ⇒ dapply (DCons t r) x0 = denote (DCons t r) x0)).
    + reflexivity.
    + intros y. exact (IH y).
  - simpl. rewrite (IHr x). reflexivity.
Qed.


Definition denv_cast {m n n' : nat} (H : n = n') (r : DEnv m n) : DEnv m n' :=
  eq_rect n (DEnv m) r n' H.

Lemma denv_cast_refl : ∀ (m n : nat) (r : DEnv m n),
  denv_cast eq_refl r = r.
Proof. intros m n r. reflexivity. Qed.

Lemma denote_denv_cast : ∀ (m n n' : nat) (H : n = n') (r : DEnv m n)
    (x : Fin.t m),
  denote (denv_cast H r) x = eq_rect n Exp (denote r x) n' H.
Proof.
  intros m n n' H r x. destruct H. reflexivity.
Qed.

Lemma eq_rect_EVar : ∀ (n n' : nat) (H : n = n') (y : Fin.t n),
  eq_rect n Exp (EVar y) n' H = EVar (eq_rect n Fin.t y n' H).
Proof.
  intros n n' H y. destruct H. reflexivity.
Qed.

Lemma shift_fuse_eq : ∀ (j k n0 : nat), j + k + n0 = j + (k + n0).
Proof. intros j k n0. lia. Qed.

Lemma fin_to_nat_eq_rect : ∀ (n n' : nat) (H : n = n') (x : Fin.t n),
  fin_to_nat (eq_rect n Fin.t x n' H) = fin_to_nat x.
Proof.
  intros n n' H x. destruct H. reflexivity.
Qed.

Lemma fin_to_nat_fin_add : ∀ (k n0 : nat) (x : Fin.t n0),
  fin_to_nat (fin_add k x) = k + fin_to_nat x.
Proof.
  intros k n0 x; induction k as [| k' IH].
  - reflexivity.
  - simpl. rewrite IH. reflexivity.
Qed.

Lemma fin_add_assoc : ∀ (j k n0 : nat) (x : Fin.t n0)
    (H : j + k + n0 = j + (k + n0)),
  eq_rect (j + k + n0) Fin.t (fin_add (j + k) x) (j + (k + n0)) H
    = fin_add j (fin_add k x).
Proof.
  intros j k n0 x H.
  apply fin_to_nat_inj.
  rewrite (fin_to_nat_eq_rect (j + k + n0) (j + (k + n0)) H (fin_add (j + k) x)).
  rewrite (fin_to_nat_fin_add (j + k) n0 x).
  rewrite (fin_to_nat_fin_add j (k + n0) (fin_add k x)).
  rewrite (fin_to_nat_fin_add k n0 x).
  lia.
Qed.


Fixpoint shift_comp_pos {n p : nat} (k' : nat) (s : DEnv (S k' + n) p)
    {struct s} : DEnv n p :=
  match s in DEnv m0 p0 return (m0 = S k' + n) → DEnv n p0 with
  | DShift j n1 ⇒ fun Heq ⇒
      denv_cast
        (eq_trans (shift_fuse_eq j (S k') n)
                  (f_equal (fun z ⇒ j + z) (eq_sym Heq)))
        (DShift (j + S k') n)
  | @DCons m1 n1 t tail ⇒ fun Heq ⇒
      match k' as k0 return (S m1 = S k0 + n) → DEnv n n1 with
      | O ⇒ fun Heq0 ⇒
          eq_rect m1 (fun x ⇒ DEnv x n1) tail n (f_equal Nat.pred Heq0)
      | S k'' ⇒ fun Heq0 ⇒
          shift_comp_pos k''
            (eq_rect m1 (fun x ⇒ DEnv x n1) tail (S k'' + n)
               (f_equal Nat.pred Heq0))
      end Heq
  | @DComp m1 n1 p1 r1 s1 ⇒ fun Heq ⇒
      DComp (shift_comp_pos k'
               (eq_rect m1 (fun x ⇒ DEnv x n1) r1 (S k' + n) Heq)) s1
  end eq_refl.


Definition shift_comp {n p : nat} (k : nat) (s : DEnv (k + n) p) : DEnv n p :=
  match k as k0 return DEnv (k0 + n) p → DEnv n p with
  | O ⇒ fun s0 ⇒ s0
  | S k' ⇒ fun s0 ⇒ shift_comp_pos k' s0
  end s.


Fixpoint dcomp {m n p : nat} (r : DEnv m n) (s : DEnv n p) {struct r}
    : DEnv m p :=
  match r in DEnv m0 n0 return DEnv n0 p → DEnv m0 p with
  | DShift k n0 ⇒ fun s0 ⇒ shift_comp k s0
  | @DCons m0 n0 t r0 ⇒ fun s0 ⇒ DCons (subst (denote s0) t) (dcomp r0 s0)
  | @DComp m0 n0 p0 r1 r2 ⇒ fun s0 ⇒ dcomp r1 (dcomp r2 s0)
  end s.


Lemma eq_rect_dom_DCons : ∀ (m1 n2 : nat) (t : Exp n2) (tail : DEnv m1 n2)
    (m2 : nat) (H : S m1 = S m2),
  eq_rect (S m1) (fun x ⇒ DEnv x n2) (DCons t tail) (S m2) H
    = DCons t (eq_rect m1 (fun x ⇒ DEnv x n2) tail m2 (f_equal Nat.pred H)).
Proof.
  intros m1 n2 t tail m2 H.
  assert (Hm : m1 = m2) by exact (f_equal Nat.pred H).
  destruct Hm.
  rewrite (Eqdep_dec.UIP_dec Nat.eq_dec H eq_refl).
  reflexivity.
Qed.


Lemma shift_comp_pos_sound_gen : ∀ (n1 p : nat) (s : DEnv n1 p)
    (k' n : nat) (Heq : n1 = S k' + n),
  sub_eq (denote (shift_comp_pos k' (eq_rect n1 (fun x ⇒ DEnv x p) s (S k' + n) Heq)))
         (comp_sub (denote (DShift (S k') n))
                    (denote (eq_rect n1 (fun x ⇒ DEnv x p) s (S k' + n) Heq))).
Proof.
  intros n1 p s.
  induction s as [j n2 | m1 n2 t tail IH | m1 n2 p1 r1 IHr s1 IHs];
    intros k' n Heq.
  -
    subst n2. simpl. intros x.
    rewrite denote_denv_cast, denote_DShift, eq_rect_EVar, fin_add_assoc.
    unfold comp_sub. rewrite subst_EVar.
    reflexivity.
  -
    change (S k' + n) with (S (k' + n)) in Heq |- ×.
    rewrite (eq_rect_dom_DCons m1 n2 t tail (k' + n) Heq).
    destruct k' as [| k'']; simpl.
    + intros x. symmetry.
      exact (cons_shift n n2 t
        (denote (eq_rect m1 (fun x0 ⇒ DEnv x0 n2) tail n (f_equal Nat.pred Heq))) x).
    + intros x.
      transitivity
        (comp_sub (denote (DShift (S k'') n))
           (denote (eq_rect m1 (fun x0 ⇒ DEnv x0 n2) tail (S (k'' + n))
                      (f_equal Nat.pred Heq))) x).
      × exact (IH k'' n (f_equal Nat.pred Heq) x).
      × unfold comp_sub.
        rewrite denote_DShift, subst_EVar, subst_EVar, fin_add_S.
        rewrite cons_sub_FS.
        reflexivity.
  -
    subst m1. simpl.
    rewrite (IHr k' n eq_refl).
    apply comp_assoc.
Qed.

Lemma shift_comp_pos_sound : ∀ (k' n p : nat) (s : DEnv (S k' + n) p),
  sub_eq (denote (shift_comp_pos k' s))
         (comp_sub (denote (DShift (S k') n)) (denote s)).
Proof.
  intros k' n p s.
  exact (shift_comp_pos_sound_gen (S k' + n) p s k' n eq_refl).
Qed.

Lemma shift_comp_sound : ∀ (k n p : nat) (s : DEnv (k + n) p),
  sub_eq (denote (shift_comp k s)) (comp_sub (denote (DShift k n)) (denote s)).
Proof.
  intros k n p s.
  destruct k as [| k'].
  - intros x. reflexivity.
  - exact (shift_comp_pos_sound k' n p s).
Qed.


Theorem dcomp_sound : ∀ (m n p : nat) (r : DEnv m n) (s : DEnv n p),
  sub_eq (denote (dcomp r s)) (comp_sub (denote r) (denote s)).
Proof.
  intros m n p r; induction r as [k n0 | m0 n0 t r0 IHr | m0 n0 p0 r1 IHr1 r2 IHr2];
    intros s.
  - exact (shift_comp_sound k n0 p s).
  - simpl. intros x.
    apply (Fin.caseS' x (fun x0 ⇒
      cons_sub (subst (denote s) t) (denote (dcomp r0 s)) x0
      = comp_sub (cons_sub t (denote r0)) (denote s) x0)).
    + reflexivity.
    + intros y. simpl. exact (IHr s y).
  - simpl. intros x.
    rewrite (IHr1 (dcomp r2 s) x).
    unfold comp_sub at 1.
    rewrite (IHr2 s).
    unfold comp_sub.
    rewrite (subst_subst n0 p0 p (denote r2) (denote s) (denote r1 x)).
    reflexivity.
Qed.


Theorem dfuse : ∀ (m n p : nat) (r : DEnv m n) (s : DEnv n p) (e : Exp m),
  subst (denote s) (subst (denote r) e) = subst (denote (dcomp r s)) e.
Proof.
  intros m n p r s e.
  rewrite (subst_subst m n p (denote r) (denote s) e).
  exact (subst_ext m p (comp_sub (denote r) (denote s)) (denote (dcomp r s))
           (fun x ⇒ eq_sym (dcomp_sound m n p r s x)) e).
Qed.


Definition dup {m n : nat} (r : DEnv m n) : DEnv (S m) (S n) :=
  DCons (EVar Fin.F1) (dcomp r (DShift 1 n)).

Theorem dup_sound : ∀ (m n : nat) (r : DEnv m n),
  sub_eq (denote (dup r)) (up (denote r)).
Proof.
  intros m n r.
  unfold dup.
  transitivity (cons_sub (EVar Fin.F1) (comp_sub (denote r) shift)).
  - intros x.
    rewrite (denote_DCons m (S n) (EVar Fin.F1) (dcomp r (DShift 1 n)) x).
    apply (Fin.caseS' x (fun x0 ⇒
      cons_sub (EVar Fin.F1) (denote (dcomp r (DShift 1 n))) x0
      = cons_sub (EVar Fin.F1) (comp_sub (denote r) shift) x0)).
    + reflexivity.
    + intros y. simpl.
      rewrite (dcomp_sound m n (S n) r (DShift 1 n) y).
      unfold comp_sub.
      exact (subst_ext n (S n) (denote (DShift 1 n)) shift
               (fun z ⇒ denote_DShift 1 n z) (denote r y)).
  - intros x. symmetry. exact (up_as_cons m n (denote r) x).
Qed.

  dcomp ex_env1 (dcomp ex_env2 ex_env3) = DShift 0 1.
