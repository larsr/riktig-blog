From Coq Require Import List ZArith.
Import ListNotations.
From Riktig Require Import Programs.Memprog.Syntax.
From Riktig Require Import Programs.Memprog.Machine.
From Riktig Require Import Programs.Memprog.Expr.
From Riktig Require Import Programs.Memprog.MC.

Local Open Scope Z_scope.


Definition sum_l1 (base n : Z) : MemT Z Z Z :=
  do! acc <- stack_alloc_m;
  do! i <- stack_alloc_m;
  write_m acc 0 >>
  write_m i 0 >>
  while_m (lt_ Z.ltb (read_m i) (ret_t n)) (
    do! iv <- read_m i;
    do! v <- read_arr base (ret_t iv);
    do! s <- read_m acc;
    write_m acc (s + v) >>
    write_m i (iv + 1)) >>
  read_m acc.



Definition sum_l1_demo : MemT Z Z Z :=
  do! base <- list_to_mem Z.of_nat [1; 2; 3; 4; 5];
  sum_l1 base 5.

Example sum_l1_result :
  fst (run_z 3000 sum_l1_demo 64) = Some 15.
Proof. vm_compute. reflexivity. Qed.
