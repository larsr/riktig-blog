Section Generic.
  Context {G : Type} `{AbGroup G}.

  Theorem swap_cancel : ∀ a b c : G, (a + b) + (c + - a) == b + c.
  Proof.
    intros a b c.
    rewrite (comm a b), assoc, (add_swap_l a c (- a)), invr, idr.
    reflexivity.
  Qed.
End Generic.

Example z_swap_cancel : (3 + 5 + (7 + - 3))%Z = (5 + 7)%Z.
Proof. exact (swap_cancel (G := Z) 3%Z 5%Z 7%Z). Qed.


Section Generic.
  Context {F U V : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}.

  Theorem lm_combo : ∀ (f : LinearMap F U V) (a b : F) (u v : U),
    f (a *: u + b *: v) == a *: f u + b *: f v.
  Proof.
    intros f a b u v.
    rewrite (lm_add f), (lm_scale f), (lm_scale f).
    reflexivity.
  Qed.
End Generic.

Section ScaleMap.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V} (c : F).

  Lemma scale_map_proper : Proper (equ ==> equ) (fun v : V ⇒ c *: v).
  Proof.
    intros u v Huv.
    rewrite Huv.
    reflexivity.
  Qed.

  Lemma scale_map_add : ∀ u v : V, c *: (u + v) == c *: u + c *: v.
  Proof.
    intros u v.
    exact (sc_scale_add c u v).
  Qed.

  Lemma scale_map_scale : ∀ (a : F) (u : V),
    c *: (a *: u) == a *: (c *: u).
  Proof.
    intros a u.
    rewrite <- (sc_scale_scale c a u), (field_mul_comm c a),
      (sc_scale_scale a c u).
    reflexivity.
  Qed.

  Definition scale_map : LinearMap F V V :=
    @Build_LinearMap F V V FF VV VV (fun v : V ⇒ c *: v)
      scale_map_proper scale_map_add scale_map_scale.
End ScaleMap.

Definition triple : LinearMap Q (Fin.t 2 → Q) (Fin.t 2 → Q) :=
  scale_map (3#1)%Q.

Example fin2_combo : ∀ u v : Fin.t 2 → Q,
  triple (2 *: u + 5 *: v) == 2 *: triple u + 5 *: triple v.
Proof. intros u v. exact (lm_combo triple (2#1)%Q (5#1)%Q u v). Qed.


Section Generic.
  Context {O : Type} `{Category O}.

  Theorem iso_absorb : ∀ (x y : O) (f : Iso x y),
    (iso_from f ∘ iso_to f) ∘ iso_from f == iso_from f.
  Proof.
    intros x y f.
    rewrite (iso_fromto f), cat_idl.
    reflexivity.
  Qed.
End Generic.

Definition negb_negb_iso : @Iso Type CoqCat bool bool := {|
  iso_to := negb; iso_from := negb;
  iso_fromto := negb_involutive; iso_tofrom := negb_involutive
|}.

Example negb_absorb : ∀ b : bool, negb (negb (negb b)) = negb b.
Proof. exact (iso_absorb _ _ negb_negb_iso). Qed.


Example ifb_true_idt_k : IFB TRUE idt k == idt.
Proof.
  apply IF_TRUE.
  - apply value_abs.
  - apply value_abs.
Qed.

Example ifb_true_idt_k_by_eval : IFB TRUE idt k == idt.
Proof. apply (conv_by_eval 20 5). reflexivity. Qed.
