Library Riktig.Programs.Memprog.Expr


From Coq Require Import List ZArith.
Import ListNotations.
From Riktig Require Import Programs.Memprog.Syntax.
From Riktig Require Import Programs.Memprog.Machine.


Inductive Expr (val : Type) : Type :=
  | Lit : val → Expr val
  | EAdd : Expr val → Expr val → Expr val
  | ESub : Expr val → Expr val → Expr val.

Arguments Lit {val}.
Arguments EAdd {val}.
Arguments ESub {val}.


Section WithOps.
  Context {addr val : Type}.
  Context (val_add val_sub : val → val → val).
  Context (val_ltb val_leb val_gtb : val → val → bool).
  Context (val_of_nat : nat → val).


  Definition plus_ (x y : MemT addr val val) : MemT addr val val :=
    do! a <- x; do! b <- y; ret_t (val_add a b).

  Definition minus_ (x y : MemT addr val val) : MemT addr val val :=
    do! a <- x; do! b <- y; ret_t (val_sub a b).


  Definition lt_ (x y : MemT addr val val) : MemT addr val bool :=
    do! a <- x; do! b <- y; ret_t (val_ltb a b).

  Definition leq_ (x y : MemT addr val val) : MemT addr val bool :=
    do! a <- x; do! b <- y; ret_t (val_leb a b).

  Definition gt_ (x y : MemT addr val val) : MemT addr val bool :=
    do! a <- x; do! b <- y; ret_t (val_gtb a b).


  Definition if_m {a : Type} (cond : MemT addr val bool)
    (thn els : MemT addr val a) : MemT addr val a :=
    do! b <- cond;
    if b then thn else els.

  Definition when_m (cond : MemT addr val bool) (body : MemT addr val unit)
    : MemT addr val unit :=
    if_m cond body (ret_t tt).


  Definition at_ (base : addr) (idx : MemT addr val val) : MemT addr val addr :=
    do! i <- idx; offset_m base i.

  Definition read_arr (base : addr) (idx : MemT addr val val) : MemT addr val val :=
    do! a <- at_ base idx; read_m a.

  Definition swap_cells (base : addr) (mi mj : MemT addr val val)
    : MemT addr val unit :=
    do! ai <- at_ base mi;
    do! aj <- at_ base mj;
    do! vi <- read_m ai;
    do! vj <- read_m aj;
    write_m ai vj >>
    write_m aj vi.


  Definition list_to_mem (xs : list val) : MemT addr val addr :=
    match xs with
    | [] ⇒ fail_t
    | _ :: _ ⇒
        do! base <- alloc_m (length xs);
        (fix fill (i : nat) (l : list val) : MemT addr val unit :=
           match l with
           | [] ⇒ ret_t tt
           | v :: rest ⇒
               do! a <- offset_m base (val_of_nat i);
               write_m a v >>
               fill (S i) rest
           end) O xs >>
        ret_t base
    end.


  Fixpoint eval_expr (e : Expr val) : val :=
    match e with
    | Lit v ⇒ v
    | EAdd x y ⇒ val_add (eval_expr x) (eval_expr y)
    | ESub x y ⇒ val_sub (eval_expr x) (eval_expr y)
    end.

  Fixpoint compile_expr (e : Expr val) : MemT addr val val :=
    match e with
    | Lit v ⇒ ret_t v
    | EAdd x y ⇒ plus_ (compile_expr x) (compile_expr y)
    | ESub x y ⇒ minus_ (compile_expr x) (compile_expr y)
    end.

End WithOps.


Local Open Scope Z_scope.

Definition demo_expr : Expr Z := ESub (EAdd (Lit 1) (Lit 2)) (ESub (Lit 3) (Lit 4)).


Example demo_expr_eval : eval_expr Z.add Z.sub demo_expr = 4.
Proof. vm_compute. reflexivity. Qed.

Example demo_expr_compiled :
  fst (run_z 100 (compile_expr Z.add Z.sub demo_expr) 8) = Some 4.
Proof. vm_compute. reflexivity. Qed.


Definition swap_demo : MemT Z Z Z :=
  do! base <- list_to_mem Z.of_nat [10; 20; 30];
  swap_cells base (ret_t 0) (ret_t 2) >>
  ret_t base.

Example swap_demo_cells :
  (let '(r, m) := run_z 1000 swap_demo 32 in mem_to_list m r 3)
  = Some [30; 20; 10].
Proof. vm_compute. reflexivity. Qed.
