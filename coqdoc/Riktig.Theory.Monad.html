Library Riktig.Theory.Monad

From Coq Require Import Morphisms.
From Riktig Require Import Ops Interface.Monad.

Section MonadTheory.
  Context {M : Type → Type} `{Monad M}.

  Definition map {A B} (f : A → B) (m : M A) : M B := bind m (fun x ⇒ ret (f x)).

  Lemma map_id : ∀ A (m : M A), map (fun x ⇒ x) m == m.
  Proof. intros A m. unfold map. apply bind_ret_r. Qed.

  Lemma map_compose : ∀ A B C (f : A → B) (g : B → C) (m : M A),
    map (fun x ⇒ g (f x)) m == map g (map f m).
  Proof.
    intros A B C f g m.
    unfold map.
    transitivity (bind m (fun x ⇒ bind (ret (f x)) (fun y ⇒ ret (g y)))).
    - apply bind_proper.
      + reflexivity.
      + intros x. symmetry. exact (bind_ret_l B C (f x) (fun y ⇒ ret (g y))).
    - symmetry. apply bind_assoc.
  Qed.

End MonadTheory.
