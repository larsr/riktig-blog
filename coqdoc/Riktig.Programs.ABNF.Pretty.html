Open Scope string_scope.


Definition lc (p : Parser string ascii) : Parser string ParseTree :=
  c <- p ;; ret (Leaf (String c EmptyString)).

Definition lcs (input : string) (p : Parser string ascii) : Parser string ParseTree :=
  cs <- many1 p input ;; ret (Leaf (list_to_string cs)).


Definition pt_ALPHA : Parser string ParseTree := lc p_alpha.
Definition pt_BIT : Parser string ParseTree := lc p_bit.
Definition pt_DIGIT : Parser string ParseTree := lc p_digit.
Definition pt_HEXDIG : Parser string ParseTree := lc p_hexdig.
Definition pt_DQUOTE : Parser string ParseTree := lc (char dquote).
Definition pt_SP : Parser string ParseTree := lc (satisfy stringSeq (fun c ⇒ Nat.eqb (Ascii.nat_of_ascii c) 32)).
Definition pt_HTAB : Parser string ParseTree := lc (satisfy stringSeq (fun c ⇒ Nat.eqb (Ascii.nat_of_ascii c) 9)).
Definition pt_WSP : Parser string ParseTree := lc p_wsp.
Definition pt_VCHAR : Parser string ParseTree := lc p_vchar.
Definition pt_CR : Parser string ParseTree := lc (satisfy stringSeq is_cr).
Definition pt_LF : Parser string ParseTree := lc (satisfy stringSeq is_lf).

Definition pt_CRLF : Parser string ParseTree :=
  
  (cr <- lc (satisfy stringSeq is_cr) ;;
   lf <- lc (satisfy stringSeq is_lf) ;;
   ret (Node "CRLF" [cr; lf]))
  <|> (lf <- lc (satisfy stringSeq is_lf) ;; ret (Node "CRLF" [lf]))
  <|> (cr <- lc (satisfy stringSeq is_cr) ;; ret (Node "CRLF" [cr])).


Definition pt_comment (input : string) : Parser string ParseTree :=
  semi <- lc (char ";"%char) ;;
  body <- many (pt_WSP <|> pt_VCHAR) input ;;
  nl <- pt_CRLF ;;
  ret (Node "comment" (semi :: body ++ [nl])).

Definition pt_c_nl (input : string) : Parser string ParseTree :=
  t <- (pt_comment input <|> pt_CRLF) ;;
  ret (Node "c-nl" [t]).

Definition pt_c_wsp (input : string) : Parser string ParseTree :=
  (t <- pt_WSP ;; ret (Node "c-wsp" [t]))
  <|>
  (nl <- pt_c_nl input ;; ws <- pt_WSP ;; ret (Node "c-wsp" [nl; ws])).

Definition pt_owsp (input : string) : Parser string ParseTree :=
  ts <- many (pt_c_wsp input) input ;;
  ret (Node "owsp" ts).

Definition pt_rulename (input : string) : Parser string ParseTree :=
  first <- pt_ALPHA ;;
  rest <- many (lc (satisfy stringSeq is_rulename_cont)) input ;;
  ret (Node "rulename" (first :: rest)).

Definition pt_defined_as (input : string) : Parser string ParseTree :=
  ws1 <- pt_owsp input ;;
  op <- ( (eq <- lc (char "="%char) ;;
            sl <- lc (char "/"%char) ;;
            ret (Node "def-incr" [eq; sl]))
           <|>
           (eq <- lc (char "="%char) ;;
            ret (Node "def-equal" [eq]))) ;;
  ws2 <- pt_owsp input ;;
  ret (Node "defined-as" [ws1; op; ws2]).

Definition pt_char_val (input : string) : Parser string ParseTree :=
  q1 <- pt_DQUOTE ;;
  body <- many (lc (satisfy stringSeq is_char_val_char)) input ;;
  q2 <- pt_DQUOTE ;;
  ret (Node "char-val" (q1 :: body ++ [q2])).

Definition pt_prose_val (input : string) : Parser string ParseTree :=
  lt <- lc (char "<"%char) ;;
  body <- many (lc (satisfy stringSeq is_prose_char)) input ;;
  gt <- lc (char ">"%char) ;;
  ret (Node "prose-val" (lt :: body ++ [gt])).

Definition pt_digits (input : string) : Parser string ParseTree :=
  lcs input p_digit.

Definition pt_hexdigs (input : string) : Parser string ParseTree :=
  lcs input p_hexdig.

Definition pt_bits (input : string) : Parser string ParseTree :=
  lcs input p_bit.

Definition pt_bin_val (input : string) : Parser string ParseTree :=
  b <- lc (char "b"%char) ;;
  ds <- pt_bits input ;;
  rest <- (
    
    (dash <- lc (char "-"%char) ;;
     ds2 <- pt_bits input ;;
     ret [dash; ds2])
    <|>
    
    (segs <- many1 (dot <- lc (char "."%char) ;;
                     d <- pt_bits input ;;
                     ret (Node "" [dot; d])) input ;;
     ret segs)
    <|>
    ret []
  ) ;;
  ret (Node "bin-val" ([b; ds] ++ rest)).

Definition pt_dec_val (input : string) : Parser string ParseTree :=
  d <- lc (char "d"%char) ;;
  ds <- pt_digits input ;;
  rest <- (
    (dash <- lc (char "-"%char) ;;
     ds2 <- pt_digits input ;;
     ret [dash; ds2])
    <|>
    (segs <- many1 (dot <- lc (char "."%char) ;;
                     d' <- pt_digits input ;;
                     ret (Node "" [dot; d'])) input ;;
     ret segs)
    <|>
    ret []
  ) ;;
  ret (Node "dec-val" ([d; ds] ++ rest)).

Definition pt_hex_val (input : string) : Parser string ParseTree :=
  x <- lc (char "x"%char) ;;
  ds <- pt_hexdigs input ;;
  rest <- (
    (dash <- lc (char "-"%char) ;;
     ds2 <- pt_hexdigs input ;;
     ret [dash; ds2])
    <|>
    (segs <- many1 (dot <- lc (char "."%char) ;;
                     d <- pt_hexdigs input ;;
                     ret (Node "" [dot; d])) input ;;
     ret segs)
    <|>
    ret []
  ) ;;
  ret (Node "hex-val" ([x; ds] ++ rest)).

Definition pt_num_val (input : string) : Parser string ParseTree :=
  pct <- lc (char "%"%char) ;;
  v <- (pt_bin_val input <|> pt_dec_val input <|> pt_hex_val input) ;;
  ret (Node "num-val" [pct; v]).

Definition pt_repeat (input : string) : Parser string ParseTree :=
  
  (lo <- many (pt_DIGIT) input ;;
   star <- lc (char "*"%char) ;;
   hi <- many (pt_DIGIT) input ;;
   ret (Node "repeat" (lo ++ [star] ++ hi)))
  <|>
  
  (ds <- many1 pt_DIGIT input ;;
   ret (Node "repeat" ds)).


Fixpoint pt_alternation (input : string) : Parser string ParseTree :=
  match input with
  | EmptyString ⇒ fail
  | String _ input' ⇒
    first <- pt_concatenation input' ;;
    rest <- many (
      ws1 <- pt_owsp input' ;;
      sl <- lc (char "/"%char) ;;
      ws2 <- pt_owsp input' ;;
      c <- pt_concatenation input' ;;
      ret (Node "" [ws1; sl; ws2; c])
    ) input' ;;
    ret (Node "alternation" (first :: rest))
  end

with pt_concatenation (input : string) : Parser string ParseTree :=
  match input with
  | EmptyString ⇒ fail
  | String _ input' ⇒
    first <- pt_repetition input' ;;
    rest <- many (
      ws <- many1 (pt_c_wsp input') input' ;;
      r <- pt_repetition input' ;;
      ret (Node "" (ws ++ [r]))
    ) input' ;;
    ret (Node "concatenation" (first :: rest))
  end

with pt_repetition (input : string) : Parser string ParseTree :=
  match input with
  | EmptyString ⇒ fail
  | String _ input' ⇒
    rep <- optional (pt_repeat input') ;;
    elem <- pt_element input' ;;
    match rep with
    | None ⇒ ret (Node "repetition" [elem])
    | Some r ⇒ ret (Node "repetition" [r; elem])
    end
  end

with pt_element (input : string) : Parser string ParseTree :=
  match input with
  | EmptyString ⇒ fail
  | String _ input' ⇒
    t <- (
      (rn <- pt_rulename input' ;; ret (Node "element" [rn]))
      <|> (pt_group input')
      <|> (pt_option_rule input')
      <|> (cv <- pt_char_val input' ;; ret (Node "element" [cv]))
      <|> (nv <- pt_num_val input' ;; ret (Node "element" [nv]))
      <|> (pv <- pt_prose_val input' ;; ret (Node "element" [pv]))
    ) ;;
    ret t
  end

with pt_group (input : string) : Parser string ParseTree :=
  match input with
  | EmptyString ⇒ fail
  | String _ input' ⇒
    lp <- lc (char "("%char) ;;
    ws1 <- pt_owsp input' ;;
    body <- pt_alternation input' ;;
    ws2 <- pt_owsp input' ;;
    rp <- lc (char ")"%char) ;;
    ret (Node "element" [Node "group" [lp; ws1; body; ws2; rp]])
  end

with pt_option_rule (input : string) : Parser string ParseTree :=
  match input with
  | EmptyString ⇒ fail
  | String _ input' ⇒
    lb <- lc (char "["%char) ;;
    ws1 <- pt_owsp input' ;;
    body <- pt_alternation input' ;;
    ws2 <- pt_owsp input' ;;
    rb <- lc (char "]"%char) ;;
    ret (Node "element" [Node "option" [lb; ws1; body; ws2; rb]])
  end.

Definition pt_elements (input : string) : Parser string ParseTree :=
  body <- pt_alternation input ;;
  ws <- pt_owsp input ;;
  ret (Node "elements" [body; ws]).

Definition pt_rule (input : string) : Parser string ParseTree :=
  name <- pt_rulename input ;;
  da <- pt_defined_as input ;;
  elts <- pt_elements input ;;
  nl <- (pt_c_nl input <|> (_ <- eos stringSeq ;; ret (Leaf ""))) ;;
  ret (Node "rule" [name; da; elts; nl]).

Definition pt_rulelist (input : string) : Parser string ParseTree :=
  rs <- many1 (
    (pt_rule input)
    <|>
    (ws <- pt_owsp input ;; nl <- pt_c_nl input ;; ret (Node "blank" [ws; nl]))
  ) input ;;
  ret (Node "rulelist" rs).


Definition parse_abnf_tree (input : string) : option ParseTree :=
  run (pt_rulelist input) input.


Definition tree_text (t : ParseTree) : string := pretty_print t.

Definition extract_rulename (t : ParseTree) : string :=
  tree_text t.

Fixpoint has_label (name : string) (t : ParseTree) : bool :=
  match t with
  | Leaf _ ⇒ false
  | Node n cs ⇒
    (String.eqb n name ||
     (fix go (ts : list ParseTree) :=
        match ts with
        | [] ⇒ false
        | t :: rest ⇒ (has_label name t || go rest)%bool
        end) cs)%bool
  end.

Definition extract_defined_as (t : ParseTree) : DefinedAs :=
  if has_label "def-incr" t then DefIncremental else DefEqual.

Fixpoint find_nodes (name : string) (ts : list ParseTree) : list ParseTree :=
  match ts with
  | [] ⇒ []
  | (Node n cs as t) :: rest ⇒
    if String.eqb n name then t :: find_nodes name rest
    else find_nodes name rest
  | _ :: rest ⇒ find_nodes name rest
  end.

Fixpoint find_first (name : string) (t : ParseTree) : option ParseTree :=
  match t with
  | Leaf _ ⇒ None
  | Node n cs ⇒
    if String.eqb n name then Some t
    else (fix go (ts : list ParseTree) :=
           match ts with
           | [] ⇒ None
           | t :: rest ⇒
             match find_first name t with
             | Some r ⇒ Some r
             | None ⇒ go rest
             end
           end) cs
  end.

Fixpoint hex_of_string (s : string) : nat :=
  match s with
  | EmptyString ⇒ 0
  | String c rest ⇒
    hexdig_val c + 16 × hex_of_string rest
  end.


Fixpoint string_to_list (s : string) : list ascii :=
  match s with
  | EmptyString ⇒ []
  | String c rest ⇒ c :: string_to_list rest
  end.


Fixpoint split_on_star (cs : list ascii) : (list ascii × list ascii) :=
  match cs with
  | [] ⇒ ([], [])
  | c :: rest ⇒
    if Ascii.eqb c "*"%char then ([], rest)
    else let '(before, after) := split_on_star rest in
         (c :: before, after)
  end.

Definition hex_of_string_be (s : string) : nat :=
  digits_value 16 (List.map hexdig_val (string_to_list s)).

Definition dec_of_string_be (s : string) : nat :=
  digits_value 10 (List.map digit_val (string_to_list s)).

Definition bin_of_string_be (s : string) : nat :=
  digits_value 2 (List.map bit_val (string_to_list s)).

Definition extract_char_val_text (t : ParseTree) : string :=
  match t with
  | Node "char-val" cs ⇒
    let inner := match cs with
                 | _ :: rest ⇒ removelast rest
                 | _ ⇒ []
                 end in
    (fix go (ts : list ParseTree) : string :=
       match ts with
       | [] ⇒ ""
       | t :: rest ⇒ (tree_text t ++ go rest)%string
       end) inner
  | _ ⇒ tree_text t
  end.

Definition extract_repeat (t : ParseTree) : option RepeatSpec :=
  match t with
  | Node "repeat" cs ⇒
    let text := tree_text t in
    if List.existsb (fun c ⇒ Ascii.eqb c "*"%char) (string_to_list text)
    then
      let parts := split_on_star (string_to_list text) in
      match parts with
      | (lo_s, hi_s) ⇒
        let lo := if String.eqb (list_to_string lo_s) "" then None
                  else Some (dec_of_string_be (list_to_string lo_s)) in
        let hi := if String.eqb (list_to_string hi_s) "" then None
                  else Some (dec_of_string_be (list_to_string hi_s)) in
        Some (mkRepeat lo hi)
      end
    else
      let n := dec_of_string_be text in
      Some (mkRepeat (Some n) (Some n))
  | _ ⇒ None
  end.

Definition extract_num_val (t : ParseTree) : option NumSpec :=
  match t with
  | Node "num-val" (Leaf "%" :: Node vtype cs :: _) ⇒
    let base_char := match cs with
                     | Leaf s :: _ ⇒ s
                     | _ ⇒ ""
                     end in
    let base := if String.eqb base_char "x" then Some HexBase
                else if String.eqb base_char "d" then Some DecBase
                else if String.eqb base_char "b" then Some BinBase
                else None in
    match base with
    | None ⇒ None
    | Some b ⇒
      let parse_num := match b with
                       | HexBase ⇒ hex_of_string_be
                       | DecBase ⇒ dec_of_string_be
                       | BinBase ⇒ bin_of_string_be
                       end in
      match cs with
      | _ :: Leaf digits :: rest ⇒
        let first := parse_num digits in
        match rest with
        | Leaf "-" :: Leaf digits2 :: _ ⇒
          Some (NumRange b first (parse_num digits2))
        | _ ⇒
          let vals := first :: List.map
            (fun seg ⇒ match seg with
                        | Node _ [Leaf "."; Leaf d] ⇒ parse_num d
                        | Node _ [_; Leaf d] ⇒ parse_num d
                        | _ ⇒ 0
                        end) rest in
          Some (NumConcat b vals)
        end
      | _ ⇒ None
      end
    end
  | _ ⇒ None
  end.


Fixpoint extract_alternation (t : ParseTree) (depth : nat) : option ABNFAlternation :=
  match depth with
  | O ⇒ None
  | Datatypes.S depth' ⇒
    match t with
    | Node "alternation" ts ⇒
      let concats := List.map (fun c ⇒ extract_concatenation c depth') ts in
      let good := List.map (fun oc ⇒ match oc with Some c ⇒ c | None ⇒ Concat [] end)
                           (List.filter (fun oc ⇒ match oc with Some _ ⇒ true | None ⇒ false end) concats) in
      Some (Alt good)
    | _ ⇒ None
    end
  end

with extract_concatenation (t : ParseTree) (depth : nat) : option ABNFConcatenation :=
  match depth with
  | O ⇒ None
  | Datatypes.S depth' ⇒
    match t with
    | Node "concatenation" ts ⇒
      let reps := List.map (fun r ⇒ extract_repetition r depth') ts in
      let good := List.map (fun or_ ⇒ match or_ with Some r ⇒ r | None ⇒ Rep None (ElemCharVal "") end)
                           (List.filter (fun or_ ⇒ match or_ with Some _ ⇒ true | None ⇒ false end) reps) in
      Some (Concat good)
    | Node "repetition" _ ⇒
      match extract_repetition t depth' with
      | Some r ⇒ Some (Concat [r])
      | None ⇒ None
      end
    | _ ⇒ None
    end
  end

with extract_repetition (t : ParseTree) (depth : nat) : option ABNFRepetition :=
  match depth with
  | O ⇒ None
  | Datatypes.S depth' ⇒
    match t with
    | Node "repetition" [elem] ⇒
      match extract_element elem depth' with
      | Some e ⇒ Some (Rep None e)
      | None ⇒ None
      end
    | Node "repetition" [rep; elem] ⇒
      let rspec := extract_repeat rep in
      match extract_element elem depth' with
      | Some e ⇒ Some (Rep rspec e)
      | None ⇒ None
      end
    | _ ⇒ None
    end
  end

with extract_element (t : ParseTree) (depth : nat) : option ABNFElement :=
  match depth with
  | O ⇒ None
  | Datatypes.S depth' ⇒
    match t with
    | Node "element" [child] ⇒
      match child with
      | Node "rulename" _ ⇒ Some (ElemRuleName (tree_text child))
      | Node "char-val" _ ⇒ Some (ElemCharVal (extract_char_val_text child))
      | Node "num-val" _ ⇒
        match extract_num_val child with
        | Some spec ⇒ Some (ElemNumVal spec)
        | None ⇒ None
        end
      | Node "prose-val" _ ⇒ Some (ElemProseVal (tree_text child))
      | Node "group" (_ :: _ :: alt :: _) ⇒
        match extract_alternation alt depth' with
        | Some a ⇒ Some (ElemGroup a)
        | None ⇒ None
        end
      | Node "option" (_ :: _ :: alt :: _) ⇒
        match extract_alternation alt depth' with
        | Some a ⇒ Some (ElemOption a)
        | None ⇒ None
        end
      | _ ⇒ None
      end
    | Node "rulename" _ ⇒ Some (ElemRuleName (tree_text t))
    | _ ⇒ None
    end
  end.

Definition extract_rule (t : ParseTree) (depth : nat) : option ABNFRule :=
  match t with
  | Node "rule" (name_t :: da_t :: elts_t :: _) ⇒
    let name := tree_text name_t in
    let da := extract_defined_as da_t in
    match elts_t with
    | Node "elements" (alt_t :: _) ⇒
      match extract_alternation alt_t depth with
      | Some body ⇒ Some (mkRule name da body)
      | None ⇒ None
      end
    | _ ⇒ None
    end
  | _ ⇒ None
  end.

Definition extract_rulelist (t : ParseTree) (depth : nat) : option ABNFRulelist :=
  match t with
  | Node "rulelist" rules ⇒
    Some (List.fold_right
      (fun r acc ⇒
         match extract_rule r depth with
         | Some rule ⇒ rule :: acc
         | None ⇒ acc
         end)
      [] rules)
  | _ ⇒ None
  end.


Definition parse_abnf_full (input : string) : option (ParseTree × ABNFRulelist) :=
  match parse_abnf_tree input with
  | None ⇒ None
  | Some tree ⇒
    match extract_rulelist tree 100 with
    | Some rules ⇒ Some (tree, rules)
    | None ⇒ None
    end
  end.

Example extract_matches_direct :
  match parse_abnf_full g1 with
  | Some (_, rules) ⇒
    match rules with
    | [r] ⇒ (String.eqb (rule_name r) "DIGIT",
              match rule_body r with
              | Alt [Concat [Rep None (ElemNumVal (NumRange HexBase 48 57))]] ⇒ true
              | _ ⇒ false
              end)
    | _ ⇒ (false, false)
    end
  | None ⇒ (false, false)
  end = (true, true).
Proof. reflexivity. Qed.
