Library Riktig.Theory.VectorSpace

From Coq Require Export Morphisms.
From Riktig Require Export Ops Props Interface.Field Interface.VectorSpace
  Theory.Group Lib.Calc.

Section VectorSpaceTheory.
  Context {F V : Type} `{VectorSpace F V}.

  #[global] Instance scale_l_proper : ∀ a : F, Proper (equ ==> equ) (@scale F V _ a).
  Proof.
    intros a u v Huv.
    rewrite Huv.
    reflexivity.
  Qed.

  #[global] Instance scale_r_proper : ∀ u : V, Proper (equ ==> equ) (fun a : F ⇒ a *: u).
  Proof.
    intros u a b Hab.
    rewrite Hab.
    reflexivity.
  Qed.

  Theorem scale_zero : ∀ a : F, a *: (0 : V) == 0.
  Proof.
    intros a.
    exact (hom_zero (scale a) (sc_scale_add a)).
  Qed.

  Theorem zero_scale : ∀ u : V, (0 : F) *: u == 0.
  Proof.
    intros u.
    exact (hom_zero (fun a : F ⇒ a *: u) (fun a b ⇒ sc_add_scale a b u)).
  Qed.

  Theorem neg_scale : ∀ (a : F) (u : V), (- a) *: u == - (a *: u).
  Proof.
    intros a u.
    exact (hom_neg (fun x : F ⇒ x *: u) (fun x y ⇒ sc_add_scale x y u) a).
  Qed.

  Theorem scale_neg : ∀ (a : F) (u : V), a *: (- u) == - (a *: u).
  Proof.
    intros a u.
    exact (hom_neg (scale a) (sc_scale_add a) u).
  Qed.

  Theorem scale_sub : ∀ (a : F) (u v : V),
    a *: (u + - v) == a *: u + - (a *: v).
  Proof.
    intros a u v.
    calc (a *: (u + - v))
      == (a *: u + a *: (- v)) { (sc_scale_add a u (- v)) }
      == (a *: u + - (a *: v)) { (scale_neg a v) }.
  Qed.

  Theorem sub_scale : ∀ (a b : F) (u : V),
    (a + - b) *: u == a *: u + - (b *: u).
  Proof.
    intros a b u.
    calc ((a + - b) *: u)
      == (a *: u + (- b) *: u) { (sc_add_scale a (- b) u) }
      == (a *: u + - (b *: u)) { (neg_scale b u) }.
  Qed.

  Theorem scale_swap : ∀ (a b : F) (u : V),
    a *: (b *: u) == b *: (a *: u).
  Proof.
    intros a b u.
    calc (a *: (b *: u))
      == ((a × b) *: u) { (sc_scale_scale a b u) }
      == ((b × a) *: u) { (comm a b) }
      == (b *: (a *: u)) { (sc_scale_scale b a u) }.
  Qed.

  Theorem neg1_scale : ∀ u : V, (- (1 : F)) *: u == - u.
  Proof.
    intros u.
    calc ((- (1 : F)) *: u)
      == (- (1 *: u)) { (neg_scale 1 u) }
      == (- u) { (sc_scale_1 u) }.
  Qed.

  Theorem scale_flip : ∀ (a b : F) (u : V),
    a *: (b *: u) == (b × a) *: u.
  Proof.
    intros a b u.
    calc (a *: (b *: u))
      == ((a × b) *: u) { (sc_scale_scale a b u) }
      == ((b × a) *: u) { (comm a b) }.
  Qed.

End VectorSpaceTheory.
