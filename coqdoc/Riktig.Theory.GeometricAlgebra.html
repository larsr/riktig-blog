  Context {F V A : Type} `{GeometricAlgebra F V A}.

  Theorem vec_zero : ga_vec 0 == (0 : A).
  Proof.
    exact (hom_zero ga_vec ga_vec_add).
  Qed.

  Lemma contract_add : ∀ u v : V,
    (norm2 u *: (1 : A) + norm2 v *: 1)
      + (ga_vec u × ga_vec v + ga_vec v × ga_vec u)
    == norm2 (u + v) *: 1.
  Proof.
    intros u v.
    calc (((norm2 u *: (1 : A) + norm2 v *: (1 : A))
            + (ga_vec u × ga_vec v + ga_vec v × ga_vec u)) : A)
      == (((ga_vec u × ga_vec u + norm2 v *: (1 : A))
            + (ga_vec u × ga_vec v + ga_vec v × ga_vec u)) : A)
        { (ga_contract u) }
      == (((ga_vec u × ga_vec u + ga_vec v × ga_vec v)
            + (ga_vec u × ga_vec v + ga_vec v × ga_vec u)) : A)
        { (ga_contract v) }
      == (((ga_vec u × ga_vec u + ga_vec v × ga_vec u)
            + (ga_vec v × ga_vec v + ga_vec u × ga_vec v)) : A)
        { (add_rotate (ga_vec u × ga_vec u) (ga_vec v × ga_vec v)
             (ga_vec u × ga_vec v) (ga_vec v × ga_vec u)) }
      == (((ga_vec u × ga_vec u + ga_vec v × ga_vec u)
            + (ga_vec u × ga_vec v + ga_vec v × ga_vec v)) : A)
        { (comm (ga_vec v × ga_vec v) (ga_vec u × ga_vec v)) }
      == (((ga_vec u + ga_vec v) × (ga_vec u + ga_vec v)) : A)
        { (mul_add_expand (ga_vec u) (ga_vec v) (ga_vec u) (ga_vec v)) }
      == ((ga_vec (u + v) × ga_vec (u + v)) : A)
        { (ga_vec_add u v) }
      == ((norm2 (u + v) *: (1 : A)) : A)
        { (ga_contract (u + v)) }.
  Qed.

  Theorem polarization : ∀ u v : V,
    ga_vec u × ga_vec v + ga_vec v × ga_vec u
    == (⟨ u , v ⟩ + ⟨ v , u ⟩) *: (1 : A).
  Proof.
    intros u v.
    apply (cancel_l (norm2 u *: (1 : A) + norm2 v *: 1)).
    calc (((norm2 u *: (1 : A) + norm2 v *: (1 : A))
            + (ga_vec u × ga_vec v + ga_vec v × ga_vec u)) : A)
      == ((norm2 (u + v) *: (1 : A)) : A)
        { (contract_add u v) }
      == ((((norm2 u + norm2 v) + (⟨ u , v ⟩ + ⟨ v , u ⟩)) *: (1 : A)) : A)
        { (norm2_add u v) }
      == (((norm2 u + norm2 v) *: (1 : A) + (⟨ u , v ⟩ + ⟨ v , u ⟩) *: (1 : A)) : A)
        { (sc_add_scale (norm2 u + norm2 v) (⟨ u , v ⟩ + ⟨ v , u ⟩) (1 : A)) }
      == (((norm2 u *: (1 : A) + norm2 v *: (1 : A)) + (⟨ u , v ⟩ + ⟨ v , u ⟩) *: (1 : A)) : A)
        { (sc_add_scale (norm2 u) (norm2 v) (1 : A)) }.
  Qed.
