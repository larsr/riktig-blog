Library Riktig.Lambda.Scoped

From Coq Require Import Setoid Morphisms.
From Coq.Vectors Require Fin.


Inductive Exp : nat → Type :=
  | EVar : ∀ {n : nat}, Fin.t n → Exp n
  | EApp : ∀ {n : nat}, Exp n → Exp n → Exp n
  | ELam : ∀ {n : nat}, Exp (S n) → Exp n.


Definition Ren (m n : nat) : Type := Fin.t m → Fin.t n.

Definition up_ren {m n : nat} (r : Ren m n) : Ren (S m) (S n) :=
  fun x : Fin.t (S m) ⇒
    Fin.caseS' x (fun _ : Fin.t (S m) ⇒ Fin.t (S n))
      Fin.F1
      (fun y : Fin.t m ⇒ Fin.FS (r y)).


Lemma up_ren_F1 : ∀ (m n : nat) (r : Ren m n),
  up_ren r Fin.F1 = Fin.F1.
Proof.
  intros m n r. reflexivity.
Qed.

Lemma up_ren_FS : ∀ (m n : nat) (r : Ren m n) (y : Fin.t m),
  up_ren r (Fin.FS y) = Fin.FS (r y).
Proof.
  intros m n r y. reflexivity.
Qed.


Fixpoint rename {m n : nat} (r : Ren m n) (e : Exp m) {struct e} : Exp n :=
  match e in Exp m0 return Ren m0 n → Exp n with
  | EVar x ⇒ fun r0 ⇒ EVar (r0 x)
  | EApp f a ⇒ fun r0 ⇒ EApp (rename r0 f) (rename r0 a)
  | ELam b ⇒ fun r0 ⇒ ELam (rename (up_ren r0) b)
  end r.

Lemma rename_EVar : ∀ (m n : nat) (r : Ren m n) (x : Fin.t m),
  rename r (EVar x) = EVar (r x).
Proof.
  intros m n r x. reflexivity.
Qed.

Lemma rename_EApp : ∀ (m n : nat) (r : Ren m n) (f a : Exp m),
  rename r (EApp f a) = EApp (rename r f) (rename r a).
Proof.
  intros m n r f a. reflexivity.
Qed.

Lemma rename_ELam : ∀ (m n : nat) (r : Ren m n) (b : Exp (S m)),
  rename r (ELam b) = ELam (rename (up_ren r) b).
Proof.
  intros m n r b. reflexivity.
Qed.


Definition Sub (m n : nat) : Type := Fin.t m → Exp n.

Definition ids {n : nat} : Sub n n :=
  fun x : Fin.t n ⇒ EVar x.

Definition shift {n : nat} : Sub n (S n) :=
  fun x : Fin.t n ⇒ EVar (Fin.FS x).

Definition cons_sub {m n : nat} (t : Exp n) (s : Sub m n) : Sub (S m) n :=
  fun x : Fin.t (S m) ⇒
    Fin.caseS' x (fun _ : Fin.t (S m) ⇒ Exp n)
      t
      (fun y : Fin.t m ⇒ s y).

Lemma cons_sub_F1 : ∀ (m n : nat) (t : Exp n) (s : Sub m n),
  cons_sub t s Fin.F1 = t.
Proof.
  intros m n t s. reflexivity.
Qed.

Lemma cons_sub_FS : ∀ (m n : nat) (t : Exp n) (s : Sub m n) (y : Fin.t m),
  cons_sub t s (Fin.FS y) = s y.
Proof.
  intros m n t s y. reflexivity.
Qed.

Definition up {m n : nat} (s : Sub m n) : Sub (S m) (S n) :=
  fun x : Fin.t (S m) ⇒
    Fin.caseS' x (fun _ : Fin.t (S m) ⇒ Exp (S n))
      (EVar Fin.F1)
      (fun y : Fin.t m ⇒ rename Fin.FS (s y)).

Lemma up_F1 : ∀ (m n : nat) (s : Sub m n),
  up s Fin.F1 = EVar Fin.F1.
Proof.
  intros m n s. reflexivity.
Qed.

Lemma up_FS : ∀ (m n : nat) (s : Sub m n) (y : Fin.t m),
  up s (Fin.FS y) = rename Fin.FS (s y).
Proof.
  intros m n s y. reflexivity.
Qed.


Fixpoint subst {m n : nat} (s : Sub m n) (e : Exp m) {struct e} : Exp n :=
  match e in Exp m0 return Sub m0 n → Exp n with
  | EVar x ⇒ fun s0 ⇒ s0 x
  | EApp f a ⇒ fun s0 ⇒ EApp (subst s0 f) (subst s0 a)
  | ELam b ⇒ fun s0 ⇒ ELam (subst (up s0) b)
  end s.

Lemma subst_EVar : ∀ (m n : nat) (s : Sub m n) (x : Fin.t m),
  subst s (EVar x) = s x.
Proof.
  intros m n s x. reflexivity.
Qed.

Lemma subst_EApp : ∀ (m n : nat) (s : Sub m n) (f a : Exp m),
  subst s (EApp f a) = EApp (subst s f) (subst s a).
Proof.
  intros m n s f a. reflexivity.
Qed.

Lemma subst_ELam : ∀ (m n : nat) (s : Sub m n) (b : Exp (S m)),
  subst s (ELam b) = ELam (subst (up s) b).
Proof.
  intros m n s b. reflexivity.
Qed.

Definition comp_sub {m n p : nat} (r : Sub m n) (s : Sub n p) : Sub m p :=
  fun x : Fin.t m ⇒ subst s (r x).

Lemma comp_sub_at : ∀ (m n p : nat) (r : Sub m n) (s : Sub n p) (x : Fin.t m),
  comp_sub r s x = subst s (r x).
Proof.
  intros m n p r s x. reflexivity.
Qed.


Definition sub_eq {m n : nat} (r s : Sub m n) : Prop :=
  ∀ x : Fin.t m, r x = s x.

#[global] Instance sub_eq_Equivalence (m n : nat) : Equivalence (@sub_eq m n).
Proof.
  split.
  - intros s x. reflexivity.
  - intros s s' Hss' x. exact (eq_sym (Hss' x)).
  - intros s1 s2 s3 H12 H23 x. exact (eq_trans (H12 x) (H23 x)).
Qed.



Lemma up_ren_ext : ∀ (m n : nat) (r r' : Ren m n),
  (∀ x : Fin.t m, r x = r' x) →
  ∀ x : Fin.t (S m), up_ren r x = up_ren r' x.
Proof.
  intros m n r r' Hrr' x.
  apply (Fin.caseS' x (fun x0 : Fin.t (S m) ⇒ up_ren r x0 = up_ren r' x0)).
  - reflexivity.
  - intros y.
    rewrite (up_ren_FS m n r y), (up_ren_FS m n r' y), (Hrr' y).
    reflexivity.
Qed.

Lemma rename_ext : ∀ (m n : nat) (r r' : Ren m n),
  (∀ x : Fin.t m, r x = r' x) →
  ∀ e : Exp m, rename r e = rename r' e.
Proof.
  intros m n r r' Hrr' e.
  revert n r r' Hrr'.
  induction e as [m0 x | m0 f IHf a IHa | m0 b IH]; intros n r r' Hrr'.
  - simpl. rewrite (Hrr' x). reflexivity.
  - simpl. rewrite (IHf n r r' Hrr'), (IHa n r r' Hrr'). reflexivity.
  - simpl.
    rewrite (IH (S n) (up_ren r) (up_ren r') (up_ren_ext m0 n r r' Hrr')).
    reflexivity.
Qed.


Lemma up_ext : ∀ (m n : nat) (s s' : Sub m n),
  sub_eq s s' → sub_eq (up s) (up s').
Proof.
  intros m n s s' Hss' x.
  apply (Fin.caseS' x (fun x0 : Fin.t (S m) ⇒ up s x0 = up s' x0)).
  - reflexivity.
  - intros y.
    rewrite (up_FS m n s y), (up_FS m n s' y), (Hss' y).
    reflexivity.
Qed.

Lemma subst_ext : ∀ (m n : nat) (s s' : Sub m n),
  sub_eq s s' → ∀ e : Exp m, subst s e = subst s' e.
Proof.
  intros m n s s' Hss' e.
  revert n s s' Hss'.
  induction e as [m0 x | m0 f IHf a IHa | m0 b IH]; intros n s s' Hss'.
  - simpl. exact (Hss' x).
  - simpl. rewrite (IHf n s s' Hss'), (IHa n s s' Hss'). reflexivity.
  - simpl.
    rewrite (IH (S n) (up s) (up s') (up_ext m0 n s s' Hss')).
    reflexivity.
Qed.


#[global] Instance subst_Proper (m n : nat) :
  Proper (@sub_eq m n ==> eq ==> eq) (@subst m n).
Proof.
  intros s s' Hss' e e' Heq.
  rewrite <- Heq.
  exact (subst_ext m n s s' Hss' e).
Qed.

#[global] Instance up_Proper (m n : nat) :
  Proper (@sub_eq m n ==> @sub_eq (S m) (S n)) (@up m n).
Proof.
  intros s s' Hss'.
  exact (up_ext m n s s' Hss').
Qed.

#[global] Instance cons_sub_Proper (m n : nat) :
  Proper (eq ==> @sub_eq m n ==> @sub_eq (S m) n) (@cons_sub m n).
Proof.
  intros t t' Heqt s s' Hss' x.
  rewrite <- Heqt.
  apply (Fin.caseS' x (fun x0 : Fin.t (S m) ⇒ cons_sub t s x0 = cons_sub t s' x0)).
  - reflexivity.
  - intros y. exact (Hss' y).
Qed.

#[global] Instance comp_sub_Proper (m n p : nat) :
  Proper (@sub_eq m n ==> @sub_eq n p ==> @sub_eq m p) (@comp_sub m n p).
Proof.
  intros r r' Hrr' s s' Hss' x.
  unfold comp_sub.
  rewrite (Hrr' x).
  exact (subst_ext n p s s' Hss' (r' x)).
Qed.


Lemma up_ren_ren : ∀ (m n p : nat) (r : Ren m n) (r' : Ren n p)
    (x : Fin.t (S m)),
  up_ren (fun y : Fin.t m ⇒ r' (r y)) x = up_ren r' (up_ren r x).
Proof.
  intros m n p r r' x.
  apply (Fin.caseS' x (fun x0 : Fin.t (S m) ⇒
    up_ren (fun y : Fin.t m ⇒ r' (r y)) x0 = up_ren r' (up_ren r x0))).
  - reflexivity.
  - intros y. reflexivity.
Qed.

Lemma rename_rename : ∀ (m n p : nat) (r : Ren m n) (r' : Ren n p)
    (e : Exp m),
  rename r' (rename r e) = rename (fun x : Fin.t m ⇒ r' (r x)) e.
Proof.
  intros m n p r r' e.
  revert n p r r'.
  induction e as [m0 x | m0 f IHf a IHa | m0 b IH]; intros n p r r'.
  - reflexivity.
  - simpl. rewrite (IHf n p r r'), (IHa n p r r'). reflexivity.
  - simpl.
    rewrite (IH (S n) (S p) (up_ren r) (up_ren r')).
    rewrite (rename_ext (S m0) (S p)
               (fun x : Fin.t (S m0) ⇒ up_ren r' (up_ren r x))
               (up_ren (fun y : Fin.t m0 ⇒ r' (r y)))
               (fun x : Fin.t (S m0) ⇒ eq_sym (up_ren_ren m0 n p r r' x)) b).
    reflexivity.
Qed.


Lemma up_evar : ∀ (m n : nat) (r : Ren m n),
  sub_eq (up (fun y : Fin.t m ⇒ EVar (r y)))
         (fun x : Fin.t (S m) ⇒ EVar (up_ren r x)).
Proof.
  intros m n r x.
  apply (Fin.caseS' x (fun x0 : Fin.t (S m) ⇒
    up (fun y : Fin.t m ⇒ EVar (r y)) x0 = EVar (up_ren r x0))).
  - reflexivity.
  - intros y. reflexivity.
Qed.

Lemma rename_as_subst : ∀ (m n : nat) (r : Ren m n) (e : Exp m),
  rename r e = subst (fun x : Fin.t m ⇒ EVar (r x)) e.
Proof.
  intros m n r e.
  revert n r.
  induction e as [m0 x | m0 f IHf a IHa | m0 b IH]; intros n r.
  - reflexivity.
  - simpl. rewrite (IHf n r), (IHa n r). reflexivity.
  - simpl.
    rewrite (IH (S n) (up_ren r)).
    rewrite (subst_ext (S m0) (S n)
               (fun x : Fin.t (S m0) ⇒ EVar (up_ren r x))
               (up (fun y : Fin.t m0 ⇒ EVar (r y)))
               (fun x : Fin.t (S m0) ⇒ eq_sym (up_evar m0 n r x)) b).
    reflexivity.
Qed.


Lemma up_ren_sub : ∀ (m n p : nat) (r : Ren m n) (s : Sub n p),

  - intros y. reflexivity.
Qed.

Lemma subst_rename : ∀ (m n p : nat) (r : Ren m n) (s : Sub n p)
    (e : Exp m),
  subst s (rename r e) = subst (fun x : Fin.t m ⇒ s (r x)) e.
Proof.
  intros m n p r s e.
  revert n p r s.
  induction e as [m0 x | m0 f IHf a IHa | m0 b IH]; intros n p r s.
  - reflexivity.
  - simpl. rewrite (IHf n p r s), (IHa n p r s). reflexivity.
  - simpl.
    rewrite (IH (S n) (S p) (up_ren r) (up s)).
    rewrite (subst_ext (S m0) (S p)
               (fun x : Fin.t (S m0) ⇒ up s (up_ren r x))
               (up (fun y : Fin.t m0 ⇒ s (r y)))
               (up_ren_sub m0 n p r s) b).
    reflexivity.
Qed.


Lemma up_sub_ren : ∀ (m n p : nat) (s : Sub m n) (r : Ren n p),
  sub_eq (fun x : Fin.t (S m) ⇒ rename (up_ren r) (up s x))
         (up (fun y : Fin.t m ⇒ rename r (s y))).
Proof.
  intros m n p s r x.
  apply (Fin.caseS' x (fun x0 : Fin.t (S m) ⇒
    rename (up_ren r) (up s x0) = up (fun y : Fin.t m ⇒ rename r (s y)) x0)).
  - reflexivity.
  - intros y.
    rewrite (up_FS m n s y),
      (up_FS m p (fun z : Fin.t m ⇒ rename r (s z)) y).
    rewrite (rename_rename n (S n) (S p) (@Fin.FS n) (up_ren r) (s y)).
    rewrite (rename_rename n p (S p) r (@Fin.FS p) (s y)).
    apply (rename_ext n (S p)
             (fun z : Fin.t n ⇒ up_ren r (Fin.FS z))
             (fun z : Fin.t n ⇒ Fin.FS (r z))).
    intros z. reflexivity.
Qed.

Lemma rename_subst : ∀ (m n p : nat) (s : Sub m n) (r : Ren n p)

Proof.
  intros m n p s r e.
  revert n p s r.
  induction e as [m0 x | m0 f IHf a IHa | m0 b IH]; intros n p s r.
  - reflexivity.
  - simpl. rewrite (IHf n p s r), (IHa n p s r). reflexivity.
  - simpl.
    rewrite (IH (S n) (S p) (up s) (up_ren r)).

               (fun x : Fin.t (S m0) ⇒ rename (up_ren r) (up s x))
               (up (fun y : Fin.t m0 ⇒ rename r (s y)))

Lemma up_ids : ∀ n : nat, sub_eq (up (@ids n)) (@ids (S n)).

Lemma up_comp : ∀ (m n p : nat) (r : Sub m n) (s : Sub n p),
  sub_eq (up (comp_sub r s)) (comp_sub (up r) (up s)).
Proof.
  intros m n p r s x.
  apply (Fin.caseS' x (fun x0 : Fin.t (S m) ⇒
    up (comp_sub r s) x0 = comp_sub (up r) (up s) x0)).
  - reflexivity.
  - intros y.
    unfold comp_sub.
    rewrite (up_FS m p (fun z : Fin.t m ⇒ subst s (r z)) y).
    rewrite (up_FS m n r y).
    rewrite (rename_subst n p (S p) s (@Fin.FS p) (r y)).
    rewrite (subst_rename n (S n) (S p) (@Fin.FS n) (up s) (r y)).
    apply (subst_ext n (S p)
             (fun z : Fin.t n ⇒ rename Fin.FS (s z))
             (fun z : Fin.t n ⇒ up s (Fin.FS z))).
    intros z.
    exact (eq_sym (up_FS n p s z)).
Qed.


Theorem subst_subst : ∀ (m n p : nat) (r : Sub m n) (s : Sub n p)
    (e : Exp m),
  subst s (subst r e) = subst (comp_sub r s) e.
Proof.
  intros m n p r s e.
  revert n p r s.
  induction e as [m0 x | m0 f IHf a IHa | m0 b IH]; intros n p r s.
  - reflexivity.
  - simpl. rewrite (IHf n p r s), (IHa n p r s). reflexivity.
  - simpl.
    rewrite (IH (S n) (S p) (up r) (up s)).
    rewrite (subst_ext (S m0) (S p)
               (comp_sub (up r) (up s))
               (up (comp_sub r s))
               (fun x : Fin.t (S m0) ⇒ eq_sym (up_comp m0 n p r s x)) b).
    reflexivity.
Qed.
