Library Riktig.Lambda.Conv


Require Import Riktig.Lambda.Syntax.
From Riktig Require Import Ops.
From Coq Require Import Morphisms Setoid.

Notation "'0'" := O.
Notation "'1'" := (S O).


Inductive conv : tm → tm → Prop :=
  | conv_step : ∀ t t', step t t' → conv t t'
  | conv_abs : ∀ b b', conv b b' → conv (Abs b) (Abs b')
  | conv_appl : ∀ f f' a, conv f f' → conv (App f a) (App f' a)
  | conv_appr : ∀ f a a', conv a a' → conv (App f a) (App f a')
  | conv_refl : ∀ t, conv t t
  | conv_sym : ∀ t u, conv t u → conv u t
  | conv_trans : ∀ t u v, conv t u → conv u v → conv t v.

#[global] Instance Tm_EqRel : EqRel tm := conv.

#[global] Instance Tm_Equivalence : Equivalence conv.
Proof.
  split.
  - exact conv_refl.
  - exact conv_sym.
  - exact conv_trans.
Qed.

#[global] Instance Abs_Proper : Proper (equ ==> equ) Abs.
Proof.
  intros b b' Hb. apply conv_abs. exact Hb.
Qed.

#[global] Instance App_Proper : Proper (equ ==> equ ==> equ) App.
Proof.
  intros f f' Hf a a' Ha.
  eapply conv_trans.
  - apply conv_appl. exact Hf.
  - apply conv_appr. exact Ha.
Qed.


Lemma conv_of_star : ∀ t t', star t t' → t == t'.
Proof.
  intros t t' Hs. induction Hs as [t0 | t0 u v Hstep _ IH].
  - apply conv_refl.
  - eapply conv_trans.
    + apply conv_step. exact Hstep.
    + exact IH.
Qed.

Lemma conv_of_eval : ∀ n t, t == eval n t.
Proof.
  intros n t. apply conv_of_star. apply eval_sound.
Qed.

Lemma conv_by_eval : ∀ n m s t, eval n s = eval m t → s == t.
Proof.
  intros n m s t Heq.
  eapply conv_trans.
  - exact (conv_of_eval n s).
  - rewrite Heq. apply conv_sym. apply conv_of_eval.
Qed.

Lemma conv_beta : ∀ b v, value v → App (Abs b) v == subst 0 v b.
Proof.
  intros b v Hv. apply conv_step. apply step_beta. exact Hv.
Qed.


Example conv_id_id : App idt idt == idt.
Proof. apply (conv_by_eval 10 10). reflexivity. Qed.

Example conv_k : App (App k idt) (Abs (Abs (Var 0))) == idt.
Proof. apply (conv_by_eval 10 10). reflexivity. Qed.
