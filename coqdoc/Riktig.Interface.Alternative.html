Library Riktig.Interface.Alternative

From Coq Require Import Morphisms.
From Riktig Require Import Ops.

Class HasFail (M : Type → Type) := fail : ∀ {A}, M A.
Class HasAlt (M : Type → Type) := alt : ∀ {A}, M A → M A → M A.

Arguments fail {M _ A}.
Arguments alt {M _ A} p q.

Infix "<|>" := alt (at level 50, left associativity).

Class Alternative (M : Type → Type) `{∀ A, EqRel (M A)} `{HasFail M} `{HasAlt M} := {
  
  #[global] alt_equiv :: ∀ A, Equivalence (@equ (M A) _);
  #[global] alt_proper :: ∀ A, Proper (equ ==> equ ==> equ) (@alt M _ A);
  alt_assoc : ∀ A (p q r : M A), alt (alt p q) r == alt p (alt q r);
  fail_alt_l : ∀ A (q : M A), alt fail q == q;
  alt_fail_r : ∀ A (p : M A), alt p fail == p
}.

#[global] Hint Mode HasFail ! : typeclass_instances.
#[global] Hint Mode HasAlt ! : typeclass_instances.
#[global] Hint Mode Alternative ! - - - : typeclass_instances.
