Section BinarySearch.

  Variable V : Type.
  Variable leb : V → V → bool.

  Variable mem : Type.
  Variable read : mem → nat → V.

  Variable m : mem.
  Variable target : V.

  Definition bsearch : nat → nat → option nat.
  apply (uncurry ∘ (Fix (well_founded_ltof (nat × nat) (fun p ⇒ S (snd p) - fst p))
           (fun _ ⇒ option nat))).
  refine
  ( fun p bsearch ⇒
      match p as x return p = x → option nat with
      | (lo, hi) ⇒
          fun e ⇒
          if dec (hi < lo)
          then None
          else
            let mid := lo + (hi - lo) / 2 in
            if dec (leb (read m mid) target = true)
            then if dec (leb target (read m mid) = true)
                 then Some mid
                 else bsearch (S mid, hi) _
            else if dec (mid = lo)
                 then None
                 else bsearch (lo, mid - 1) _
      end eq_refl
  );
  (subst p; unfold ltof; cbn [fst snd]; pose proof (half_le (hi - lo)); abstract lia).
  Defined.

  Lemma bsearch_step lo hi :
    bsearch lo hi =
      if dec (hi < lo)
      then None
      else
        let mid := lo + (hi - lo) / 2 in
        if dec (leb (read m mid) target = true)
        then if dec (leb target (read m mid) = true)
             then Some mid
             else bsearch (S mid) hi
        else if dec (mid = lo)
             then None
             else bsearch lo (mid - 1).
  Proof.
    unfold bsearch at 1, compose, uncurry.
    rewrite Fix_eq.
    - reflexivity.
    - intros [lo' hi'] f g H.
      destruct (dec (hi' < lo')); [reflexivity|].
      destruct (dec (leb (read m (lo' + (hi' - lo') / 2)) target = true));
        [ destruct (dec (leb target (read m (lo' + (hi' - lo') / 2)) = true));
          [reflexivity | rewrite H; reflexivity]
        | destruct (dec (lo' + (hi' - lo') / 2 = lo')); [reflexivity | rewrite H; reflexivity] ].
  Qed.

End BinarySearch.


Section TestBsearch.
Import ListNotations.

Definition l : list nat := [1;3;5;7;9;11;13].

Check Logic.eq_refl : bsearch nat Nat.leb (list nat) (nth_default 0) l 7 0 6 = Some 3.
Check Logic.eq_refl : bsearch nat Nat.leb (list nat) (nth_default 0) l 1 0 6 = Some 0.
Check Logic.eq_refl : bsearch nat Nat.leb (list nat) (nth_default 0) l 13 0 6 = Some 6.
Check Logic.eq_refl : bsearch nat Nat.leb (list nat) (nth_default 0) l 4 0 6 = None.
Check Logic.eq_refl : bsearch nat Nat.leb (list nat) (nth_default 0) l 100 0 6 = None.

End TestBsearch.
