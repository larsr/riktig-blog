Lemma half_le n : (n - 1) / 2 ≤ n - 1.
Proof. apply Nat.Div0.div_le_upper_bound. lia. Qed.

Lemma half_lt n : n ≠ 0 → (n - 1) / 2 < n.
Proof. intros H. pose proof (half_le n). lia. Qed.

Section BinarySearch.

  Variable V : Type.
  Variable leb : V → V → bool.

  Variable mem : Type.
  Variable read : mem → nat → V.

  Variable m : mem.
  Variable target : V.

  Definition bsearch_aux : nat → nat → option nat.
  refine (Fix lt_wf (fun _ ⇒ nat → option nat)
    (fun n rec lo ⇒
       if dec (n = 0)
       then None
       else
         let hi := lo + n - 1 in
         let mid := lo + (n - 1) / 2 in
         if dec (leb (read m mid) target = true)
         then if dec (leb target (read m mid) = true)
              then Some mid
              else rec (hi - mid) _ (S mid)
         else rec (mid - lo) _ lo)).
  - abstract lia.
  - abstract (pose proof (half_lt n ltac:(assumption)); lia).
  Defined.

  Lemma bsearch_aux_step n lo :
    bsearch_aux n lo =
      if dec (n = 0)
      then None
      else
        let hi := lo + n - 1 in
        let mid := lo + (n - 1) / 2 in
        if dec (leb (read m mid) target = true)
        then if dec (leb target (read m mid) = true)
             then Some mid
             else bsearch_aux (hi - mid) (S mid)
        else bsearch_aux (mid - lo) lo.
  Proof.
    unfold bsearch_aux at 1. rewrite Fix_eq.
    - reflexivity.
    - intros x f g Hfg.
      apply functional_extensionality; intro lo'.
      destruct (dec (x = 0)); [reflexivity|].
      destruct (dec (_ = true)); [|rewrite Hfg; reflexivity].
      destruct (dec (_ = true)); [reflexivity|rewrite Hfg; reflexivity].
  Qed.

  Definition bsearch (lo hi : nat) : option nat :=
    bsearch_aux (S hi - lo) lo.

End BinarySearch.


Section TestBsearch.
Import ListNotations.

Definition l : list nat := [1;3;5;7;9;11;13].

Check Logic.eq_refl : bsearch nat Nat.leb (list nat) (nth_default 0) l 7 0 6 = Some 3.
Check Logic.eq_refl : bsearch nat Nat.leb (list nat) (nth_default 0) l 1 0 6 = Some 0.
Check Logic.eq_refl : bsearch nat Nat.leb (list nat) (nth_default 0) l 13 0 6 = Some 6.
Check Logic.eq_refl : bsearch nat Nat.leb (list nat) (nth_default 0) l 4 0 6 = None.
Check Logic.eq_refl : bsearch nat Nat.leb (list nat) (nth_default 0) l 100 0 6 = None.

End TestBsearch.
