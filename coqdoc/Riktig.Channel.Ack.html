Library Riktig.Channel.Ack


From Coq Require Export List Bool Lia.
From Riktig.Theory Require Export Subseq.
From Riktig.Channel Require Export Trace ABP Fairness.
Import ListNotations.

Section Ack.
  Context {M : Type}.


  Fixpoint abp_ack_run (b : bit) (ms : list M) (fwire : list Packet) (awire : list bit) : Prop :=
    match ms with
    | [] ⇒ fwire = [] ∧ awire = []
    | m :: ms' ⇒
        ∃ (k : nat) (w1 : list Packet),
          lossy_dup (sender_burst b m k) w1 ∧
          ( (deliver b w1 = [] ∧ fwire = w1 ∧ awire = [])
          ∨ (∃ a1 : list bit,
                deliver b w1 = [m] ∧ lossy_dup [b] a1 ∧
                ( (a1 = [] ∧ fwire = w1 ∧ awire = a1)
                ∨ (a1 ≠ [] ∧
                    ∃ fwire2 awire2,
                      fwire = w1 ++ fwire2 ∧ awire = a1 ++ awire2 ∧
                      abp_ack_run (negb b) ms' fwire2 awire2) ) ) )
    end.


  Theorem abp_ack_prefix : ∀ (b : bit) (ms : list M) (fwire : list Packet) (awire : list bit),
    abp_ack_run b ms fwire awire → prefix (deliver b fwire) ms.
  Proof.
    intros b ms.
    revert b.
    induction ms as [| m ms' IH]; intros b fwire awire H.
    - simpl in H. destruct H as [Hf Ha]. rewrite Hf. simpl. apply prefix_refl.
    - simpl in H.
      destruct H as [k [w1 [Hburst Hcase]]].
      destruct Hcase as [[Hd [Hf Ha]] | [a1 [Hd [Hack Hcase2]]]].
      +
        rewrite Hf. rewrite Hd. apply prefix_nil_l.
      + destruct Hcase2 as [[Ha1 [Hf Ha]] | [Ha1 [fwire2 [awire2 [Hf [Ha Hrun]]]]]].
        ×
          rewrite Hf. rewrite Hd. unfold prefix. ∃ ms'. reflexivity.
        ×
          rewrite Hf.
          rewrite (deliver_app w1 fwire2 b).
          rewrite Hd.
          assert (Hnext : next_bit b w1 = negb b).
          { destruct (burst_delivery b m k w1 b Hburst) as [[Hd0 Hn0] | [Hd0 Hn0]].
            - rewrite Hd0 in Hd. discriminate Hd.
            - exact Hn0. }
          rewrite Hnext. simpl.
          unfold prefix.
          destruct (IH (negb b) fwire2 awire2 Hrun) as [ms2 Heq2].
          ∃ ms2. rewrite Heq2. reflexivity.
  Qed.


  Fixpoint ack_witness (b : bit) (ms : list M) : list bit :=
    match ms with
    | [] ⇒ []
    | m :: ms' ⇒ b :: ack_witness (negb b) ms'
    end.

  Lemma abp_ack_run_witness : ∀ (b : bit) (ms : list M),
    abp_ack_run b ms (sender_stream b ms 1) (ack_witness b ms).
  Proof.
    intros b ms.
    revert b.
    induction ms as [| m ms' IH]; intros b.
    - simpl. split; reflexivity.
    - simpl.
      ∃ 1, (sender_burst b m 1).
      split.
      + apply lossy_dup_refl.
      + right.
        ∃ [b].
        split.
        × simpl. rewrite (eqb_bit_true b b eq_refl). reflexivity.
        × split.
          -- apply lossy_dup_refl.
          -- right.
             split.
             ++ discriminate.
             ++ ∃ (sender_stream (negb b) ms' 1), (ack_witness (negb b) ms').
                split.
                ** reflexivity.
                ** split.
                   --- reflexivity.
                   --- apply IH.
  Qed.

  Corollary abp_ack_run_delivers_all : ∀ (b : bit) (ms : list M),
    ∃ fwire awire, abp_ack_run b ms fwire awire ∧ deliver b fwire = ms.
  Proof.
    intros b ms.
    ∃ (sender_stream b ms 1), (ack_witness b ms).
    split.
    - apply abp_ack_run_witness.
    - apply (abp_collapse_perfect_gen 1 ms b). lia.
  Qed.

End Ack.
