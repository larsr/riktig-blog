From Riktig Require Import Ops Props Interface.GeometricAlgebra
  Interface.InnerProduct Theory.Group Theory.AbGroup Theory.GeometricAlgebra.

Section ExteriorTheory.
  Context {F V A : Type} `{GeometricAlgebra F V A}.

  Definition wedge (u v : V) : A :=
    ga_vec u × ga_vec v + - (ga_vec v × ga_vec u).

  Theorem wedge_self_zero : ∀ u : V, wedge u u == 0.
  Proof.
    intros u.
    exact (invr (ga_vec u × ga_vec u)).
  Qed.

  Theorem wedge_anti : ∀ u v : V, wedge u v == - wedge v u.
  Proof.
    intros u v.
    unfold wedge.
    rewrite (neg_add (ga_vec v × ga_vec u) (- (ga_vec u × ga_vec v))),
      (neg_involutive (ga_vec u × ga_vec v)).
    apply comm.
  Qed.

  Theorem ga_symmetric_part : ∀ u v : V,
    ga_vec u × ga_vec v + ga_vec v × ga_vec u
    == (⟨ u , v ⟩ + ⟨ u , v ⟩) *: (1 : A).
  Proof.
    intros u v.
    rewrite (polarization u v), (ip_sym v u).
    reflexivity.
  Qed.

  Theorem ga_decompose : ∀ u v : V,
    ga_vec u × ga_vec v + ga_vec u × ga_vec v
    == (⟨ u , v ⟩ + ⟨ u , v ⟩) *: (1 : A) + wedge u v.
  Proof.
    intros u v.
    rewrite <- (ga_symmetric_part u v).
    unfold wedge.
    rewrite (add_interchange (ga_vec u × ga_vec v) (ga_vec v × ga_vec u)
      (ga_vec u × ga_vec v) (- (ga_vec v × ga_vec u))),
      (invr (ga_vec v × ga_vec u)), idr.
    reflexivity.
  Qed.

End ExteriorTheory.
