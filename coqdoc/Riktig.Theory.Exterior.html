Section ExteriorTheory.
  Context {F V A : Type} `{GeometricAlgebra F V A}.

  Definition wedge (u v : V) : A :=
    ga_vec u × ga_vec v + - (ga_vec v × ga_vec u).

  Theorem wedge_self_zero : ∀ u : V, wedge u u == 0.
  Proof.
    intros u.
    exact (invr (ga_vec u × ga_vec u)).
  Qed.

  Theorem wedge_anti : ∀ u v : V, wedge u v == - wedge v u.
  Proof.
    intros u v.
    unfold wedge.
    rewrite (neg_add (ga_vec v × ga_vec u) (- (ga_vec u × ga_vec v))),
      (neg_involutive (ga_vec u × ga_vec v)).
    apply comm.
  Qed.

  Theorem ga_symmetric_part : ∀ u v : V,
    ga_vec u × ga_vec v + ga_vec v × ga_vec u
    == (⟨ u , v ⟩ + ⟨ u , v ⟩) *: (1 : A).
  Proof.
    intros u v.
    rewrite (polarization u v), (ip_sym v u).
    reflexivity.
  Qed.

  Theorem ga_decompose : ∀ u v : V,
    ga_vec u × ga_vec v + ga_vec u × ga_vec v
    == (⟨ u , v ⟩ + ⟨ u , v ⟩) *: (1 : A) + wedge u v.
  Proof.
    intros u v.
    rewrite <- (ga_symmetric_part u v).
    unfold wedge.
    rewrite (add_interchange (ga_vec u × ga_vec v) (ga_vec v × ga_vec u)
      (ga_vec u × ga_vec v) (- (ga_vec v × ga_vec u))),
      (invr (ga_vec v × ga_vec u)), idr.
    reflexivity.
  Qed.

End ExteriorTheory.


Section VolumeElement.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V} {p : nat}.
  Context (B : Basis (S p) F V).

  Lemma mat_of_fam_lm : ∀ (f : LinearMap F V V) (vs : Fin.t (S p) → V)
      (i j : Fin.t (S p)),
    mat_of_fam B (fun k ⇒ f (vs k)) i j
      == mat_mul (mat_of B B f) (mat_of_fam B vs) i j.
  Proof.
    intros f vs i j.
    exact (mat_of_coords B B f (vs j) i).
  Qed.

  Section VolumeElementConditional.
    Context (detM_mul : ∀ (m : nat) (A A' : Matrix m m F),
      detM (mat_mul A A') == detM A × detM A').

    Theorem omega_B_lm : ∀ (f : LinearMap F V V) (vs : Fin.t (S p) → V),
      omega_B B (fun k ⇒ f (vs k)) == det B f × omega_B B vs.
    Proof.
      intros f vs.
      calc (omega_B B (fun k ⇒ f (vs k)))
        == (detM (mat_of_fam B (fun k ⇒ f (vs k))))
        == (detM (mat_mul (mat_of B B f) (mat_of_fam B vs)))
        == (detM (mat_of B B f) × detM (mat_of_fam B vs))
        == (det B f × omega_B B vs).
      - reflexivity.
      - exact (detM_proper (S p) (mat_of_fam B (fun k ⇒ f (vs k)))
                 (mat_mul (mat_of B B f) (mat_of_fam B vs))
                 (fun i j ⇒ mat_of_fam_lm f vs i j)).
      - exact (detM_mul (S p) (mat_of B B f) (mat_of_fam B vs)).
      - reflexivity.
    Qed.

  End VolumeElementConditional.

  Theorem omega_B_lm_id : ∀ vs : Fin.t (S p) → V,
    omega_B B (fun k ⇒ (@lm_id F V FF VV) (vs k)) == omega_B B vs.
  Proof.
    intros vs.
    apply (omega_B_proper B).
    intros k.
    reflexivity.
  Qed.

End VolumeElement.

Corollary omega_B_lm_Q : ∀ (V : Type) (VV : VectorSpace Q V) (p : nat)
    (B : Basis (S p) Q V) (f : LinearMap Q V V) (vs : Fin.t (S p) → V),
  omega_B B (fun k ⇒ f (vs k)) == det B f × omega_B B vs.
Proof.
  intros V VV p B f vs.
  exact (omega_B_lm B (fun m A A' ⇒ detM_mul_Q m A A') f vs).
Qed.
