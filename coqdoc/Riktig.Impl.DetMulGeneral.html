Section CountNonfixed.
  Variable n : nat.

  Definition nonfixed (f : Fin.t n → Fin.t n) : list (Fin.t n) :=
    filter (fun i ⇒ if Fin.eq_dec (f i) i then false else true) (fin_all n).

  Definition count_nonfixed (f : Fin.t n → Fin.t n) : nat := length (nonfixed f).

  Lemma count_nonfixed_zero_id : ∀ f : Fin.t n → Fin.t n,
    count_nonfixed f = 0%nat → ∀ i, f i = i.
  Proof.
    intros f Hc i.
    unfold count_nonfixed, nonfixed in Hc.
    apply length_zero_iff_nil in Hc.
    destruct (Fin.eq_dec (f i) i) as [Heq | Hne].
    - exact Heq.
    - exfalso.
      assert (Hin : In i (filter (fun i0 ⇒ if Fin.eq_dec (f i0) i0 then false else true) (fin_all n))).
      { apply filter_In. split.
        - exact (fin_all_complete n i).
        - destruct (Fin.eq_dec (f i) i) as [Heq2 | Hne2].
          + destruct (Hne Heq2).
          + reflexivity. }
      rewrite Hc in Hin. exact Hin.
  Qed.

End CountNonfixed.

Section SwapDecrease.
  Variable n : nat.

  Lemma nonfixed_incl : ∀ (f f' : Fin.t n → Fin.t n) (p b : Fin.t n),
    p ≠ b → f p ≠ p → f' b = b →
    (∀ i, i ≠ p → i ≠ b → f' i = f i) →
    incl (nonfixed n f') (nonfixed n f).
  Proof.
    intros f f' p b Hpb Hfp Hf'b Hrest i Hin.
    apply filter_In in Hin. destruct Hin as [HinAll Htest].
    destruct (Fin.eq_dec (f' i) i) as [Heq | Hne]; [discriminate Htest |].
    apply filter_In. split; [exact HinAll |].
    destruct (Fin.eq_dec i p) as [Hip | Hip].
    - subst i. destruct (Fin.eq_dec (f p) p) as [Hc | Hc]; [destruct (Hfp Hc) | reflexivity].
    - destruct (Fin.eq_dec i b) as [Hib | Hib].
      + subst i. destruct (Hne Hf'b).
      + rewrite (Hrest i Hip Hib) in Hne.
        destruct (Fin.eq_dec (f i) i) as [Hc | Hc]; [destruct (Hne Hc) | reflexivity].
  Qed.

  Theorem count_nonfixed_swap_decrease : ∀ (f f' : Fin.t n → Fin.t n) (p b : Fin.t n),
    p ≠ b → f p ≠ p → f b ≠ b → f' b = b →
    (∀ i, i ≠ p → i ≠ b → f' i = f i) →
    (count_nonfixed n f' < count_nonfixed n f)%nat.
  Proof.
    intros f f' p b Hpb Hfp Hfb Hf'b Hrest.
    unfold count_nonfixed.
    assert (Hincl := nonfixed_incl f f' p b Hpb Hfp Hf'b Hrest).
    assert (HbIn : In b (nonfixed n f)).
    { apply filter_In. split; [exact (fin_all_complete n b) |].
      destruct (Fin.eq_dec (f b) b) as [Hc | Hc]; [destruct (Hfb Hc) | reflexivity]. }
    assert (HbNotIn : ¬ In b (nonfixed n f')).
    { intros Hin. apply filter_In in Hin. destruct Hin as [_ Htest].
      destruct (Fin.eq_dec (f' b) b) as [Hc | Hc]; [discriminate Htest | exact (Hc Hf'b)]. }
    assert (HndF' : NoDup (nonfixed n f')).
    { apply NoDup_filter. exact (fin_all_NoDup n). }
    assert (Hincl2 : incl (nonfixed n f') (remove Fin.eq_dec b (nonfixed n f))).
    { intros x Hx.
      apply in_in_remove.
      - intros Hxb. subst x. exact (HbNotIn Hx).
      - exact (Hincl x Hx). }
    assert (Hle := NoDup_incl_length HndF' Hincl2).
    assert (Hlt := remove_length_lt Fin.eq_dec (nonfixed n f) b HbIn).
    lia.
  Qed.

End SwapDecrease.

  Lemma upd_row_comm : ∀ (n:nat) (A:Matrix n n F) (i j:Fin.t n) (r1 r2:Fin.t n→F),
    i ≠ j → upd_row (upd_row A i r1) j r2 == upd_row (upd_row A j r2) i r1.
  Proof.
    intros n A i j r1 r2 Hij r' c.
    unfold upd_row.
    destruct (dec (r' = j)) as [Hj | Hj]; destruct (dec (r' = i)) as [Hi | Hi].
    - exfalso. apply Hij. rewrite <- Hi, Hj. reflexivity.
    - reflexivity.
    - reflexivity.
    - reflexivity.
  Qed.

  Lemma upd_row_row_swap_eq : ∀ (n:nat)(A:Matrix n n F)(i j:Fin.t n), i≠j →
    upd_row (upd_row A i (A j)) j (A i) == row_swap A i j.
  Proof.
    intros n A i j Hij r c.
    unfold upd_row, row_swap.
    destruct (dec (r = j)) as [Hj | Hj]; destruct (dec (r = i)) as [Hi | Hi].
    - exfalso. apply Hij. rewrite <- Hi, Hj. reflexivity.
    - reflexivity.
    - reflexivity.
    - reflexivity.
  Qed.

  Lemma upd_row_change_pointwise : ∀ (n:nat)(A:Matrix n n F)(i:Fin.t n)(r1 r2:Fin.t n→F),
    (∀ k, r1 k == r2 k) → upd_row A i r1 == upd_row A i r2.
  Proof.
    intros n A i r1 r2 Heq r' c.
    unfold upd_row.
    destruct (dec (r' = i)) as [Hi | Hi].
    - exact (Heq c).
    - reflexivity.
  Qed.

  Theorem g_swap_antisym : ∀ (n:nat) (g : Matrix n n F → F),
    Proper (equ ==> equ) g →
    (∀ (i:Fin.t n)(A:Matrix n n F)(r1 r2:Fin.t n→F),
       g (upd_row A i (fun k⇒r1 k+r2 k)) == g(upd_row A i r1)+g(upd_row A i r2)) →
    (∀ (A:Matrix n n F)(i j:Fin.t n), i≠j → (∀ c, A i c==A j c) → g A == 0) →
    ∀ (A:Matrix n n F)(i j:Fin.t n), i≠j → g (row_swap A i j) == - g A.
  Proof.
    intros n g Hproper Hadd Halt A i j Hij.
    assert (Hji : j ≠ i) by (intros Heq; apply Hij; exact (eq_sym Heq)).
    set (u := A i). set (v := A j).
    set (N1 := upd_row A j (fun k⇒u k+v k)).
    set (M1 := upd_row N1 i (fun k⇒u k+v k)).
    assert (HM1i : ∀ c, M1 i c == u c + v c).
    { intros c. exact (upd_row_at n N1 i (fun k⇒u k+v k) c). }
    assert (HM1j : ∀ c, M1 j c == u c + v c).
    { intros c.
      transitivity (N1 j c).
      - exact (upd_row_ne n N1 i j (fun k⇒u k+v k) c Hji).
      - exact (upd_row_at n A j (fun k⇒u k+v k) c). }
    assert (HM1zero : g M1 == 0).
    { apply (Halt M1 i j Hij). intros c. rewrite (HM1i c), (HM1j c). reflexivity. }
    
    assert (HM1exp : g M1 == g (upd_row N1 i u) + g (upd_row N1 i v)).
    { exact (Hadd i N1 u v). }
    
    assert (Hleft : g (upd_row N1 i u) == g A).
    { assert (HN1i : ∀ c, N1 i c == u c).
      { intros c. exact (upd_row_ne n A j i (fun k⇒u k+v k) c Hij). }
      assert (Heq1 : upd_row N1 i u == upd_row N1 i (N1 i)).
      { apply upd_row_change_pointwise. intros k. symmetry. exact (HN1i k). }
      assert (Heq2 : upd_row N1 i (N1 i) == N1).
      { exact (upd_row_self n N1 i). }
      rewrite Heq1, Heq2.
      unfold N1.
      transitivity (g (upd_row A j u) + g (upd_row A j v)).
      - exact (Hadd j A u v).
      - assert (H1 : g (upd_row A j u) == 0).
        { apply (Halt (upd_row A j u) i j Hij). intros c.
          transitivity (A i c).
          - exact (upd_row_ne n A j i u c Hij).
          - transitivity (u c); [reflexivity |].
            symmetry. exact (upd_row_at n A j u c). }
        assert (H2 : g (upd_row A j v) == g A).
        { assert (HAjv : upd_row A j v == A).
          { assert (Hveq : ∀ c, v c == A j c) by (intros c; reflexivity).
            transitivity (upd_row A j (A j)).
            - apply upd_row_change_pointwise. exact Hveq.
            - exact (upd_row_self n A j). }
          apply Hproper. exact HAjv. }
        rewrite H1, H2.
        exact (monoid_left_identity (g A)). }
    
    assert (Hright : g (upd_row N1 i v) == g (row_swap A i j)).
    { assert (Hcomm : upd_row N1 i v == upd_row (upd_row A i v) j (fun k⇒u k+v k)).
      { unfold N1. exact (upd_row_comm n A j i (fun k⇒u k+v k) v Hji). }
      rewrite Hcomm.
      set (A2 := upd_row A i v).
      transitivity (g (upd_row A2 j u) + g (upd_row A2 j v)).
      - exact (Hadd j A2 u v).
      - assert (H1 : g (upd_row A2 j u) == g (row_swap A i j)).
        { assert (Heq3 : upd_row A2 j u == row_swap A i j).
          { unfold A2. exact (upd_row_row_swap_eq n A i j Hij). }
          apply Hproper. exact Heq3. }
        assert (H2 : g (upd_row A2 j v) == 0).
        { apply (Halt (upd_row A2 j v) i j Hij). intros c.
          transitivity (v c).
          - transitivity (A2 i c).
            + exact (upd_row_ne n A2 j i v c Hij).
            + exact (upd_row_at n A i v c).
          - symmetry. exact (upd_row_at n A2 j v c). }
        rewrite H1, H2.
        exact (monoid_right_identity (g (row_swap A i j))). }
    assert (HM1full : g M1 == g A + g (row_swap A i j)).
    { rewrite HM1exp, Hleft, Hright. reflexivity. }
    assert (Hsum : g A + g (row_swap A i j) == 0).
    { transitivity (g M1); [symmetry; exact HM1full | exact HM1zero]. }
    exact (neg_unique_r (g A) (g (row_swap A i j)) Hsum).
  Qed.


  Definition upd_rowP {p n : nat} (A : Matrix p n F) (i : Fin.t p)
      (r : Fin.t n → F) : Matrix p n F :=
    fun i' j ⇒ if dec (i' = i) then r j else A i' j.

  Lemma upd_rowP_at : ∀ (p n:nat)(A:Matrix p n F)(i:Fin.t p)(r:Fin.t n→F)(j:Fin.t n),
    upd_rowP A i r i j == r j.
  Proof.
    intros p n A i r j. unfold upd_rowP.
    destruct (dec (i = i)) as [_ | Hne]; [reflexivity | destruct (Hne eq_refl)].
  Qed.

  Lemma upd_rowP_ne : ∀ (p n:nat)(A:Matrix p n F)(i i':Fin.t p)(r:Fin.t n→F)(j:Fin.t n),
    i' ≠ i → upd_rowP A i r i' j == A i' j.
  Proof.
    intros p n A i i' r j Hne. unfold upd_rowP.
    destruct (dec (i' = i)) as [Heq | _]; [destruct (Hne Heq) | reflexivity].
  Qed.

  Lemma upd_rowP_self : ∀ (p n:nat)(A:Matrix p n F)(i:Fin.t p),
    upd_rowP A i (A i) == A.
  Proof.
    intros p n A i i' j. unfold upd_rowP.
    destruct (dec (i' = i)) as [Heq | _]; [rewrite Heq; reflexivity | reflexivity].
  Qed.

  Lemma upd_rowP_change_pointwise : ∀ (p n:nat)(A:Matrix p n F)(i:Fin.t p)(r1 r2:Fin.t n→F),
    (∀ k, r1 k == r2 k) → upd_rowP A i r1 == upd_rowP A i r2.
  Proof.
    intros p n A i r1 r2 Heq i' j. unfold upd_rowP.
    destruct (dec (i' = i)) as [_ | _]; [exact (Heq j) | reflexivity].
  Qed.

  Definition cons_rowP {p n : nat} (r0 : Fin.t n → F) (C : Matrix p n F)
      : Matrix (S p) n F :=
    fun i' j ⇒ Fin.caseS' i' (fun _ ⇒ F) (r0 j) (fun s ⇒ C s j).

  Lemma cons_rowP_upd_FS : ∀ (p n:nat)(r0:Fin.t n→F)(C:Matrix p n F)(s:Fin.t p)(r:Fin.t n→F),
    cons_rowP r0 (upd_rowP C s r) == upd_rowP (cons_rowP r0 C) (Fin.FS s) r.
  Proof.
    intros p n r0 C s r i' j.
    apply (Fin.caseS' i' (fun i0 ⇒
      cons_rowP r0 (upd_rowP C s r) i0 j == upd_rowP (cons_rowP r0 C) (Fin.FS s) r i0 j)).
    - transitivity (r0 j).
      + reflexivity.
      + symmetry.
        exact (upd_rowP_ne (S p) n (cons_rowP r0 C) (Fin.FS s) Fin.F1 r j (fin_F1_neq_FS p s)).
    - intros s'.
      destruct (dec (s' = s)) as [Heq | Hne].
      + transitivity (r j).
        × transitivity (upd_rowP C s r s' j).
          -- reflexivity.
          -- rewrite Heq. exact (upd_rowP_at p n C s r j).
        × symmetry. rewrite Heq. exact (upd_rowP_at (S p) n (cons_rowP r0 C) (Fin.FS s) r j).
      + transitivity (C s' j).
        × transitivity (upd_rowP C s r s' j).
          -- reflexivity.
          -- exact (upd_rowP_ne p n C s s' r j Hne).
        × symmetry.
          transitivity (cons_rowP r0 C (Fin.FS s') j).
          -- exact (upd_rowP_ne (S p) n (cons_rowP r0 C) (Fin.FS s) (Fin.FS s') r j
                      (fun Heq ⇒ Hne (Fin.FS_inj s' s Heq))).
          -- reflexivity.
  Qed.

  Definition fun_cons {n : nat} {p : nat} (k : Fin.t n) (f : Fin.t p → Fin.t n)
      : Fin.t (S p) → Fin.t n :=
    fun i ⇒ Fin.caseS' i (fun _ ⇒ Fin.t n) k f.

  Definition basis_rowsP {p n : nat} (f : Fin.t p → Fin.t n) : Matrix p n F :=
    fun i j ⇒ e (f i) j.

  Lemma cons_rowP_basis : ∀ (n p:nat)(k:Fin.t n)(f':Fin.t p→Fin.t n),
    cons_rowP (e k) (basis_rowsP f') == basis_rowsP (fun_cons k f').
  Proof.
    intros n p k f' i j.
    apply (Fin.caseS' i (fun i0 ⇒
      cons_rowP (e k) (basis_rowsP f') i0 j == basis_rowsP (fun_cons k f') i0 j)).
    - reflexivity.
    - intros s. reflexivity.
  Qed.

  Fixpoint fsum (n p : nat) (h : (Fin.t p → Fin.t n) → F) {struct p} : F :=
    match p as p0 return ((Fin.t p0 → Fin.t n) → F) → F with
    | O ⇒ fun h0 ⇒ h0 (fun i ⇒ Fin.case0 (fun _ ⇒ Fin.t n) i)
    | S q ⇒ fun h0 ⇒ bigopF (fun k : Fin.t n ⇒
        fsum n q (fun f' : Fin.t q → Fin.t n ⇒ h0 (fun_cons k f')))
    end h.

  Lemma fsum_O : ∀ (n:nat)(h : (Fin.t 0 → Fin.t n) → F),
    fsum n 0 h == h (fun i ⇒ Fin.case0 (fun _ ⇒ Fin.t n) i).
  Proof. intros n h. reflexivity. Qed.

  Lemma fsum_S : ∀ (n q:nat)(h : (Fin.t (S q) → Fin.t n) → F),
    fsum n (S q) h == bigopF (fun k : Fin.t n ⇒
      fsum n q (fun f' : Fin.t q → Fin.t n ⇒ h (fun_cons k f'))).
  Proof. intros n q h. reflexivity. Qed.

  Lemma fsum_ext : ∀ (n p:nat)(h1 h2 : (Fin.t p → Fin.t n) → F),
    (∀ f, h1 f == h2 f) → fsum n p h1 == fsum n p h2.
  Proof.
    intros n p; induction p as [| q IH]; intros h1 h2 Heq.
    - exact (Heq _).
    - rewrite (fsum_S n q h1), (fsum_S n q h2).
      apply bigopF_ext. intros k. apply IH. intros f'. exact (Heq _).
  Qed.

  Lemma fsum_scale : ∀ (n p:nat)(c:F)(h : (Fin.t p → Fin.t n) → F),
    fsum n p (fun f ⇒ c × h f) == c × fsum n p h.
  Proof.
    intros n p; induction p as [| q IH]; intros c h.
    - reflexivity.
    - calc (fsum n (S q) (fun f ⇒ c × h f))
        == (bigopF (fun k : Fin.t n ⇒ fsum n q (fun f' ⇒ c × h (fun_cons k f'))))
        == (bigopF (fun k : Fin.t n ⇒ c × fsum n q (fun f' ⇒ h (fun_cons k f'))))
        == (c × bigopF (fun k : Fin.t n ⇒ fsum n q (fun f' ⇒ h (fun_cons k f'))))
        == (c × fsum n (S q) h).
      + exact (fsum_S n q (fun f ⇒ c × h f)).
      + apply bigopF_ext. intros k. exact (IH c (fun f' ⇒ h (fun_cons k f'))).
      + exact (bigopF_mul_l n c (fun k ⇒ fsum n q (fun f' ⇒ h (fun_cons k f')))).
      + apply ring_mul_proper; [reflexivity |]. symmetry. exact (fsum_S n q h).
  Qed.

  Lemma g_zero_row : ∀ (p n:nat)(g:Matrix p n F → F),
    Proper (equ ==> equ) g →
    (∀ (i:Fin.t p)(A:Matrix p n F)(c:F)(r0:Fin.t n→F),
       g (upd_rowP A i (fun k ⇒ c × r0 k)) == c × g (upd_rowP A i r0)) →
    ∀ (i:Fin.t p)(A:Matrix p n F), g (upd_rowP A i (fun _ ⇒ 0)) == 0.
  Proof.
    intros p n g Hproper Hscale i A.
    calc (g (upd_rowP A i (fun _ ⇒ 0)))
      == (g (upd_rowP A i (fun k ⇒ (0:F) × A i k)))
      == ((0:F) × g (upd_rowP A i (A i)))
      == (0:F).
    - apply Hproper. apply upd_rowP_change_pointwise. intros k. symmetry. exact (mul_0_l (A i k)).
    - exact (Hscale i A 0 (A i)).
    - exact (mul_0_l (g (upd_rowP A i (A i)))).
  Qed.

  Theorem g_row_lincomb : ∀ (p n : nat) (g : Matrix p n F → F),
    Proper (equ ==> equ) g →
    (∀ (i:Fin.t p)(A:Matrix p n F)(r1 r2:Fin.t n→F),
       g (upd_rowP A i (fun k⇒r1 k+r2 k)) == g(upd_rowP A i r1)+g(upd_rowP A i r2)) →
    (∀ (i:Fin.t p)(A:Matrix p n F)(c:F)(r0:Fin.t n→F),
       g (upd_rowP A i (fun k⇒c×r0 k)) == c×g(upd_rowP A i r0)) →
    ∀ (i:Fin.t p)(A:Matrix p n F)(m:nat)(idx:Fin.t m → Fin.t n)(c:Fin.t m → F),
    g (upd_rowP A i (fun j ⇒ bigopF (fun k ⇒ c k × e (idx k) j)))
      == bigopF (fun k ⇒ c k × g (upd_rowP A i (e (idx k)))).
  Proof.
    intros p n g Hproper Hadd Hscale i A m.
    induction m as [| m' IH]; intros idx c.
    - calc (g (upd_rowP A i (fun j ⇒ bigopF (fun k : Fin.t 0 ⇒ c k × e (idx k) j))))
        == (g (upd_rowP A i (fun _ : Fin.t n ⇒ (0:F))))
        == (0:F)
        == (bigopF (fun k : Fin.t 0 ⇒ c k × g (upd_rowP A i (e (idx k))))).
      + apply Hproper. apply upd_rowP_change_pointwise. intros j.
        exact (bigopF_0 (fun k : Fin.t 0 ⇒ c k × e (idx k) j)).
      + exact (g_zero_row p n g Hproper Hscale i A).
      + symmetry. exact (bigopF_0 (fun k : Fin.t 0 ⇒ c k × g (upd_rowP A i (e (idx k))))).
    - calc (g (upd_rowP A i (fun j ⇒ bigopF (fun k ⇒ c k × e (idx k) j))))
        == (g (upd_rowP A i (fun j ⇒
              c Fin.F1 × e (idx Fin.F1) j
              + bigopF (fun k' : Fin.t m' ⇒ c (Fin.FS k') × e (idx (Fin.FS k')) j))))
        == (g (upd_rowP A i (fun j ⇒ c Fin.F1 × e (idx Fin.F1) j))
            + g (upd_rowP A i (fun j ⇒
                  bigopF (fun k' : Fin.t m' ⇒ c (Fin.FS k') × e (idx (Fin.FS k')) j))))
        == (c Fin.F1 × g (upd_rowP A i (e (idx Fin.F1)))
            + bigopF (fun k' : Fin.t m' ⇒
                c (Fin.FS k') × g (upd_rowP A i (e (idx (Fin.FS k'))))))
        == (bigopF (fun k ⇒ c k × g (upd_rowP A i (e (idx k))))).
      + apply Hproper. apply upd_rowP_change_pointwise. intros j.
        exact (bigopF_S m' (fun k ⇒ c k × e (idx k) j)).
      + exact (Hadd i A (fun j ⇒ c Fin.F1 × e (idx Fin.F1) j)
                 (fun j ⇒ bigopF (fun k' : Fin.t m' ⇒ c (Fin.FS k') × e (idx (Fin.FS k')) j))).
      + apply monoid_add_proper.
        × exact (Hscale i A (c Fin.F1) (e (idx Fin.F1))).
        × exact (IH (fun k' ⇒ idx (Fin.FS k')) (fun k' ⇒ c (Fin.FS k'))).
      + symmetry. exact (bigopF_S m' (fun k ⇒ c k × g (upd_rowP A i (e (idx k))))).
  Qed.

  Theorem row_expand : ∀ (n p : nat) (g : Matrix p n F → F),
    Proper (equ ==> equ) g →
    (∀ (i:Fin.t p)(A:Matrix p n F)(r1 r2:Fin.t n→F),
       g (upd_rowP A i (fun k⇒r1 k+r2 k)) == g(upd_rowP A i r1)+g(upd_rowP A i r2)) →
    (∀ (i:Fin.t p)(A:Matrix p n F)(c:F)(r0:Fin.t n→F),
       g (upd_rowP A i (fun k⇒c×r0 k)) == c×g(upd_rowP A i r0)) →
    ∀ A : Matrix p n F,
    g A == fsum n p (fun f : Fin.t p → Fin.t n ⇒
      prodF (fun i : Fin.t p ⇒ A i (f i)) × g (basis_rowsP f)).
  Proof.
    intros n p; induction p as [| q IH]; intros g Hproper Hadd Hscale A.
    - transitivity (g (basis_rowsP (fun i : Fin.t 0 ⇒ Fin.case0 (fun _ ⇒ Fin.t n) i))).
      + apply Hproper. intros i j. exact (Fin.case0 (fun i0 ⇒ A i0 j == basis_rowsP _ i0 j) i).
      + rewrite (fsum_O n (fun f : Fin.t 0 → Fin.t n ⇒
          prodF (fun i : Fin.t 0 ⇒ A i (f i)) × g (basis_rowsP f))).
        transitivity ((1:F) × g (basis_rowsP (fun i : Fin.t 0 ⇒ Fin.case0 (fun _ ⇒ Fin.t n) i))).
        × symmetry. exact (ring_mul_1l _).
        × apply ring_mul_proper; [symmetry; exact (prodF_0 _) | reflexivity].
    - assert (HAeq : A == upd_rowP A Fin.F1 (fun j ⇒ bigopF (fun k : Fin.t n ⇒ A Fin.F1 k × e k j))).
      { intros i' j.
        apply (Fin.caseS' i' (fun i0 ⇒
          A i0 j == upd_rowP A Fin.F1 (fun j0 ⇒ bigopF (fun k : Fin.t n ⇒ A Fin.F1 k × e k j0)) i0 j)).
        - transitivity (bigopF (fun k : Fin.t n ⇒ A Fin.F1 k × e k j)).
          + symmetry. exact (lincomb_std_delta n (A Fin.F1) j).
          + symmetry. exact (upd_rowP_at (S q) n A Fin.F1 _ j).
        - intros s.
          symmetry.
          exact (upd_rowP_ne (S q) n A Fin.F1 (Fin.FS s) _ j (fin_FS_neq_F1 q s)). }
      rewrite (Hproper A _ HAeq).
      rewrite (g_row_lincomb (S q) n g Hproper Hadd Hscale Fin.F1 A n (fun k ⇒ k) (A Fin.F1)).
      transitivity (bigopF (fun k : Fin.t n ⇒ A Fin.F1 k ×
        fsum n q (fun f' : Fin.t q → Fin.t n ⇒
          prodF (fun s : Fin.t q ⇒ A (Fin.FS s) (f' s)) × g (basis_rowsP (fun_cons k f'))))).
      + apply bigopF_ext. intros k.
        apply ring_mul_proper; [reflexivity |].
        set (g_k := fun C : Matrix q n F ⇒ g (cons_rowP (e k) C)).
        assert (Hgk_proper : Proper (equ ==> equ) g_k).
        { intros C C' HCC'. unfold g_k. apply Hproper. intros i' j.
          apply (Fin.caseS' i' (fun i0 ⇒ cons_rowP (e k) C i0 j == cons_rowP (e k) C' i0 j)).
          - reflexivity.
          - intros s. exact (HCC' s j). }
        assert (Hgk_add : ∀ (s:Fin.t q)(C:Matrix q n F)(r1 r2:Fin.t n→F),
          g_k (upd_rowP C s (fun k0⇒r1 k0+r2 k0)) == g_k (upd_rowP C s r1) + g_k (upd_rowP C s r2)).
        { intros s C r1 r2. unfold g_k.
          rewrite (Hproper (cons_rowP (e k) (upd_rowP C s (fun k0 ⇒ r1 k0 + r2 k0)))
                     (upd_rowP (cons_rowP (e k) C) (Fin.FS s) (fun k0 ⇒ r1 k0 + r2 k0))
                     (cons_rowP_upd_FS q n (e k) C s (fun k0 ⇒ r1 k0 + r2 k0))).
          rewrite (Hproper (cons_rowP (e k) (upd_rowP C s r1))
                     (upd_rowP (cons_rowP (e k) C) (Fin.FS s) r1)
                     (cons_rowP_upd_FS q n (e k) C s r1)).
          rewrite (Hproper (cons_rowP (e k) (upd_rowP C s r2))
                     (upd_rowP (cons_rowP (e k) C) (Fin.FS s) r2)
                     (cons_rowP_upd_FS q n (e k) C s r2)).
          exact (Hadd (Fin.FS s) (cons_rowP (e k) C) r1 r2). }
        assert (Hgk_scale : ∀ (s:Fin.t q)(C:Matrix q n F)(c0:F)(r0:Fin.t n→F),
          g_k (upd_rowP C s (fun k0⇒c0×r0 k0)) == c0 × g_k (upd_rowP C s r0)).
        { intros s C c0 r0. unfold g_k.
          rewrite (Hproper (cons_rowP (e k) (upd_rowP C s (fun k0 ⇒ c0 × r0 k0)))
                     (upd_rowP (cons_rowP (e k) C) (Fin.FS s) (fun k0 ⇒ c0 × r0 k0))
                     (cons_rowP_upd_FS q n (e k) C s (fun k0 ⇒ c0 × r0 k0))).
          rewrite (Hproper (cons_rowP (e k) (upd_rowP C s r0))
                     (upd_rowP (cons_rowP (e k) C) (Fin.FS s) r0)
                     (cons_rowP_upd_FS q n (e k) C s r0)).
          exact (Hscale (Fin.FS s) (cons_rowP (e k) C) c0 r0). }
        assert (Hkey : g_k (fun s ⇒ A (Fin.FS s)) ==
          fsum n q (fun f' : Fin.t q → Fin.t n ⇒
            prodF (fun s : Fin.t q ⇒ A (Fin.FS s) (f' s)) × g_k (basis_rowsP f'))).
        { exact (IH g_k Hgk_proper Hgk_add Hgk_scale (fun s ⇒ A (Fin.FS s))). }
        transitivity (g_k (fun s ⇒ A (Fin.FS s))).
        × unfold g_k. apply Hproper. intros i' j.
          apply (Fin.caseS' i' (fun i0 ⇒
            upd_rowP A Fin.F1 (e k) i0 j == cons_rowP (e k) (fun s ⇒ A (Fin.FS s)) i0 j)).
          -- transitivity (e k j).
             ++ exact (upd_rowP_at (S q) n A Fin.F1 (e k) j).
             ++ reflexivity.
          -- intros s.
             transitivity (A (Fin.FS s) j).
             ++ exact (upd_rowP_ne (S q) n A Fin.F1 (Fin.FS s) (e k) j (fin_FS_neq_F1 q s)).
             ++ reflexivity.
        × rewrite Hkey.
          apply fsum_ext. intros f'.
          apply ring_mul_proper; [reflexivity |].
          unfold g_k. apply Hproper. exact (cons_rowP_basis n q k f').
      + rewrite (fsum_S n q (fun f : Fin.t (S q) → Fin.t n ⇒
          prodF (fun i : Fin.t (S q) ⇒ A i (f i)) × g (basis_rowsP f))).
        apply bigopF_ext. intros k.
        transitivity (fsum n q (fun f' : Fin.t q → Fin.t n ⇒
          A Fin.F1 k × (prodF (fun s : Fin.t q ⇒ A (Fin.FS s) (f' s)) × g (basis_rowsP (fun_cons k f'))))).
        × symmetry.
          exact (fsum_scale n q (A Fin.F1 k)
            (fun f' : Fin.t q → Fin.t n ⇒
              prodF (fun s : Fin.t q ⇒ A (Fin.FS s) (f' s)) × g (basis_rowsP (fun_cons k f')))).
        × apply fsum_ext. intros f'.
          transitivity ((A Fin.F1 k × prodF (fun s : Fin.t q ⇒ A (Fin.FS s) (f' s)))
                        × g (basis_rowsP (fun_cons k f'))).
          -- symmetry. exact (ring_mul_assoc (A Fin.F1 k)
               (prodF (fun s : Fin.t q ⇒ A (Fin.FS s) (f' s))) (g (basis_rowsP (fun_cons k f')))).
          -- apply ring_mul_proper; [| reflexivity].
             symmetry.
             transitivity (A Fin.F1 (fun_cons k f' Fin.F1) ×
               prodF (fun s : Fin.t q ⇒ A (Fin.FS s) (fun_cons k f' (Fin.FS s)))).
             ++ exact (prodF_S q (fun i : Fin.t (S q) ⇒ A i (fun_cons k f' i))).
             ++ reflexivity.
  Qed.


  Lemma basis_rows_row_swap_precompose : ∀ (N:nat)(f:Fin.t N → Fin.t N)(a b:Fin.t N),
    row_swap (basis_rowsP f) a b == basis_rowsP (fun x ⇒ f (swap_fin a b x)).
  Proof.
    intros N f a b r c.
    unfold row_swap, basis_rowsP, swap_fin.
    destruct (dec (r = a)) as [Hra | Hra].
    - reflexivity.
    - destruct (dec (r = b)) as [Hrb | Hrb]; reflexivity.
  Qed.

  Lemma basis_rows_repeat_zero : ∀ (N:nat)(g:Matrix N N F → F),
    (∀ A i j, i≠j → (∀ c, A i c==A j c) → g A==0) →
    ∀ (f:Fin.t N → Fin.t N)(i1 i2:Fin.t N), i1≠i2 → f i1 = f i2 →
    g (basis_rowsP f) == 0.
  Proof.
    intros N g Halt f i1 i2 Hne Heq.
    apply (Halt (basis_rowsP f) i1 i2 Hne).
    intros c. unfold basis_rowsP. rewrite Heq. reflexivity.
  Qed.

  Theorem basis_rows_g_eq_detM_bound : ∀ (p0 : nat) (g : Matrix (S p0) (S p0) F → F),
    Proper (equ ==> equ) g →
    (∀ (i:Fin.t (S p0))(A:Matrix (S p0) (S p0) F)(r1 r2:Fin.t (S p0)→F),
       g (upd_row A i (fun k⇒r1 k+r2 k)) == g(upd_row A i r1)+g(upd_row A i r2)) →
    (∀ (A:Matrix (S p0) (S p0) F)(i j:Fin.t (S p0)), i≠j → (∀ c, A i c==A j c) → g A == 0) →
    ∀ (m:nat)(f:Fin.t (S p0) → Fin.t (S p0)), (count_nonfixed (S p0) f ≤ m)%nat →
    g (basis_rowsP f) == g mat_id × detM (basis_rowsP f).
  Proof.
    intros p0 g Hproper Hadd Halt.
    induction m as [| m' IHm]; intros f Hle.
    - assert (Hid : ∀ i, f i = i).
      { apply count_nonfixed_zero_id. lia. }
      assert (Heq : basis_rowsP f == @mat_id F FF (S p0)).
      { intros i j. unfold basis_rowsP. rewrite (Hid i). reflexivity. }
      transitivity (g mat_id).
      + apply Hproper. exact Heq.
      + transitivity (g mat_id × 1).
        × symmetry. exact (ring_mul_1r (g mat_id)).
        × apply ring_mul_proper; [reflexivity |].
          transitivity (detM (@mat_id F FF (S p0))).
          -- symmetry. exact (det_id (S p0)).
          -- symmetry. apply detM_proper. exact Heq.
    - destruct (Nat.le_gt_cases (count_nonfixed (S p0) f) m') as [Hle' | Hgt].
      + exact (IHm f Hle').
      + assert (HcountSm' : count_nonfixed (S p0) f = S m') by lia.
        assert (Hnonempty : nonfixed (S p0) f ≠ nil).
        { intros Hnil. unfold count_nonfixed in HcountSm'. rewrite Hnil in HcountSm'.
          simpl in HcountSm'. discriminate HcountSm'. }
        destruct (nonfixed (S p0) f) as [| p rest] eqn:Hlist.
        × destruct (Hnonempty eq_refl).
        × assert (Hpin : In p (nonfixed (S p0) f)) by (rewrite Hlist; left; reflexivity).
          apply filter_In in Hpin. destruct Hpin as [_ Hptest].
          assert (Hfp : f p ≠ p).
          { destruct (Fin.eq_dec (f p) p) as [Hc | Hc]; [discriminate Hptest | exact Hc]. }
          set (b := f p).
          assert (Hpb : p ≠ b).
          { intros Heq. apply Hfp. unfold b in Heq. symmetry. exact Heq. }
          destruct (Fin.eq_dec (f b) b) as [Hfb_eq | Hfb_ne].
          --
             assert (Hrepeat : f p = f b) by (symmetry; exact Hfb_eq).
             assert (Hg0 : g (basis_rowsP f) == 0).
             { exact (basis_rows_repeat_zero (S p0) g Halt f p b Hpb Hrepeat). }
             assert (Hd0 : detM (basis_rowsP f) == 0).
             { apply (detM_alternating (S p0) (basis_rowsP f) p b Hpb).
               intros c. unfold basis_rowsP. rewrite Hrepeat. reflexivity. }
             rewrite Hg0, Hd0. symmetry. exact (mul_0_r (g mat_id)).
          --
             set (f' := fun x ⇒ f (swap_fin p b x)).
             assert (Hf'b : f' b = b).
             { unfold f'. rewrite (swap_fin_at_j (S p0) p b Hpb). exact eq_refl. }
             assert (Hrest : ∀ i, i ≠ p → i ≠ b → f' i = f i).
             { intros i Hip Hib. unfold f'. rewrite (swap_fin_ne (S p0) p b i Hip Hib). reflexivity. }
             assert (Hdecr := count_nonfixed_swap_decrease (S p0) f f' p b Hpb Hfp Hfb_ne Hf'b Hrest).
             assert (Hle'' : (count_nonfixed (S p0) f' ≤ m')%nat) by lia.
             assert (Hswap_eq : basis_rowsP f' == row_swap (basis_rowsP f) p b).
             { intros i c. symmetry. exact (basis_rows_row_swap_precompose (S p0) f p b i c). }
             assert (IHf' := IHm f' Hle'').
             rewrite (Hproper (basis_rowsP f') (row_swap (basis_rowsP f) p b) Hswap_eq),
               (detM_proper (S p0) (basis_rowsP f') (row_swap (basis_rowsP f) p b) Hswap_eq) in IHf'.
             assert (Hgswap : g (row_swap (basis_rowsP f) p b) == - g (basis_rowsP f)).
             { exact (g_swap_antisym (S p0) g Hproper Hadd Halt (basis_rowsP f) p b Hpb). }
             assert (Hdswap : detM (row_swap (basis_rowsP f) p b) == - detM (basis_rowsP f)).
             { exact (g_swap_antisym (S p0) detM (@detM_proper F FF (S p0))
                        (det_rowI_add p0) (detM_alternating (S p0)) (basis_rowsP f) p b Hpb). }
             rewrite Hgswap, Hdswap in IHf'.
             assert (IHf'' : - g (basis_rowsP f) == - (g mat_id × detM (basis_rowsP f))).
             { rewrite IHf'. exact (mul_neg_r (g mat_id) (detM (basis_rowsP f))). }
             exact (neg_inj (g (basis_rowsP f)) (g mat_id × detM (basis_rowsP f)) IHf'').
  Qed.

  Theorem basis_rows_g_eq_detM : ∀ (N : nat) (g : Matrix N N F → F),
    Proper (equ ==> equ) g →
    (∀ (i:Fin.t N)(A:Matrix N N F)(r1 r2:Fin.t N→F),
       g (upd_row A i (fun k⇒r1 k+r2 k)) == g(upd_row A i r1)+g(upd_row A i r2)) →
    (∀ (A:Matrix N N F)(i j:Fin.t N), i≠j → (∀ c, A i c==A j c) → g A == 0) →
    ∀ f : Fin.t N → Fin.t N,
    g (basis_rowsP f) == g mat_id × detM (basis_rowsP f).
  Proof.
    intros N; destruct N as [| p0]; intros g Hproper Hadd Halt f.
    - assert (Heq0 : basis_rowsP f == @mat_id F FF 0).
      { intros i j. exact (Fin.case0 (fun i0 ⇒ basis_rowsP f i0 j == mat_id i0 j) i). }
      transitivity (g mat_id).
      + apply Hproper. exact Heq0.
      + transitivity (g mat_id × 1).
        × symmetry. exact (ring_mul_1r (g mat_id)).
        × apply ring_mul_proper; [reflexivity |].
          transitivity (detM (@mat_id F FF 0)).
          -- symmetry. exact (det_id 0).
          -- symmetry. apply detM_proper. exact Heq0.
    - exact (basis_rows_g_eq_detM_bound p0 g Hproper Hadd Halt (count_nonfixed (S p0) f)
               f (le_n (count_nonfixed (S p0) f))).
  Qed.


  Theorem det_row_expand : ∀ (N : nat) (A : Matrix N N F),
    detM A == fsum N N (fun f : Fin.t N → Fin.t N ⇒
      prodF (fun i : Fin.t N ⇒ A i (f i)) × detM (basis_rowsP f)).
  Proof.
    intros N; destruct N as [| p0]; intros A.
    - assert (Heq0 : A == basis_rowsP (fun i : Fin.t 0 ⇒ Fin.case0 (fun _ ⇒ Fin.t 0) i)).
      { intros i j. exact (Fin.case0 (fun i0 ⇒ A i0 j == basis_rowsP _ i0 j) i). }
      rewrite (fsum_O 0 (fun f : Fin.t 0 → Fin.t 0 ⇒
        prodF (fun i : Fin.t 0 ⇒ A i (f i)) × detM (basis_rowsP f))).
      transitivity ((1:F) × detM (basis_rowsP (fun i : Fin.t 0 ⇒ Fin.case0 (fun _ ⇒ Fin.t 0) i))).
      + transitivity (detM (basis_rowsP (fun i : Fin.t 0 ⇒ Fin.case0 (fun _ ⇒ Fin.t 0) i))).
        × exact (detM_proper 0 A _ Heq0).
        × symmetry. exact (ring_mul_1l _).
      + apply ring_mul_proper; [symmetry; exact (prodF_0 _) | reflexivity].
    - exact (row_expand (S p0) (S p0) (@detM F FF (S p0)) (@detM_proper F FF (S p0))
               (det_rowI_add p0) (det_rowI_scale p0) A).
  Qed.

  Theorem g_row_expand_eq : ∀ (N:nat) (g : Matrix N N F → F),
    Proper (equ ==> equ) g →
    (∀ (i:Fin.t N)(A:Matrix N N F)(r1 r2:Fin.t N→F),
       g (upd_row A i (fun k⇒r1 k+r2 k)) == g(upd_row A i r1)+g(upd_row A i r2)) →
    (∀ (i:Fin.t N)(A:Matrix N N F)(c:F)(r0:Fin.t N→F),
       g (upd_row A i (fun k⇒c×r0 k)) == c×g(upd_row A i r0)) →
    (∀ (A:Matrix N N F)(i j:Fin.t N), i≠j → (∀ c, A i c==A j c) → g A == 0) →
    ∀ A : Matrix N N F, g A == g mat_id × detM A.
  Proof.
    intros N g Hproper Hadd Hscale Halt A.
    assert (Hg_exp := row_expand N N g Hproper Hadd Hscale A).
    assert (Hd_exp := det_row_expand N A).
    assert (Hpoint : ∀ f : Fin.t N → Fin.t N,
      prodF (fun i ⇒ A i (f i)) × g (basis_rowsP f)
        == g mat_id × (prodF (fun i ⇒ A i (f i)) × detM (basis_rowsP f))).
    { intros f.
      calc (prodF (fun i ⇒ A i (f i)) × g (basis_rowsP f))
        == (prodF (fun i ⇒ A i (f i)) × (g mat_id × detM (basis_rowsP f)))
             { (basis_rows_g_eq_detM N g Hproper Hadd Halt f) }
        == ((prodF (fun i ⇒ A i (f i)) × g mat_id) × detM (basis_rowsP f))
             { (ring_mul_assoc (prodF (fun i ⇒ A i (f i))) (g mat_id)
                  (detM (basis_rowsP f))) }
        == ((g mat_id × prodF (fun i ⇒ A i (f i))) × detM (basis_rowsP f))
             { (field_mul_comm (prodF (fun i ⇒ A i (f i))) (g mat_id)) }
        == (g mat_id × (prodF (fun i ⇒ A i (f i)) × detM (basis_rowsP f)))
             { (ring_mul_assoc (g mat_id) (prodF (fun i ⇒ A i (f i)))
                  (detM (basis_rowsP f))) }. }
    calc (g A)
      == (fsum N N (fun f ⇒ prodF (fun i ⇒ A i (f i)) × g (basis_rowsP f)))
      == (fsum N N (fun f ⇒
            g mat_id × (prodF (fun i ⇒ A i (f i)) × detM (basis_rowsP f))))
      == (g mat_id × fsum N N (fun f ⇒
            prodF (fun i ⇒ A i (f i)) × detM (basis_rowsP f)))
      == (g mat_id × detM A).
    - exact Hg_exp.
    - exact (fsum_ext N N
        (fun f ⇒ prodF (fun i ⇒ A i (f i)) × g (basis_rowsP f))
        (fun f ⇒ g mat_id × (prodF (fun i ⇒ A i (f i)) × detM (basis_rowsP f)))
        Hpoint).
    - exact (fsum_scale N N (g mat_id)
        (fun f ⇒ prodF (fun i ⇒ A i (f i)) × detM (basis_rowsP f))).
    - apply ring_mul_proper; [reflexivity |]. symmetry. exact Hd_exp.
  Qed.


  Definition g_mulB {N:nat} (B : Matrix N N F) (A : Matrix N N F) : F := detM (mat_mul A B).

  Lemma mat_mul_upd_row : ∀ (N:nat)(A B:Matrix N N F)(i:Fin.t N)(r:Fin.t N→F),
    mat_mul (upd_row A i r) B == upd_row (mat_mul A B) i (fun k ⇒ bigopF (fun l ⇒ r l × B l k)).
  Proof.
    intros N A B i r i' k.
    destruct (dec (i' = i)) as [Heq | Hne].
    - transitivity (bigopF (fun l ⇒ r l × B l k)).
      + unfold mat_mul. apply bigopF_ext. intros l.
        apply ring_mul_proper; [| reflexivity]. rewrite Heq. exact (upd_row_at N A i r l).
      + rewrite Heq. symmetry. exact (upd_row_at N (mat_mul A B) i (fun k0 ⇒ bigopF (fun l ⇒ r l × B l k0)) k).
    - transitivity (mat_mul A B i' k).
      + unfold mat_mul. apply bigopF_ext. intros l.
        apply ring_mul_proper; [| reflexivity]. exact (upd_row_ne N A i i' r l Hne).
      + symmetry. exact (upd_row_ne N (mat_mul A B) i i' (fun k0 ⇒ bigopF (fun l ⇒ r l × B l k0)) k Hne).
  Qed.

  Lemma g_mulB_proper : ∀ (N:nat)(B:Matrix N N F), Proper (equ ==> equ) (@g_mulB N B).
  Proof.
    intros N B A A' HAA'. unfold g_mulB.
    apply detM_proper. exact (mat_mul_proper N N N A A' HAA' B B (fun i j ⇒ reflexivity (B i j))).
  Qed.

  Lemma g_mulB_add : ∀ (N p0:nat), N = S p0 →
    ∀ (B:Matrix N N F)(i:Fin.t N)(A:Matrix N N F)(r1 r2:Fin.t N→F),
    g_mulB B (upd_row A i (fun k⇒r1 k+r2 k)) == g_mulB B (upd_row A i r1) + g_mulB B (upd_row A i r2).
  Proof.
    intros N p0 HN B i A r1 r2. unfold g_mulB. subst N.
    rewrite (detM_proper (S p0) (mat_mul (upd_row A i (fun k⇒r1 k+r2 k)) B)
               (upd_row (mat_mul A B) i (fun k⇒bigopF (fun l⇒(r1 l+r2 l)*B l k)))
               (mat_mul_upd_row (S p0) A B i (fun k⇒r1 k+r2 k))).
    rewrite (detM_proper (S p0) (mat_mul (upd_row A i r1) B)
               (upd_row (mat_mul A B) i (fun k⇒bigopF (fun l⇒r1 l×B l k)))
               (mat_mul_upd_row (S p0) A B i r1)).
    rewrite (detM_proper (S p0) (mat_mul (upd_row A i r2) B)
               (upd_row (mat_mul A B) i (fun k⇒bigopF (fun l⇒r2 l×B l k)))
               (mat_mul_upd_row (S p0) A B i r2)).
    assert (Hrow : ∀ k, bigopF (fun l ⇒ (r1 l + r2 l) × B l k)
      == bigopF (fun l ⇒ r1 l × B l k) + bigopF (fun l ⇒ r2 l × B l k)).
    { intros k.
      transitivity (bigopF (fun l ⇒ r1 l × B l k + r2 l × B l k)).
      - apply bigopF_ext. intros l. exact (ring_distr (r1 l) (r2 l) (B l k)).
      - exact (bigopF_add (S p0) (fun l ⇒ r1 l × B l k) (fun l ⇒ r2 l × B l k)). }
    rewrite (detM_proper (S p0)
               (upd_row (mat_mul A B) i (fun k⇒bigopF (fun l⇒(r1 l+r2 l)*B l k)))
               (upd_row (mat_mul A B) i (fun k⇒bigopF (fun l⇒r1 l×B l k)+bigopF (fun l⇒r2 l×B l k)))
               (upd_row_change_pointwise (S p0) (mat_mul A B) i _ _ Hrow)).
    exact (det_rowI_add p0 i (mat_mul A B) (fun k⇒bigopF (fun l⇒r1 l×B l k)) (fun k⇒bigopF (fun l⇒r2 l×B l k))).
  Qed.

  Lemma g_mulB_scale : ∀ (N p0:nat), N = S p0 →
    ∀ (B:Matrix N N F)(i:Fin.t N)(A:Matrix N N F)(c:F)(r0:Fin.t N→F),
    g_mulB B (upd_row A i (fun k⇒c×r0 k)) == c × g_mulB B (upd_row A i r0).
  Proof.
    intros N p0 HN B i A c r0. unfold g_mulB. subst N.
    rewrite (detM_proper (S p0) (mat_mul (upd_row A i (fun k⇒c×r0 k)) B)
               (upd_row (mat_mul A B) i (fun k⇒bigopF (fun l⇒(c×r0 l)*B l k)))
               (mat_mul_upd_row (S p0) A B i (fun k⇒c×r0 k))).
    rewrite (detM_proper (S p0) (mat_mul (upd_row A i r0) B)
               (upd_row (mat_mul A B) i (fun k⇒bigopF (fun l⇒r0 l×B l k)))
               (mat_mul_upd_row (S p0) A B i r0)).
    assert (Hrow : ∀ k, bigopF (fun l ⇒ (c×r0 l) × B l k) == c × bigopF (fun l ⇒ r0 l × B l k)).
    { intros k.
      transitivity (bigopF (fun l ⇒ c × (r0 l × B l k))).
      - apply bigopF_ext. intros l. exact (ring_mul_assoc c (r0 l) (B l k)).
      - exact (bigopF_mul_l (S p0) c (fun l ⇒ r0 l × B l k)). }
    rewrite (detM_proper (S p0)
               (upd_row (mat_mul A B) i (fun k⇒bigopF (fun l⇒(c×r0 l)*B l k)))
               (upd_row (mat_mul A B) i (fun k⇒c×bigopF (fun l⇒r0 l×B l k)))
               (upd_row_change_pointwise (S p0) (mat_mul A B) i _ _ Hrow)).
    exact (det_rowI_scale p0 i (mat_mul A B) c (fun k⇒bigopF (fun l⇒r0 l×B l k))).
  Qed.

  Lemma g_mulB_alt : ∀ (N:nat)(B:Matrix N N F)(A:Matrix N N F)(i j:Fin.t N),
    i≠j → (∀ c, A i c == A j c) → g_mulB B A == 0.
  Proof.
    intros N B A i j Hij Heq. unfold g_mulB.
    apply (detM_alternating N (mat_mul A B) i j Hij).
    intros c. unfold mat_mul. apply bigopF_ext. intros l.
    apply ring_mul_proper; [exact (Heq l) | reflexivity].
  Qed.


  Theorem detM_mul : ∀ (N:nat)(A B:Matrix N N F), detM (mat_mul A B) == detM A × detM B.
  Proof.
    intros N A B.
    destruct N as [| p0].
    - calc (detM (mat_mul A B))
        == (1:F)
        == ((1:F)*1)
        == (detM A × detM B).
      + exact (detM_0 (mat_mul A B)).
      + symmetry. exact (ring_mul_1l 1).
      + apply ring_mul_proper; symmetry; [exact (detM_0 A) | exact (detM_0 B)].
    - assert (Hg := g_row_expand_eq (S p0) (g_mulB B) (g_mulB_proper (S p0) B)
                 (fun i A' r1 r2 ⇒ g_mulB_add (S p0) p0 eq_refl B i A' r1 r2)
                 (fun i A' c r0 ⇒ g_mulB_scale (S p0) p0 eq_refl B i A' c r0)
                 (g_mulB_alt (S p0) B) A).
      unfold g_mulB in Hg.
      calc (detM (mat_mul A B))
        == (detM (mat_mul mat_id B) × detM A)
        == (detM A × detM (mat_mul mat_id B))
        == (detM A × detM B).
      + exact Hg.
      + exact (field_mul_comm (detM (mat_mul mat_id B)) (detM A)).
      + apply ring_mul_proper; [reflexivity |].
        apply detM_proper. exact (mat_mul_1l (S p0) (S p0) B).
  Qed.

End DetMulField.


Definition row3 {A : Type} (a b c : A) : Fin.t 3 → A :=
  fun j ⇒ match j with
           | Fin.F1 ⇒ a
           | Fin.FS Fin.F1 ⇒ b
           | Fin.FS (Fin.FS _) ⇒ c
           end.

Definition mat3 {A : Type} (a11 a12 a13 a21 a22 a23 a31 a32 a33 : A) : Matrix 3 3 A :=
  fun i ⇒ match i with
           | Fin.F1 ⇒ row3 a11 a12 a13
           | Fin.FS Fin.F1 ⇒ row3 a21 a22 a23
           | Fin.FS (Fin.FS _) ⇒ row3 a31 a32 a33
           end.

Definition mat3_ns_g : Matrix 3 3 Q :=
  mat3 (1#1)%Q (2#1)%Q (3#1)%Q (4#1)%Q (5#1)%Q (6#1)%Q (7#1)%Q (8#1)%Q (10#1)%Q.

Definition mat3_b_g : Matrix 3 3 Q :=
  mat3 (2#1)%Q (0#1)%Q (1#1)%Q (1#1)%Q (1#1)%Q (0#1)%Q (0#1)%Q (3#1)%Q (1#1)%Q.
