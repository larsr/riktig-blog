Section DetAlternatingAdjacent.
  Context {F : Type} {FF : Field F}.

  Theorem detM_alternating_adjacent : ∀ (n : nat) (k : Fin.t n)
      (A : Matrix (S n) (S n) F),
    (∀ c : Fin.t (S n), A (fweaken k) c == A (Fin.FS k) c) →
    detM A == 0.
  Proof.
    intros n; induction n as [| n0 IHn]; intros k A Heq.
    - exact (Fin.case0 (fun k0 : Fin.t 0 ⇒ detM A == 0) k).
    - rewrite (detM_S (S n0) A).
      set (g := fun c0 : Fin.t (S (S n0)) ⇒
        sign_at c0 × (A c0 Fin.F1 × detM (minor A c0))).
      transitivity (g (fweaken k) + g (Fin.FS k)).
      + apply (bigopF_extract2 (M := F) n0 (fweaken k) (Fin.FS k) g (fweaken_neq_FS (S n0) k)).
        intros c0 Hc1 Hc2.
        destruct (unskip_adjacent_shape n0 k c0 Hc1 Hc2) as [k' [Hu1 Hu2]].
        assert (Hsk1 : skip c0 (fweaken k') = fweaken k)
          by exact (unskip_correct (S n0) c0 (fweaken k) (fweaken k') Hu1).
        assert (Hsk2 : skip c0 (Fin.FS k') = Fin.FS k)
          by exact (unskip_correct (S n0) c0 (Fin.FS k) (Fin.FS k') Hu2).
        assert (Hzero : detM (minor A c0) == 0).
        { apply (IHn k' (minor A c0)).
          intros c.
          rewrite (minor_def (S n0) A c0 (fweaken k') c), Hsk1,
            (minor_def (S n0) A c0 (Fin.FS k') c), Hsk2.
          exact (Heq (Fin.FS c)). }
        unfold g.
        rewrite Hzero, (mul_0_r (A c0 Fin.F1)), (mul_0_r (sign_at c0)).
        reflexivity.
      + unfold g.
        assert (Hminor : minor A (Fin.FS k) == minor A (fweaken k)).
        { intros s c.
          destruct (Fin.eq_dec s k) as [Hsk | Hsk].
          - rewrite Hsk, (minor_def (S n0) A (Fin.FS k) k c), (skip_FS_self (S n0) k),
              (minor_def (S n0) A (fweaken k) k c), (skip_fweaken_self (S n0) k).
            exact (Heq (Fin.FS c)).
          - rewrite (minor_def (S n0) A (Fin.FS k) s c),
              (minor_def (S n0) A (fweaken k) s c),
              <- (skip_fweaken_eq_skip_FS (S n0) k s Hsk).
            reflexivity. }
        rewrite (detM_proper (S n0) (minor A (Fin.FS k)) (minor A (fweaken k)) Hminor).
        rewrite (sign_at_fweaken_FS (S n0) k).
        assert (HF1 : A (Fin.FS k) Fin.F1 == A (fweaken k) Fin.F1).
        { symmetry. exact (Heq Fin.F1). }
        rewrite HF1.
        rewrite (mul_neg_l (sign_at (fweaken k))
                   (A (fweaken k) Fin.F1 × detM (minor A (fweaken k)))).
        exact (grp_invr (sign_at (fweaken k)
                 × (A (fweaken k) Fin.F1 × detM (minor A (fweaken k))))).
  Qed.

End DetAlternatingAdjacent.


Section DetAlternatingGeneral.
  Context {F : Type} {FF : Field F}.

  Theorem detM_alternating_pos : ∀ (n0 : nat) (i k : Fin.t (S n0))
      (A : Matrix (S n0) (S n0) F),
    fin_to_nat k = S (fin_to_nat i) → (∀ c, A i c == A k c) →
    detM A == 0.
  Proof.
    intros n0 i k A Hk Heq.
    pose proof (fin_to_nat_bound (S n0) k) as Hbound.
    assert (Hlt : (fin_to_nat i < n0)%nat) by lia.
    destruct (fweaken_surj n0 i Hlt) as [m Hm].
    assert (Hkm : k = Fin.FS m).
    { apply (fin_to_nat_inj (S n0)).
      rewrite (fin_to_nat_FS n0 m), <- (fin_to_nat_fweaken n0 m), <- Hm.
      exact Hk. }
    apply (detM_alternating_adjacent n0 m A).
    intros c. rewrite <- Hm, <- Hkm. exact (Heq c).
  Qed.

  Theorem detM_alternating_lt : ∀ (n0 : nat) (A : Matrix (S n0) (S n0) F)
      (i j : Fin.t (S n0)),
    (fin_to_nat i < fin_to_nat j)%nat → (∀ c, A i c == A j c) →
    detM A == 0.
  Proof.
    intros n0 A i j Hlt Heq.
    assert (Hij : i ≠ j).
    { intros Heqij. rewrite Heqij in Hlt. lia. }
    destruct (Nat.eq_dec (fin_to_nat j) (S (fin_to_nat i))) as [Hadj | Hnadj].
    - exact (detM_alternating_pos n0 i j A Hadj Heq).
    - pose proof (fin_to_nat_bound (S n0) j) as Hboundj.
      assert (Hlt' : (fin_to_nat i < n0)%nat) by lia.
      destruct (fweaken_surj n0 i Hlt') as [m Hm].
      set (k := Fin.FS m).
      assert (Hik_adj : fin_to_nat k = S (fin_to_nat i)).
      { unfold k. rewrite (fin_to_nat_FS n0 m), Hm, (fin_to_nat_fweaken n0 m).
        reflexivity. }
      assert (Hkj : k ≠ j).
      { intros Heqkj. apply Hnadj. rewrite <- Heqkj. exact Hik_adj. }
      assert (Hik_ne : i ≠ k).
      { intros Heqik. rewrite Heqik in Hik_adj. lia. }
      assert (Hswap := detM_swap_general (S n0) k j Hkj A).
      assert (HA'i : ∀ c, row_swap A k j i c == A i c).
      { intros c. exact (row_swap_ne (S n0) A k j i c Hik_ne Hij). }
      assert (HA'k : ∀ c, row_swap A k j k c == A j c).
      { intros c. exact (row_swap_at_i (S n0) A k j c). }
      assert (HeqA'ik : ∀ c, row_swap A k j i c == row_swap A k j k c).
      { intros c. rewrite (HA'i c), (HA'k c). exact (Heq c). }
      assert (Hzero' := detM_alternating_pos n0 i k (row_swap A k j) Hik_adj HeqA'ik).
      rewrite Hzero' in Hswap.
      assert (Hswap' : - detM A == 0) by (symmetry; exact Hswap).
      rewrite <- (neg_involutive (detM A)).
      rewrite Hswap'.
      exact neg_zero.
  Qed.

  Theorem detM_alternating : ∀ (N : nat) (A : Matrix N N F) (i j : Fin.t N),
    i ≠ j → (∀ c, A i c == A j c) → detM A == 0.
  Proof.
    intros N A i j Hij Heq.
    destruct N as [| n0].
    - exact (Fin.case0 (fun i0 : Fin.t 0 ⇒ detM A == 0) i).
    - destruct (Nat.lt_trichotomy (fin_to_nat i) (fin_to_nat j)) as [Hlt | [Heqn | Hgt]].
      + exact (detM_alternating_lt n0 A i j Hlt Heq).
      + exfalso. apply Hij. exact (fin_to_nat_inj (S n0) i j Heqn).
      + apply (detM_alternating_lt n0 A j i Hgt).
        intros c. symmetry. exact (Heq c).
  Qed.

End DetAlternatingGeneral.


Section RowLinearityGeneral.
  Context {F : Type} {FF : Field F}.

  Lemma upd_row_swap_comm : ∀ (p : nat) (A : Matrix (S p) (S p) F)
      (i : Fin.t (S p)) (r : Fin.t (S p) → F),
    i ≠ Fin.F1 →
    row_swap (upd_row A i r) Fin.F1 i == upd_row (row_swap A Fin.F1 i) Fin.F1 r.
  Proof.
    intros p A i r Hi r' c.
    destruct (Fin.eq_dec r' Fin.F1) as [Hr1 | Hr1].
    - rewrite Hr1, (row_swap_at_i (S p) (upd_row A i r) Fin.F1 i c),
        (upd_row_at (S p) A i r c),
        (upd_row_at (S p) (row_swap A Fin.F1 i) Fin.F1 r c).
      reflexivity.
    - destruct (Fin.eq_dec r' i) as [Hri | Hri].
      + rewrite Hri,
          (row_swap_at_j (S p) (upd_row A i r) Fin.F1 i c (fun Heq ⇒ Hi (eq_sym Heq))),
          (upd_row_ne (S p) A i Fin.F1 r c (fun Heq ⇒ Hi (eq_sym Heq))),
          (upd_row_ne (S p) (row_swap A Fin.F1 i) Fin.F1 i r c Hi),
          (row_swap_at_j (S p) A Fin.F1 i c (fun Heq ⇒ Hi (eq_sym Heq))).
        reflexivity.
      + rewrite (row_swap_ne (S p) (upd_row A i r) Fin.F1 i r' c Hr1 Hri),
          (upd_row_ne (S p) A i r' r c Hri),
          (upd_row_ne (S p) (row_swap A Fin.F1 i) Fin.F1 r' r c Hr1),
          (row_swap_ne (S p) A Fin.F1 i r' c Hr1 Hri).
        reflexivity.
  Qed.

  Theorem det_rowI_conj : ∀ (p : nat) (i : Fin.t (S p))
      (A : Matrix (S p) (S p) F) (r : Fin.t (S p) → F),
    i ≠ Fin.F1 →
    detM (upd_row A i r) == - detM (upd_row (row_swap A Fin.F1 i) Fin.F1 r).
  Proof.
    intros p i A r Hi.
    assert (Hswap := detM_swap_general (S p) Fin.F1 i (fun Heq ⇒ Hi (eq_sym Heq))
                        (upd_row A i r)).
    rewrite (detM_proper (S p) (row_swap (upd_row A i r) Fin.F1 i)
               (upd_row (row_swap A Fin.F1 i) Fin.F1 r)
               (upd_row_swap_comm p A i r Hi)) in Hswap.
    rewrite Hswap.
    symmetry.
    exact (neg_involutive (detM (upd_row A i r))).
  Qed.

  Theorem det_rowI_add : ∀ (p : nat) (i : Fin.t (S p)) (A : Matrix (S p) (S p) F)
      (r1 r2 : Fin.t (S p) → F),
    detM (upd_row A i (fun k ⇒ r1 k + r2 k))
      == detM (upd_row A i r1) + detM (upd_row A i r2).
  Proof.
    intros p i.
    destruct (Fin.eq_dec i Fin.F1) as [Hi1 | Hi1].
    - intros A r1 r2. rewrite Hi1. exact (det_rowF1_add p A r1 r2).
    - intros A r1 r2.
      rewrite (det_rowI_conj p i A (fun k ⇒ r1 k + r2 k) Hi1),
        (det_rowI_conj p i A r1 Hi1), (det_rowI_conj p i A r2 Hi1).
      rewrite (det_rowF1_add p (row_swap A Fin.F1 i) r1 r2).
      exact (neg_add (detM (upd_row (row_swap A Fin.F1 i) Fin.F1 r1))
               (detM (upd_row (row_swap A Fin.F1 i) Fin.F1 r2))).
  Qed.

  Theorem det_rowI_scale : ∀ (p : nat) (i : Fin.t (S p)) (A : Matrix (S p) (S p) F)
      (c : F) (r0 : Fin.t (S p) → F),
    detM (upd_row A i (fun k ⇒ c × r0 k)) == c × detM (upd_row A i r0).
  Proof.
    intros p i.
    destruct (Fin.eq_dec i Fin.F1) as [Hi1 | Hi1].
    - intros A c r0. rewrite Hi1. exact (det_rowF1_scale p A c r0).
    - intros A c r0.
      rewrite (det_rowI_conj p i A (fun k ⇒ c × r0 k) Hi1),
        (det_rowI_conj p i A r0 Hi1).
      rewrite (det_rowF1_scale p (row_swap A Fin.F1 i) c r0).
      symmetry.
      exact (mul_neg_r c (detM (upd_row (row_swap A Fin.F1 i) Fin.F1 r0))).
  Qed.

End RowLinearityGeneral.


Section DetElemAdd.
  Context {F : Type} {FF : Field F}.

  Lemma elem_add_mul_eq_upd_row : ∀ (n : nat) (i j : Fin.t n) (c : F)
      (A : Matrix n n F),
    mat_mul (elem_add i j c) A == upd_row A i (fun k ⇒ A i k + c × A j k).
  Proof.
    intros n i j c A r' c'.
    destruct (Fin.eq_dec r' i) as [Hri | Hri].
    - rewrite Hri, (elem_add_mul_eq n n i j c A c'),
        (upd_row_at n A i (fun k ⇒ A i k + c × A j k) c').
      reflexivity.
    - rewrite (elem_add_mul_ne n n i j r' c A c' Hri),
        (upd_row_ne n A i r' (fun k ⇒ A i k + c × A j k) c' Hri).
      reflexivity.
  Qed.

  Theorem det_elem_add : ∀ (n : nat) (i j : Fin.t n) (c : F) (A : Matrix n n F),
    i ≠ j → detM (mat_mul (elem_add i j c) A) == detM A.
  Proof.
    intros n i j c A Hij.
    destruct n as [| p].
    - exact (Fin.case0 (fun i0 : Fin.t 0 ⇒ detM (mat_mul (elem_add i0 j c) A) == detM A) i).
    - rewrite (detM_proper (S p) (mat_mul (elem_add i j c) A)
                 (upd_row A i (fun k ⇒ A i k + c × A j k))
                 (elem_add_mul_eq_upd_row (S p) i j c A)).
      rewrite (det_rowI_add p i A (A i) (fun k ⇒ c × A j k)).
      rewrite (det_rowI_scale p i A c (A j)).
      rewrite (detM_proper (S p) (upd_row A i (A i)) A (upd_row_self (S p) A i)).
      assert (Hji : j ≠ i) by (intros Heq; apply Hij; exact (eq_sym Heq)).
      assert (Halt : detM (upd_row A i (A j)) == 0).
      { apply (detM_alternating (S p) (upd_row A i (A j)) i j Hij).
        intros c0.
        rewrite (upd_row_at (S p) A i (A j) c0), (upd_row_ne (S p) A i j (A j) c0 Hji).
        reflexivity. }
      rewrite Halt, (mul_0_r c).
      exact (mon_idr (detM A)).
  Qed.

End DetElemAdd.

Qed.

    == detM mat3_ns.
Proof.
  reflexivity.
Qed.

Example det_elem_add_Q :
  detM (mat_mul (elem_add Fin.F1 (Fin.FS (Fin.FS Fin.F1)) (5#1)%Q) mat3_ns)
    == detM mat3_ns.
Proof.
  exact (det_elem_add 3 Fin.F1 (Fin.FS (Fin.FS Fin.F1)) (5#1)%Q
    mat3_ns (fin_F1_neq_FS 2 (Fin.FS Fin.F1))).
Qed.
