Library Riktig.Interface.AbGroup

From Riktig Require Export Ops Props Interface.Group.

Class AbGroup G := {
  #[global] ab_group :: Group G;
  ab_comm : ∀ a b : G, a + b == b + a
}.

#[global] Instance AbGroup_Comm G (H : AbGroup G) : Commutes G equ add := ab_comm.
