From Riktig Require Import Ops Interface.VectorSpace Interface.InnerProduct
  Interface.Order.

Class PosInner (F V : Type) `{OrderedField F} {VS : VectorSpace F V}
    {IP : InnerProduct F V} := {
  pi_nonneg : ∀ u : V, 0 ≤ ⟨ u , u ⟩;
  pi_definite : ∀ u : V, ⟨ u , u ⟩ == 0 → u == 0
}.
