Definition assemble (binds : list tm) (atom : tm) : tm :=
  fold_right (fun c acc ⇒ App (Abs acc) c) atom binds.

Fixpoint mapi_from (i : nat) (g : nat → tm → tm) (l : list tm) : list tm :=
  match l with
  | [] ⇒ []
  | x :: xs ⇒ g i x :: mapi_from (S i) g xs
  end.

Fixpoint flat (t : tm) : list tm × tm :=
  match t with
  | Var n ⇒ ([], Var n)
  | Abs b ⇒ let (bb, ab) := flat b in ([], Abs (assemble bb ab))
  | App f a ⇒
      let (bf, af) := flat f in
      let (ba, aa) := flat a in
      let m := length bf in
      let p := length ba in
      let ba' := mapi_from 0 (fun j c ⇒ lift m j c) ba in
      let c := App (lift p 0 af) (lift m p aa) in
      (bf ++ ba' ++ [c], Var 0)
  end.

Definition anf (t : tm) : tm := let (b, a) := flat t in assemble b a.


Inductive Atom : tm → Prop :=
  | at_var : ∀ n, Atom (Var n)
  | at_abs : ∀ b, Anf b → Atom (Abs b)
with Anf : tm → Prop :=
  | anf_atom : ∀ a, Atom a → Anf a
  | anf_comp : ∀ a1 a2, Atom a1 → Atom a2 → Anf (App a1 a2)
  | anf_let : ∀ a1 a2 body,
      Atom a1 → Atom a2 → Anf body → Anf (App (Abs body) (App a1 a2)).

Inductive Comp : tm → Prop :=
  | comp_intro : ∀ a1 a2, Atom a1 → Atom a2 → Comp (App a1 a2).

Scheme Atom_ind2 := Induction for Atom Sort Prop
  with Anf_ind2 := Induction for Anf Sort Prop.
Combined Scheme Atom_Anf_mutind from Atom_ind2, Anf_ind2.


Lemma lift_pred :
  (∀ t, Atom t → ∀ d k, Atom (lift d k t)) ∧
  (∀ t, Anf t → ∀ d k, Anf (lift d k t)).
Proof.
  apply Atom_Anf_mutind; intros; simpl.
  - constructor.
  - constructor. apply H.
  - constructor. apply H.
  - constructor; [apply H | apply H0].
  - constructor; [apply H | apply H0 | apply H1].
Qed.

Lemma lift_Atom : ∀ t d k, Atom t → Atom (lift d k t).
Proof. intros t d k H. apply (proj1 lift_pred); exact H. Qed.

Lemma lift_Anf : ∀ t d k, Anf t → Anf (lift d k t).
Proof. intros t d k H. apply (proj2 lift_pred); exact H. Qed.

Lemma lift_Comp : ∀ c d k, Comp c → Comp (lift d k c).
Proof.
  intros c d k H. destruct H as [a1 a2 H1 H2]. simpl.
  constructor; apply lift_Atom; assumption.
Qed.

Lemma Atom_Anf : ∀ t, Atom t → Anf t.
Proof. intros t H. apply anf_atom. exact H. Qed.

Lemma assemble_Anf : ∀ binds atom,
  Atom atom → Forall Comp binds → Anf (assemble binds atom).
Proof.
  induction binds as [| c rest IH]; intros atom Hatom Hbinds; simpl.
  - apply Atom_Anf. exact Hatom.
  - inversion Hbinds as [| c0 rest0 Hc Hrest]; subst.
    destruct Hc as [a1 a2 H1 H2].
    apply anf_let.
    + exact H1.
    + exact H2.
    + apply IH; assumption.
Qed.

Lemma Forall_mapi : ∀ g l i,
  (∀ j x, Comp x → Comp (g j x)) →
  Forall Comp l → Forall Comp (mapi_from i g l).
Proof.
  induction l as [| x xs IH]; intros i Hg Hl; simpl.
  - constructor.
  - inversion Hl; subst. constructor.
    + apply Hg. assumption.
    + apply IH; assumption.
Qed.

Lemma flat_inv : ∀ t,
  Atom (snd (flat t)) ∧ Forall Comp (fst (flat t)).
Proof.
  induction t as [n | b IH | f IHf a IHa]; simpl.
  - split; [constructor | constructor].
  - destruct (flat b) as [bb ab]. simpl in IH. destruct IH as [IHat IHco].
    split; simpl.
    + constructor. apply assemble_Anf; assumption.
    + constructor.
  - destruct (flat f) as [bf af]. destruct (flat a) as [ba aa].
    simpl in IHf, IHa. destruct IHf as [Haf Hbf]. destruct IHa as [Haa Hba].
    split; simpl.
    + constructor.
    + apply Forall_app. split; [exact Hbf |].
      apply Forall_app. split.
      × apply Forall_mapi; [| exact Hba].
        intros j x Hx. apply lift_Comp. exact Hx.
      × constructor; [| constructor].
        constructor; apply lift_Atom; assumption.
Qed.

Theorem anf_is_anf : ∀ t, Anf (anf t).
Proof.
  intros t. unfold anf.
  destruct (flat_inv t) as [Hatom Hbinds].
  destruct (flat t) as [b a]. simpl in Hatom, Hbinds.
  apply assemble_Anf; assumption.
Qed.


Lemma let_correct : ∀ body rhs v,
  value v → star rhs v → App (Abs body) rhs == subst 0 v body.
Proof.
  intros body rhs v Hv Hstar.
  calc (App (Abs body) rhs)
    == (App (Abs body) v)
    == (subst 0 v body).
  - apply conv_appr. apply conv_of_star. exact Hstar.
  - apply conv_beta. exact Hv.
Qed.


Example anf_id_id_shape :
  anf (App idt idt)
  = App (Abs (Var 0)) (App (Abs (Var 0)) (Abs (Var 0))).
Proof. reflexivity. Qed.

Example anf_id_id_is_anf : Anf (anf (App idt idt)).
Proof. apply anf_is_anf. Qed.

Example anf_preserves_id_id :
  eval 50 (anf (App idt idt)) = eval 50 (App idt idt).
Proof. reflexivity. Qed.

Example anf_preserves_nested :
  eval 80 (anf (App idt (App idt idt))) = eval 80 (App idt (App idt idt)).
Proof. reflexivity. Qed.

Example anf_preserves_left_nested :
  eval 80 (anf (App (App idt idt) idt)) = eval 80 (App (App idt idt) idt).
Proof. reflexivity. Qed.

Example anf_conv_id_id : anf (App idt idt) == App idt idt.
Proof. apply (conv_by_eval 50 50). reflexivity. Qed.

Example anf_conv_k :
  anf (App (App k idt) (Abs (Abs (Var 0)))) == App (App k idt) (Abs (Abs (Var 0))).
Proof. apply (conv_by_eval 100 100). reflexivity. Qed.
