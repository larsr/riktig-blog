Section IntegralTheory.
  Context {F V : Type} `{Integral F V}.

  Theorem int_empty : ∀ (f : F → V) (a : F), int f a a == 0.
  Proof.
    intros f a.
    apply (cancel_l (int f a a)).
    rewrite int_additivity, idr.
    reflexivity.
  Qed.

  Theorem int_rev : ∀ (f : F → V) (a b : F),
    int f a b == - int f b a.
  Proof.
    intros f a b.
    apply neg_unique_l.
    rewrite int_additivity.
    apply int_empty.
  Qed.

  Theorem int_zero_fn : ∀ a b : F,
    int (fun _ : F ⇒ (0 : V)) a b == 0.
  Proof.
    intros a b.
    rewrite (int_const 0 a b).
    apply scale_zero.
  Qed.

End IntegralTheory.
