Definition abnf_section4 :=
  ("rulelist = 1*( rule / (*c-wsp c-nl) )" ++ lf ++
  lf ++
  "rule = rulename defined-as elements c-nl" ++ lf ++
  " ; continues if next line starts" ++ lf ++
  " ; with white space" ++ lf ++
  lf ++
  "rulename = ALPHA *(ALPHA / DIGIT / ""-"")" ++ lf ++
  lf ++
  "defined-as = *c-wsp (""=/"" / ""="") *c-wsp" ++ lf ++
  " ; basic rules definition and" ++ lf ++
  " ; incremental alternatives" ++ lf ++
  lf ++
  "elements = alternation *c-wsp" ++ lf ++
  lf ++
  "c-wsp = WSP / (c-nl WSP)" ++ lf ++
  lf ++
  "c-nl = comment / CRLF" ++ lf ++
  " ; comment or newline" ++ lf ++
  lf ++
  "comment = "";"" *(WSP / VCHAR) CRLF" ++ lf ++
  lf ++
  "alternation = concatenation" ++ lf ++
  " *(*c-wsp ""/"" *c-wsp concatenation)" ++ lf ++
  lf ++
  "concatenation = repetition *(1*c-wsp repetition)" ++ lf ++
  lf ++
  "repetition = [repeat] element" ++ lf ++
  lf ++
  "repeat = (*DIGIT ""*"" *DIGIT) / 1*DIGIT" ++ lf ++
  lf ++
  "element = rulename / group / option /" ++ lf ++
  " char-val / num-val / prose-val" ++ lf ++
  lf ++
  "group = ""("" *c-wsp alternation *c-wsp "")""" ++ lf ++
  lf ++
  "option = ""["" *c-wsp alternation *c-wsp ""]""" ++ lf ++
  lf ++
  "char-val = DQUOTE *(%x20-21 / %x23-7E) DQUOTE" ++ lf ++
  " ; quoted string of SP and VCHAR" ++ lf ++
  " ; without DQUOTE" ++ lf ++
  lf ++
  "num-val = ""%"" (bin-val / dec-val / hex-val)" ++ lf ++
  lf ++
  "bin-val = ""b"" 1*BIT" ++ lf ++
  " [ 1*(""."" 1*BIT) / (""-"" 1*BIT) ]" ++ lf ++
  " ; series of concatenated bit values" ++ lf ++
  " ; or single ONEOF range" ++ lf ++
  lf ++
  "dec-val = ""d"" 1*DIGIT" ++ lf ++
  " [ 1*(""."" 1*DIGIT) / (""-"" 1*DIGIT) ]" ++ lf ++
  lf ++
  "hex-val = ""x"" 1*HEXDIG" ++ lf ++
  " [ 1*(""."" 1*HEXDIG) / (""-"" 1*HEXDIG) ]" ++ lf ++
  lf ++
  "prose-val = ""<"" *(%x20-3D / %x3F-7E) "">""" ++ lf ++
  " ; bracketed string of SP and VCHAR" ++ lf ++
  " ; without angles" ++ lf ++
  " ; prose description, to be used as" ++ lf ++
  " ; last resort" ++ lf)%string.

Definition abnf_grammar :=
  
  ("; RFC 5234 - Section 4: ABNF Definition of ABNF" ++ lf ++
  lf ++
  "rulelist = 1*( rule / (*c-wsp c-nl) )" ++ lf ++
  lf ++
  "rule = rulename defined-as elements c-nl" ++ lf ++
  " ; continues if next line starts" ++ lf ++
  " ; with white space" ++ lf ++
  lf ++
  "rulename = ALPHA *(ALPHA / DIGIT / ""-"")" ++ lf ++
  lf ++
  "defined-as = *c-wsp (""=/"" / ""="") *c-wsp" ++ lf ++
  " ; basic rules definition and" ++ lf ++
  " ; incremental alternatives" ++ lf ++
  lf ++
  "elements = alternation *c-wsp" ++ lf ++
  lf ++
  "c-wsp = WSP / (c-nl WSP)" ++ lf ++
  lf ++
  "c-nl = comment / CRLF" ++ lf ++
  " ; comment or newline" ++ lf ++
  lf ++
  "comment = "";"" *(WSP / VCHAR) CRLF" ++ lf ++
  lf ++
  "alternation = concatenation" ++ lf ++
  " *(*c-wsp ""/"" *c-wsp concatenation)" ++ lf ++
  lf ++
  "concatenation = repetition *(1*c-wsp repetition)" ++ lf ++
  lf ++
  "repetition = [repeat] element" ++ lf ++
  lf ++
  "repeat = (*DIGIT ""*"" *DIGIT) / 1*DIGIT" ++ lf ++
  lf ++
  "element = rulename / group / option /" ++ lf ++
  " char-val / num-val / prose-val" ++ lf ++
  lf ++
  "group = ""("" *c-wsp alternation *c-wsp "")""" ++ lf ++
  lf ++
  "option = ""["" *c-wsp alternation *c-wsp ""]""" ++ lf ++
  lf ++
  "char-val = DQUOTE *(%x20-21 / %x23-7E) DQUOTE" ++ lf ++
  " ; quoted string of SP and VCHAR" ++ lf ++
  " ; without DQUOTE" ++ lf ++
  lf ++
  "num-val = ""%"" (bin-val / dec-val / hex-val)" ++ lf ++
  lf ++
  "bin-val = ""b"" 1*BIT" ++ lf ++
  " [ 1*(""."" 1*BIT) / (""-"" 1*BIT) ]" ++ lf ++
  " ; series of concatenated bit values" ++ lf ++
  " ; or single ONEOF range" ++ lf ++
  lf ++
  "dec-val = ""d"" 1*DIGIT" ++ lf ++

  lf ++
  "hex-val = ""x"" 1*HEXDIG" ++ lf ++
  " [ 1*(""."" 1*HEXDIG) / (""-"" 1*HEXDIG) ]" ++ lf ++
  lf ++
  "prose-val = ""<"" *(%x20-3D / %x3F-7E) "">""" ++ lf ++
  " ; bracketed string of SP and VCHAR" ++ lf ++
  " ; without angles" ++ lf ++
  " ; prose description, to be used as" ++ lf ++
  " ; last resort" ++ lf ++
  lf ++
  "; RFC 5234 - Appendix B.1: Core Rules" ++ lf ++
  lf ++
  "ALPHA = %x41-5A / %x61-7A ; A-Z / a-z" ++ lf ++
  lf ++
  "BIT = ""0"" / ""1""" ++ lf ++
  lf ++
  "CHAR = %x01-7F" ++ lf ++
  " ; any 7-bit US-ASCII character," ++ lf ++
  " ; excluding NUL" ++ lf ++
  lf ++
  "CR = %x0D" ++ lf ++
  " ; carriage return" ++ lf ++
  lf ++
  "CRLF = CR LF" ++ lf ++
  " ; Internet standard newline" ++ lf ++
  lf ++
  "CTL = %x00-1F / %x7F" ++ lf ++

  lf ++

  lf ++

  " ; white space" ++ lf)%string.


Example grammar_length :
  String.length abnf_grammar = 3388.
Proof. vm_compute. reflexivity. Qed.


Example parse_abnf_grammar_succeeds :
  match parse_abnf_tree abnf_grammar with
  | Some _ ⇒ true
  | None ⇒ false
  end = true.
Proof. compute. reflexivity. Qed.


Example roundtrip_abnf_grammar :
  match parse_abnf_tree abnf_grammar with
  | Some t ⇒ String.eqb (pretty_print t) abnf_grammar
  | None ⇒ false
  end = true.
Proof. compute. reflexivity. Qed.


Definition abnf_section4_rules : option ABNFRulelist :=
  parse_abnf abnf_section4.

Example selfhost_abnf_section4 :
  match abnf_section4_rules with
  | None ⇒ None
  | Some rules ⇒ print_with_grammar rules "rulelist" abnf_section4
  end = Some abnf_section4.
Proof. vm_compute. reflexivity. Qed.
