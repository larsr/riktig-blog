Library Riktig.Interface.InnerProduct

From Coq Require Import Morphisms.
From Riktig Require Import Ops Interface.VectorSpace.

Class InnerProduct (F V : Type) `{VectorSpace F V} := {
  #[global] ip_inner :: HasInner F V;
  #[global] ip_proper :: Proper (equ ==> equ ==> equ) (@inner F V _);
  ip_sym : ∀ u v : V, ⟨ u , v ⟩ == ⟨ v , u ⟩;
  ip_add_l : ∀ u v w : V, ⟨ u + v , w ⟩ == ⟨ u , w ⟩ + ⟨ v , w ⟩;
  ip_scale_l : ∀ (a : F) (u v : V), ⟨ a *: u , v ⟩ == a × ⟨ u , v ⟩
}.

Definition norm2 {F V : Type} `{InnerProduct F V} (u : V) : F := ⟨ u , u ⟩.
