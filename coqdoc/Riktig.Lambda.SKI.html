Library Riktig.Lambda.SKI


Require Import Riktig.Lambda.Syntax.

Inductive comb : Type :=
  | CS : comb
  | CK : comb
  | CI : comb
  | CVar : nat → comb
  | CApp : comb → comb → comb.

Notation "f @@ x" := (CApp f x) (at level 40, left associativity).


Inductive cstep : comb → comb → Prop :=
  | cstep_S : ∀ f g x,
      cstep (CS @@ f @@ g @@ x) (f @@ x @@ (g @@ x))
  | cstep_K : ∀ x y,
      cstep (CK @@ x @@ y) x
  | cstep_I : ∀ x,
      cstep (CI @@ x) x
  | cstep_app1 : ∀ f f' a,
      cstep f f' →
      cstep (f @@ a) (f' @@ a)
  | cstep_app2 : ∀ f a a',
      cstep a a' →
      cstep (f @@ a) (f @@ a').

Inductive cstar : comb → comb → Prop :=
  | cstar_refl : ∀ t, cstar t t
  | cstar_step : ∀ t u v, cstep t u → cstar u v → cstar t v.

Lemma cstar_trans : ∀ t u v, cstar t u → cstar u v → cstar t v.
Proof.
  intros t u v H1. induction H1 as [t1 | t1 u1 v1 Hs Hst IH]; intros H2.
  - exact H2.
  - eapply cstar_step. { exact Hs. } apply IH. exact H2.
Qed.

Lemma cstar_app1 : ∀ f f' a, cstar f f' → cstar (f @@ a) (f' @@ a).
Proof.
  intros f f' a H. induction H as [t | t u v Hs Hst IH].
  - apply cstar_refl.
  - eapply cstar_step. { apply cstep_app1. exact Hs. } exact IH.
Qed.

Lemma cstar_app2 : ∀ f a a', cstar a a' → cstar (f @@ a) (f @@ a').
Proof.
  intros f a a' H. induction H as [t | t u v Hs Hst IH].
  - apply cstar_refl.
  - eapply cstar_step. { apply cstep_app2. exact Hs. } exact IH.
Qed.


Definition contract (t : comb) : option comb :=
  match t with
  | CApp (CApp (CApp CS f) g) x ⇒ Some (f @@ x @@ (g @@ x))
  | CApp (CApp CK x) _ ⇒ Some x
  | CApp CI x ⇒ Some x
  | _ ⇒ None
  end.

Lemma contract_sound : ∀ t t', contract t = Some t' → cstep t t'.
Proof.
  intros t t' H.
  destruct t as [ | | | n | f a]; simpl in H; try discriminate.
  destruct f as [ | | | n | f1 a1]; try discriminate.
  - injection H as <-. apply cstep_I.
  - destruct f1 as [ | | | n | f2 a2]; try discriminate.
    + injection H as <-. apply cstep_K.
    + destruct f2; try discriminate.
      injection H as <-. apply cstep_S.
Qed.

Arguments contract : simpl never.

Fixpoint cstep_fn (t : comb) : option comb :=
  match t with
  | CApp f a ⇒
      match contract (CApp f a) with
      | Some t' ⇒ Some t'
      | None ⇒
          match cstep_fn f with
          | Some f' ⇒ Some (f' @@ a)
          | None ⇒ option_map (CApp f) (cstep_fn a)
          end
      end
  | _ ⇒ None
  end.

Lemma cstep_fn_sound : ∀ t t', cstep_fn t = Some t' → cstep t t'.
Proof.
  induction t as [ | | | n | f IHf a IHa]; intros t' H; simpl in H;
    try discriminate.
  destruct (contract (CApp f a)) as [u|] eqn:Hc.
  - injection H as <-. apply contract_sound. exact Hc.
  - destruct (cstep_fn f) as [f'|].
    + injection H as <-. apply cstep_app1. apply IHf. reflexivity.
    + destruct (cstep_fn a) as [a'|]; [| discriminate].
      injection H as <-. apply cstep_app2. apply IHa. reflexivity.
Qed.

Lemma cstep_fn_none_no_step : ∀ t t', cstep t t' → cstep_fn t ≠ None.
Proof.
  intros t t' H.
  induction H as [f g x | x y | x | f f' a Hs IH | f a a' Hs IH]; intros E.
  - discriminate E.
  - discriminate E.
  - discriminate E.
  - simpl in E. destruct (contract (CApp f a)) as [u|].
    + discriminate E.
    + destruct (cstep_fn f) as [f0|].
      × discriminate E.
      × apply IH. reflexivity.
  - simpl in E. destruct (contract (CApp f a)) as [u|].
    + discriminate E.
    + destruct (cstep_fn f) as [f0|].
      × discriminate E.
      × destruct (cstep_fn a) as [a0|].
        { discriminate E. }
        { apply IH. reflexivity. }
Qed.

Fixpoint nf (fuel : nat) (t : comb) : comb :=
  match fuel with
  | O ⇒ t
  | S fuel' ⇒
      match cstep_fn t with
      | Some t' ⇒ nf fuel' t'
      | None ⇒ t
      end
  end.

Lemma nf_sound : ∀ fuel t, cstar t (nf fuel t).
Proof.
  induction fuel as [| fuel IH]; intros t; simpl.
  - apply cstar_refl.
  - destruct (cstep_fn t) as [t'|] eqn:H.
    + eapply cstar_step. { apply cstep_fn_sound. exact H. } apply IH.
    + apply cstar_refl.
Qed.

Lemma nf_cstar : ∀ fuel t u, nf fuel t = u → cstar t u.
Proof.
  intros fuel t u H. rewrite <- H. apply nf_sound.
Qed.

Example nf_SKK : nf 10 (CS @@ CK @@ CK @@ CVar 0) = CVar 0.
Proof. reflexivity. Qed.


Definition Bc : comb := CS @@ (CK @@ CS) @@ CK.
Definition Cc : comb := CS @@ (CS @@ (CK @@ Bc) @@ CS) @@ (CK @@ CK).
Definition Wc : comb := CS @@ CS @@ (CS @@ CK).

Example B_spec :
  cstar (Bc @@ CVar 0 @@ CVar 1 @@ CVar 2) (CVar 0 @@ (CVar 1 @@ CVar 2)).
Proof. apply (nf_cstar 50). reflexivity. Qed.

Example C_spec :
  cstar (Cc @@ CVar 0 @@ CVar 1 @@ CVar 2) (CVar 0 @@ CVar 2 @@ CVar 1).
Proof. apply (nf_cstar 50). reflexivity. Qed.

Example W_spec :
  cstar (Wc @@ CVar 0 @@ CVar 1) (CVar 0 @@ CVar 1 @@ CVar 1).
Proof. apply (nf_cstar 50). reflexivity. Qed.


Definition SfromBCW : comb := Bc @@ (Bc @@ (Bc @@ Wc) @@ Cc) @@ (Bc @@ Bc).
Definition IfromWK : comb := Wc @@ CK.

Example SfromBCW_spec :
  cstar (SfromBCW @@ CVar 0 @@ CVar 1 @@ CVar 2)
        (CVar 0 @@ CVar 2 @@ (CVar 1 @@ CVar 2)).
Proof. apply (nf_cstar 500). reflexivity. Qed.

Example IfromWK_spec : cstar (IfromWK @@ CVar 0) (CVar 0).
Proof. apply (nf_cstar 50). reflexivity. Qed.


Fixpoint babs (t : comb) : comb :=
  match t with
  | CVar O ⇒ CI
  | CVar (S n) ⇒ CK @@ CVar n
  | CApp f a ⇒ CS @@ babs f @@ babs a
  | c ⇒ CK @@ c
  end.

Fixpoint csub (u : comb) (t : comb) : comb :=
  match t with
  | CVar O ⇒ u
  | CVar (S n) ⇒ CVar n
  | CApp f a ⇒ csub u f @@ csub u a
  | c ⇒ c
  end.

Lemma babs_correct : ∀ t u, cstar (babs t @@ u) (csub u t).
Proof.
  induction t as [ | | | n | f IHf a IHa]; intros u; simpl.
  - eapply cstar_step. { apply cstep_K. } apply cstar_refl.
  - eapply cstar_step. { apply cstep_K. } apply cstar_refl.
  - eapply cstar_step. { apply cstep_K. } apply cstar_refl.
  - destruct n as [| n]; simpl.
    + eapply cstar_step. { apply cstep_I. } apply cstar_refl.
    + eapply cstar_step. { apply cstep_K. } apply cstar_refl.
  - eapply cstar_step. { apply cstep_S. }
    eapply cstar_trans. { apply cstar_app1. apply IHf. }
    apply cstar_app2. apply IHa.
Qed.


Fixpoint translate (t : tm) : comb :=
  match t with
  | Var n ⇒ CVar n
  | Abs b ⇒ babs (translate b)
  | App f a ⇒ translate f @@ translate a
  end.

Lemma translate_beta_step : ∀ b v,
  cstar (translate (App (Abs b) v)) (csub (translate v) (translate b)).
Proof.
  intros b v. simpl. apply babs_correct.
Qed.


Example translate_id : translate idt = CI.
Proof. reflexivity. Qed.

Example translate_id_beta : cstar (translate idt @@ CVar 5) (CVar 5).
Proof. apply (nf_cstar 10). reflexivity. Qed.

Example translate_k : translate k = CS @@ (CK @@ CK) @@ CI.
Proof. reflexivity. Qed.

Example translate_k_fst :
  cstar (translate k @@ CVar 0 @@ CVar 1) (CVar 0).
Proof. apply (nf_cstar 20). reflexivity. Qed.

Example translate_app_id_id : cstar (translate (App idt idt)) CI.
Proof. apply (nf_cstar 10). reflexivity. Qed.

Example translate_flipapp :
  cstar (translate (Abs (Abs (App (Var 0) (Var 1)))) @@ CVar 0 @@ CVar 1)
        (CVar 1 @@ CVar 0).
Proof. apply (nf_cstar 50). reflexivity. Qed.
