Library Riktig.Interface.Field

From Coq Require Import Morphisms.
From Riktig Require Import Ops Interface.Ring.

Class Field F := {
  #[global] field_ring :: Ring F;
  field_mul_comm : ∀ a b : F, a × b == b × a;
  #[global] field_inv :: HasInv F;
  field_invl : ∀ a : F, a =/= 0 → inv a × a == 1;
  #[global] field_inv_proper :: Proper (equ ==> equ) (@inv F _);
  field_nontrivial : (1 : F) =/= 0
}.
