Library Riktig.Theory.Fin

From Coq Require Import Arith Lia List.
From Coq.Vectors Require Fin.


Lemma fin_F1_neq_FS : ∀ (n : nat) (i : Fin.t n),
  Fin.F1 ≠ Fin.FS i.
Proof.
  intros n i Heq.
  discriminate Heq.
Qed.

Lemma fin_FS_neq_F1 : ∀ (n : nat) (i : Fin.t n),
  Fin.FS i ≠ Fin.F1.
Proof.
  intros n i Heq.
  exact (fin_F1_neq_FS n i (eq_sym Heq)).
Qed.


Fixpoint skip {n : nat} (i : Fin.t (S n)) {struct n}
    : Fin.t n → Fin.t (S n) :=
  match n as n0 return Fin.t (S n0) → Fin.t n0 → Fin.t (S n0) with
  | O ⇒ fun _ j ⇒ Fin.case0 (fun _ ⇒ Fin.t 1) j
  | S m ⇒ fun i0 ⇒
      Fin.caseS' i0 (fun _ ⇒ Fin.t (S m) → Fin.t (S (S m)))
        Fin.FS
        (fun (i' : Fin.t (S m)) (j : Fin.t (S m)) ⇒
           Fin.caseS' j (fun _ ⇒ Fin.t (S (S m)))
             Fin.F1
             (fun j' : Fin.t m ⇒ Fin.FS (skip i' j')))
  end i.

Lemma skip_F1 : ∀ (n : nat) (j : Fin.t n),
  skip Fin.F1 j = Fin.FS j.
Proof.
  intros n j.
  destruct j as [m | m j'].
  - reflexivity.
  - reflexivity.
Qed.

Lemma skip_FS_F1 : ∀ (n : nat) (i : Fin.t (S n)),
  skip (Fin.FS i) Fin.F1 = Fin.F1.
Proof.
  intros n i.
  reflexivity.
Qed.

Lemma skip_FS_FS : ∀ (n : nat) (i : Fin.t (S n)) (j : Fin.t n),
  skip (Fin.FS i) (Fin.FS j) = Fin.FS (skip i j).
Proof.
  intros n i j.
  reflexivity.
Qed.


Lemma skip_ne : ∀ (n : nat) (i : Fin.t (S n)) (s : Fin.t n),
  skip i s ≠ i.
Proof.
  intros n; induction n as [| n IH]; intros i s.
  - exact (Fin.case0 (fun s0 ⇒ skip i s0 ≠ i) s).
  - revert s.
    apply (Fin.caseS' i (fun i0 ⇒ ∀ s : Fin.t (S n), skip i0 s ≠ i0)).
    + intros s Heq.
      rewrite (skip_F1 (S n) s) in Heq.
      exact (fin_FS_neq_F1 (S n) s Heq).
    + intros i' s.
      apply (Fin.caseS' s (fun s0 ⇒ skip (Fin.FS i') s0 ≠ Fin.FS i')).
      × intros Heq.
        rewrite (skip_FS_F1 n i') in Heq.
        exact (fin_F1_neq_FS (S n) i' Heq).
      × intros s' Heq.
        rewrite (skip_FS_FS n i' s') in Heq.
        exact (IH i' s' (Fin.FS_inj (skip i' s') i' Heq)).
Qed.


Lemma skip_inj : ∀ (n : nat) (i : Fin.t (S n)) (s1 s2 : Fin.t n),
  skip i s1 = skip i s2 → s1 = s2.
Proof.
  intros n; induction n as [| n IH]; intros i s1 s2 Heq.
  - exact (Fin.case0 (fun s ⇒ s = s2) s1).
  - revert s1 s2 Heq.
    apply (Fin.caseS' i (fun i0 ⇒ ∀ s1 s2 : Fin.t (S n),
      skip i0 s1 = skip i0 s2 → s1 = s2)).
    + intros s1 s2 Heq.
      rewrite (skip_F1 (S n) s1), (skip_F1 (S n) s2) in Heq.
      exact (Fin.FS_inj s1 s2 Heq).
    + intros i' s1 s2 Heq.
      apply (Fin.caseS' s1 (fun s10 ⇒ skip (Fin.FS i') s10 = skip (Fin.FS i') s2 →
        s10 = s2)).
      × apply (Fin.caseS' s2 (fun s20 ⇒ skip (Fin.FS i') Fin.F1 = skip (Fin.FS i') s20 →
          Fin.F1 = s20)).
        -- intros _. reflexivity.
        -- intros s2' Heq2.
           rewrite (skip_FS_F1 n i'), (skip_FS_FS n i' s2') in Heq2.
           destruct (fin_F1_neq_FS (S n) (skip i' s2') Heq2).
      × intros s1'.
        apply (Fin.caseS' s2 (fun s20 ⇒ skip (Fin.FS i') (Fin.FS s1') = skip (Fin.FS i') s20 →
          Fin.FS s1' = s20)).
        -- intros Heq2.
           rewrite (skip_FS_F1 n i'), (skip_FS_FS n i' s1') in Heq2.
           destruct (fin_FS_neq_F1 (S n) (skip i' s1') Heq2).
        -- intros s2' Heq2.
           rewrite (skip_FS_FS n i' s1'), (skip_FS_FS n i' s2') in Heq2.
           exact (f_equal Fin.FS (IH i' s1' s2' (Fin.FS_inj (skip i' s1') (skip i' s2') Heq2))).
      × exact Heq.
Qed.


Fixpoint unskip {n : nat} (r i : Fin.t (S n)) {struct n} : option (Fin.t n) :=
  match n as n0 return Fin.t (S n0) → Fin.t (S n0) → option (Fin.t n0) with
  | O ⇒ fun _ _ ⇒ None
  | S m ⇒ fun r0 i0 ⇒
      Fin.caseS' r0 (fun _ ⇒ option (Fin.t (S m)))
        (Fin.caseS' i0 (fun _ ⇒ option (Fin.t (S m)))
           None
           (fun i' ⇒ Some i'))
        (fun r' : Fin.t (S m) ⇒
           Fin.caseS' i0 (fun _ ⇒ option (Fin.t (S m)))
             (Some Fin.F1)
             (fun i' : Fin.t (S m) ⇒
                match unskip r' i' with
                | Some s' ⇒ Some (Fin.FS s')
                | None ⇒ None
                end))
  end r i.


Lemma unskip_F1_FS : ∀ (n : nat) (i' : Fin.t n),
  unskip Fin.F1 (Fin.FS i') = Some i'.
Proof.
  intros n; destruct n as [| m]; intros i'.
  - exact (Fin.case0 (fun i'0 ⇒ unskip Fin.F1 (Fin.FS i'0) = Some i'0) i').
  - reflexivity.
Qed.

Lemma unskip_FS_F1 : ∀ (n : nat) (r' : Fin.t (S n)),
  unskip (Fin.FS r') Fin.F1 = Some Fin.F1.
Proof.
  intros n r'. reflexivity.
Qed.

Lemma unskip_FS_FS : ∀ (n : nat) (r' i' : Fin.t (S n)),
  unskip (Fin.FS r') (Fin.FS i') =
    match unskip r' i' with
    | Some s' ⇒ Some (Fin.FS s')
    | None ⇒ None
    end.
Proof.
  intros n r' i'. reflexivity.
Qed.

Lemma unskip_F1_F1 : ∀ (n : nat), unskip (n := S n) Fin.F1 Fin.F1 = None.
Proof.
  intros n. reflexivity.
Qed.

Lemma unskip_some : ∀ (n : nat) (r i : Fin.t (S n)),
  r ≠ i → ∃ s : Fin.t n, unskip r i = Some s ∧ skip r s = i.
Proof.
  intros n; induction n as [| n IH]; intros r i Hri.
  - revert i Hri.
    apply (Fin.caseS' r (fun r0 ⇒ ∀ i0 : Fin.t 1, r0 ≠ i0 →
      ∃ s : Fin.t 0, unskip r0 i0 = Some s ∧ skip r0 s = i0)).
    + intros i0.
      apply (Fin.caseS' i0 (fun i1 ⇒ Fin.F1 ≠ i1 →
        ∃ s : Fin.t 0, unskip Fin.F1 i1 = Some s ∧ skip Fin.F1 s = i1)).
      × intros Hc. destruct (Hc eq_refl).
      × intros i' _.
        exact (Fin.case0 (fun i'0 ⇒ ∃ s : Fin.t 0,
          unskip Fin.F1 (Fin.FS i'0) = Some s
          ∧ skip Fin.F1 s = Fin.FS i'0) i').
    + intros r' i0 _.
      exact (Fin.case0 (fun r'0 ⇒ ∃ s : Fin.t 0,
        unskip (Fin.FS r'0) i0 = Some s ∧ skip (Fin.FS r'0) s = i0) r').
  - revert i Hri.
    apply (Fin.caseS' r (fun r0 ⇒ ∀ i0 : Fin.t (S (S n)), r0 ≠ i0 →
      ∃ s : Fin.t (S n), unskip r0 i0 = Some s ∧ skip r0 s = i0)).
    + intros i0.
      apply (Fin.caseS' i0 (fun i1 ⇒ Fin.F1 ≠ i1 →
        ∃ s : Fin.t (S n), unskip Fin.F1 i1 = Some s
          ∧ skip Fin.F1 s = i1)).
      × intros Hc. destruct (Hc eq_refl).
      × intros i' _.
        ∃ i'. split.
        -- exact (unskip_F1_FS (S n) i').
        -- exact (skip_F1 (S n) i').
    + intros r' i0.
      apply (Fin.caseS' i0 (fun i1 ⇒ Fin.FS r' ≠ i1 →
        ∃ s : Fin.t (S n), unskip (Fin.FS r') i1 = Some s
          ∧ skip (Fin.FS r') s = i1)).
      × intros _.
        ∃ Fin.F1. split.
        -- exact (unskip_FS_F1 n r').
        -- exact (skip_FS_F1 n r').
      × intros i' Hne.
        assert (Hne' : r' ≠ i').
        { intros Heq. apply Hne. exact (f_equal Fin.FS Heq). }
        destruct (IH r' i' Hne') as [s' [Hus Hsk]].
        ∃ (Fin.FS s'). split.
        -- rewrite (unskip_FS_FS n r' i'), Hus. reflexivity.
        -- rewrite (skip_FS_FS n r' s'), Hsk. reflexivity.
Qed.


Lemma unskip_skip : ∀ (n : nat) (r : Fin.t (S n)) (s : Fin.t n),
  unskip r (skip r s) = Some s.
Proof.
  intros n r s.
  destruct (unskip_some n r (skip r s) (fun Heq ⇒ skip_ne n r s (eq_sym Heq)))
    as [s' [Hu Hsk]].
  rewrite Hu.
  rewrite (skip_inj n r s' s Hsk).
  reflexivity.
Qed.


Lemma unskip_correct : ∀ (n : nat) (r i : Fin.t (S n)) (s : Fin.t n),
  unskip r i = Some s → skip r s = i.
Proof.
  intros n; induction n as [| n IH]; intros r i s Hu.
  - simpl in Hu. discriminate Hu.
  - revert i Hu.
    apply (Fin.caseS' r (fun r0 ⇒ ∀ i0 : Fin.t (S (S n)),
      unskip r0 i0 = Some s → skip r0 s = i0)).
    + intros i0.
      apply (Fin.caseS' i0 (fun i1 ⇒
        unskip Fin.F1 i1 = Some s → skip Fin.F1 s = i1)).
      × intros Hu0. rewrite (unskip_F1_F1 n) in Hu0. discriminate Hu0.
      × intros i' Hu0.
        rewrite (unskip_F1_FS (S n) i') in Hu0.
        injection Hu0 as Hu0.
        rewrite <- Hu0.
        exact (skip_F1 (S n) i').
    + intros r' i0.
      apply (Fin.caseS' i0 (fun i1 ⇒
        unskip (Fin.FS r') i1 = Some s → skip (Fin.FS r') s = i1)).
      × intros Hu0.
        rewrite (unskip_FS_F1 n r') in Hu0.
        injection Hu0 as Hu0.
        rewrite <- Hu0.
        exact (skip_FS_F1 n r').
      × intros i' Hu0.
        rewrite (unskip_FS_FS n r' i') in Hu0.
        destruct (unskip r' i') as [s' |] eqn:Hu'.
        -- injection Hu0 as Hu0.
           rewrite <- Hu0, (skip_FS_FS n r' s'), (IH r' i' s' Hu').
           reflexivity.
        -- discriminate Hu0.
Qed.


Definition swap_fin {n : nat} (i j r : Fin.t n) : Fin.t n :=
  match Fin.eq_dec r i with
  | left _ ⇒ j
  | right _ ⇒ match Fin.eq_dec r j with
               | left _ ⇒ i
               | right _ ⇒ r
               end
  end.

Lemma swap_fin_at_i : ∀ (n : nat) (i j : Fin.t n), swap_fin i j i = j.
Proof.
  intros n i j.
  unfold swap_fin.
  destruct (Fin.eq_dec i i) as [_ | Hne].
  - reflexivity.
  - destruct (Hne eq_refl).
Qed.

Lemma swap_fin_at_j : ∀ (n : nat) (i j : Fin.t n), i ≠ j → swap_fin i j j = i.
Proof.
  intros n i j Hij.
  unfold swap_fin.
  destruct (Fin.eq_dec j i) as [Heq | _].
  - destruct (Hij (eq_sym Heq)).
  - destruct (Fin.eq_dec j j) as [_ | Hne].
    + reflexivity.
    + destruct (Hne eq_refl).
Qed.

Lemma swap_fin_ne : ∀ (n : nat) (i j r : Fin.t n),
  r ≠ i → r ≠ j → swap_fin i j r = r.
Proof.
  intros n i j r Hri Hrj.
  unfold swap_fin.
  destruct (Fin.eq_dec r i) as [Heq | _].
  - destruct (Hri Heq).
  - destruct (Fin.eq_dec r j) as [Heq2 | _].
    + destruct (Hrj Heq2).
    + reflexivity.
Qed.

Lemma swap_fin_F1_fixed : ∀ (n : nat) (a b : Fin.t n),
  swap_fin (Fin.FS a) (Fin.FS b) Fin.F1 = Fin.F1.
Proof.
  intros n a b.
  exact (swap_fin_ne (S n) (Fin.FS a) (Fin.FS b) Fin.F1
           (fin_F1_neq_FS n a) (fin_F1_neq_FS n b)).
Qed.

Lemma swap_fin_FS : ∀ (n : nat) (a b y : Fin.t n),
  swap_fin (Fin.FS a) (Fin.FS b) (Fin.FS y) = Fin.FS (swap_fin a b y).
Proof.
  intros n a b y.
  destruct (Fin.eq_dec y a) as [Heq | Hne].
  - rewrite Heq, (swap_fin_at_i n a b), (swap_fin_at_i (S n) (Fin.FS a) (Fin.FS b)).
    reflexivity.
  - destruct (Fin.eq_dec y b) as [Heq2 | Hne2].
    + rewrite Heq2.
      destruct (Fin.eq_dec a b) as [Heqab | Hneab].
      × rewrite Heqab, (swap_fin_at_i n b b), (swap_fin_at_i (S n) (Fin.FS b) (Fin.FS b)).
        reflexivity.
      × rewrite (swap_fin_at_j n a b Hneab),
          (swap_fin_at_j (S n) (Fin.FS a) (Fin.FS b)
             (fun Heq ⇒ Hneab (Fin.FS_inj a b Heq))).
        reflexivity.
    + rewrite (swap_fin_ne n a b y Hne Hne2),
        (swap_fin_ne (S n) (Fin.FS a) (Fin.FS b) (Fin.FS y)
           (fun Heq ⇒ Hne (Fin.FS_inj y a Heq))
           (fun Heq ⇒ Hne2 (Fin.FS_inj y b Heq))).
      reflexivity.
Qed.

Lemma swap_fin_invol : ∀ (n : nat) (i j x : Fin.t n), swap_fin i j (swap_fin i j x) = x.
Proof.
  intros n i j x.
  destruct (Fin.eq_dec x i) as [Hxi | Hxi].
  - rewrite Hxi, (swap_fin_at_i n i j).
    destruct (Fin.eq_dec i j) as [Hij | Hij].
    + rewrite <- Hij. exact (swap_fin_at_i n i i).
    + exact (swap_fin_at_j n i j Hij).
  - destruct (Fin.eq_dec x j) as [Hxj | Hxj].
    + assert (Hij : i ≠ j).
      { intros Heq. apply Hxi. rewrite Hxj. symmetry. exact Heq. }
      rewrite Hxj, (swap_fin_at_j n i j Hij), (swap_fin_at_i n i j).
      reflexivity.
    + rewrite (swap_fin_ne n i j x Hxi Hxj), (swap_fin_ne n i j x Hxi Hxj).
      reflexivity.
Qed.

Lemma swap_fin_comm : ∀ (n : nat) (i j x : Fin.t n), swap_fin i j x = swap_fin j i x.
Proof.
  intros n i j x.
  destruct (Fin.eq_dec x i) as [Hxi | Hxi].
  - rewrite Hxi, (swap_fin_at_i n i j).
    destruct (Fin.eq_dec i j) as [Hij | Hij].
    + rewrite Hij, (swap_fin_at_i n j j). reflexivity.
    + exact (eq_sym (swap_fin_at_j n j i (fun Heq ⇒ Hij (eq_sym Heq)))).
  - destruct (Fin.eq_dec x j) as [Hxj | Hxj].
    + rewrite Hxj.
      destruct (Fin.eq_dec i j) as [Hij | Hij].
      × rewrite Hij, (swap_fin_at_i n j j). reflexivity.
      × rewrite (swap_fin_at_j n i j Hij).
        exact (eq_sym (swap_fin_at_i n j i)).
    + rewrite (swap_fin_ne n i j x Hxi Hxj), (swap_fin_ne n j i x Hxj Hxi).
      reflexivity.
Qed.

Lemma swap_fin_conj : ∀ (n : nat) (i j k : Fin.t n),
  i ≠ j → i ≠ k → j ≠ k →
  ∀ x : Fin.t n,
  swap_fin i j x = swap_fin i k (swap_fin k j (swap_fin i k x)).
Proof.
  intros n i j k Hij Hik Hjk x.
  destruct (Fin.eq_dec x i) as [Hxi | Hxi].
  - rewrite Hxi, (swap_fin_at_i n i j), (swap_fin_at_i n i k), (swap_fin_at_i n k j).
    exact (eq_sym (swap_fin_ne n i k j (fun Heq ⇒ Hij (eq_sym Heq)) Hjk)).
  - destruct (Fin.eq_dec x j) as [Hxj | Hxj].
    + rewrite Hxj, (swap_fin_at_j n i j Hij),
        (swap_fin_ne n i k j (fun Heq ⇒ Hij (eq_sym Heq)) Hjk),
        (swap_fin_at_j n k j (fun Heq ⇒ Hjk (eq_sym Heq))).
      exact (eq_sym (swap_fin_at_j n i k Hik)).
    + destruct (Fin.eq_dec x k) as [Hxk | Hxk].
      × rewrite Hxk, (swap_fin_ne n i j k (fun Heq ⇒ Hik (eq_sym Heq)) (fun Heq ⇒ Hjk (eq_sym Heq))),
          (swap_fin_at_j n i k Hik), (swap_fin_ne n k j i Hik Hij).
        exact (eq_sym (swap_fin_at_i n i k)).
      × rewrite (swap_fin_ne n i k x Hxi Hxk), (swap_fin_ne n k j x Hxk Hxj),
          (swap_fin_ne n i k x Hxi Hxk).
        exact (swap_fin_ne n i j x Hxi Hxj).
Qed.


Fixpoint fweaken {n : nat} (k : Fin.t n) {struct k} : Fin.t (S n) :=
  match k with
  | Fin.F1 ⇒ Fin.F1
  | Fin.FS k' ⇒ Fin.FS (fweaken k')
  end.

Lemma fweaken_F1 : ∀ n : nat, @fweaken (S n) Fin.F1 = Fin.F1.
Proof. intros n. reflexivity. Qed.

Lemma fweaken_FS : ∀ (n : nat) (k' : Fin.t n),
  fweaken (Fin.FS k') = Fin.FS (fweaken k').
Proof. intros n k'. reflexivity. Qed.


Lemma skip_fweaken_self : ∀ (n : nat) (k : Fin.t n),
  skip (fweaken k) k = Fin.FS k.
Proof.
  intros n k; induction k as [n0 | n0 k' IH].
  - exact (skip_F1 (S n0) Fin.F1).
  - rewrite (fweaken_FS n0 k'), (skip_FS_FS n0 (fweaken k') k'), IH.
    reflexivity.
Qed.

Lemma skip_FS_self : ∀ (n : nat) (k : Fin.t n),
  skip (Fin.FS k) k = fweaken k.
Proof.
  intros n k; induction k as [n0 | n0 k' IH].
  - exact (skip_FS_F1 n0 Fin.F1).
  - rewrite (fweaken_FS n0 k'), (skip_FS_FS n0 (Fin.FS k') k'), IH.
    reflexivity.
Qed.

Lemma fweaken_neq_FS : ∀ (n : nat) (k : Fin.t n), fweaken k ≠ Fin.FS k.
Proof.
  intros n k; induction k as [n0 | n0 k' IH].
  - exact (fin_F1_neq_FS (S n0) Fin.F1).
  - rewrite (fweaken_FS n0 k').
    intros Heq.
    exact (IH (Fin.FS_inj (fweaken k') (Fin.FS k') Heq)).
Qed.


Lemma skip_fweaken_neq : ∀ (n : nat) (k y : Fin.t n),
  y ≠ k → skip (fweaken k) y ≠ Fin.FS k.
Proof.
  intros n k y Hne Heq.
  apply Hne.
  apply (skip_inj n (fweaken k) y k).
  rewrite Heq.
  symmetry.
  exact (skip_fweaken_self n k).
Qed.

Lemma skip_FS_neq : ∀ (n : nat) (k y : Fin.t n),
  y ≠ k → skip (Fin.FS k) y ≠ fweaken k.
Proof.
  intros n k y Hne Heq.
  apply Hne.
  apply (skip_inj n (Fin.FS k) y k).
  rewrite Heq.
  symmetry.
  exact (skip_FS_self n k).
Qed.


Lemma skip_fweaken_eq_skip_FS : ∀ (n : nat) (k y : Fin.t n),
  y ≠ k → skip (fweaken k) y = skip (Fin.FS k) y.
Proof.
  intros n k; induction k as [n0 | n0 k' IH]; intros y Hne.
  - revert Hne.
    apply (Fin.caseS' y (fun y0 ⇒ y0 ≠ Fin.F1 →
      skip (fweaken (n := S n0) Fin.F1) y0 = skip (Fin.FS (n := S n0) Fin.F1) y0)).
    + intros Hc. destruct (Hc eq_refl).
    + intros y' _.
      rewrite (fweaken_F1 n0), (skip_F1 (S n0) (Fin.FS y')),
        (skip_FS_FS n0 Fin.F1 y'), (skip_F1 n0 y').
      reflexivity.
  - revert Hne.
    apply (Fin.caseS' y (fun y0 ⇒ y0 ≠ Fin.FS k' →
      skip (fweaken (n := S n0) (Fin.FS k')) y0 = skip (Fin.FS (n := S n0) (Fin.FS k')) y0)).
    + intros _.
      rewrite (fweaken_FS n0 k'), (skip_FS_F1 n0 (fweaken k')), (skip_FS_F1 n0 (Fin.FS k')).
      reflexivity.
    + intros y' Hne'.
      assert (Hne'' : y' ≠ k').
      { intros Heq. apply Hne'. exact (f_equal Fin.FS Heq). }
      rewrite (fweaken_FS n0 k'), (skip_FS_FS n0 (fweaken k') y'),
        (skip_FS_FS n0 (Fin.FS k') y'), (IH y' Hne'').
      reflexivity.
Qed.


Lemma unskip_fweaken_FS : ∀ (n : nat) (k : Fin.t n),
  unskip (fweaken k) (Fin.FS k) = Some k.
Proof.
  intros n k; induction k as [n0 | n0 k' IH].
  - exact (unskip_F1_FS (S n0) Fin.F1).
  - rewrite (fweaken_FS n0 k'), (unskip_FS_FS n0 (fweaken k') (Fin.FS k')), IH.
    reflexivity.
Qed.

Lemma unskip_FS_fweaken : ∀ (n : nat) (k : Fin.t n),
  unskip (Fin.FS k) (fweaken k) = Some k.
Proof.
  intros n k; induction k as [n0 | n0 k' IH].
  - exact (unskip_FS_F1 n0 Fin.F1).
  - rewrite (fweaken_FS n0 k'), (unskip_FS_FS n0 (Fin.FS k') (fweaken k')), IH.
    reflexivity.
Qed.


Lemma unskip_adjacent_shape : ∀ (n0 : nat) (k : Fin.t (S n0)) (c : Fin.t (S (S n0))),
  c ≠ fweaken k → c ≠ Fin.FS k →
  ∃ k' : Fin.t n0,
    unskip c (fweaken k) = Some (fweaken k') ∧ unskip c (Fin.FS k) = Some (Fin.FS k').
Proof.
  intros n0; induction n0 as [| n0 IHn]; intros k c Hci Hcj.
  - revert c Hci Hcj.
    apply (Fin.caseS' k (fun k0 ⇒ ∀ c0 : Fin.t 2,
      c0 ≠ fweaken k0 → c0 ≠ Fin.FS k0 →
      ∃ w : Fin.t 0, unskip c0 (fweaken k0) = Some (fweaken w)
        ∧ unskip c0 (Fin.FS k0) = Some (Fin.FS w))).
    + intros c0.
      apply (Fin.caseS' c0 (fun c1 ⇒
        c1 ≠ fweaken (n := 1) Fin.F1 → c1 ≠ Fin.FS Fin.F1 →
        ∃ w : Fin.t 0, unskip c1 (fweaken (n := 1) Fin.F1) = Some (fweaken w)
          ∧ unskip c1 (Fin.FS Fin.F1) = Some (Fin.FS w))).
      × intros Hc _. rewrite fweaken_F1 in Hc. destruct (Hc eq_refl).
      × intros c1.
        apply (Fin.caseS' c1 (fun c2 ⇒
          Fin.FS c2 ≠ fweaken (n := 1) Fin.F1 → Fin.FS c2 ≠ Fin.FS Fin.F1 →
          ∃ w : Fin.t 0, unskip (Fin.FS c2) (fweaken (n := 1) Fin.F1) = Some (fweaken w)
            ∧ unskip (Fin.FS c2) (Fin.FS Fin.F1) = Some (Fin.FS w))).
        -- intros _ Hc. destruct (Hc eq_refl).
        -- intros c3.
           exact (Fin.case0 (fun c4 ⇒
             Fin.FS (Fin.FS c4) ≠ fweaken (n := 1) Fin.F1 → Fin.FS (Fin.FS c4) ≠ Fin.FS Fin.F1 →
             ∃ w : Fin.t 0, unskip (Fin.FS (Fin.FS c4)) (fweaken (n := 1) Fin.F1) = Some (fweaken w)
               ∧ unskip (Fin.FS (Fin.FS c4)) (Fin.FS Fin.F1) = Some (Fin.FS w)) c3).
    + intros k0. exact (Fin.case0 (fun k1 ⇒ ∀ c0 : Fin.t 2,
        c0 ≠ fweaken (Fin.FS k1) → c0 ≠ Fin.FS (Fin.FS k1) →
        ∃ w : Fin.t 0, unskip c0 (fweaken (Fin.FS k1)) = Some (fweaken w)
          ∧ unskip c0 (Fin.FS (Fin.FS k1)) = Some (Fin.FS w)) k0).
  - revert c Hci Hcj.
    apply (Fin.caseS' k (fun k0 ⇒ ∀ c0 : Fin.t (S (S (S n0))),
      c0 ≠ fweaken k0 → c0 ≠ Fin.FS k0 →
      ∃ w : Fin.t (S n0), unskip c0 (fweaken k0) = Some (fweaken w)
        ∧ unskip c0 (Fin.FS k0) = Some (Fin.FS w))).
    + intros c0.
      apply (Fin.caseS' c0 (fun c1 ⇒
        c1 ≠ fweaken Fin.F1 → c1 ≠ Fin.FS Fin.F1 →
        ∃ w : Fin.t (S n0), unskip c1 (fweaken Fin.F1) = Some (fweaken w)
          ∧ unskip c1 (Fin.FS Fin.F1) = Some (Fin.FS w))).
      × intros Hc _. rewrite fweaken_F1 in Hc. destruct (Hc eq_refl).
      × intros c1 _ Hcj.
        revert Hcj.
        apply (Fin.caseS' c1 (fun c2 ⇒
          Fin.FS c2 ≠ Fin.FS Fin.F1 →
          ∃ w : Fin.t (S n0), unskip (Fin.FS c2) (fweaken Fin.F1) = Some (fweaken w)
            ∧ unskip (Fin.FS c2) (Fin.FS Fin.F1) = Some (Fin.FS w))).
        -- intros Hc. destruct (Hc eq_refl).
        -- intros c2 _.
           ∃ Fin.F1.
           rewrite fweaken_F1.
           split.
           ++ exact (unskip_FS_F1 (S n0) (Fin.FS c2)).
           ++ rewrite (unskip_FS_FS (S n0) (Fin.FS c2) Fin.F1),
                (unskip_FS_F1 n0 c2).
              reflexivity.
    + intros k'.
      intros c0.
      apply (Fin.caseS' c0 (fun c1 ⇒
        c1 ≠ fweaken (Fin.FS k') → c1 ≠ Fin.FS (Fin.FS k') →
        ∃ w : Fin.t (S n0), unskip c1 (fweaken (Fin.FS k')) = Some (fweaken w)
          ∧ unskip c1 (Fin.FS (Fin.FS k')) = Some (Fin.FS w))).
      × intros _ _.
        ∃ k'.
        rewrite fweaken_FS.
        split.
        -- exact (unskip_F1_FS (S (S n0)) (fweaken k')).
        -- exact (unskip_F1_FS (S (S n0)) (Fin.FS k')).
      × intros c1 Hci Hcj.
        assert (Hci' : c1 ≠ fweaken k').
        { intros Heq. apply Hci. rewrite fweaken_FS. exact (f_equal Fin.FS Heq). }
        assert (Hcj' : c1 ≠ Fin.FS k').
        { intros Heq. apply Hcj. exact (f_equal Fin.FS Heq). }
        destruct (IHn k' c1 Hci' Hcj') as [w [Hu1 Hu2]].
        ∃ (Fin.FS w).
        rewrite fweaken_FS.
        split.
        -- rewrite (unskip_FS_FS (S n0) c1 (fweaken k')), Hu1, fweaken_FS.
           reflexivity.
        -- rewrite (unskip_FS_FS (S n0) c1 (Fin.FS k')), Hu2.
           reflexivity.
Qed.


Definition bstep2 {m : nat} (c : Fin.t (S m))
    (tau tau' : Fin.t (S m) → Fin.t (S m)) (y : Fin.t m) : Fin.t m :=
  match unskip (tau c) (tau (skip c y)) with
  | Some t ⇒ t
  | None ⇒ y
  end.

Definition bstep2' {m : nat} (c : Fin.t (S m))
    (tau tau' : Fin.t (S m) → Fin.t (S m)) (t : Fin.t m) : Fin.t m :=
  match unskip c (tau' (skip (tau c) t)) with
  | Some s ⇒ s
  | None ⇒ t
  end.

Lemma bstep2_spec : ∀ (m : nat) (c : Fin.t (S m))
    (tau tau' : Fin.t (S m) → Fin.t (S m)),
  (∀ x, tau' (tau x) = x) → (∀ y, tau (tau' y) = y) →
  ∀ y : Fin.t m, skip (tau c) (bstep2 c tau tau' y) = tau (skip c y).
Proof.
  intros m c tau tau' H1 H2 y.
  assert (Hne : tau c ≠ tau (skip c y)).
  { intros Heq.
    assert (Hc : c = skip c y).
    { transitivity (tau' (tau c)).
      - symmetry. exact (H1 c).
      - rewrite Heq. exact (H1 (skip c y)). }
    exact (skip_ne m c y (eq_sym Hc)). }
  destruct (unskip_some m (tau c) (tau (skip c y)) Hne) as [t [Hu Hsk]].
  unfold bstep2. rewrite Hu. exact Hsk.
Qed.

Lemma bstep2'_spec : ∀ (m : nat) (c : Fin.t (S m))
    (tau tau' : Fin.t (S m) → Fin.t (S m)),
  (∀ x, tau' (tau x) = x) → (∀ y, tau (tau' y) = y) →
  ∀ t : Fin.t m, skip c (bstep2' c tau tau' t) = tau' (skip (tau c) t).
Proof.
  intros m c tau tau' H1 H2 t.
  assert (Hne : c ≠ tau' (skip (tau c) t)).
  { intros Heq.
    assert (Hp : tau c = skip (tau c) t).
    { transitivity (tau (tau' (skip (tau c) t))).
      - exact (f_equal tau Heq).
      - exact (H2 (skip (tau c) t)). }
    exact (skip_ne m (tau c) t (eq_sym Hp)). }
  destruct (unskip_some m c (tau' (skip (tau c) t)) Hne) as [s [Hu Hsk]].
  unfold bstep2'. rewrite Hu. exact Hsk.
Qed.

Lemma bstep2_bstep2' : ∀ (m : nat) (c : Fin.t (S m))
    (tau tau' : Fin.t (S m) → Fin.t (S m)),
  (∀ x, tau' (tau x) = x) → (∀ y, tau (tau' y) = y) →
  ∀ y : Fin.t m, bstep2' c tau tau' (bstep2 c tau tau' y) = y.
Proof.
  intros m c tau tau' H1 H2 y.
  apply (skip_inj m c _ y).
  rewrite (bstep2'_spec m c tau tau' H1 H2 (bstep2 c tau tau' y)),
    (bstep2_spec m c tau tau' H1 H2 y), (H1 (skip c y)).
  reflexivity.
Qed.

Lemma bstep2'_bstep2 : ∀ (m : nat) (c : Fin.t (S m))
    (tau tau' : Fin.t (S m) → Fin.t (S m)),
  (∀ x, tau' (tau x) = x) → (∀ y, tau (tau' y) = y) →
  ∀ t : Fin.t m, bstep2 c tau tau' (bstep2' c tau tau' t) = t.
Proof.
  intros m c tau tau' H1 H2 t.
  apply (skip_inj m (tau c) _ t).
  rewrite (bstep2_spec m c tau tau' H1 H2 (bstep2' c tau tau' t)),
    (bstep2'_spec m c tau tau' H1 H2 t), (H2 (skip (tau c) t)).
  reflexivity.
Qed.


Lemma bstep2_id : ∀ (n : nat) (c : Fin.t (S n)) (tau tau' : Fin.t (S n) → Fin.t (S n)),
  (∀ x, tau x = x) → ∀ y : Fin.t n, bstep2 c tau tau' y = y.
Proof.
  intros n c tau tau' Htau y.
  unfold bstep2.
  rewrite (Htau c), (Htau (skip c y)), (unskip_skip n c y).
  reflexivity.
Qed.

Lemma bstep2'_id : ∀ (n : nat) (c : Fin.t (S n)) (tau tau' : Fin.t (S n) → Fin.t (S n)),
  (∀ x, tau x = x) → (∀ x, tau' x = x) →
  ∀ t : Fin.t n, bstep2' c tau tau' t = t.
Proof.
  intros n c tau tau' Htau Htau' t.
  unfold bstep2'.
  rewrite (Htau c), (Htau' (skip c t)), (unskip_skip n c t).
  reflexivity.
Qed.


Lemma bstep2_ext : ∀ (n : nat) (c : Fin.t (S n)) (tau1 tau2 tau1' tau2' : Fin.t (S n) → Fin.t (S n)),
  (∀ x, tau1 x = tau2 x) → ∀ y : Fin.t n, bstep2 c tau1 tau1' y = bstep2 c tau2 tau2' y.
Proof.
  intros n c tau1 tau2 tau1' tau2' Heq y.
  unfold bstep2.
  rewrite (Heq c), (Heq (skip c y)).
  reflexivity.
Qed.


Lemma bstep2_compose : ∀ (m : nat) (c : Fin.t (S m))
    (tau1 tau1' tau2 tau2' : Fin.t (S m) → Fin.t (S m)),
  (∀ x, tau1' (tau1 x) = x) → (∀ y, tau1 (tau1' y) = y) →
  (∀ x, tau2' (tau2 x) = x) → (∀ y, tau2 (tau2' y) = y) →
  ∀ y : Fin.t m,
  bstep2 c (fun x ⇒ tau2 (tau1 x)) (fun x ⇒ tau1' (tau2' x)) y
    = bstep2 (tau1 c) tau2 tau2' (bstep2 c tau1 tau1' y).
Proof.
  intros m c tau1 tau1' tau2 tau2' Ha1 Ha2 Hb1 Hb2 y.
  assert (Hspec : skip (tau1 c) (bstep2 c tau1 tau1' y) = tau1 (skip c y)).
  { exact (bstep2_spec m c tau1 tau1' Ha1 Ha2 y). }
  assert (Hne : tau2 (tau1 c) ≠ tau2 (tau1 (skip c y))).
  { intros Heq.
    assert (Htau1eq : tau1 c = tau1 (skip c y)).
    { assert (Heq2 := f_equal tau2' Heq).
      rewrite (Hb1 (tau1 c)), (Hb1 (tau1 (skip c y))) in Heq2.
      exact Heq2. }
    assert (Hceq : c = skip c y).
    { assert (Heq3 := f_equal tau1' Htau1eq).
      rewrite (Ha1 c), (Ha1 (skip c y)) in Heq3.
      exact Heq3. }
    exact (skip_ne m c y (eq_sym Hceq)). }
  destruct (unskip_some m (tau2 (tau1 c)) (tau2 (tau1 (skip c y))) Hne) as [t [Hu Hsk]].
  unfold bstep2 at 1 2.
  rewrite Hu, Hspec, Hu.
  reflexivity.
Qed.


Lemma bstep2_swap_adjacent_left : ∀ (n : nat) (k : Fin.t n) (y : Fin.t n),
  bstep2 (fweaken k) (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k)) y = y.
Proof.
  intros n k; induction k as [n0 | n0 k' IH]; intros y.
  - apply (Fin.caseS' y (fun y0 ⇒
      bstep2 (fweaken Fin.F1) (swap_fin (fweaken Fin.F1) (Fin.FS Fin.F1))
        (swap_fin (fweaken Fin.F1) (Fin.FS Fin.F1)) y0 = y0)).
    + unfold bstep2.
      rewrite fweaken_F1, swap_fin_at_i, skip_F1,
        (swap_fin_at_j _ Fin.F1 (Fin.FS Fin.F1) (fin_F1_neq_FS _ Fin.F1)),
        unskip_FS_F1.
      reflexivity.
    + intros y'.
      unfold bstep2.
      rewrite fweaken_F1, swap_fin_at_i, skip_F1,
        (swap_fin_ne _ Fin.F1 (Fin.FS Fin.F1) (Fin.FS (Fin.FS y'))
           (fin_FS_neq_F1 _ (Fin.FS y'))
           (fun Heq ⇒ fin_FS_neq_F1 _ y' (Fin.FS_inj (Fin.FS y') Fin.F1 Heq))),
        unskip_FS_FS, unskip_F1_FS.
      reflexivity.
  - apply (Fin.caseS' y (fun y0 ⇒
      bstep2 (fweaken (Fin.FS k')) (swap_fin (fweaken (Fin.FS k')) (Fin.FS (Fin.FS k')))
        (swap_fin (fweaken (Fin.FS k')) (Fin.FS (Fin.FS k'))) y0 = y0)).
    + unfold bstep2.
      rewrite fweaken_FS, swap_fin_at_i, skip_FS_F1, swap_fin_F1_fixed, unskip_FS_F1.
      reflexivity.
    + intros y'.
      unfold bstep2.
      rewrite fweaken_FS, swap_fin_at_i, skip_FS_FS, swap_fin_FS.
      destruct (Fin.eq_dec y' k') as [Heq | Hne].
      × rewrite Heq, skip_fweaken_self,
          (swap_fin_at_j _ (fweaken k') (Fin.FS k')
             (fun Heq2 ⇒ fweaken_neq_FS _ k' Heq2)),
          unskip_FS_FS, unskip_FS_fweaken.
        reflexivity.
      × rewrite (swap_fin_ne _ (fweaken k') (Fin.FS k') (skip (fweaken k') y')
                   (skip_ne _ (fweaken k') y') (skip_fweaken_neq _ k' y' Hne)),
          unskip_FS_FS.
        rewrite (skip_fweaken_eq_skip_FS _ k' y' Hne), unskip_skip.
        reflexivity.
Qed.

Lemma bstep2_swap_adjacent_right : ∀ (n : nat) (k : Fin.t n) (y : Fin.t n),
  bstep2 (Fin.FS k) (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k)) y = y.
Proof.
  intros n k; induction k as [n0 | n0 k' IH]; intros y.
  - apply (Fin.caseS' y (fun y0 ⇒
      bstep2 (Fin.FS Fin.F1) (swap_fin (fweaken Fin.F1) (Fin.FS Fin.F1))
        (swap_fin (fweaken Fin.F1) (Fin.FS Fin.F1)) y0 = y0)).
    + unfold bstep2.
      rewrite fweaken_F1,
        (swap_fin_at_j _ Fin.F1 (Fin.FS Fin.F1) (fin_F1_neq_FS _ Fin.F1)),
        skip_FS_F1, swap_fin_at_i, unskip_F1_FS.
      reflexivity.
    + intros y'.
      unfold bstep2.
      rewrite fweaken_F1,
        (swap_fin_at_j _ Fin.F1 (Fin.FS Fin.F1) (fin_F1_neq_FS _ Fin.F1)),
        skip_FS_FS, skip_F1,
        (swap_fin_ne _ Fin.F1 (Fin.FS Fin.F1) (Fin.FS (Fin.FS y'))
           (fin_FS_neq_F1 _ (Fin.FS y'))
           (fun Heq ⇒ fin_FS_neq_F1 _ y' (Fin.FS_inj (Fin.FS y') Fin.F1 Heq))),
        unskip_F1_FS.
      reflexivity.
  - apply (Fin.caseS' y (fun y0 ⇒
      bstep2 (Fin.FS (Fin.FS k')) (swap_fin (fweaken (Fin.FS k')) (Fin.FS (Fin.FS k')))
        (swap_fin (fweaken (Fin.FS k')) (Fin.FS (Fin.FS k'))) y0 = y0)).
    + unfold bstep2.
      rewrite fweaken_FS,
        (swap_fin_at_j _ (Fin.FS (fweaken k')) (Fin.FS (Fin.FS k'))
           (fun Heq2 ⇒ fweaken_neq_FS _ k' (Fin.FS_inj (fweaken k') (Fin.FS k') Heq2))),
        skip_FS_F1, swap_fin_F1_fixed, unskip_FS_F1.
      reflexivity.
    + intros y'.
      unfold bstep2.
      rewrite fweaken_FS,
        (swap_fin_at_j _ (Fin.FS (fweaken k')) (Fin.FS (Fin.FS k'))
           (fun Heq2 ⇒ fweaken_neq_FS _ k' (Fin.FS_inj (fweaken k') (Fin.FS k') Heq2))),
        skip_FS_FS, swap_fin_FS.
      destruct (Fin.eq_dec y' k') as [Heq | Hne].
      × rewrite Heq, skip_FS_self,
          (swap_fin_at_i _ (fweaken k') (Fin.FS k')),
          unskip_FS_FS, unskip_fweaken_FS.
        reflexivity.
      × rewrite (swap_fin_ne _ (fweaken k') (Fin.FS k') (skip (Fin.FS k') y')
                   (skip_FS_neq _ k' y' Hne) (skip_ne _ (Fin.FS k') y')),
          unskip_FS_FS.
        rewrite <- (skip_fweaken_eq_skip_FS _ k' y' Hne), unskip_skip.
        reflexivity.
Qed.

Lemma bstep2_swap_adjacent_ne : ∀ (n0 : nat) (k : Fin.t (S n0)) (c : Fin.t (S (S n0)))
    (k' : Fin.t n0),
  c ≠ fweaken k → c ≠ Fin.FS k →
  unskip c (fweaken k) = Some (fweaken k') → unskip c (Fin.FS k) = Some (Fin.FS k') →
  ∀ y : Fin.t (S n0),
  bstep2 c (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k)) y
    = swap_fin (fweaken k') (Fin.FS k') y.
Proof.
  intros n0 k c k' Hci Hcj Hu1 Hu2 y.
  unfold bstep2.
  rewrite (swap_fin_ne _ (fweaken k) (Fin.FS k) c Hci Hcj).
  destruct (Fin.eq_dec y (fweaken k')) as [Heq | Hne].
  - rewrite Heq, (unskip_correct (S n0) c (fweaken k) (fweaken k') Hu1),
      (swap_fin_at_i _ (fweaken k) (Fin.FS k)), Hu2,
      (swap_fin_at_i _ (fweaken k') (Fin.FS k')).
    reflexivity.
  - destruct (Fin.eq_dec y (Fin.FS k')) as [Heq2 | Hne2].
    + rewrite Heq2, (unskip_correct (S n0) c (Fin.FS k) (Fin.FS k') Hu2),
        (swap_fin_at_j _ (fweaken k) (Fin.FS k) (fweaken_neq_FS (S n0) k)),
        Hu1, (swap_fin_at_j _ (fweaken k') (Fin.FS k') (fweaken_neq_FS n0 k')).
      reflexivity.
    + assert (Hney : skip c y ≠ fweaken k).
      { intros Heq3. apply Hne.
        apply (skip_inj (S n0) c y (fweaken k')).
        rewrite Heq3.
        exact (eq_sym (unskip_correct (S n0) c (fweaken k) (fweaken k') Hu1)). }
      assert (Hney2 : skip c y ≠ Fin.FS k).
      { intros Heq3. apply Hne2.
        apply (skip_inj (S n0) c y (Fin.FS k')).
        rewrite Heq3.
        exact (eq_sym (unskip_correct (S n0) c (Fin.FS k) (Fin.FS k') Hu2)). }
      rewrite (swap_fin_ne _ (fweaken k) (Fin.FS k) (skip c y) Hney Hney2),
        (unskip_skip (S n0) c y),
        (swap_fin_ne _ (fweaken k') (Fin.FS k') y Hne Hne2).
      reflexivity.
Qed.


Fixpoint fin_to_nat {n : nat} (i : Fin.t n) : nat :=
  match i with
  | Fin.F1 ⇒ O
  | Fin.FS j ⇒ S (fin_to_nat j)
  end.

Lemma fin_to_nat_F1 : ∀ n : nat, fin_to_nat (@Fin.F1 n) = O.
Proof. intros n. reflexivity. Qed.

Lemma fin_to_nat_FS : ∀ (n : nat) (i : Fin.t n),
  fin_to_nat (Fin.FS i) = S (fin_to_nat i).
Proof.
  intros n i.
  reflexivity.
Qed.

Lemma fin_to_nat_inj : ∀ (n : nat) (i j : Fin.t n),
  fin_to_nat i = fin_to_nat j → i = j.
Proof.
  intros n; induction n as [| n IH]; intros i j.
  - exact (Fin.case0 (fun _ ⇒ fin_to_nat i = fin_to_nat j → i = j) i).
  - apply (Fin.caseS' i (fun i0 ⇒ fin_to_nat i0 = fin_to_nat j → i0 = j)).
    + apply (Fin.caseS' j (fun j0 ⇒ fin_to_nat Fin.F1 = fin_to_nat j0 → Fin.F1 = j0)).
      × intros _. reflexivity.
      × intros j' Heq.
        rewrite (fin_to_nat_FS n j') in Heq.
        discriminate Heq.
    + intros i'.
      apply (Fin.caseS' j (fun j0 ⇒
               fin_to_nat (Fin.FS i') = fin_to_nat j0 → Fin.FS i' = j0)).
      × intros Heq.
        rewrite (fin_to_nat_FS n i') in Heq.
        discriminate Heq.
      × intros j' Heq.
        rewrite (fin_to_nat_FS n i'), (fin_to_nat_FS n j') in Heq.
        rewrite (IH i' j' (eq_add_S (fin_to_nat i') (fin_to_nat j') Heq)).
        reflexivity.
Qed.

Lemma fin_to_nat_bound : ∀ (n : nat) (i : Fin.t n), (fin_to_nat i < n)%nat.
Proof.
  intros n; induction n as [| n IH]; intros i.
  - exact (Fin.case0 (fun i0 ⇒ (fin_to_nat i0 < 0)%nat) i).
  - apply (Fin.caseS' i (fun i0 ⇒ (fin_to_nat i0 < S n)%nat)).
    + rewrite (fin_to_nat_F1 n). lia.
    + intros i'.
      rewrite (fin_to_nat_FS n i').
      assert (H := IH i'). lia.
Qed.

Lemma fin_to_nat_fweaken : ∀ (n : nat) (k : Fin.t n), fin_to_nat (fweaken k) = fin_to_nat k.
Proof.
  intros n; induction n as [| n IH]; intros k.
  - exact (Fin.case0 (fun k0 ⇒ fin_to_nat (fweaken k0) = fin_to_nat k0) k).
  - apply (Fin.caseS' k (fun k0 ⇒ fin_to_nat (fweaken k0) = fin_to_nat k0)).
    + reflexivity.
    + intros k'.
      rewrite (fweaken_FS n k'), (fin_to_nat_FS (S n) (fweaken k')), (fin_to_nat_FS n k').
      exact (f_equal S (IH k')).
Qed.

Lemma fweaken_surj : ∀ (n0 : nat) (i : Fin.t (S n0)),
  (fin_to_nat i < n0)%nat → ∃ k : Fin.t n0, i = fweaken k.
Proof.
  intros n0; induction n0 as [| m IH]; intros i Hlt.
  - exfalso. lia.
  - revert Hlt.
    apply (Fin.caseS' i (fun i0 ⇒ (fin_to_nat i0 < S m)%nat → ∃ k : Fin.t (S m), i0 = fweaken k)).
    + intros _. ∃ Fin.F1. symmetry. exact (fweaken_F1 m).
    + intros i' Hlt.
      assert (Hlt' : (fin_to_nat i' < m)%nat).
      { rewrite (fin_to_nat_FS (S m) i') in Hlt. lia. }
      destruct (IH i' Hlt') as [k' Hk'].
      ∃ (Fin.FS k').
      rewrite Hk'. symmetry. exact (fweaken_FS m k').
Qed.


Fixpoint fin_all (n : nat) : list (Fin.t n) :=
  match n with
  | O ⇒ nil
  | S m ⇒ cons Fin.F1 (map Fin.FS (fin_all m))
  end.

Lemma fin_all_complete : ∀ (n : nat) (i : Fin.t n), In i (fin_all n).
Proof.
  intros n; induction n as [| m IH]; intros i.
  - exact (Fin.case0 (fun i0 : Fin.t 0 ⇒ In i0 (fin_all 0)) i).
  - apply (Fin.caseS' i (fun i0 ⇒ In i0 (fin_all (S m)))).
    + left. reflexivity.
    + intros s. right. apply in_map. exact (IH s).
Qed.

Lemma NoDup_map_FS : ∀ (n : nat) (l : list (Fin.t n)),
  NoDup l → NoDup (map Fin.FS l).
Proof.
  intros n l; induction l as [| a l' IH]; intros Hnd.
  - constructor.
  - inversion Hnd as [| x xs Hnotin Hnd' Hxs]; subst.
    simpl. constructor.
    + intros Hin. apply in_map_iff in Hin.
      destruct Hin as [x [Hx Hinx]].
      apply Hnotin.
      rewrite <- (Fin.FS_inj x a Hx). exact Hinx.
    + exact (IH Hnd').
Qed.

Lemma fin_all_NoDup : ∀ n : nat, NoDup (fin_all n).
Proof.
  intros n; induction n as [| m IH].
  - constructor.
  - simpl. constructor.
    + intros Hin. apply in_map_iff in Hin.
      destruct Hin as [x [Hx _]].
      exact (fin_FS_neq_F1 m x Hx).
    + exact (NoDup_map_FS m (fin_all m) IH).
Qed.
