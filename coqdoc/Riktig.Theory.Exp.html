Library Riktig.Theory.Exp

From Coq Require Import Morphisms.
From Riktig Require Import Ops Props Interface.Field Interface.Order
  Interface.Exp Theory.Ring Theory.Field Theory.Order.

Section ExpTheory.
  Context {F : Type} `{Exp F}.

  Lemma exp_add_neg : ∀ a : F, exp_fn a × exp_fn (- a) == 1.
  Proof.
    intros a.
    rewrite <- exp_add, invr, exp_zero.
    reflexivity.
  Qed.

  Lemma exp_nonzero : ∀ a : F, exp_fn a =/= 0.
  Proof.
    intros a Heq.
    apply field_nontrivial.
    rewrite <- (exp_add_neg a), Heq, mul_0_l.
    reflexivity.
  Qed.

  Lemma exp_neg : ∀ a : F, exp_fn (- a) == inv (exp_fn a).
  Proof.
    intros a.
    apply (mul_cancel_l (exp_fn a) (exp_fn (- a)) (inv (exp_fn a))
      (exp_nonzero a)).
    rewrite (exp_add_neg a), (field_invr (exp_fn a) (exp_nonzero a)).
    reflexivity.
  Qed.

  Lemma exp_strict_pos : ∀ a : F, 0 < exp_fn a.
  Proof.
    intros a.
    split.
    - apply exp_pos.
    - intros Heq.
      apply (exp_nonzero a).
      symmetry.
      exact Heq.
  Qed.

  Lemma exp_gt_one_of_pos : ∀ a : F, 0 < a → 1 < exp_fn a.
  Proof.
    intros a Ha.
    rewrite <- exp_zero.
    apply exp_mono.
    exact Ha.
  Qed.

  Lemma exp_lt_one_of_neg : ∀ a : F, a < 0 → exp_fn a < 1.
  Proof.
    intros a Ha.
    rewrite <- exp_zero.
    apply exp_mono.
    exact Ha.
  Qed.

  Lemma exp_le_of_eq : ∀ a b : F, exp_fn a == exp_fn b → a ≤ b.
  Proof.
    intros a b Heq.
    apply le_stable.
    intros Hnab.
    assert (Hane : a =/= b).
    { intros Heqab. apply Hnab. rewrite Heqab. apply le_refl. }
    destruct (le_total a b) as [Hab | Hba].
    - exact (Hnab Hab).
    - assert (Hblt : b < a).
      { split.
        - exact Hba.
        - intros Hbeq. apply Hane. symmetry. exact Hbeq. }
      destruct (exp_mono b a Hblt) as [_ Hneq].
      apply Hneq.
      symmetry.
      exact Heq.
  Qed.

  Lemma exp_injective : ∀ a b : F, exp_fn a == exp_fn b → a == b.
  Proof.
    intros a b Heq.
    apply le_antisym.
    - exact (exp_le_of_eq a b Heq).
    - apply (exp_le_of_eq b a).
      symmetry.
      exact Heq.
  Qed.

End ExpTheory.
