Section Action.
  Context {F : Type} `{OrderedField F}.
  Context {Int : Integral F F}.

  Context (ftc : ∀ (G g : F → F) (a b : F),
    (∀ t : F, is_deriv G t (lm_scale_map (g t))) →
    int g a b == G b + - G a).

  Theorem int_substitution : ∀ (f W u du : F → F) (a b : F),
    Proper (equ ==> equ) W →
    (∀ x : F, is_deriv W x (lm_scale_map (f x))) →
    (∀ t : F, is_deriv u t (lm_scale_map (du t))) →
    int (fun t : F ⇒ f (u t) × du t) a b == int f (u a) (u b).
  Proof.
    intros f W u du a b PW HW Hu.
    calc (int (fun t : F ⇒ f (u t) × du t) a b)
      == (W (u b) + - W (u a))
      == (int f (u a) (u b)).
    - apply (ftc (fun t : F ⇒ W (u t)) (fun t : F ⇒ f (u t) × du t) a b).
      intros t.
      apply (is_deriv_L_equ (fun s : F ⇒ W (u s)) t
        (lm_comp (lm_scale_map (f (u t))) (lm_scale_map (du t)))).
      + exact (lm_comp_scale_scale (du t) (f (u t))).
      + exact (deriv_comp u W t (lm_scale_map (du t))
          (lm_scale_map (f (u t))) PW (Hu t) (HW (u t))).
    - symmetry. exact (ftc W f (u a) (u b) HW).
  Qed.

  Section AbbreviatedAction.
    Context (m : F) (q v phi Phi : F → F).
    Context (Phi_proper : Proper (equ ==> equ) Phi).
    Context (Phi_antideriv : ∀ x : F, is_deriv Phi x (lm_scale_map (phi x))).
    Context (q_deriv : ∀ t : F, is_deriv q t (lm_scale_map (v t))).
    Context (speed_of_position : ∀ t : F, v t == phi (q t)).

    Theorem abbreviated_action : ∀ a b : F,
      int (fun x : F ⇒ m × phi x) (q a) (q b)
        == int (fun t : F ⇒ m × (v t × v t)) a b.
    Proof.
      intros a b.
      calc (int (fun x : F ⇒ m × phi x) (q a) (q b))
        == (m *: int phi (q a) (q b))
        == (m *: int (fun t : F ⇒ phi (q t) × v t) a b)
        == (int (fun t : F ⇒ m × (phi (q t) × v t)) a b)
        == (int (fun t : F ⇒ m × (v t × v t)) a b).
      - exact (int_linear_scale m phi (q a) (q b)).
      - apply field_sc_proper.
        + reflexivity.
        + symmetry.
          exact (int_substitution phi Phi q v a b
            Phi_proper Phi_antideriv q_deriv).
      - symmetry.
        exact (int_linear_scale m (fun t : F ⇒ phi (q t) × v t) a b).
      - assert (Hpt : pointwise_relation F equ
            (fun t : F ⇒ m × (phi (q t) × v t))
            (fun t : F ⇒ m × (v t × v t))).
        { intros t.
          apply ring_mul_proper; [ reflexivity | ].
          apply ring_mul_proper; [ symmetry; exact (speed_of_position t) | ].
          reflexivity. }
        exact (int_proper (fun t : F ⇒ m × (phi (q t) × v t))
          (fun t : F ⇒ m × (v t × v t)) Hpt
          a a (reflexivity a) b b (reflexivity b)).
    Qed.

    Theorem action_mass_linear : ∀ (c a b : F),
      int (fun t : F ⇒ (c × m) × (v t × v t)) a b
        == c *: int (fun t : F ⇒ m × (v t × v t)) a b.
    Proof.
      intros c a b.
      calc (int (fun t : F ⇒ (c × m) × (v t × v t)) a b)
        == (int (fun t : F ⇒ c *: (m × (v t × v t))) a b)
        == (c *: int (fun t : F ⇒ m × (v t × v t)) a b).
      - assert (Hpt : pointwise_relation F equ
            (fun t : F ⇒ (c × m) × (v t × v t))
            (fun t : F ⇒ c *: (m × (v t × v t)))).
        { intros t. exact (ring_mul_assoc c m (v t × v t)). }
        exact (int_proper (fun t : F ⇒ (c × m) × (v t × v t))
          (fun t : F ⇒ c *: (m × (v t × v t))) Hpt
          a a (reflexivity a) b b (reflexivity b)).
      - exact (int_linear_scale c (fun t : F ⇒ m × (v t × v t)) a b).
    Qed.

    Theorem action_path_additive : ∀ a b c : F,
      int (fun t : F ⇒ m × (v t × v t)) a b
        + int (fun t : F ⇒ m × (v t × v t)) b c
        == int (fun t : F ⇒ m × (v t × v t)) a c.
    Proof.
      intros a b c.
      exact (int_additivity (fun t : F ⇒ m × (v t × v t)) a b c).
    Qed.

  End AbbreviatedAction.

End Action.
