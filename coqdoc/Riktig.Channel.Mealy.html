Library Riktig.Channel.Mealy


From Coq Require Import List.
From Riktig.Channel Require Import Trace.
Import ListNotations.


Record Mealy (S I O : Type) := {
  m_init : S;
  m_step : S → I → S × O;
}.

Arguments m_init {S I O} m.
Arguments m_step {S I O} m s i.


Fixpoint run {S I O : Type} (m : Mealy S I O) (s : S) (is : list I)
    : S × list O :=
  match is with
  | [] ⇒ (s, [])
  | i :: is' ⇒
      let (s1, o) := m_step m s i in
      let (s2, os) := run m s1 is' in
      (s2, o :: os)
  end.

Definition run_from_init {S I O : Type} (m : Mealy S I O) (is : list I)
    : S × list O :=
  run m (m_init m) is.


Lemma run_nil : ∀ {S I O : Type} (m : Mealy S I O) (s : S),
  run m s [] = (s, []).
Proof.
  intros S I O m s. reflexivity.
Qed.

Lemma run_app : ∀ {S I O : Type} (m : Mealy S I O)
    (is1 is2 : list I) (s : S),
  run m s (is1 ++ is2) =
    (fst (run m (fst (run m s is1)) is2),
     snd (run m s is1) ++ snd (run m (fst (run m s is1)) is2)).
Proof.
  intros S I O m is1.
  induction is1 as [| i is1 IH]; intros is2 s.
  - simpl. destruct (run m s is2) as [s2 os2]. reflexivity.
  - simpl.
    destruct (m_step m s i) as [s1 o].
    destruct (run m s1 is1) as [s2 os] eqn:Hrun1.
    simpl.
    rewrite IH.
    rewrite Hrun1.
    simpl.
    reflexivity.
Qed.


Section Composition.
  Context {Ssnd Srcv App Wire : Type}.
  Context (snd_m : Mealy Ssnd App Wire).
  Context (rcv_m : Mealy Srcv Wire App).
  Context (chan : list Wire → list Wire → Prop).

  Definition composed_trace (app_in app_out : list App) : Prop :=
    ∃ wire_sent wire_recv,
      snd (run_from_init snd_m app_in) = wire_sent ∧
      chan wire_sent wire_recv ∧
      snd (run_from_init rcv_m wire_recv) = app_out.

End Composition.


Section Sanity.
  Context {App : Type}.

  Definition relay_m : Mealy unit App App :=
    {| m_init := tt; m_step := fun _ a ⇒ (tt, a) |}.

  Lemma run_relay : ∀ (is : list App) (s : unit),
    run relay_m s is = (tt, is).
  Proof.
    intros is.
    induction is as [| a is IH]; intros s.
    - destruct s. reflexivity.
    - simpl. rewrite IH. reflexivity.
  Qed.

  Theorem composed_trace_perfect_relay : ∀ app_in app_out : list App,
    composed_trace relay_m relay_m perfect app_in app_out →
    app_out = app_in.
  Proof.
    intros app_in app_out [wire_sent [wire_recv [Hsent [Hperfect Hrecv]]]].
    unfold run_from_init in Hsent, Hrecv.
    rewrite run_relay in Hsent.
    simpl in Hsent.
    unfold perfect in Hperfect.
    rewrite run_relay in Hrecv.
    simpl in Hrecv.
    rewrite <- Hsent in Hperfect.
    rewrite Hperfect in Hrecv.
    rewrite <- Hrecv.
    reflexivity.
  Qed.

End Sanity.
