Library Riktig.Channel.Mealy


From Coq Require Export List.
From Riktig Require Export Lib.Calc.
From Riktig.Channel Require Export Trace.
Import ListNotations.


Record Mealy (S I O : Type) := {
  m_init : S;
  m_step : S → I → S × O;
}.

Arguments m_init {S I O} m.
Arguments m_step {S I O} m s i.


Fixpoint run {S I O : Type} (m : Mealy S I O) (s : S) (is : list I)
    : S × list O :=
  match is with
  | [] ⇒ (s, [])
  | i :: is' ⇒
      let (s1, o) := m_step m s i in
      let (s2, os) := run m s1 is' in
      (s2, o :: os)
  end.

Definition run_from_init {S I O : Type} (m : Mealy S I O) (is : list I)
    : S × list O :=
  run m (m_init m) is.


Lemma run_nil : ∀ {S I O : Type} (m : Mealy S I O) (s : S),
  run m s [] = (s, []).
Proof.
  intros S I O m s. reflexivity.
Qed.

Lemma run_app : ∀ {S I O : Type} (m : Mealy S I O)
    (is1 is2 : list I) (s : S),
  run m s (is1 ++ is2) =
    (fst (run m (fst (run m s is1)) is2),
     snd (run m s is1) ++ snd (run m (fst (run m s is1)) is2)).
Proof.
  intros S I O m is1.
  induction is1 as [| i is1 IH]; intros is2 s.
  - simpl. destruct (run m s is2) as [s2 os2]. reflexivity.
  - simpl.
    destruct (m_step m s i) as [s1 o].
    destruct (run m s1 is1) as [s2 os] eqn:Hrun1.
    simpl.
    rewrite IH.
    rewrite Hrun1.
    simpl.
    reflexivity.
Qed.


Section Composition.
  Context {Ssnd Srcv App Wire : Type}.
  Context (snd_m : Mealy Ssnd App Wire).
  Context (rcv_m : Mealy Srcv Wire App).
  Context (chan : list Wire → list Wire → Prop).

  Definition composed_trace (app_in app_out : list App) : Prop :=
    ∃ wire_sent wire_recv,
      snd (run_from_init snd_m app_in) = wire_sent ∧
      chan wire_sent wire_recv ∧
      snd (run_from_init rcv_m wire_recv) = app_out.

End Composition.


Section Sanity.
  Context {App : Type}.

  Definition relay_m : Mealy unit App App :=
    {| m_init := tt; m_step := fun _ a ⇒ (tt, a) |}.

  Lemma run_relay : ∀ (is : list App) (s : unit),
    run relay_m s is = (tt, is).
  Proof.
    intros is.
    induction is as [| a is IH]; intros s.
    - destruct s. reflexivity.
    - simpl. rewrite IH. reflexivity.
  Qed.

  Theorem composed_trace_perfect_relay : ∀ app_in app_out : list App,
    composed_trace relay_m relay_m perfect app_in app_out →
    app_out = app_in.
  Proof.
    intros app_in app_out [wire_sent [wire_recv [Hsent [Hperfect Hrecv]]]].
    unfold run_from_init in Hsent, Hrecv.
    rewrite run_relay in Hsent. simpl in Hsent.
    rewrite run_relay in Hrecv. simpl in Hrecv.
    unfold perfect in Hperfect.
    calc app_out
      == wire_recv { Hrecv }
      == wire_sent { Hperfect }
      == app_in { Hsent }.
  Qed.

End Sanity.


Section FaultyVsPerfect.
  Context {Ssnd Srcv App Wire : Type}.
  Context (snd_m : Mealy Ssnd App Wire).
  Context (rcv_m : Mealy Srcv Wire App).
  Context (chan : list Wire → list Wire → Prop).
  Context (R : list App → list App → Prop).

  Theorem composed_trace_faulty_vs_perfect :
    (∀ ws wr : list Wire,
        chan ws wr →
        R (snd (run_from_init rcv_m ws)) (snd (run_from_init rcv_m wr))) →
    ∀ app_in app_out : list App,
      composed_trace snd_m rcv_m chan app_in app_out →
      ∃ app_perfect : list App,
        composed_trace snd_m rcv_m perfect app_in app_perfect ∧
        R app_perfect app_out.
  Proof.
    intros Hlift app_in app_out [ws [wr [Hsent [Hchan Hrecv]]]].
    ∃ (snd (run_from_init rcv_m ws)).
    split.
    -
      ∃ ws, ws.
      split.
      + exact Hsent.
      + split.
        × unfold perfect. reflexivity.
        × reflexivity.
    - rewrite <- Hrecv. apply Hlift. exact Hchan.
  Qed.

End FaultyVsPerfect.


Definition mapper_m {I O : Type} (f : I → O) : Mealy unit I O :=
  {| m_init := tt; m_step := fun _ i ⇒ (tt, f i) |}.

Lemma run_mapper : ∀ {I O : Type} (f : I → O) (is : list I) (s : unit),
  run (mapper_m f) s is = (tt, map f is).
Proof.
  intros I O f is.
  induction is as [| i is IH]; intros s.
  - destruct s. reflexivity.
  - simpl. rewrite IH. reflexivity.
Qed.


Lemma lossy_dup_map : ∀ {I O : Type} (f : I → O) (s r : list I),
  lossy_dup s r → lossy_dup (map f s) (map f r).
Proof.
  intros I O f s r H.
  induction H as [| x s' r' H' IH | x s' r' H' IH | x s' r' H' IH]; simpl.
  - apply lossy_dup_nil.
  - apply lossy_dup_keep. exact IH.
  - apply lossy_dup_drop. exact IH.
  - apply lossy_dup_stutter. exact IH.
Qed.


Lemma map_roundtrip : ∀ {A B : Type} (enc : A → B) (dec : B → A),
  (∀ a : A, dec (enc a) = a) →
  ∀ l : list A, map dec (map enc l) = l.
Proof.
  intros A B enc dec Hround l.
  induction l as [| a l IH]; simpl.
  - reflexivity.
  - rewrite Hround, IH. reflexivity.
Qed.


Theorem composed_trace_relay_transparent :
  ∀ {App : Type} (chan : list App → list App → Prop)
         (app_in app_out : list App),
  composed_trace relay_m relay_m chan app_in app_out ↔
  chan app_in app_out.
Proof.
  intros App chan app_in app_out.
  split.
  - intros [wire_sent [wire_recv [Hsent [Hchan Hrecv]]]].
    unfold run_from_init in Hsent, Hrecv.
    rewrite run_relay in Hsent. simpl in Hsent.
    rewrite run_relay in Hrecv. simpl in Hrecv.
    rewrite <- Hsent, Hrecv in Hchan.
    exact Hchan.
  - intros Hchan.
    ∃ app_in, app_out.
    split.
    + unfold run_from_init. rewrite run_relay. reflexivity.
    + split.
      × exact Hchan.
      × unfold run_from_init. rewrite run_relay. reflexivity.
Qed.


Corollary composed_trace_lossy_dup_relay :
  ∀ {App : Type} (app_in app_out : list App),
  composed_trace relay_m relay_m lossy_dup app_in app_out →
  lossy_dup app_in app_out.
Proof.
  intros App app_in app_out Hcomp.
  exact (proj1 (composed_trace_relay_transparent lossy_dup app_in app_out)
           Hcomp).
Qed.


Corollary composed_trace_lossy_relay :
  ∀ {App : Type} (app_in app_out : list App),
  composed_trace relay_m relay_m lossy app_in app_out →
  subseq app_out app_in.
Proof.
  intros App app_in app_out Hcomp.
  exact (proj1 (composed_trace_relay_transparent lossy app_in app_out)
           Hcomp).
Qed.


Theorem composed_trace_perfect_codec :
  ∀ {App Wire : Type} (enc : App → Wire) (dec : Wire → App),
  (∀ a : App, dec (enc a) = a) →
  ∀ app_in app_out : list App,
  composed_trace (mapper_m enc) (mapper_m dec) perfect app_in app_out →
  app_out = app_in.
Proof.
  intros App Wire enc dec Hround app_in app_out
    [wire_sent [wire_recv [Hsent [Hperfect Hrecv]]]].
  unfold run_from_init in Hsent, Hrecv.
  rewrite run_mapper in Hsent. simpl in Hsent.
  rewrite run_mapper in Hrecv. simpl in Hrecv.
  unfold perfect in Hperfect.
  calc app_out
    == (map dec wire_recv) { Hrecv }
    == (map dec wire_sent) { Hperfect }
    == (map dec (map enc app_in)) { Hsent }
    == app_in { (map_roundtrip enc dec Hround app_in) }.
Qed.

Theorem composed_trace_lossy_dup_codec :
  ∀ {App Wire : Type} (enc : App → Wire) (dec : Wire → App),
  (∀ a : App, dec (enc a) = a) →
  ∀ app_in app_out : list App,
  composed_trace (mapper_m enc) (mapper_m dec) lossy_dup app_in app_out →
  lossy_dup app_in app_out.
Proof.
  intros App Wire enc dec Hround app_in app_out Hcomp.
  assert (Hlift : ∀ ws wr : list Wire,
      lossy_dup ws wr →
      lossy_dup (snd (run_from_init (mapper_m dec) ws))
                (snd (run_from_init (mapper_m dec) wr))).
  { intros ws wr Hchan.
    unfold run_from_init. rewrite !run_mapper. simpl.
    apply lossy_dup_map. exact Hchan. }
  
  destruct (composed_trace_faulty_vs_perfect
              (mapper_m enc) (mapper_m dec) lossy_dup lossy_dup
              Hlift app_in app_out Hcomp)
    as [app_perfect [Hperfect Hfault]].
  assert (Heq : app_perfect = app_in).
  { apply (composed_trace_perfect_codec enc dec Hround app_in app_perfect).
    exact Hperfect. }
  rewrite Heq in Hfault. exact Hfault.
Qed.
