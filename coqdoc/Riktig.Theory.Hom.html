Section EvalPair.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V}.

  Local Notation DualV := (LinearMap F V F).

  Let FV : VectorSpace F F := @Field_VectorSpace F FF.
  Let DV : VectorSpace F DualV := @Hom_VectorSpace F V F FF VV FV.

  Lemma eval_proper :
    Proper (equ ==> equ ==> equ) (fun (f : DualV) (v : V) ⇒ f v).
  Proof.
    intros f g Hfg v v' Hv.
    rewrite (lm_proper f v v' Hv).
    exact (Hfg v').
  Qed.

  Lemma eval_add_l : ∀ (f g : DualV) (v : V), (f + g) v == f v + g v.
  Proof.
    intros f g v.
    reflexivity.
  Qed.

  Lemma eval_scale_l : ∀ (a : F) (f : DualV) (v : V),
    (a *: f) v == a *: f v.
  Proof.
    intros a f v.
    reflexivity.
  Qed.

  Lemma eval_add_r : ∀ (f : DualV) (v v' : V),
    f (v + v') == f v + f v'.
  Proof.
    intros f v v'.
    exact (lm_add f v v').
  Qed.

  Lemma eval_scale_r : ∀ (a : F) (f : DualV) (v : V),
    f (a *: v) == a *: f v.
  Proof.
    intros a f v.
    exact (lm_scale f a v).
  Qed.

  Definition eval_pair : BilinearMap F (Dual F V) V F :=
    @Build_BilinearMap F (Dual F V) V F FF DV VV FV
      (fun (f : DualV) (v : V) ⇒ f v)
      eval_proper eval_add_l eval_scale_l eval_add_r eval_scale_r.

End EvalPair.


Section CurryUncurry.
  Context {F U V W : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}
          {VW : VectorSpace F W}.

  Local Notation HomVW := (LinearMap F V W).

  Let HVW : VectorSpace F HomVW := @Hom_VectorSpace F V W FF VV VW.

  #[global] Instance Bilin_EqRel : EqRel (BilinearMap F U V W) :=
    fun b b' ⇒ ∀ (u : U) (v : V), b u v == b' u v.

  #[global] Instance Bilin_Equivalence :
    Equivalence (@equ (BilinearMap F U V W) Bilin_EqRel).
  Proof.
    apply Build_Equivalence.
    - intros b u v. reflexivity.
    - intros b b' Hb u v. symmetry. apply Hb.
    - intros b c d Hbc Hcd u v. rewrite (Hbc u v). apply Hcd.
  Qed.

  Section Curry.
    Context (b : BilinearMap F U V W).

    Lemma curry_proper : Proper (equ ==> equ) (fun u : U ⇒ bm_appl b u).
    Proof.
      intros u u' Hu v.
      exact (bm_proper b u u' Hu v v (reflexivity v)).
    Qed.

    Lemma curry_add : ∀ u u' : U,
      bm_appl b (u + u') == bm_appl b u + bm_appl b u'.
    Proof.
      intros u u' v.
      exact (bm_add_l b u u' v).
    Qed.

    Lemma curry_scale : ∀ (a : F) (u : U),
      bm_appl b (a *: u) == a *: bm_appl b u.
    Proof.
      intros a u v.
      exact (bm_scale_l b a u v).
    Qed.

    Definition curry : LinearMap F U HomVW :=
      @Build_LinearMap F U HomVW FF VU HVW
        (fun u : U ⇒ bm_appl b u)
        curry_proper curry_add curry_scale.

  End Curry.

  Section Uncurry.
    Context (g : LinearMap F U HomVW).

    Lemma uncurry_proper :
      Proper (equ ==> equ ==> equ) (fun (u : U) (v : V) ⇒ g u v).
    Proof.
      intros u u' Hu v v' Hv.
      rewrite (lm_proper g u u' Hu v).
      exact (lm_proper (g u') v v' Hv).
    Qed.

    Lemma uncurry_add_l : ∀ (u u' : U) (v : V),
      g (u + u') v == g u v + g u' v.
    Proof.
      intros u u' v.
      exact (lm_add g u u' v).
    Qed.

    Lemma uncurry_scale_l : ∀ (a : F) (u : U) (v : V),
      g (a *: u) v == a *: g u v.
    Proof.
      intros a u v.
      exact (lm_scale g a u v).
    Qed.

    Lemma uncurry_add_r : ∀ (u : U) (v v' : V),
      g u (v + v') == g u v + g u v'.
    Proof.
      intros u v v'.
      exact (lm_add (g u) v v').
    Qed.

    Lemma uncurry_scale_r : ∀ (a : F) (u : U) (v : V),
      g u (a *: v) == a *: g u v.
    Proof.
      intros a u v.
      exact (lm_scale (g u) a v).
    Qed.

    Definition uncurry : BilinearMap F U V W :=
      @Build_BilinearMap F U V W FF VU VV VW
        (fun (u : U) (v : V) ⇒ g u v)
        uncurry_proper uncurry_add_l uncurry_scale_l
        uncurry_add_r uncurry_scale_r.

  End Uncurry.


  Theorem curry_uncurry : ∀ g : LinearMap F U HomVW,
    curry (uncurry g) == g.
  Proof.
    intros g u v.
    reflexivity.
  Qed.

  Theorem uncurry_curry : ∀ b : BilinearMap F U V W,
    uncurry (curry b) == b.
  Proof.
    intros b u v.
    reflexivity.
  Qed.


End CurryUncurry.


Section DoubleDual.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V}.

  Local Notation DDualV := (LinearMap F (LinearMap F V F) F).

  Let FV : VectorSpace F F := @Field_VectorSpace F FF.
  Let DV : VectorSpace F (LinearMap F V F) :=
    @Hom_VectorSpace F V F FF VV FV.
  Let DDV : VectorSpace F DDualV :=
    @Hom_VectorSpace F (LinearMap F V F) F FF DV FV.

  Lemma dd_proper :
    Proper (equ ==> equ) (fun v : V ⇒ bm_appr eval_pair v).
  Proof.
    intros v v' Hv f.
    exact (lm_proper f v v' Hv).
  Qed.

  Lemma dd_add : ∀ u v : V,
    bm_appr eval_pair (u + v) == bm_appr eval_pair u + bm_appr eval_pair v.
  Proof.
    intros u v f.
    exact (lm_add f u v).
  Qed.

  Lemma dd_scale : ∀ (a : F) (v : V),
    bm_appr eval_pair (a *: v) == a *: bm_appr eval_pair v.
  Proof.
    intros a v f.
    exact (lm_scale f a v).
  Qed.

  Definition into_double_dual : LinearMap F V (Dual F (Dual F V)) :=
    @Build_LinearMap F V (Dual F (Dual F V)) FF VV DDV
      (fun v : V ⇒ bm_appr eval_pair v)
      dd_proper dd_add dd_scale.


End DoubleDual.


Example double_dual_eval (F V : Type) (FF : Field F) (VV : VectorSpace F V) :
  ∀ (v : V) (f : LinearMap F V F), into_double_dual v f == f v.
Proof.
  intros v f.
  reflexivity.
Qed.

Example curry_eval_check (F V : Type) (FF : Field F) (VV : VectorSpace F V) :
  ∀ f : LinearMap F V F, curry eval_pair f == f.
Proof.
  intros f v.
  reflexivity.
Qed.
