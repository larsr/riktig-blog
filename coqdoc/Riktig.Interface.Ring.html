Library Riktig.Interface.Ring

From Coq Require Export Morphisms.
From Riktig Require Export Ops Props Interface.AbGroup.

Class Ring R := {
  #[global] ring_ab :: AbGroup R;
  #[global] ring_one :: One R;
  #[global] ring_mul :: HasMult R;
  ring_mul_assoc : ∀ a b c : R, (a × b) × c == a × (b × c);
  ring_mul_1l : ∀ a : R, 1 × a == a;
  ring_mul_1r : ∀ a : R, a × 1 == a;
  ring_distl : ∀ a b c : R, a × (b + c) == a × b + a × c;
  ring_distr : ∀ a b c : R, (a + b) × c == a × c + b × c;
  #[global] ring_mul_proper :: Proper (equ ==> equ ==> equ) (@mult R _)
}.

#[global] Instance Ring_MulAssoc R (H : Ring R) : Assoc R equ mult := ring_mul_assoc.
#[global] Instance Ring_MulIdl R (H : Ring R) : Idl R equ mult 1 := ring_mul_1l.
#[global] Instance Ring_MulIdr R (H : Ring R) : Idr R equ mult 1 := ring_mul_1r.
#[global] Instance Ring_DistL R (H : Ring R) : DistL R equ add mult := ring_distl.
#[global] Instance Ring_DistR R (H : Ring R) : DistR R equ add mult := ring_distr.

#[global] Hint Mode Ring ! : typeclass_instances.
