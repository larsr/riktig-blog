Library Riktig.Theory.KleisliIso


From Riktig Require Import Ops Interface.Monad Lib.Calc.


Record KleisliIso (M : Type → Type) `{Monad M} (A B : Type) := {
  ki_to : A → M B;
  ki_from : B → M A;
  ki_fromto : ∀ a : A, bind (ki_to a) ki_from == ret a;
  ki_tofrom : ∀ b : B, bind (ki_from b) ki_to == ret b
}.

Arguments ki_to {M H H0 H1 H2 A B}.
Arguments ki_from {M H H0 H1 H2 A B}.
Arguments ki_fromto {M H H0 H1 H2 A B}.
Arguments ki_tofrom {M H H0 H1 H2 A B}.


Section ChannelInsertion.
  Context {M : Type → Type} `{Monad M} {A B B' C : Type}.
  Context (prog1 : A → M B) (prog2 : B → M C) (codec : KleisliIso M B B').

  Lemma ki_insert_pointwise : ∀ b : B,
    bind (ki_to codec b) (fun b' ⇒ bind (ki_from codec b') prog2) == prog2 b.
  Proof.
    intro b.
    calc (bind (ki_to codec b) (fun b' ⇒ bind (ki_from codec b') prog2))
      == (bind (bind (ki_to codec b) (ki_from codec)) prog2)
      == (bind (ret b) prog2)
      == (prog2 b).
    - symmetry. apply bind_assoc.
    - apply bind_proper.
      + exact (ki_fromto codec b).
      + intro x. reflexivity.
    - apply bind_ret_l.
  Qed.

  Theorem channel_insertion : ∀ a : A,
    (prog1 >=> prog2) a == (prog1 >=> ki_to codec >=> ki_from codec >=> prog2) a.
  Proof.
    intro a.
    apply bind_proper.
    - reflexivity.
    - intro b. symmetry. exact (ki_insert_pointwise b).
  Qed.

End ChannelInsertion.


From Coq Require Import Cantor.
From Riktig Require Import Impl.OptionMonad.

Lemma nat_pair_fromto : ∀ p : nat × nat,
  bind (Some (to_nat p)) (fun n ⇒ Some (of_nat n)) == Some p.
Proof.
  intro p.
  unfold bind, Option_HasBind, equ, Option_EqRel.
  rewrite cancel_of_to.
  reflexivity.
Qed.

Lemma nat_pair_tofrom : ∀ n : nat,
  bind (Some (of_nat n)) (fun p : nat × nat ⇒ Some (to_nat p)) == Some n.
Proof.
  intro n.
  unfold bind, Option_HasBind, equ, Option_EqRel.
  rewrite cancel_to_of.
  reflexivity.
Qed.

Definition nat_pair_codec : KleisliIso option (nat × nat) nat :=
  {| ki_to := fun p : nat × nat ⇒ Some (to_nat p);
     ki_from := fun n : nat ⇒ Some (of_nat n);
     ki_fromto := nat_pair_fromto;
     ki_tofrom := nat_pair_tofrom |}.

Definition split_n (n : nat) : option (nat × nat) := Some (n, n).
Definition sum_pair (p : nat × nat) : option nat := Some (Nat.add (fst p) (snd p)).

Example channel_insertion_nat_pair_check :=
  channel_insertion split_n sum_pair nat_pair_codec.
