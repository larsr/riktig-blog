Section GroupTheory.
  Context {G : Type} `{Group G}.

  Theorem neg_unique_l : ∀ g h : G, h + g == 0 → h == - g.
  Proof.
    intros g h Heq.
    calc h
      == (h + 0) { idr }
      == (h + (g + - g)) { invr }
      == ((h + g) + - g) { assoc }
      == (0 + - g) { Heq }
      == (- g) { idl }.
  Qed.

  Theorem neg_unique_r : ∀ g h : G, g + h == 0 → h == - g.
  Proof.
    intros g h Heq.
    calc h
      == (0 + h) { idl }
      == ((- g + g) + h) { invl }
      == (- g + (g + h)) { assoc }
      == (- g + 0) { Heq }
      == (- g) { idr }.
  Qed.

  Theorem cancel_l : ∀ a b c : G, a + b == a + c → b == c.
  Proof.
    intros a b c Heq.
    calc b
      == (0 + b) { idl }
      == ((- a + a) + b) { invl }
      == (- a + (a + b)) { assoc }
      == (- a + (a + c)) { Heq }
      == ((- a + a) + c) { assoc }
      == (0 + c) { invl }
      == c { idl }.
  Qed.

  Theorem cancel_r : ∀ a b c : G, b + a == c + a → b == c.
  Proof.
    intros a b c Heq.
    calc b
      == (b + 0) { idr }
      == (b + (a + - a)) { invr }
      == ((b + a) + - a) { assoc }
      == ((c + a) + - a) { Heq }
      == (c + (a + - a)) { assoc }
      == (c + 0) { invr }
      == c { idr }.
  Qed.

  Theorem neg_involutive : ∀ g : G, - (- g) == g.
  Proof.
    intros g.
    apply (cancel_l (- g)).
    calc ((- g) + - - g)
      == (0 : G) { (invr (- g)) }
      == ((- g) + g) { (invl g) }.
  Qed.

  Theorem neg_zero : - (0 : G) == 0.
  Proof.
    apply (cancel_l 0).
    calc ((0 : G) + - 0)
      == (0 : G) { (invr 0) }
      == ((0 : G) + 0) { (idr 0) }.
  Qed.

  Theorem neg_add_flip : ∀ a b : G, - (a + b) == (- b) + (- a).
  Proof.
    intros a b. symmetry. apply neg_unique_r.
    calc ((a + b) + (- b + - a))
      == (((a + b) + - b) + - a) { assoc }
      == ((a + (b + - b)) + - a) { assoc }
      == ((a + 0) + - a) { invr }
      == (a + - a) { idr }
      == (0 : G) { invr }.
  Qed.

  Theorem neg_inj : ∀ a b : G, - a == - b → a == b.
  Proof.
    intros a b Heq.
    calc a
      == (- - a) { (neg_involutive a) }
      == (- - b) { Heq }
      == b { (neg_involutive b) }.
  Qed.

  Theorem add_neg_telescope : ∀ a b c : G,
    (a + - b) + (b + - c) == a + - c.
  Proof.
    intros a b c.
    calc ((a + - b) + (b + - c))
      == (a + (- b + (b + - c))) { assoc }
      == (a + ((- b + b) + - c)) { assoc }
      == (a + (0 + - c)) { invl }
      == (a + - c) { idl }.
  Qed.

  Theorem sub_eq_zero : ∀ a b : G, a + - b == 0 → a == b.
  Proof.
    intros a b Heq.
    apply (cancel_r (- b)).
    calc (a + - b)
      == (0 : G) { Heq }
      == (b + - b) { (invr b) }.
  Qed.

End GroupTheory.

#[global] Instance Group_NegInjective G (H : Group G) :
  Injective G G equ equ neg := neg_inj.

Section GroupHomTheory.
  Context {G K : Type} `{Group G} `{Group K}.
  Context (f : G → K).
  Context `{fp : Proper _ (equ ==> equ) f}.
  Context (fadd : ∀ a b : G, f (a + b) == f a + f b).

  Theorem hom_zero : f 0 == 0.
  Proof.
    apply (cancel_l (f 0)).
    calc (f 0 + f 0)
      == (f (0 + 0)) { (fadd 0 0) }
      == (f 0) { idl }
      == (f 0 + 0) { idr }.
  Qed.

  Theorem hom_neg : ∀ a : G, f (- a) == - f a.
  Proof.
    intros a.
    apply neg_unique_l.
    calc (f (- a) + f a)
      == (f (- a + a)) { (fadd (- a) a) }
      == (f 0) { invl }
      == (0 : K) { hom_zero }.
  Qed.

End GroupHomTheory.

Section GroupRewriteExamples.
  Context {G : Type} `{Group G}.

  Example autorewrite_double_neg_idr : ∀ g : G, - (- g) + 0 == g.
  Proof. intros g. autorewrite with grp. reflexivity. Qed.

  Example autorewrite_multi : ∀ g : G, (0 + - (- g)) + 0 == g.
  Proof. intros g. autorewrite with grp. reflexivity. Qed.

  Example autorewrite_invl_invr : ∀ g : G,
    ((- g) + g) + (g + - g) == 0.
  Proof. intros g. autorewrite with grp. reflexivity. Qed.

End GroupRewriteExamples.
