From Coq Require Import Morphisms.
From Coq.Vectors Require Fin.
From Riktig Require Import Ops Interface.AbGroup Interface.Ring

Definition Matrix (m n : nat) (F : Type) : Type :=
  Fin.t m → Fin.t n → F.

Example matrix_vs_check (F : Type) (FF : Field F) (m n : nat) :
  VectorSpace F (Matrix m n F) := _.

Section MatrixOps.
  Context {F : Type} {FF : Field F}.


  Definition mat_id {n : nat} : Matrix n n F := e.


  Definition mat_mul {m n p : nat} (A : Matrix m n F) (B : Matrix n p F)
      : Matrix m p F :=
    fun i k ⇒ bigopF (fun j ⇒ A i j × B j k).

  Lemma mat_mul_proper : ∀ (m n p : nat),
    Proper (equ ==> equ ==> equ) (@mat_mul m n p).
  Proof.
    intros m n p A A' HA B B' HB i k.
    exact (bigopF_ext n (fun j ⇒ A i j × B j k) (fun j ⇒ A' i j × B' j k)
             (fun j ⇒ ring_mul_proper (A i j) (A' i j) (HA i j)
                         (B j k) (B' j k) (HB j k))).
  Qed.


  Lemma mat_mul_expand_l : ∀ (m n p q : nat) (A : Matrix m n F)
      (B : Matrix n p F) (C : Matrix p q F) (i : Fin.t m) (l : Fin.t q),
    mat_mul (mat_mul A B) C i l
      == bigopF (fun k ⇒ bigopF (fun j ⇒ A i j × (B j k × C k l))).
  Proof.
    intros m n p q A B C i l.
    transitivity (bigopF (fun k ⇒ bigopF (fun j ⇒ A i j × B j k) × C k l)).
    - reflexivity.
    - apply bigopF_ext. intros k.
      transitivity (bigopF (fun j ⇒ A i j × B j k × C k l)).
      + symmetry.
        exact (bigopF_mul_r n (C k l) (fun j ⇒ A i j × B j k)).
      + apply bigopF_ext. intros j.
        exact (ring_mul_assoc (A i j) (B j k) (C k l)).
  Qed.

  Lemma mat_mul_expand_r : ∀ (m n p q : nat) (A : Matrix m n F)
      (B : Matrix n p F) (C : Matrix p q F) (i : Fin.t m) (l : Fin.t q),
    mat_mul A (mat_mul B C) i l
      == bigopF (fun j ⇒ bigopF (fun k ⇒ A i j × (B j k × C k l))).
  Proof.
    intros m n p q A B C i l.
    transitivity (bigopF (fun j ⇒ A i j × bigopF (fun k ⇒ B j k × C k l))).
    - reflexivity.
    - apply bigopF_ext. intros j.
      symmetry.
      exact (bigopF_mul_l p (A i j) (fun k ⇒ B j k × C k l)).
  Qed.

  Theorem mat_mul_assoc : ∀ (m n p q : nat) (A : Matrix m n F)
      (B : Matrix n p F) (C : Matrix p q F),
    mat_mul (mat_mul A B) C == mat_mul A (mat_mul B C).
  Proof.
    intros m n p q A B C i l.
    rewrite (mat_mul_expand_l m n p q A B C i l),
      (mat_mul_expand_r m n p q A B C i l).
    exact (bigopF_swap p n (fun k j ⇒ A i j × (B j k × C k l))).
  Qed.


  Lemma e_sym : ∀ (n : nat) (i j : Fin.t n), e i j == e j i.
  Proof.
    intros n i j.
    destruct (Fin.eq_dec i j) as [Heq | Hne].
    - rewrite (e_at_eq n i j Heq), (e_at_eq n j i (eq_sym Heq)).
      reflexivity.
    - rewrite (e_at_neq n i j Hne),
        (e_at_neq n j i (fun Heq ⇒ Hne (eq_sym Heq))).
      reflexivity.
  Qed.

  Lemma e_mul_comm : ∀ (n : nat) (a : F) (i j : Fin.t n),
    e i j × a == a × e j i.
  Proof.
    intros n a i j.
    rewrite (field_mul_comm (e i j) a), (e_sym n i j).
    reflexivity.
  Qed.

  Theorem mat_mul_1l : ∀ (m n : nat) (A : Matrix m n F),
    mat_mul mat_id A == A.
  Proof.
    intros m n A i k.
    transitivity (bigopF (fun j ⇒ A j k × e j i)).
    - apply bigopF_ext. intros j.
      exact (e_mul_comm m (A j k) i j).
    - exact (lincomb_std_delta m (fun j ⇒ A j k) i).
  Qed.

  Theorem mat_mul_1r : ∀ (m n : nat) (A : Matrix m n F),
    mat_mul A mat_id == A.
  Proof.
    intros m n A i k.
    exact (lincomb_std_delta n (fun j ⇒ A i j) k).
  Qed.


  Theorem mat_mul_add_l : ∀ (m n p : nat) (A A' : Matrix m n F)
      (B : Matrix n p F),
    mat_mul (A + A') B == mat_mul A B + mat_mul A' B.
  Proof.
    intros m n p A A' B i k.
    transitivity (bigopF (fun j ⇒ A i j × B j k + A' i j × B j k)).
    - apply bigopF_ext. intros j.
      exact (ring_distr (A i j) (A' i j) (B j k)).
    - exact (bigopF_add n (fun j ⇒ A i j × B j k)
               (fun j ⇒ A' i j × B j k)).
  Qed.

  Theorem mat_mul_add_r : ∀ (m n p : nat) (A : Matrix m n F)
      (B B' : Matrix n p F),
    mat_mul A (B + B') == mat_mul A B + mat_mul A B'.
  Proof.
    intros m n p A B B' i k.
    transitivity (bigopF (fun j ⇒ A i j × B j k + A i j × B' j k)).
    - apply bigopF_ext. intros j.
      exact (ring_distl (A i j) (B j k) (B' j k)).
    - exact (bigopF_add n (fun j ⇒ A i j × B j k)
               (fun j ⇒ A i j × B' j k)).
  Qed.

  Theorem mat_mul_scale_l : ∀ (m n p : nat) (a : F)
      (A : Matrix m n F) (B : Matrix n p F),
    mat_mul (a *: A) B == a *: mat_mul A B.
  Proof.
    intros m n p a A B i k.
    transitivity (bigopF (fun j ⇒ a × (A i j × B j k))).
    - apply bigopF_ext. intros j.
      exact (ring_mul_assoc a (A i j) (B j k)).
    - exact (bigopF_mul_l n a (fun j ⇒ A i j × B j k)).
  Qed.

  Theorem mat_mul_scale_r : ∀ (m n p : nat) (a : F)
      (A : Matrix m n F) (B : Matrix n p F),
    mat_mul A (a *: B) == a *: mat_mul A B.
  Proof.
    intros m n p a A B i k.
    transitivity (bigopF (fun j ⇒ a × (A i j × B j k))).
    - apply bigopF_ext. intros j.
      exact (mul_comm_swap (A i j) a (B j k)).
    - exact (bigopF_mul_l n a (fun j ⇒ A i j × B j k)).
  Qed.


  #[global] Instance Matrix_One (n : nat) : One (Matrix n n F) := mat_id.

  #[global] Instance Matrix_Mult (n : nat) : HasMult (Matrix n n F) :=
    @mat_mul n n n.

  Lemma matrix_mul_assoc (n : nat) : ∀ A B C : Matrix n n F,
    (A × B) × C == A × (B × C).
  Proof.
    intros A B C.
    exact (mat_mul_assoc n n n n A B C).
  Qed.

  Lemma matrix_mul_1l (n : nat) : ∀ A : Matrix n n F, 1 × A == A.
  Proof.
    intros A.
    exact (mat_mul_1l n n A).
  Qed.

  Lemma matrix_mul_1r (n : nat) : ∀ A : Matrix n n F, A × 1 == A.
  Proof.
    intros A.
    exact (mat_mul_1r n n A).
  Qed.

  Lemma matrix_distl (n : nat) : ∀ A B C : Matrix n n F,
    A × (B + C) == A × B + A × C.
  Proof.
    intros A B C.
    exact (mat_mul_add_r n n n A B C).
  Qed.

  Lemma matrix_distr (n : nat) : ∀ A B C : Matrix n n F,
    (A + B) × C == A × C + B × C.
  Proof.
    intros A B C.
    exact (mat_mul_add_l n n n A B C).
  Qed.

  Lemma matrix_mul_proper (n : nat) :
    Proper (equ ==> equ ==> equ) (@mult (Matrix n n F) (Matrix_Mult n)).
  Proof.
    exact (mat_mul_proper n n n).
  Qed.

  #[global] Instance Matrix_Ring (n : nat) : Ring (Matrix n n F) := {|
    ring_ab := @PW_AbGroup (Fin.t n) (Fin.t n → F)
                         (@PW_AbGroup (Fin.t n) F
                            (@ring_ab F (@field_ring F FF)));
    ring_one := Matrix_One n;
    ring_mul := Matrix_Mult n;
    ring_mul_assoc := matrix_mul_assoc n;
    ring_mul_1l := matrix_mul_1l n;
    ring_mul_1r := matrix_mul_1r n;
    ring_distl := matrix_distl n;
    ring_distr := matrix_distr n;
    ring_mul_proper := matrix_mul_proper n
  |}.

End MatrixOps.
