From Coq Require Import List ZArith.
Import ListNotations.
From Riktig Require Export Programs.Memprog.Syntax.



Inductive Cell (addr val : Type) : Type :=
  | CellVal : val → Cell addr val
  | CellAddr : addr → Cell addr val
  | CellInstr : Instr val → Cell addr val.

Arguments CellVal {addr val}.
Arguments CellAddr {addr val}.
Arguments CellInstr {addr val}.


Definition AStore (addr val : Type) := list (addr × Cell addr val).

Fixpoint store_lookup {addr val : Type} (eqb : addr → addr → bool)
  (a : addr) (s : AStore addr val) : option (Cell addr val) :=
  match s with
  | [] ⇒ None
  | (k, c) :: rest ⇒ if eqb a k then Some c else store_lookup eqb a rest
  end.

Definition store_insert {addr val : Type}
  (a : addr) (c : Cell addr val) (s : AStore addr val) : AStore addr val :=
  (a, c) :: s.

Fixpoint store_member {addr val : Type} (eqb : addr → addr → bool)
  (a : addr) (s : AStore addr val) : bool :=
  match s with
  | [] ⇒ false
  | (k, _) :: rest ⇒ if eqb a k then true else store_member eqb a rest
  end.



Record Machine (addr val : Type) : Type := MkMachine {
  mach_store : AStore addr val;
  mach_sp : addr;
  mach_hp : addr
}.

Arguments MkMachine {addr val}.
Arguments mach_store {addr val}.
Arguments mach_sp {addr val}.
Arguments mach_hp {addr val}.


Section Interp.
  Context {addr val : Type}.
  Context (addr_eqb : addr → addr → bool).
  Context (addr_add addr_sub : addr → addr → addr).
  Context (addr_leb addr_geb : addr → addr → bool).
  Context (addr_of_nat : nat → addr).
  Context (addr1 : addr).
  Context (to_addr : val → addr).
  Context (val_default : val).


  Fixpoint interp {a : Type} (fuel : nat)
    (p : Mem addr val a) (mach : Machine addr val) {struct fuel}
    : option a × Machine addr val :=
    match fuel with
    | O ⇒ (None, mach)
    | S fuel' ⇒
      match p in Mem _ _ t return option t × Machine addr val with
      | Ret x ⇒ (Some x, mach)
      | ReadVal ad k ⇒
          match store_lookup addr_eqb ad (mach_store mach) with
          | Some (CellVal v) ⇒ interp fuel' (k v) mach
          | _ ⇒ (None, mach)
          end
      | WriteVal ad v m' ⇒
          if store_member addr_eqb ad (mach_store mach)
          then interp fuel' m'
                 (MkMachine (store_insert ad (CellVal v) (mach_store mach))
                            (mach_sp mach) (mach_hp mach))
          else (None, mach)
      | ReadAddr ad k ⇒
          match store_lookup addr_eqb ad (mach_store mach) with
          | Some (CellAddr p') ⇒ interp fuel' (k p') mach
          | _ ⇒ (None, mach)
          end
      | WriteAddr ad p' m' ⇒
          if store_member addr_eqb ad (mach_store mach)
          then interp fuel' m'
                 (MkMachine (store_insert ad (CellAddr p') (mach_store mach))
                            (mach_sp mach) (mach_hp mach))
          else (None, mach)
      | ReadInstr ad k ⇒
          match store_lookup addr_eqb ad (mach_store mach) with
          | Some (CellInstr i) ⇒ interp fuel' (k i) mach
          | _ ⇒ (None, mach)
          end
      | WriteInstr ad i m' ⇒
          if store_member addr_eqb ad (mach_store mach)
          then interp fuel' m'
                 (MkMachine (store_insert ad (CellInstr i) (mach_store mach))
                            (mach_sp mach) (mach_hp mach))
          else (None, mach)
      | StackAlloc k ⇒
          if addr_leb (mach_sp mach) (mach_hp mach)
          then let ad := mach_sp mach in
               interp fuel' (k ad)
                 (MkMachine (store_insert ad (CellVal val_default) (mach_store mach))
                            (addr_add ad addr1) (mach_hp mach))
          else (None, mach)
      | Alloc n k ⇒
          let base := addr_sub (mach_hp mach) (addr_of_nat (n - 1)) in
          if addr_geb base (mach_sp mach)
          then let st :=
                 fold_left
                   (fun s i ⇒ store_insert (addr_add base (addr_of_nat i))
                                            (CellVal val_default) s)
                   (seq 0 n) (mach_store mach) in
               interp fuel' (k base)
                 (MkMachine st (mach_sp mach) (addr_sub base addr1))
          else (None, mach)
      | Offset b o k ⇒
          interp fuel' (k (addr_add b (to_addr o))) mach
      | GetSp k ⇒
          interp fuel' (k (mach_sp mach)) mach
      | SetSp sp m' ⇒
          interp fuel' m' (MkMachine (mach_store mach) sp (mach_hp mach))
      | While c body m' ⇒
          let '(rc, mach1) := interp fuel' c mach in
          match rc with
          | Some (Some true) ⇒
              let '(rb, mach2) := interp fuel' body mach1 in
              match rb with
              | Some (Some _) ⇒ interp fuel' (While c body m') mach2
              | _ ⇒ (None, mach2)
              end
          | Some (Some false) ⇒ interp fuel' m' mach1
          | _ ⇒ (None, mach1)
          end
      end
    end.

  Definition run_t {a : Type} (fuel : nat)
    (p : MemT addr val a) (mach : Machine addr val)
    : option a × Machine addr val :=
    let '(r, mach') := interp fuel p mach in
    (match r with Some (Some x) ⇒ Some x | _ ⇒ None end, mach').

End Interp.


Local Open Scope Z_scope.


Definition z_machine (memsize : nat) : Machine Z Z :=
  MkMachine [] 0 (Z.of_nat memsize - 1).

Definition run_z {a : Type} (fuel : nat) (p : MemT Z Z a) (memsize : nat)
  : option a × Machine Z Z :=
  run_t Z.eqb Z.add Z.sub Z.leb Z.geb Z.of_nat 1 (fun v : Z ⇒ v) 0
        fuel p (z_machine memsize).


Fixpoint mem_read_cells (n : nat) (base : Z) (mach : Machine Z Z)
  : option (list Z) :=
  match n with
  | O ⇒ Some []
  | S n' ⇒
      match store_lookup Z.eqb base (mach_store mach) with
      | Some (CellVal v) ⇒ option_map (cons v) (mem_read_cells n' (base + 1) mach)
      | _ ⇒ None
      end
  end.

Definition mem_to_list (mach : Machine Z Z) (mbase : option Z) (n : nat)
  : option (list Z) :=
  match mbase with
  | Some base ⇒ mem_read_cells n base mach
  | None ⇒ None
  end.


Definition write_then_read : MemT Z Z Z :=
  do! a <- stack_alloc_m;
  write_m a 42 >>
  read_m a.

Example write_then_read_result :
  fst (run_z 100 write_then_read 16) = Some 42.
Proof. vm_compute. reflexivity. Qed.

Definition count_up : MemT Z Z Z :=
  do! base <- alloc_m 5;
  do! i <- stack_alloc_m;
  write_m i 0 >>
  while_m (do! v <- read_m i; ret_t (v <? 5))
          (do! v <- read_m i;
           do! a <- offset_m base v;
           write_m a v >>
           write_m i (v + 1)) >>
  ret_t base.

Example count_up_cells :
  (let '(r, m) := run_z 2000 count_up 64 in mem_to_list m r 5)
  = Some [0; 1; 2; 3; 4].
Proof. vm_compute. reflexivity. Qed.

Example loop_forever_no_answer :
  fst (run_z 5000 (while_m (ret_t true) (ret_t tt) : MemT Z Z unit) 16)
  = None.
Proof. vm_compute. reflexivity. Qed.
