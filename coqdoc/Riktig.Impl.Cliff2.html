Record C2 (F : Type) := mkC2 {
  c2_s : F;
  c2_1 : F;
  c2_2 : F;
  c2_12 : F
}.

Arguments mkC2 {F} c2_s c2_1 c2_2 c2_12.
Arguments c2_s {F} c.
Arguments c2_1 {F} c.
Arguments c2_2 {F} c.
Arguments c2_12 {F} c.


Section C2Ops.
  Context {F : Type} `{FF : Field F}.

  #[global] Instance C2_EqRel : EqRel (C2 F) :=
    fun x y ⇒ c2_s x == c2_s y ∧ c2_1 x == c2_1 y
               ∧ c2_2 x == c2_2 y ∧ c2_12 x == c2_12 y.

  #[global] Instance C2_Zero : Zero (C2 F) := mkC2 0 0 0 0.

  #[global] Instance C2_One : One (C2 F) := mkC2 1 0 0 0.

  #[global] Instance C2_Add : HasAdd (C2 F) :=
    fun x y ⇒ mkC2 (c2_s x + c2_s y) (c2_1 x + c2_1 y)
                    (c2_2 x + c2_2 y) (c2_12 x + c2_12 y).

  #[global] Instance C2_Neg : HasNeg (C2 F) :=
    fun x ⇒ mkC2 (- c2_s x) (- c2_1 x) (- c2_2 x) (- c2_12 x).

  #[global] Instance C2_Mult : HasMult (C2 F) :=
    fun x y ⇒ mkC2
      (c2_s x × c2_s y + (c2_1 x × c2_1 y + (c2_2 x × c2_2 y
         + - (c2_12 x × c2_12 y))))
      (c2_s x × c2_1 y + (c2_1 x × c2_s y + (c2_12 x × c2_2 y
         + - (c2_2 x × c2_12 y))))
      (c2_s x × c2_2 y + (c2_2 x × c2_s y + (c2_1 x × c2_12 y
         + - (c2_12 x × c2_1 y))))
      (c2_s x × c2_12 y + (c2_12 x × c2_s y + (c2_1 x × c2_2 y
         + - (c2_2 x × c2_1 y)))).

  #[global] Instance C2_Equivalence : Equivalence C2_EqRel.
  Proof.
    apply Build_Equivalence.
    - intros x. repeat split; reflexivity.
    - intros x y [Hs [H1 [H2 H12]]]. repeat split.
      + symmetry. exact Hs.
      + symmetry. exact H1.
      + symmetry. exact H2.
      + symmetry. exact H12.
    - intros x y v [Hs1 [H11 [H21 H121]]] [Hs2 [H12' [H22 H122]]].
      repeat split.
      + rewrite Hs1. exact Hs2.
      + rewrite H11. exact H12'.
      + rewrite H21. exact H22.
      + rewrite H121. exact H122.
  Qed.

End C2Ops.

  Lemma c2_add_assoc : ∀ x y v : C2 F, (x + y) + v == x + (y + v).
  Proof.
    intros x y v. repeat split.
    - exact (monoid_associative (c2_s x) (c2_s y) (c2_s v)).
    - exact (monoid_associative (c2_1 x) (c2_1 y) (c2_1 v)).
    - exact (monoid_associative (c2_2 x) (c2_2 y) (c2_2 v)).
    - exact (monoid_associative (c2_12 x) (c2_12 y) (c2_12 v)).
  Qed.

  Lemma c2_add_idl : ∀ x : C2 F, 0 + x == x.
  Proof.
    intros x. repeat split.
    - exact (monoid_left_identity (c2_s x)).
    - exact (monoid_left_identity (c2_1 x)).
    - exact (monoid_left_identity (c2_2 x)).
    - exact (monoid_left_identity (c2_12 x)).
  Qed.

  Lemma c2_add_idr : ∀ x : C2 F, x + 0 == x.
  Proof.
    intros x. repeat split.
    - exact (monoid_right_identity (c2_s x)).
    - exact (monoid_right_identity (c2_1 x)).
    - exact (monoid_right_identity (c2_2 x)).
    - exact (monoid_right_identity (c2_12 x)).
  Qed.

  #[global] Instance c2_add_proper : Proper (equ ==> equ ==> equ) (@add (C2 F) _).
  Proof.
    intros x x' [Hs [H1 [H2 H12]]] y y' [Hs' [H1' [H2' H12']]].
    repeat split.
    - exact (monoid_add_proper (c2_s x) (c2_s x') Hs (c2_s y) (c2_s y') Hs').
    - exact (monoid_add_proper (c2_1 x) (c2_1 x') H1 (c2_1 y) (c2_1 y') H1').
    - exact (monoid_add_proper (c2_2 x) (c2_2 x') H2 (c2_2 y) (c2_2 y') H2').
    - exact (monoid_add_proper (c2_12 x) (c2_12 x') H12 (c2_12 y) (c2_12 y') H12').
  Qed.

  Lemma c2_add_invl : ∀ x : C2 F, (- x) + x == 0.
  Proof.
    intros x. repeat split.
    - exact (group_left_inverse (c2_s x)).
    - exact (group_left_inverse (c2_1 x)).
    - exact (group_left_inverse (c2_2 x)).
    - exact (group_left_inverse (c2_12 x)).
  Qed.

  Lemma c2_add_invr : ∀ x : C2 F, x + (- x) == 0.
  Proof.
    intros x. repeat split.
    - exact (group_right_inverse (c2_s x)).
    - exact (group_right_inverse (c2_1 x)).
    - exact (group_right_inverse (c2_2 x)).
    - exact (group_right_inverse (c2_12 x)).
  Qed.

  #[global] Instance c2_neg_proper : Proper (equ ==> equ) (@neg (C2 F) _).
  Proof.
    intros x y [Hs [H1 [H2 H12]]]. repeat split.
    - exact (group_neg_proper (c2_s x) (c2_s y) Hs).
    - exact (group_neg_proper (c2_1 x) (c2_1 y) H1).
    - exact (group_neg_proper (c2_2 x) (c2_2 y) H2).
    - exact (group_neg_proper (c2_12 x) (c2_12 y) H12).
  Qed.

  Lemma c2_add_comm : ∀ x y : C2 F, x + y == y + x.
  Proof.
    intros x y. repeat split.
    - exact (ab_comm (c2_s x) (c2_s y)).
    - exact (ab_comm (c2_1 x) (c2_1 y)).
    - exact (ab_comm (c2_2 x) (c2_2 y)).
    - exact (ab_comm (c2_12 x) (c2_12 y)).
  Qed.

  Lemma cliff2_add4_swap : ∀ x1 y1 x2 y2 x3 y3 x4 y4 : F,
    (x1 + y1) + ((x2 + y2) + ((x3 + y3) + (x4 + y4)))
    == (x1 + (x2 + (x3 + x4))) + (y1 + (y2 + (y3 + y4))).
  Proof.
    intros x1 y1 x2 y2 x3 y3 x4 y4.
    rewrite (add_interchange x3 y3 x4 y4).
    rewrite (add_interchange x2 y2 (x3 + x4) (y3 + y4)).
    rewrite (add_interchange x1 y1 (x2 + (x3 + x4)) (y2 + (y3 + y4))).
    reflexivity.
  Qed.

    a1 × (b1 + c1) + (a2 × (b2 + c2) + (a3 × (b3 + c3) + - (a4 × (b4 + c4))))
    == (a1 × b1 + (a2 × b2 + (a3 × b3 + - (a4 × b4))))
       + (a1 × c1 + (a2 × c2 + (a3 × c3 + - (a4 × c4)))).

    rewrite ring_distl, ring_distl, ring_distl, ring_distl.
    rewrite neg_add.

    a3 == a3' → b3 == b3' → a4 == a4' → b4 == b4' →

    == a1' × b1' + (a2' × b2' + (a3' × b3' + - (a4' × b4'))).

  Qed.
