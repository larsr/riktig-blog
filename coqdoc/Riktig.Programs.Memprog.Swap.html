From Coq Require Import List ZArith.
Import ListNotations.
From Riktig Require Import Programs.Memprog.Syntax.
From Riktig Require Import Programs.Memprog.Machine.
From Riktig Require Import Programs.Memprog.Expr.
From Riktig Require Import Programs.Memprog.Frame.
From Riktig Require Import Programs.Memprog.MC.

Local Open Scope Z_scope.


Definition swap_l1 (base i j : Z) : MemT Z Z unit :=
  swap_cells base (ret_t i) (ret_t j).



Definition swap_l3_body : MemT Z Z unit :=
  with_frame (
    do! fp <- get_sp_m;
    do! j <- read_slot_z fp 2;
    do! i <- read_slot_z fp 3;
    do! base <- read_slot_addr_z fp 4;
    swap_cells base (ret_t i) (ret_t j)).


Definition swap_l3 (base i j : Z) : MemT Z Z unit :=
  call_frame (push_addr base >> push_val i >> push_val j)
             (fun _ ⇒ swap_l3_body).



Definition swap_l1_demo : MemT Z Z Z :=
  do! base <- list_to_mem Z.of_nat [10; 20; 30; 40; 50];
  swap_l1 base 1 3 >>
  ret_t base.

Example swap_l1_cells :
  (let '(r, m) := run_z 1000 swap_l1_demo 64 in mem_to_list m r 5)
  = Some [10; 40; 30; 20; 50].
Proof. vm_compute. reflexivity. Qed.
