Section AttentionLayer.
  Context {F : Type} {FF : Field F}.
  Context (n d : nat).
  Context (Q K V : Matrix n d F).

  Definition attn_simple : Matrix n d F :=
    mat_mul (mat_mul Q (mat_tr K)) V.

  Definition attn_efficient : Matrix n d F :=
    mat_mul Q (mat_mul (mat_tr K) V).

  Lemma attn_simple_is_naive : attn_simple = attn_naive n d Q K V.
  Proof. reflexivity. Qed.

  Lemma attn_efficient_is_fast : attn_efficient = attn_fast n d Q K V.
  Proof. reflexivity. Qed.


  Check (mat_mul_proper
    : ∀ m n p : nat,
        Proper (equ ==> equ ==> equ) (@mat_mul F FF m n p)).

  Theorem attn_equal : attn_simple == attn_efficient.
  Proof.
    unfold attn_simple, attn_efficient.
    rewrite mat_mul_assoc.
    reflexivity.
  Qed.


  Check (@mat_vec_lm F FF
    : ∀ m k : nat, Matrix m k F →
        LinearMap F (Fin.t k → F) (Fin.t m → F)).

  Check (mat_vec_lm_mul
    : ∀ (m k p : nat) (A : Matrix m k F) (B : Matrix k p F),
        mat_vec_lm (mat_mul A B) == lm_comp (mat_vec_lm A) (mat_vec_lm B)).

  Theorem attn_composite :
    mat_vec_lm attn_simple
      == lm_comp (mat_vec_lm Q)
           (lm_comp (mat_vec_lm (mat_tr K)) (mat_vec_lm V)).
  Proof.
    intros w.
    exact (attn_pipeline n d Q K V w).
  Qed.

  Corollary attn_apply : ∀ w : Fin.t d → F,
    mat_vec attn_simple w
      == mat_vec Q (mat_vec (mat_tr K) (mat_vec V w)).
  Proof.
    exact (attn_pipeline n d Q K V).
  Qed.

End AttentionLayer.

Example attn_simple_Q :
     attn_simple 3 2 Qm3 Km3 Vm3 Fin.F1 Fin.F1 == (9#1)%Q
  ∧ attn_simple 3 2 Qm3 Km3 Vm3 Fin.F1 (Fin.FS Fin.F1) == (7#1)%Q
  ∧ attn_simple 3 2 Qm3 Km3 Vm3 (Fin.FS Fin.F1) Fin.F1 == (2#1)%Q
  ∧ attn_simple 3 2 Qm3 Km3 Vm3 (Fin.FS Fin.F1) (Fin.FS Fin.F1) == (3#1)%Q
  ∧ attn_simple 3 2 Qm3 Km3 Vm3 (Fin.FS (Fin.FS Fin.F1)) Fin.F1 == (3#1)%Q
  ∧ attn_simple 3 2 Qm3 Km3 Vm3 (Fin.FS (Fin.FS Fin.F1)) (Fin.FS Fin.F1)
       == ((-2)#1)%Q.
Proof. exact attn_naive_Q_compute. Qed.

Example attn_efficient_Q :
     attn_efficient 3 2 Qm3 Km3 Vm3 Fin.F1 Fin.F1 == (9#1)%Q
  ∧ attn_efficient 3 2 Qm3 Km3 Vm3 Fin.F1 (Fin.FS Fin.F1) == (7#1)%Q
  ∧ attn_efficient 3 2 Qm3 Km3 Vm3 (Fin.FS Fin.F1) Fin.F1 == (2#1)%Q
  ∧ attn_efficient 3 2 Qm3 Km3 Vm3 (Fin.FS Fin.F1) (Fin.FS Fin.F1) == (3#1)%Q
  ∧ attn_efficient 3 2 Qm3 Km3 Vm3 (Fin.FS (Fin.FS Fin.F1)) Fin.F1 == (3#1)%Q
  ∧ attn_efficient 3 2 Qm3 Km3 Vm3 (Fin.FS (Fin.FS Fin.F1)) (Fin.FS Fin.F1)
       == ((-2)#1)%Q.
Proof. exact attn_fast_Q_compute. Qed.

Example attn_equal_Q :
  attn_simple 3 2 Qm3 Km3 Vm3 == attn_efficient 3 2 Qm3 Km3 Vm3.
Proof. exact (attn_equal 3 2 Qm3 Km3 Vm3). Qed.

Example attn_apply_Q :
  mat_vec (attn_simple 3 2 Qm3 Km3 Vm3) w2
    == mat_vec Qm3 (mat_vec (mat_tr Km3) (mat_vec Vm3 w2)).
Proof. exact (attn_apply 3 2 Qm3 Km3 Vm3 w2). Qed.
