Section NormTheory.
  Context {F V : Type} `{NormedSpace F V}.

  Theorem norm_zero : ‖ (0 : V) ‖ == 0.
  Proof.
    calc (‖ (0 : V) ‖)
      == (‖ (0 : F) *: (0 : V) ‖) { zero_scale }
      == (abs (0 : F) × ‖ (0 : V) ‖) { norm_scale }
      == ((0 : F) × ‖ (0 : V) ‖) { abs_zero }
      == (0 : F) { mul_0_l }.
  Qed.

  Theorem norm_neg : ∀ u : V, ‖ - u ‖ == ‖ u ‖.
  Proof.
    intros u.
    calc (‖ - u ‖)
      == (‖ (- (1 : F)) *: u ‖) { neg1_scale }
      == (abs (- (1 : F)) × ‖ u ‖) { norm_scale }
      == (abs (1 : F) × ‖ u ‖) { abs_neg }
      == ((1 : F) × ‖ u ‖) { abs_one }
      == (‖ u ‖) { ring_mul_1l }.
  Qed.

  Theorem norm_sub_self : ∀ u : V, ‖ u + - u ‖ == 0.
  Proof.
    intros u.
    calc (‖ u + - u ‖)
      == (‖ (0 : V) ‖) { invr }
      == (0 : F) { norm_zero }.
  Qed.

  Theorem norm_sub_self_lt : ∀ (u : V) (d : F),
    0 < d → ‖ u + - u ‖ < d.
  Proof.
    intros u d Hd.
    apply (le_lt_trans (‖ u + - u ‖) 0 d).
    - apply eq_le. apply norm_sub_self.
    - exact Hd.
  Qed.

  Theorem norm_neg_add : ∀ u v : V, ‖ - u + - v ‖ == ‖ u + v ‖.
  Proof.
    intros u v.
    calc (‖ - u + - v ‖)
      == (‖ - (u + v) ‖) { neg_add }
      == (‖ u + v ‖) { norm_neg }.
  Qed.

  Theorem norm_sub_swap : ∀ u v : V, ‖ u + - v ‖ == ‖ v + - u ‖.
  Proof.
    intros u v.
    calc (‖ u + - v ‖)
      == (‖ - (u + - v) ‖) { norm_neg }
      == (‖ - u + - (- v) ‖) { neg_add }
      == (‖ - u + v ‖) { neg_involutive }
      == (‖ v + - u ‖) { ab_comm }.
  Qed.

  Theorem norm_sub_triangle : ∀ u v w : V,
    ‖ u + - w ‖ ≤ ‖ u + - v ‖ + ‖ v + - w ‖.
  Proof.
    intros u v w.
    calc (‖ u + - w ‖)
      == (‖ (u + - v) + (v + - w) ‖) { add_neg_telescope }
      == (‖ u + - v ‖ + ‖ v + - w ‖) { (norm_triangle (u + - v) (v + - w)) }.
  Qed.

  Theorem norm_sub_pair : ∀ u v a b : V,
    ‖ (u + v) + - (a + b) ‖ ≤ ‖ u + - a ‖ + ‖ v + - b ‖.
  Proof.
    intros u v a b.
    calc (‖ (u + v) + - (a + b) ‖)
      == (‖ (u + - a) + (v + - b) ‖) { (sub_pair u v a b) }
      == (‖ u + - a ‖ + ‖ v + - b ‖) { (norm_triangle (u + - a) (v + - b)) }.
  Qed.

  Theorem norm_scale_sub : ∀ (a : F) (u v : V),
    ‖ a *: u + - (a *: v) ‖ == abs a × ‖ u + - v ‖.
  Proof.
    intros a u v.
    calc (‖ a *: u + - (a *: v) ‖)
      == (‖ a *: (u + - v) ‖) { scale_sub }
      == (abs a × ‖ u + - v ‖) { norm_scale }.
  Qed.

  Theorem norm_scale_nonneg : ∀ (a : F) (u : V),
    0 ≤ a → ‖ a *: u ‖ == a × ‖ u ‖.
  Proof.
    intros a u Ha.
    calc (‖ a *: u ‖)
      == (abs a × ‖ u ‖) { norm_scale }
      == (a × ‖ u ‖) { (abs_nonneg_id a Ha) }.
  Qed.

  Theorem norm_sub_le_add : ∀ u v : V, ‖ u + - v ‖ ≤ ‖ u ‖ + ‖ v ‖.
  Proof.
    intros u v.
    calc (‖ u + - v ‖)
      == (‖ u ‖ + ‖ - v ‖) { (norm_triangle u (- v)) }
      == (‖ u ‖ + ‖ v ‖) { norm_neg }.
  Qed.

End NormTheory.
