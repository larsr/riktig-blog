From Coq Require Import RelationClasses.

Ltac calc_ends x y :=
  lazymatch goal with
  | |- _ ?l ?r ⇒ unify l x; unify r y
  end.

Ltac calc_by r :=
  first [ now rewrite r
        | now rewrite <- r
        | now apply r
        | now (symmetry; apply r) ].
