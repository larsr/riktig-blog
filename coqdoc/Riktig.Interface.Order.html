Library Riktig.Interface.Order

From Coq Require Export Morphisms.
From Riktig Require Export Ops Interface.Ring Interface.Field.

Class HasLe T := le : Rel T.
Infix "≤" := le (at level 70, no associativity).

Class HasAbs T := abs : T → T.

#[global] Hint Mode HasLe ! : typeclass_instances.
#[global] Hint Mode HasAbs ! : typeclass_instances.

Definition lt {T : Type} `{EqRel T} `{HasLe T} (a b : T) : Prop :=
  a ≤ b ∧ a =/= b.
Infix "<" := lt (at level 70, no associativity).

Class OrderedField F `{Field F} := {
  #[global] of_le :: HasLe F;
  #[global] of_abs :: HasAbs F;
  le_refl : ∀ a : F, a ≤ a;
  le_trans : ∀ a b c : F, a ≤ b → b ≤ c → a ≤ c;
  le_antisym : ∀ a b : F, a ≤ b → b ≤ a → a == b;
  le_total : ∀ a b : F, a ≤ b ∨ b ≤ a;
  
  le_stable : ∀ a b : F, ¬ ¬ a ≤ b → a ≤ b;
  #[global] le_proper :: Proper (equ ==> equ ==> iff) (@le F _);
  add_le_mono_l : ∀ a b c : F, a ≤ b → c + a ≤ c + b;
  mul_nonneg : ∀ a b : F, 0 ≤ a → 0 ≤ b → 0 ≤ a × b;
  #[global] abs_proper :: Proper (equ ==> equ) (@abs F _);
  abs_nonneg : ∀ a : F, 0 ≤ abs a;
  abs_zero_iff : ∀ a : F, abs a == 0 ↔ a == 0;
  abs_mul : ∀ a b : F, abs (a × b) == abs a × abs b;
  
  abs_neg : ∀ a : F, abs (- a) == abs a;
  abs_triangle : ∀ a b : F, abs (a + b) ≤ abs a + abs b;
  abs_nonneg_id : ∀ a : F, 0 ≤ a → abs a == a
}.

#[global] Hint Mode OrderedField ! - : typeclass_instances.
