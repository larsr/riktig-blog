Library Riktig.Impl.CoqCat


From Coq Require Export Morphisms FunctionalExtensionality.
From Riktig Require Export Interface.Category.


Definition fn_equ (X Y : Type) (f g : X → Y) : Prop :=
  ∀ x : X, f x = g x.

Definition fn_id (X : Type) : X → X := fun x ⇒ x.

Definition fn_comp (X Y Z : Type) (g : Y → Z) (f : X → Y) : X → Z :=
  fun x ⇒ g (f x).

Lemma fn_equ_refl (X Y : Type) (f : X → Y) : fn_equ X Y f f.
Proof.
  intros x. reflexivity.
Qed.

Lemma fn_equ_sym (X Y : Type) (f g : X → Y) :
  fn_equ X Y f g → fn_equ X Y g f.
Proof.
  intros Hfg x. symmetry. exact (Hfg x).
Qed.

Lemma fn_equ_trans (X Y : Type) (f g h : X → Y) :
  fn_equ X Y f g → fn_equ X Y g h → fn_equ X Y f h.
Proof.
  intros Hfg Hgh x. rewrite (Hfg x). exact (Hgh x).
Qed.

#[global] Instance fn_equ_equivalence (X Y : Type) : Equivalence (fn_equ X Y).
Proof.
  apply Build_Equivalence.
  - exact (fn_equ_refl X Y).
  - exact (fn_equ_sym X Y).
  - exact (fn_equ_trans X Y).
Qed.

Lemma fn_comp_ext (X Y Z : Type) (g g' : Y → Z) (f f' : X → Y) :
  fn_equ Y Z g g' → fn_equ X Y f f' →
  ∀ x : X, g (f x) = g' (f' x).
Proof.
  intros Hg Hf x. rewrite (Hf x). exact (Hg (f' x)).
Qed.

#[global] Instance fn_comp_proper (X Y Z : Type) :
  Proper (fn_equ Y Z ==> fn_equ X Y ==> fn_equ X Z) (fn_comp X Y Z).
Proof.
  intros g g' Hg f f' Hf. exact (fn_comp_ext X Y Z g g' f f' Hg Hf).
Qed.

Lemma fn_idl (X Y : Type) (f : X → Y) :
  fn_equ X Y (fn_comp X Y Y (fn_id Y) f) f.
Proof.
  intros x. reflexivity.
Qed.

Lemma fn_idr (X Y : Type) (f : X → Y) :
  fn_equ X Y (fn_comp X X Y f (fn_id X)) f.
Proof.
  intros x. reflexivity.
Qed.

Lemma fn_assoc (W X Y Z : Type) (f : Y → Z) (g : X → Y) (h : W → X) :
  fn_equ W Z (fn_comp W Y Z f (fn_comp W X Y g h))
             (fn_comp W X Z (fn_comp X Y Z f g) h).
Proof.
  intros x. reflexivity.
Qed.

#[global] Instance CoqCat : Category Type := {|
  hom := fun X Y : Type ⇒ X → Y;
  hom_equ := fn_equ;
  hom_equiv := fn_equ_equivalence;
  cat_id := fn_id;
  comp := fn_comp;
  comp_proper := fn_comp_proper;
  cat_idl := fn_idl;
  cat_idr := fn_idr;
  cat_assoc := fn_assoc
|}.


Lemma unit_bang_unique (X : Type) (f : X → unit) :
  fn_equ X unit f (fun _ : X ⇒ tt).
Proof.
  intros x. destruct (f x). reflexivity.
Qed.

#[global] Instance CoqTerminal : Terminal CoqCat :=
  @Build_Terminal Type CoqCat
    unit
    (fun X : Type ⇒ fun _ : X ⇒ tt)
    unit_bang_unique.


Definition empty_emit (X : Type) : Empty_set → X :=
  fun e ⇒ match e with end.

Lemma empty_emit_unique (X : Type) (f : Empty_set → X) :
  fn_equ Empty_set X f (empty_emit X).
Proof.
  intros e. destruct e.
Qed.

#[global] Instance CoqInitial : Initial CoqCat :=
  @Build_Initial Type CoqCat
    Empty_set
    empty_emit
    empty_emit_unique.


Definition fn_pair (X A B : Type) (f : X → A) (g : X → B)
  : X → prod A B :=
  fun x ⇒ (f x, g x).

Lemma fn_pair_ext (X A B : Type) (f f' : X → A) (g g' : X → B) :
  fn_equ X A f f' → fn_equ X B g g' →
  ∀ x : X, (f x, g x) = (f' x, g' x).
Proof.
  intros Hf Hg x. rewrite (Hf x), (Hg x). reflexivity.
Qed.

#[global] Instance fn_pair_proper (X A B : Type) :
  Proper (fn_equ X A ==> fn_equ X B ==> fn_equ X (prod A B))
         (fn_pair X A B).
Proof.
  intros f f' Hf g g' Hg. exact (fn_pair_ext X A B f f' g g' Hf Hg).
Qed.

Lemma fn_pair_pi1 (X A B : Type) (f : X → A) (g : X → B) :
  fn_equ X A (fn_comp X (prod A B) A (@fst A B) (fn_pair X A B f g)) f.
Proof.
  intros x. reflexivity.
Qed.

Lemma fn_pair_pi2 (X A B : Type) (f : X → A) (g : X → B) :
  fn_equ X B (fn_comp X (prod A B) B (@snd A B) (fn_pair X A B f g)) g.
Proof.
  intros x. reflexivity.
Qed.

Lemma fn_pair_unique (X A B : Type) (f : X → A) (g : X → B)
    (h : X → prod A B) :
  (∀ x : X, fst (h x) = f x) →
  (∀ x : X, snd (h x) = g x) →
  ∀ x : X, h x = (f x, g x).
Proof.
  intros Hf Hg x.
  rewrite <- (Hf x), <- (Hg x).
  destruct (h x). reflexivity.
Qed.

#[global] Instance CoqProducts : HasProducts CoqCat :=
  @Build_HasProducts Type CoqCat
    prod
    (@fst)
    (@snd)
    fn_pair
    fn_pair_proper
    fn_pair_pi1
    fn_pair_pi2
    fn_pair_unique.


Definition fn_case (X A B : Type) (f : A → X) (g : B → X)
  : sum A B → X :=
  fun s ⇒ match s with
           | inl a ⇒ f a
           | inr b ⇒ g b
           end.

Lemma fn_case_ext (X A B : Type) (f f' : A → X) (g g' : B → X) :
  fn_equ A X f f' → fn_equ B X g g' →
  ∀ s : sum A B, fn_case X A B f g s = fn_case X A B f' g' s.
Proof.
  intros Hf Hg s. destruct s as [a | b].
  - exact (Hf a).
  - exact (Hg b).
Qed.

#[global] Instance fn_case_proper (X A B : Type) :
  Proper (fn_equ A X ==> fn_equ B X ==> fn_equ (sum A B) X)
         (fn_case X A B).
Proof.
  intros f f' Hf g g' Hg. exact (fn_case_ext X A B f f' g g' Hf Hg).
Qed.

Lemma fn_case_inj1 (X A B : Type) (f : A → X) (g : B → X) :
  fn_equ A X (fn_comp A (sum A B) X (fn_case X A B f g) (@inl A B)) f.
Proof.
  intros a. reflexivity.
Qed.

Lemma fn_case_inj2 (X A B : Type) (f : A → X) (g : B → X) :
  fn_equ B X (fn_comp B (sum A B) X (fn_case X A B f g) (@inr A B)) g.
Proof.
  intros b. reflexivity.
Qed.

Lemma fn_case_unique (X A B : Type) (f : A → X) (g : B → X)
    (h : sum A B → X) :
  (∀ a : A, h (inl a) = f a) →
  (∀ b : B, h (inr b) = g b) →
  ∀ s : sum A B, h s = fn_case X A B f g s.
Proof.
  intros Hf Hg s. destruct s as [a | b].
  - exact (Hf a).
  - exact (Hg b).
Qed.

#[global] Instance CoqCoproducts : HasCoproducts CoqCat :=
  @Build_HasCoproducts Type CoqCat
    sum
    (@inl)
    (@inr)
    fn_case
    fn_case_proper
    fn_case_inj1
    fn_case_inj2
    fn_case_unique.


Definition fn_eval (A B : Type) : prod (A → B) A → B :=
  fun p ⇒ fst p (snd p).

Definition fn_curry (X A B : Type) (h : prod X A → B) : X → A → B :=
  fun x a ⇒ h (x, a).

Lemma fn_curry_ext (X A B : Type) (h h' : prod X A → B) :
  fn_equ (prod X A) B h h' →
  ∀ x : X, fn_curry X A B h x = fn_curry X A B h' x.
Proof.
  intros Hh x.
  apply functional_extensionality. intros a.
  exact (Hh (x, a)).
Qed.

#[global] Instance fn_curry_proper (X A B : Type) :
  Proper (fn_equ (prod X A) B ==> fn_equ X (A → B)) (fn_curry X A B).
Proof.
  intros h h' Hh. exact (fn_curry_ext X A B h h' Hh).
Qed.

Lemma fn_curry_beta (X A B : Type) (h : prod X A → B) :
  ∀ p : prod X A, h (fst p, snd p) = h p.
Proof.
  intros p. destruct p as [x a]. reflexivity.
Qed.

Lemma fn_curry_unique (X A B : Type) (h : prod X A → B)
    (k : X → A → B) :
  (∀ p : prod X A, k (fst p) (snd p) = h p) →
  ∀ x : X, k x = fn_curry X A B h x.
Proof.
  intros Hk x.
  apply functional_extensionality. intros a.
  exact (Hk (x, a)).
Qed.

#[global] Instance CoqExponentials :
  @HasExponentials Type CoqCat CoqProducts :=
  @Build_HasExponentials Type CoqCat CoqProducts
    (fun A B : Type ⇒ A → B)
    fn_eval
    fn_curry
    fn_curry_proper
    fn_curry_beta
    fn_curry_unique.


#[global] Instance CoqCCC : CCC CoqCat :=
  @Build_CCC Type CoqCat
    CoqTerminal
    CoqProducts
    CoqExponentials.
