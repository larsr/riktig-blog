Section BMBounded.
  Context {F U V W : Type} `{NormedSpace F U}.
  Context {VSV : VectorSpace F V} {NSV : NormedSpace F V}.
  Context {VSW : VectorSpace F W} {NSW : NormedSpace F W}.

  Definition bm_bounded (B : BilinearMap F U V W) : Prop :=
    ∃ C : F, 0 ≤ C ∧ ∀ (u : U) (v : V), ‖ B u v ‖ ≤ C × ‖ u ‖ × ‖ v ‖.

End BMBounded.


Section BilinearDerivative.
  Context {F U V W : Type} `{NormedSpace F U}.
  Context {VSV : VectorSpace F V} {NSV : NormedSpace F V}.
  Context {VSW : VectorSpace F W} {NSW : NormedSpace F W}.
  Context (B : BilinearMap F U V W) (u0 : U) (v0 : V).


  Definition bilinear_deriv_fun (p : U × V) : W :=
    B (fst p) v0 + B u0 (snd p).

  Lemma bilinear_deriv_proper : Proper (equ ==> equ) bilinear_deriv_fun.
  Proof.
    intros [du1 dv1] [du2 dv2] [Hu Hv].
    unfold bilinear_deriv_fun; simpl.
    rewrite (bm_proper B du1 du2 Hu v0 v0 (reflexivity v0)),
      (bm_proper B u0 u0 (reflexivity u0) dv1 dv2 Hv).
    reflexivity.
  Qed.

  Lemma bilinear_deriv_add : ∀ p q : U × V,
    bilinear_deriv_fun (p + q) == bilinear_deriv_fun p + bilinear_deriv_fun q.
  Proof.
    intros [du1 dv1] [du2 dv2].
    unfold bilinear_deriv_fun; simpl.
    rewrite (bm_add_l B du1 du2 v0), (bm_add_r B u0 dv1 dv2).
    apply add_interchange.
  Qed.

  Lemma bilinear_deriv_scale : ∀ (a : F) (p : U × V),
    bilinear_deriv_fun (a *: p) == a *: bilinear_deriv_fun p.
  Proof.
    intros a [du dv].
    unfold bilinear_deriv_fun; simpl.
    rewrite (bm_scale_l B a du v0), (bm_scale_r B a u0 dv).
    symmetry.
    apply sc_scale_add.
  Qed.

  Definition bilinear_deriv_lm : LinearMap F (U × V) W :=
    @Build_LinearMap F (U × V) W _ _ VSW bilinear_deriv_fun
      bilinear_deriv_proper bilinear_deriv_add bilinear_deriv_scale.


  Lemma bilinear_deriv_bounded : bm_bounded B → lm_bounded bilinear_deriv_lm.
  Proof.
    intros [C [HC HCb]].
    ∃ (C × ‖ v0 ‖ + C × ‖ u0 ‖).
    split.
    - apply (le_trans 0 (0 + 0) (C × ‖ v0 ‖ + C × ‖ u0 ‖)).
      + apply eq_le. symmetry. apply mon_idl.
      + apply add_le_mono.
        × apply mul_nonneg; [exact HC | apply norm_nonneg].
        × apply mul_nonneg; [exact HC | apply norm_nonneg].
    - intros [du dv].
      unfold lm_fun, bilinear_deriv_lm, bilinear_deriv_fun; simpl.
      unfold vnorm, Prod_HasNorm, prod_norm; simpl.
      apply (le_trans (‖ B du v0 + B u0 dv ‖) (‖ B du v0 ‖ + ‖ B u0 dv ‖)
        ((C × ‖ v0 ‖ + C × ‖ u0 ‖) × (‖ du ‖ + ‖ dv ‖))).
      + apply norm_triangle.
      + apply (le_trans (‖ B du v0 ‖ + ‖ B u0 dv ‖)
          ((C × ‖ v0 ‖ + C × ‖ u0 ‖) × ‖ du ‖
             + (C × ‖ v0 ‖ + C × ‖ u0 ‖) × ‖ dv ‖)
          ((C × ‖ v0 ‖ + C × ‖ u0 ‖) × (‖ du ‖ + ‖ dv ‖))).
        × apply add_le_mono.
          -- apply (le_trans (‖ B du v0 ‖) (C × ‖ v0 ‖ × ‖ du ‖)
               ((C × ‖ v0 ‖ + C × ‖ u0 ‖) × ‖ du ‖)).
             ++ apply (le_trans (‖ B du v0 ‖) (C × ‖ du ‖ × ‖ v0 ‖)
                  (C × ‖ v0 ‖ × ‖ du ‖)).
                ** exact (HCb du v0).
                ** apply eq_le. apply mul_assoc_comm_r.
             ++ apply mul_le_mono_r.
                ** apply norm_nonneg.
                ** apply le_add_nonneg_r.
                   apply mul_nonneg; [exact HC | apply norm_nonneg].
          -- apply (le_trans (‖ B u0 dv ‖) (C × ‖ u0 ‖ × ‖ dv ‖)
               ((C × ‖ v0 ‖ + C × ‖ u0 ‖) × ‖ dv ‖)).
             ++ exact (HCb u0 dv).
             ++ apply mul_le_mono_r.
                ** apply norm_nonneg.
                ** apply le_add_nonneg_l.
                   apply mul_nonneg; [exact HC | apply norm_nonneg].
        × apply eq_le. symmetry. apply ring_distl.
  Qed.


  Lemma bilinear_remainder_eq : ∀ (du : U) (dv : V),
    B (u0 + du) (v0 + dv) + - (B u0 v0 + (B du v0 + B u0 dv)) == B du dv.
  Proof.
    intros du dv.
    rewrite (bm_add_r B (u0 + du) v0 dv), (bm_add_l B u0 du v0),
      (bm_add_l B u0 du dv).
    exact (four_term_cancel (B u0 v0) (B du v0) (B u0 dv) (B du dv)).
  Qed.


  Lemma bilinear_remainder_bound : ∀ C : F, 0 ≤ C →
    (∀ (u : U) (v : V), ‖ B u v ‖ ≤ C × ‖ u ‖ × ‖ v ‖) →
    ∀ eps : F, 0 < eps →
    ∀ (du : U) (dv : V),
      ‖ du ‖ + ‖ dv ‖ < eps × inv (1 + C) →
      ‖ B du dv ‖ ≤ eps × (‖ du ‖ + ‖ dv ‖).
  Proof.
    intros C HC HCb eps Heps du dv Hh.
    assert (Hdu_h : ‖ du ‖ ≤ ‖ du ‖ + ‖ dv ‖).
    { apply le_add_nonneg_r. apply norm_nonneg. }
    assert (Hdv_h : ‖ dv ‖ ≤ ‖ du ‖ + ‖ dv ‖).
    { apply le_add_nonneg_l. apply norm_nonneg. }
    assert (Hh_nonneg : 0 ≤ ‖ du ‖ + ‖ dv ‖).
    { apply (le_trans 0 (‖ du ‖) (‖ du ‖ + ‖ dv ‖)).
      - apply norm_nonneg.
      - exact Hdu_h. }
    apply (le_trans (‖ B du dv ‖) (C × ‖ du ‖ × ‖ dv ‖)
      (eps × (‖ du ‖ + ‖ dv ‖))).
    - exact (HCb du dv).
    - apply (le_trans (C × ‖ du ‖ × ‖ dv ‖)
        (C × ((‖ du ‖ + ‖ dv ‖) × (‖ du ‖ + ‖ dv ‖)))
        (eps × (‖ du ‖ + ‖ dv ‖))).
      + apply (le_trans (C × ‖ du ‖ × ‖ dv ‖) (C × (‖ du ‖ × ‖ dv ‖))
          (C × ((‖ du ‖ + ‖ dv ‖) × (‖ du ‖ + ‖ dv ‖)))).
        × apply eq_le. apply ring_mul_assoc.
        × apply mul_le_mono_l.
          -- exact HC.
          -- apply (le_trans (‖ du ‖ × ‖ dv ‖)
               ((‖ du ‖ + ‖ dv ‖) × ‖ dv ‖)
               ((‖ du ‖ + ‖ dv ‖) × (‖ du ‖ + ‖ dv ‖))).
             ++ apply mul_le_mono_r; [apply norm_nonneg | exact Hdu_h].
             ++ apply mul_le_mono_l; [exact Hh_nonneg | exact Hdv_h].
      + apply (le_trans (C × ((‖ du ‖ + ‖ dv ‖) × (‖ du ‖ + ‖ dv ‖)))
          ((C × (‖ du ‖ + ‖ dv ‖)) × (‖ du ‖ + ‖ dv ‖))
          (eps × (‖ du ‖ + ‖ dv ‖))).
        × apply eq_le. symmetry. apply ring_mul_assoc.
        × apply mul_le_mono_r.
          -- exact Hh_nonneg.
          -- apply (le_trans (C × (‖ du ‖ + ‖ dv ‖)) (C × (eps × inv (1 + C)))
               eps).
             ++ apply mul_le_mono_l; [exact HC | apply lt_le; exact Hh].
             ++ apply mul_pad_div_le; [apply lt_le; exact Heps | exact HC].
  Qed.


  Theorem bilinear_is_deriv : bm_bounded B →
    is_deriv (fun p : U × V ⇒ B (fst p) (snd p)) (u0, v0) bilinear_deriv_lm.
  Proof.
    intros Hbb.
    split.
    - exact (bilinear_deriv_bounded Hbb).
    - destruct Hbb as [C [HC HCb]].
      intros eps Heps.
      ∃ (eps × inv (1 + C)).
      split.
      + apply mul_pos.
        × exact Heps.
        × apply pos_inv. apply one_plus_nonneg_pos. exact HC.
      + intros [du dv] Hh.
        simpl.
        unfold lm_fun, bilinear_deriv_lm, bilinear_deriv_fun; simpl.
        apply (le_trans (‖ B (u0 + du) (v0 + dv) + - (B u0 v0 + (B du v0 + B u0 dv)) ‖)
          (‖ B du dv ‖) (eps × ‖ (du, dv) ‖)).
        × apply eq_le. apply norm_proper. exact (bilinear_remainder_eq du dv).
        × unfold vnorm, Prod_HasNorm, prod_norm; simpl.
          exact (bilinear_remainder_bound C HC HCb eps Heps du dv Hh).
  Qed.

End BilinearDerivative.
