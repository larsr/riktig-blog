Library Riktig.Impl.OptionMonad

From Coq Require Import Morphisms.
From Riktig Require Import Ops Interface.Monad.

#[global] Instance Option_EqRel (A : Type) : EqRel (option A) := eq.

#[global] Instance Option_HasBind : HasBind option :=
  fun A B m f ⇒ match m with
                 | None ⇒ None
                 | Some x ⇒ f x
                 end.

#[global] Instance Option_HasReturn : HasReturn option :=
  fun A x ⇒ Some x.

Lemma option_bind_proper : ∀ A B,
  Proper (equ ==> (fun f g ⇒ ∀ x, f x == g x) ==> equ) (@bind option _ A B).
Proof.
  intros A B m1 m2 Hm f g Hfg.
  unfold equ, Option_EqRel in Hm.
  destruct Hm.
  destruct m1 as [x |]; simpl.
  - exact (Hfg x).
  - reflexivity.
Qed.

Lemma option_bind_ret_l : ∀ A B (x : A) (f : A → option B),
  bind (ret x) f == f x.
Proof. intros A B x f. reflexivity. Qed.

Lemma option_bind_ret_r : ∀ A (m : option A), bind m ret == m.
Proof. intros A m. destruct m as [x |]; reflexivity. Qed.

Lemma option_bind_assoc : ∀ A B C (m : option A) (f : A → option B) (g : B → option C),
  bind (bind m f) g == bind m (fun x ⇒ bind (f x) g).
Proof. intros A B C m f g. destruct m as [x |]; reflexivity. Qed.

#[global] Instance Option_Monad : Monad option := {|
  monad_equiv := fun A ⇒ eq_equivalence;
  bind_proper := option_bind_proper;
  bind_ret_l := option_bind_ret_l;
  bind_ret_r := option_bind_ret_r;
  bind_assoc := option_bind_assoc
|}.

Example option_monad_check : Monad option := _.
