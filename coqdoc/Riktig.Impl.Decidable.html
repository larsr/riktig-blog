Library Riktig.Impl.Decidable


Require Import Arith PeanoNat Bool.
From Riktig Require Import Interface.Decidable.


#[global] Instance nat_eq_dec : EqDec nat := Nat.eq_dec.

#[global] Instance nat_lt_dec (n m : nat) : Decidable (n < m) := lt_dec n m.
#[global] Instance nat_le_dec (n m : nat) : Decidable (n ≤ m) := le_dec n m.
#[global] Instance nat_gt_dec (n m : nat) : Decidable (n > m) := lt_dec m n.
#[global] Instance nat_ge_dec (n m : nat) : Decidable (n ≥ m) := le_dec m n.


#[global] Instance bool_eq_dec : EqDec bool := Bool.bool_dec.

#[global] Instance bool_true_dec (b : bool) : Decidable (b = true).
destruct b; [left; reflexivity | right; discriminate].
Defined.

#[global] Instance bool_false_dec (b : bool) : Decidable (b = false).
destruct b; [right; discriminate | left; reflexivity].
Defined.
