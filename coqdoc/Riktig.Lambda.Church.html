Library Riktig.Lambda.Church


Require Import Riktig.Lambda.Syntax.
From Riktig Require Import Ops.
Require Import Riktig.Lambda.Conv. From Coq Require Import Morphisms.



Definition TRUE : tm := Abs (Abs (Var 1)). Definition FALSE : tm := Abs (Abs (Var 0)).
Definition IFB (b t e : tm) : tm := App (App b t) e.

Definition PAIR : tm :=
  Abs (Abs (Abs (App (App (Var 0) (Var 2)) (Var 1)))).
Definition FST (p : tm) : tm := App p TRUE.
Definition SND (p : tm) : tm := App p FALSE.

#[global] Instance IFB_Proper : Proper (equ ==> equ ==> equ ==> equ) IFB.
Proof.
  intros b b' Hb t t' Ht e e' He.
  apply App_Proper.
  - apply App_Proper.
    + exact Hb.
    + exact Ht.
  - exact He.
Qed.


Lemma IF_TRUE : ∀ g h, value g → value h → IFB TRUE g h == g.
Proof.
  intros g h Hg Hh.
  eapply conv_trans.
  { apply conv_appl. apply conv_beta. exact Hg. }
  cbn.
  eapply conv_trans.
  { apply conv_beta. exact Hh. }
  rewrite subst_lift. reflexivity.
Qed.

Lemma IF_FALSE : ∀ g h, value g → value h → IFB FALSE g h == h.
Proof.
  intros g h Hg Hh.
  eapply conv_trans.
  { apply conv_appl. apply conv_beta. exact Hg. }
  cbn.
  eapply conv_trans.
  { apply conv_beta. exact Hh. }
  cbn. rewrite lift_0. reflexivity.
Qed.

Lemma FST_PAIR : ∀ v w, value v → value w →
  FST (App (App PAIR v) w) == v.
Proof.
  intros v w Hv Hw.
  eapply conv_trans.
  { apply conv_appl. apply conv_appl. apply conv_beta. exact Hv. }
  cbn.
  eapply conv_trans.
  { apply conv_appl. apply conv_beta. exact Hw. }
  cbn. rewrite subst_lift_lower.
  eapply conv_trans.
  { apply conv_beta. apply value_abs. }
  cbn. rewrite (subst_lift v), (subst_lift w).
  exact (IF_TRUE v w Hv Hw).
Qed.

Lemma SND_PAIR : ∀ v w, value v → value w →
  SND (App (App PAIR v) w) == w.
Proof.
  intros v w Hv Hw.
  eapply conv_trans.
  { apply conv_appl. apply conv_appl. apply conv_beta. exact Hv. }
  cbn.
  eapply conv_trans.
  { apply conv_appl. apply conv_beta. exact Hw. }
  cbn. rewrite subst_lift_lower.
  eapply conv_trans.
  { apply conv_beta. apply value_abs. }
  cbn. rewrite (subst_lift v), (subst_lift w).
  exact (IF_FALSE v w Hv Hw).
Qed.


Lemma if_of_true : ∀ b g h,
  value g → value h → b == TRUE → IFB b g h == g.
Proof.
  intros b g h Hg Hh Hb. rewrite Hb. apply IF_TRUE; assumption.
Qed.

Lemma if_of_false : ∀ b g h,
  value g → value h → b == FALSE → IFB b g h == h.
Proof.
  intros b g h Hg Hh Hb. rewrite Hb. apply IF_FALSE; assumption.
Qed.

Example fst_pair_closed : FST (App (App PAIR idt) k) == idt.
Proof. apply (conv_by_eval 20 10). reflexivity. Qed.
