Fixpoint nth0 {F : Type} {ZF : Zero F} (xs : list F) (i : nat) : F :=
  match xs with
  | nil ⇒ 0
  | cons x t ⇒
      match i with
      | O ⇒ x
      | S j ⇒ nth0 t j
      end
  end.

Fixpoint ladd {F : Type} {AF : HasAdd F} (xs ys : list F) : list F :=
  match xs with
  | nil ⇒ ys
  | cons x xt ⇒
      match ys with
      | nil ⇒ cons x xt
      | cons y yt ⇒ cons (x + y) (ladd xt yt)
      end
  end.

Fixpoint lneg {F : Type} {NF : HasNeg F} (xs : list F) : list F :=
  match xs with
  | nil ⇒ nil
  | cons x t ⇒ cons (- x) (lneg t)
  end.

Fixpoint lscale {F : Type} {MF : HasMult F} (a : F) (xs : list F) : list F :=
  match xs with
  | nil ⇒ nil
  | cons x t ⇒ cons (a × x) (lscale a t)
  end.

#[global] Instance LV_EqRel {F : Type} `{EqRel F} `{Zero F} :
  EqRel (list F) :=
  fun xs ys ⇒ ∀ i : nat, nth0 xs i == nth0 ys i.

#[global] Instance LV_Zero {F : Type} : Zero (list F) := nil.

#[global] Instance LV_Add {F : Type} `{HasAdd F} : HasAdd (list F) := ladd.

#[global] Instance LV_Neg {F : Type} `{HasNeg F} : HasNeg (list F) := lneg.

#[global] Instance LV_Scale {F : Type} `{HasMult F} : HasScale F (list F) :=
  lscale.

#[global] Instance LV_Equivalence {F : Type} {e : EqRel F} {z : Zero F}
    {E : Equivalence e} : Equivalence (@LV_EqRel F e z).
Proof.
  apply Build_Equivalence.
  - intros xs i. reflexivity.
  - intros xs ys Hxy i. symmetry. apply Hxy.
  - intros xs ys zs Hxy Hyz i. rewrite (Hxy i). apply Hyz.
Qed.


Section ListVSMonoid.
  Context {F : Type} `{Monoid F}.

  Lemma nth0_add : ∀ (xs ys : list F) (i : nat),
    nth0 (xs + ys) i == nth0 xs i + nth0 ys i.
  Proof.
    intros xs.
    induction xs as [| x xt IH]; intros ys i.
    - symmetry. apply mon_idl.
    - destruct ys as [| y yt].
      + symmetry. apply mon_idr.
      + destruct i as [| j].
        × reflexivity.
        × exact (IH yt j).
  Qed.

  Lemma lv_assoc : ∀ xs ys zs : list F,
    (xs + ys) + zs == xs + (ys + zs).
  Proof.
    intros xs ys zs i.
    rewrite nth0_add, nth0_add, nth0_add, nth0_add.
    apply mon_assoc.
  Qed.

  Lemma lv_idl : ∀ xs : list F, 0 + xs == xs.
  Proof.
    intros xs i.
    rewrite nth0_add.
    apply mon_idl.
  Qed.

  Lemma lv_idr : ∀ xs : list F, xs + 0 == xs.
  Proof.
    intros xs i.
    rewrite nth0_add.
    apply mon_idr.
  Qed.

  Lemma lv_add_proper : Proper (equ ==> equ ==> equ) (@add (list F) _).
  Proof.
    intros xs xs' Hx ys ys' Hy i.
    rewrite nth0_add, nth0_add.
    apply mon_add_proper; [apply Hx | apply Hy].
  Qed.

  #[global] Instance LV_Monoid : Monoid (list F) := {|
    mon_equ := LV_EqRel;
    mon_equiv := LV_Equivalence;
    mon_zero := LV_Zero;
    mon_add := LV_Add;
    mon_assoc := lv_assoc;
    mon_idl := lv_idl;
    mon_idr := lv_idr;
    mon_add_proper := lv_add_proper
  |}.

End ListVSMonoid.


Section ListVSGroup.
  Context {F : Type} `{Group F}.

  Lemma nth0_neg : ∀ (xs : list F) (i : nat),
    nth0 (- xs) i == - nth0 xs i.
  Proof.
    intros xs.
    induction xs as [| x xt IH]; intros i.
    - symmetry. apply neg_zero.
    - destruct i as [| j].
      + reflexivity.
      + exact (IH j).
  Qed.

  Lemma lv_invl : ∀ xs : list F, (- xs) + xs == 0.
  Proof.
    intros xs i.
    rewrite nth0_add, nth0_neg.
    apply grp_invl.
  Qed.

  Lemma lv_invr : ∀ xs : list F, xs + (- xs) == 0.
  Proof.
    intros xs i.
    rewrite nth0_add, nth0_neg.
    apply grp_invr.
  Qed.

  Lemma lv_neg_proper : Proper (equ ==> equ) (@neg (list F) _).
  Proof.
    intros xs xs' Hx i.
    rewrite nth0_neg, nth0_neg.
    apply grp_neg_proper, Hx.
  Qed.

  #[global] Instance LV_Group : Group (list F) := {|
    grp_monoid := LV_Monoid;
    grp_neg := LV_Neg;
    grp_invl := lv_invl;
    grp_invr := lv_invr;
    grp_neg_proper := lv_neg_proper
  |}.

End ListVSGroup.


Section ListVSAbGroup.
  Context {F : Type} `{AbGroup F}.

  Lemma lv_comm : ∀ xs ys : list F, xs + ys == ys + xs.
  Proof.
    intros xs ys i.
    rewrite nth0_add, nth0_add.
    apply ab_comm.
  Qed.

  #[global] Instance LV_AbGroup : AbGroup (list F) := {|
    ab_group := LV_Group;
    ab_comm := lv_comm
  |}.

End ListVSAbGroup.


Section ListVSVectorSpace.
  Context {F : Type} `{Field F}.

  Lemma nth0_scale : ∀ (a : F) (xs : list F) (i : nat),
    nth0 (a *: xs) i == a × nth0 xs i.
  Proof.
    intros a xs.
    induction xs as [| x xt IH]; intros i.
    - symmetry. apply mul_0_r.
    - destruct i as [| j].
      + reflexivity.
      + exact (IH j).
  Qed.

  Lemma lv_sc_proper : Proper (equ ==> equ ==> equ) (@scale F (list F) _).
  Proof.
    intros a a' Ha xs xs' Hx i.
    rewrite nth0_scale, nth0_scale.
    apply ring_mul_proper; [exact Ha | apply Hx].
  Qed.

  Lemma lv_scale_add : ∀ (a : F) (u v : list F),
    a *: (u + v) == a *: u + a *: v.
  Proof.
    intros a u v i.
    rewrite nth0_scale, nth0_add, nth0_add, nth0_scale, nth0_scale.
    apply ring_distl.
  Qed.

  Lemma lv_add_scale : ∀ (a b : F) (u : list F),
    (a + b) *: u == a *: u + b *: u.
  Proof.
    intros a b u i.
    rewrite nth0_scale, nth0_add, nth0_scale, nth0_scale.
    apply ring_distr.
  Qed.

  Lemma lv_scale_scale : ∀ (a b : F) (u : list F),
    (a × b) *: u == a *: (b *: u).
  Proof.
    intros a b u i.
    rewrite nth0_scale, nth0_scale, nth0_scale.
    apply ring_mul_assoc.
  Qed.

  Lemma lv_scale_1 : ∀ u : list F, 1 *: u == u.
  Proof.
    intros u i.
    rewrite nth0_scale.
    apply ring_mul_1l.
  Qed.

  #[global] Instance LV_ScaleLaws : ScaleLaws F (list F) :=
    @Build_ScaleLaws F (list F) _ LV_AbGroup LV_Scale
      lv_sc_proper lv_scale_add lv_add_scale lv_scale_scale lv_scale_1.

  #[global] Instance LV_VectorSpace : VectorSpace F (list F) :=
    @Build_VectorSpace F (list F) _ LV_AbGroup LV_Scale LV_ScaleLaws.

End ListVSVectorSpace.
