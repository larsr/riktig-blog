Library Riktig.Interface.DecTotalOrder


From Coq Require Import RelationClasses.
From Riktig Require Import Interface.Decidable.

Class DecTotalOrder (A : Type) := {
  olt : A → A → Prop;
  ole : A → A → Prop;
  #[global] olt_dec :: ∀ x y, Decidable (olt x y);
  #[global] olt_strict :: StrictOrder olt;
  olt_total : ∀ x y, olt x y ∨ x = y ∨ olt y x;
  ole_nlt : ∀ x y, ole x y ↔ ¬ olt y x;
}.

#[global] Hint Mode DecTotalOrder ! : typeclass_instances.

Definition ogt {A} `{DecTotalOrder A} (x y : A) : Prop := olt y x.
Definition oge {A} `{DecTotalOrder A} (x y : A) : Prop := ole y x.

Declare Scope order_scope.
Infix "<" := olt : order_scope.
Infix "≤" := ole : order_scope.
Infix ">" := ogt : order_scope.
Infix "≥" := oge : order_scope.
