Library Riktig.Programs.Gcd


Require Import Arith Lia.
Require Import Coq.Arith.Wf_nat.
Require Import Riktig.Interface.Decidable Riktig.Impl.Decidable.
Require Import Riktig.Lib.Fun.

Definition gcd : nat → nat → nat.
 apply (uncurry ∘ (Fix (well_founded_ltof (nat × nat) snd) (fun _ ⇒ nat))).
  refine
  ( fun p gcd ⇒
      match p as x return p=x → nat with
      | (a, b) ⇒
          fun e ⇒
          if dec (b = 0)
            then a
            else gcd (b, a mod b) _
      end eq_refl );
  (subst p; unfold ltof; simpl; abstract (now apply Nat.mod_upper_bound)).
 Defined.

Lemma gcd_step: ∀ a b : nat,
    gcd a b = if dec (b = 0) then a else gcd b (a mod b).
  intros; unfold gcd at 1, compose, uncurry.
  rewrite Fix_eq.
  - destruct dec; reflexivity.
  - intros [x y] f g H.
    destruct dec; [reflexivity | apply H].
 Qed.

Check Logic.eq_refl : gcd 48 18 = 6.
Check Logic.eq_refl : gcd 17 5 = 1.
Check Logic.eq_refl : gcd 0 5 = 5.
Check Logic.eq_refl : gcd 5 0 = 5.
Check Logic.eq_refl : gcd 270 192 = 6.
