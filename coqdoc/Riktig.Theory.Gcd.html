Library Riktig.Theory.Gcd


Require Import Arith Lia.
From Riktig Require Import Interface.Decidable Impl.Decidable Lib.MeasureInd.
Require Import Riktig.Programs.Gcd.

Section GcdProps.

  Lemma gcd_eq_stdlib : ∀ a b, gcd a b = Nat.gcd a b.
  Proof.
    apply (measure_ind2 (fun (_ b : nat) ⇒ b)
             (fun a b ⇒ gcd a b = Nat.gcd a b)).
    intros a b IH.
    rewrite gcd_step. destruct (dec (b = 0)) as [->|Hb].
    - now rewrite Nat.gcd_0_r.
    - rewrite (IH b (a mod b)) by (apply Nat.mod_upper_bound; exact Hb).
      rewrite (Nat.gcd_comm b (a mod b)), Nat.Lcm0.gcd_mod.
      apply Nat.gcd_comm.
  Qed.


  Lemma gcd_divides_l : ∀ a b, Nat.divide (gcd a b) a.
  Proof. intros a b. rewrite gcd_eq_stdlib. apply Nat.gcd_divide_l. Qed.

  Lemma gcd_divides_r : ∀ a b, Nat.divide (gcd a b) b.
  Proof. intros a b. rewrite gcd_eq_stdlib. apply Nat.gcd_divide_r. Qed.

  Lemma gcd_greatest : ∀ a b c,
      Nat.divide c a → Nat.divide c b → Nat.divide c (gcd a b).
  Proof.
    intros a b c Ha Hb. rewrite gcd_eq_stdlib.
    now apply Nat.gcd_greatest.
  Qed.

End GcdProps.
