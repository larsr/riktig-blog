Definition Blade : Type := bool × bool × bool.

Definition bxor (A B : Blade) : Blade :=
  match A, B with
  | (a1,a2,a3), (b1,b2,b3) ⇒ (xorb a1 b1, xorb a2 b2, xorb a3 b3)
  end.

Definition bpar (A B : Blade) : bool :=
  match A, B with
  | (_,a2,a3), (b1,b2,_) ⇒ xorb (andb a2 b1) (andb a3 (xorb b1 b2))
  end.

Lemma bxor_invol : ∀ A B : Blade, bxor A (bxor A B) = B.
Proof.
  intros [[a1 a2] a3] [[b1 b2] b3]. simpl.
  destruct a1, a2, a3, b1, b2, b3; reflexivity.
Qed.

Lemma bxor_assoc : ∀ A B C : Blade, bxor (bxor A B) C = bxor A (bxor B C).
Proof.
  intros [[a1 a2] a3] [[b1 b2] b3] [[c1 c2] c3]. simpl.
  destruct a1, a2, a3, b1, b2, b3, c1, c2, c3; reflexivity.
Qed.

Lemma bxor_comm : ∀ A B : Blade, bxor A B = bxor B A.
Proof.
  intros [[a1 a2] a3] [[b1 b2] b3]. simpl.
  destruct a1, a2, a3, b1, b2, b3; reflexivity.
Qed.

Lemma bpar_cocycle : ∀ A B D0 : Blade,
  xorb (bpar (bxor A B) (bxor (bxor A B) D0)) (bpar A B)
  = xorb (bpar A (bxor A D0)) (bpar B (bxor (bxor A B) D0)).
Proof.
  intros [[a1 a2] a3] [[b1 b2] b3] [[d1 d2] d3]. simpl.
  destruct a1, a2, a3, b1, b2, b3, d1, d2, d3; reflexivity.
Qed.


Definition fin_to_blade (i : Fin.t 8) : Blade :=
  Fin.caseS' i (fun _ ⇒ Blade) (false,false,false)
    (fun i1 ⇒ Fin.caseS' i1 (fun _ ⇒ Blade) (true,false,false)
      (fun i2 ⇒ Fin.caseS' i2 (fun _ ⇒ Blade) (false,true,false)
        (fun i3 ⇒ Fin.caseS' i3 (fun _ ⇒ Blade) (false,false,true)
          (fun i4 ⇒ Fin.caseS' i4 (fun _ ⇒ Blade) (true,true,false)
            (fun i5 ⇒ Fin.caseS' i5 (fun _ ⇒ Blade) (true,false,true)
              (fun i6 ⇒ Fin.caseS' i6 (fun _ ⇒ Blade) (false,true,true)
                (fun i7 ⇒ Fin.caseS' i7 (fun _ ⇒ Blade) (true,true,true)
                  (fun i8 ⇒ Fin.case0 (fun _ ⇒ Blade) i8)))))))).

Definition blade_to_fin (A : Blade) : Fin.t 8 :=
  match A with
  | (false,false,false) ⇒ Fin.F1
  | (true,false,false) ⇒ Fin.FS Fin.F1
  | (false,true,false) ⇒ Fin.FS (Fin.FS Fin.F1)
  | (false,false,true) ⇒ Fin.FS (Fin.FS (Fin.FS Fin.F1))
  | (true,true,false) ⇒ Fin.FS (Fin.FS (Fin.FS (Fin.FS Fin.F1)))
  | (true,false,true) ⇒ Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS Fin.F1))))
  | (false,true,true) ⇒ Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS Fin.F1)))))
  | (true,true,true) ⇒ Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS Fin.F1))))))
  end.

Lemma blade_to_fin_to_blade : ∀ A : Blade, fin_to_blade (blade_to_fin A) = A.
Proof.
  intros [[a1 a2] a3]. destruct a1, a2, a3; reflexivity.
Qed.

Lemma fin_to_blade_to_fin : ∀ i : Fin.t 8, blade_to_fin (fin_to_blade i) = i.
Proof.
  intros i.
  apply (Fin.caseS' i (fun i0 ⇒ blade_to_fin (fin_to_blade i0) = i0)).
  - reflexivity.
  - intros i1. apply (Fin.caseS' i1 (fun i0 ⇒ blade_to_fin (fin_to_blade (Fin.FS i0)) = Fin.FS i0)).
    + reflexivity.
    + intros i2. apply (Fin.caseS' i2 (fun i0 ⇒ blade_to_fin (fin_to_blade (Fin.FS (Fin.FS i0))) = Fin.FS (Fin.FS i0))).
      × reflexivity.
      × intros i3. apply (Fin.caseS' i3 (fun i0 ⇒ blade_to_fin (fin_to_blade (Fin.FS (Fin.FS (Fin.FS i0)))) = Fin.FS (Fin.FS (Fin.FS i0)))).
        -- reflexivity.
        -- intros i4. apply (Fin.caseS' i4 (fun i0 ⇒ blade_to_fin (fin_to_blade (Fin.FS (Fin.FS (Fin.FS (Fin.FS i0))))) = Fin.FS (Fin.FS (Fin.FS (Fin.FS i0))))).
           ++ reflexivity.
           ++ intros i5. apply (Fin.caseS' i5 (fun i0 ⇒ blade_to_fin (fin_to_blade (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i0)))))) = Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i0)))))).
              ** reflexivity.
              ** intros i6. apply (Fin.caseS' i6 (fun i0 ⇒ blade_to_fin (fin_to_blade (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i0))))))) = Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i0))))))).
                 --- reflexivity.
                 --- intros i7. apply (Fin.caseS' i7 (fun i0 ⇒ blade_to_fin (fin_to_blade (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i0)))))))) = Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i0)))))))).
                     +++ reflexivity.
                     +++ intros i8. exact (Fin.case0 (fun i0 ⇒ blade_to_fin (fin_to_blade (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i0))))))))) = Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i0)))))))) i8).
Qed.

  #[global] Instance C3_Equivalence : Equivalence C3_EqRel.
  Proof.
    apply Build_Equivalence.
    - intros x. repeat split; reflexivity.
    - intros x y H. repeat split; symmetry; apply H.
    - intros x y v Hxy Hyv. repeat split.
      + rewrite (proj1 Hxy). exact (proj1 Hyv).
      + rewrite (proj1 (proj2 Hxy)). exact (proj1 (proj2 Hyv)).
      + rewrite (proj1 (proj2 (proj2 Hxy))). exact (proj1 (proj2 (proj2 Hyv))).
      + rewrite (proj1 (proj2 (proj2 (proj2 Hxy)))).
        exact (proj1 (proj2 (proj2 (proj2 Hyv)))).
      + rewrite (proj1 (proj2 (proj2 (proj2 (proj2 Hxy))))).
        exact (proj1 (proj2 (proj2 (proj2 (proj2 Hyv))))).
      + rewrite (proj1 (proj2 (proj2 (proj2 (proj2 (proj2 Hxy)))))).
        exact (proj1 (proj2 (proj2 (proj2 (proj2 (proj2 Hyv)))))).
      + rewrite (proj1 (proj2 (proj2 (proj2 (proj2 (proj2 (proj2 Hxy))))))).
        exact (proj1 (proj2 (proj2 (proj2 (proj2 (proj2 (proj2 Hyv))))))).
      + rewrite (proj2 (proj2 (proj2 (proj2 (proj2 (proj2 (proj2 Hxy))))))).
        exact (proj2 (proj2 (proj2 (proj2 (proj2 (proj2 (proj2 Hyv))))))).
  Qed.

  Definition c3_proj (x : C3 F) (A : Blade) : F :=
    match A with
    | (false,false,false) ⇒ c3_s x
    | (true,false,false) ⇒ c3_1 x
    | (false,true,false) ⇒ c3_2 x
    | (false,false,true) ⇒ c3_3 x
    | (true,true,false) ⇒ c3_12 x
    | (true,false,true) ⇒ c3_13 x
    | (false,true,true) ⇒ c3_23 x
    | (true,true,true) ⇒ c3_123 x
    end.

  Lemma c3_proj_add : ∀ (x y : C3 F) (A : Blade),
    c3_proj (x + y) A == c3_proj x A + c3_proj y A.
  Proof.
    intros x y [[a1 a2] a3]. destruct a1, a2, a3; reflexivity.
  Qed.

  Lemma c3_proj_neg : ∀ (x : C3 F) (A : Blade),
    c3_proj (- x) A == - c3_proj x A.
  Proof.
    intros x [[a1 a2] a3]. destruct a1, a2, a3; reflexivity.
  Qed.

  Lemma c3_proj_zero : ∀ A : Blade, c3_proj (0 : C3 F) A == 0.
  Proof.
    intros [[a1 a2] a3]. destruct a1, a2, a3; reflexivity.
  Qed.

  Lemma c3_proj_one_scalar : c3_proj (1 : C3 F) (false,false,false) == 1.
  Proof. reflexivity. Qed.

  Lemma c3_proj_one_other : ∀ A : Blade,
    A ≠ (false,false,false) → c3_proj (1 : C3 F) A == 0.
  Proof.
    intros [[a1 a2] a3] Hne. destruct a1, a2, a3; try reflexivity.
    destruct (Hne eq_refl).
  Qed.

  Lemma c3_proj_proper : ∀ (x y : C3 F), x == y →
    ∀ A : Blade, c3_proj x A == c3_proj y A.
  Proof.
    intros x y Hxy [[a1 a2] a3].
    destruct Hxy as [Hs [H1 [H2 [H3 [H12 [H13 [H23 H123]]]]]]].
    destruct a1, a2, a3; simpl; assumption.
  Qed.

  #[global] Instance c3_proj_proper_inst : Proper (equ ==> eq ==> equ) c3_proj.
  Proof.
    intros x y Hxy A B HAB. subst B. exact (c3_proj_proper x y Hxy A).
  Qed.

  Lemma c3_eq_intro : ∀ x y : C3 F,
    (∀ A : Blade, c3_proj x A == c3_proj y A) → x == y.
  Proof.
    intros x y H. repeat split.
    - exact (H (false,false,false)).
    - exact (H (true,false,false)).
    - exact (H (false,true,false)).
    - exact (H (false,false,true)).
    - exact (H (true,true,false)).
    - exact (H (true,false,true)).
    - exact (H (false,true,true)).
    - exact (H (true,true,true)).
  Qed.

End C3Ops.

Section BSum.
  Context {F : Type} `{FF : Field F}.

  Definition bsum (f : Blade → F) : F := bigopF (fun i ⇒ f (fin_to_blade i)).

  Lemma bsum_unfold : ∀ f : Blade → F,
    bsum f == f (false,false,false)
      + (f (true,false,false)
      + (f (false,true,false)
      + (f (false,false,true)
      + (f (true,true,false)
      + (f (true,false,true)
      + (f (false,true,true)
      + f (true,true,true))))))).
  Proof.
    intros f. unfold bsum.
    rewrite (bigopF_S 7 (fun i ⇒ f (fin_to_blade i))).
    rewrite (bigopF_S 6 (fun i ⇒ f (fin_to_blade (Fin.FS i)))).
    rewrite (bigopF_S 5 (fun i ⇒ f (fin_to_blade (Fin.FS (Fin.FS i))))).
    rewrite (bigopF_S 4 (fun i ⇒ f (fin_to_blade (Fin.FS (Fin.FS (Fin.FS i)))))).
    rewrite (bigopF_S 3 (fun i ⇒ f (fin_to_blade (Fin.FS (Fin.FS (Fin.FS (Fin.FS i))))))).
    rewrite (bigopF_S 2 (fun i ⇒ f (fin_to_blade (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i)))))))).
    rewrite (bigopF_S 1 (fun i ⇒ f (fin_to_blade (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i))))))))).
    rewrite (bigopF_S 0 (fun i ⇒ f (fin_to_blade (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i)))))))))).
    rewrite (bigopF_0 (fun i ⇒ f (fin_to_blade (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS (Fin.FS i)))))))))) ).
    simpl fin_to_blade.
    rewrite (idr (f (true,true,true) : F)).
    reflexivity.
  Qed.

  Lemma bsum_ext : ∀ f g : Blade → F,
    (∀ b : Blade, f b == g b) → bsum f == bsum g.
  Proof.
    intros f g Hfg. unfold bsum.
    exact (bigopF_ext 8 (fun i ⇒ f (fin_to_blade i)) (fun i ⇒ g (fin_to_blade i))
             (fun i ⇒ Hfg (fin_to_blade i))).
  Qed.

  Lemma bsum_add : ∀ f g : Blade → F,
    bsum (fun b ⇒ f b + g b) == bsum f + bsum g.
  Proof.
    intros f g. unfold bsum.
    exact (bigopF_add 8 (fun i ⇒ f (fin_to_blade i)) (fun i ⇒ g (fin_to_blade i))).
  Qed.

  Lemma bsum_mul_l : ∀ (a : F) (f : Blade → F),
    bsum (fun b ⇒ a × f b) == a × bsum f.
  Proof.
    intros a f. unfold bsum.
    exact (bigopF_mul_l 8 a (fun i ⇒ f (fin_to_blade i))).
  Qed.

  Lemma bsum_mul_r : ∀ (a : F) (f : Blade → F),
    bsum (fun b ⇒ f b × a) == bsum f × a.
  Proof.
    intros a f. unfold bsum.
    exact (bigopF_mul_r 8 a (fun i ⇒ f (fin_to_blade i))).
  Qed.

  Lemma bsum_swap : ∀ f : Blade → Blade → F,
    bsum (fun A ⇒ bsum (fun B ⇒ f A B)) == bsum (fun B ⇒ bsum (fun A ⇒ f A B)).
  Proof.
    intros f. unfold bsum.
    exact (bigopF_swap 8 8 (fun i j ⇒ f (fin_to_blade i) (fin_to_blade j))).
  Qed.

  Lemma bsum_reindex_bxor : ∀ (f : Blade → F) (A : Blade),
    bsum f == bsum (fun D ⇒ f (bxor A D)).
  Proof.
    intros f A. unfold bsum.
    assert (Hinv : ∀ i : Fin.t 8,
        blade_to_fin (bxor A (fin_to_blade (blade_to_fin (bxor A (fin_to_blade i)))))
        = i).
    { intros i.
      rewrite (blade_to_fin_to_blade (bxor A (fin_to_blade i))).
      rewrite (bxor_invol A (fin_to_blade i)).
      exact (fin_to_blade_to_fin i). }
    rewrite (bigopF_reindex 8 (fun i ⇒ f (fin_to_blade i))
               (fun i ⇒ blade_to_fin (bxor A (fin_to_blade i)))
               (fun i ⇒ blade_to_fin (bxor A (fin_to_blade i)))
               Hinv Hinv).
    apply bigopF_ext. intros i.
    rewrite (blade_to_fin_to_blade (bxor A (fin_to_blade i))).
    reflexivity.
  Qed.

  Lemma bsum_single_scalar : ∀ f : Blade → F,
    (∀ b : Blade, b ≠ (false,false,false) → f b == 0) →
    bsum f == f (false,false,false).
  Proof.
    intros f Hz.
    rewrite (bsum_unfold f).
    rewrite (Hz (true,false,false) ltac:(discriminate)),
      (Hz (false,true,false) ltac:(discriminate)),
      (Hz (false,false,true) ltac:(discriminate)),
      (Hz (true,true,false) ltac:(discriminate)),
      (Hz (true,false,true) ltac:(discriminate)),
      (Hz (false,true,true) ltac:(discriminate)),
      (Hz (true,true,true) ltac:(discriminate)).
    autorewrite with grp.
    reflexivity.
  Qed.

  Lemma bsum_sandwich_mul : ∀ (a b : F) (f : Blade → F),
    a × bsum f × b == bsum (fun A ⇒ a × f A × b).
  Proof.
    intros a b f.
    rewrite <- (bsum_mul_l a f).
    rewrite <- (bsum_mul_r b (fun A ⇒ a × f A)).
    reflexivity.
  Qed.

End BSum.

Section C3Mul.
  Context {F : Type} `{FF : Field F}.

  Definition bsign (A B : Blade) : F := if bpar A B then - (1) else 1.

  Lemma sign_xor : ∀ p q : bool,
    (if xorb p q then - (1) else 1 : F)
    == (if p then - (1) else 1 : F) × (if q then - (1) else 1 : F).
  Proof.
    intros [] []; simpl.
    - symmetry. rewrite (mul_neg_neg 1 1). exact (ring_mul_1l 1).
    - symmetry. exact (ring_mul_1r (- (1))).
    - symmetry. exact (ring_mul_1l (- (1))).
    - symmetry. exact (ring_mul_1l (1:F)).
  Qed.

  Lemma bsign_cocycle : ∀ A B D0 : Blade,
    bsign (bxor A B) (bxor (bxor A B) D0) × bsign A B
    == bsign A (bxor A D0) × bsign B (bxor (bxor A B) D0).
  Proof.
    intros A B D0. unfold bsign.
    rewrite <- (sign_xor (bpar (bxor A B) (bxor (bxor A B) D0)) (bpar A B)).
    rewrite (bpar_cocycle A B D0).
    exact (sign_xor (bpar A (bxor A D0)) (bpar B (bxor (bxor A B) D0))).
  Qed.

  Definition c3_mul_at (D : Blade) (x y : C3 F) : F :=
    bsum (fun A ⇒ bsign A (bxor A D) × c3_proj x A × c3_proj y (bxor A D)).

  #[global] Instance C3_Mult : HasMult (C3 F) :=
    fun x y ⇒ mkC3
      (c3_mul_at (false,false,false) x y)
      (c3_mul_at (true,false,false) x y)
      (c3_mul_at (false,true,false) x y)
      (c3_mul_at (false,false,true) x y)
      (c3_mul_at (true,true,false) x y)
      (c3_mul_at (true,false,true) x y)
      (c3_mul_at (false,true,true) x y)
      (c3_mul_at (true,true,true) x y).

  Lemma c3_proj_mul : ∀ (D : Blade) (x y : C3 F),
    c3_proj (x × y) D == c3_mul_at D x y.
  Proof.
    intros [[d1 d2] d3] x y. destruct d1, d2, d3; reflexivity.
  Qed.

  Lemma five_factor_regroup : ∀ S1 S2 S3 S4 cx cy cz : F,
    S1 × S2 == S3 × S4 →
    S1 × (S2 × cx × cy) × cz == S3 × cx × (S4 × cy × cz).
  Proof.
    intros S1 S2 S3 S4 cx cy cz H.
    rewrite <- (ring_mul_assoc S1 (S2×cx) cy).
    rewrite <- (ring_mul_assoc S1 S2 cx).
    rewrite (ring_mul_assoc (S1×S2) cx cy).
    rewrite H.
    rewrite <- (ring_mul_assoc (S3×S4) cx cy).
    rewrite (ring_mul_assoc S3 S4 cx).
    rewrite (field_mul_comm S4 cx).
    rewrite <- (ring_mul_assoc S3 cx S4).
    rewrite (ring_mul_assoc (S3×cx) S4 cy).
    rewrite (ring_mul_assoc (S3×cx) (S4×cy) cz).
    reflexivity.
  Qed.

  Lemma c3_mul_at_assoc_term : ∀ (D0 A B : Blade) (x y z : C3 F),
    bsign (bxor A B) (bxor (bxor A B) D0) × (bsign A B × c3_proj x A × c3_proj y B)
      × c3_proj z (bxor (bxor A B) D0)
    == bsign A (bxor A D0) × c3_proj x A
       × (bsign B (bxor B (bxor A D0)) × c3_proj y B × c3_proj z (bxor B (bxor A D0))).
  Proof.
    intros D0 A B x y z.
    rewrite (bxor_lemma1 A B D0).
    apply (five_factor_regroup (bsign (bxor A B) (bxor B (bxor A D0))) (bsign A B)
             (bsign A (bxor A D0)) (bsign B (bxor B (bxor A D0)))
             (c3_proj x A) (c3_proj y B) (c3_proj z (bxor B (bxor A D0)))).
    rewrite <- (bxor_lemma1 A B D0).
    exact (bsign_cocycle A B D0).
  Qed.

                 (fun A ⇒ bsign A (bxor A D) × c3_proj x A × c3_proj y (bxor A D))).
  Qed.

  Theorem c3_reverse_anti : ∀ x y : C3 F,
    c3_reverse (x × y) == c3_reverse y × c3_reverse x.
  Proof.
    intros x y. apply c3_eq_intro. intros D.
    rewrite (c3_proj_reverse (x × y) D), (c3_proj_mul D x y),
      (c3_proj_mul D (c3_reverse y) (c3_reverse x)).
    symmetry. exact (c3_mul_at_reverse D x y).
  Qed.

  Definition c3_rotate (r x : C3 F) : C3 F := r × x × c3_reverse r.

  Lemma bsum_neg : ∀ f : Blade → F, bsum (fun A ⇒ - f A) == - bsum f.
  Proof.
    intros f.
    transitivity (bsum (fun A ⇒ - (1) × f A)).
    - apply bsum_ext. intros A. symmetry. exact (mul_neg1_l (f A)).
    - rewrite (bsum_mul_l (- (1)) f). exact (mul_neg1_l (bsum f)).
  Qed.

  Lemma c3_neg_mul_l : ∀ a b : C3 F, (- a) × b == - (a × b).
  Proof.
    intros a b. apply c3_eq_intro. intros A.
    rewrite (c3_proj_mul A (- a) b), (c3_proj_neg (a × b) A), (c3_proj_mul A a b).
    unfold c3_mul_at.
    rewrite <- (bsum_neg (fun D ⇒ bsign D (bxor D A) × c3_proj a D × c3_proj b (bxor D A))).
    apply bsum_ext. intros D.
    rewrite (c3_proj_neg a D).
    rewrite (mul_neg_r (bsign D (bxor D A)) (c3_proj a D)).
    exact (mul_neg_l (bsign D (bxor D A) × c3_proj a D) (c3_proj b (bxor D A))).
  Qed.

  Lemma c3_neg_mul_r : ∀ a b : C3 F, a × (- b) == - (a × b).
  Proof.
    intros a b. apply c3_eq_intro. intros A.
    rewrite (c3_proj_mul A a (- b)), (c3_proj_neg (a × b) A), (c3_proj_mul A a b).
    unfold c3_mul_at.
    rewrite <- (bsum_neg (fun D ⇒ bsign D (bxor D A) × c3_proj a D × c3_proj b (bxor D A))).
    apply bsum_ext. intros D.
    rewrite (c3_proj_neg b (bxor D A)).
    exact (mul_neg_r (bsign D (bxor D A) × c3_proj a D) (c3_proj b (bxor D A))).
  Qed.

  Lemma c3_neg_involutive : ∀ a : C3 F, - (- a) == a.
  Proof.
    intros a. apply c3_eq_intro. intros A.
    rewrite (c3_proj_neg (- a) A), (c3_proj_neg a A).
    exact (neg_involutive (c3_proj a A)).
  Qed.

  Lemma c3_rotate_trivector_vanish : ∀ (r : C3 F) (v : Fin.t 3 → F),
    c3_even r → c3_123 (c3_rotate r (c3_vec v)) == 0.
  Proof.
    intros r v Hr.
    change (c3_123 (c3_rotate r (c3_vec v)))
      with (c3_proj (c3_rotate r (c3_vec v)) (true,true,true)).
    unfold c3_rotate.
    rewrite (c3_even_shape r Hr).
    unfold c3_vec.
    exact (c3_rotate_trivector_vanish_explicit (c3_s r) (c3_12 r) (c3_13 r) (c3_23 r)
             (v Fin.F1) (v (Fin.FS Fin.F1)) (v (Fin.FS (Fin.FS Fin.F1)))).
  Qed.

  Theorem c3_rotate_vector : ∀ (r : C3 F) (v : Fin.t 3 → F),
    c3_even r → ∃ w : Fin.t 3 → F, c3_rotate r (c3_vec v) == c3_vec w.
  Proof.
    intros r v Hr.
    assert (Hcodd : c3_codd (c3_rotate r (c3_vec v))).
    { unfold c3_rotate.
      apply c3_codd_even_mul_closed.
      - apply c3_even_codd_mul_closed.
        + exact Hr.
        + exact (c3_vec_codd v).
      - exact (c3_reverse_even r Hr). }
    destruct Hcodd as [Hs [H12 [H13 H23]]].
    pose (w := fun i : Fin.t 3 ⇒
      match i with
      | Fin.F1 ⇒ c3_1 (c3_rotate r (c3_vec v))
      | Fin.FS Fin.F1 ⇒ c3_2 (c3_rotate r (c3_vec v))
      | Fin.FS (Fin.FS _) ⇒ c3_3 (c3_rotate r (c3_vec v))
      end).
    ∃ w. repeat split; unfold w.
    - exact Hs.
    - reflexivity.
    - reflexivity.
    - reflexivity.
    - exact H12.
    - exact H13.
    - exact H23.
    - exact (c3_rotate_trivector_vanish r v Hr).
  Qed.

  Theorem c3_rotate_neg : ∀ r x : C3 F, c3_rotate (- r) x == c3_rotate r x.
  Proof.
    intros r x. unfold c3_rotate.
    assert (Hrev : c3_reverse (- r) == - c3_reverse r).
    { apply c3_eq_intro. intros A.
      rewrite (c3_proj_neg (c3_reverse r) A), (c3_proj_reverse (- r) A),
        (c3_proj_reverse r A), (c3_proj_neg r A).
      exact (mul_neg_r (c3_rsign A) (c3_proj r A)). }
    rewrite Hrev.
    rewrite (c3_neg_mul_l r x).
    rewrite (c3_neg_mul_r (- (r × x)) (c3_reverse r)).
    rewrite (c3_neg_mul_l (r × x) (c3_reverse r)).
    exact (c3_neg_involutive (r × x × c3_reverse r)).
  Qed.
