Lemma Perm_S_eta : ∀ (n : nat) (sigma : Perm (S n)),
  ∃ (c : Fin.t (S n)) (p : Perm n), sigma = Perm_S c p.
Proof.
  intros n sigma.
  set (P := fun (n0 : nat) (s : Perm n0) ⇒
    match n0 return Perm n0 → Prop with
    | O ⇒ fun _ ⇒ True
    | S m ⇒ fun s0 ⇒ ∃ c p, s0 = Perm_S c p
    end s).
  change (P (S n) sigma).
  destruct sigma as [| m c p].
  - exact I.
  - exact (ex_intro _ c (ex_intro _ p eq_refl)).
Qed.

Section PermPostcompose.

  Fixpoint perm_postcompose {n : nat} (sigma : Perm n) {struct sigma}
      : (Fin.t n → Fin.t n) → (Fin.t n → Fin.t n) → Perm n :=
    match sigma in Perm n0
      return (Fin.t n0 → Fin.t n0) → (Fin.t n0 → Fin.t n0) → Perm n0
    with
    | Perm_O ⇒ fun _ _ ⇒ Perm_O
    | @Perm_S m c p ⇒ fun tau tau' ⇒
        Perm_S (tau c) (perm_postcompose p (bstep2 c tau tau') (bstep2' c tau tau'))
    end.

  Lemma perm_postcompose_O : ∀ tau tau' : Fin.t O → Fin.t O,
    perm_postcompose Perm_O tau tau' = Perm_O.
  Proof. intros tau tau'. reflexivity. Qed.

  Lemma perm_postcompose_S : ∀ (m : nat) (c : Fin.t (S m)) (p : Perm m)
      (tau tau' : Fin.t (S m) → Fin.t (S m)),
    perm_postcompose (Perm_S c p) tau tau'
      = Perm_S (tau c) (perm_postcompose p (bstep2 c tau tau') (bstep2' c tau tau')).
  Proof. intros m c p tau tau'. reflexivity. Qed.

  Theorem perm_apply_postcompose : ∀ (n : nat) (sigma : Perm n)
      (tau tau' : Fin.t n → Fin.t n),
    (∀ x, tau' (tau x) = x) → (∀ y, tau (tau' y) = y) →
    ∀ x : Fin.t n, perm_apply (perm_postcompose sigma tau tau') x = tau (perm_apply sigma x).
  Proof.
    intros n sigma; induction sigma as [| m c p IH]; intros tau tau' H1 H2 x.
    - exact (Fin.case0 (fun x0 ⇒ perm_apply (perm_postcompose Perm_O tau tau') x0
               = tau (perm_apply Perm_O x0)) x).
    - revert tau tau' H1 H2.
      apply (Fin.caseS' x (fun x0 ⇒ ∀ tau tau' : Fin.t (S m) → Fin.t (S m),
        (∀ z, tau' (tau z) = z) → (∀ z, tau (tau' z) = z) →
        perm_apply (perm_postcompose (Perm_S c p) tau tau') x0
          = tau (perm_apply (Perm_S c p) x0))).
      + intros tau tau' H1 H2.
        rewrite (perm_postcompose_S m c p tau tau'), (perm_apply_F1 m (tau c)
          (perm_postcompose p (bstep2 c tau tau') (bstep2' c tau tau'))),
          (perm_apply_F1 m c p).
        reflexivity.
      + intros x' tau tau' H1 H2.
        rewrite (perm_postcompose_S m c p tau tau'),
          (perm_apply_FS m (tau c)
             (perm_postcompose p (bstep2 c tau tau') (bstep2' c tau tau')) x'),
          (IH (bstep2 c tau tau') (bstep2' c tau tau')
             (bstep2_bstep2' m c tau tau' H1 H2) (bstep2'_bstep2 m c tau tau' H1 H2) x'),
          (bstep2_spec m c tau tau' H1 H2 (perm_apply p x')),
          (perm_apply_FS m c p x').
        reflexivity.
  Qed.

End PermPostcompose.



Section PermSignPostcompose.

  Theorem perm_sign_postcompose_ext : ∀ (n : nat) (sigma : Perm n)
      (tau1 tau2 tau1' tau2' : Fin.t n → Fin.t n),
    (∀ x, tau1 x = tau2 x) →
    perm_sign (perm_postcompose sigma tau1 tau1') == perm_sign (perm_postcompose sigma tau2 tau2').
  Proof.
    intros n sigma; induction sigma as [| m c p IH]; intros tau1 tau2 tau1' tau2' Heq.
    - reflexivity.
    - rewrite (perm_postcompose_S m c p tau1 tau1'), (perm_postcompose_S m c p tau2 tau2'),
        (perm_sign_S m (tau1 c) (perm_postcompose p (bstep2 c tau1 tau1') (bstep2' c tau1 tau1'))),
        (perm_sign_S m (tau2 c) (perm_postcompose p (bstep2 c tau2 tau2') (bstep2' c tau2 tau2'))),
        (Heq c).
      apply ring_mul_proper.
      + reflexivity.
      + exact (IH (bstep2 c tau1 tau1') (bstep2 c tau2 tau2') (bstep2' c tau1 tau1')
                 (bstep2' c tau2 tau2') (bstep2_ext m c tau1 tau2 tau1' tau2' Heq)).
  Qed.


  Theorem perm_sign_postcompose_id : ∀ (n : nat) (sigma : Perm n)
      (tau tau' : Fin.t n → Fin.t n),
    (∀ x, tau x = x) → (∀ x, tau' x = x) →
    perm_sign (perm_postcompose sigma tau tau') == perm_sign sigma.
  Proof.
    intros n sigma; induction sigma as [| m c p IH]; intros tau tau' Htau Htau'.
    - reflexivity.
    - rewrite (perm_postcompose_S m c p tau tau'),
        (perm_sign_S m (tau c) (perm_postcompose p (bstep2 c tau tau') (bstep2' c tau tau'))),
        (Htau c), (perm_sign_S m c p).
      apply ring_mul_proper.
      + reflexivity.
      + exact (IH (bstep2 c tau tau') (bstep2' c tau tau')
                 (bstep2_id m c tau tau' Htau) (bstep2'_id m c tau tau' Htau Htau')).
  Qed.


  Theorem sign_at_fweaken_FS : ∀ (n : nat) (k : Fin.t n),
    sign_at (Fin.FS k) == - sign_at (fweaken k).
  Proof.
    intros n k; induction k as [n0 | n0 k' IH].
    - rewrite fweaken_F1, (sign_at_FS (S n0) Fin.F1), (sign_at_F1 n0), (sign_at_F1 (S n0)).
      reflexivity.
    - rewrite fweaken_FS, (sign_at_FS (S n0) (Fin.FS k')), (sign_at_FS (S n0) (fweaken k')), IH.
      reflexivity.
  Qed.


  Theorem perm_sign_postcompose_adj : ∀ (n : nat) (k : Fin.t n) (sigma : Perm (S n)),
    perm_sign (perm_postcompose sigma (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k)))
      == - perm_sign sigma.
  Proof.
    intros n; induction n as [| n0 IHn]; intros k sigma.
    - revert sigma. exact (Fin.case0 (fun k0 ⇒ ∀ sigma : Perm 1,
        perm_sign (perm_postcompose sigma (swap_fin (fweaken k0) (Fin.FS k0))
          (swap_fin (fweaken k0) (Fin.FS k0))) == - perm_sign sigma) k).
    - destruct (Perm_S_eta (S n0) sigma) as [c [p Heq]].
      rewrite Heq,
        (perm_postcompose_S (S n0) c p (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k))),
        (perm_sign_S (S n0) (swap_fin (fweaken k) (Fin.FS k) c)
           (perm_postcompose p (bstep2 c (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k)))
              (bstep2' c (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k))))),
        (perm_sign_S (S n0) c p).
      destruct (Fin.eq_dec c (fweaken k)) as [Hceq1 | Hcne1].
      + rewrite Hceq1, (swap_fin_at_i _ (fweaken k) (Fin.FS k)),
          (perm_sign_postcompose_ext (S n0) p
             (bstep2 (fweaken k) (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k)))
             (fun x ⇒ x)
             (bstep2' (fweaken k) (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k)))
             (fun x ⇒ x)
             (bstep2_swap_adjacent_left (S n0) k)),
          (perm_sign_postcompose_id (S n0) p (fun x ⇒ x) (fun x ⇒ x)
             (fun x ⇒ eq_refl) (fun x ⇒ eq_refl)),
          (sign_at_fweaken_FS (S n0) k).
        exact (mul_neg_l (sign_at (fweaken k)) (perm_sign p)).
      + destruct (Fin.eq_dec c (Fin.FS k)) as [Hceq2 | Hcne2].
        × rewrite Hceq2,
            (swap_fin_at_j _ (fweaken k) (Fin.FS k) (fweaken_neq_FS (S n0) k)),
            (perm_sign_postcompose_ext (S n0) p
               (bstep2 (Fin.FS k) (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k)))
               (fun x ⇒ x)
               (bstep2' (Fin.FS k) (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k)))
               (fun x ⇒ x)
               (bstep2_swap_adjacent_right (S n0) k)),
            (perm_sign_postcompose_id (S n0) p (fun x ⇒ x) (fun x ⇒ x)
               (fun x ⇒ eq_refl) (fun x ⇒ eq_refl)).
          symmetry.
          rewrite (sign_at_fweaken_FS (S n0) k), (mul_neg_l (sign_at (fweaken k)) (perm_sign p)),
            (neg_involutive (sign_at (fweaken k) × perm_sign p)).
          reflexivity.
        × destruct (unskip_adjacent_shape n0 k c Hcne1 Hcne2) as [k' [Hu1 Hu2]].
          rewrite (swap_fin_ne _ (fweaken k) (Fin.FS k) c Hcne1 Hcne2),
            (perm_sign_postcompose_ext (S n0) p
               (bstep2 c (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k)))
               (swap_fin (fweaken k') (Fin.FS k'))
               (bstep2' c (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k)))
               (swap_fin (fweaken k') (Fin.FS k'))
               (bstep2_swap_adjacent_ne n0 k c k' Hcne1 Hcne2 Hu1 Hu2)),
            (IHn k' p).
          exact (mul_neg_r (sign_at c) (perm_sign p)).
  Qed.

End PermSignPostcompose.


Section PsumReindex.
  Context {F : Type} {FF : Field F}.

  Theorem psum_reindex : ∀ (n : nat) (h : Perm n → F) (tau tau' : Fin.t n → Fin.t n),
    (∀ x, tau' (tau x) = x) → (∀ y, tau (tau' y) = y) →
    psum h == psum (fun sigma ⇒ h (perm_postcompose sigma tau tau')).
  Proof.
    intros n; induction n as [| m IH]; intros h tau tau' H1 H2.
    - reflexivity.
    - rewrite (psum_S m h).
      transitivity (bigopF (fun i : Fin.t (S m) ⇒ psum (fun p : Perm m ⇒ h (Perm_S (tau i) p)))).
      + exact (bigopF_reindex (M := F) (S m)
                 (fun i : Fin.t (S m) ⇒ psum (fun p : Perm m ⇒ h (Perm_S i p))) tau tau' H1 H2).
      + transitivity (bigopF (fun i : Fin.t (S m) ⇒
          psum (fun p : Perm m ⇒ h (Perm_S (tau i)
            (perm_postcompose p (bstep2 i tau tau') (bstep2' i tau tau')))))).
        × apply bigopF_ext. intros i.
          exact (IH (fun p ⇒ h (Perm_S (tau i) p)) (bstep2 i tau tau') (bstep2' i tau tau')
                   (bstep2_bstep2' m i tau tau' H1 H2) (bstep2'_bstep2 m i tau tau' H1 H2)).
        × symmetry.
          rewrite (psum_S m (fun sigma ⇒ h (perm_postcompose sigma tau tau'))).
          apply bigopF_ext. intros i.
          apply psum_ext. intros p.
          rewrite (perm_postcompose_S m i p tau tau').
          reflexivity.
  Qed.

End PsumReindex.



Section PsumNeg.
  Context {F : Type} {FF : Field F}.

  Theorem psum_neg : ∀ (n : nat) (f : Perm n → F), psum (fun p ⇒ - f p) == - psum f.
  Proof.
    intros n f.
    transitivity (psum (fun p ⇒ (- (1)) × f p)).
    - apply psum_ext. intros p. symmetry. exact (mul_neg1_l (f p)).
    - rewrite (psum_scale n (- (1)) f). exact (mul_neg1_l (psum f)).
  Qed.

End PsumNeg.


Section DetSwapAdjacent.
  Context {F : Type} {FF : Field F}.

  Theorem detM_leibniz_swap_adjacent : ∀ (n : nat) (k : Fin.t n) (A : Matrix (S n) (S n) F),
    detM_leibniz (row_swap A (fweaken k) (Fin.FS k)) == - detM_leibniz A.
  Proof.
    intros n k A.
    unfold detM_leibniz at 1.
    transitivity (psum (fun sigma : Perm (S n) ⇒ perm_sign sigma ×
      prodF (fun c ⇒ A (swap_fin (fweaken k) (Fin.FS k) (perm_apply sigma c)) c))).
    - apply psum_ext. intros sigma.
      apply ring_mul_proper.
      + reflexivity.
      + exact (perm_prod_row_swap (S n) A (fweaken k) (Fin.FS k) sigma).
    - rewrite (psum_reindex (S n)
        (fun sigma : Perm (S n) ⇒ perm_sign sigma ×
          prodF (fun c ⇒ A (swap_fin (fweaken k) (Fin.FS k) (perm_apply sigma c)) c))
        (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k))
        (swap_fin_invol (S n) (fweaken k) (Fin.FS k)) (swap_fin_invol (S n) (fweaken k) (Fin.FS k))).
      transitivity (psum (fun sigma : Perm (S n) ⇒ - (perm_sign sigma × perm_prod A sigma))).
      + apply psum_ext. intros sigma.
        transitivity ((- perm_sign sigma) ×
          prodF (fun c ⇒ A (swap_fin (fweaken k) (Fin.FS k)
            (perm_apply (perm_postcompose sigma (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k))) c)) c)).
        × apply ring_mul_proper.
          -- exact (perm_sign_postcompose_adj n k sigma).
          -- reflexivity.
        × rewrite (mul_neg_l (perm_sign sigma)
            (prodF (fun c ⇒ A (swap_fin (fweaken k) (Fin.FS k)
              (perm_apply (perm_postcompose sigma (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k))) c)) c))).
          apply group_neg_proper.
          apply ring_mul_proper.
          -- reflexivity.
          -- unfold perm_prod.
             apply prodF_ext. intros c.
             rewrite (perm_apply_postcompose (S n) sigma (swap_fin (fweaken k) (Fin.FS k)) (swap_fin (fweaken k) (Fin.FS k))
                        (swap_fin_invol (S n) (fweaken k) (Fin.FS k)) (swap_fin_invol (S n) (fweaken k) (Fin.FS k)) c),
                (swap_fin_invol (S n) (fweaken k) (Fin.FS k) (perm_apply sigma c)).
             reflexivity.
      + exact (psum_neg (S n) (fun sigma ⇒ perm_sign sigma × perm_prod A sigma)).
  Qed.

  Theorem detM_swap_adjacent : ∀ (n : nat) (k : Fin.t n) (A : Matrix (S n) (S n) F),
    detM (row_swap A (fweaken k) (Fin.FS k)) == - detM A.
  Proof.
    intros n k A.
    rewrite <- (detM_leibniz_eq_detM (S n) (row_swap A (fweaken k) (Fin.FS k))).
    rewrite (detM_leibniz_swap_adjacent n k A).
    apply group_neg_proper.
    exact (detM_leibniz_eq_detM (S n) A).
  Qed.

End DetSwapAdjacent.


Definition row3 {A : Type} (a b c : A) : Fin.t 3 → A :=
  fun j ⇒ match j with
           | Fin.F1 ⇒ a
           | Fin.FS Fin.F1 ⇒ b
           | Fin.FS (Fin.FS _) ⇒ c
           end.

Definition mat3 {A : Type} (a11 a12 a13 a21 a22 a23 a31 a32 a33 : A) : Matrix 3 3 A :=
  fun i ⇒ match i with
           | Fin.F1 ⇒ row3 a11 a12 a13
           | Fin.FS Fin.F1 ⇒ row3 a21 a22 a23
           | Fin.FS (Fin.FS _) ⇒ row3 a31 a32 a33
           end.

Definition mat3_ns : Matrix 3 3 Q :=
  mat3 (1#1)%Q (2#1)%Q (3#1)%Q (4#1)%Q (5#1)%Q (6#1)%Q (7#1)%Q (8#1)%Q (10#1)%Q.

Example detM_mat3_ns_Q : detM mat3_ns == ((-3)#1)%Q.
Proof.
  reflexivity.
Qed.

Example detM_swap_adjacent_mid_Q_compute :
  detM (row_swap mat3_ns (fweaken (Fin.FS Fin.F1)) (Fin.FS (Fin.FS Fin.F1)))
    == (3#1)%Q.
Proof.
  reflexivity.
Qed.

Example detM_swap_adjacent_mid_Q :
  detM (row_swap mat3_ns (fweaken (Fin.FS Fin.F1)) (Fin.FS (Fin.FS Fin.F1)))
    == - detM mat3_ns.
Proof.
  exact (detM_swap_adjacent 2 (Fin.FS Fin.F1) mat3_ns).
Qed.
