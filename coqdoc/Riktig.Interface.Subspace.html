Library Riktig.Interface.Subspace

From Riktig Require Import Ops Interface.VectorSpace.

Class SubspacePred (F V : Type) `{VectorSpace F V} (P : V → Prop) := {
  sp_zero : P 0;
  sp_add : ∀ u v : V, P u → P v → P (u + v);
  sp_scale : ∀ (a : F) (u : V), P u → P (a *: u)
}.
