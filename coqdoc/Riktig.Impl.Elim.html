Section RowSurgery.
  Context {F : Type} {FF : Field F}.


  Definition upd_row {n : nat} (A : Matrix n n F) (i : Fin.t n)
      (r : Fin.t n → F) : Matrix n n F :=
    fun i' j ⇒ if dec (i' = i) then r j else A i' j.

  Lemma upd_row_at : ∀ (n : nat) (A : Matrix n n F) (i : Fin.t n)
      (r : Fin.t n → F) (j : Fin.t n),
    upd_row A i r i j == r j.
  Proof.
    intros n A i r j.
    unfold upd_row.
    destruct (dec (i = i)) as [_ | Hne].
    - reflexivity.
    - destruct (Hne eq_refl).
  Qed.

  Lemma upd_row_ne : ∀ (n : nat) (A : Matrix n n F) (i i' : Fin.t n)
      (r : Fin.t n → F) (j : Fin.t n),
    i' ≠ i → upd_row A i r i' j == A i' j.
  Proof.
    intros n A i i' r j Hne.
    unfold upd_row.
    destruct (dec (i' = i)) as [Heq | _].
    - destruct (Hne Heq).
    - reflexivity.
  Qed.

  Lemma upd_row_self : ∀ (n : nat) (A : Matrix n n F) (i : Fin.t n),
    upd_row A i (A i) == A.
  Proof.
    intros n A i i' j.
    unfold upd_row.
    destruct (dec (i' = i)) as [Heq | _].
    - rewrite Heq. reflexivity.
    - reflexivity.
  Qed.

  Lemma minor_upd_row_self : ∀ (p : nat) (A : Matrix (S p) (S p) F)
      (i : Fin.t (S p)) (r : Fin.t (S p) → F),
    minor (upd_row A i r) i == minor A i.
  Proof.
    intros p A i r s c.
    exact (upd_row_ne (S p) A i (skip i s) r (Fin.FS c) (skip_ne p i s)).
  Qed.


  Lemma minor_upd_row_ne : ∀ (p : nat) (A : Matrix (S p) (S p) F)
      (i r' : Fin.t (S p)) (r : Fin.t (S p) → F) (t : Fin.t p),
    r' ≠ i → skip r' t = i →
    minor (upd_row A i r) r' == upd_row (minor A r') t (fun c ⇒ r (Fin.FS c)).
  Proof.
    intros p A i r' r t Hne Hsk s c.
    rewrite (minor_def p (upd_row A i r) r' s c).
    destruct (Fin.eq_dec s t) as [Heq | Hneq].
    - rewrite Heq.
      rewrite (upd_row_at p (minor A r') t (fun c0 ⇒ r (Fin.FS c0)) c).
      rewrite Hsk.
      exact (upd_row_at (S p) A i r (Fin.FS c)).
    - assert (Hne2 : skip r' s ≠ i).
      { intros Heq2. apply Hneq. apply (skip_inj p r' s t). rewrite Heq2, Hsk. reflexivity. }
      rewrite (upd_row_ne (S p) A i (skip r' s) r (Fin.FS c) Hne2).
      rewrite (upd_row_ne p (minor A r') t s (fun c0 ⇒ r (Fin.FS c0)) c Hneq).
      exact (minor_def p A r' s c).
  Qed.

End RowSurgery.

Section RowLinearity.
  Context {F : Type} {FF : Field F}.

  Lemma detM_1 : ∀ A : Matrix 1 1 F, detM A == A Fin.F1 Fin.F1.
  Proof.
    intros A.
    rewrite (detM_S 0 A),
      (bigopF_S 0 (fun i ⇒ sign_at i × (A i Fin.F1 × detM (minor A i)))),
      (bigopF_0 (fun i : Fin.t 0 ⇒
         sign_at (Fin.FS i) × (A (Fin.FS i) Fin.F1 × detM (minor A (Fin.FS i))))),
      (sign_at_F1 0), (detM_0 (minor A Fin.F1)),
      (ring_mul_1r (A Fin.F1 Fin.F1)), (ring_mul_1l (A Fin.F1 Fin.F1)).
    exact (monoid_right_identity (A Fin.F1 Fin.F1)).
  Qed.


  Lemma det_rowF1_expand : ∀ (p : nat) (A : Matrix (S (S p)) (S (S p)) F)
      (r : Fin.t (S (S p)) → F),
    detM (upd_row A Fin.F1 r)
      == r Fin.F1 × detM (minor A Fin.F1)
         + bigopF (fun s : Fin.t (S p) ⇒
             sign_at (Fin.FS s) × (A (Fin.FS s) Fin.F1 ×
               detM (upd_row (minor A (Fin.FS s)) Fin.F1 (fun c ⇒ r (Fin.FS c))))).
  Proof.
    intros p A r.
    rewrite (detM_S (S p) (upd_row A Fin.F1 r)),
      (bigopF_S (S p) (fun i ⇒ sign_at i ×
         (upd_row A Fin.F1 r i Fin.F1 × detM (minor (upd_row A Fin.F1 r) i)))).
    apply monoid_add_proper.
    - rewrite (upd_row_at (S (S p)) A Fin.F1 r Fin.F1),
        (detM_proper (S p) (minor (upd_row A Fin.F1 r) Fin.F1) (minor A Fin.F1)
           (minor_upd_row_self (S p) A Fin.F1 r)),
        (sign_at_F1 (S p)).
      exact (ring_mul_1l (r Fin.F1 × detM (minor A Fin.F1))).
    - apply bigopF_ext. intros s.
      rewrite (upd_row_ne (S (S p)) A Fin.F1 (Fin.FS s) r Fin.F1
                 (fin_FS_neq_F1 (S p) s)),
        (detM_proper (S p) (minor (upd_row A Fin.F1 r) (Fin.FS s))
           (upd_row (minor A (Fin.FS s)) Fin.F1 (fun c ⇒ r (Fin.FS c)))
           (minor_upd_row_ne (S p) A Fin.F1 (Fin.FS s) r Fin.F1
              (fin_FS_neq_F1 (S p) s) (skip_FS_F1 p s))).
      reflexivity.
  Qed.

  Lemma mul_last_distr : ∀ s a x y : F,
    s × (a × (x + y)) == s × (a × x) + s × (a × y).
  Proof.
    intros s a x y.
    rewrite (ring_distl a x y).
    exact (ring_distl s (a × x) (a × y)).
  Qed.

  Theorem det_rowF1_add : ∀ (p : nat) (A : Matrix (S p) (S p) F)
      (r1 r2 : Fin.t (S p) → F),
    detM (upd_row A Fin.F1 (fun k ⇒ r1 k + r2 k))
      == detM (upd_row A Fin.F1 r1) + detM (upd_row A Fin.F1 r2).
  Proof.
    intros p; induction p as [| p IH]; intros A r1 r2.
    - rewrite (detM_1 (upd_row A Fin.F1 (fun k ⇒ r1 k + r2 k))),
        (detM_1 (upd_row A Fin.F1 r1)), (detM_1 (upd_row A Fin.F1 r2)),
        (upd_row_at 1 A Fin.F1 (fun k ⇒ r1 k + r2 k) Fin.F1),
        (upd_row_at 1 A Fin.F1 r1 Fin.F1), (upd_row_at 1 A Fin.F1 r2 Fin.F1).
      reflexivity.
    - rewrite (det_rowF1_expand p A (fun k ⇒ r1 k + r2 k)),
        (det_rowF1_expand p A r1), (det_rowF1_expand p A r2).
      transitivity
        ((r1 Fin.F1 + r2 Fin.F1) × detM (minor A Fin.F1)
         + bigopF (fun s : Fin.t (S p) ⇒
             sign_at (Fin.FS s) × (A (Fin.FS s) Fin.F1 ×
               detM (upd_row (minor A (Fin.FS s)) Fin.F1 (fun c ⇒ r1 (Fin.FS c))))
             + sign_at (Fin.FS s) × (A (Fin.FS s) Fin.F1 ×
                 detM (upd_row (minor A (Fin.FS s)) Fin.F1
                   (fun c ⇒ r2 (Fin.FS c)))))).
      + apply monoid_add_proper.
        × reflexivity.
        × apply bigopF_ext. intros s.
          rewrite (IH (minor A (Fin.FS s)) (fun c ⇒ r1 (Fin.FS c))
                     (fun c ⇒ r2 (Fin.FS c))).
          exact (mul_last_distr (sign_at (Fin.FS s)) (A (Fin.FS s) Fin.F1)
                   (detM (upd_row (minor A (Fin.FS s)) Fin.F1
                      (fun c ⇒ r1 (Fin.FS c))))
                   (detM (upd_row (minor A (Fin.FS s)) Fin.F1
                      (fun c ⇒ r2 (Fin.FS c))))).
      + rewrite (ring_distr (r1 Fin.F1) (r2 Fin.F1) (detM (minor A Fin.F1))),
          (bigopF_add (S p)
             (fun s : Fin.t (S p) ⇒ sign_at (Fin.FS s) × (A (Fin.FS s) Fin.F1 ×
                detM (upd_row (minor A (Fin.FS s)) Fin.F1 (fun c ⇒ r1 (Fin.FS c)))))
             (fun s : Fin.t (S p) ⇒ sign_at (Fin.FS s) × (A (Fin.FS s) Fin.F1 ×
                detM (upd_row (minor A (Fin.FS s)) Fin.F1 (fun c ⇒ r2 (Fin.FS c)))))).
        exact (add_interchange
          (r1 Fin.F1 × detM (minor A Fin.F1))
          (r2 Fin.F1 × detM (minor A Fin.F1))
          (bigopF (fun s : Fin.t (S p) ⇒ sign_at (Fin.FS s) × (A (Fin.FS s) Fin.F1 ×
             detM (upd_row (minor A (Fin.FS s)) Fin.F1 (fun c ⇒ r1 (Fin.FS c))))))
          (bigopF (fun s : Fin.t (S p) ⇒ sign_at (Fin.FS s) × (A (Fin.FS s) Fin.F1 ×
             detM (upd_row (minor A (Fin.FS s)) Fin.F1 (fun c ⇒ r2 (Fin.FS c))))))).
  Qed.

  Lemma mul_last_scale : ∀ s a c x : F, s × (a × (c × x)) == c × (s × (a × x)).
  Proof.
    intros s a c x.
    rewrite (mul_comm_swap a c x).
    exact (mul_comm_swap s c (a × x)).
  Qed.

  Theorem det_rowF1_scale : ∀ (p : nat) (A : Matrix (S p) (S p) F)
      (c : F) (r0 : Fin.t (S p) → F),
    detM (upd_row A Fin.F1 (fun k ⇒ c × r0 k))
      == c × detM (upd_row A Fin.F1 r0).
  Proof.
    intros p; induction p as [| p IH]; intros A c r0.
    - rewrite (detM_1 (upd_row A Fin.F1 (fun k ⇒ c × r0 k))),
        (detM_1 (upd_row A Fin.F1 r0)),
        (upd_row_at 1 A Fin.F1 (fun k ⇒ c × r0 k) Fin.F1),
        (upd_row_at 1 A Fin.F1 r0 Fin.F1).
      reflexivity.
    - rewrite (det_rowF1_expand p A (fun k ⇒ c × r0 k)),
        (det_rowF1_expand p A r0).
      transitivity
        (c × (r0 Fin.F1 × detM (minor A Fin.F1))
         + bigopF (fun s : Fin.t (S p) ⇒
             c × (sign_at (Fin.FS s) × (A (Fin.FS s) Fin.F1 ×
               detM (upd_row (minor A (Fin.FS s)) Fin.F1 (fun c0 ⇒ r0 (Fin.FS c0))))))).
      + apply monoid_add_proper.
        × exact (ring_mul_assoc c (r0 Fin.F1) (detM (minor A Fin.F1))).
        × apply bigopF_ext. intros s.
          rewrite (IH (minor A (Fin.FS s)) c (fun c0 ⇒ r0 (Fin.FS c0))).
          exact (mul_last_scale (sign_at (Fin.FS s)) (A (Fin.FS s) Fin.F1) c
                   (detM (upd_row (minor A (Fin.FS s)) Fin.F1
                      (fun c0 ⇒ r0 (Fin.FS c0))))).
      + rewrite (bigopF_mul_l (S p) c
          (fun s : Fin.t (S p) ⇒ sign_at (Fin.FS s) × (A (Fin.FS s) Fin.F1 ×
             detM (upd_row (minor A (Fin.FS s)) Fin.F1 (fun c0 ⇒ r0 (Fin.FS c0)))))).
        symmetry.
        exact (ring_distl c (r0 Fin.F1 × detM (minor A Fin.F1))
                 (bigopF (fun s : Fin.t (S p) ⇒ sign_at (Fin.FS s) ×
                    (A (Fin.FS s) Fin.F1 × detM (upd_row (minor A (Fin.FS s)) Fin.F1
                      (fun c0 ⇒ r0 (Fin.FS c0))))))).
  Qed.

End RowLinearity.


Section ElementaryMatrices.
  Context {F : Type} {FF : Field F}.


  Definition elem_swap {n : nat} (i j : Fin.t n) : Matrix n n F :=
    fun r c ⇒ mat_id (swap_fin i j r) c.

  Definition elem_scale {n : nat} (i : Fin.t n) (c : F) : Matrix n n F :=
    fun r c0 ⇒ if dec (r = i) then c × mat_id r c0 else mat_id r c0.

  Definition elem_add {n : nat} (i j : Fin.t n) (c : F) : Matrix n n F :=
    fun r c0 ⇒
      if dec (r = i) then mat_id i c0 + c × mat_id j c0 else mat_id r c0.


  Lemma elem_swap_mul : ∀ (n p : nat) (i j : Fin.t n) (A : Matrix n p F)
      (r : Fin.t n) (c : Fin.t p),
    mat_mul (elem_swap i j) A r c == A (swap_fin i j r) c.
  Proof.
    intros n p i j A r c.
    exact (mat_mul_1l n p A (swap_fin i j r) c).
  Qed.

  Lemma elem_scale_mul_eq : ∀ (n p : nat) (i : Fin.t n) (c : F)
      (A : Matrix n p F) (c0 : Fin.t p),
    mat_mul (elem_scale i c) A i c0 == c × A i c0.
  Proof.
    intros n p i c A c0.
    transitivity (bigopF (fun k ⇒ c × (mat_id i k × A k c0))).
    - apply bigopF_ext. intros k.
      unfold elem_scale.
      destruct (dec (i = i)) as [_ | Hne].
      + exact (ring_mul_assoc c (mat_id i k) (A k c0)).
      + destruct (Hne eq_refl).
    - rewrite (bigopF_mul_l n c (fun k ⇒ mat_id i k × A k c0)).
      apply ring_mul_proper.
      + reflexivity.
      + exact (mat_mul_1l n p A i c0).
  Qed.

  Lemma elem_scale_mul_ne : ∀ (n p : nat) (i r : Fin.t n) (c : F)
      (A : Matrix n p F) (c0 : Fin.t p),
    r ≠ i → mat_mul (elem_scale i c) A r c0 == A r c0.
  Proof.
    intros n p i r c A c0 Hri.
    transitivity (bigopF (fun k ⇒ mat_id r k × A k c0)).
    - apply bigopF_ext. intros k.
      unfold elem_scale.
      destruct (dec (r = i)) as [Heq | _].
      + destruct (Hri Heq).
      + reflexivity.
    - exact (mat_mul_1l n p A r c0).
  Qed.

  Lemma elem_add_mul_eq : ∀ (n p : nat) (i j : Fin.t n) (c : F)
      (A : Matrix n p F) (c0 : Fin.t p),
    mat_mul (elem_add i j c) A i c0 == A i c0 + c × A j c0.
  Proof.
    intros n p i j c A c0.
    transitivity (bigopF (fun k ⇒ mat_id i k × A k c0 + c × (mat_id j k × A k c0))).
    - apply bigopF_ext. intros k.
      unfold elem_add.
      destruct (dec (i = i)) as [_ | Hne].
      + transitivity ((mat_id i k + c × mat_id j k) × A k c0).
        × reflexivity.
        × rewrite (ring_distr (mat_id i k) (c × mat_id j k) (A k c0)).
          apply monoid_add_proper.
          -- reflexivity.
          -- exact (ring_mul_assoc c (mat_id j k) (A k c0)).
      + destruct (Hne eq_refl).
    - rewrite (bigopF_add n (fun k ⇒ mat_id i k × A k c0)
                 (fun k ⇒ c × (mat_id j k × A k c0))),
        (bigopF_mul_l n c (fun k ⇒ mat_id j k × A k c0)).
      apply monoid_add_proper.
      + exact (mat_mul_1l n p A i c0).
      + apply ring_mul_proper.
        × reflexivity.
        × exact (mat_mul_1l n p A j c0).
  Qed.

  Lemma elem_add_mul_ne : ∀ (n p : nat) (i j r : Fin.t n) (c : F)
      (A : Matrix n p F) (c0 : Fin.t p),
    r ≠ i → mat_mul (elem_add i j c) A r c0 == A r c0.
  Proof.
    intros n p i j r c A c0 Hri.
    transitivity (bigopF (fun k ⇒ mat_id r k × A k c0)).
    - apply bigopF_ext. intros k.
      unfold elem_add.
      destruct (dec (r = i)) as [Heq | _].
      + destruct (Hri Heq).
      + reflexivity.
    - exact (mat_mul_1l n p A r c0).
  Qed.

End ElementaryMatrices.


Section DetElemScale.
  Context {F : Type} {FF : Field F}.

  Lemma elem_scale_F1_col0 : ∀ (p : nat) (s : Fin.t p) (c : F),
    @elem_scale F FF (S p) Fin.F1 c (Fin.FS s) Fin.F1 == 0.
  Proof.
    intros p s c.
    unfold elem_scale.
    destruct (dec (Fin.FS s = Fin.F1)) as [Heq | _].
    - destruct (fin_FS_neq_F1 p s Heq).
    - exact (mat_id_col_0 p s).
  Qed.

  Lemma elem_scale_F1_diag : ∀ (p : nat) (c : F),
    @elem_scale F FF (S p) Fin.F1 c Fin.F1 Fin.F1 == c.
  Proof.
    intros p c.
    unfold elem_scale.
    destruct (dec ((Fin.F1 : Fin.t (S p)) = Fin.F1)) as [_ | Hne].
    - rewrite (mat_id_diag (S p) Fin.F1). exact (ring_mul_1r c).
    - destruct (Hne eq_refl).
  Qed.

  Lemma minor_elem_scale_F1 : ∀ (p : nat) (c : F),
    minor (@elem_scale F FF (S p) Fin.F1 c) Fin.F1 == @mat_id F FF p.
  Proof.
    intros p c s c0.
    rewrite (minor_def p (elem_scale Fin.F1 c) Fin.F1 s c0), (skip_F1 p s).
    unfold elem_scale.
    destruct (dec (Fin.FS s = Fin.F1)) as [Heq | _].
    - destruct (fin_FS_neq_F1 p s Heq).
    - exact (e_FS p s c0).
  Qed.

  Lemma elem_scale_FS_col0 : ∀ (p : nat) (i' s : Fin.t p) (c : F),
    @elem_scale F FF (S p) (Fin.FS i') c (Fin.FS s) Fin.F1 == 0.
  Proof.
    intros p i' s c.
    unfold elem_scale.
    destruct (dec (Fin.FS s = Fin.FS i')) as [_ | _].
    - rewrite (mat_id_col_0 p s). exact (mul_0_r c).
    - exact (mat_id_col_0 p s).
  Qed.

  Lemma elem_scale_FS_diag : ∀ (p : nat) (i' : Fin.t p) (c : F),
    @elem_scale F FF (S p) (Fin.FS i') c Fin.F1 Fin.F1 == 1.
  Proof.
    intros p i' c.
    unfold elem_scale.
    destruct (dec ((Fin.F1 : Fin.t (S p)) = Fin.FS i')) as [Heq | _].
    - destruct (fin_F1_neq_FS p i' Heq).
    - exact (mat_id_diag (S p) Fin.F1).
  Qed.

  Lemma minor_elem_scale_FS : ∀ (p : nat) (i' : Fin.t p) (c : F),
    minor (@elem_scale F FF (S p) (Fin.FS i') c) Fin.F1 == @elem_scale F FF p i' c.
  Proof.
    intros p i' c s c0.
    rewrite (minor_def p (elem_scale (Fin.FS i') c) Fin.F1 s c0), (skip_F1 p s).
    unfold elem_scale.
    destruct (dec (Fin.FS s = Fin.FS i')) as [Heq | Hne].
    - destruct (dec (s = i')) as [_ | Hne2].
      + apply ring_mul_proper.
        × reflexivity.
        × exact (e_FS p s c0).
      + destruct (Hne2 (Fin.FS_inj s i' Heq)).
    - destruct (dec (s = i')) as [Heq2 | _].
      + destruct (Hne (f_equal Fin.FS Heq2)).
      + exact (e_FS p s c0).
  Qed.

  Theorem det_elem_scale : ∀ (n : nat) (i : Fin.t n) (c : F),
    detM (elem_scale i c) == c.
  Proof.
    intros n; induction n as [| p IH]; intros i c.
    - exact (Fin.case0 (fun i0 ⇒ detM (elem_scale i0 c) == c) i).
    - apply (Fin.caseS' i (fun i0 ⇒ detM (elem_scale i0 c) == c)).
      + rewrite (detM_S p (elem_scale (n := S p) Fin.F1 c)),
          (bigopF_S p (fun i0 ⇒ sign_at i0 ×
             (elem_scale (n := S p) Fin.F1 c i0 Fin.F1
              × detM (minor (elem_scale (n := S p) Fin.F1 c) i0)))),
          (bigopF_ext p
             (fun s : Fin.t p ⇒ sign_at (Fin.FS s) ×
                (elem_scale (n := S p) Fin.F1 c (Fin.FS s) Fin.F1
                 × detM (minor (elem_scale (n := S p) Fin.F1 c) (Fin.FS s))))
             (fun _ : Fin.t p ⇒ (0 : F))
             (fun s ⇒ det_term_col0 p (elem_scale (n := S p) Fin.F1 c) (Fin.FS s)
                          (elem_scale_F1_col0 p s c))),
          (bigopF_zero p).
        rewrite (sign_at_F1 p), (elem_scale_F1_diag p c),
          (detM_proper p (minor (elem_scale (n := S p) Fin.F1 c) Fin.F1)
             (@mat_id F FF p) (minor_elem_scale_F1 p c)),
          (det_id p), (ring_mul_1r c), (ring_mul_1l c).
        exact (monoid_right_identity c).
      + intros i'.
        rewrite (detM_S p (elem_scale (n := S p) (Fin.FS i') c)),
          (bigopF_S p (fun i0 ⇒ sign_at i0 ×
             (elem_scale (n := S p) (Fin.FS i') c i0 Fin.F1
              × detM (minor (elem_scale (n := S p) (Fin.FS i') c) i0)))),
          (bigopF_ext p
             (fun s : Fin.t p ⇒ sign_at (Fin.FS s) ×
                (elem_scale (n := S p) (Fin.FS i') c (Fin.FS s) Fin.F1
                 × detM (minor (elem_scale (n := S p) (Fin.FS i') c) (Fin.FS s))))
             (fun _ : Fin.t p ⇒ (0 : F))
             (fun s ⇒ det_term_col0 p (elem_scale (n := S p) (Fin.FS i') c) (Fin.FS s)
                          (elem_scale_FS_col0 p i' s c))),
          (bigopF_zero p).
        rewrite (sign_at_F1 p), (elem_scale_FS_diag p i' c),
          (detM_proper p (minor (elem_scale (n := S p) (Fin.FS i') c) Fin.F1)
             (@elem_scale F FF p i' c) (minor_elem_scale_FS p i' c)),
          (IH i' c), (ring_mul_1l c), (ring_mul_1l c).
        exact (monoid_right_identity c).
  Qed.

End DetElemScale.


Section RowSwap.
  Context {F : Type} {FF : Field F}.

  Definition row_swap {n : nat} (A : Matrix n n F) (i j : Fin.t n)
      : Matrix n n F :=
    fun r c ⇒
      if dec (r = i) then A j c
      else if dec (r = j) then A i c
      else A r c.

  Lemma row_swap_at_i : ∀ (n : nat) (A : Matrix n n F) (i j : Fin.t n)
      (c : Fin.t n),
    row_swap A i j i c == A j c.
  Proof.
    intros n A i j c.
    unfold row_swap.
    destruct (dec (i = i)) as [_ | Hne].
    - reflexivity.
    - destruct (Hne eq_refl).
  Qed.

  Lemma row_swap_at_j : ∀ (n : nat) (A : Matrix n n F) (i j : Fin.t n)
      (c : Fin.t n),
    i ≠ j → row_swap A i j j c == A i c.
  Proof.
    intros n A i j c Hij.
    unfold row_swap.
    destruct (dec (j = i)) as [Heq | _].
    - destruct (Hij (eq_sym Heq)).
    - destruct (dec (j = j)) as [_ | Hne].
      + reflexivity.
      + destruct (Hne eq_refl).
  Qed.

  Lemma row_swap_ne : ∀ (n : nat) (A : Matrix n n F) (i j r : Fin.t n)
      (c : Fin.t n),
    r ≠ i → r ≠ j → row_swap A i j r c == A r c.
  Proof.
    intros n A i j r c Hri Hrj.
    unfold row_swap.
    destruct (dec (r = i)) as [Heq | _].
    - destruct (Hri Heq).
    - destruct (dec (r = j)) as [Heq2 | _].
      + destruct (Hrj Heq2).
      + reflexivity.
  Qed.

End RowSwap.

      (ring_mul_1l (A (Fin.FS Fin.F1) Fin.F1 × detM (minor A (Fin.FS Fin.F1)))),

             (A (Fin.FS Fin.F1) Fin.F1 × detM (minor A (Fin.FS Fin.F1)))).
  Qed.

    rewrite (row_swap_ne (S (S (S p))) A Fin.F1 (Fin.FS Fin.F1) (Fin.FS (Fin.FS s))

         (row_swap (minor A (Fin.FS (Fin.FS s))) Fin.F1 (Fin.FS Fin.F1))
         (minor_row_swap_F1FS_other p A s)),
      (IHp (minor A (Fin.FS (Fin.FS s)))),
      (mul_neg_r (A (Fin.FS (Fin.FS s)) Fin.F1) (detM (minor A (Fin.FS (Fin.FS s))))),
      (mul_neg_r (sign_at (Fin.FS (Fin.FS s)))
         (A (Fin.FS (Fin.FS s)) Fin.F1 × detM (minor A (Fin.FS (Fin.FS s))))).

  Qed.

  Theorem det_swap_F1_FS2 : ∀ (p : nat) (A : Matrix (S (S p)) (S (S p)) F),
    detM (row_swap A Fin.F1 (Fin.FS Fin.F1)) == - detM A.

        (sign_at (Fin.F1 : Fin.t 2) × (row_swap A Fin.F1 (Fin.FS Fin.F1) Fin.F1 Fin.F1
            × detM (minor (row_swap A Fin.F1 (Fin.FS Fin.F1)) Fin.F1))

                 × detM (minor (row_swap A Fin.F1 (Fin.FS Fin.F1)) (Fin.FS Fin.F1))))).

        (sign_at (Fin.F1 : Fin.t 2) × (A Fin.F1 Fin.F1 × detM (minor A Fin.F1))
         + sign_at (Fin.FS (Fin.F1 : Fin.t 1))
             × (A (Fin.FS Fin.F1) Fin.F1 × detM (minor A (Fin.FS Fin.F1))))).
      exact (swap_terms_negate 0 A).

           × (A (Fin.FS (Fin.FS s)) Fin.F1 × detM (minor A (Fin.FS (Fin.FS s))))))
        (fun s ⇒ detM_swap_tail_term_negates p IH A s)).
      rewrite (bigopF_neg (S p) (fun s : Fin.t (S p) ⇒ sign_at (Fin.FS (Fin.FS s))
           × (A (Fin.FS (Fin.FS s)) Fin.F1 × detM (minor A (Fin.FS (Fin.FS s)))))).
      rewrite (neg_add
        (sign_at (Fin.F1 : Fin.t (S (S (S p)))) × (A Fin.F1 Fin.F1 × detM (minor A Fin.F1))
         + sign_at (Fin.FS (Fin.F1 : Fin.t (S (S p))))
             × (A (Fin.FS Fin.F1) Fin.F1 × detM (minor A (Fin.FS Fin.F1))))
        (bigopF (fun s : Fin.t (S p) ⇒ sign_at (Fin.FS (Fin.FS s))
           × (A (Fin.FS (Fin.FS s)) Fin.F1 × detM (minor A (Fin.FS (Fin.FS s))))))).
      exact (monoid_add_proper _ _ (swap_terms_negate (S p) A) _ _ (reflexivity _)).
  Qed.

    exact (group_right_inverse (1 : F)).
  Qed.

  Lemma detM_eq_rows_tail_term_zero : ∀ (p : nat),
    (∀ A' : Matrix (S (S p)) (S (S p)) F,
       (∀ c, A' Fin.F1 c == A' (Fin.FS Fin.F1) c) → detM A' == 0) →
    ∀ (A : Matrix (S (S (S p))) (S (S (S p))) F),
    (∀ c, A Fin.F1 c == A (Fin.FS Fin.F1) c) →
    ∀ s : Fin.t (S p),
    sign_at (Fin.FS (Fin.FS s))
      × (A (Fin.FS (Fin.FS s)) Fin.F1 × detM (minor A (Fin.FS (Fin.FS s)))) == 0.
  Proof.
    intros p IHp A Heq s.
    rewrite (IHp (minor A (Fin.FS (Fin.FS s))) (minor_eq_rows_F1_FS2_other p A Heq s)),
      (mul_0_r (A (Fin.FS (Fin.FS s)) Fin.F1)), (mul_0_r (sign_at (Fin.FS (Fin.FS s)))).
    reflexivity.
  Qed.

  Theorem det_eq_rows_F1_FS2 : ∀ (p : nat) (A : Matrix (S (S p)) (S (S p)) F),
    (∀ c, A Fin.F1 c == A (Fin.FS Fin.F1) c) → detM A == 0.
  Proof.
    intros p; induction p as [| p IH]; intros A Heq.
    - rewrite (detM_expand2 0 A),
        (bigopF_0 (fun s : Fin.t 0 ⇒ sign_at (Fin.FS (Fin.FS s))
           × (A (Fin.FS (Fin.FS s)) Fin.F1 × detM (minor A (Fin.FS (Fin.FS s)))))),
        (monoid_right_identity (sign_at (Fin.F1 : Fin.t 2) × (A Fin.F1 Fin.F1 × detM (minor A Fin.F1))
           + sign_at (Fin.FS (Fin.F1 : Fin.t 1))
               × (A (Fin.FS Fin.F1) Fin.F1 × detM (minor A (Fin.FS Fin.F1))))).
      rewrite <- (Heq Fin.F1),
        (detM_proper 1 (minor A (Fin.FS Fin.F1)) (minor A Fin.F1)
           (minor_eq_rows_F1_FS2_sym 0 A Heq)),
        <- (ring_distr (sign_at (Fin.F1 : Fin.t 2)) (sign_at (Fin.FS (Fin.F1 : Fin.t 1)))
             (A Fin.F1 Fin.F1 × detM (minor A Fin.F1))),
        (sign_at_F1_FS_sum_zero 0).
      exact (mul_0_l (A Fin.F1 Fin.F1 × detM (minor A Fin.F1))).
    - rewrite (detM_expand2 (S p) A).
      rewrite <- (Heq Fin.F1),
        (detM_proper (S (S p)) (minor A (Fin.FS Fin.F1)) (minor A Fin.F1)
           (minor_eq_rows_F1_FS2_sym (S p) A Heq)),
        <- (ring_distr (sign_at (Fin.F1 : Fin.t (S (S (S p)))))
             (sign_at (Fin.FS (Fin.F1 : Fin.t (S (S p)))))
             (A Fin.F1 Fin.F1 × detM (minor A Fin.F1))),
        (sign_at_F1_FS_sum_zero (S p)),
        (mul_0_l (A Fin.F1 Fin.F1 × detM (minor A Fin.F1))),
        (monoid_left_identity (bigopF (fun s : Fin.t (S p) ⇒ sign_at (Fin.FS (Fin.FS s))
           × (A (Fin.FS (Fin.FS s)) Fin.F1 × detM (minor A (Fin.FS (Fin.FS s))))))),
        (bigopF_ext (S p)
           (fun s : Fin.t (S p) ⇒ sign_at (Fin.FS (Fin.FS s))
              × (A (Fin.FS (Fin.FS s)) Fin.F1 × detM (minor A (Fin.FS (Fin.FS s)))))
           (fun _ : Fin.t (S p) ⇒ (0 : F))
           (fun s ⇒ detM_eq_rows_tail_term_zero p IH A Heq s)).
      exact (bigopF_zero (S p)).
  Qed.

End DetEqRowsF1FS2.
