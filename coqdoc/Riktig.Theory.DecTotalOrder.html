Library Riktig.Theory.DecTotalOrder


From Coq Require Export RelationClasses.
From Riktig Require Export Interface.Decidable Interface.DecTotalOrder.

Section DecTotalOrderTheory.
  Context {A : Type} `{O : DecTotalOrder A}.


  Lemma ole_refl : ∀ x : A, ole x x.
    intros x; apply ole_nlt; exact (StrictOrder_Irreflexive x).
  Qed.

  Lemma olt_ole : ∀ x y : A, olt x y → ole x y.
    intros x y L; apply ole_nlt; intros L'.
    exact (StrictOrder_Irreflexive x (transitivity L L')).
  Qed.

  Lemma nolt_ole : ∀ x y : A, ¬ olt x y → ole y x.
    intros x y N; now apply ole_nlt.
  Qed.

  Lemma ole_trans : Transitive (ole (A:=A)).
    intros x y z X Y; apply ole_nlt; intros L.
    destruct (olt_total x y) as [XY|[<-|YX]].
    - exact (proj1 (ole_nlt y z) Y (transitivity L XY)).
    - exact (proj1 (ole_nlt x z) Y L).
    - exact (proj1 (ole_nlt x y) X YX).
  Qed.

  Lemma ole_antisym : ∀ x y : A, ole x y → ole y x → x = y.
    intros x y X Y; destruct (olt_total x y) as [XY|[E|YX]].
    - now destruct (proj1 (ole_nlt y x) Y).
    - exact E.
    - now destruct (proj1 (ole_nlt x y) X).
  Qed.

  #[global] Instance ord_eq_dec : EqDec A.
  Proof.
    intros x y.
    destruct (dec (olt x y)) as [XY|NXY].
    - right; intros E; subst y; exact (StrictOrder_Irreflexive x XY).
    - destruct (dec (olt y x)) as [YX|NYX].
      + right; intros E; subst y; exact (StrictOrder_Irreflexive x YX).
      + left; destruct (olt_total x y) as [XY|[E|YX]];
          [contradiction | exact E | contradiction].
  Defined.

End DecTotalOrderTheory.
