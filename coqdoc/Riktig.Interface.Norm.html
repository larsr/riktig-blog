Library Riktig.Interface.Norm

From Coq Require Export Morphisms.
From Riktig Require Export Ops Interface.VectorSpace Interface.Order.

Class HasNorm F V := vnorm : V → F.
Notation "‖ u ‖" := (vnorm u) (at level 20, u at level 200).

#[global] Hint Mode HasNorm - ! : typeclass_instances.

Class NormedSpace (F V : Type) `{OrderedField F} {VS : VectorSpace F V} := {
  #[global] ns_norm :: HasNorm F V;
  #[global] norm_proper :: Proper (equ ==> equ) (@vnorm F V _);
  norm_nonneg : ∀ u : V, 0 ≤ ‖ u ‖;
  norm_def : ∀ u : V, ‖ u ‖ == 0 → u == 0;
  norm_scale : ∀ (a : F) (u : V), ‖ a *: u ‖ == abs a × ‖ u ‖;
  norm_triangle : ∀ u v : V, ‖ u + v ‖ ≤ ‖ u ‖ + ‖ v ‖
}.

#[global] Hint Mode NormedSpace - ! - - - : typeclass_instances.
