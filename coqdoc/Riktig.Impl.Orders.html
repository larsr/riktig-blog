Definition DecTotalOrder_pullback {A B : Type} `{OB : DecTotalOrder B}
    (f : A → B) (inj : ∀ x y : A, f x = f y → x = y) : DecTotalOrder A.
Proof.
  refine {| olt := fun x y : A ⇒ olt (f x) (f y);
            ole := fun x y : A ⇒ ole (f x) (f y);
            olt_dec := fun x y : A ⇒ olt_dec (f x) (f y);
            olt_strict := _;
            olt_total := _;
            ole_nlt := fun x y : A ⇒ ole_nlt (f x) (f y) |}.
  - split.
    + intros x L; exact (StrictOrder_Irreflexive (f x) L).
    + intros x y z L1 L2; exact (transitivity L1 L2).
  - intros x y; destruct (olt_total (f x) (f y)) as [L|[E|G]].
    + left; exact L.
    + right; left; exact (inj x y E).
    + right; right; exact G.
Defined.

#[global] Instance Prod_DecTotalOrder {A B : Type}
    `{OA : DecTotalOrder A} `{OB : DecTotalOrder B} : DecTotalOrder (A × B).
Proof.
  refine {| olt := fun p q : A × B ⇒
              olt (fst p) (fst q) ∨ (fst p = fst q ∧ olt (snd p) (snd q));
            ole := fun p q : A × B ⇒
              ¬ (olt (fst q) (fst p) ∨ (fst q = fst p ∧ olt (snd q) (snd p)));
            olt_dec := fun p q : A × B ⇒ dec _;
            olt_strict := _;
            olt_total := _;
            ole_nlt := fun p q : A × B ⇒ iff_refl _ |}.
  -
    exact (dec _).
  - split.
    + intros p [L|[_ L]]; exact (StrictOrder_Irreflexive _ L).
    + intros p q r [L1|[E1 L1]] [L2|[E2 L2]].
      × left; exact (transitivity L1 L2).
      × left; rewrite <- E2; exact L1.
      × left; rewrite E1; exact L2.
      × right; split; [congruence | exact (transitivity L1 L2)].
  - intros [a b] [c d]; simpl.
    destruct (olt_total a c) as [L|[E|G]].
    + left; left; exact L.
    + subst c; destruct (olt_total b d) as [L|[E|G]].
      × left; right; split; [reflexivity | exact L].
      × subst d; right; left; reflexivity.
      × right; right; right; split; [reflexivity | exact G].
    + right; right; left; exact G.
Defined.

Section ListOrder.
  Context {A : Type} `{OA : DecTotalOrder A}.

  Fixpoint list_olt (l1 l2 : list A) : Prop :=
    match l1, l2 with
    | [], [] ⇒ False
    | [], _ :: _ ⇒ True
    | _ :: _, [] ⇒ False
    | x :: l1', y :: l2' ⇒ olt x y ∨ (x = y ∧ list_olt l1' l2')
    end.

  Definition list_olt_dec : ∀ l1 l2 : list A, Decidable (list_olt l1 l2).
  Proof.
    induction l1 as [|x l1 IH]; intros [|y l2]; simpl.
    - exact False_dec.
    - exact True_dec.
    - exact False_dec.
    - destruct (dec (olt x y)) as [L|NL].
      + left; now left.
      + destruct (dec (x = y)) as [E|NE].
        × destruct (IH l2) as [T|NT]; [left; now right | right; tauto].
        × right; tauto.
  Defined.

  Lemma list_olt_irrefl : ∀ l : list A, ¬ list_olt l l.
  Proof.
    induction l as [|x l IH]; simpl.
    - exact (fun F ⇒ F).
    - intros [L|[_ T]]; [exact (StrictOrder_Irreflexive x L) | exact (IH T)].
  Qed.

  Lemma list_olt_trans : ∀ l1 l2 l3 : list A,
      list_olt l1 l2 → list_olt l2 l3 → list_olt l1 l3.
  Proof.
    induction l1 as [|x l1 IH]; intros [|y l2] [|z l3]; simpl; try tauto.
    intros [L1|[E1 T1]] [L2|[E2 T2]].
    - left; exact (transitivity L1 L2).
    - left; rewrite <- E2; exact L1.
    - left; rewrite E1; exact L2.
    - right; split; [congruence | exact (IH l2 l3 T1 T2)].
  Qed.

  Lemma list_olt_total : ∀ l1 l2 : list A,
      list_olt l1 l2 ∨ l1 = l2 ∨ list_olt l2 l1.
  Proof.
    induction l1 as [|x l1 IH]; intros [|y l2]; simpl.
    - right; left; reflexivity.
    - left; exact I.
    - right; right; exact I.
    - destruct (olt_total x y) as [L|[E|G]].
      + left; now left.
      + subst y; destruct (IH l2) as [T|[E|T]].
        × left; right; auto.
        × subst l2; right; left; reflexivity.
        × right; right; right; auto.
      + right; right; now left.
  Qed.

  #[global] Instance List_DecTotalOrder : DecTotalOrder (list A).
  Proof.
    refine {| olt := list_olt;
              ole := fun l1 l2 : list A ⇒ ¬ list_olt l2 l1;
              olt_dec := list_olt_dec;
              olt_strict := _;
              olt_total := list_olt_total;
              ole_nlt := fun l1 l2 : list A ⇒ iff_refl _ |}.
    split; [exact list_olt_irrefl | exact list_olt_trans].
  Defined.

End ListOrder.



Lemma nat_ole_nlt : ∀ x y : nat, (x ≤ y)%nat ↔ ¬ (y < x)%nat.
Proof. intros; split; lia. Qed.

#[global] Instance Nat_DecTotalOrder : DecTotalOrder nat :=
  {| olt := Peano.lt;
     ole := Peano.le;
     olt_dec := nat_lt_dec;
     olt_strict := Nat.lt_strorder;
     olt_total := Nat.lt_trichotomy;
     ole_nlt := nat_ole_nlt |}.

Definition N_lt_dec : ∀ x y : N, Decidable (x < y)%N.
Proof.
  intros x y; unfold N.lt; destruct (x ?= y)%N;
    [right; discriminate | left; reflexivity | right; discriminate].
Defined.

#[global] Instance N_DecTotalOrder : DecTotalOrder N :=
  {| olt := N.lt;
     ole := N.le;
     olt_dec := N_lt_dec;
     olt_strict := N.lt_strorder;
     olt_total := N.lt_trichotomy;
     ole_nlt := N.le_ngt |}.

Definition Qc_lt_dec : ∀ x y : Qc, Decidable (x < y).
Proof.
  intros x y; destruct (Qclt_le_dec x y) as [L|R];
    [left; exact L | right; exact (Qcle_not_lt y x R)].
Defined.

Lemma Qclt_strorder : StrictOrder Qclt.
Proof.
  split.
  - intros x L; exact (Qclt_not_eq x x L eq_refl).
  - exact Qclt_trans.
Qed.

Lemma Qclt_total : ∀ x y : Qc, x < y ∨ x = y ∨ y < x.
Proof. intros x y; destruct (Qc_dec x y) as [[L|G]|E]; auto. Qed.

Lemma Qcle_nlt : ∀ x y : Qc, x ≤ y ↔ ¬ y < x.
Proof. intros x y; split; [exact (Qcle_not_lt x y) | exact (Qcnot_lt_le y x)]. Qed.

#[global] Instance Qc_DecTotalOrder : DecTotalOrder Qc :=
  {| olt := Qclt;
     ole := Qcle;
     olt_dec := Qc_lt_dec;
     olt_strict := Qclt_strorder;
     olt_total := Qclt_total;
     ole_nlt := Qcle_nlt |}.


Lemma nat_of_ascii_inj : ∀ x y : ascii, nat_of_ascii x = nat_of_ascii y → x = y.
Proof.
  intros x y E.
  rewrite <- (ascii_nat_embedding x), <- (ascii_nat_embedding y), E; reflexivity.
Qed.

#[global] Instance Ascii_DecTotalOrder : DecTotalOrder ascii :=
  DecTotalOrder_pullback nat_of_ascii nat_of_ascii_inj.

Lemma byte_to_N_inj : ∀ x y : byte, Byte.to_N x = Byte.to_N y → x = y.
Proof.
  intros x y E.
  pose proof (Byte.of_to_N x) as Hx; pose proof (Byte.of_to_N y) as Hy.
  rewrite E, Hy in Hx; now injection Hx.
Qed.

#[global] Instance Byte_DecTotalOrder : DecTotalOrder byte :=
  DecTotalOrder_pullback Byte.to_N byte_to_N_inj.

Lemma list_ascii_of_string_inj : ∀ s t : string,
    list_ascii_of_string s = list_ascii_of_string t → s = t.
Proof.
  intros s t E.
  rewrite <- (string_of_list_ascii_of_string s),
          <- (string_of_list_ascii_of_string t), E; reflexivity.
Qed.

#[global] Instance String_DecTotalOrder : DecTotalOrder string :=
  DecTotalOrder_pullback list_ascii_of_string list_ascii_of_string_inj.
