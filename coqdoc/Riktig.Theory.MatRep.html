Section LincombDelta.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V}.

  Lemma scale_e_eq : ∀ (k : nat) (i j : Fin.t k) (v : V),
    i = j → e i j *: v == v.
  Proof.
    intros k i j v Hij.
    rewrite (e_at_eq k i j Hij).
    exact (sc_scale_1 v).
  Qed.

  Lemma scale_e_neq : ∀ (k : nat) (i j : Fin.t k) (v : V),
    i ≠ j → e i j *: v == 0.
  Proof.
    intros k i j v Hij.
    rewrite (e_at_neq k i j Hij).
    exact (zero_scale v).
  Qed.

  Lemma scale_e_FS : ∀ (k : nat) (i j : Fin.t k) (v : V),
    e (Fin.FS i) (Fin.FS j) *: v == e i j *: v.
  Proof.
    intros k i j v.
    rewrite (e_FS k i j).
    reflexivity.
  Qed.

  Lemma lincomb_delta_sum : ∀ (k : nat) (b : Fin.t k → V)
      (j : Fin.t k),
    bigopF (fun i ⇒ e j i *: b i) == b j.
  Proof.
    intros k; induction k as [| k IH]; intros b j.
    - exact (Fin.case0
               (fun j0 ⇒ bigopF (fun i ⇒ e j0 i *: b i) == b j0) j).
    - apply (Fin.caseS' j
               (fun j0 ⇒ bigopF (fun i ⇒ e j0 i *: b i) == b j0)).
      + transitivity (@e F FF (S k) Fin.F1 Fin.F1 *: b Fin.F1
            + bigopF (fun i ⇒ e Fin.F1 (Fin.FS i) *: b (Fin.FS i))).
        × reflexivity.
        × rewrite (scale_e_eq (S k) Fin.F1 Fin.F1 (b Fin.F1) eq_refl),
            (bigopF_ext k
               (fun i ⇒ e Fin.F1 (Fin.FS i) *: b (Fin.FS i))
               (fun _ : Fin.t k ⇒ (0 : V))
               (fun i ⇒ scale_e_neq (S k) Fin.F1 (Fin.FS i)
                           (b (Fin.FS i)) (fin_F1_neq_FS k i))),
            (bigopF_zero k).
          exact (mon_idr (b Fin.F1)).
      + intros j'.
        transitivity (@e F FF (S k) (Fin.FS j') Fin.F1 *: b Fin.F1
            + bigopF (fun i ⇒ e (Fin.FS j') (Fin.FS i) *: b (Fin.FS i))).
        × reflexivity.
        × rewrite (scale_e_neq (S k) (Fin.FS j') Fin.F1 (b Fin.F1)
              (fin_FS_neq_F1 k j')),
            (bigopF_ext k
               (fun i ⇒ e (Fin.FS j') (Fin.FS i) *: b (Fin.FS i))
               (fun i ⇒ e j' i *: b (Fin.FS i))
               (fun i ⇒ scale_e_FS k j' i (b (Fin.FS i)))),
            (IH (fun i ⇒ b (Fin.FS i)) j').
          exact (mon_idl (b (Fin.FS j'))).
  Qed.

  Lemma lincomb_delta : ∀ (k : nat) (b : Fin.t k → V) (j : Fin.t k),
    lincomb b (e j) == b j.
  Proof.
    intros k b j.
    exact (lincomb_delta_sum k b j).
  Qed.

  Lemma lincomb_double : ∀ (k l : nat) (b : Fin.t k → V)
      (M : Fin.t k → Fin.t l → F) (c : Fin.t l → F),
    bigopF (fun j ⇒ c j *: bigopF (fun i ⇒ M i j *: b i))
      == lincomb b (fun i ⇒ bigopF (fun j ⇒ M i j × c j)).
  Proof.
    intros k l b M c.
    transitivity (bigopF (fun j ⇒ bigopF (fun i ⇒ c j *: (M i j *: b i)))).
    - apply bigopF_ext. intros j.
      symmetry.
      exact (bigopF_scale k (c j) (fun i ⇒ M i j *: b i)).
    - rewrite (bigopF_swap l k (fun j i ⇒ c j *: (M i j *: b i))).
      apply bigopF_ext. intros i.
      transitivity (bigopF (fun j ⇒ (M i j × c j) *: b i)).
      + apply bigopF_ext. intros j.
        exact (scale_flip (c j) (M i j) (b i)).
      + exact (bigopF_scale_r l (fun j ⇒ M i j × c j) (b i)).
  Qed.

End LincombDelta.


Section LinearMapSums.
  Context {F U V : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}.
  Context (f : LinearMap F U V).

  Theorem lm_bigopF : ∀ (k : nat) (g : Fin.t k → U),
    f (bigopF g) == bigopF (fun i ⇒ f (g i)).
  Proof.
    intros k; induction k as [| k IH]; intros g.
    - exact (lm_zero f).
    - transitivity (f (g Fin.F1) + f (bigopF (fun i ⇒ g (Fin.FS i)))).
      + exact (lm_add f (g Fin.F1) (bigopF (fun i ⇒ g (Fin.FS i)))).
      + exact (mon_add_proper
                 (f (g Fin.F1)) (f (g Fin.F1)) (reflexivity (f (g Fin.F1)))
                 (f (bigopF (fun i ⇒ g (Fin.FS i))))
                 (bigopF (fun i ⇒ f (g (Fin.FS i))))
                 (IH (fun i ⇒ g (Fin.FS i)))).
  Qed.

  Theorem lm_lincomb : ∀ (k : nat) (b : Fin.t k → U)
      (c : Fin.t k → F),
    f (lincomb b c) == lincomb (fun i ⇒ f (b i)) c.
  Proof.
    intros k b c.
    transitivity (bigopF (fun i ⇒ f (c i *: b i))).
    - exact (lm_bigopF k (fun i ⇒ c i *: b i)).
    - exact (bigopF_ext k (fun i ⇒ f (c i *: b i))
               (fun i ⇒ c i *: f (b i))
               (fun i ⇒ lm_scale f (c i) (b i))).
  Qed.

End LinearMapSums.


Section CoordsPointwise.
  Context {F V : Type} {FF : Field F} {VV : VectorSpace F V}.
  Context {k : nat} (B : Basis k F V).

  Lemma coords_proper : ∀ u v : V,
    u == v → ∀ i : Fin.t k, basis_coords B u i == basis_coords B v i.
  Proof.
    intros u v Huv i.
    exact (lm_proper (basis_coords B) u v Huv i).
  Qed.

  Lemma coords_add : ∀ (u v : V) (i : Fin.t k),
    basis_coords B (u + v) i == basis_coords B u i + basis_coords B v i.
  Proof.
    intros u v i.
    exact (lm_add (basis_coords B) u v i).
  Qed.

  Lemma coords_scale : ∀ (a : F) (u : V) (i : Fin.t k),
    basis_coords B (a *: u) i == a × basis_coords B u i.
  Proof.
    intros a u i.
    exact (lm_scale (basis_coords B) a u i).
  Qed.

  Lemma coords_vec : ∀ j : Fin.t k,
    basis_coords B (basis_vec B j) == e j.
  Proof.
    intros j.
    transitivity (basis_coords B (lincomb (basis_vec B) (e j))).
    - apply (lm_proper (basis_coords B)).
      symmetry.
      exact (lincomb_delta k (basis_vec B) j).
    - exact (basis_coords_lincomb B (e j)).
  Qed.

End CoordsPointwise.


Section MatRep.
  Context {F U V : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}.
  Context {m n : nat} (BU : Basis n F U) (BV : Basis m F V).

  Definition mat_of (f : LinearMap F U V) : Matrix m n F :=
    fun i j ⇒ basis_coords BV (f (basis_vec BU j)) i.

  Definition hom_of_coeff (A : Matrix m n F) (u : U) : Fin.t m → F :=
    fun i ⇒ bigopF (fun j ⇒ A i j × basis_coords BU u j).

  Definition hom_of_fun (A : Matrix m n F) (u : U) : V :=
    lincomb (basis_vec BV) (hom_of_coeff A u).


  Lemma hom_of_coeff_proper : ∀ (A : Matrix m n F) (u u' : U),
    u == u' →
    ∀ i : Fin.t m, hom_of_coeff A u i == hom_of_coeff A u' i.
  Proof.
    intros A u u' Hu i.
    exact (bigopF_ext n (fun j ⇒ A i j × basis_coords BU u j)
             (fun j ⇒ A i j × basis_coords BU u' j)
             (fun j ⇒ ring_mul_proper (A i j) (A i j) (reflexivity (A i j))
                         (basis_coords BU u j) (basis_coords BU u' j)
                         (coords_proper BU u u' Hu j))).
  Qed.

  Lemma hom_of_coeff_add : ∀ (A : Matrix m n F) (u v : U)
      (i : Fin.t m),
    hom_of_coeff A (u + v) i == hom_of_coeff A u i + hom_of_coeff A v i.
  Proof.
    intros A u v i.
    transitivity (bigopF (fun j ⇒ A i j × basis_coords BU u j
                                 + A i j × basis_coords BU v j)).
    - apply bigopF_ext. intros j.
      rewrite (coords_add BU u v j).
      exact (ring_distl (A i j) (basis_coords BU u j)
               (basis_coords BU v j)).
    - exact (bigopF_add n (fun j ⇒ A i j × basis_coords BU u j)
               (fun j ⇒ A i j × basis_coords BU v j)).
  Qed.

  Lemma hom_of_coeff_scale : ∀ (A : Matrix m n F) (a : F) (u : U)
      (i : Fin.t m),
    hom_of_coeff A (a *: u) i == a × hom_of_coeff A u i.
  Proof.
    intros A a u i.
    transitivity (bigopF (fun j ⇒ a × (A i j × basis_coords BU u j))).
    - apply bigopF_ext. intros j.
      rewrite (coords_scale BU a u j).
      exact (mul_comm_swap (A i j) a (basis_coords BU u j)).
    - exact (bigopF_mul_l n a (fun j ⇒ A i j × basis_coords BU u j)).
  Qed.


  Lemma hom_of_fun_proper : ∀ A : Matrix m n F,
    Proper (equ ==> equ) (hom_of_fun A).
  Proof.
    intros A u u' Hu.
    apply (lincomb_proper m (basis_vec BV)
             (hom_of_coeff A u) (hom_of_coeff A u')).
    intros i.
    exact (hom_of_coeff_proper A u u' Hu i).
  Qed.

  Lemma hom_of_fun_add : ∀ (A : Matrix m n F) (u v : U),
    hom_of_fun A (u + v) == hom_of_fun A u + hom_of_fun A v.
  Proof.
    intros A u v.
    transitivity (lincomb (basis_vec BV)
        (hom_of_coeff A u + hom_of_coeff A v)).
    - apply (lincomb_proper m (basis_vec BV) (hom_of_coeff A (u + v))
               (hom_of_coeff A u + hom_of_coeff A v)).
      intros i.
      exact (hom_of_coeff_add A u v i).
    - exact (lincomb_add m (basis_vec BV) (hom_of_coeff A u)
               (hom_of_coeff A v)).
  Qed.

  Lemma hom_of_fun_scale : ∀ (A : Matrix m n F) (a : F) (u : U),
    hom_of_fun A (a *: u) == a *: hom_of_fun A u.
  Proof.
    intros A a u.
    transitivity (lincomb (basis_vec BV) (a *: hom_of_coeff A u)).
    - apply (lincomb_proper m (basis_vec BV) (hom_of_coeff A (a *: u))
               (a *: hom_of_coeff A u)).
      intros i.
      exact (hom_of_coeff_scale A a u i).
    - exact (lincomb_scale m a (basis_vec BV) (hom_of_coeff A u)).
  Qed.

  Definition hom_of (A : Matrix m n F) : LinearMap F U V :=
    @Build_LinearMap F U V FF VU VV (hom_of_fun A)
      (hom_of_fun_proper A) (hom_of_fun_add A) (hom_of_fun_scale A).


  Lemma hom_expand : ∀ (f : LinearMap F U V) (u : U),
    f u == lincomb (basis_vec BV) (hom_of_coeff (mat_of f) u).
  Proof.
    intros f u.
    transitivity (lincomb (fun j ⇒ f (basis_vec BU j))
        (basis_coords BU u)).
    - transitivity (f (lincomb (basis_vec BU) (basis_coords BU u))).
      + apply (lm_proper f).
        symmetry.
        exact (basis_recon BU u).
      + exact (lm_lincomb f n (basis_vec BU) (basis_coords BU u)).
    - transitivity (bigopF (fun j ⇒ basis_coords BU u j
          *: bigopF (fun i ⇒ mat_of f i j *: basis_vec BV i))).
      + apply bigopF_ext. intros j.
        apply sc_proper.
        × reflexivity.
        × symmetry.
          exact (basis_recon BV (f (basis_vec BU j))).
      + exact (lincomb_double m n (basis_vec BV) (mat_of f)
                 (basis_coords BU u)).
  Qed.

  Theorem mat_of_coords : ∀ (f : LinearMap F U V) (u : U)
      (i : Fin.t m),
    basis_coords BV (f u) i == hom_of_coeff (mat_of f) u i.
  Proof.
    intros f u i.
    transitivity (basis_coords BV
        (lincomb (basis_vec BV) (hom_of_coeff (mat_of f) u)) i).
    - exact (lm_proper (basis_coords BV) (f u)
               (lincomb (basis_vec BV) (hom_of_coeff (mat_of f) u))
               (hom_expand f u) i).
    - exact (basis_coords_lincomb BV (hom_of_coeff (mat_of f) u) i).
  Qed.


  Theorem hom_of_mat_of : ∀ f : LinearMap F U V,
    hom_of (mat_of f) == f.
  Proof.
    intros f u.
    transitivity (lincomb (basis_vec BV) (basis_coords BV (f u))).
    - apply (lincomb_proper m (basis_vec BV)
               (hom_of_coeff (mat_of f) u) (basis_coords BV (f u))).
      intros i.
      symmetry.
      exact (mat_of_coords f u i).
    - exact (basis_recon BV (f u)).
  Qed.


  Lemma hom_of_coeff_vec : ∀ (A : Matrix m n F) (j : Fin.t n)
      (i : Fin.t m),
    hom_of_coeff A (basis_vec BU j) i == A i j.
  Proof.
    intros A j i.
    transitivity (bigopF (fun j' ⇒ A i j' × e j' j)).
    - apply bigopF_ext. intros j'.
      rewrite (coords_vec BU j j'), (e_sym n j j').
      reflexivity.
    - exact (lincomb_std_delta n (fun j' ⇒ A i j') j).
  Qed.

  Theorem mat_of_hom_of : ∀ A : Matrix m n F,
    mat_of (hom_of A) == A.
  Proof.
    intros A i j.
    transitivity (basis_coords BV
        (lincomb (basis_vec BV) (fun i' : Fin.t m ⇒ A i' j)) i).
    - exact (lm_proper (basis_coords BV)
               (hom_of A (basis_vec BU j))
               (lincomb (basis_vec BV) (fun i' : Fin.t m ⇒ A i' j))
               (lincomb_proper m (basis_vec BV)
                  (hom_of_coeff A (basis_vec BU j))
                  (fun i' : Fin.t m ⇒ A i' j)
                  (fun i' ⇒ hom_of_coeff_vec A j i')) i).
    - exact (basis_coords_lincomb BV (fun i' : Fin.t m ⇒ A i' j) i).
  Qed.


  Lemma mat_of_proper : Proper (equ ==> equ) mat_of.
  Proof.
    intros f g Hfg i j.
    exact (coords_proper BV (f (basis_vec BU j)) (g (basis_vec BU j))
             (Hfg (basis_vec BU j)) i).
  Qed.

  Theorem mat_of_add : ∀ f g : LinearMap F U V,
    mat_of (f + g) == mat_of f + mat_of g.
  Proof.
    intros f g i j.
    exact (coords_add BV (f (basis_vec BU j)) (g (basis_vec BU j)) i).
  Qed.

  Theorem mat_of_scale : ∀ (a : F) (f : LinearMap F U V),
    mat_of (a *: f) == a *: mat_of f.
  Proof.
    intros a f i j.
    exact (coords_scale BV a (f (basis_vec BU j)) i).
  Qed.


  Lemma hom_of_coeff_matadd : ∀ (A A' : Matrix m n F) (u : U)
      (i : Fin.t m),
    hom_of_coeff (A + A') u i == hom_of_coeff A u i + hom_of_coeff A' u i.
  Proof.
    intros A A' u i.
    transitivity (bigopF (fun j ⇒ A i j × basis_coords BU u j
                                 + A' i j × basis_coords BU u j)).
    - apply bigopF_ext. intros j.
      exact (ring_distr (A i j) (A' i j) (basis_coords BU u j)).
    - exact (bigopF_add n (fun j ⇒ A i j × basis_coords BU u j)
               (fun j ⇒ A' i j × basis_coords BU u j)).
  Qed.

  Lemma hom_of_coeff_matscale : ∀ (a : F) (A : Matrix m n F) (u : U)
      (i : Fin.t m),
    hom_of_coeff (a *: A) u i == a × hom_of_coeff A u i.
  Proof.
    intros a A u i.
    transitivity (bigopF (fun j ⇒ a × (A i j × basis_coords BU u j))).
    - apply bigopF_ext. intros j.
      exact (ring_mul_assoc a (A i j) (basis_coords BU u j)).
    - exact (bigopF_mul_l n a (fun j ⇒ A i j × basis_coords BU u j)).
  Qed.

  Lemma hom_of_matproper : Proper (equ ==> equ) hom_of.
  Proof.
    intros A A' HA u.
    apply (lincomb_proper m (basis_vec BV)
             (hom_of_coeff A u) (hom_of_coeff A' u)).
    intros i.
    apply bigopF_ext. intros j.
    exact (ring_mul_proper (A i j) (A' i j) (HA i j)
             (basis_coords BU u j) (basis_coords BU u j)
             (reflexivity (basis_coords BU u j))).
  Qed.

  Theorem hom_of_matadd : ∀ A A' : Matrix m n F,
    hom_of (A + A') == hom_of A + hom_of A'.
  Proof.
    intros A A' u.
    transitivity (lincomb (basis_vec BV)
        (hom_of_coeff A u + hom_of_coeff A' u)).
    - apply (lincomb_proper m (basis_vec BV)
               (hom_of_coeff (A + A') u)
               (hom_of_coeff A u + hom_of_coeff A' u)).
      intros i.
      exact (hom_of_coeff_matadd A A' u i).
    - exact (lincomb_add m (basis_vec BV) (hom_of_coeff A u)
               (hom_of_coeff A' u)).
  Qed.

  Theorem hom_of_matscale : ∀ (a : F) (A : Matrix m n F),
    hom_of (a *: A) == a *: hom_of A.
  Proof.
    intros a A u.
    transitivity (lincomb (basis_vec BV) (a *: hom_of_coeff A u)).
    - apply (lincomb_proper m (basis_vec BV)
               (hom_of_coeff (a *: A) u) (a *: hom_of_coeff A u)).
      intros i.
      exact (hom_of_coeff_matscale a A u i).
    - exact (lincomb_scale m a (basis_vec BV) (hom_of_coeff A u)).
  Qed.


  Local Notation HomUV := (LinearMap F U V).

  Let VHom : VectorSpace F HomUV := @Hom_VectorSpace F U V FF VU VV.
  Let VMat : VectorSpace F (Matrix m n F) :=
    @PW_VectorSpace (Fin.t m) F (Fin.t n → F) FF
      (@PW_VectorSpace (Fin.t n) F F FF (@Field_VectorSpace F FF)).

  Definition mat_of_lm : LinearMap F HomUV (Matrix m n F) :=
    @Build_LinearMap F HomUV (Matrix m n F) FF VHom VMat
      mat_of mat_of_proper mat_of_add mat_of_scale.

  Definition hom_of_lm : LinearMap F (Matrix m n F) HomUV :=
    @Build_LinearMap F (Matrix m n F) HomUV FF VMat VHom
      hom_of hom_of_matproper hom_of_matadd hom_of_matscale.

  Definition vsiso_matrix : @VSIso F HomUV (Matrix m n F) FF VHom VMat :=
    @Build_VSIso F HomUV (Matrix m n F) FF VHom VMat
      mat_of_lm hom_of_lm hom_of_mat_of mat_of_hom_of.

End MatRep.


Section MatRepComp.
  Context {F U V W : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}
          {VW : VectorSpace F W}.
  Context {n m p : nat} (BU : Basis n F U) (BV : Basis m F V)
          (BW : Basis p F W).

  Theorem mat_of_comp : ∀ (g : LinearMap F V W) (f : LinearMap F U V),
    mat_of BU BW (lm_comp g f)
      == mat_mul (mat_of BV BW g) (mat_of BU BV f).
  Proof.
    intros g f i j.
    exact (mat_of_coords BV BW g (f (basis_vec BU j)) i).
  Qed.

End MatRepComp.
