Section AbGroupTheory.
  Context {G : Type} `{AbGroup G}.

  Theorem add_swap_r : ∀ a b c : G, (a + b) + c == (a + c) + b.
  Proof.
    intros a b c.
    calc ((a + b) + c)
      == (a + (b + c)) { assoc }
      == (a + (c + b)) { (comm b c) }
      == ((a + c) + b) { assoc }.
  Qed.

  Theorem add_swap_l : ∀ a b c : G, a + (b + c) == b + (a + c).
  Proof.
    intros a b c.
    calc (a + (b + c))
      == ((a + b) + c) { assoc }
      == ((b + a) + c) { (comm a b) }
      == (b + (a + c)) { assoc }.
  Qed.

  Theorem add_interchange : ∀ a b c d : G,
    (a + b) + (c + d) == (a + c) + (b + d).
  Proof.
    intros a b c d.
    calc ((a + b) + (c + d))
      == (a + (b + (c + d))) { assoc }
      == (a + (c + (b + d))) { (add_swap_l b c d) }
      == ((a + c) + (b + d)) { assoc }.
  Qed.

  Theorem add_rotate : ∀ a b c d : G,
    (a + b) + (c + d) == (a + d) + (b + c).
  Proof.
    intros a b c d.
    calc ((a + b) + (c + d))
      == ((a + b) + (d + c)) { (comm c d) }
      == ((a + d) + (b + c)) { (add_interchange a b d c) }.
  Qed.

  Theorem neg_add : ∀ a b : G, - (a + b) == - a + - b.
  Proof.
    intros a b.
    calc (- (a + b))
      == (- b + - a) { (neg_add_flip a b) }
      == (- a + - b) { (comm (- b) (- a)) }.
  Qed.

  Theorem add_sub_cancel : ∀ a b : G, a + (b + - a) == b.
  Proof.
    intros a b.
    calc (a + (b + - a))
      == (a + (- a + b)) { (comm b (- a)) }
      == ((a + - a) + b) { assoc }
      == (0 + b) { invr }
      == b { idl }.
  Qed.

  Theorem sub_pair : ∀ a b c d : G,
    (a + b) + - (c + d) == (a + - c) + (b + - d).
  Proof.
    intros a b c d.
    calc ((a + b) + - (c + d))
      == ((a + b) + (- c + - d)) { (neg_add c d) }
      == ((a + - c) + (b + - d)) { (add_interchange a b (- c) (- d)) }.
  Qed.

  Theorem sub_add_cancel_r : ∀ p q x : G, (p + - (q + x)) + x == p + - q.
  Proof.
    intros p q x.
    calc ((p + - (q + x)) + x)
      == ((p + (- q + - x)) + x) { (neg_add q x) }
      == (((p + - q) + - x) + x) { (assoc p (- q) (- x)) }
      == ((p + - q) + (- x + x)) { (assoc (p + - q) (- x) x) }
      == ((p + - q) + 0) { invl }
      == (p + - q) { idr }.
  Qed.

  Theorem neg_add_assoc : ∀ a b c : G, a + - (b + c) == (a + - b) + - c.
  Proof.
    intros a b c.
    calc (a + - (b + c))
      == (a + (- b + - c)) { (neg_add b c) }
      == ((a + - b) + - c) { (assoc a (- b) (- c)) }.
  Qed.

  Theorem sub_common_cancel : ∀ p q x y : G,
    (p + - (q + x)) + - (p + - (q + y)) == y + - x.
  Proof.
    intros p q x y.
    calc ((p + - (q + x)) + - (p + - (q + y)))
      ==[negp] ((p + - (q + x)) + (- p + (q + y)))
      ==[interL] (- (q + x) + (q + y))
      ==[interR] (- x + y)
      ==[swap] (y + - x).
    [negp]: now rewrite (neg_add p (- (q + y))), (neg_involutive (q + y)).
    [interL]: now rewrite (add_interchange p (- (q + x)) (- p) (q + y)), invr, idl.
    [interR]: now rewrite (neg_add q x), (add_interchange (- q) (- x) q y), invl, idl.
    [swap]: apply comm.
  Qed.

  Theorem add_neg_cancel_l : ∀ a b : G, (a + b) + - a == b.
  Proof.
    intros a b.
    calc ((a + b) + - a)
      == ((b + a) + - a) { (comm a b) }
      == (b + (a + - a)) { assoc }
      == (b + 0) { invr }
      == b { idr }.
  Qed.

  Theorem sub_mid_cancel : ∀ p q x y : G,
    (p + - (q + x)) + (x + - y) == p + - (q + y).
  Proof.
    intros p q x y.
    calc ((p + - (q + x)) + (x + - y))
      ==[expand] ((p + (- q + - x)) + (x + - y))
      ==[assoc4] ((p + - q) + ((- x + x) + - y))
      ==[cancel] ((p + - q) + - y)
      ==[fold] (p + - (q + y)).
    [expand]: now rewrite (neg_add q x).
    [assoc4]: now rewrite (assoc (- x) x (- y)),
                <- (assoc (p + - q) (- x) (x + - y)),
                (assoc p (- q) (- x)).
    [cancel]: now rewrite invl, idl.
    [fold]: now rewrite (assoc p (- q) (- y)), <- (neg_add q y).
  Qed.

  Theorem four_term_cancel : ∀ a b c d : G,
    (a + b) + (c + d) + - (a + (b + c)) == d.
  Proof.
    intros a b c d.
    calc (((a + b) + (c + d)) + - (a + (b + c)))
      == ((a + (b + (c + d))) + - (a + (b + c))) { (assoc a b (c + d)) }
      == ((a + ((b + c) + d)) + - (a + (b + c))) { (assoc b c d) }
      == (((a + (b + c)) + d) + - (a + (b + c))) { (assoc a (b + c) d) }
      == d { (add_neg_cancel_l (a + (b + c)) d) }.
  Qed.

End AbGroupTheory.
