Section Determinant.
  Context {F : Type} {FF : Field F}.


  Fixpoint sign_at {n : nat} (i : Fin.t n) : F :=
    match i with
    | Fin.F1 ⇒ 1
    | Fin.FS j ⇒ - sign_at j
    end.

  Lemma sign_at_F1 : ∀ n : nat, @sign_at (S n) Fin.F1 == 1.
  Proof.
    intros n.
    reflexivity.
  Qed.

  Lemma sign_at_FS : ∀ (n : nat) (i : Fin.t n),
    sign_at (Fin.FS i) == - sign_at i.
  Proof.
    intros n i.
    reflexivity.
  Qed.

  Lemma sign_at_mul_self : ∀ (n : nat) (i : Fin.t n),
    sign_at i × sign_at i == 1.
  Proof.
    intros n i; induction i as [m | m j IH].
    - exact (ring_mul_1l 1).
    - rewrite (sign_at_FS m j), (mul_neg_neg (sign_at j) (sign_at j)).
      exact IH.
  Qed.


  Definition minor {p : nat} (A : Matrix (S p) (S p) F) (i : Fin.t (S p))
      : Matrix p p F :=
    fun r c ⇒ A (skip i r) (Fin.FS c).

  Lemma minor_def : ∀ (p : nat) (A : Matrix (S p) (S p) F)
      (i : Fin.t (S p)) (r c : Fin.t p),
    minor A i r c == A (skip i r) (Fin.FS c).
  Proof.
    intros p A i r c.
    reflexivity.
  Qed.

  Lemma minor_proper : ∀ (p : nat) (A B : Matrix (S p) (S p) F),
    A == B → ∀ i : Fin.t (S p), minor A i == minor B i.
  Proof.
    intros p A B HAB i r c.
    exact (HAB (skip i r) (Fin.FS c)).
  Qed.


  Fixpoint detM {n : nat} (A : Matrix n n F) {struct n} : F :=
    match n as n0 return Matrix n0 n0 F → F with
    | O ⇒ fun _ ⇒ 1
    | S m ⇒ fun B ⇒
        bigopF (fun i : Fin.t (S m) ⇒
          sign_at i × (B i Fin.F1 × detM (minor B i)))
    end A.

  Lemma detM_0 : ∀ A : Matrix O O F, detM A == 1.
  Proof.
    intros A.
    reflexivity.
  Qed.

  Lemma detM_S : ∀ (p : nat) (A : Matrix (S p) (S p) F),
    detM A == bigopF (fun i : Fin.t (S p) ⇒
      sign_at i × (A i Fin.F1 × detM (minor A i))).
  Proof.
    intros p A.
    reflexivity.
  Qed.


  Lemma detM_proper : ∀ n : nat, Proper (equ ==> equ) (@detM n).
  Proof.
    intros n; induction n as [| m IH].
    - intros A B HAB.
      reflexivity.
    - intros A B HAB.
      rewrite (detM_S m A), (detM_S m B).
      apply bigopF_ext. intros i.
      exact (ring_mul_proper (sign_at i) (sign_at i)
               (reflexivity (sign_at i))
               (A i Fin.F1 × detM (minor A i))
               (B i Fin.F1 × detM (minor B i))
               (ring_mul_proper (A i Fin.F1) (B i Fin.F1) (HAB i Fin.F1)
                  (detM (minor A i)) (detM (minor B i))
                  (IH (minor A i) (minor B i)
                     (minor_proper m A B HAB i)))).
  Qed.


  Lemma mul_0_mid : ∀ a b : F, a × (0 × b) == 0.
  Proof.
    intros a b.
    rewrite (mul_0_l b).
    exact (mul_0_r a).
  Qed.

  Lemma mul_mid_distr : ∀ s b c x : F,
    s × ((b + c) × x) == s × (b × x) + s × (c × x).
  Proof.
    intros s b c x.
    rewrite (ring_distr b c x).
    exact (ring_distl s (b × x) (c × x)).
  Qed.

  Lemma mul_mid_pull : ∀ s a b x : F,
    s × ((a × b) × x) == a × (s × (b × x)).
  Proof.
    intros s a b x.
    rewrite (ring_mul_assoc a b x).
    exact (mul_comm_swap s a (b × x)).
  Qed.

  Lemma det_term_col0 : ∀ (p : nat) (A : Matrix (S p) (S p) F)
      (i : Fin.t (S p)),
    A i Fin.F1 == 0 →
    sign_at i × (A i Fin.F1 × detM (minor A i)) == 0.
  Proof.
    intros p A i HAi.
    rewrite HAi.
    exact (mul_0_mid (sign_at i) (detM (minor A i))).
  Qed.


  Lemma mat_id_diag : ∀ (n : nat) (i : Fin.t n),
    @mat_id F FF n i i == 1.
  Proof.
    intros n i.
    exact (e_at_eq n i i eq_refl).
  Qed.

  Lemma mat_id_col_0 : ∀ (n : nat) (i : Fin.t n),
    @mat_id F FF (S n) (Fin.FS i) Fin.F1 == 0.
  Proof.
    intros n i.
    exact (e_at_neq (S n) (Fin.FS i) Fin.F1 (fin_FS_neq_F1 n i)).
  Qed.

  Lemma mat_id_minor_F1 : ∀ p : nat,
    minor (@mat_id F FF (S p)) Fin.F1 == @mat_id F FF p.
  Proof.
    intros p r c.
    rewrite (minor_def p (@mat_id F FF (S p)) Fin.F1 r c), (skip_F1 p r).
    exact (e_FS p r c).
  Qed.

  Theorem det_id : ∀ n : nat, detM (@mat_id F FF n) == 1.
  Proof.
    intros n; induction n as [| m IH].
    - reflexivity.
    - transitivity
        (sign_at (Fin.F1 : Fin.t (S m))
           × (@mat_id F FF (S m) Fin.F1 Fin.F1
               × detM (minor (@mat_id F FF (S m)) Fin.F1))
         + bigopF (fun i : Fin.t m ⇒
             sign_at (Fin.FS i)
               × (@mat_id F FF (S m) (Fin.FS i) Fin.F1
                   × detM (minor (@mat_id F FF (S m)) (Fin.FS i))))).
      + reflexivity.
      + rewrite (sign_at_F1 m),
          (mat_id_diag (S m) Fin.F1),
          (detM_proper m (minor (@mat_id F FF (S m)) Fin.F1)
             (@mat_id F FF m) (mat_id_minor_F1 m)),
          IH,
          (bigopF_ext m
             (fun i : Fin.t m ⇒
                sign_at (Fin.FS i)
                  × (@mat_id F FF (S m) (Fin.FS i) Fin.F1
                      × detM (minor (@mat_id F FF (S m)) (Fin.FS i))))
             (fun _ : Fin.t m ⇒ (0 : F))
             (fun i : Fin.t m ⇒
                det_term_col0 m (@mat_id F FF (S m)) (Fin.FS i)
                  (mat_id_col_0 m i))),
          (bigopF_zero m),
          (ring_mul_1l ((1 : F) × 1)),
          (ring_mul_1l (1 : F)).
        exact (mon_idr (1 : F)).
  Qed.


  Definition upd_col1 {m p : nat} (A : Matrix m (S p) F)
      (c : Fin.t m → F) : Matrix m (S p) F :=
    fun i j ⇒
      Fin.caseS' j (fun _ ⇒ F) (c i) (fun j' : Fin.t p ⇒ A i (Fin.FS j')).

  Lemma upd_col1_F1 : ∀ (m p : nat) (A : Matrix m (S p) F)
      (c : Fin.t m → F) (i : Fin.t m),
    upd_col1 A c i Fin.F1 == c i.
  Proof.
    intros m p A c i.
    reflexivity.
  Qed.

  Lemma upd_col1_FS : ∀ (m p : nat) (A : Matrix m (S p) F)
      (c : Fin.t m → F) (i : Fin.t m) (j : Fin.t p),
    upd_col1 A c i (Fin.FS j) == A i (Fin.FS j).
  Proof.
    intros m p A c i j.
    reflexivity.
  Qed.

  Lemma minor_upd_col1 : ∀ (p : nat) (A : Matrix (S p) (S p) F)
      (c : Fin.t (S p) → F) (i : Fin.t (S p)),
    minor (upd_col1 A c) i == minor A i.
  Proof.
    intros p A c i r s.
    reflexivity.
  Qed.

  Lemma det_col1_term : ∀ (p : nat) (A : Matrix (S p) (S p) F)
      (c : Fin.t (S p) → F) (i : Fin.t (S p)),
    sign_at i × (upd_col1 A c i Fin.F1 × detM (minor (upd_col1 A c) i))
      == sign_at i × (c i × detM (minor A i)).
  Proof.
    intros p A c i.
    rewrite (upd_col1_F1 (S p) p A c i),
      (detM_proper p (minor (upd_col1 A c) i) (minor A i)
         (minor_upd_col1 p A c i)).
    reflexivity.
  Qed.

  Lemma det_col1_expand : ∀ (p : nat) (A : Matrix (S p) (S p) F)
      (c : Fin.t (S p) → F),
    detM (upd_col1 A c)
      == bigopF (fun i : Fin.t (S p) ⇒
           sign_at i × (c i × detM (minor A i))).
  Proof.
    intros p A c.
    rewrite (detM_S p (upd_col1 A c)).
    apply bigopF_ext. intros i.
    exact (det_col1_term p A c i).
  Qed.

  Theorem det_col1_add : ∀ (p : nat) (A : Matrix (S p) (S p) F)
      (c d : Fin.t (S p) → F),
    detM (upd_col1 A (fun i ⇒ c i + d i))
      == detM (upd_col1 A c) + detM (upd_col1 A d).
  Proof.
    intros p A c d.
    transitivity (bigopF (fun i : Fin.t (S p) ⇒
      sign_at i × ((c i + d i) × detM (minor A i)))).
    - exact (det_col1_expand p A (fun i ⇒ c i + d i)).
    - rewrite (bigopF_ext (S p)
        (fun i : Fin.t (S p) ⇒
           sign_at i × ((c i + d i) × detM (minor A i)))
        (fun i : Fin.t (S p) ⇒
           sign_at i × (c i × detM (minor A i))
           + sign_at i × (d i × detM (minor A i)))
        (fun i : Fin.t (S p) ⇒
           mul_mid_distr (sign_at i) (c i) (d i) (detM (minor A i)))),
        (bigopF_add (S p)
          (fun i : Fin.t (S p) ⇒ sign_at i × (c i × detM (minor A i)))
          (fun i : Fin.t (S p) ⇒ sign_at i × (d i × detM (minor A i)))),
        (det_col1_expand p A c), (det_col1_expand p A d).
      reflexivity.
  Qed.

  Theorem det_col1_scale : ∀ (p : nat) (A : Matrix (S p) (S p) F)
      (a : F) (c : Fin.t (S p) → F),
    detM (upd_col1 A (fun i ⇒ a × c i)) == a × detM (upd_col1 A c).
  Proof.
    intros p A a c.
    transitivity (bigopF (fun i : Fin.t (S p) ⇒
      sign_at i × ((a × c i) × detM (minor A i)))).
    - exact (det_col1_expand p A (fun i ⇒ a × c i)).
    - rewrite (bigopF_ext (S p)
        (fun i : Fin.t (S p) ⇒
           sign_at i × ((a × c i) × detM (minor A i)))
        (fun i : Fin.t (S p) ⇒
           a × (sign_at i × (c i × detM (minor A i))))
        (fun i : Fin.t (S p) ⇒
           mul_mid_pull (sign_at i) a (c i) (detM (minor A i)))),
        (bigopF_mul_l (S p) a
          (fun i : Fin.t (S p) ⇒ sign_at i × (c i × detM (minor A i)))),
        (det_col1_expand p A c).
      reflexivity.
  Qed.


  Theorem det_col1_zero : ∀ (p : nat) (A : Matrix (S p) (S p) F),
    (∀ i : Fin.t (S p), A i Fin.F1 == 0) → detM A == 0.
  Proof.
    intros p A HA.
    rewrite (detM_S p A),
      (bigopF_ext (S p)
        (fun i : Fin.t (S p) ⇒
           sign_at i × (A i Fin.F1 × detM (minor A i)))
        (fun _ : Fin.t (S p) ⇒ (0 : F))
        (fun i : Fin.t (S p) ⇒ det_term_col0 p A i (HA i))).
    exact (bigopF_zero (S p)).
  Qed.

End Determinant.


Definition row2 {A : Type} (a b : A) : Fin.t 2 → A :=
  fun j ⇒ match j with
           | Fin.F1 ⇒ a
           | Fin.FS _ ⇒ b
           end.

Definition mat2 {A : Type} (a b c d : A) : Matrix 2 2 A :=
  fun i ⇒ match i with
           | Fin.F1 ⇒ row2 a b
           | Fin.FS _ ⇒ row2 c d
           end.

Example det2_Q :
  detM (mat2 (1#1)%Q (2#1)%Q (3#1)%Q (4#1)%Q) == ((-2)#1)%Q.
Proof.
  reflexivity.
Qed.
