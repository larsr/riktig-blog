Section StandardBasisIsBasis.
  Context {F : Type} {FF : Field F}.

  Lemma e_at_eq : ∀ (n : nat) (i j : Fin.t n), i = j → e i j == 1.
  Proof.
    intros n i j Hij.
    unfold e; destruct (dec (i = j)) as [_ | Hne].
    - reflexivity.
    - destruct (Hne Hij).
  Qed.

  Lemma e_at_neq : ∀ (n : nat) (i j : Fin.t n), i ≠ j → e i j == 0.
  Proof.
    intros n i j Hij.
    unfold e; destruct (dec (i = j)) as [Heq | _].
    - destruct (Hij Heq).
    - reflexivity.
  Qed.

  Lemma e_sym : ∀ (n : nat) (i j : Fin.t n), e i j == e j i.
  Proof.
    intros n i j.
    destruct (Fin.eq_dec i j) as [Heq | Hne].
    - rewrite (e_at_eq n i j Heq), (e_at_eq n j i (eq_sym Heq)).
      reflexivity.
    - rewrite (e_at_neq n i j Hne),
        (e_at_neq n j i (fun Heq ⇒ Hne (eq_sym Heq))).
      reflexivity.
  Qed.

  Lemma e_FS : ∀ (n : nat) (i j : Fin.t n),
    e (Fin.FS i) (Fin.FS j) == e i j.
  Proof.
    intros n i j.
    unfold e; destruct (dec (Fin.FS i = Fin.FS j)) as [Heq | Hne];
      destruct (dec (i = j)) as [Heq' | Hne'].
    - reflexivity.
    - destruct (Hne' (Fin.FS_inj i j Heq)).
    - destruct (Hne (f_equal Fin.FS Heq')).
    - reflexivity.
  Qed.

  Lemma mul_e_eq : ∀ (n : nat) (a : F) (i j : Fin.t n),
    i = j → a × e i j == a.
  Proof.
    intros n a i j Hij.
    rewrite (e_at_eq n i j Hij).
    exact (ring_mul_1r a).
  Qed.

  Lemma mul_e_neq : ∀ (n : nat) (a : F) (i j : Fin.t n),
    i ≠ j → a × e i j == 0.
  Proof.
    intros n a i j Hij.
    rewrite (e_at_neq n i j Hij).
    exact (mul_0_r a).
  Qed.

  Lemma mul_e_FS : ∀ (n : nat) (a : F) (i j : Fin.t n),
    a × e (Fin.FS i) (Fin.FS j) == a × e i j.
  Proof.
    intros n a i j.
    rewrite (e_FS n i j).
    reflexivity.
  Qed.

  Lemma lincomb_std_delta : ∀ (n : nat) (c : Fin.t n → F)
      (j : Fin.t n),
    bigopF (fun i ⇒ c i × e i j) == c j.
  Proof.
    intros n; induction n as [| n IH]; intros c j.
    - exact (Fin.case0
               (fun j0 ⇒ bigopF (fun i ⇒ c i × e i j0) == c j0) j).
    - apply (Fin.caseS' j
               (fun j0 ⇒ bigopF (fun i ⇒ c i × e i j0) == c j0)).
      + calc (bigopF (fun i ⇒ c i × e i Fin.F1))
          == (c Fin.F1 × @e F FF (S n) Fin.F1 Fin.F1
              + bigopF (fun i ⇒ c (Fin.FS i) × e (Fin.FS i) Fin.F1))
          == (c Fin.F1
              + bigopF (fun i ⇒ c (Fin.FS i) × e (Fin.FS i) Fin.F1))
          == (c Fin.F1 + bigopF (fun _ : Fin.t n ⇒ (0 : F)))
          == (c Fin.F1 + 0)
          == (c Fin.F1).
        × exact (bigopF_S n (fun i ⇒ c i × e i Fin.F1)).
        × rewrite (mul_e_eq (S n) (c Fin.F1) Fin.F1 Fin.F1 eq_refl).
          reflexivity.
        × apply mon_add_proper; [reflexivity |].
          apply bigopF_ext. intros i.
          exact (mul_e_neq (S n) (c (Fin.FS i)) (Fin.FS i)
                   Fin.F1 (fin_FS_neq_F1 n i)).
        × apply mon_add_proper; [reflexivity | exact (bigopF_zero n)].
        × exact (mon_idr (c Fin.F1)).
      + intros j'.
        calc (bigopF (fun i ⇒ c i × e i (Fin.FS j')))
          == (c Fin.F1 × @e F FF (S n) Fin.F1 (Fin.FS j')
              + bigopF (fun i ⇒ c (Fin.FS i) × e (Fin.FS i) (Fin.FS j')))
          == ((0 : F)
              + bigopF (fun i ⇒ c (Fin.FS i) × e (Fin.FS i) (Fin.FS j')))
          == ((0 : F) + bigopF (fun i ⇒ c (Fin.FS i) × e i j'))
          == ((0 : F) + c (Fin.FS j'))
          == (c (Fin.FS j')).
        × exact (bigopF_S n (fun i ⇒ c i × e i (Fin.FS j'))).
        × apply mon_add_proper; [| reflexivity].
          exact (mul_e_neq (S n) (c Fin.F1) Fin.F1 (Fin.FS j')
                   (fin_F1_neq_FS n j')).
        × apply mon_add_proper; [reflexivity |].
          apply bigopF_ext. intros i.
          exact (mul_e_FS n (c (Fin.FS i)) i j').
        × apply mon_add_proper; [reflexivity |].
          exact (IH (fun i ⇒ c (Fin.FS i)) j').
        × exact (mon_idl (c (Fin.FS j'))).
  Qed.

  Lemma lincomb_std : ∀ (n : nat) (c : Fin.t n → F),
    lincomb e c == c.
  Proof.
    intros n c j.
    transitivity (bigopF (fun i ⇒ (c i *: e i) j)).
    - exact (bigopF_app n (fun i ⇒ c i *: e i) j).
    - exact (lincomb_std_delta n c j).
  Qed.

  Definition FinVec_Basis (n : nat) : Basis n F (Fin.t n → F) :=
    @Build_Basis n F (Fin.t n → F) FF
      (@PW_VectorSpace (Fin.t n) F F FF (@Field_VectorSpace F FF))
      (@e F FF n)
      (@lm_id F (Fin.t n → F) FF
         (@PW_VectorSpace (Fin.t n) F F FF (@Field_VectorSpace F FF)))
      (fun v ⇒ lincomb_std n v)
      (fun c ⇒ lincomb_std n c).

End StandardBasisIsBasis.

Example finvec_basis_check (F : Type) (FF : Field F) (n : nat) :
  Basis n F (Fin.t n → F) := FinVec_Basis n.
