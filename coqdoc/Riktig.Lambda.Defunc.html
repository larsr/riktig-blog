Library Riktig.Lambda.Defunc


Require Import Riktig.Lambda.Syntax.
From Coq Require Import List.
Import ListNotations.
Require Import Riktig.Lambda.Conv. From Riktig Require Import Lib.Calc.
From Coq Require Import Morphisms Setoid.


Inductive instr : Type :=
  | IArg : tm → instr
  | ILam : instr.

Fixpoint build (base : tm) (is : list instr) (k : tm → tm) : tm :=
  match is with
  | [] ⇒ k base
  | IArg t :: is' ⇒ build base is' (fun r ⇒ k (App r t))
  | ILam :: is' ⇒ build base is' (fun r ⇒ k (Abs r))
  end.


Inductive cont : Type :=
  | Kid : cont
  | Kapp : tm → cont → cont
  | Kabs : cont → cont.

Fixpoint apply_clo (k : cont) (v : tm) : tm :=
  match k with
  | Kid ⇒ v
  | Kapp t k' ⇒ apply_clo k' (App v t)
  | Kabs k' ⇒ apply_clo k' (Abs v)
  end.

Fixpoint build_d (base : tm) (is : list instr) (k : cont) : tm :=
  match is with
  | [] ⇒ apply_clo k base
  | IArg t :: is' ⇒ build_d base is' (Kapp t k)
  | ILam :: is' ⇒ build_d base is' (Kabs k)
  end.


Fixpoint denote (k : cont) : tm → tm :=
  match k with
  | Kid ⇒ fun v ⇒ v
  | Kapp t k' ⇒ fun r ⇒ denote k' (App r t)
  | Kabs k' ⇒ fun r ⇒ denote k' (Abs r)
  end.

Lemma apply_denote : ∀ k v, apply_clo k v = denote k v.
Proof.
  induction k as [| t k' IH | k' IH]; intros v; simpl.
  - reflexivity.
  - rewrite IH. reflexivity.
  - rewrite IH. reflexivity.
Qed.

Lemma build_defunc_correct : ∀ is base k,
  build_d base is k = build base is (denote k).
Proof.
  induction is as [| i is' IH]; intros base k; simpl.
  - apply apply_denote.
  - destruct i as [t |].
    + apply IH.
    + apply IH.
Qed.

Corollary build_defunc_correct_id : ∀ base is,
  build_d base is Kid = build base is (fun x ⇒ x).
Proof.
  intros base is. exact (build_defunc_correct is base Kid).
Qed.

Corollary build_defunc_conv : ∀ base is,
  build_d base is Kid == build base is (fun x ⇒ x).
Proof.
  intros base is. rewrite build_defunc_correct_id. reflexivity.
Qed.


Example build_worked :
  build_d (Var 0) [IArg (Var 1); ILam] Kid = App (Abs (Var 0)) (Var 1).
Proof.
  calc (build_d (Var 0) [IArg (Var 1); ILam] Kid)
    == (build (Var 0) [IArg (Var 1); ILam] (fun x ⇒ x))
    == (App (Abs (Var 0)) (Var 1)).
  - apply build_defunc_correct_id.
  - reflexivity.
Qed.

Example build_d_computes :
  build_d (Var 0) [IArg (Var 1); ILam] Kid = App (Abs (Var 0)) (Var 1).
Proof. reflexivity. Qed.

Example build_worked_eval :
  eval 10 (build_d (Var 0) [IArg (Var 1); ILam] Kid) = Var 1.
Proof. reflexivity. Qed.
