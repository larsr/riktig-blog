Local Open Scope Z_scope.

#[local] Instance Z_lt_decidable (x y : Z) : Decidable (x < y) := Z_lt_dec x y.
#[local] Instance Z_le_decidable (x y : Z) : Decidable (x ≤ y) := Z_le_dec x y.

Definition range_measure (p : Z × Z) : nat := Z.to_nat (snd p - fst p + 1).



Definition partition_l1 (base lo hi : Z) : MemT Z Z Z :=
  do! pivot <- read_arr base (ret_t hi);
  do! iv <- stack_alloc_m;
  do! jv <- stack_alloc_m;
  write_m iv lo >>
  write_m jv lo >>
  while_m (lt_ Z.ltb (read_m jv) (ret_t hi)) (
    when_m (leq_ Z.leb (read_arr base (read_m jv)) (ret_t pivot)) (
      swap_cells base (read_m iv) (read_m jv) >>
      do! i <- read_m iv;
      write_m iv (i + 1)) >>
    do! j <- read_m jv;
    write_m jv (j + 1)) >>
  swap_cells base (read_m iv) (ret_t hi) >>
  read_m iv.


Definition qsort_l1 (base : Z) : Z → Z → MemT Z Z unit.
  apply (uncurry ∘ (Fix (well_founded_ltof (Z × Z) range_measure)
                        (fun _ ⇒ MemT Z Z unit))).
  refine
    (fun p qsort ⇒
       match p as x return p = x → MemT Z Z unit with
       | (lo, hi) ⇒
           fun e ⇒
             when_m (lt_ Z.ltb (ret_t lo) (ret_t hi)) (
               do! pv <- partition_l1 base lo hi;
               match dec (lo ≤ pv ≤ hi) with
               | left Hin ⇒ qsort (lo, pv - 1) _ >> qsort (pv + 1, hi) _
               | right _ ⇒ fail_t
               end)
       end eq_refl);
    (subst p; unfold ltof, range_measure; simpl; abstract lia).
Defined.




Definition qsort_l2_eval (base : Z) : Z → Z → MemT Z Z unit.
  apply (uncurry ∘ (Fix (well_founded_ltof (Z × Z) range_measure)
                        (fun _ ⇒ MemT Z Z unit))).
  refine
    (fun p qsort ⇒
       match p as x return p = x → MemT Z Z unit with
       | (lo, hi) ⇒
           fun e ⇒
             when_m (lt_ Z.ltb (ret_t lo) (ret_t hi)) (
               do! pv <- partition_l1 base lo hi;
               match dec (lo ≤ pv ≤ hi) with
               | left Hin ⇒
                   qsort (lo, eval_expr Z.add Z.sub
                                (ESub (Lit pv) (Lit 1))) _ >>
                   qsort (eval_expr Z.add Z.sub
                            (EAdd (Lit pv) (Lit 1)), hi) _
               | right _ ⇒ fail_t
               end)
       end eq_refl);
    (subst p; unfold ltof, range_measure; simpl; abstract lia).
Defined.


Definition qsort_l2_compile (base : Z) : Z → Z → MemT Z Z unit.
  apply (uncurry ∘ (Fix (well_founded_ltof (Z × Z) range_measure)
                        (fun _ ⇒ MemT Z Z unit))).
  refine
    (fun p qsort ⇒
       match p as x return p = x → MemT Z Z unit with
       | (lo, hi) ⇒
           fun e ⇒
             when_m (lt_ Z.ltb (ret_t lo) (ret_t hi)) (
               do! pv <- partition_l1 base lo hi;
               do! pm1 <- compile_expr Z.add Z.sub (ESub (Lit pv) (Lit 1));
               do! pp1 <- compile_expr Z.add Z.sub (EAdd (Lit pv) (Lit 1));
               match dec (lo - 1 ≤ pm1 < hi ∧ lo < pp1 ≤ hi + 1) with
               | left Hin ⇒ qsort (lo, pm1) _ >> qsort (pp1, hi) _
               | right _ ⇒ fail_t
               end)
       end eq_refl);
    (subst p; unfold ltof, range_measure; simpl; abstract lia).
Defined.



Definition partition_l3 : MemT Z Z Z :=
  with_frame (
    stack_alloc_n 2 >>
    do! fp <- get_sp_m;
    do! hi <- read_slot_z fp 4;
    do! lo <- read_slot_z fp 5;
    do! base <- read_slot_addr_z fp 6;
    do! iv <- slot_addr_z fp 2;
    do! jv <- slot_addr_z fp 1;
    do! pivot <- read_arr base (ret_t hi);
    write_m iv lo >>
    write_m jv lo >>
    while_m (lt_ Z.ltb (read_m jv) (ret_t hi)) (
      when_m (leq_ Z.leb (read_arr base (read_m jv)) (ret_t pivot)) (
        swap_cells base (read_m iv) (read_m jv) >>
        do! i <- read_m iv;
        write_m iv (i + 1)) >>
      do! j <- read_m jv;
      write_m jv (j + 1)) >>
    swap_cells base (read_m iv) (ret_t hi) >>
    read_m iv).


Definition qsort_l3 (base : Z) : Z → Z → MemT Z Z unit.
  apply (uncurry ∘ (Fix (well_founded_ltof (Z × Z) range_measure)
                        (fun _ ⇒ MemT Z Z unit))).
  refine
    (fun p qsort ⇒
       match p as x return p = x → MemT Z Z unit with
       | (lo, hi) ⇒
           fun e ⇒
             call_frame (push_addr base >> push_val lo >> push_val hi)
               (fun _ ⇒
                  with_frame (
                    do! fp <- get_sp_m;
                    do! hi' <- read_slot_z fp 2;
                    do! lo' <- read_slot_z fp 3;
                    do! base' <- read_slot_addr_z fp 4;
                    when_m (lt_ Z.ltb (ret_t lo') (ret_t hi')) (
                      do! pv <- call_frame
                                  (push_addr base' >>
                                   push_val lo' >> push_val hi')
                                  (fun _ ⇒ partition_l3);
                      match dec (lo ≤ pv ≤ hi) with
                      | left Hin ⇒
                          qsort (lo, pv - 1) _ >> qsort (pv + 1, hi) _
                      | right _ ⇒ fail_t
                      end)))
       end eq_refl);
    (subst p; unfold ltof, range_measure; simpl; abstract lia).
Defined.

Definition call_qsort_l1 (base n : Z) : MemT Z Z unit :=
  qsort_l1 base 0 (n - 1).

Definition call_qsort_l2_eval (base n : Z) : MemT Z Z unit :=
  qsort_l2_eval base 0 (eval_expr Z.add Z.sub (ESub (Lit n) (Lit 1))).

Definition call_qsort_l2_compile (base n : Z) : MemT Z Z unit :=
  do! nm1 <- compile_expr Z.add Z.sub (ESub (Lit n) (Lit 1));
  qsort_l2_compile base 0 nm1.

Definition call_qsort_l3 (base n : Z) : MemT Z Z unit :=
  qsort_l3 base 0 (n - 1).




Definition partition_demo : MemT Z Z (Z × Z) :=
  do! base <- list_to_mem Z.of_nat [3; 1; 2];
  do! pv <- partition_l1 base 0 2;
  ret_t (base, pv).

Example partition_l1_pivot :
  (let '(r, m) := run_z 3000 partition_demo 64 in
   (option_map snd r, mem_to_list m (option_map fst r) 3))
  = (Some 1, Some [1; 2; 3]).
Proof. vm_compute. reflexivity. Qed.
