Section BilinearPartial.
  Context {F U V W : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}
          {VW : VectorSpace F W}.
  Context (b : BilinearMap F U V W).

  Lemma bm_appl_proper : ∀ u : U,
    Proper (equ ==> equ) (fun v : V ⇒ b u v).
  Proof.
    intros u v v' Hv.
    exact (bm_proper b u u (reflexivity u) v v' Hv).
  Qed.

  Definition bm_appl (u : U) : LinearMap F V W :=
    @Build_LinearMap F V W FF VV VW (fun v : V ⇒ b u v)
      (bm_appl_proper u)
      (bm_add_r b u)
      (fun (a : F) (v : V) ⇒ bm_scale_r b a u v).

  Lemma bm_appr_proper : ∀ v : V,
    Proper (equ ==> equ) (fun u : U ⇒ b u v).
  Proof.
    intros v u u' Hu.
    exact (bm_proper b u u' Hu v v (reflexivity v)).
  Qed.

  Definition bm_appr (v : V) : LinearMap F U W :=
    @Build_LinearMap F U W FF VU VW (fun u : U ⇒ b u v)
      (bm_appr_proper v)
      (fun u u' : U ⇒ bm_add_l b u u' v)
      (fun (a : F) (u : U) ⇒ bm_scale_l b a u v).

  Theorem bm_zero_l : ∀ v : V, b 0 v == 0.
  Proof.
    intros v.
    exact (lm_zero (bm_appr v)).
  Qed.

  Theorem bm_zero_r : ∀ u : U, b u 0 == 0.
  Proof.
    intros u.
    exact (lm_zero (bm_appl u)).
  Qed.

End BilinearPartial.


Lemma upd_at {V : Type} {EV : EqRel V} (RV : Reflexive (@equ V EV)) :
  ∀ (n : nat) (vs : Fin.t n → V) (i : Fin.t n) (x : V),
    upd vs i x i == x.
Proof.
  intros n vs i x.
  unfold upd; destruct (Fin.eq_dec i i) as [_ | Hne].
  - reflexivity.
  - destruct (Hne eq_refl).
Qed.

Lemma upd_off {V : Type} {EV : EqRel V} (RV : Reflexive (@equ V EV)) :
  ∀ (n : nat) (vs : Fin.t n → V) (i j : Fin.t n) (x : V),
    i ≠ j → upd vs i x j == vs j.
Proof.
  intros n vs i j x Hij.
  unfold upd; destruct (Fin.eq_dec i j) as [Heq | _].
  - destruct (Hij Heq).
  - reflexivity.
Qed.


Section ZeroBilinear.
  Context {F U V W : Type} {FF : Field F}
          {VU : VectorSpace F U} {VV : VectorSpace F V}
          {VW : VectorSpace F W}.

  Lemma bm0_proper :
    Proper (equ ==> equ ==> equ) (fun (_ : U) (_ : V) ⇒ (0 : W)).
  Proof.
    intros u u' _ v v' _.
    reflexivity.
  Qed.

  Lemma bm0_add_l : ∀ (u u' : U) (v : V), (0 : W) == 0 + 0.
  Proof.
    intros u u' v.
    rewrite (idl (0 : W)).
    reflexivity.
  Qed.

  Lemma bm0_add_r : ∀ (u : U) (v v' : V), (0 : W) == 0 + 0.
  Proof.
    intros u v v'.
    rewrite (idl (0 : W)).
    reflexivity.
  Qed.

  Lemma bm0_scale : ∀ (a : F) (u : U) (v : V), (0 : W) == a *: 0.
  Proof.
    intros a u v.
    rewrite (scale_zero a).
    reflexivity.
  Qed.

  Example bm_zero_map : BilinearMap F U V W :=
    @Build_BilinearMap F U V W FF VU VV VW
      (fun (_ : U) (_ : V) ⇒ (0 : W))
      bm0_proper bm0_add_l bm0_scale bm0_add_r bm0_scale.

End ZeroBilinear.
