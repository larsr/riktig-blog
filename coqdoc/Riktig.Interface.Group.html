Library Riktig.Interface.Group

From Coq Require Export Morphisms.
From Riktig Require Export Ops Props Interface.Monoid.

Class Group G := {
  #[global] grp_monoid :: Monoid G;
  #[global] grp_neg :: HasNeg G;
  grp_invl : ∀ a : G, (- a) + a == 0;
  grp_invr : ∀ a : G, a + (- a) == 0;
  #[global] grp_neg_proper :: Proper (equ ==> equ) (@neg G _)
}.

#[global] Instance Group_Invl G (H : Group G) : Invl G equ add neg zero := grp_invl.
#[global] Instance Group_Invr G (H : Group G) : Invr G equ add neg zero := grp_invr.

#[global] Hint Mode Group ! : typeclass_instances.
