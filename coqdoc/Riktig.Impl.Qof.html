#[global] Instance Q_Le : HasLe Q := Qle.
#[global] Instance Q_Abs : HasAbs Q := Qabs.

Lemma Qle_total : ∀ a b : Q, (a ≤ b)%Q ∨ (b ≤ a)%Q.
Proof.
  intros a b.
  destruct (Qlt_le_dec a b) as [Hab | Hba].
  - left. exact (Qlt_le_weak a b Hab).
  - right. exact Hba.
Qed.

Lemma Qle_bool_false : ∀ a b : Q,
  Qle_bool a b = false → ¬ (a ≤ b)%Q.
Proof.
  intros a b E Hle.
  rewrite (proj2 (Qle_bool_iff a b) Hle) in E.
  discriminate E.
Qed.

Lemma Qle_stable : ∀ a b : Q, ¬ ¬ (a ≤ b)%Q → (a ≤ b)%Q.
Proof.
  intros a b Hnn.
  destruct (Qle_bool a b) eqn:E.
  - exact (proj1 (Qle_bool_iff a b) E).
  - destruct (Hnn (Qle_bool_false a b E)).
Qed.

Lemma Qplus_le_mono_l : ∀ a b c : Q, (a ≤ b)%Q → (c + a ≤ c + b)%Q.
Proof.
  intros a b c Hab.
  exact (proj2 (Qplus_le_r a b c) Hab).
Qed.

Lemma Qeq_le : ∀ x y : Q, (x == y)%Q → (x ≤ y)%Q.
Proof.
  intros x y Hxy.
  rewrite Hxy.
  apply Qle_refl.
Qed.

Lemma Qabs_zero_iff : ∀ a : Q, (Qabs a == 0)%Q ↔ (a == 0)%Q.
Proof.
  intros a.
  split.
  - intros Ha.
    destruct (proj1 (Qabs_Qle_condition a 0) (Qeq_le (Qabs a) 0 Ha))
      as [H1 H2].
    apply Qle_antisym.
    + exact H2.
    + exact H1.
  - intros Ha.
    rewrite Ha.
    exact (Qabs_pos 0 (Qle_refl 0)).
Qed.

Lemma Qmult_neg_neg : ∀ a b : Q, (- a × - b == a × b)%Q.
Proof.
  intros a b.
  exact (mul_neg_neg a b).
Qed.

Lemma Qsquare_nonneg : ∀ a : Q, (0 ≤ a × a)%Q.
Proof.
  intros a.
  destruct (Qlt_le_dec a 0) as [Ha | Ha].
  - rewrite <- (Qmult_neg_neg a a).
    apply Qmult_le_0_compat.
    + exact (Qopp_le_compat a 0 (Qlt_le_weak a 0 Ha)).
    + exact (Qopp_le_compat a 0 (Qlt_le_weak a 0 Ha)).
  - apply Qmult_le_0_compat.
    + exact Ha.
    + exact Ha.
Qed.

Lemma Qplus_zero_nonpos_l : ∀ x y : Q,
  (0 ≤ y)%Q → (x + y == 0)%Q → (x ≤ 0)%Q.
Proof.
  intros x y Hy Hxy.
  rewrite <- Hxy.
  apply (Qle_trans x (x + 0) (x + y)).
  - rewrite Qplus_0_r. apply Qle_refl.
  - exact (proj2 (Qplus_le_r 0 y x) Hy).
Qed.

Lemma Qsquare_sum_zero_l : ∀ a b : Q,
  (a × a + b × b == 0)%Q → (a × a == 0)%Q.
Proof.
  intros a b Hab.
  apply Qle_antisym.
  - exact (Qplus_zero_nonpos_l (a × a) (b × b) (Qsquare_nonneg b) Hab).
  - exact (Qsquare_nonneg a).
Qed.

Lemma Qfr_sos : ∀ a b : Q, (a × a + b × b == 0)%Q → (a == 0)%Q.
Proof.
  intros a b Hab.
  destruct (Qmult_integral a a (Qsquare_sum_zero_l a b Hab)) as [Ha | Ha].
  - exact Ha.
  - exact Ha.
Qed.

#[global] Instance Q_FormallyReal : FormallyReal Q := Qfr_sos.
