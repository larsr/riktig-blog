Inductive Perm : nat → Type :=
| Perm_O : Perm O
| Perm_S : ∀ {n : nat}, Fin.t (S n) → Perm n → Perm (S n).

Fixpoint perm_apply {n : nat} (sigma : Perm n) {struct sigma} : Fin.t n → Fin.t n :=
  match sigma in Perm n0 return Fin.t n0 → Fin.t n0 with
  | Perm_O ⇒ fun x ⇒ Fin.case0 (fun _ ⇒ Fin.t 0) x
  | @Perm_S m c p ⇒ fun x ⇒
      Fin.caseS' x (fun _ ⇒ Fin.t (S m)) c (fun x' ⇒ skip c (perm_apply p x'))
  end.

Lemma perm_apply_F1 : ∀ (n : nat) (c : Fin.t (S n)) (p : Perm n),
  perm_apply (Perm_S c p) Fin.F1 = c.
Proof.
  intros n c p. reflexivity.
Qed.

Lemma perm_apply_FS : ∀ (n : nat) (c : Fin.t (S n)) (p : Perm n) (x : Fin.t n),
  perm_apply (Perm_S c p) (Fin.FS x) = skip c (perm_apply p x).
Proof.
  intros n c p x. reflexivity.
Qed.


Section DetLeibniz.
  Context {F : Type} {FF : Field F}.


  Fixpoint perm_sign {n : nat} (sigma : Perm n) {struct sigma} : F :=
    match sigma with
    | Perm_O ⇒ 1
    | Perm_S c p ⇒ sign_at c × perm_sign p
    end.

  Lemma perm_sign_O : perm_sign Perm_O == 1.
  Proof. reflexivity. Qed.

  Lemma perm_sign_S : ∀ (n : nat) (c : Fin.t (S n)) (p : Perm n),
    perm_sign (Perm_S c p) == sign_at c × perm_sign p.
  Proof. intros n c p. reflexivity. Qed.



  Definition perm_prod {n : nat} (A : Matrix n n F) (sigma : Perm n) : F :=
    prodF (fun j ⇒ A (perm_apply sigma j) j).

  Lemma perm_prod_proper : ∀ (n : nat) (A B : Matrix n n F),
    A == B → ∀ sigma : Perm n, perm_prod A sigma == perm_prod B sigma.
  Proof.
    intros n A B HAB sigma.
    unfold perm_prod.
    apply prodF_ext. intros j.
    exact (HAB (perm_apply sigma j) j).
  Qed.

  Lemma perm_prod_O : ∀ A : Matrix O O F, perm_prod A Perm_O == 1.
  Proof.
    intros A.
    exact (prodF_0 (fun j : Fin.t 0 ⇒ A (perm_apply Perm_O j) j)).
  Qed.

  Lemma perm_prod_S : ∀ (m : nat) (A : Matrix (S m) (S m) F)
      (c : Fin.t (S m)) (p : Perm m),
    perm_prod A (Perm_S c p) == A c Fin.F1 × perm_prod (minor A c) p.
  Proof.
    intros m A c p.
    unfold perm_prod.
    rewrite (prodF_S m (fun j ⇒ A (perm_apply (Perm_S c p) j) j)),
      (perm_apply_F1 m c p).
    apply ring_mul_proper.
    - reflexivity.
    - apply prodF_ext. intros i.
      rewrite (perm_apply_FS m c p i).
      symmetry.
      exact (minor_def m A c (perm_apply p i) i).
  Qed.


  Fixpoint psum {n : nat} (h : Perm n → F) {struct n} : F :=
    match n as n0 return (Perm n0 → F) → F with
    | O ⇒ fun h0 ⇒ h0 Perm_O
    | S m ⇒ fun h0 ⇒
        bigopF (fun i : Fin.t (S m) ⇒ psum (fun p : Perm m ⇒ h0 (Perm_S i p)))
    end h.

  Lemma psum_O : ∀ h : Perm O → F, psum h == h Perm_O.
  Proof. intros h. reflexivity. Qed.

  Lemma psum_S : ∀ (m : nat) (h : Perm (S m) → F),
    psum h == bigopF (fun i : Fin.t (S m) ⇒
      psum (fun p : Perm m ⇒ h (Perm_S i p))).
  Proof. intros m h. reflexivity. Qed.

  Lemma psum_ext : ∀ (n : nat) (h1 h2 : Perm n → F),
    (∀ sigma, h1 sigma == h2 sigma) → psum h1 == psum h2.
  Proof.
    intros n; induction n as [| m IH]; intros h1 h2 Hh.
    - exact (Hh Perm_O).
    - rewrite (psum_S m h1), (psum_S m h2).
      apply bigopF_ext. intros i.
      apply IH. intros p.
      exact (Hh (Perm_S i p)).
  Qed.

  Theorem psum_scale : ∀ (n : nat) (c : F) (f : Perm n → F),
    psum (fun p ⇒ c × f p) == c × psum f.
  Proof.
    intros n; induction n as [| m IH]; intros c f.
    - reflexivity.
    - calc (psum (fun p : Perm (S m) ⇒ c × f p))
        == (bigopF (fun i : Fin.t (S m) ⇒
              psum (fun p : Perm m ⇒ c × f (Perm_S i p))))
        == (bigopF (fun i : Fin.t (S m) ⇒
              c × psum (fun p : Perm m ⇒ f (Perm_S i p))))
        == (c × bigopF (fun i : Fin.t (S m) ⇒
              psum (fun p : Perm m ⇒ f (Perm_S i p))))
        == (c × psum f).
      + exact (psum_S m (fun p ⇒ c × f p)).
      + apply bigopF_ext. intros i.
        exact (IH c (fun p ⇒ f (Perm_S i p))).
      + exact (bigopF_mul_l (S m) c
          (fun i : Fin.t (S m) ⇒ psum (fun p : Perm m ⇒ f (Perm_S i p)))).
      + apply ring_mul_proper; [reflexivity |].
        symmetry. exact (psum_S m f).
  Qed.


  Definition detM_leibniz {n : nat} (A : Matrix n n F) : F :=
    psum (fun sigma : Perm n ⇒ perm_sign sigma × perm_prod A sigma).

  #[global] Instance detM_leibniz_proper : ∀ n : nat, Proper (equ ==> equ) (@detM_leibniz n).
  Proof.
    intros n A B HAB.
    unfold detM_leibniz.
    apply psum_ext. intros sigma.
    apply ring_mul_proper.
    - reflexivity.
    - exact (perm_prod_proper n A B HAB sigma).
  Qed.

  Lemma mul_four_regroup : ∀ a b c d : F, (a × b) × (c × d) == (a × c) × (b × d).
  Proof.
    intros a b c d.
    calc ((a × b) × (c × d))
      == (a × (b × (c × d))) { (ring_mul_assoc a b (c × d)) }
      == (a × ((b × c) × d)) { (ring_mul_assoc b c d) }
      == (a × ((c × b) × d)) { (field_mul_comm b c) }
      == (a × (c × (b × d))) { (ring_mul_assoc c b d) }
      == ((a × c) × (b × d)) { (ring_mul_assoc a c (b × d)) }.
  Qed.

  Theorem detM_leibniz_eq_detM : ∀ (n : nat) (A : Matrix n n F),
    detM_leibniz A == detM A.
  Proof.
    intros n; induction n as [| m IH]; intros A.
    - transitivity (1 × (1 : F)).
      + unfold detM_leibniz.
        rewrite (psum_O (fun sigma : Perm O ⇒ perm_sign sigma × perm_prod A sigma)),
          perm_sign_O, (perm_prod_O A).
        reflexivity.
      + rewrite (ring_mul_1l (1 : F)).
        symmetry. exact (detM_0 A).
    - unfold detM_leibniz.
      rewrite (psum_S m (fun sigma : Perm (S m) ⇒ perm_sign sigma × perm_prod A sigma)),
        (detM_S m A).
      apply bigopF_ext. intros i.
      calc (psum (fun p : Perm m ⇒
              perm_sign (Perm_S i p) × perm_prod A (Perm_S i p)))
        == (psum (fun p : Perm m ⇒
              (sign_at i × perm_sign p) × (A i Fin.F1 × perm_prod (minor A i) p)))
        == (psum (fun p : Perm m ⇒
              (sign_at i × A i Fin.F1) × (perm_sign p × perm_prod (minor A i) p)))
        == ((sign_at i × A i Fin.F1) × detM_leibniz (minor A i))
        == ((sign_at i × A i Fin.F1) × detM (minor A i))
        == (sign_at i × (A i Fin.F1 × detM (minor A i))).
      + apply psum_ext. intros p.
        apply ring_mul_proper.
        × exact (perm_sign_S m i p).
        × exact (perm_prod_S m A i p).
      + apply psum_ext. intros p.
        exact (mul_four_regroup (sign_at i) (perm_sign p) (A i Fin.F1)
                 (perm_prod (minor A i) p)).
      + exact (psum_scale m (sign_at i × A i Fin.F1)
          (fun p ⇒ perm_sign p × perm_prod (minor A i) p)).
      + apply ring_mul_proper; [reflexivity | exact (IH (minor A i))].
      + exact (ring_mul_assoc (sign_at i) (A i Fin.F1) (detM (minor A i))).
  Qed.


  Lemma row_swap_eq_swap_fin : ∀ (n : nat) (A : Matrix n n F)
      (i j r c : Fin.t n),
    row_swap A i j r c == A (swap_fin i j r) c.
  Proof.
    intros n A i j r c.
    unfold row_swap, swap_fin.
    destruct (dec (r = i)) as [Heq | Hne].
    - reflexivity.
    - destruct (dec (r = j)) as [Heq2 | Hne2].
      + reflexivity.
      + reflexivity.
  Qed.

  Theorem perm_prod_row_swap : ∀ (n : nat) (A : Matrix n n F)
      (i j : Fin.t n) (sigma : Perm n),
    perm_prod (row_swap A i j) sigma
      == prodF (fun c : Fin.t n ⇒ A (swap_fin i j (perm_apply sigma c)) c).
  Proof.
    intros n A i j sigma.
    unfold perm_prod.
    apply prodF_ext. intros c.
    exact (row_swap_eq_swap_fin n A i j (perm_apply sigma c) c).
  Qed.

End DetLeibniz.



Example detM_leibniz_eq_detM_2x2_Q :
  ∀ a b c d : Q, detM_leibniz (mat2 a b c d) == detM (mat2 a b c d).
Proof.
  intros a b c d.
  exact (detM_leibniz_eq_detM 2 (mat2 a b c d)).
Qed.

Example detM_leibniz_2x2_Q :
  detM_leibniz (mat2 (1#1)%Q (2#1)%Q (3#1)%Q (4#1)%Q) == ((-2)#1)%Q.
Proof.
  rewrite (detM_leibniz_eq_detM 2 (mat2 (1#1)%Q (2#1)%Q (3#1)%Q (4#1)%Q)).
  exact det2_Q.
Qed.


Definition swap01 : Fin.t 3 → Fin.t 3 :=
  fun i ⇒ match i with
           | Fin.F1 ⇒ Fin.FS Fin.F1
           | Fin.FS Fin.F1 ⇒ Fin.F1
           | Fin.FS (Fin.FS _) ⇒ Fin.FS (Fin.FS Fin.F1)
           end.

Lemma swap01_invol : ∀ i : Fin.t 3, swap01 (swap01 i) = i.
Proof.
  intros i.
  apply (Fin.caseS' i (fun i0 ⇒ swap01 (swap01 i0) = i0)).
  - reflexivity.
  - intros i'.
    apply (Fin.caseS' i' (fun i1 ⇒ swap01 (swap01 (Fin.FS i1)) = Fin.FS i1)).
    + reflexivity.
    + intros i''.
      apply (Fin.caseS' i''
        (fun i2 ⇒ swap01 (swap01 (Fin.FS (Fin.FS i2))) = Fin.FS (Fin.FS i2))).
      × reflexivity.
      × intros i3.
        exact (Fin.case0
          (fun i4 ⇒ swap01 (swap01 (Fin.FS (Fin.FS (Fin.FS i4))))
                     = Fin.FS (Fin.FS (Fin.FS i4))) i3).
Qed.

Definition f123 : Fin.t 3 → Q :=
  fun i ⇒ match i with
           | Fin.F1 ⇒ (1#1)%Q
           | Fin.FS Fin.F1 ⇒ (2#1)%Q
           | Fin.FS (Fin.FS _) ⇒ (3#1)%Q
           end.

Example bigopF_reindex_swap01_Q_compute :
  bigopF f123 == bigopF (fun i ⇒ f123 (swap01 i)).
Proof.
  reflexivity.
Qed.

Example bigopF_reindex_swap01_Q :
  ∀ (f : Fin.t 3 → Q), bigopF f == bigopF (fun i ⇒ f (swap01 i)).
Proof.
  intros f.
  exact (bigopF_reindex (M := Q) 3 f swap01 swap01
           swap01_invol swap01_invol).
Qed.
