#[global] Instance Prod_EqRel {A B : Type} `{EqRel A} `{EqRel B} :
  EqRel (A × B) :=
  fun p q ⇒ fst p == fst q ∧ snd p == snd q.

#[global] Instance Prod_Zero {A B : Type} `{Zero A} `{Zero B} :
  Zero (A × B) := (0, 0).

#[global] Instance Prod_Add {A B : Type} `{HasAdd A} `{HasAdd B} :
  HasAdd (A × B) :=
  fun p q ⇒ (fst p + fst q, snd p + snd q).

#[global] Instance Prod_Neg {A B : Type} `{HasNeg A} `{HasNeg B} :
  HasNeg (A × B) :=
  fun p ⇒ (- fst p, - snd p).

#[global] Instance Prod_Scale {F A B : Type} `{HasScale F A} `{HasScale F B} :
  HasScale F (A × B) :=
  fun a p ⇒ (a *: fst p, a *: snd p).

#[global] Instance Prod_Equivalence {A B : Type} {eA : EqRel A} {eB : EqRel B}
    {EA : Equivalence eA} {EB : Equivalence eB} :
  Equivalence (@Prod_EqRel A B eA eB).
Proof.
  apply Build_Equivalence.
  - intros p. split; reflexivity.
  - intros p q [H1 H2]. split.
    + symmetry. exact H1.
    + symmetry. exact H2.
  - intros p q r [H1 H2] [H3 H4]. split.
    + rewrite H1. exact H3.
    + rewrite H2. exact H4.
Qed.


Section ProdMonoid.
  Context {A B : Type} `{Monoid A} `{Monoid B}.

  Lemma prod_assoc : ∀ p q r : A × B, (p + q) + r == p + (q + r).
  Proof. intros [a1 b1] [a2 b2] [a3 b3]. split; apply monoid_associative. Qed.

  Lemma prod_idl : ∀ p : A × B, 0 + p == p.
  Proof. intros [a b]. split; apply monoid_left_identity. Qed.

  Lemma prod_idr : ∀ p : A × B, p + 0 == p.
  Proof. intros [a b]. split; apply monoid_right_identity. Qed.

  #[global] Instance prod_add_proper : Proper (equ ==> equ ==> equ) (@add (A × B) _).
  Proof.
    intros p p' [Hp1 Hp2] q q' [Hq1 Hq2].
    split.
    - apply monoid_add_proper; [exact Hp1 | exact Hq1].
    - apply monoid_add_proper; [exact Hp2 | exact Hq2].
  Qed.

  #[global] Instance Prod_Monoid : Monoid (A × B) := {|
    monoid_eq := Prod_EqRel;
    monoid_equivalence := Prod_Equivalence;
    monoid_zero := Prod_Zero;
    monoid_add := Prod_Add;
    monoid_associative := prod_assoc;
    monoid_left_identity := prod_idl;
    monoid_right_identity := prod_idr;
    monoid_add_proper := prod_add_proper
  |}.

End ProdMonoid.


Section ProdGroup.
  Context {A B : Type} `{Group A} `{Group B}.

  Lemma prod_invl : ∀ p : A × B, (- p) + p == 0.
  Proof. intros [a b]. split; apply group_left_inverse. Qed.

  Lemma prod_invr : ∀ p : A × B, p + (- p) == 0.
  Proof. intros [a b]. split; apply group_right_inverse. Qed.

  #[global] Instance prod_neg_proper : Proper (equ ==> equ) (@neg (A × B) _).
  Proof.
    intros p p' [H1 H2].
    split.
    - apply group_neg_proper. exact H1.
    - apply group_neg_proper. exact H2.
  Qed.

  #[global] Instance Prod_Group : Group (A × B) := {|
    group_monoid := Prod_Monoid;
    group_neg := Prod_Neg;
    group_left_inverse := prod_invl;
    group_right_inverse := prod_invr;
    group_neg_proper := prod_neg_proper
  |}.

End ProdGroup.

Section ProdVectorSpace.
  Context {F A B : Type} `{VectorSpace F A} `{VS2 : !VectorSpace F B}.

  #[global] Instance prod_sc_proper : Proper (equ ==> equ ==> equ) (@scale F (A × B) _).
  Proof.
    intros a a' Ha p p' [H1 H2].
    split.
    - apply sc_proper; [exact Ha | exact H1].
    - apply sc_proper; [exact Ha | exact H2].
  Qed.

  Lemma prod_scale_add : ∀ (a : F) (u v : A × B),
    a *: (u + v) == a *: u + a *: v.
  Proof. intros a [u1 u2] [v1 v2]. split; apply sc_scale_add. Qed.

  Lemma prod_add_scale : ∀ (a b : F) (u : A × B),
    (a + b) *: u == a *: u + b *: u.
  Proof. intros a b [u1 u2]. split; apply sc_add_scale. Qed.

  Lemma prod_scale_scale : ∀ (a b : F) (u : A × B),
    (a × b) *: u == a *: (b *: u).
  Proof. intros a b [u1 u2]. split; apply sc_scale_scale. Qed.

  Lemma prod_scale_1 : ∀ u : A × B, 1 *: u == u.
  Proof. intros [u1 u2]. split; apply sc_scale_1. Qed.

  #[global] Instance Prod_ScaleLaws : ScaleLaws F (A × B) :=
    @Build_ScaleLaws F (A × B) _ Prod_AbGroup Prod_Scale
      prod_sc_proper prod_scale_add prod_add_scale prod_scale_scale
      prod_scale_1.

  #[global] Instance Prod_VectorSpace : VectorSpace F (A × B) :=
    @Build_VectorSpace F (A × B) _ Prod_AbGroup Prod_Scale Prod_ScaleLaws.

End ProdVectorSpace.

Section ProdNormedSpace.
  Context {F A B : Type} `{NormedSpace F A}.
  Context {VB : VectorSpace F B} {NB : NormedSpace F B}.

  Definition prod_norm (p : A × B) : F := ‖ fst p ‖ + ‖ snd p ‖.

  #[global] Instance Prod_HasNorm : HasNorm F (A × B) := prod_norm.

  #[global] Instance prod_norm_proper : Proper (equ ==> equ) (@vnorm F (A × B) Prod_HasNorm).
  Proof.
    intros [u1 v1] [u2 v2] [Hu Hv].
    unfold vnorm, Prod_HasNorm, prod_norm; simpl.
    rewrite Hu, Hv.
    reflexivity.
  Qed.

  Lemma prod_norm_nonneg : ∀ p : A × B, 0 ≤ ‖ p ‖.
  Proof.
    intros [u v].
    unfold vnorm, Prod_HasNorm, prod_norm; simpl.
    apply (le_rewrite_l ((0 : F) + 0)).
    - apply monoid_left_identity.
    - apply add_le_mono; apply norm_nonneg.
  Qed.

  Lemma prod_norm_def : ∀ p : A × B, ‖ p ‖ == 0 → p == 0.
  Proof.
    intros [u v] Heq.
    unfold vnorm, Prod_HasNorm, prod_norm in Heq; simpl in Heq.
    destruct (nonneg_add_eq_zero (‖ u ‖) (‖ v ‖) (norm_nonneg u)
      (norm_nonneg v) Heq) as [Hu Hv].
    split; simpl.
    - apply norm_def. exact Hu.
    - apply norm_def. exact Hv.
  Qed.

  Lemma prod_norm_scale : ∀ (a : F) (p : A × B), ‖ a *: p ‖ == abs a × ‖ p ‖.
  Proof.
    intros a [u v].
    unfold vnorm, Prod_HasNorm, prod_norm; simpl.
    rewrite norm_scale, norm_scale, ring_distl.
    reflexivity.
  Qed.

  Lemma prod_norm_triangle : ∀ p q : A × B, ‖ p + q ‖ ≤ ‖ p ‖ + ‖ q ‖.
  Proof.
    intros [u1 v1] [u2 v2].
    unfold vnorm, Prod_HasNorm, prod_norm; simpl.
    apply (le_trans (‖ u1 + u2 ‖ + ‖ v1 + v2 ‖)
      ((‖ u1 ‖ + ‖ u2 ‖) + (‖ v1 ‖ + ‖ v2 ‖))
      ((‖ u1 ‖ + ‖ v1 ‖) + (‖ u2 ‖ + ‖ v2 ‖))).
    - apply add_le_mono; apply norm_triangle.
    - apply eq_le. apply add_interchange.
  Qed.

  #[global] Instance Prod_NormedSpace : NormedSpace F (A × B) := {|
    ns_norm := Prod_HasNorm;
    norm_proper := prod_norm_proper;
    norm_nonneg := prod_norm_nonneg;
    norm_def := prod_norm_def;
    norm_scale := prod_norm_scale;
    norm_triangle := prod_norm_triangle
  |}.

End ProdNormedSpace.
