Section FunEqu.
  Context {F : Type} `{OrderedField F}.

  Lemma is_deriv_fun_equ : ∀ (f g : F → F) (a : F) (L : LinearMap F F F),
    (∀ x : F, f x == g x) → is_deriv f a L → is_deriv g a L.
  Proof.
    intros f g a L Hfg [Hb HD].
    split.
    - exact Hb.
    - intros eps Heps.
      destruct (HD eps Heps) as [d [Hd HR]].
      ∃ d.
      split.
      + exact Hd.
      + intros h Hh.
        apply (le_rewrite_l (‖ f (a + h) + - (f a + L h) ‖)).
        × rewrite (Hfg (a + h)), (Hfg a).
          reflexivity.
        × exact (HR h Hh).
  Qed.

End FunEqu.


Section MulBilinear.
  Context {F : Type} `{OrderedField F}.

  Lemma mul_bm_scale_l : ∀ (a u v : F), (a *: u) × v == a *: (u × v).
  Proof.
    intros a u v.
    exact (ring_mul_assoc a u v).
  Qed.

  Lemma mul_bm_scale_r : ∀ (a u v : F), u × (a *: v) == a *: (u × v).
  Proof.
    intros a u v.
    exact (mul_comm_swap u a v).
  Qed.

  Definition mul_bm : BilinearMap F F F F :=
    @Build_BilinearMap F F F F _ Field_VectorSpace Field_VectorSpace
      Field_VectorSpace mult ring_mul_proper ring_distr mul_bm_scale_l
      ring_distl mul_bm_scale_r.

  Lemma mul_bm_bounded : bm_bounded mul_bm.
  Proof.
    ∃ 1.
    split.
    - exact zero_le_one.
    - intros u v.
      apply eq_le.
      calc (‖ mul_bm u v ‖)
        == (‖ u ‖ × ‖ v ‖)
        == (1 × ‖ u ‖ × ‖ v ‖).
      + exact (abs_mul u v).
      + symmetry.
        apply ring_mul_proper; [ exact (ring_mul_1l (‖ u ‖)) | reflexivity ].
  Qed.

  Lemma mul_pair_proper :
    Proper (equ ==> equ) (fun pr : F × F ⇒ mul_bm (fst pr) (snd pr)).
  Proof.
    intros [x1 y1] [x2 y2] [Hx Hy].
    exact (bm_proper mul_bm x1 x2 Hx y1 y2 Hy).
  Qed.

End MulBilinear.


Section PairLM.
  Context {F : Type} `{OrderedField F}.
  Context (Lf Lg : LinearMap F F F).

  Definition pair_fun (h : F) : F × F := (Lf h, Lg h).

  Lemma pair_fun_proper : Proper (equ ==> equ) pair_fun.
  Proof.
    intros h h' Hh.
    split.
    - apply (lm_proper Lf). exact Hh.
    - apply (lm_proper Lg). exact Hh.
  Qed.

  Lemma pair_fun_add : ∀ h1 h2 : F,
    pair_fun (h1 + h2) == pair_fun h1 + pair_fun h2.
  Proof.
    intros h1 h2.
    split.
    - exact (lm_add Lf h1 h2).
    - exact (lm_add Lg h1 h2).
  Qed.

  Lemma pair_fun_scale : ∀ (a h : F), pair_fun (a *: h) == a *: pair_fun h.
  Proof.
    intros a h.
    split.
    - exact (lm_scale Lf a h).
    - exact (lm_scale Lg a h).
  Qed.

  Definition lm_pair : LinearMap F F (F × F) :=
    @Build_LinearMap F F (F × F) _ Field_VectorSpace _
      pair_fun pair_fun_proper pair_fun_add pair_fun_scale.

  Lemma lm_pair_bounded : lm_bounded Lf → lm_bounded Lg → lm_bounded lm_pair.
  Proof.
    intros [Cf [HCf Hbf]] [Cg [HCg Hbg]].
    ∃ (Cf + Cg).
    split.
    - apply (le_rewrite_l ((0 : F) + 0)).
      + apply idl.
      + apply add_le_mono.
        × exact HCf.
        × exact HCg.
    - intros u.
      apply (le_rewrite_l (‖ Lf u ‖ + ‖ Lg u ‖)).
      + reflexivity.
      + apply (le_trans _ (Cf × ‖ u ‖ + Cg × ‖ u ‖) _).
        × apply add_le_mono.
          -- exact (Hbf u).
          -- exact (Hbg u).
        × apply eq_le. symmetry. exact (ring_distr Cf Cg (‖ u ‖)).
  Qed.

  Lemma deriv_pair : ∀ (f g : F → F) (a : F),
    is_deriv f a Lf → is_deriv g a Lg →
    is_deriv (fun t : F ⇒ (f t, g t)) a lm_pair.
  Proof.
    intros f g a [Hbf HDf] [Hbg HDg].
    split.
    - exact (lm_pair_bounded Hbf Hbg).
    - intros eps Heps.
      destruct (HDf (half eps) (half_pos eps Heps)) as [d1 [Hd1 HR1]].
      destruct (HDg (half eps) (half_pos eps Heps)) as [d2 [Hd2 HR2]].
      destruct (pos_min_exists d1 d2 Hd1 Hd2) as [d [Hd [Hdd1 Hdd2]]].
      ∃ d.
      split.
      + exact Hd.
      + intros h Hh.
        apply (le_rewrite_l
          (‖ f (a + h) + - (f a + Lf h) ‖ + ‖ g (a + h) + - (g a + Lg h) ‖)).
        × reflexivity.
        × apply (le_trans _ (half eps × ‖ h ‖ + half eps × ‖ h ‖) _).
          -- apply add_le_mono.
             ++ apply HR1. exact (lt_le_trans (‖ h ‖) d d1 Hh Hdd1).
             ++ apply HR2. exact (lt_le_trans (‖ h ‖) d d2 Hh Hdd2).
          -- apply eq_le. apply half_mul_sum.
  Qed.

End PairLM.


Section ProductRule.
  Context {F : Type} `{OrderedField F}.

  Theorem deriv_mul : ∀ (f g : F → F) (a : F) (fd gd : F),
    is_deriv f a (lm_scale_map fd) → is_deriv g a (lm_scale_map gd) →
    is_deriv (fun t : F ⇒ f t × g t) a
      (lm_scale_map (fd × g a + f a × gd)).
  Proof.
    intros f g a fd gd HDf HDg.
    apply (is_deriv_L_equ (fun t : F ⇒ f t × g t) a
      (lm_comp (bilinear_deriv_lm mul_bm (f a) (g a))
               (lm_pair (lm_scale_map fd) (lm_scale_map gd)))).
    -
      intros h.
      change ((fd × h) × g a + f a × (gd × h)
                == (fd × g a + f a × gd) × h).
      rewrite (mul_assoc_comm_r fd h (g a)),
        <- (ring_mul_assoc (f a) gd h),
        <- (ring_distr (fd × g a) (f a × gd) h).
      reflexivity.
    - exact (deriv_comp (fun t : F ⇒ (f t, g t))
        (fun pr : F × F ⇒ mul_bm (fst pr) (snd pr)) a
        (lm_pair (lm_scale_map fd) (lm_scale_map gd))
        (bilinear_deriv_lm mul_bm (f a) (g a))
        mul_pair_proper
        (deriv_pair (lm_scale_map fd) (lm_scale_map gd) f g a HDf HDg)
        (bilinear_is_deriv mul_bm (f a) (g a) mul_bm_bounded)).
  Qed.

End ProductRule.


Section Hamiltonian.
  Context {F : Type} `{OrderedField F}.

  Context (m : F) (V Vd : F → F) (q p v : F → F).
  Context (V_proper : Proper (equ ==> equ) V).
  Context (potential_deriv : ∀ x : F, is_deriv V x (lm_scale_map (Vd x))).

  Context (hamilton_q : ∀ t : F, is_deriv q t (lm_scale_map (v t))).
  Context (hamilton_p : ∀ t : F, is_deriv p t (lm_scale_map (- Vd (q t)))).
  Context (momentum : ∀ t : F, m × v t == p t).

  Definition energy (t : F) : F := p t × p t + (two × m) × V (q t).

  Lemma potential_along_path_deriv : ∀ t : F,
    is_deriv (fun s : F ⇒ V (q s)) t (lm_scale_map (Vd (q t) × v t)).
  Proof.
    intros t.
    apply (is_deriv_L_equ (fun s : F ⇒ V (q s)) t
      (lm_comp (lm_scale_map (Vd (q t))) (lm_scale_map (v t)))).
    - exact (lm_comp_scale_scale (v t) (Vd (q t))).
    - exact (deriv_comp q V t (lm_scale_map (v t)) (lm_scale_map (Vd (q t)))
        V_proper (hamilton_q t) (potential_deriv (q t))).
  Qed.

  Lemma scaled_potential_deriv : ∀ t : F,
    is_deriv (fun s : F ⇒ (two × m) × V (q s)) t
      (lm_scale_map ((two × m) × (Vd (q t) × v t))).
  Proof.
    intros t.
    apply (is_deriv_L_equ (fun s : F ⇒ (two × m) × V (q s)) t
      (lm_comp (lm_scale_map (two × m)) (lm_scale_map (Vd (q t) × v t)))).
    - exact (lm_comp_scale_scale (Vd (q t) × v t) (two × m)).
    - exact (deriv_comp (fun s : F ⇒ V (q s)) (scale_map (two × m)) t
        (lm_scale_map (Vd (q t) × v t)) (lm_scale_map (two × m))
        (scale_map_proper (two × m))
        (potential_along_path_deriv t)
        (is_deriv_scale_map (two × m) (V (q t)))).
  Qed.

  Lemma kinetic_deriv : ∀ t : F,
    is_deriv (fun s : F ⇒ p s × p s) t
      (lm_scale_map ((- Vd (q t)) × p t + p t × (- Vd (q t)))).
  Proof.
    intros t.
    exact (deriv_mul p p t (- Vd (q t)) (- Vd (q t))
      (hamilton_p t) (hamilton_p t)).
  Qed.

  Lemma hamilton_cancel : ∀ t : F,
    ((- Vd (q t)) × p t + p t × (- Vd (q t))) + (two × m) × (Vd (q t) × v t)
      == 0.
  Proof.
    intros t.
    rewrite (mul_neg_l (Vd (q t)) (p t)),
      (mul_neg_r (p t) (Vd (q t))),
      (field_mul_comm (p t) (Vd (q t))),
      <- (neg_add (Vd (q t) × p t) (Vd (q t) × p t)),
      (ring_mul_assoc two m (Vd (q t) × v t)),
      (mul_comm_swap m (Vd (q t)) (v t)),
      (momentum t),
      (field_mul_comm two (Vd (q t) × p t)),
      (mul_two (Vd (q t) × p t)).
    exact (grp_invl (Vd (q t) × p t + Vd (q t) × p t)).
  Qed.

  Lemma lm_add_scale_scale : ∀ c d : F,
    (lm_scale_map c + lm_scale_map d : LinearMap F F F)
      == lm_scale_map (c + d).
  Proof.
    intros c d h.
    symmetry.
    exact (ring_distr c d h).
  Qed.

  Theorem energy_conserved : ∀ t : F, is_deriv energy t lm_const0.
  Proof.
    intros t.
    apply (is_deriv_L_equ energy t
      (lm_scale_map ((- Vd (q t)) × p t + p t × (- Vd (q t)))
         + lm_scale_map ((two × m) × (Vd (q t) × v t)))).
    - intros h.
      calc ((lm_scale_map ((- Vd (q t)) × p t + p t × (- Vd (q t)))
               + lm_scale_map ((two × m) × (Vd (q t) × v t))) h)
        == ((((- Vd (q t)) × p t + p t × (- Vd (q t)))
               + (two × m) × (Vd (q t) × v t)) × h)
        == ((0 : F) × h)
        == (0 : F).
      + exact (lm_add_scale_scale ((- Vd (q t)) × p t + p t × (- Vd (q t)))
          ((two × m) × (Vd (q t) × v t)) h).
      + apply ring_mul_proper; [ exact (hamilton_cancel t) | reflexivity ].
      + exact (mul_0_l h).
    - exact (deriv_add (fun s : F ⇒ p s × p s)
        (fun s : F ⇒ (two × m) × V (q s)) t
        (lm_scale_map ((- Vd (q t)) × p t + p t × (- Vd (q t))))
        (lm_scale_map ((two × m) × (Vd (q t) × v t)))
        (kinetic_deriv t) (scaled_potential_deriv t)).
  Qed.

  Theorem newton_second_law : ∀ t : F,
    is_deriv (fun s : F ⇒ m × v s) t (lm_scale_map (- Vd (q t))).
  Proof.
    intros t.
    apply (is_deriv_fun_equ p (fun s : F ⇒ m × v s) t
      (lm_scale_map (- Vd (q t)))).
    - intros x. symmetry. exact (momentum x).
    - exact (hamilton_p t).
  Qed.

  Definition hamiltonian (t : F) : F := p t × p t × inv (two × m) + V (q t).

  Section WithMassNonzero.
    Context (m_neq_0 : m =/= 0).

    Lemma two_m_neq_0 : two × m =/= 0.
    Proof.
      exact (field_mul_neq_0 two m two_neq_0 m_neq_0).
    Qed.

    Lemma hamiltonian_from_energy : ∀ t : F,
      inv (two × m) × energy t == hamiltonian t.
    Proof.
      intros t.
      unfold energy, hamiltonian.
      calc (inv (two × m) × (p t × p t + (two × m) × V (q t)))
        == (inv (two × m) × (p t × p t)
              + inv (two × m) × ((two × m) × V (q t)))
        == (p t × p t × inv (two × m) + V (q t)).
      - apply ring_distl.
      - rewrite (field_mul_comm (inv (two × m)) (p t × p t)),
          (inv_mul_cancel (two × m) (V (q t)) two_m_neq_0).
        reflexivity.
    Qed.

    Theorem hamiltonian_conserved : ∀ t : F,
      is_deriv hamiltonian t lm_const0.
    Proof.
      intros t.
      apply (is_deriv_fun_equ (fun s : F ⇒ inv (two × m) × energy s)
        hamiltonian t lm_const0).
      - exact hamiltonian_from_energy.
      - apply (is_deriv_L_equ (fun s : F ⇒ inv (two × m) × energy s) t
          (lm_comp (lm_scale_map (inv (two × m))) lm_const0)).
        + intros h.
          exact (mul_0_r (inv (two × m))).
        + exact (deriv_comp energy (scale_map (inv (two × m))) t
            lm_const0 (lm_scale_map (inv (two × m)))
            (scale_map_proper (inv (two × m)))
            (energy_conserved t)
            (is_deriv_scale_map (inv (two × m)) (energy t))).
    Qed.

  End WithMassNonzero.

End Hamiltonian.
