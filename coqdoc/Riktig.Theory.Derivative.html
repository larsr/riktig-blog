Section Derivative.
  Context {F U V : Type} `{NormedSpace F U}.
  Context {VSV : VectorSpace F V} {NSV : NormedSpace F V}.

  Definition is_deriv (f : U → V) (a : U) (L : LinearMap F U V) : Prop :=
    lm_bounded L ∧
    ∀ eps : F, 0 < eps →
      ∃ delta : F, 0 < delta ∧
        ∀ h : U, ‖ h ‖ < delta →
          ‖ f (a + h) + - (f a + L h) ‖ ≤ eps × ‖ h ‖.


  Lemma bounded_const0 : lm_bounded (lm_const0 : LinearMap F U V).
  Proof.
    ∃ 0.
    split.
    - apply le_refl.
    - intros u.
      apply (le_rewrite_l (‖ (0 : V) ‖)).
      + reflexivity.
      + rewrite norm_zero, (mul_0_l (‖ u ‖)).
        apply le_refl.
  Qed.

  Lemma bounded_add : ∀ f g : LinearMap F U V,
    lm_bounded f → lm_bounded g → lm_bounded (f + g).
  Proof.
    intros f g [C1 [HC1 Hb1]] [C2 [HC2 Hb2]].
    ∃ (C1 + C2).
    split.
    - apply (le_rewrite_l ((0 : F) + 0)).
      + apply idl.
      + apply add_le_mono.
        × exact HC1.
        × exact HC2.
    - intros u.
      apply (le_trans _ (‖ f u ‖ + ‖ g u ‖) _).
      + apply norm_triangle.
      + apply (le_trans _ (C1 × ‖ u ‖ + C2 × ‖ u ‖) _).
        × apply add_le_mono.
          -- exact (Hb1 u).
          -- exact (Hb2 u).
        × apply eq_le. symmetry. apply ring_distr.
  Qed.

  Lemma bounded_scale : ∀ (c : F) (f : LinearMap F U V),
    lm_bounded f → lm_bounded (c *: f).
  Proof.
    intros c f [C [HC Hb]].
    ∃ (abs c × C).
    split.
    - apply mul_nonneg.
      + apply abs_nonneg.
      + exact HC.
    - intros u.
      apply (le_rewrite_l (abs c × ‖ f u ‖)).
      + symmetry. apply norm_scale.
      + apply (le_trans _ (abs c × (C × ‖ u ‖)) _).
        × apply mul_le_mono_l.
          -- apply abs_nonneg.
          -- exact (Hb u).
        × apply eq_le. symmetry. apply ring_mul_assoc.
  Qed.

  Lemma bounded_neg : ∀ f : LinearMap F U V,
    lm_bounded f → lm_bounded (- f).
  Proof.
    intros f [C [HC Hb]].
    ∃ C.
    split.
    - exact HC.
    - intros u.
      apply (le_rewrite_l (‖ f u ‖)).
      + symmetry. apply norm_neg.
      + exact (Hb u).
  Qed.


  Lemma const_remainder_zero : ∀ v : V, ‖ v + - (v + (0 : V)) ‖ == 0.
  Proof.
    intros v.
    rewrite (idr v).
    apply norm_sub_self.
  Qed.

  Lemma lm_remainder_zero : ∀ (g : LinearMap F U V) (a h : U),
    ‖ g (a + h) + - (g a + g h) ‖ == 0.
  Proof.
    intros g a h.
    rewrite (lm_add g a h).
    apply norm_sub_self.
  Qed.

  Lemma add_remainder_split : ∀ p p' q q' x x' : V,
    ‖ (p + p') + - ((q + q') + (x + x')) ‖ ≤
    ‖ p + - (q + x) ‖ + ‖ p' + - (q' + x') ‖.
  Proof.
    intros p p' q q' x x'.
    rewrite (add_interchange q q' x x').
    apply norm_sub_pair.
  Qed.

  Lemma scale_remainder_norm : ∀ (c : F) (p q x : V),
    ‖ c *: p + - (c *: q + c *: x) ‖ == abs c × ‖ p + - (q + x) ‖.
  Proof.
    intros c p q x.
    rewrite <- (sc_scale_add c q x).
    apply norm_scale_sub.
  Qed.

  Lemma neg_remainder_norm : ∀ p q x : V,
    ‖ - p + - (- q + - x) ‖ == ‖ p + - (q + x) ‖.
  Proof.
    intros p q x.
    rewrite (norm_neg_add p (- q + - x)), <- (neg_add q x).
    reflexivity.
  Qed.

  Lemma deriv_increment_bound : ∀ (f : U → V) (a h : U)
      (L : LinearMap F U V) (e C : F),
    0 ≤ C → (∀ u : U, ‖ L u ‖ ≤ C × ‖ u ‖) → e ≤ 1 →
    ‖ f (a + h) + - (f a + L h) ‖ ≤ e × ‖ h ‖ →
    ‖ f (a + h) + - f a ‖ ≤ (1 + C) × ‖ h ‖.
  Proof.
    intros f a h L e C HC Hb He Hr.
    rewrite <- (sub_add_cancel_r (f (a + h)) (f a) (L h)).
    apply (le_trans _ (‖ f (a + h) + - (f a + L h) ‖ + ‖ L h ‖) _).
    - apply norm_triangle.
    - apply (le_trans _ (e × ‖ h ‖ + C × ‖ h ‖) _).
      + apply add_le_mono.
        × exact Hr.
        × exact (Hb h).
      + apply (le_rewrite_l ((e + C) × ‖ h ‖)).
        × apply ring_distr.
        × apply mul_le_mono_r.
          -- apply norm_nonneg.
          -- apply add_le_mono_r. exact He.
  Qed.


  Theorem deriv_const : ∀ (v : V) (a : U),
    is_deriv (fun _ : U ⇒ v) a lm_const0.
  Proof.
    intros v a.
    split.
    - exact bounded_const0.
    - intros eps Heps.
      ∃ 1.
      split.
      + exact lt_0_1.
      + intros h Hh.
        apply (le_rewrite_l (0 : F)).
        × symmetry. exact (const_remainder_zero v).
        × apply mul_nonneg.
          -- apply lt_le. exact Heps.
          -- apply norm_nonneg.
  Qed.

  Theorem deriv_lm : ∀ (g : LinearMap F U V) (a : U),
    lm_bounded g → is_deriv (lm_fun g) a g.
  Proof.
    intros g a Hb.
    split.
    - exact Hb.
    - intros eps Heps.
      ∃ 1.
      split.
      + exact lt_0_1.
      + intros h Hh.
        apply (le_rewrite_l (0 : F)).
        × symmetry. exact (lm_remainder_zero g a h).
        × apply mul_nonneg.
          -- apply lt_le. exact Heps.
          -- apply norm_nonneg.
  Qed.

  Theorem deriv_add : ∀ (f g : U → V) (a : U)
      (Lf Lg : LinearMap F U V),
    is_deriv f a Lf → is_deriv g a Lg →
    is_deriv (fun u : U ⇒ f u + g u) a (Lf + Lg).
  Proof.
    intros f g a Lf Lg [Hbf HDf] [Hbg HDg].
    split.
    - exact (bounded_add Lf Lg Hbf Hbg).
    - intros eps Heps.
      destruct (HDf (half eps) (half_pos eps Heps)) as [d1 [Hd1 HR1]].
      destruct (HDg (half eps) (half_pos eps Heps)) as [d2 [Hd2 HR2]].
      destruct (pos_min_exists d1 d2 Hd1 Hd2) as [d [Hd [Hdd1 Hdd2]]].
      ∃ d.
      split.
      + exact Hd.
      + intros h Hh.
        apply (le_trans _
          (‖ f (a + h) + - (f a + Lf h) ‖ + ‖ g (a + h) + - (g a + Lg h) ‖)
          _).
        × exact (add_remainder_split (f (a + h)) (g (a + h)) (f a) (g a)
            (Lf h) (Lg h)).
        × apply (le_trans _ (half eps × ‖ h ‖ + half eps × ‖ h ‖) _).
          -- apply add_le_mono.
             ++ apply HR1. exact (lt_le_trans (‖ h ‖) d d1 Hh Hdd1).
             ++ apply HR2. exact (lt_le_trans (‖ h ‖) d d2 Hh Hdd2).
          -- apply eq_le. apply half_mul_sum.
  Qed.

  Theorem deriv_scale : ∀ (c : F) (f : U → V) (a : U)
      (Lf : LinearMap F U V),
    is_deriv f a Lf → is_deriv (fun u : U ⇒ c *: f u) a (c *: Lf).
  Proof.
    intros c f a Lf [Hbf HDf].
    split.
    - exact (bounded_scale c Lf Hbf).
    - intros eps Heps.
      destruct (HDf (eps × inv (1 + abs c))
          (mul_pos eps (inv (1 + abs c)) Heps
             (pos_inv (1 + abs c)
                (one_plus_nonneg_pos (abs c) (abs_nonneg c)))))
        as [d [Hd HR]].
      ∃ d.
      split.
      + exact Hd.
      + intros h Hh.
        apply (le_rewrite_l (abs c × ‖ f (a + h) + - (f a + Lf h) ‖)).
        × symmetry.
          exact (scale_remainder_norm c (f (a + h)) (f a) (Lf h)).
        × apply (le_trans _ (abs c × (eps × inv (1 + abs c) × ‖ h ‖)) _).
          -- apply mul_le_mono_l.
             ++ apply abs_nonneg.
             ++ apply HR. exact Hh.
          -- apply (le_rewrite_l ((abs c × (eps × inv (1 + abs c))) × ‖ h ‖)).
             ++ apply ring_mul_assoc.
             ++ apply mul_le_mono_r.
                ** apply norm_nonneg.
                ** apply mul_pad_div_le.
                   --- apply lt_le. exact Heps.
                   --- apply abs_nonneg.
  Qed.

  Theorem deriv_neg : ∀ (f : U → V) (a : U) (Lf : LinearMap F U V),
    is_deriv f a Lf → is_deriv (fun u : U ⇒ - f u) a (- Lf).
  Proof.
    intros f a Lf [Hbf HDf].
    split.
    - exact (bounded_neg Lf Hbf).
    - intros eps Heps.
      destruct (HDf eps Heps) as [d [Hd HR]].
      ∃ d.
      split.
      + exact Hd.
      + intros h Hh.
        apply (le_rewrite_l (‖ f (a + h) + - (f a + Lf h) ‖)).
        × symmetry.
          exact (neg_remainder_norm (f (a + h)) (f a) (Lf h)).
        × apply HR. exact Hh.
  Qed.


  Lemma deriv_diff_bound : ∀ (f : U → V) (a : U)
      (L1 L2 : LinearMap F U V) (eps : F) (u : U),
    is_deriv f a L1 → is_deriv f a L2 → 0 < eps →
    ‖ L1 u + - L2 u ‖ ≤ eps × ‖ u ‖.
  Proof.
    intros f a L1 L2 eps u [_ HD1] [_ HD2] Heps.
    destruct (HD1 (half eps) (half_pos eps Heps)) as [d1 [Hd1 HR1]].
    destruct (HD2 (half eps) (half_pos eps Heps)) as [d2 [Hd2 HR2]].
    destruct (pos_min_exists d1 d2 Hd1 Hd2) as [d [Hd [Hdd1 Hdd2]]].
    destruct (small_scale_exists d (‖ u ‖) Hd (norm_nonneg u))
      as [t [Ht Htu]].
    apply (mul_le_cancel_l (‖ L1 u + - L2 u ‖) (eps × ‖ u ‖) t Ht).
    rewrite <- (norm_scale_nonneg t (L1 u + - L2 u) (lt_le 0 t Ht)),
      (scale_sub t (L1 u) (L2 u)), <- (lm_scale L1 t u), <- (lm_scale L2 t u),
      <- (sub_common_cancel (f (a + t *: u)) (f a) (L2 (t *: u))
            (L1 (t *: u))).
    apply (le_trans _
      (‖ f (a + t *: u) + - (f a + L2 (t *: u)) ‖
       + ‖ f (a + t *: u) + - (f a + L1 (t *: u)) ‖) _).
    - apply norm_sub_le_add.
    - apply (le_trans _ (half eps × ‖ t *: u ‖ + half eps × ‖ t *: u ‖) _).
      + apply add_le_mono.
        × apply HR2.
          apply (le_lt_trans (‖ t *: u ‖) (t × ‖ u ‖) d2).
          -- apply eq_le. apply norm_scale_nonneg. apply lt_le. exact Ht.
          -- exact (lt_le_trans (t × ‖ u ‖) d d2 Htu Hdd2).
        × apply HR1.
          apply (le_lt_trans (‖ t *: u ‖) (t × ‖ u ‖) d1).
          -- apply eq_le. apply norm_scale_nonneg. apply lt_le. exact Ht.
          -- exact (lt_le_trans (t × ‖ u ‖) d d1 Htu Hdd1).
      + apply (le_rewrite_l (eps × ‖ t *: u ‖)).
        × symmetry. apply half_mul_sum.
        × apply eq_le.
          rewrite (norm_scale_nonneg t u (lt_le 0 t Ht)).
          apply mul_comm_swap.
  Qed.

  Theorem deriv_unique : ∀ (f : U → V) (a : U)
      (L1 L2 : LinearMap F U V),
    is_deriv f a L1 → is_deriv f a L2 → L1 == L2.
  Proof.
    intros f a L1 L2 HD1 HD2 u.
    apply sub_eq_zero.
    apply norm_def.
    apply nonneg_below_all_pos_is_zero.
    - apply norm_nonneg.
    - intros eps Heps.
      apply (le_trans (‖ L1 u + - L2 u ‖)
        ((eps × inv (1 + ‖ u ‖)) × ‖ u ‖) eps).
      + apply (deriv_diff_bound f a L1 L2 (eps × inv (1 + ‖ u ‖)) u HD1 HD2).
        apply mul_pos.
        × exact Heps.
        × apply pos_inv. apply one_plus_nonneg_pos. apply norm_nonneg.
      + apply (le_rewrite_l (‖ u ‖ × (eps × inv (1 + ‖ u ‖)))).
        × apply field_mul_comm.
        × apply mul_pad_div_le.
          -- apply lt_le. exact Heps.
          -- apply norm_nonneg.
  Qed.


  Theorem deriv_continuous : ∀ (f : U → V) (a : U)
      (L : LinearMap F U V),
    Proper (equ ==> equ) f → is_deriv f a L → continuous_at f a.
  Proof.
    intros f a L Pf [[C [HC Hb]] HD].
    intros eps Heps.
    destruct (HD 1 lt_0_1) as [d0 [Hd0 HR]].
    destruct (pos_min_exists d0 (eps × inv (1 + (1 + C))) Hd0
        (mul_pos eps (inv (1 + (1 + C))) Heps
           (pos_inv (1 + (1 + C))
              (one_plus_nonneg_pos (1 + C)
                 (lt_le 0 (1 + C) (one_plus_nonneg_pos C HC))))))
      as [delta [Hdelta [Hdd0 Hdd2]]].
    ∃ delta.
    split.
    - exact Hdelta.
    - intros u Hu.
      rewrite <- (add_sub_cancel a u),
        <- (sub_add_cancel_r (f (a + (u + - a))) (f a) (L (u + - a))).
      apply (le_lt_trans _
        (‖ f (a + (u + - a)) + - (f a + L (u + - a)) ‖ + ‖ L (u + - a) ‖)
        eps).
      + apply norm_triangle.
      + apply (le_lt_trans _ (1 × ‖ u + - a ‖ + C × ‖ u + - a ‖) eps).
        × apply add_le_mono.
          -- apply HR. exact (lt_le_trans (‖ u + - a ‖) delta d0 Hu Hdd0).
          -- exact (Hb (u + - a)).
        × apply (le_lt_trans _ ((1 + C) × ‖ u + - a ‖) eps).
          -- apply eq_le. symmetry. apply ring_distr.
          -- apply (lt_le_trans _
               ((1 + (1 + C)) × (eps × inv (1 + (1 + C)))) eps).
             ++ apply mul_lt_pad.
                ** apply lt_le. apply one_plus_nonneg_pos. exact HC.
                ** apply norm_nonneg.
                ** exact (lt_le_trans (‖ u + - a ‖) delta
                     (eps × inv (1 + (1 + C))) Hu Hdd2).
             ++ apply eq_le.
                apply field_mul_div_cancel.
                apply pos_neq_0.
                apply one_plus_nonneg_pos.
                apply lt_le.
                apply one_plus_nonneg_pos.
                exact HC.
  Qed.

End Derivative.


Section DerivativeComp.
  Context {F U V W : Type} `{NormedSpace F U}.
  Context {VSV : VectorSpace F V} {NSV : NormedSpace F V}.
  Context {VSW : VectorSpace F W} {NSW : NormedSpace F W}.

  Lemma bounded_comp : ∀ (g : LinearMap F V W) (f : LinearMap F U V),
    lm_bounded g → lm_bounded f → lm_bounded (lm_comp g f).
  Proof.
    intros g f [Cg [HCg Hbg]] [Cf [HCf Hbf]].
    ∃ (Cg × Cf).
    split.
    - apply mul_nonneg.
      + exact HCg.
      + exact HCf.
    - intros u.
      apply (le_trans _ (Cg × ‖ f u ‖) _).
      + exact (Hbg (f u)).
      + apply (le_trans _ (Cg × (Cf × ‖ u ‖)) _).
        × apply mul_le_mono_l.
          -- exact HCg.
          -- exact (Hbf u).
        × apply eq_le. symmetry. apply ring_mul_assoc.
  Qed.

  Lemma comp_remainder_split : ∀ (g : V → W) (Lg : LinearMap F V W),
    Proper (equ ==> equ) g →
    ∀ b y x : V,
    ‖ g y + - (g b + Lg x) ‖ ≤
    ‖ g (b + (y + - b)) + - (g b + Lg (y + - b)) ‖ + ‖ Lg (y + - (b + x)) ‖.
  Proof.
    intros g Lg Pg b y x.
    apply (le_rewrite_l
      (‖ (g (b + (y + - b)) + - (g b + Lg (y + - b)))
         + Lg (y + - (b + x)) ‖)).
    - apply norm_proper.
      rewrite (add_sub_cancel b y),
        (lm_proper Lg (y + - (b + x)) ((y + - b) + - x)
           (neg_add_assoc y b x)),
        (lm_add Lg (y + - b) (- x)), (lm_neg Lg x).
      apply sub_mid_cancel.
    - apply norm_triangle.
  Qed.

  Theorem deriv_comp : ∀ (f : U → V) (g : V → W) (a : U)
      (Lf : LinearMap F U V) (Lg : LinearMap F V W),
    Proper (equ ==> equ) g →
    is_deriv f a Lf → is_deriv g (f a) Lg →
    is_deriv (fun u : U ⇒ g (f u)) a (lm_comp Lg Lf).
  Proof.
    intros f g a Lf Lg Pg [Hbf HDf] [Hbg HDg].
    split.
    - exact (bounded_comp Lg Lf Hbg Hbf).
    - destruct Hbf as [Cf [HCf HbF]].
      destruct Hbg as [Cg [HCg HbG]].
      intros eps Heps.
      destruct (pos_min_exists 1 (half eps × inv (1 + Cg)) lt_0_1
          (mul_pos (half eps) (inv (1 + Cg)) (half_pos eps Heps)
             (pos_inv (1 + Cg) (one_plus_nonneg_pos Cg HCg))))
        as [e1 [He1 [He1_1 He1_g]]].
      destruct (HDf e1 He1) as [df [Hdf HRf]].
      destruct (HDg (half eps × inv (1 + (1 + Cf)))
          (mul_pos (half eps) (inv (1 + (1 + Cf))) (half_pos eps Heps)
             (pos_inv (1 + (1 + Cf))
                (one_plus_nonneg_pos (1 + Cf)
                   (lt_le 0 (1 + Cf) (one_plus_nonneg_pos Cf HCf))))))
        as [dg [Hdg HRg]].
      destruct (pos_min_exists df (dg × inv (1 + Cf)) Hdf
          (mul_pos dg (inv (1 + Cf)) Hdg
             (pos_inv (1 + Cf) (one_plus_nonneg_pos Cf HCf))))
        as [d [Hd [Hd_f Hd_g]]].
      ∃ d.
      split.
      + exact Hd.
      + intros h Hh.
        apply (le_trans _
          (‖ g (f a + (f (a + h) + - f a))
             + - (g (f a) + Lg (f (a + h) + - f a)) ‖
           + ‖ Lg (f (a + h) + - (f a + Lf h)) ‖) _).
        × exact (comp_remainder_split g Lg Pg (f a) (f (a + h)) (Lf h)).
        × apply (le_trans _ (half eps × ‖ h ‖ + half eps × ‖ h ‖) _).
          -- apply add_le_mono.
             ++ apply (le_trans _
                  ((half eps × inv (1 + (1 + Cf)))
                   × ‖ f (a + h) + - f a ‖) _).
                ** apply HRg.
                   apply (le_lt_trans _ ((1 + Cf) × ‖ h ‖) _).
                   --- exact (deriv_increment_bound f a h Lf e1 Cf HCf HbF
                         He1_1 (HRf h (lt_le_trans (‖ h ‖) d df Hh Hd_f))).
                   --- apply (lt_le_trans _
                         ((1 + Cf) × (dg × inv (1 + Cf))) _).
                       +++ apply mul_lt_mono_l.
                           *** apply one_plus_nonneg_pos. exact HCf.
                           *** exact (lt_le_trans (‖ h ‖) d
                                 (dg × inv (1 + Cf)) Hh Hd_g).
                       +++ apply eq_le.
                           apply field_mul_div_cancel.
                           apply pos_neq_0.
                           apply one_plus_nonneg_pos.
                           exact HCf.
                ** apply (le_trans _
                     ((half eps × inv (1 + (1 + Cf)))
                      × ((1 + Cf) × ‖ h ‖)) _).
                   --- apply mul_le_mono_l.
                       +++ apply mul_nonneg.
                           *** apply lt_le. apply half_pos. exact Heps.
                           *** apply lt_le. apply pos_inv.
                               apply one_plus_nonneg_pos.
                               apply lt_le. apply one_plus_nonneg_pos.
                               exact HCf.
                       +++ exact (deriv_increment_bound f a h Lf e1 Cf HCf
                             HbF He1_1
                             (HRf h (lt_le_trans (‖ h ‖) d df Hh Hd_f))).
                   --- apply (le_rewrite_l
                         (((half eps × inv (1 + (1 + Cf))) × (1 + Cf))
                          × ‖ h ‖)).
                       +++ apply ring_mul_assoc.
                       +++ apply mul_le_mono_r.

Section Partials.
  Context {F U V : Type} `{NormedSpace F U}.
  Context {VSV : VectorSpace F V} {NSV : NormedSpace F V}.

  Definition partial_at {n : nat} (L : LinearMap F U V) (B : Basis n F U)
      (i : Fin.t n) : V :=
    L (basis_vec B i).

  Lemma partial_at_welldef : ∀ (n : nat) (f : U → V) (a : U)
      (L1 L2 : LinearMap F U V) (B : Basis n F U) (i : Fin.t n),
    is_deriv f a L1 → is_deriv f a L2 →
    partial_at L1 B i == partial_at L2 B i.
  Proof.
    intros n f a L1 L2 B i HD1 HD2.
    exact (deriv_unique f a L1 L2 HD1 HD2 (basis_vec B i)).
  Qed.

End Partials.
