  Context {F V : Type} `{PosInner F V}.


  Fixpoint gs_fin_aux (acc : list V) (n : nat)
      {struct n} : (Fin.t n → V) → Fin.t n → V :=
    match n as n0 return (Fin.t n0 → V) → Fin.t n0 → V with
    | O ⇒ fun v i ⇒ v i
    | S m ⇒ fun v ⇒
        let w := gs_aux acc (v Fin.F1) in
        fun i ⇒ Fin.caseS' i (fun _ ⇒ V) w
                   (fun i' ⇒ gs_fin_aux (acc ++ (w :: nil)) m
                                (fun j ⇒ v (Fin.FS j)) i')
    end.

  Definition gs_fin {n : nat} (v : Fin.t n → V) : Fin.t n → V :=
    gs_fin_aux nil n v.

  Lemma gs_fin_aux_F1 : ∀ (acc : list V) (m : nat) (v : Fin.t (S m) → V),
    gs_fin_aux acc (S m) v Fin.F1 == gs_aux acc (v Fin.F1).
  Proof.
    intros acc m v.
    reflexivity.
  Qed.

  Lemma gs_fin_aux_FS : ∀ (acc : list V) (m : nat) (v : Fin.t (S m) → V)
      (i' : Fin.t m),
    gs_fin_aux acc (S m) v (Fin.FS i')
      == gs_fin_aux (acc ++ (gs_aux acc (v Fin.F1) :: nil)) m
           (fun j ⇒ v (Fin.FS j)) i'.
  Proof.
    intros acc m v i'.
    reflexivity.
  Qed.


  Fixpoint fin_indices_before (n : nat) {struct n} : Fin.t n → list (Fin.t n) :=
    match n as n0 return Fin.t n0 → list (Fin.t n0) with
    | O ⇒ fun k ⇒ Fin.case0 (fun _ ⇒ list (Fin.t 0)) k
    | S m ⇒ fun k ⇒
        Fin.caseS' k (fun _ ⇒ list (Fin.t (S m)))
          nil
          (fun k' ⇒ Fin.F1 :: map Fin.FS (fin_indices_before m k'))
    end.

  Lemma fin_indices_before_F1 : ∀ m : nat,
    fin_indices_before (S m) Fin.F1 = nil.
  Proof.
    intros m.
    reflexivity.
  Qed.

  Lemma fin_indices_before_FS : ∀ (m : nat) (k' : Fin.t m),
    fin_indices_before (S m) (Fin.FS k')
      = Fin.F1 :: map Fin.FS (fin_indices_before m k').
  Proof.
    intros m k'.
    reflexivity.
  Qed.


  Lemma map_gs_fin_aux_FS : ∀ (acc : list V) (m : nat)
      (v : Fin.t (S m) → V) (l : list (Fin.t m)),
    map (gs_fin_aux acc (S m) v) (map Fin.FS l)
      = map (gs_fin_aux (acc ++ (gs_aux acc (v Fin.F1) :: nil)) m
               (fun j ⇒ v (Fin.FS j))) l.
  Proof.
    intros acc m v; induction l as [| x l' IH].
    - reflexivity.
    - exact (f_equal2 cons eq_refl IH).
  Qed.

  Lemma gs_fin_aux_eq_rhs_step : ∀ (acc : list V) (m : nat)
      (v : Fin.t (S m) → V) (k' : Fin.t m),
    acc ++ map (gs_fin_aux acc (S m) v) (fin_indices_before (S m) (Fin.FS k'))
      = (acc ++ (gs_aux acc (v Fin.F1) :: nil))
          ++ map (gs_fin_aux (acc ++ (gs_aux acc (v Fin.F1) :: nil)) m
                    (fun j ⇒ v (Fin.FS j))) (fin_indices_before m k').
  Proof.
    intros acc m v k'.
    rewrite (fin_indices_before_FS m k').
    change (map (gs_fin_aux acc (S m) v)
              (Fin.F1 :: map Fin.FS (fin_indices_before m k')))
      with (gs_fin_aux acc (S m) v Fin.F1
              :: map (gs_fin_aux acc (S m) v)
                   (map Fin.FS (fin_indices_before m k'))).
    rewrite (map_gs_fin_aux_FS acc m v (fin_indices_before m k')).
    exact (app_assoc acc (gs_aux acc (v Fin.F1) :: nil)
             (map (gs_fin_aux (acc ++ (gs_aux acc (v Fin.F1) :: nil)) m
                     (fun j ⇒ v (Fin.FS j))) (fin_indices_before m k'))).
  Qed.

  Lemma gs_fin_aux_eq : ∀ (n : nat) (v : Fin.t n → V) (acc : list V)
      (k : Fin.t n),
    gs_fin_aux acc n v k
      == gs_aux (acc ++ map (gs_fin_aux acc n v) (fin_indices_before n k))
           (v k).
  Proof.
    intros n; induction n as [| m IH]; intros v acc k.
    - exact (Fin.case0
               (fun k0 ⇒ gs_fin_aux acc 0 v k0
                  == gs_aux (acc ++ map (gs_fin_aux acc 0 v)
                               (fin_indices_before 0 k0)) (v k0)) k).
    - apply (Fin.caseS' k
               (fun k0 ⇒ gs_fin_aux acc (S m) v k0
                  == gs_aux (acc ++ map (gs_fin_aux acc (S m) v)
                               (fin_indices_before (S m) k0)) (v k0))).
      + change (fin_indices_before (S m) Fin.F1) with (@nil (Fin.t (S m))).
        change (map (gs_fin_aux acc (S m) v) (@nil (Fin.t (S m)))) with (@nil V).
        rewrite (app_nil_r acc).
        exact (gs_fin_aux_F1 acc m v).
      + intros k'.
        rewrite (gs_fin_aux_eq_rhs_step acc m v k'), (gs_fin_aux_FS acc m v k').
        exact (IH (fun j ⇒ v (Fin.FS j)) (acc ++ (gs_aux acc (v Fin.F1) :: nil)) k').
  Qed.


  Lemma fin_indices_before_in : ∀ (n : nat) (k i : Fin.t n),
    In i (fin_indices_before n k) ↔ (fin_to_nat i < fin_to_nat k)%nat.
  Proof.
    intros n; induction n as [| m IH]; intros k i.
    - exact (Fin.case0 (fun k0 ⇒ In i (fin_indices_before 0 k0)
                 ↔ (fin_to_nat i < fin_to_nat k0)%nat) k).
    - apply (Fin.caseS' k
               (fun k0 ⇒ In i (fin_indices_before (S m) k0)
                  ↔ (fin_to_nat i < fin_to_nat k0)%nat)).
      + change (fin_indices_before (S m) Fin.F1) with (@nil (Fin.t (S m))).
        split.
        × intros Hin. destruct Hin.
        × intros Hlt. destruct (Nat.nlt_0_r (fin_to_nat i) Hlt).
      + intros k'.
        rewrite (fin_indices_before_FS m k').
        apply (Fin.caseS' i
                 (fun i0 ⇒ In i0 (Fin.F1 :: map Fin.FS (fin_indices_before m k'))
                    ↔ (fin_to_nat i0 < fin_to_nat (Fin.FS k'))%nat)).
        × split.
          -- intros _.
             rewrite (fin_to_nat_FS m k').
             exact (Nat.lt_0_succ (fin_to_nat k')).
          -- intros _.
             left. reflexivity.
        × intros i'.
          rewrite (fin_to_nat_FS m i'), (fin_to_nat_FS m k').
          split.
          -- intros Hin.
             destruct Hin as [Heq | Hin].
             ++ destruct (fin_FS_neq_F1 m i' (eq_sym Heq)).
             ++ destruct (proj1 (in_map_iff Fin.FS (fin_indices_before m k') (Fin.FS i'))
                            Hin) as [j [Heqj Hj]].
                rewrite (Fin.FS_inj j i' Heqj) in Hj.
                rewrite <- (Nat.succ_lt_mono (fin_to_nat i') (fin_to_nat k')).
                exact (proj1 (IH k' i') Hj).
          -- intros Hlt.
             right.
             apply (in_map_iff Fin.FS (fin_indices_before m k') (Fin.FS i')).
             ∃ i'.
             split.
             ++ reflexivity.
             ++ apply (proj2 (IH k' i')).
                exact (proj2 (Nat.succ_lt_mono (fin_to_nat i') (fin_to_nat k')) Hlt).
  Qed.

  Corollary gs_fin_eq : ∀ (n : nat) (v : Fin.t n → V) (j : Fin.t n),
    gs_fin v j == gs_aux (map (gs_fin v) (fin_indices_before n j)) (v j).
  Proof.
    intros n v j.
    exact (gs_fin_aux_eq n v nil j).
  Qed.


  Lemma gs_fin_orth_bound : ∀ (n : nat) (v : Fin.t n → V),
    (∀ k : Fin.t n, norm2 (gs_fin v k) =/= 0) →
    ∀ p : nat, ∀ j : Fin.t n, (fin_to_nat j < p)%nat →
      ∀ i : Fin.t n, (fin_to_nat i < fin_to_nat j)%nat →
        orthogonal (gs_fin v j) (gs_fin v i).
  Proof.
    intros n v Hnz p; induction p as [| p' IH]; intros j Hjp i Hij.
    - destruct (Nat.nlt_0_r (fin_to_nat j) Hjp).
    - assert (Hjp' : (fin_to_nat j ≤ p')%nat) by (apply Nat.lt_succ_r; exact Hjp).
      assert (Hpw : ∀ d1 d2 : V,
          In d1 (map (gs_fin v) (fin_indices_before n j)) →
          In d2 (map (gs_fin v) (fin_indices_before n j)) →
          orthogonal d1 d2 ∨ d1 = d2).
      { intros d1 d2 Hd1 Hd2.
        destruct (proj1 (in_map_iff (gs_fin v) (fin_indices_before n j) d1) Hd1)
          as [i1 [Heq1 Hi1]].
        destruct (proj1 (in_map_iff (gs_fin v) (fin_indices_before n j) d2) Hd2)
          as [i2 [Heq2 Hi2]].
        destruct (Nat.lt_trichotomy (fin_to_nat i1) (fin_to_nat i2)) as [Hlt | [Heq | Hlt]].
        - left.
          rewrite <- Heq1, <- Heq2.
          apply orthogonal_sym.
          apply (IH i2 (Nat.lt_le_trans (fin_to_nat i2) (fin_to_nat j) p'
                   (proj1 (fin_indices_before_in n j i2) Hi2) Hjp')).
          exact Hlt.
        - right.
          rewrite <- Heq1, <- Heq2, (fin_to_nat_inj n i1 i2 Heq).
          reflexivity.
        - left.
          rewrite <- Heq1, <- Heq2.
          apply (IH i1 (Nat.lt_le_trans (fin_to_nat i1) (fin_to_nat j) p'
                   (proj1 (fin_indices_before_in n j i1) Hi1) Hjp')).
          exact Hlt.
      }
      assert (Hnzd : ∀ d : V,
          In d (map (gs_fin v) (fin_indices_before n j)) → norm2 d =/= 0).
      { intros d Hd.
        destruct (proj1 (in_map_iff (gs_fin v) (fin_indices_before n j) d) Hd)
          as [i0 [Heq0 _]].
        rewrite <- Heq0.
        exact (Hnz i0).
      }
      apply orthogonal_def.
      rewrite (gs_fin_eq n v j).
      apply orthogonal_eq.
      apply (gs_aux_orth (map (gs_fin v) (fin_indices_before n j)) Hpw Hnzd (v j)).
      apply (in_map (gs_fin v) (fin_indices_before n j) i).
      exact (proj2 (fin_indices_before_in n j i) Hij).
  Qed.

  Corollary gs_fin_orth_before : ∀ (n : nat) (v : Fin.t n → V),
    (∀ k : Fin.t n, norm2 (gs_fin v k) =/= 0) →
    ∀ j i : Fin.t n, (fin_to_nat i < fin_to_nat j)%nat →
      orthogonal (gs_fin v j) (gs_fin v i).
  Proof.
    intros n v Hnz j i Hij.
    exact (gs_fin_orth_bound n v Hnz (S (fin_to_nat j)) j
             (Nat.lt_succ_diag_r (fin_to_nat j)) i Hij).
  Qed.


  Lemma inner_sub_proj_eq : ∀ d0 d w : V,
    orthogonal d0 d → ⟨ w + - proj d0 w , d ⟩ == ⟨ w , d ⟩.
  Proof.
    intros d0 d w Hd0d.
    rewrite (ip_add_l w (- proj d0 w) d), (ip_neg_l (proj d0 w) d),
      (proj_eq w d0), (ip_scale_l (⟨ w , d0 ⟩ × inv (norm2 d0)) d0 d),
      (orthogonal_eq d0 d Hd0d), (mul_0_r (⟨ w , d0 ⟩ × inv (norm2 d0))),
      neg_zero.
    apply idr.
  Qed.

  Lemma list_lincomb_map_ext : ∀ (l : list V) (g g' : V → F),
    (∀ d : V, In d l → g d == g' d) →
    list_lincomb (map g l) l == list_lincomb (map g' l) l.
  Proof.
    intros l; induction l as [| d l' IH]; intros g g' Hext.
    - reflexivity.
    - rewrite (list_lincomb_cons (g d) (map g l') d l'),
        (list_lincomb_cons (g' d) (map g' l') d l'),
        (Hext d (or_introl eq_refl)),
        (IH g g' (fun d' Hd' ⇒ Hext d' (or_intror Hd'))).
      reflexivity.
  Qed.

  Definition proj_coeffs (w : V) (done : list V) : list F :=
    map (fun d ⇒ - (⟨ w , d ⟩ × inv (norm2 d))) done.

  Lemma neg_proj_coeff_scale : ∀ (w d : V),
    - (⟨ w , d ⟩ × inv (norm2 d)) *: d == - proj d w.
  Proof.
    intros w d.
    rewrite (neg_scale (⟨ w , d ⟩ × inv (norm2 d)) d).
    rewrite (proj_eq w d).
    reflexivity.
  Qed.

  Lemma gs_aux_orth_reconstruct : ∀ (done : list V),
    NoDup done →
    (∀ d1 d2 : V, In d1 done → In d2 done → d1 ≠ d2 → orthogonal d1 d2) →
    (∀ d : V, In d done → norm2 d =/= 0) →
    ∀ w : V,
      gs_aux done w == w + list_lincomb (proj_coeffs w done) done.
  Proof.
    intros done; induction done as [| d0 done' IH]; intros Hnd Hpw Hnz w.
    - rewrite (gs_aux_nil w).
      unfold proj_coeffs.
      rewrite (list_lincomb_nil_l (nil : list V)), idr.
      reflexivity.
    - rewrite (gs_aux_cons d0 done' w).
      destruct (proj1 (NoDup_cons_iff d0 done') Hnd) as [Hnotin Hnd'].
      assert (Hpw' : ∀ d1 d2 : V,
          In d1 done' → In d2 done' → d1 ≠ d2 → orthogonal d1 d2).
      { intros d1 d2 Hd1 Hd2 Hneq.
        apply (Hpw d1 d2 (or_intror Hd1) (or_intror Hd2) Hneq). }
      assert (Hnz' : ∀ d : V, In d done' → norm2 d =/= 0).
      { intros d Hd. apply (Hnz d (or_intror Hd)). }
      rewrite (IH Hnd' Hpw' Hnz' (w + - proj d0 w)).
      assert (Hext : ∀ d : V, In d done' →
          - (⟨ w + - proj d0 w , d ⟩ × inv (norm2 d))
            == - (⟨ w , d ⟩ × inv (norm2 d))).
      { intros d Hd.
        assert (Hd0d : orthogonal d0 d).
        { apply (Hpw d0 d (or_introl eq_refl) (or_intror Hd)).
          intros Heq.
          apply Hnotin.
          rewrite Heq.
          exact Hd. }
        rewrite (inner_sub_proj_eq d0 d w Hd0d).
        reflexivity. }
      unfold proj_coeffs.
      rewrite (list_lincomb_map_ext done'
                 (fun d ⇒ - (⟨ w + - proj d0 w , d ⟩ × inv (norm2 d)))
                 (fun d ⇒ - (⟨ w , d ⟩ × inv (norm2 d))) Hext).
      change (map (fun d : V ⇒ - (⟨ w , d ⟩ × inv (norm2 d))) (d0 :: done'))
        with ((- (⟨ w , d0 ⟩ × inv (norm2 d0)))
                :: map (fun d : V ⇒ - (⟨ w , d ⟩ × inv (norm2 d))) done').
      rewrite (list_lincomb_cons (- (⟨ w , d0 ⟩ × inv (norm2 d0)))
                 (map (fun d : V ⇒ - (⟨ w , d ⟩ × inv (norm2 d))) done') d0 done'),
        (neg_proj_coeff_scale w d0).
      exact (assoc w (- proj d0 w)
               (list_lincomb (map (fun d : V ⇒ - (⟨ w , d ⟩ × inv (norm2 d))) done')
                  done')).
  Qed.


  Lemma NoDup_map_inj_list : ∀ (A B : Type) (f : A → B) (l : list A),
    (∀ x y : A, In x l → In y l → f x = f y → x = y) →
    NoDup l → NoDup (map f l).
  Proof.
    intros A B f l; induction l as [| x l' IH]; intros Hinj Hnd.
    - exact (NoDup_nil B).
    - destruct (proj1 (NoDup_cons_iff x l') Hnd) as [Hnotin Hnd'].
      apply NoDup_cons.
      + intros Hin.
        destruct (proj1 (in_map_iff f l' (f x)) Hin) as [y [Heqy Hy]].
        pose proof (Hinj x y (or_introl eq_refl) (or_intror Hy) (eq_sym Heqy))
          as Hxy.
        apply Hnotin.
        rewrite Hxy.
        exact Hy.
      + apply (IH (fun a b Ha Hb ⇒ Hinj a b (or_intror Ha) (or_intror Hb)) Hnd').
  Qed.

  Lemma fin_indices_before_NoDup : ∀ (n : nat) (k : Fin.t n),
    NoDup (fin_indices_before n k).
  Proof.
    intros n; induction n as [| m IH]; intros k.
    - exact (Fin.case0 (fun k0 ⇒ NoDup (fin_indices_before 0 k0)) k).
    - apply (Fin.caseS' k (fun k0 ⇒ NoDup (fin_indices_before (S m) k0))).
      + change (fin_indices_before (S m) Fin.F1) with (@nil (Fin.t (S m))).
        exact (NoDup_nil (Fin.t (S m))).
      + intros k'.
        rewrite (fin_indices_before_FS m k').
        apply NoDup_cons.
        × intros Hin.
          destruct (proj1 (in_map_iff Fin.FS (fin_indices_before m k') Fin.F1) Hin)
            as [y [Heqy _]].
          exact (fin_F1_neq_FS m y (eq_sym Heqy)).
        × apply (NoDup_map_inj_list (Fin.t m) (Fin.t (S m)) Fin.FS
                   (fin_indices_before m k')
                   (fun x y _ _ Heq ⇒ Fin.FS_inj x y Heq)
                   (IH k')).
  Qed.

  Lemma gs_fin_distinct : ∀ (n : nat) (v : Fin.t n → V),
    (∀ k : Fin.t n, norm2 (gs_fin v k) =/= 0) →
    ∀ i1 i2 : Fin.t n, (fin_to_nat i1 < fin_to_nat i2)%nat →
      gs_fin v i1 ≠ gs_fin v i2.
  Proof.
    intros n v Hnz i1 i2 Hlt Heq.
    pose proof (gs_fin_orth_before n v Hnz i2 i1 Hlt) as Horth.
    rewrite Heq in Horth.
    exact (Hnz i2 (orthogonal_eq (gs_fin v i2) (gs_fin v i2) Horth)).
  Qed.

  Lemma gs_fin_distinct_ne : ∀ (n : nat) (v : Fin.t n → V),
    (∀ k : Fin.t n, norm2 (gs_fin v k) =/= 0) →
    ∀ i1 i2 : Fin.t n, i1 ≠ i2 → gs_fin v i1 ≠ gs_fin v i2.
  Proof.
    intros n v Hnz i1 i2 Hne.
    destruct (Nat.lt_trichotomy (fin_to_nat i1) (fin_to_nat i2))
      as [Hlt | [Heq | Hlt]].
    - exact (gs_fin_distinct n v Hnz i1 i2 Hlt).
    - destruct (Hne (fin_to_nat_inj n i1 i2 Heq)).
    - intros Heq.
      exact (gs_fin_distinct n v Hnz i2 i1 Hlt (eq_sym Heq)).
  Qed.

  Lemma gs_fin_prefix_NoDup : ∀ (n : nat) (v : Fin.t n → V),
    (∀ k : Fin.t n, norm2 (gs_fin v k) =/= 0) →
    ∀ j : Fin.t n, NoDup (map (gs_fin v) (fin_indices_before n j)).
  Proof.
    intros n v Hnz j.
    apply (NoDup_map_inj_list (Fin.t n) V (gs_fin v) (fin_indices_before n j)).
    - intros x y _ _ Heq.
      destruct (Fin.eq_dec x y) as [Heqxy | Hnexy].
      + exact Heqxy.
      + destruct (gs_fin_distinct_ne n v Hnz x y Hnexy Heq).
    - exact (fin_indices_before_NoDup n j).
  Qed.

  Lemma gs_fin_prefix_pairwise : ∀ (n : nat) (v : Fin.t n → V),
    (∀ k : Fin.t n, norm2 (gs_fin v k) =/= 0) →
    ∀ (j : Fin.t n) (d1 d2 : V),
      In d1 (map (gs_fin v) (fin_indices_before n j)) →
      In d2 (map (gs_fin v) (fin_indices_before n j)) →
      d1 ≠ d2 → orthogonal d1 d2.
  Proof.
    intros n v Hnz j d1 d2 Hd1 Hd2 Hneq.
    destruct (proj1 (in_map_iff (gs_fin v) (fin_indices_before n j) d1) Hd1)
      as [i1 [Heq1 _]].
    destruct (proj1 (in_map_iff (gs_fin v) (fin_indices_before n j) d2) Hd2)
      as [i2 [Heq2 _]].
    destruct (Nat.lt_trichotomy (fin_to_nat i1) (fin_to_nat i2))
      as [Hlt | [Heqi | Hlt]].
    - rewrite <- Heq1, <- Heq2.
      apply orthogonal_sym.
      exact (gs_fin_orth_before n v Hnz i2 i1 Hlt).
    - destruct (Hneq (eq_trans (eq_sym Heq1)
                 (eq_trans (f_equal (gs_fin v) (fin_to_nat_inj n i1 i2 Heqi)) Heq2))).
    - rewrite <- Heq1, <- Heq2.
      exact (gs_fin_orth_before n v Hnz i1 i2 Hlt).
  Qed.


  Lemma if_lt_dec_0_S : ∀ (X : Type) (b : nat) (x y : X),
    (if lt_dec 0 (S b) then x else y) = x.
  Proof.
    intros X b x y.
    destruct (lt_dec 0 (S b)) as [_ | Hnlt].
    - reflexivity.
    - destruct (Hnlt (Nat.lt_0_succ b)).
  Qed.

  Lemma if_lt_dec_succ : ∀ (X : Type) (a b : nat) (x y : X),
    (if lt_dec (S a) (S b) then x else y) = (if lt_dec a b then x else y).
  Proof.
    intros X a b x y.
    destruct (lt_dec (S a) (S b)) as [Hlt1 | Hnlt1];
      destruct (lt_dec a b) as [Hlt2 | Hnlt2].
    - reflexivity.
    - destruct (Hnlt2 (proj2 (Nat.succ_lt_mono a b) Hlt1)).
    - destruct (Hnlt1 (proj1 (Nat.succ_lt_mono a b) Hlt2)).
    - reflexivity.
  Qed.

  Lemma if_lt_dec_nlt_0 : ∀ (X : Type) (a : nat) (x y : X),
    (if lt_dec a 0 then x else y) = y.
  Proof.
    intros X a x y.
    destruct (lt_dec a 0) as [Hlt | _].
    - destruct (Nat.nlt_0_r a Hlt).
    - reflexivity.
  Qed.

  Lemma if_lt_dec_nlt_0_scale : ∀ (a : nat) (x : F) (w : V),
    (if lt_dec a 0 then x else 0) *: w == 0.
  Proof.
    intros a x w.
    rewrite (if_lt_dec_nlt_0 F a x 0).
    exact (zero_scale w).
  Qed.

  Lemma list_lincomb_prefix_bigopF : ∀ (n : nat) (f : Fin.t n → V)
      (g : Fin.t n → F) (j : Fin.t n),
    list_lincomb (map g (fin_indices_before n j)) (map f (fin_indices_before n j))
      == bigopF (fun i : Fin.t n ⇒
           (if lt_dec (fin_to_nat i) (fin_to_nat j) then g i else 0) *: f i).
  Proof.
    intros n; induction n as [| m IH]; intros f g j.
    - exact (Fin.case0
               (fun j0 ⇒ list_lincomb (map g (fin_indices_before 0 j0))
                            (map f (fin_indices_before 0 j0))
                  == bigopF (fun i : Fin.t 0 ⇒
                       (if lt_dec (fin_to_nat i) (fin_to_nat j0) then g i else 0)
                         *: f i)) j).
    - apply (Fin.caseS' j
               (fun j0 ⇒ list_lincomb (map g (fin_indices_before (S m) j0))
                            (map f (fin_indices_before (S m) j0))
                  == bigopF (fun i : Fin.t (S m) ⇒
                       (if lt_dec (fin_to_nat i) (fin_to_nat j0) then g i else 0)
                         *: f i))).
      + change (fin_indices_before (S m) Fin.F1) with (@nil (Fin.t (S m))).
        change (map g (@nil (Fin.t (S m)))) with (@nil F).
        change (map f (@nil (Fin.t (S m)))) with (@nil V).
        rewrite (list_lincomb_nil_l (nil : list V)).
        symmetry.
        rewrite (bigopF_ext (S m)
                   (fun i : Fin.t (S m) ⇒
                      (if lt_dec (fin_to_nat i) (fin_to_nat Fin.F1) then g i else 0)
                        *: f i)
                   (fun _ : Fin.t (S m) ⇒ (0 : V))
                   (fun i ⇒ if_lt_dec_nlt_0_scale (fin_to_nat i) (g i) (f i))).
        exact (bigopF_zero (S m)).
      + intros j'.
        rewrite (fin_indices_before_FS m j').
        change (map g (Fin.F1 :: map Fin.FS (fin_indices_before m j')))
          with (g Fin.F1 :: map g (map Fin.FS (fin_indices_before m j'))).
        change (map f (Fin.F1 :: map Fin.FS (fin_indices_before m j')))
          with (f Fin.F1 :: map f (map Fin.FS (fin_indices_before m j'))).
        rewrite (list_lincomb_cons (g Fin.F1) (map g (map Fin.FS (fin_indices_before m j')))
                   (f Fin.F1) (map f (map Fin.FS (fin_indices_before m j')))),
          (map_map Fin.FS g (fin_indices_before m j')),
          (map_map Fin.FS f (fin_indices_before m j')),
          (IH (fun i ⇒ f (Fin.FS i)) (fun i ⇒ g (Fin.FS i)) j').
        rewrite (bigopF_S m (fun i : Fin.t (S m) ⇒
                     (if lt_dec (fin_to_nat i) (fin_to_nat (Fin.FS j')) then g i else 0)
                       *: f i)).
        apply mon_add_proper.
        × change (fin_to_nat Fin.F1) with 0%nat.
          rewrite (fin_to_nat_FS m j').
          symmetry.
          rewrite (if_lt_dec_0_S F (fin_to_nat j') (g Fin.F1) 0).
          reflexivity.
        × apply bigopF_ext.
          intros i'.
          rewrite (fin_to_nat_FS m i'), (fin_to_nat_FS m j').
          rewrite (if_lt_dec_succ F (fin_to_nat i') (fin_to_nat j') (g (Fin.FS i')) 0).
          reflexivity.
  Qed.


  Lemma solve_for_b : ∀ a b x : V, a == b + x → b == a + - x.
  Proof.
    intros a b x Heq.
    rewrite Heq, assoc, invr.
    symmetry.
    exact (idr b).
  Qed.


  Theorem gs_fin_reconstruct : ∀ (n : nat) (v : Fin.t n → V),
    (∀ k : Fin.t n, norm2 (gs_fin v k) =/= 0) →
    ∀ j : Fin.t n,
      v j == gs_fin v j
        + bigopF (fun i : Fin.t n ⇒
            (if lt_dec (fin_to_nat i) (fin_to_nat j)
               then ⟨ v j , gs_fin v i ⟩ × inv (norm2 (gs_fin v i)) else 0)
              *: gs_fin v i).
  Proof.
    intros n v Hnz j.
    assert (Heq : gs_fin v j == v j
        + bigopF (fun i : Fin.t n ⇒
            (if lt_dec (fin_to_nat i) (fin_to_nat j)
               then - (⟨ v j , gs_fin v i ⟩ × inv (norm2 (gs_fin v i))) else 0)
              *: gs_fin v i)).
    { rewrite (gs_fin_eq n v j).
      rewrite (gs_aux_orth_reconstruct (map (gs_fin v) (fin_indices_before n j))
                 (gs_fin_prefix_NoDup n v Hnz j)
                 (gs_fin_prefix_pairwise n v Hnz j)
                 (fun d Hd ⇒
                    match proj1 (in_map_iff (gs_fin v) (fin_indices_before n j) d) Hd
                    with
                    | ex_intro _ i0 (conj Heq0 _) ⇒
                        eq_ind_r (fun x ⇒ norm2 x =/= 0) (Hnz i0) (eq_sym Heq0)
                    end)
                 (v j)).
      unfold proj_coeffs.
      rewrite (map_map (gs_fin v) (fun d ⇒ - (⟨ v j , d ⟩ × inv (norm2 d)))
                 (fin_indices_before n j)).
      apply mon_add_proper.
      - reflexivity.
      - exact (list_lincomb_prefix_bigopF n (gs_fin v)
                 (fun i ⇒ - (⟨ v j , gs_fin v i ⟩ × inv (norm2 (gs_fin v i))))
                 j). }
    transitivity (gs_fin v j + - bigopF (fun i : Fin.t n ⇒
        (if lt_dec (fin_to_nat i) (fin_to_nat j)
           then - (⟨ v j , gs_fin v i ⟩ × inv (norm2 (gs_fin v i))) else 0)
          *: gs_fin v i)).
    - exact (solve_for_b (gs_fin v j) (v j) _ Heq).
    - apply mon_add_proper.
      + reflexivity.
      + rewrite <- (bigopF_neg n (fun i ⇒
          (if lt_dec (fin_to_nat i) (fin_to_nat j)
             then - (⟨ v j , gs_fin v i ⟩ × inv (norm2 (gs_fin v i))) else 0)
            *: gs_fin v i)).
        apply bigopF_ext.
        intros i.
        destruct (lt_dec (fin_to_nat i) (fin_to_nat j)) as [Hlt | Hnlt].
        × rewrite (neg_scale (⟨ v j , gs_fin v i ⟩ × inv (norm2 (gs_fin v i)))
                     (gs_fin v i)).
          exact (neg_involutive
                   ((⟨ v j , gs_fin v i ⟩ × inv (norm2 (gs_fin v i))) *: gs_fin v i)).
        × rewrite (zero_scale (gs_fin v i)).
          exact neg_zero.
  Qed.

End GSFin.


Section QR.
  Context {F V : Type} `{PosInner F V}.
  Context {n : nat} (B : Basis n F V).

  Definition qr_Q (v : Fin.t n → V) : Matrix n n F :=
    fun i j ⇒ basis_coords B (gs_fin v j) i.

  Definition qr_R_coeff (v : Fin.t n → V) (j k : Fin.t n) : F :=
    if lt_dec (fin_to_nat k) (fin_to_nat j)
      then ⟨ v j , gs_fin v k ⟩ × inv (norm2 (gs_fin v k)) else 0.

  Definition qr_R (v : Fin.t n → V) : Matrix n n F :=
    fun i j ⇒ e j i + qr_R_coeff v j i.

  Definition qr_A (v : Fin.t n → V) : Matrix n n F :=
    fun i j ⇒ basis_coords B (v j) i.


  Theorem qr_R_upper : ∀ (v : Fin.t n → V) (i j : Fin.t n),
    (fin_to_nat j < fin_to_nat i)%nat → qr_R v i j == 0.
  Proof.
    intros v i j Hji.
    assert (Hne : j ≠ i).
    { intros Heq. rewrite Heq in Hji. exact (Nat.lt_irrefl (fin_to_nat i) Hji). }
    unfold qr_R, qr_R_coeff.
    rewrite (e_at_neq n j i Hne).
    destruct (lt_dec (fin_to_nat i) (fin_to_nat j)) as [Hlt | _].
    - destruct (Nat.lt_asymm (fin_to_nat j) (fin_to_nat i) Hji Hlt).
    - exact (idl (0 : F)).
  Qed.

  Theorem qr_R_diag : ∀ (v : Fin.t n → V) (i : Fin.t n), qr_R v i i == 1.
  Proof.
    intros v i.
    unfold qr_R, qr_R_coeff.
    rewrite (e_at_eq n i i eq_refl).
    destruct (lt_dec (fin_to_nat i) (fin_to_nat i)) as [Hlt | _].
    - destruct (Nat.lt_irrefl (fin_to_nat i) Hlt).
    - exact (idr (1 : F)).
  Qed.

  Theorem qr_R_diag_nonzero : ∀ (v : Fin.t n → V) (i : Fin.t n),
    qr_R v i i =/= 0.
  Proof.
    intros v i Hcontra.
    apply field_nontrivial.
    transitivity (qr_R v i i).
    - symmetry. exact (qr_R_diag v i).
    - exact Hcontra.
  Qed.


  Lemma coords_bigopF : ∀ (m : nat) (c : Fin.t m → F) (w : Fin.t m → V)
      (i : Fin.t n),
    basis_coords B (bigopF (fun l ⇒ c l *: w l)) i
      == bigopF (fun l ⇒ c l × basis_coords B (w l) i).
  Proof.
    intros m c w i.
    transitivity (bigopF (fun l ⇒ basis_coords B (c l *: w l)) i).
    - exact (lm_bigopF (basis_coords B) m (fun l ⇒ c l *: w l) i).
    - transitivity (bigopF (fun l ⇒ basis_coords B (c l *: w l) i)).
      + exact (bigopF_app m (fun l ⇒ basis_coords B (c l *: w l)) i).
      + apply bigopF_ext.
        intros l.
        exact (coords_scale B (c l) (w l) i).
  Qed.

  Lemma qr_Q_e_collapse : ∀ (v : Fin.t n → V) (i j : Fin.t n),
    bigopF (fun k ⇒ qr_Q v i k × e j k) == basis_coords B (gs_fin v j) i.
  Proof.
    intros v i j.
    transitivity (bigopF (fun k ⇒ basis_coords B (gs_fin v k) i × e k j)).
    - apply bigopF_ext.
      intros k.
      rewrite (e_sym n j k).
      reflexivity.
    - exact (lincomb_std_delta n (fun k ⇒ basis_coords B (gs_fin v k) i) j).
  Qed.

  Definition qr_common (v : Fin.t n → V) (i j : Fin.t n) : F :=
    basis_coords B (gs_fin v j) i
      + bigopF (fun k ⇒ qr_R_coeff v j k × basis_coords B (gs_fin v k) i).

  Lemma qr_mat_mul_common : ∀ (v : Fin.t n → V) (i j : Fin.t n),
    mat_mul (qr_Q v) (qr_R v) i j == qr_common v i j.
  Proof.
    intros v i j.
    unfold mat_mul, qr_common.
    transitivity (bigopF (fun k ⇒ qr_Q v i k × e j k
        + qr_Q v i k × qr_R_coeff v j k)).
    - apply bigopF_ext.
      intros k.
      exact (ring_distl (qr_Q v i k) (e j k) (qr_R_coeff v j k)).
    - rewrite (bigopF_add n (fun k ⇒ qr_Q v i k × e j k)
                 (fun k ⇒ qr_Q v i k × qr_R_coeff v j k)).
      apply mon_add_proper.
      + exact (qr_Q_e_collapse v i j).
      + apply bigopF_ext.
        intros k.
        unfold qr_Q.
        exact (field_mul_comm (basis_coords B (gs_fin v k) i) (qr_R_coeff v j k)).
  Qed.

  Lemma qr_A_common : ∀ (v : Fin.t n → V),
    (∀ k : Fin.t n, norm2 (gs_fin v k) =/= 0) →
    ∀ i j : Fin.t n, qr_A v i j == qr_common v i j.
  Proof.
    intros v Hnz i j.
    unfold qr_A, qr_common.
    transitivity (basis_coords B (gs_fin v j
        + bigopF (fun k ⇒ qr_R_coeff v j k *: gs_fin v k)) i).
    - apply (coords_proper B).
      exact (gs_fin_reconstruct n v Hnz j).
    - rewrite (coords_add B (gs_fin v j)
                 (bigopF (fun k ⇒ qr_R_coeff v j k *: gs_fin v k)) i).
      apply mon_add_proper.
      + reflexivity.
      + exact (coords_bigopF n (qr_R_coeff v j) (gs_fin v) i).
  Qed.


  Theorem qr_factorization : ∀ (v : Fin.t n → V),
    (∀ k : Fin.t n, norm2 (gs_fin v k) =/= 0) →
    qr_A v == mat_mul (qr_Q v) (qr_R v).
  Proof.
    intros v Hnz i j.
    transitivity (qr_common v i j).
    - exact (qr_A_common v Hnz i j).
    - symmetry.
      exact (qr_mat_mul_common v i j).
  Qed.

End QR.


Section QRSmoke.
  Context {F V : Type} `{PosInner F V}.
  Context (B : Basis 2 F V).
  Variable v0 v1 : V.

  Definition qr_smoke_v : Fin.t 2 → V :=
    fun i ⇒ Fin.caseS' i (fun _ ⇒ V) v0 (fun _ ⇒ v1).

  Example qr_smoke_statement : Prop :=
    qr_A B qr_smoke_v == mat_mul (qr_Q B qr_smoke_v) (qr_R qr_smoke_v).

  Hypothesis Hnz : ∀ k : Fin.t 2, norm2 (gs_fin qr_smoke_v k) =/= 0.

  Example qr_smoke_factorization :
    qr_A B qr_smoke_v == mat_mul (qr_Q B qr_smoke_v) (qr_R qr_smoke_v) :=
    qr_factorization B qr_smoke_v Hnz.

  Example qr_smoke_upper :
    qr_R qr_smoke_v (Fin.FS Fin.F1) Fin.F1 == 0 :=
    qr_R_upper qr_smoke_v (Fin.FS Fin.F1) Fin.F1 (Nat.lt_0_succ 0).

  Example qr_smoke_diag_F1 : qr_R qr_smoke_v Fin.F1 Fin.F1 == 1 :=
    qr_R_diag qr_smoke_v Fin.F1.

  Example qr_smoke_diag_FS : qr_R qr_smoke_v (Fin.FS Fin.F1) (Fin.FS Fin.F1) == 1 :=
    qr_R_diag qr_smoke_v (Fin.FS Fin.F1).

  Example qr_smoke_offdiag :
    qr_R qr_smoke_v Fin.F1 (Fin.FS Fin.F1)
      == e (n := 2) (Fin.FS Fin.F1) Fin.F1
           + ⟨ v1 , gs_fin qr_smoke_v Fin.F1 ⟩
               × inv (norm2 (gs_fin qr_smoke_v Fin.F1)).
  Proof.
    reflexivity.
  Qed.

End QRSmoke.
