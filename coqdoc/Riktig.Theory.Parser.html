Section AbstractSeqPrimitives.
  Context {C S : Type}.
  Implicit Types (seq : Seq C S) (pred : C → bool) (input rest : S) (c : C).

  Lemma satisfy_sound : ∀ seq pred input c rest,
    satisfy seq pred input = Some (c, rest) →
    uncons seq input = Some (c, rest) ∧ pred c = true.
  Proof.
    intros seq pred input c rest H.
    unfold satisfy in H.
    destruct (uncons seq input) as [[c0 rest0] |] eqn:Hun; [ | discriminate].
    destruct (pred c0) eqn:Hpred; [ | discriminate].
    injection H as → →.
    split; [reflexivity | exact Hpred].
  Qed.

  Lemma satisfy_complete : ∀ seq pred input c rest,
    uncons seq input = Some (c, rest) → pred c = true →
    satisfy seq pred input = Some (c, rest).
  Proof.
    intros seq pred input c rest Hun Hpred.
    unfold satisfy. rewrite Hun, Hpred. reflexivity.
  Qed.

  Lemma satisfy_spec : ∀ seq pred input c rest,
    satisfy seq pred input = Some (c, rest) ↔
    uncons seq input = Some (c, rest) ∧ pred c = true.
  Proof.
    intros seq pred input c rest. split.
    - apply satisfy_sound.
    - intros [Hun Hpred]. apply satisfy_complete; assumption.
  Qed.

  Lemma satisfy_fail_iff : ∀ seq pred input,
    satisfy seq pred input = None ↔
    uncons seq input = None ∨
    (∃ c rest, uncons seq input = Some (c, rest) ∧ pred c = false).
  Proof.
    intros seq pred input. unfold satisfy.
    destruct (uncons seq input) as [[c rest] |] eqn:Hun.
    - destruct (pred c) eqn:Hpred; split.
      + discriminate.
      + intros [H | (c' & rest' & Heq & Hf)].
        × discriminate.
        × injection Heq as → →. congruence.
      + intros _. right. ∃ c, rest. split; [reflexivity | exact Hpred].
      + intros _. reflexivity.
    - split; intros _; [left; reflexivity | reflexivity].
  Qed.

  Lemma any_spec : ∀ seq input c rest,
    any seq input = Some (c, rest) ↔ uncons seq input = Some (c, rest).
  Proof.
    intros seq input c rest. unfold any. rewrite satisfy_spec.
    split; [intros [H _] | intros H]; auto.
  Qed.

  Lemma exact_spec : ∀ seq eq_dec input c c' rest,
    exact seq eq_dec c input = Some (c', rest) ↔
    uncons seq input = Some (c', rest) ∧ eq_dec c c' = true.
  Proof. intros. unfold exact. apply satisfy_spec. Qed.

  Lemma eos_succeeds : ∀ seq input,
    uncons seq input = None → eos seq input = Some (tt, input).
  Proof. intros seq input H. unfold eos. rewrite H. reflexivity. Qed.

  Lemma eos_fails : ∀ seq input c rest,
    uncons seq input = Some (c, rest) → eos seq input = None.
  Proof. intros seq input c rest H. unfold eos. rewrite H. reflexivity. Qed.

  Lemma eos_spec : ∀ seq input,
    eos seq input = Some (tt, input) ↔ uncons seq input = None.
  Proof.
    intros seq input. unfold eos.
    destruct (uncons seq input) as [[c rest] |] eqn:Hun.
    - split; discriminate.
    - split; intros _; reflexivity.
  Qed.

  Lemma eos_none_iff : ∀ seq input,
    eos seq input = None ↔ ∃ c rest, uncons seq input = Some (c, rest).
  Proof.
    intros seq input. unfold eos.
    destruct (uncons seq input) as [[c rest] |] eqn:Hun.
    - split; [intros _; ∃ c, rest; reflexivity | intros _; reflexivity].
    - split; [discriminate | intros (c & rest & Hc); discriminate].
  Qed.

End AbstractSeqPrimitives.


Lemma stringSeq_uncons_iff : ∀ input c rest,
  uncons stringSeq input = Some (c, rest) ↔ input = String c rest.
Proof.
  intros input c rest. unfold stringSeq, uncons.
  destruct input as [| c0 rest0]; simpl.
  - split; discriminate.
  - split.
    + intros H. injection H as → →. reflexivity.
    + intros H. injection H as → →. reflexivity.
Qed.

Lemma char_spec : ∀ c input c' rest,
  char c input = Some (c', rest) ↔ input = String c' rest ∧ c = c'.
Proof.
  intros c input c' rest. unfold char.
  rewrite exact_spec, stringSeq_uncons_iff, Ascii.eqb_eq.
  reflexivity.
Qed.

Lemma char_succeeds_iff : ∀ c input rest,
  char c input = Some (c, rest) ↔ input = String c rest.
Proof.
  intros c input rest. rewrite char_spec.
  split.
  - intros [H _]. exact H.
  - intros H. split; [exact H | reflexivity].
Qed.

Lemma char_result_is_c : ∀ c input c' rest,
  char c input = Some (c', rest) → c' = c.
Proof.
  intros c input c' rest H.
  rewrite char_spec in H.
  destruct H as [_ Hc].
  symmetry. exact Hc.
Qed.


Lemma string_append_assoc : ∀ s1 s2 s3 : string,
  (s1 ++ s2) ++ s3 = s1 ++ (s2 ++ s3).
Proof.
  induction s1 as [| c s1 IH]; intros s2 s3; simpl.
  - reflexivity.
  - rewrite IH. reflexivity.
Qed.


Definition ParserSound {T : Type} (p : Parser string T) : Prop :=
  ∀ s v s', p s = Some (v, s') → ∃ prefix, s = prefix ++ s'.

Lemma ret_ParserSound : ∀ T (x : T), ParserSound (ret x : Parser string T).
Proof.
  intros T x s v s' H.
  unfold ret, Parser_HasReturn in H.
  injection H as <- <-.
  ∃ EmptyString. reflexivity.
Qed.

Lemma bind_ParserSound : ∀ T U (p : Parser string T) (f : T → Parser string U),
  ParserSound p → (∀ x, ParserSound (f x)) → ParserSound (bind p f).
Proof.
  intros T U p f Hp Hf s v s' H.
  unfold bind, Parser_HasBind in H.
  destruct (p s) as [[x rest] |] eqn:Hps; [ | discriminate].
  destruct (Hp s x rest Hps) as [prefix1 Heq1].
  destruct (Hf x rest v s' H) as [prefix2 Heq2].
  ∃ (prefix1 ++ prefix2).
  rewrite Heq1, Heq2, string_append_assoc.
  reflexivity.
Qed.

Lemma map_ParserSound : ∀ T U (f : T → U) (p : Parser string T),
  ParserSound p → ParserSound (map f p).
Proof.
  intros T U f p Hp. unfold map. apply bind_ParserSound.
  - exact Hp.
  - intros x. apply ret_ParserSound.
Qed.

Lemma alt_ParserSound : ∀ T (p q : Parser string T),
  ParserSound p → ParserSound q → ParserSound (alt p q).
Proof.
  intros T p q Hp Hq s v s' H.
  unfold alt, Parser_HasAlt in H.
  remember (p s) as ps eqn:Hps.
  destruct ps as [[v0 s0] |].
  - simpl in H.
    injection H as Heqv Heqs'.
    rewrite <- Heqs'.
    exact (Hp s v0 s0 (eq_sym Hps)).
  - simpl in H.
    exact (Hq s v s' H).
Qed.

Lemma seqL_ParserSound : ∀ T U (p : Parser string T) (q : Parser string U),
  ParserSound p → ParserSound q → ParserSound (seqL p q).
Proof.
  intros T U p q Hp Hq. unfold seqL. apply bind_ParserSound.
  - exact Hp.
  - intros x. apply bind_ParserSound.
    + exact Hq.
    + intros _. apply ret_ParserSound.
Qed.

Lemma seqR_ParserSound : ∀ T U (p : Parser string T) (q : Parser string U),
  ParserSound p → ParserSound q → ParserSound (seqR p q).
Proof.
  intros T U p q Hp Hq. unfold seqR. apply bind_ParserSound.
  - exact Hp.
  - intros _. exact Hq.
Qed.

Lemma between_ParserSound : ∀ A B T
    (open : Parser string A) (close : Parser string B) (p : Parser string T),
  ParserSound open → ParserSound close → ParserSound p →
  ParserSound (between open close p).
Proof.
  intros A B T open close p Hopen Hclose Hp. unfold between.
  apply bind_ParserSound.
  - exact Hopen.
  - intros _. apply bind_ParserSound.
    + exact Hp.
    + intros x. apply bind_ParserSound.
      × exact Hclose.
      × intros _. apply ret_ParserSound.
Qed.

Lemma optional_ParserSound : ∀ T (p : Parser string T),
  ParserSound p → ParserSound (optional p).
Proof.
  intros T p Hp. unfold optional. apply alt_ParserSound.
  - apply map_ParserSound. exact Hp.
  - apply ret_ParserSound.
Qed.

Lemma satisfy_stringSeq_ParserSound : ∀ pred,
  ParserSound (satisfy stringSeq pred).
Proof.
  intros pred s v s' H. apply satisfy_sound in H as [Hun _].
  apply stringSeq_uncons_iff in Hun. subst s.
  ∃ (String v ""). reflexivity.
Qed.

Lemma any_stringSeq_ParserSound : ParserSound (any stringSeq).
Proof. unfold any. apply satisfy_stringSeq_ParserSound. Qed.

Lemma exact_stringSeq_ParserSound : ∀ (eq_dec : ascii → ascii → bool) (c : ascii),
  ParserSound (exact stringSeq eq_dec c).
Proof. intros eq_dec c. unfold exact. apply satisfy_stringSeq_ParserSound. Qed.

Lemma char_ParserSound : ∀ c, ParserSound (char c).
Proof.
  intros c s v s' H.
  rewrite char_spec in H. destruct H as [Hs Hcv].
  ∃ (String c ""). rewrite Hs, Hcv. reflexivity.
Qed.


Section AlwaysSucceeds.
  Context {S : Type}.

  Definition AlwaysSucceeds {A} (p : Parser S A) : Prop := ∀ s, ∃ r, p s = Some r.

  Lemma ret_always_succeeds : ∀ A (x : A), AlwaysSucceeds (ret x : Parser S A).
  Proof. intros A x s. ∃ (x, s). reflexivity. Qed.

  Lemma bind_always_succeeds : ∀ A B (m : Parser S A) (g : A → Parser S B),
    AlwaysSucceeds m → (∀ v rest, ∃ r, g v rest = Some r) →
    ∀ s, ∃ r, bind m g s = Some r.
  Proof.
    intros A B m g Hm Hg s.
    unfold bind, Parser_HasBind.
    destruct (Hm s) as [[v rest] Hv].
    rewrite Hv. apply Hg.
  Qed.

  Lemma alt_always_succeeds_of_r : ∀ A (p q : Parser S A),
    AlwaysSucceeds q → ∀ s, ∃ r, alt p q s = Some r.
  Proof.
    intros A p q Hq s.
    unfold alt, Parser_HasAlt.
    destruct (p s) as [pr |] eqn:Hp.
    - ∃ pr. reflexivity.
    - apply Hq.
  Qed.

  Lemma bind_never_fails_after_success : ∀ A B (p : Parser S A) (f : A → Parser S B),
    (∀ v rest, ∃ r, f v rest = Some r) →
    ∀ s, bind p f s = None ↔ p s = None.
  Proof.
    intros A B p f Hf s.
    unfold bind, Parser_HasBind.
    destruct (p s) as [[v rest] |] eqn:Hp.
    - destruct (Hf v rest) as [r Hr]. rewrite Hr. split; discriminate.
    - split; reflexivity.
  Qed.

End AlwaysSucceeds.


Lemma many_never_fails : ∀ T (p : Parser string T) fuel input,
  ∃ r, many p fuel input = Some r.
Proof.
  intros T p fuel input.
  destruct fuel as [| c fuel'].
  - simpl. apply ret_always_succeeds.
  - simpl. apply alt_always_succeeds_of_r. apply ret_always_succeeds.
Qed.

Lemma many_step_none : ∀ T (p : Parser string T) (c : ascii) (fuel' s : string),
  p s = None → many p (String c fuel') s = Some ([], s).
Proof.
  intros T p c fuel' s Hp.
  simpl. unfold alt, Parser_HasAlt, bind, Parser_HasBind.
  rewrite Hp. reflexivity.
Qed.

Lemma many_step_some : ∀ T (p : Parser string T) (c : ascii) (fuel' s : string) (v : T) (s' : string),
  p s = Some (v, s') →
  ∃ xs rest, many p fuel' s' = Some (xs, rest) ∧
                  many p (String c fuel') s = Some (v :: xs, rest).
Proof.
  intros T p c fuel' s v s' Hp.
  destruct (many_never_fails T p fuel' s') as [[xs rest] Hm].
  ∃ xs, rest. split; [exact Hm | ].
  simpl. unfold alt, Parser_HasAlt, bind, Parser_HasBind.
  rewrite Hp, Hm. reflexivity.
Qed.

Lemma many_ParserSound : ∀ T (p : Parser string T) fuel,
  ParserSound p → ParserSound (many p fuel).
Proof.
  intros T p fuel Hp.
  induction fuel as [| c fuel' IH]; simpl.
  - apply ret_ParserSound.
  - apply alt_ParserSound.
    + apply bind_ParserSound.
      × exact Hp.
      × intros x. apply bind_ParserSound.
        -- exact IH.
        -- intros xs. apply ret_ParserSound.
    + apply ret_ParserSound.
Qed.

Lemma many1_ParserSound : ∀ T (p : Parser string T) fuel,
  ParserSound p → ParserSound (many1 p fuel).
Proof.
  intros T p fuel Hp. unfold many1. apply bind_ParserSound.
  - exact Hp.
  - intros x. apply bind_ParserSound.
    + apply many_ParserSound. exact Hp.
    + intros xs. apply ret_ParserSound.
Qed.

Lemma many1_fails_iff : ∀ T (p : Parser string T) fuel input,
  many1 p fuel input = None ↔ p input = None.
Proof.
  intros T p fuel input. unfold many1.
  apply bind_never_fails_after_success.
  intros v rest. apply bind_always_succeeds.
  - intros s. apply many_never_fails.
  - intros xs rest'. apply ret_always_succeeds.
Qed.

Definition HeadConsuming {T : Type} (p : Parser string T) : Prop :=
  ∀ s v s', p s = Some (v, s') → ∃ c, s = String c s'.

Lemma satisfy_stringSeq_HeadConsuming : ∀ pred, HeadConsuming (satisfy stringSeq pred).
Proof.
  intros pred s v s' H. apply satisfy_sound in H as [Hun _].
  apply stringSeq_uncons_iff in Hun. ∃ v. exact Hun.
Qed.

Lemma any_stringSeq_HeadConsuming : HeadConsuming (any stringSeq).
Proof. unfold any. apply satisfy_stringSeq_HeadConsuming. Qed.

Lemma exact_stringSeq_HeadConsuming : ∀ (eq_dec : ascii → ascii → bool) (c : ascii),
  HeadConsuming (exact stringSeq eq_dec c).
Proof. intros eq_dec c. unfold exact. apply satisfy_stringSeq_HeadConsuming. Qed.

Lemma char_HeadConsuming : ∀ c, HeadConsuming (char c).
Proof. intros c s v s' H. apply char_spec in H as [-> _]. ∃ v. reflexivity. Qed.

Theorem many_canonical_unfold : ∀ T (p : Parser string T), HeadConsuming p →
  ∀ s, many p s s =
    match p s with
    | None ⇒ Some ([], s)
    | Some (v, s') ⇒
      match many p s' s' with
      | Some (xs, rest) ⇒ Some (v :: xs, rest)
      | None ⇒ None
      end
    end.
Proof.
  intros T p Hhc s.
  destruct s as [| c s0].
  - destruct (p "") as [[v s'] |] eqn:Hp.
    + destruct (Hhc _ _ _ Hp) as [c Heq]. discriminate Heq.
    + reflexivity.
  - destruct (p (String c s0)) as [[v s1] |] eqn:Hp.
    + destruct (Hhc _ _ _ Hp) as [c' Heq].
      injection Heq as _ Hs01.
      rewrite <- Hs01 in Hp.
      rewrite <- Hs01.
      destruct (many_step_some T p c s0 (String c s0) v s0 Hp) as (xs & rest & Hmfuel & Heq').
      rewrite Hmfuel. exact Heq'.
    + exact (many_step_none T p c s0 (String c s0) Hp).
Qed.

Lemma many_fuel_invariant : ∀ T (p : Parser string T), HeadConsuming p →
  ∀ fuel s, String.length s ≤ String.length fuel → many p fuel s = many p s s.
Proof.
  intros T p Hhc fuel.
  induction fuel as [| d fuel0 IH]; intros s Hlen.
  - destruct s as [| e s0]; [reflexivity | ].
    simpl in Hlen. inversion Hlen.
  - destruct s as [| e s0].
    + destruct (p "") as [[v s'] |] eqn:Hp.
      × destruct (Hhc _ _ _ Hp) as [c Heq]. discriminate Heq.
      × exact (many_step_none T p d fuel0 "" Hp).
    + simpl in Hlen. apply le_S_n in Hlen.
      destruct (p (String e s0)) as [[v s1] |] eqn:Hp.
      × destruct (Hhc _ _ _ Hp) as [c Heq].
        injection Heq as _ Hs01.
        pose proof (many_step_some T p d fuel0 (String e s0) v s1 Hp) as (xsL & restL & HmL & HeqL).
        pose proof (many_step_some T p e s0 (String e s0) v s1 Hp) as (xsR & restR & HmR & HeqR).
        rewrite <- Hs01 in HmL.
        rewrite <- Hs01 in HmR.
        rewrite (IH s0 Hlen) in HmL.
        rewrite HmL in HmR.
        injection HmR as HxsEq HrestEq.
        rewrite HeqL, HeqR, HxsEq, HrestEq. reflexivity.
      × rewrite (many_step_none T p d fuel0 (String e s0) Hp).
        rewrite (many_step_none T p e s0 (String e s0) Hp).
        reflexivity.
Qed.

Theorem many1_canonical_unfold : ∀ T (p : Parser string T), HeadConsuming p →
  ∀ s, many1 p s s =
    match p s with
    | None ⇒ None
    | Some (v, s') ⇒
      match many p s' s' with
      | Some (xs, rest) ⇒ Some (v :: xs, rest)
      | None ⇒ None
      end
    end.
Proof.
  intros T p Hhc s.
  destruct (p s) as [[v s'] |] eqn:Hp.
  - assert (Hlen : String.length s' ≤ String.length s).
    { destruct (Hhc _ _ _ Hp) as [c Heq]. subst s. simpl. lia. }
    unfold many1, bind, Parser_HasBind.
    rewrite Hp. simpl. unfold bind, Parser_HasBind.
    rewrite (many_fuel_invariant T p Hhc s s' Hlen).
    destruct (many p s' s') as [[xs rest] |].
    + unfold ret, Parser_HasReturn. reflexivity.
    + reflexivity.
  - unfold many1, bind, Parser_HasBind. rewrite Hp. reflexivity.
Qed.


Lemma sepBy1_fails_iff : ∀ T U (p : Parser string T) (sep : Parser string U) fuel input,
  sepBy1 p sep fuel input = None ↔ p input = None.
Proof.
  intros T U p sep fuel input.
  destruct fuel as [| c fuel'].
  - simpl. apply bind_never_fails_after_success. intros v rest. apply ret_always_succeeds.
  - simpl. apply bind_never_fails_after_success. intros v rest.
    apply alt_always_succeeds_of_r. apply ret_always_succeeds.
Qed.

Lemma sepBy1_ParserSound : ∀ T U (p : Parser string T) (sep : Parser string U) fuel,
  ParserSound p → ParserSound sep → ParserSound (sepBy1 p sep fuel).
Proof.
  intros T U p sep fuel Hp Hsep.
  induction fuel as [| c fuel' IH]; simpl.
  - apply bind_ParserSound.
    + exact Hp.
    + intros x. apply ret_ParserSound.
  - apply bind_ParserSound.
    + exact Hp.
    + intros x. apply alt_ParserSound.
      × apply bind_ParserSound.
        -- exact Hsep.
        -- intros _. apply bind_ParserSound.
           ++ exact IH.
           ++ intros xs. apply ret_ParserSound.
      × apply ret_ParserSound.
Qed.

Lemma skipMany_never_fails : ∀ T (p : Parser string T) fuel input,
  ∃ r, skipMany p fuel input = Some r.
Proof.
  intros T p fuel input.
  destruct fuel as [| c fuel'].
  - simpl. apply ret_always_succeeds.
  - simpl. apply alt_always_succeeds_of_r. apply ret_always_succeeds.
Qed.

Lemma skipMany_ParserSound : ∀ T (p : Parser string T) fuel,
  ParserSound p → ParserSound (skipMany p fuel).
Proof.
  intros T p fuel Hp.
  induction fuel as [| c fuel' IH]; simpl.
  - apply ret_ParserSound.
  - apply alt_ParserSound.
    + apply bind_ParserSound.
      × exact Hp.
      × intros _. exact IH.
    + apply ret_ParserSound.
Qed.


Lemma run_spec : ∀ S T (p : Parser S T) input v,
  run p input = Some v ↔ ∃ rest, p input = Some (v, rest).
Proof.
  intros S T p input v. unfold run.
  destruct (p input) as [[v0 rest] |] eqn:Hp.
  - split.
    + intros H. injection H as <-. ∃ rest. reflexivity.
    + intros [rest' H]. injection H as <- <-. reflexivity.
  - split.
    + discriminate.
    + intros [rest' H]. discriminate.
Qed.

Lemma runExact_sound : ∀ S C T (seq : Seq C S) (p : Parser S T) input v,
  runExact seq p input = Some v →
  ∃ rest, p input = Some (v, rest) ∧ uncons seq rest = None.
Proof.
  intros S C T seq p input v H. unfold runExact in H.
  destruct (p input) as [[v0 rest] |] eqn:Hp; [ | discriminate].
  destruct (uncons seq rest) as [[c rest'] |] eqn:Hun; [discriminate | ].
  injection H as →.
  ∃ rest. split; [reflexivity | exact Hun].
Qed.

Lemma runExact_complete : ∀ S C T (seq : Seq C S) (p : Parser S T) input v rest,
  p input = Some (v, rest) → uncons seq rest = None →
  runExact seq p input = Some v.
Proof.
  intros S C T seq p input v rest Hp Hun. unfold runExact.
  rewrite Hp, Hun. reflexivity.
Qed.
