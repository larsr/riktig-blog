Library Riktig.Theory.Log

From Coq Require Import Morphisms.
From Riktig Require Import Ops Props Interface.Field Interface.Order
  Interface.Exp Theory.Ring Theory.Field Theory.Order Theory.Exp.

Section Log.
  Context {F : Type} `{Exp F}.

  Context (exp_surj : ∀ y : F, 0 < y → {x : F | exp_fn x == y}).

  Definition log_fn (y : F) (Hy : 0 < y) : F := proj1_sig (exp_surj y Hy).

  Lemma exp_log : ∀ (y : F) (Hy : 0 < y), exp_fn (log_fn y Hy) == y.
  Proof.
    intros y Hy.
    exact (proj2_sig (exp_surj y Hy)).
  Qed.

  Lemma log_proof_irrel : ∀ (y : F) (Hy1 Hy2 : 0 < y),
    log_fn y Hy1 == log_fn y Hy2.
  Proof.
    intros y Hy1 Hy2.
    apply exp_injective.
    rewrite (exp_log y Hy1), (exp_log y Hy2).
    reflexivity.
  Qed.

  Lemma log_exp : ∀ x : F, log_fn (exp_fn x) (exp_strict_pos x) == x.
  Proof.
    intros x.
    apply exp_injective.
    rewrite (exp_log (exp_fn x) (exp_strict_pos x)).
    reflexivity.
  Qed.

  Lemma log_injective : ∀ (a b : F) (Ha : 0 < a) (Hb : 0 < b),
    log_fn a Ha == log_fn b Hb → a == b.
  Proof.
    intros a b Ha Hb Heq.
    rewrite <- (exp_log a Ha), <- (exp_log b Hb), Heq.
    reflexivity.
  Qed.

  Lemma log_one : ∀ Hlt01 : (0 : F) < 1, log_fn 1 Hlt01 == 0.
  Proof.
    intros Hlt01.
    apply exp_injective.
    rewrite (exp_log 1 Hlt01), exp_zero.
    reflexivity.
  Qed.

  Lemma log_mul : ∀ (a b : F) (Ha : 0 < a) (Hb : 0 < b),
    log_fn (a × b) (mul_pos a b Ha Hb) == log_fn a Ha + log_fn b Hb.
  Proof.
    intros a b Ha Hb.
    apply exp_injective.
    rewrite (exp_log (a × b) (mul_pos a b Ha Hb)), exp_add,
      (exp_log a Ha), (exp_log b Hb).
    reflexivity.
  Qed.

  Lemma log_inv : ∀ (a : F) (Ha : 0 < a),
    log_fn (inv a) (pos_inv a Ha) == - log_fn a Ha.
  Proof.
    intros a Ha.
    apply exp_injective.
    rewrite (exp_log (inv a) (pos_inv a Ha)), (exp_neg (log_fn a Ha)),
      (exp_log a Ha).
    reflexivity.
  Qed.

  Lemma exp_lt_reflect : ∀ a b : F, exp_fn a < exp_fn b → a < b.
  Proof.
    intros a b [Hle Hne].
    split.
    - apply le_stable.
      intros Hnab.
      destruct (le_total a b) as [Hab' | Hba].
      + exact (Hnab Hab').
      + assert (Hbane : b =/= a).
        { intros Heqba. apply Hnab. rewrite Heqba. apply le_refl. }
        pose proof (conj Hba Hbane) as Hblt.
        destruct (exp_mono b a Hblt) as [Hle2 Hne2].
        exact (Hne (le_antisym _ _ Hle Hle2)).
    - intros Heqab.
      apply Hne.
      rewrite Heqab.
      reflexivity.
  Qed.

  Lemma log_mono : ∀ (a b : F) (Ha : 0 < a) (Hb : 0 < b),
    a < b → log_fn a Ha < log_fn b Hb.
  Proof.
    intros a b Ha Hb Hab.
    apply exp_lt_reflect.
    rewrite (exp_log a Ha), (exp_log b Hb).
    exact Hab.
  Qed.

End Log.
