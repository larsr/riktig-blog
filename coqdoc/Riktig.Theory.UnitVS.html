#[global] Instance Unit_EqRel : EqRel unit := fun _ _ ⇒ True.
#[global] Instance Unit_Zero : Zero unit := tt.
#[global] Instance Unit_Add : HasAdd unit := fun _ _ ⇒ tt.
#[global] Instance Unit_Neg : HasNeg unit := fun _ ⇒ tt.
#[global] Instance Unit_Scale {F : Type} : HasScale F unit := fun _ _ ⇒ tt.

#[global] Instance Unit_Equivalence : Equivalence Unit_EqRel.
Proof.
  apply Build_Equivalence.
  - intros a. exact I.
  - intros a b Hab. exact I.
  - intros a b c Hab Hbc. exact I.
Qed.


Lemma unit_assoc : ∀ a b c : unit, (a + b) + c == a + (b + c).
Proof. intros a b c. exact I. Qed.

Lemma unit_idl : ∀ a : unit, 0 + a == a.
Proof. intros a. exact I. Qed.

Lemma unit_idr : ∀ a : unit, a + 0 == a.
Proof. intros a. exact I. Qed.

#[global] Instance unit_add_proper : Proper (equ ==> equ ==> equ) (@add unit _).
Proof. intros a a' Ha b b' Hb. exact I. Qed.

#[global] Instance Unit_Monoid : Monoid unit := {|
  mon_equ := Unit_EqRel;
  mon_equiv := Unit_Equivalence;
  mon_zero := Unit_Zero;
  mon_add := Unit_Add;
  mon_assoc := unit_assoc;
  mon_idl := unit_idl;
  mon_idr := unit_idr;
  mon_add_proper := unit_add_proper
|}.


Lemma unit_invl : ∀ a : unit, (- a) + a == 0.
Proof. intros a. exact I. Qed.

Lemma unit_invr : ∀ a : unit, a + (- a) == 0.
Proof. intros a. exact I. Qed.

#[global] Instance unit_neg_proper : Proper (equ ==> equ) (@neg unit _).
Proof. intros a a' Ha. exact I. Qed.

#[global] Instance Unit_Group : Group unit := {|
  grp_monoid := Unit_Monoid;
  grp_neg := Unit_Neg;
  grp_invl := unit_invl;
  grp_invr := unit_invr;
  grp_neg_proper := unit_neg_proper
|}.


Lemma unit_comm : ∀ a b : unit, a + b == b + a.
Proof. intros a b. exact I. Qed.

#[global] Instance Unit_AbGroup : AbGroup unit := {|
  ab_group := Unit_Group;
  ab_comm := unit_comm
|}.


Section UnitVectorSpace.
  Context {F : Type} `{Field F}.

  #[global] Instance unit_sc_proper : Proper (equ ==> equ ==> equ) (@scale F unit _).
  Proof. intros a a' Ha u u' Hu. exact I. Qed.

  Lemma unit_scale_add : ∀ (a : F) (u v : unit),
    a *: (u + v) == a *: u + a *: v.
  Proof. intros a u v. exact I. Qed.

  Lemma unit_add_scale : ∀ (a b : F) (u : unit),
    (a + b) *: u == a *: u + b *: u.
  Proof. intros a b u. exact I. Qed.

  Lemma unit_scale_scale : ∀ (a b : F) (u : unit),
    (a × b) *: u == a *: (b *: u).
  Proof. intros a b u. exact I. Qed.

  Lemma unit_scale_1 : ∀ u : unit, (1 : F) *: u == u.
  Proof. intros u. exact I. Qed.

  #[global] Instance Unit_ScaleLaws : ScaleLaws F unit :=
    @Build_ScaleLaws F unit _ Unit_AbGroup Unit_Scale
      unit_sc_proper unit_scale_add unit_add_scale unit_scale_scale
      unit_scale_1.

  #[global] Instance Unit_VectorSpace : VectorSpace F unit :=
    @Build_VectorSpace F unit _ Unit_AbGroup Unit_Scale Unit_ScaleLaws.

End UnitVectorSpace.
