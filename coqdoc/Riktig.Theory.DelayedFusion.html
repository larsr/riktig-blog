Library Riktig.Theory.DelayedFusion

From Coq Require Import Morphisms.
From Riktig Require Import Ops Props Interface.Monoid Lib.Calc.
From Riktig Require Import Impl.Zr.


Inductive Tree (A : Type) : Type :=
| Leaf (a : A)
| Node (l r : Tree A).

Arguments Leaf {A} a.
Arguments Node {A} l r.

Inductive DTree (M A : Type) : Type :=
| DLeaf (a : A)
| DNode (l r : DTree M A)
| DDelay (m : M) (t : DTree M A).

Arguments DLeaf {M A} a.
Arguments DNode {M A} l r.
Arguments DDelay {M A} m t.

Section DelayedFusion.
  Context {M : Type} `{Monoid M}.

  Context {A : Type} {AE : EqRel A} {AEq : Equivalence (@equ A AE)}.

  Context (act : M → A → A).
  Context (act_proper : Proper (equ ==> equ ==> equ) act).
  Context (act_unit : ∀ a : A, act 0 a == a).
  Context (act_compose : ∀ (m n : M) (a : A),
    act m (act n a) == act (m + n) a).


  Inductive tree_equiv : Tree A → Tree A → Prop :=
  | TE_Leaf : ∀ a b : A, a == b → tree_equiv (Leaf a) (Leaf b)
  | TE_Node : ∀ l l' r r' : Tree A,
      tree_equiv l l' → tree_equiv r r' →
      tree_equiv (Node l r) (Node l' r').

  Lemma tree_equiv_refl : Reflexive tree_equiv.
  Proof.
    intros t; induction t as [a | l IHl r IHr].
    - apply TE_Leaf. reflexivity.
    - apply TE_Node; [exact IHl | exact IHr].
  Qed.

  Lemma tree_equiv_sym : Symmetric tree_equiv.
  Proof.
    intros t u Htu.
    induction Htu as [a b Hab | l l' r r' Hl IHl Hr IHr].
    - apply TE_Leaf. symmetry. exact Hab.
    - apply TE_Node; [exact IHl | exact IHr].
  Qed.

  Lemma tree_equiv_trans : Transitive tree_equiv.
  Proof.
    intros t u v Htu.
    revert v.
    induction Htu as [a b Hab | l l' r r' Hl IHl Hr IHr]; intros v Huv.
    - inversion Huv as [b' c Hbc | ]. subst.
      apply TE_Leaf. transitivity b; [exact Hab | exact Hbc].
    - inversion Huv as [| l2 l3 r2 r3 Hl2 Hr2]. subst.
      apply TE_Node; [apply IHl; exact Hl2 | apply IHr; exact Hr2].
  Qed.

  Instance tree_equiv_Equivalence : Equivalence tree_equiv :=
    Build_Equivalence tree_equiv tree_equiv_refl tree_equiv_sym tree_equiv_trans.

  Instance Leaf_proper : Proper (equ ==> tree_equiv) (@Leaf A).
  Proof.
    intros a b Hab. apply TE_Leaf. exact Hab.
  Qed.

  Instance Node_proper :
    Proper (tree_equiv ==> tree_equiv ==> tree_equiv) (@Node A).
  Proof.
    intros l l' Hl r r' Hr. apply TE_Node; [exact Hl | exact Hr].
  Qed.


  Fixpoint map_act (m : M) (t : Tree A) : Tree A :=
    match t with
    | Leaf a ⇒ Leaf (act m a)
    | Node l r ⇒ Node (map_act m l) (map_act m r)
    end.

  Instance map_act_proper : Proper (equ ==> tree_equiv ==> tree_equiv) map_act.
  Proof.
    intros m m' Hm t u Htu.
    induction Htu as [a b Hab | l l' r r' Hl IHl Hr IHr].
    - apply TE_Leaf. exact (act_proper m m' Hm a b Hab).
    - apply TE_Node; [exact IHl | exact IHr].
  Qed.

  Theorem map_act_unit : ∀ t : Tree A, tree_equiv (map_act 0 t) t.
  Proof.
    intros t; induction t as [a | l IHl r IHr].
    - apply TE_Leaf. exact (act_unit a).
    - apply TE_Node; [exact IHl | exact IHr].
  Qed.

  Theorem map_act_fuse : ∀ (m n : M) (t : Tree A),
    tree_equiv (map_act m (map_act n t)) (map_act (m + n) t).
  Proof.
    intros m n t; induction t as [a | l IHl r IHr].
    - apply TE_Leaf. exact (act_compose m n a).
    - apply TE_Node; [exact IHl | exact IHr].
  Qed.


  Fixpoint force_acc (m : M) (t : DTree M A) : Tree A :=
    match t with
    | DLeaf a ⇒ Leaf (act m a)
    | DNode l r ⇒ Node (force_acc m l) (force_acc m r)
    | DDelay n u ⇒ force_acc (m + n) u
    end.

  Definition force (t : DTree M A) : Tree A := force_acc 0 t.

  Definition delay (m : M) (t : DTree M A) : DTree M A :=
    match t with
    | DDelay n u ⇒ DDelay (m + n) u
    | _ ⇒ DDelay m t
    end.

  Fixpoint lift (t : Tree A) : DTree M A :=
    match t with
    | Leaf a ⇒ DLeaf a
    | Node l r ⇒ DNode (lift l) (lift r)
    end.


  Lemma force_acc_ext : ∀ (m m' : M) (t : DTree M A),
    m == m' → tree_equiv (force_acc m t) (force_acc m' t).
  Proof.
    intros m m' t; revert m m'.
    induction t as [a | l IHl r IHr | n u IH]; intros m m' Hmm'.
    - apply TE_Leaf. apply act_proper; [exact Hmm' | reflexivity].
    - apply TE_Node; [apply IHl; exact Hmm' | apply IHr; exact Hmm'].
    - simpl. apply IH. apply monoid_add_proper; [exact Hmm' | reflexivity].
  Qed.

  Lemma force_acc_compose : ∀ (m n : M) (t : DTree M A),
    tree_equiv (force_acc (m + n) t) (map_act m (force_acc n t)).
  Proof.
    intros m n t; revert m n.
    induction t as [a | l IHl r IHr | k u IH]; intros m n.
    -
      apply TE_Leaf. symmetry. exact (act_compose m n a).
    - apply TE_Node; [apply IHl | apply IHr].
    -
      simpl.
      calc (force_acc ((m + n) + k) u)
        == (force_acc (m + (n + k)) u)
             { (force_acc_ext ((m + n) + k) (m + (n + k)) u (monoid_associative m n k)) }
        == (map_act m (force_acc (n + k) u))
             { (IH m (n + k)) }.
  Qed.

  Theorem force_acc_correct : ∀ (m : M) (t : DTree M A),
    tree_equiv (force_acc m t) (map_act m (force t)).
  Proof.
    intros m t.
    calc (force_acc m t)
      == (force_acc (m + 0) t)
           { (force_acc_ext (m + 0) m t (monoid_right_identity m)) }
      == (map_act m (force t))
           { (force_acc_compose m 0 t) }.
  Qed.

  Lemma force_acc_delay : ∀ (k m : M) (t : DTree M A),
    tree_equiv (force_acc k (delay m t)) (force_acc (k + m) t).
  Proof.
    intros k m t; destruct t as [a | l r | n u].
    - reflexivity.
    - reflexivity.
    -
      symmetry.
      exact (force_acc_ext ((k + m) + n) (k + (m + n)) u (monoid_associative k m n)).
  Qed.

  Lemma zero_add_swap : ∀ m : M, 0 + m == m + 0.
  Proof.
    intros m.
    calc (0 + m)
      == m { idl }
      == (m + 0) { idr }.
  Qed.


  Theorem force_delay : ∀ (m : M) (t : DTree M A),
    tree_equiv (force (delay m t)) (map_act m (force t)).
  Proof.
    intros m t.
    calc (force (delay m t))
      == (force_acc (0 + m) t)
           { (force_acc_delay 0 m t) }
      == (force_acc (m + 0) t)
           { (force_acc_ext (0 + m) (m + 0) t (zero_add_swap m)) }
      == (map_act m (force t))
           { (force_acc_compose m 0 t) }.
  Qed.


  Lemma delay_DDelay : ∀ (m n : M) (t : DTree M A),
    delay m (DDelay n t) = DDelay (m + n) t.
  Proof.
    intros m n t. reflexivity.
  Qed.

  Lemma delay_delay_def : ∀ (m n : M) (t : DTree M A),
    ∃ (k : M) (u : DTree M A),
      delay n t = DDelay k u ∧ delay m (delay n t) = DDelay (m + k) u.
  Proof.
    intros m n t; destruct t as [a | l r | p u].
    - ∃ n, (DLeaf a). split; reflexivity.
    - ∃ n, (DNode l r). split; reflexivity.
    - ∃ (n + p), u. split; reflexivity.
  Qed.

  Theorem delay_delay : ∀ (m n : M) (t : DTree M A),
    tree_equiv (force (delay m (delay n t))) (force (delay (m + n) t)).
  Proof.
    intros m n t.
    calc (force (delay m (delay n t)))
      == (map_act m (force (delay n t)))
           { (force_delay m (delay n t)) }
      == (map_act m (map_act n (force t)))
           { (force_delay n t) }
      == (map_act (m + n) (force t))
           { (map_act_fuse m n (force t)) }
      == (force (delay (m + n) t))
           { (force_delay (m + n) t) }.
  Qed.


  Lemma force_lift : ∀ t : Tree A, tree_equiv (force (lift t)) t.
  Proof.
    intros t; induction t as [a | l IHl r IHr].
    - apply TE_Leaf. exact (act_unit a).
    - apply TE_Node; [exact IHl | exact IHr].
  Qed.

  Corollary map_map_delay : ∀ (m n : M) (u : Tree A),
    tree_equiv (map_act m (map_act n u)) (force (delay m (delay n (lift u)))).
  Proof.
    intros m n u.
    symmetry.
    calc (force (delay m (delay n (lift u))))
      == (map_act m (force (delay n (lift u))))
           { (force_delay m (delay n (lift u))) }
      == (map_act m (map_act n (force (lift u))))
           { (force_delay n (lift u)) }
      == (map_act m (map_act n u))
           { (force_lift u) }.
  Qed.

  Corollary fusion_via_delay : ∀ (m n : M) (u : Tree A),
    tree_equiv (map_act m (map_act n u)) (map_act (m + n) u).
  Proof.
    intros m n u.
    calc (map_act m (map_act n u))
      == (force (delay m (delay n (lift u))))
           { (map_map_delay m n u) }
      == (force (delay (m + n) (lift u)))
           { (delay_delay m n (lift u)) }
      == (map_act (m + n) (force (lift u)))
           { (force_delay (m + n) (lift u)) }
      == (map_act (m + n) u)
           { (force_lift u) }.
  Qed.

End DelayedFusion.


Section ZExample.
  Local Open Scope Z_scope.

  Definition ztree : Tree Z := Node (Leaf 1) (Node (Leaf 2) (Leaf 3)).

  Example zdelay_combines :
    delay 5 (delay 7 (lift ztree))
    = DDelay 12 (DNode (DLeaf 1) (DNode (DLeaf 2) (DLeaf 3))).
  Proof.
    vm_compute. reflexivity.
  Qed.

  Example zforce :
    force Z.add (delay 5 (delay 7 (lift ztree)))
    = Node (Leaf 13) (Node (Leaf 14) (Leaf 15)).
  Proof.
    vm_compute. reflexivity.
  Qed.

  Example zfuse :
    map_act Z.add 5 (map_act Z.add 7 ztree)
    = force Z.add (delay 5 (delay 7 (lift ztree))).
  Proof.
    vm_compute. reflexivity.
  Qed.

End ZExample.
