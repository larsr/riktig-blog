Library Riktig.Theory.Product

From Riktig Require Import Ops Interface.Category Theory.Category.

Section ProductTheory.
  Context {O : Type} {C : Category O} `{@HasProducts O C}.

  Theorem pair_eta : ∀ (x a b : O) (h : x ~> prod_obj a b),
      pair_mor (pi1 ∘ h) (pi2 ∘ h) == h.
  Proof.
    intros x a b h.
    symmetry.
    apply pair_unique; reflexivity.
  Qed.

  Theorem pair_comp : ∀ (x y a b : O) (f : y ~> a) (g : y ~> b) (h : x ~> y),
      pair_mor f g ∘ h == pair_mor (f ∘ h) (g ∘ h).
  Proof.
    intros x y a b f g h.
    apply pair_unique.
    - rewrite cat_assoc, pair_pi1. reflexivity.
    - rewrite cat_assoc, pair_pi2. reflexivity.
  Qed.

  Theorem par_pi1 : ∀ (a b c d : O) (f : a ~> c) (g : b ~> d),
      pi1 ∘ (f ×× g) == f ∘ pi1.
  Proof.
    intros a b c d f g.
    apply pair_pi1.
  Qed.

  Theorem par_pi2 : ∀ (a b c d : O) (f : a ~> c) (g : b ~> d),
      pi2 ∘ (f ×× g) == g ∘ pi2.
  Proof.
    intros a b c d f g.
    apply pair_pi2.
  Qed.

  Theorem pair_pi_id : ∀ a b : O,
      pair_mor pi1 pi2 == (cat_id : prod_obj a b ~> prod_obj a b).
  Proof.
    intros a b.
    symmetry.
    apply pair_unique; apply cat_idr.
  Qed.

  Lemma swap_invol : ∀ a b : O,
      pair_mor pi2 pi1 ∘ pair_mor pi2 pi1
        == (cat_id : prod_obj a b ~> prod_obj a b).
  Proof.
    intros a b.
    rewrite pair_comp, pair_pi2, pair_pi1.
    apply pair_pi_id.
  Qed.

  Definition prod_swap (a b : O) : Iso (prod_obj a b) (prod_obj b a) := {|
    iso_to := pair_mor pi2 pi1;
    iso_from := pair_mor pi2 pi1;
    iso_fromto := swap_invol a b;
    iso_tofrom := swap_invol b a
  |}.

End ProductTheory.
