Library Riktig.Interface.GeometricAlgebra

From Coq Require Import Morphisms.
From Riktig Require Import Ops Interface.Ring Interface.VectorSpace
  Interface.InnerProduct.

Class GeometricAlgebra (F V A : Type) `{InnerProduct F V} := {
  #[global] ga_ring :: Ring A;
  #[global] ga_scale :: HasScale F A;
  #[global] ga_scale_laws :: ScaleLaws F A;
  ga_scale_mul_l : ∀ (a : F) (x y : A), (a *: x) × y == a *: (x × y);
  ga_scale_mul_r : ∀ (a : F) (x y : A), x × (a *: y) == a *: (x × y);
  ga_vec : V → A;
  #[global] ga_vec_proper :: Proper (equ ==> equ) ga_vec;
  ga_vec_add : ∀ u v : V, ga_vec (u + v) == ga_vec u + ga_vec v;
  ga_vec_scale : ∀ (a : F) (u : V), ga_vec (a *: u) == a *: ga_vec u;
  ga_contract : ∀ u : V, ga_vec u × ga_vec u == norm2 u *: 1
}.
