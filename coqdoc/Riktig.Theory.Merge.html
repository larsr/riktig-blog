Library Riktig.Theory.Merge


Require Import Arith Lia.
Require Import List Permutation Sorted. Import ListNotations.
From Riktig Require Import Interface.Decidable Lib.Calc Lib.MeasureInd.
From Riktig Require Import Interface.DecTotalOrder Theory.DecTotalOrder Theory.Sorted.
Require Import Riktig.Programs.Merge Riktig.Impl.Orders.

Section MergeProps.
  Context {A : Type} `{O : DecTotalOrder A}.


  Lemma merge_perm: ∀ a b : list A, Permutation (a ++ b) (merge a b).
    apply (measure_ind2 (fun (a b : list A) ⇒ length a + length b)).
    intros [|x a] [|y b] IH.
    - reflexivity.
    - reflexivity.
    - now rewrite app_nil_r.
    - destruct (dec (olt x y)) as [L|L].
      + calc ((x::a) ++ (y::b))
          == (x :: (a ++ (y::b)))
          == (x :: merge a (y::b))
          == (merge (x::a) (y::b)).
        × reflexivity.
        × apply perm_skip, IH; simpl; lia.
        × rewrite merge_cons; destruct (dec (olt x y)); [reflexivity | contradiction].
      + calc ((x::a) ++ (y::b))
          == (y :: ((x::a) ++ b))
          == (y :: merge (x::a) b)
          == (merge (x::a) (y::b)).
        × apply Permutation_sym, Permutation_middle.
        × apply perm_skip, IH; simpl; lia.
        × rewrite merge_cons; destruct (dec (olt x y)); [contradiction | reflexivity].
   Qed.

  Lemma merge_comm_perm: ∀ a b : list A, Permutation (merge a b) (merge b a).
    intros a b.
    calc (merge a b) == (a ++ b) == (b ++ a) == (merge b a).
    - symmetry; apply merge_perm.
    - apply Permutation_app_comm.
    - apply merge_perm.
  Qed.


  Lemma merge_hdrel: ∀ (x : A) a b,
      HdRel ole x a → HdRel ole x b → HdRel ole x (merge a b).
    intros x [|p a] [|q b] HA HB;
      rewrite ?merge_nil_l, ?merge_nil_r, ?merge_cons; auto.
    destruct (dec (olt p q)); constructor; [now inversion HA | now inversion HB].
   Qed.

  Lemma merge_sorted: ∀ a b : list A,
      Sorted ole a → Sorted ole b → Sorted ole (merge a b).
   Proof.
    apply (measure_ind2 (fun (a b : list A) ⇒ length a + length b)
             (fun a b ⇒ Sorted ole a → Sorted ole b → Sorted ole (merge a b))).
    intros [|x a] [|y b] IH SA SB.
    - constructor.
    - now rewrite merge_nil_l.
    - now rewrite merge_nil_r.
    - rewrite merge_cons; inversion_clear SA; inversion_clear SB;
        destruct (dec (olt x y)); constructor;
        eauto using merge_hdrel, HdRel_cons, olt_ole, nolt_ole, Sorted_cons;
        apply IH; simpl; eauto using Sorted_cons; lia.
     Qed.

  Lemma merge_comm: ∀ a b : list A,
      Sorted ole a → Sorted ole b → merge a b = merge b a.
    intros; apply sorted_perm_unique;
      eauto using merge_sorted, merge_comm_perm.
   Qed.

End MergeProps.

Example merge_comm_nat: ∀ a b : list nat,
    Sorted Peano.le a → Sorted Peano.le b → merge a b = merge b a.
Proof. exact (merge_comm (O:=Nat_DecTotalOrder)). Qed.

Check Logic.eq_refl : merge [1] [1;0] = [1;0;1].
Check Logic.eq_refl : merge [1;0] [1] = [1;1;0].
