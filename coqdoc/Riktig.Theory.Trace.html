Section MatrixTrace.
  Context {F : Type} {FF : Field F}.

  Definition trM {n : nat} (A : Matrix n n F) : F :=
    bigopF (fun i ⇒ A i i).

  Lemma trM_proper : ∀ n : nat, Proper (equ ==> equ) (@trM n).
  Proof.
    intros n A A' HA.
    exact (bigopF_ext n (fun i ⇒ A i i) (fun i ⇒ A' i i)
             (fun i ⇒ HA i i)).
  Qed.


  Lemma trM_mul_expand : ∀ (m n : nat) (A : Matrix m n F)
      (B : Matrix n m F),
    trM (mat_mul A B)
      == bigopF (fun i : Fin.t m ⇒ bigopF (fun j : Fin.t n ⇒ A i j × B j i)).
  Proof.
    intros m n A B. reflexivity.
  Qed.

  Theorem tr_mul_comm_rect : ∀ (m n : nat) (A : Matrix m n F)
      (B : Matrix n m F),
    trM (mat_mul A B) == trM (mat_mul B A).
  Proof.
    intros m n A B.
    rewrite (trM_mul_expand m n A B), (trM_mul_expand n m B A).
    transitivity (bigopF (fun j : Fin.t n ⇒
        bigopF (fun i : Fin.t m ⇒ A i j × B j i))).
    - exact (bigopF_swap m n (fun i j ⇒ A i j × B j i)).
    - apply bigopF_ext. intros j.
      apply bigopF_ext. intros i.
      exact (field_mul_comm (A i j) (B j i)).
  Qed.

  Theorem tr_mul_comm : ∀ (n : nat) (A B : Matrix n n F),
    trM (mat_mul A B) == trM (mat_mul B A).
  Proof.
    intros n A B.
    exact (tr_mul_comm_rect n n A B).
  Qed.

End MatrixTrace.


Section MatOfId.
  Context {F U : Type} {FF : Field F} {VU : VectorSpace F U}.
  Context {n : nat} (B : Basis n F U).

  Theorem mat_of_id : mat_of B B (@lm_id F U FF VU) == mat_id.
  Proof.
    intros i j.
    transitivity (@e F FF n j i).
    - exact (coords_vec B j i).
    - exact (e_sym n j i).
  Qed.

End MatOfId.


Section ChangeOfBasis.
  Context {F U : Type} {FF : Field F} {VU : VectorSpace F U}.
  Context {n : nat}.

  Definition cob (B B' : Basis n F U) : Matrix n n F :=
    mat_of B' B (@lm_id F U FF VU).


  Lemma lm_comp_id_id :
    lm_comp (@lm_id F U FF VU) (@lm_id F U FF VU) == @lm_id F U FF VU.
  Proof.
    intros u. reflexivity.
  Qed.

  Lemma lm_id_comp_expand : ∀ f : LinearMap F U U,
    f == lm_comp (@lm_id F U FF VU) (lm_comp f (@lm_id F U FF VU)).
  Proof.
    intros f u. reflexivity.
  Qed.

  Theorem cob_inv : ∀ B B' : Basis n F U,
    mat_mul (cob B' B) (cob B B') == mat_id.
  Proof.
    intros B B'.
    transitivity (mat_of B' B'
        (lm_comp (@lm_id F U FF VU) (@lm_id F U FF VU))).
    - symmetry.
      exact (mat_of_comp B' B B' (@lm_id F U FF VU) (@lm_id F U FF VU)).
    - transitivity (mat_of B' B' (@lm_id F U FF VU)).
      + exact (mat_of_proper B' B'
                 (lm_comp (@lm_id F U FF VU) (@lm_id F U FF VU))
                 (@lm_id F U FF VU) lm_comp_id_id).
      + exact (mat_of_id B').
  Qed.

  Theorem mat_of_conj : ∀ (B B' : Basis n F U) (f : LinearMap F U U),
    mat_of B' B' f
      == mat_mul (cob B' B) (mat_mul (mat_of B B f) (cob B B')).
  Proof.
    intros B B' f.
    transitivity (mat_of B' B'
        (lm_comp (@lm_id F U FF VU) (lm_comp f (@lm_id F U FF VU)))).
    - exact (mat_of_proper B' B' f
               (lm_comp (@lm_id F U FF VU) (lm_comp f (@lm_id F U FF VU)))
               (lm_id_comp_expand f)).
    - transitivity (mat_mul (cob B' B)
          (mat_of B' B (lm_comp f (@lm_id F U FF VU)))).
      + exact (mat_of_comp B' B B' (@lm_id F U FF VU)
                 (lm_comp f (@lm_id F U FF VU))).
      + exact (mat_mul_proper n n n
                 (cob B' B) (cob B' B) (reflexivity (cob B' B))
                 (mat_of B' B (lm_comp f (@lm_id F U FF VU)))
                 (mat_mul (mat_of B B f) (cob B B'))
                 (mat_of_comp B' B B f (@lm_id F U FF VU))).
  Qed.

End ChangeOfBasis.


Section TraceBasisIndep.
  Context {F U : Type} {FF : Field F} {VU : VectorSpace F U}.
  Context {n : nat}.

  Theorem tr_basis_indep : ∀ (B B' : Basis n F U)
      (f : LinearMap F U U),
    trM (mat_of B' B' f) == trM (mat_of B B f).
  Proof.
    intros B B' f.
    transitivity (trM (mat_mul (cob B' B)
        (mat_mul (mat_of B B f) (cob B B')))).
    - exact (trM_proper n
               (mat_of B' B' f)
               (mat_mul (cob B' B) (mat_mul (mat_of B B f) (cob B B')))
               (mat_of_conj B B' f)).
    - transitivity (trM (mat_mul
          (mat_mul (mat_of B B f) (cob B B')) (cob B' B))).
      + exact (tr_mul_comm n (cob B' B)
                 (mat_mul (mat_of B B f) (cob B B'))).
      + transitivity (trM (mat_mul (mat_of B B f)
            (mat_mul (cob B B') (cob B' B)))).
        × exact (trM_proper n
                   (mat_mul (mat_mul (mat_of B B f) (cob B B')) (cob B' B))
                   (mat_mul (mat_of B B f) (mat_mul (cob B B') (cob B' B)))
                   (mat_mul_assoc n n n n (mat_of B B f)
                      (cob B B') (cob B' B))).
        × transitivity (trM (mat_mul (mat_of B B f) (@mat_id F FF n))).
          -- exact (trM_proper n
                      (mat_mul (mat_of B B f) (mat_mul (cob B B') (cob B' B)))
                      (mat_mul (mat_of B B f) (@mat_id F FF n))
                      (mat_mul_proper n n n
                         (mat_of B B f) (mat_of B B f)
                         (reflexivity (mat_of B B f))
                         (mat_mul (cob B B') (cob B' B)) (@mat_id F FF n)
                         (cob_inv B' B))).
          -- exact (trM_proper n
                      (mat_mul (mat_of B B f) (@mat_id F FF n))
                      (mat_of B B f)
                      (mat_mul_1r n n (mat_of B B f))).
  Qed.

  Definition tr (B : Basis n F U) (f : LinearMap F U U) : F :=
    trM (mat_of B B f).

  Theorem tr_indep : ∀ (B B' : Basis n F U) (f : LinearMap F U U),
    tr B' f == tr B f.
  Proof.
    intros B B' f.
    exact (tr_basis_indep B B' f).
  Qed.

End TraceBasisIndep.

Example tr_check (F : Type) (FF : Field F) (n : nat) : F :=
  tr (FinVec_Basis (F := F) n) lm_id.
