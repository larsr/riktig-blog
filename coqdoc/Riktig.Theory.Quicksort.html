Lemma perm_transpose {A : Type} (a b c : list A) (x y : A) :
  Permutation (a ++ x :: b ++ y :: c) (a ++ y :: b ++ x :: c).
Proof.
  apply Permutation_app_head.
  calc (x :: b ++ y :: c)
    == (x :: y :: b ++ c)
    == (y :: x :: b ++ c)
    == (y :: b ++ x :: c).
  - apply perm_skip, Permutation_sym, Permutation_middle.
  - apply perm_swap.
  - apply perm_skip, Permutation_middle.
Qed.

Section QuicksortCorrect.

  Variable V : Type.
  Variable leb : V → V → bool.

  Variable mem : Type.
  Variable read : mem → nat → V.
  Variable swap : mem → nat → nat → mem.

  Hypothesis read_swap_l : ∀ m i j, read (swap m i j) i = read m j.
  Hypothesis read_swap_r : ∀ m i j, read (swap m i j) j = read m i.
  Hypothesis read_swap_other : ∀ m i j k,
      k ≠ i → k ≠ j → read (swap m i j) k = read m k.

  Hypothesis leb_total : ∀ x y, leb x y = false → leb y x = true.
  Hypothesis leb_trans : ∀ x y z,
      leb x y = true → leb y z = true → leb x z = true.

  Local Notation part := (Quicksort.part V leb mem read swap).
  Local Notation qsort := (Quicksort.qsort V leb mem read swap).
  Local Notation leP := (fun x y : V ⇒ leb x y = true).

  Definition slice (m : mem) (lo n : nat) : list V := map (read m) (seq lo n).


  Lemma slice_0 m lo : slice m lo 0 = [].
  Proof. reflexivity. Qed.

  Lemma slice_1 m lo : slice m lo 1 = [read m lo].
  Proof. reflexivity. Qed.

  Lemma slice_cons m lo n : slice m lo (S n) = read m lo :: slice m (S lo) n.
  Proof. reflexivity. Qed.

  Lemma slice_split m lo a b :
    slice m lo (a + b) = slice m lo a ++ slice m (lo + a) b.
  Proof. unfold slice. now rewrite seq_app, map_app. Qed.

  Lemma slice_ext m1 m2 lo n :
    (∀ k, lo ≤ k < lo + n → read m1 k = read m2 k) →
    slice m1 lo n = slice m2 lo n.
  Proof.
    intros H. apply map_ext_in. intros k Hk.
    apply H. now apply in_seq in Hk.
  Qed.

  Lemma slice_in x m lo n :
    In x (slice m lo n) → ∃ k, lo ≤ k < lo + n ∧ x = read m k.
  Proof.
    unfold slice. intros H. apply in_map_iff in H as (k & E & I).
    apply in_seq in I. ∃ k. split; [lia | now symmetry].
  Qed.

  Lemma slice_split_pivot m lo n p :
    lo ≤ p < lo + n →
    slice m lo n = slice m lo (p - lo) ++ read m p :: slice m (S p) (lo + n - S p).
  Proof.
    intros H. replace n with (p - lo + S (lo + n - S p)) at 1 by lia.
    rewrite slice_split. replace (lo + (p - lo)) with p by lia.
    now rewrite slice_cons.
  Qed.

  Lemma slice_split3 m lo n i j :
    lo ≤ i → i < j → j < lo + n →
    slice m lo n = slice m lo (i - lo)
      ++ read m i :: slice m (S i) (j - S i)
      ++ read m j :: slice m (S j) (lo + n - S j).
  Proof.
    intros. rewrite (slice_split_pivot m lo n i) by lia.
    rewrite (slice_split_pivot m (S i) (lo + n - S i) j) by lia.
    now replace (S i + (lo + n - S i) - S j) with (lo + n - S j) by lia.
  Qed.

  Lemma slice_cons_perm m1 m2 lo n :
    read m1 lo = read m2 lo →
    Permutation (slice m1 (S lo) n) (slice m2 (S lo) n) →
    Permutation (slice m1 lo (S n)) (slice m2 lo (S n)).
  Proof. intros E P. rewrite !slice_cons, E. now apply perm_skip. Qed.


  Lemma slice_swap_comm m i j lo n :
    slice (swap m i j) lo n = slice (swap m j i) lo n.
  Proof.
    apply slice_ext. intros k _.
    destruct (Nat.eq_dec k i) as [->|]; [now rewrite read_swap_l, read_swap_r|].
    destruct (Nat.eq_dec k j) as [->|]; [now rewrite read_swap_r, read_swap_l|].
    rewrite !read_swap_other by assumption. reflexivity.
  Qed.

  Lemma swap_perm_lt m i j lo n :
    lo ≤ i → i < j → j < lo + n →
    Permutation (slice m lo n) (slice (swap m i j) lo n).
  Proof.
    intros Hi Hij Hj.
    rewrite (slice_split3 m lo n i j) by lia.
    rewrite (slice_split3 (swap m i j) lo n i j) by lia.
    rewrite read_swap_l, read_swap_r.
    rewrite (slice_ext (swap m i j) m lo (i - lo))
      by (intros; apply read_swap_other; lia).
    rewrite (slice_ext (swap m i j) m (S i) (j - S i))
      by (intros; apply read_swap_other; lia).
    rewrite (slice_ext (swap m i j) m (S j) (lo + n - S j))
      by (intros; apply read_swap_other; lia).
    apply perm_transpose.
  Qed.

  Lemma swap_perm m i j lo n :
    lo ≤ i < lo + n → lo ≤ j < lo + n →
    Permutation (slice m lo n) (slice (swap m i j) lo n).
  Proof.
    intros Hi Hj. destruct (lt_eq_lt_dec i j) as [[L| ->]|L].
    - apply swap_perm_lt; lia.
    - replace (slice (swap m j j) lo n) with (slice m lo n); [reflexivity|].
      apply slice_ext. intros k _.
      destruct (Nat.eq_dec k j) as [->|];
        [now rewrite read_swap_l | now rewrite read_swap_other].
    - rewrite slice_swap_comm. apply swap_perm_lt; lia.
  Qed.


  Lemma sorted_glue (p : V) : ∀ l1 l2,
    Sorted leP l1 → Sorted leP l2 →
    (∀ x, In x l1 → leP x p) →
    (∀ y, In y l2 → leP p y) →
    Sorted leP (l1 ++ p :: l2).
  Proof.
    induction l1 as [|a l1 IH]; intros l2 S1 S2 B1 B2; simpl.
    - constructor; [exact S2|]. destruct l2; constructor. apply B2, in_eq.
    - apply Sorted_inv in S1 as [S1 H1].
      constructor.
      + apply IH; auto. intros x Hx. apply B1. now right.
      + destruct l1; simpl; constructor.
        × apply B1, in_eq.
        × now inversion H1.
  Qed.


  Definition scan_inv (m : mem) (lo store i hi : nat) : Prop :=
    lo ≤ store ∧ store ≤ i ∧ i ≤ hi ∧
    (∀ k, lo ≤ k < store → leb (read m k) (read m hi) = true) ∧
    (∀ k, store ≤ k < i → leb (read m k) (read m hi) = false).

  Definition part_out (m : mem) (lo store hi : nat) (m' : mem) (p : nat) : Prop :=
    store ≤ p ≤ hi ∧
    (∀ k, k < store ∨ hi < k → read m' k = read m k) ∧
    Permutation (slice m store (S hi - store)) (slice m' store (S hi - store)) ∧
    (∀ k, lo ≤ k < p → leb (read m' k) (read m' p) = true) ∧
    (∀ k, p < k ≤ hi → leb (read m' p) (read m' k) = true).

  Lemma scan_inv_step_le m lo store i hi :
    i < hi → leb (read m i) (read m hi) = true →
    scan_inv m lo store i hi →
    scan_inv (swap m i store) lo (S store) (S i) hi.
  Proof.
    intros L E (H1 & H2 & H3 & H4 & H5).
    repeat split; try lia; intros k Hk;
      rewrite (read_swap_other _ _ _ hi) by lia.
    - destruct (Nat.eq_dec k store) as [->|Ne].
      + now rewrite read_swap_r.
      + rewrite read_swap_other by lia. apply H4; lia.
    - destruct (Nat.eq_dec k i) as [->|Ne].
      + rewrite read_swap_l. apply H5; lia.
      + rewrite read_swap_other by lia. apply H5; lia.
  Qed.

  Lemma scan_inv_step_gt m lo store i hi :
    i < hi → leb (read m i) (read m hi) ≠ true →
    scan_inv m lo store i hi →
    scan_inv m lo store (S i) hi.
  Proof.
    intros L E (H1 & H2 & H3 & H4 & H5).
    repeat split; try lia; auto.
    intros k Hk. destruct (Nat.eq_dec k i) as [->|Ne].
    - now apply Bool.not_true_is_false.
    - apply H5; lia.
  Qed.

  Lemma part_out_swap m lo store i hi m' p :
    lo ≤ store → store ≤ i → i < hi →
    part_out (swap m i store) lo (S store) hi m' p →
    part_out m lo store hi m' p.
  Proof.
    intros Hls Hsi Hih ((Hp1 & Hp2) & Hfr & Hpm & Hc & Hd).
    repeat split; try lia; auto.
    - intros k Hk. rewrite Hfr by lia. apply read_swap_other; lia.
    - replace (S hi - store) with (S (S hi - S store)) by lia.
      apply perm_trans with (slice (swap m i store) store (S (S hi - S store))).
      + replace (S (S hi - S store)) with (S hi - store) by lia.
        apply swap_perm; lia.
      + apply slice_cons_perm; [symmetry; apply Hfr; lia | exact Hpm].
  Qed.

  Lemma part_out_base m lo store hi :
    scan_inv m lo store hi hi →
    part_out m lo store hi (swap m store hi) store.
  Proof.
    intros (H1 & H2 & _ & H4 & H5).
    repeat split; try lia.
    - intros k Hk. apply read_swap_other; lia.
    - apply swap_perm; lia.
    - intros k Hk. rewrite read_swap_l.
      rewrite read_swap_other by lia. apply H4; lia.
    - intros k Hk. rewrite read_swap_l.
      destruct (Nat.eq_dec k hi) as [->|Ne].
      + rewrite read_swap_r. apply leb_total, H5. lia.
      + rewrite read_swap_other by lia. apply leb_total, H5. lia.
  Qed.

  Lemma part_post hi : ∀ i m store lo m' p,
    scan_inv m lo store i hi →
    part m store i hi = (m', p) →
    part_out m lo store hi m' p.
  Proof.
    apply (measure_ind (fun i ⇒ hi - i)
             (fun i ⇒ ∀ m store lo m' p,
                  scan_inv m lo store i hi →
                  part m store i hi = (m', p) →
                  part_out m lo store hi m' p)).
    intros i IH m store lo m' p Inv Ep.
    rewrite part_step in Ep.
    destruct (dec (i < hi)) as [L|L].
    - destruct (dec (leb (read m i) (read m hi) = true)) as [E|E].
      + apply part_out_swap with (i := i);
          try (destruct Inv as (?&?&?&?&?); lia).
        apply (IH (S i)); [lia | now apply scan_inv_step_le | exact Ep].
      + apply (IH (S i)); [lia | now apply scan_inv_step_gt | exact Ep].
    - assert (i = hi) by (destruct Inv as (?&?&?&?&?); lia). subst i.
      injection Ep as <- <-. now apply part_out_base.
  Qed.

  Lemma part_correct m lo hi m' p :
    lo ≤ hi → part m lo lo hi = (m', p) → part_out m lo lo hi m' p.
  Proof.
    intros H Ep. eapply part_post; [|exact Ep].
    repeat split; try lia; intros k Hk; lia.
  Qed.


  Lemma qsort_small m lo hi : S hi - lo < 2 → qsort m lo hi = m.
  Proof.
    intros L. rewrite qsort_step.
    destruct (dec (S hi - lo < 2)); [reflexivity | lia].
  Qed.

  Lemma qsort_step_part m lo hi m' piv :
    ¬ S hi - lo < 2 →
    part m lo lo hi = (m', piv) →
    lo ≤ piv ≤ hi →
    qsort m lo hi = qsort (qsort m' lo (piv - 1)) (S piv) hi.
  Proof.
    intros H2 Ep Hp. rewrite qsort_step.
    destruct (dec (S hi - lo < 2)); [lia|].
    cbv zeta. rewrite Ep.
    destruct (dec (S (piv - 1) - lo < S hi - lo)); [|lia].
    destruct (dec (S hi - S piv < S hi - lo)); [|lia].
    reflexivity.
  Qed.


  Theorem qsort_frame : ∀ lo hi m k,
    k < lo ∨ hi < k → read (qsort m lo hi) k = read m k.
  Proof.
    apply (measure_ind2 (fun lo hi ⇒ S hi - lo)
             (fun lo hi ⇒ ∀ m k,
                  k < lo ∨ hi < k → read (qsort m lo hi) k = read m k)).
    intros lo hi IH m k Hk.
    destruct (dec (S hi - lo < 2)) as [L|L]; [now rewrite qsort_small|].
    destruct (part m lo lo hi) as [m' piv] eqn:Ep.
    destruct (part_correct m lo hi m' piv ltac:(lia) Ep) as ((Hp1 & Hp2) & Hfr & _).
    rewrite (qsort_step_part m lo hi m' piv L Ep ltac:(lia)).
    rewrite (IH (S piv) hi ltac:(lia) _ k ltac:(lia)).
    rewrite (IH lo (piv - 1) ltac:(lia) _ k ltac:(lia)).
    apply Hfr; lia.
  Qed.

  Lemma qsort_slice_out m lo hi lo' n' :
    lo' + n' ≤ lo ∨ hi < lo' →
    slice (qsort m lo hi) lo' n' = slice m lo' n'.
  Proof.
    intros H. apply slice_ext. intros k Hk. apply qsort_frame; lia.
  Qed.

  Lemma qsort_left_pivot m' lo piv :
    lo ≤ piv → read (qsort m' lo (piv - 1)) piv = read m' piv.
  Proof.
    intros Hle. destruct (dec (piv = lo)) as [->|Hneq].
    - rewrite qsort_small; [reflexivity | lia].
    - apply qsort_frame; lia.
  Qed.

  Theorem qsort_perm : ∀ lo hi m,
    Permutation (slice m lo (S hi - lo)) (slice (qsort m lo hi) lo (S hi - lo)).
  Proof.
    apply (measure_ind2 (fun lo hi ⇒ S hi - lo)
             (fun lo hi ⇒ ∀ m,
                  Permutation (slice m lo (S hi - lo)) (slice (qsort m lo hi) lo (S hi - lo)))).
    intros lo hi IH m.
    destruct (dec (S hi - lo < 2)) as [L|L]; [now rewrite qsort_small|].
    destruct (part m lo lo hi) as [m' piv] eqn:Ep.
    destruct (part_correct m lo hi m' piv ltac:(lia) Ep) as ((Hp1 & Hp2) & Hfr & Hpm & _).
    rewrite (qsort_step_part m lo hi m' piv L Ep ltac:(lia)).
    apply perm_trans with (slice m' lo (S hi - lo)).
    { exact Hpm. }
    rewrite (slice_split_pivot m' lo (S hi - lo) piv) by lia.
    rewrite (slice_split_pivot
               (qsort (qsort m' lo (piv - 1)) (S piv) hi)
               lo (S hi - lo) piv) by lia.
    rewrite (qsort_slice_out (qsort m' lo (piv - 1)) (S piv) hi lo (piv - lo)) by lia.
    rewrite (qsort_frame (S piv) hi) by lia.
    rewrite (qsort_left_pivot m' lo piv Hp1).
    replace (lo + (S hi - lo) - S piv) with (hi - piv) by lia.
    apply Permutation_app.
    -
      destruct (dec (piv = lo)) as [->|Hneq].
      + replace (lo - lo) with 0 by lia. rewrite !slice_0. constructor.
      + replace (piv - lo) with (S (piv - 1) - lo) by lia. apply IH; lia.
    - apply perm_skip.
      rewrite <- (qsort_slice_out m' lo (piv - 1) (S piv) (hi - piv)) by lia.
      replace (hi - piv) with (S hi - S piv) by lia.
      apply IH; lia.
  Qed.

  Theorem qsort_sorted : ∀ lo hi m,
    Sorted leP (slice (qsort m lo hi) lo (S hi - lo)).
  Proof.
    apply (measure_ind2 (fun lo hi ⇒ S hi - lo)
             (fun lo hi ⇒ ∀ m, Sorted leP (slice (qsort m lo hi) lo (S hi - lo)))).
    intros lo hi IH m.
    destruct (dec (S hi - lo < 2)) as [L|L].
    { rewrite qsort_small by lia.
      replace (S hi - lo) with (if dec (hi < lo) then 0 else 1) by (destruct (dec (hi < lo)); lia).
      destruct (dec (hi < lo)); [rewrite slice_0 | rewrite slice_1];
        repeat constructor. }
    destruct (part m lo lo hi) as [m' piv] eqn:Ep.
    destruct (part_correct m lo hi m' piv ltac:(lia) Ep) as ((Hp1 & Hp2) & Hfr & Hpm & Hc & Hd).
    rewrite (qsort_step_part m lo hi m' piv L Ep ltac:(lia)).
    rewrite (slice_split_pivot
               (qsort (qsort m' lo (piv - 1)) (S piv) hi)
               lo (S hi - lo) piv) by lia.
    rewrite (qsort_slice_out (qsort m' lo (piv - 1)) (S piv) hi lo (piv - lo)) by lia.
    rewrite (qsort_frame (S piv) hi) by lia.
    rewrite (qsort_left_pivot m' lo piv Hp1).
    replace (lo + (S hi - lo) - S piv) with (hi - piv) by lia.
    apply sorted_glue.
    -
      destruct (dec (piv = lo)) as [->|Hneq].
      + replace (lo - lo) with 0 by lia. rewrite slice_0. constructor.
      + replace (piv - lo) with (S (piv - 1) - lo) by lia. apply IH; lia.
    - apply IH; lia.
    -
      intros x Hx.
      destruct (dec (piv = lo)) as [->|Hneq].
      + replace (lo - lo) with 0 in Hx by lia.
        rewrite slice_0 in Hx. destruct Hx.
      + apply Permutation_in with (l' := slice m' lo (piv - lo)) in Hx;
          [|replace (piv - lo) with (S (piv - 1) - lo) by lia;
            apply Permutation_sym, qsort_perm].
        destruct (slice_in _ _ _ _ Hx) as (k & Hk & ->).
        apply Hc; lia.
    -
      intros y Hy.
      apply Permutation_in with
        (l' := slice (qsort m' lo (piv - 1)) (S piv) (hi - piv)) in Hy;
        [|replace (hi - piv) with (S hi - S piv) by lia;
          apply Permutation_sym, qsort_perm].
      rewrite (qsort_slice_out m' lo (piv - 1) (S piv) (hi - piv)) in Hy by lia.
      destruct (slice_in _ _ _ _ Hy) as (k & Hk & ->).
      apply Hd; lia.
  Qed.

  Corollary qsort_strongly_sorted : ∀ lo hi m,
    StronglySorted leP (slice (qsort m lo hi) lo (S hi - lo)).
  Proof.
    intros. apply Sorted_StronglySorted;
      [exact leb_trans | apply qsort_sorted].
  Qed.

  Corollary qsort_correct : ∀ m lo hi,
    Permutation (slice m lo (S hi - lo)) (slice (qsort m lo hi) lo (S hi - lo))
    ∧ Sorted leP (slice (qsort m lo hi) lo (S hi - lo)).
  Proof. intros. split; [apply qsort_perm | apply qsort_sorted]. Qed.

End QuicksortCorrect.


Section FunMem.

  Definition fread (m : nat → nat) (i : nat) : nat := m i.

  Definition fswap (m : nat → nat) (i j : nat) : nat → nat :=
    fun k ⇒ if dec (k = i) then m j else if dec (k = j) then m i else m k.

  Lemma fswap_l m i j : fread (fswap m i j) i = fread m j.
  Proof. unfold fread, fswap. now destruct (dec (i = i)). Qed.

  Lemma fswap_r m i j : fread (fswap m i j) j = fread m i.
  Proof.
    unfold fread, fswap.
    destruct (dec (j = i)) as [->|]; [reflexivity|].
    now destruct (dec (j = j)).
  Qed.

  Lemma fswap_other m i j k : k ≠ i → k ≠ j → fread (fswap m i j) k = fread m k.
  Proof.
    intros. unfold fread, fswap.
    destruct (dec (k = i)); [contradiction|].
    destruct (dec (k = j)); [contradiction | reflexivity].
  Qed.

  Lemma nat_leb_total x y : Nat.leb x y = false → Nat.leb y x = true.
  Proof. intros H. apply Nat.leb_gt in H. apply Nat.leb_le. lia. Qed.

  Example fun_qsort_correct (m : nat → nat) (lo hi : nat) :
    let m' := qsort nat Nat.leb (nat → nat) fread fswap m lo hi in
    Permutation (slice nat (nat → nat) fread m lo (S hi - lo))
                (slice nat (nat → nat) fread m' lo (S hi - lo))
    ∧ Sorted (fun x y ⇒ Nat.leb x y = true)
              (slice nat (nat → nat) fread m' lo (S hi - lo)).
  Proof.
    apply qsort_correct;
      auto using fswap_l, fswap_r, fswap_other, nat_leb_total.
  Qed.

End FunMem.
