  Context (addr_add addr_sub : addr → addr → addr).

  Local Notation interp :=
    (interp addr_eqb addr_add addr_sub addr_leb addr_geb
            addr_of_nat addr1 to_addr val_default).


  Lemma interp_mono :
    ∀ (fuel : nat) (a : Type) (p : Mem addr val a)
           (m : Machine addr val) (r : a) (m' : Machine addr val)
           (fuel' : nat),
      fuel ≤ fuel' →
      interp fuel p m = (Some r, m') →
      interp fuel' p m = (Some r, m').
  Proof.
    induction fuel as [|f IH]; intros a p m r m' fuel' Hle Hrun.
    - cbn [interp] in Hrun. discriminate.
    - destruct fuel' as [|f']; [ inversion Hle |].
      apply le_S_n in Hle.
      destruct p; cbn [interp] in Hrun |- ×.
      + exact Hrun.
      +
        destruct (store_lookup addr_eqb _ (mach_store m)) as [[v|q|i]|];
          try discriminate; exact (IH _ _ _ _ _ _ Hle Hrun).
      +
        destruct (store_member addr_eqb _ (mach_store m));
          try discriminate; exact (IH _ _ _ _ _ _ Hle Hrun).
      +
        destruct (store_lookup addr_eqb _ (mach_store m)) as [[v|q|i]|];
          try discriminate; exact (IH _ _ _ _ _ _ Hle Hrun).
      +
        destruct (store_member addr_eqb _ (mach_store m));
          try discriminate; exact (IH _ _ _ _ _ _ Hle Hrun).
      +
        destruct (store_lookup addr_eqb _ (mach_store m)) as [[v|q|i]|];
          try discriminate; exact (IH _ _ _ _ _ _ Hle Hrun).
      +
        destruct (store_member addr_eqb _ (mach_store m));
          try discriminate; exact (IH _ _ _ _ _ _ Hle Hrun).
      +
        destruct (addr_leb (mach_sp m) (mach_hp m));
          try discriminate; exact (IH _ _ _ _ _ _ Hle Hrun).
      +
        destruct (addr_geb _ (mach_sp m));
          try discriminate; exact (IH _ _ _ _ _ _ Hle Hrun).
      + exact (IH _ _ _ _ _ _ Hle Hrun).
      + exact (IH _ _ _ _ _ _ Hle Hrun).
      + exact (IH _ _ _ _ _ _ Hle Hrun).
      +
        destruct (interp f _ m) as [rc mach1] eqn:Ec.
        destruct rc as [[b|]|]; try discriminate.
        destruct b.
        × destruct (interp f _ mach1) as [rb mach2] eqn:Eb.
          destruct rb as [[u|]|]; try discriminate.
          rewrite (IH _ _ _ _ _ _ Hle Ec).
          rewrite (IH _ _ _ _ _ _ Hle Eb).
          exact (IH _ _ _ _ _ _ Hle Hrun).
        × rewrite (IH _ _ _ _ _ _ Hle Ec).
          exact (IH _ _ _ _ _ _ Hle Hrun).
  Qed.

  Lemma interp_deterministic :
    ∀ (a : Type) (f1 f2 : nat) (p : Mem addr val a)
           (m : Machine addr val) (r1 r2 : a) (m1 m2 : Machine addr val),
      interp f1 p m = (Some r1, m1) →
      interp f2 p m = (Some r2, m2) →
      r1 = r2 ∧ m1 = m2.
  Proof.
    intros a f1 f2 p m r1 r2 m1 m2 H1 H2.
    pose proof (interp_mono f1 _ _ _ _ _ (Nat.max f1 f2) (Nat.le_max_l _ _) H1) as H1'.
    pose proof (interp_mono f2 _ _ _ _ _ (Nat.max f1 f2) (Nat.le_max_r _ _) H2) as H2'.
    rewrite H1' in H2'. inversion H2'. auto.
  Qed.


  Inductive Eval : ∀ (a : Type),
      Mem addr val a → Machine addr val → a → Machine addr val → Prop :=
  | EvRet : ∀ (a : Type) (x : a) (mach : Machine addr val),
      Eval a (Ret x) mach x mach
  | EvReadVal : ∀ (a : Type) (ad : addr) (k : val → Mem addr val a)
                       (v : val) (mach : Machine addr val) r m',
      store_lookup addr_eqb ad (mach_store mach) = Some (CellVal v) →
      Eval a (k v) mach r m' →
      Eval a (ReadVal ad k) mach r m'
  | EvWriteVal : ∀ (a : Type) (ad : addr) (v : val) (p : Mem addr val a)
                        (mach : Machine addr val) r m',
      store_member addr_eqb ad (mach_store mach) = true →
      Eval a p (MkMachine (store_insert ad (CellVal v) (mach_store mach))
                          (mach_sp mach) (mach_hp mach)) r m' →
      Eval a (WriteVal ad v p) mach r m'
  | EvReadAddr : ∀ (a : Type) (ad : addr) (k : addr → Mem addr val a)
                        (q : addr) (mach : Machine addr val) r m',
      store_lookup addr_eqb ad (mach_store mach) = Some (CellAddr q) →
      Eval a (k q) mach r m' →
      Eval a (ReadAddr ad k) mach r m'
  | EvWriteAddr : ∀ (a : Type) (ad q : addr) (p : Mem addr val a)
                         (mach : Machine addr val) r m',
      store_member addr_eqb ad (mach_store mach) = true →
      Eval a p (MkMachine (store_insert ad (CellAddr q) (mach_store mach))
                          (mach_sp mach) (mach_hp mach)) r m' →
      Eval a (WriteAddr ad q p) mach r m'
  | EvReadInstr : ∀ (a : Type) (ad : addr) (k : Instr val → Mem addr val a)
                         (i : Instr val) (mach : Machine addr val) r m',
      store_lookup addr_eqb ad (mach_store mach) = Some (CellInstr i) →
      Eval a (k i) mach r m' →
      Eval a (ReadInstr ad k) mach r m'
  | EvWriteInstr : ∀ (a : Type) (ad : addr) (i : Instr val)
                          (p : Mem addr val a) (mach : Machine addr val) r m',
      store_member addr_eqb ad (mach_store mach) = true →
      Eval a p (MkMachine (store_insert ad (CellInstr i) (mach_store mach))
                          (mach_sp mach) (mach_hp mach)) r m' →
      Eval a (WriteInstr ad i p) mach r m'
  | EvStackAlloc : ∀ (a : Type) (k : addr → Mem addr val a)
                          (mach : Machine addr val) r m',
      addr_leb (mach_sp mach) (mach_hp mach) = true →
      Eval a (k (mach_sp mach))
           (MkMachine (store_insert (mach_sp mach) (CellVal val_default)
                                    (mach_store mach))
                      (addr_add (mach_sp mach) addr1) (mach_hp mach)) r m' →
      Eval a (StackAlloc k) mach r m'
  | EvAlloc : ∀ (a : Type) (n : nat) (k : addr → Mem addr val a)
                     (mach : Machine addr val) (base : addr)
                     (st : AStore addr val) r m',
      base = addr_sub (mach_hp mach) (addr_of_nat (n - 1)) →
      st = fold_left
             (fun s i ⇒ store_insert (addr_add base (addr_of_nat i))
                                      (CellVal val_default) s)
             (seq 0 n) (mach_store mach) →
      addr_geb base (mach_sp mach) = true →
      Eval a (k base) (MkMachine st (mach_sp mach) (addr_sub base addr1)) r m' →
      Eval a (Alloc n k) mach r m'
  | EvOffset : ∀ (a : Type) (base : addr) (off : val)
                      (k : addr → Mem addr val a) (mach : Machine addr val) r m',
      Eval a (k (addr_add base (to_addr off))) mach r m' →
      Eval a (Offset base off k) mach r m'
  | EvGetSp : ∀ (a : Type) (k : addr → Mem addr val a)
                     (mach : Machine addr val) r m',
      Eval a (k (mach_sp mach)) mach r m' →
      Eval a (GetSp k) mach r m'
  | EvSetSp : ∀ (a : Type) (sp : addr) (p : Mem addr val a)
                     (mach : Machine addr val) r m',
      Eval a p (MkMachine (mach_store mach) sp (mach_hp mach)) r m' →
      Eval a (SetSp sp p) mach r m'
  | EvWhileTrue : ∀ (a : Type) (c : Mem addr val (option bool))
                         (body : Mem addr val (option unit))
                         (k : Mem addr val a)
                         (mach mach1 mach2 : Machine addr val) r m',
      Eval (option bool) c mach (Some true) mach1 →
      Eval (option unit) body mach1 (Some tt) mach2 →
      Eval a (While c body k) mach2 r m' →
      Eval a (While c body k) mach r m'
  | EvWhileFalse : ∀ (a : Type) (c : Mem addr val (option bool))
                          (body : Mem addr val (option unit))
                          (k : Mem addr val a)
                          (mach mach1 : Machine addr val) r m',
      Eval (option bool) c mach (Some false) mach1 →
      Eval a k mach1 r m' →
      Eval a (While c body k) mach r m'.

  Arguments Eval {a}.


  Lemma interp_sound :
    ∀ (fuel : nat) (a : Type) (p : Mem addr val a)
           (m : Machine addr val) (r : a) (m' : Machine addr val),
      interp fuel p m = (Some r, m') →
      Eval p m r m'.
  Proof.
    induction fuel as [|f IH]; intros a p.
    - intros m r m' Hrun. cbn [interp] in Hrun. discriminate.
    -
      refine
        (match p as p0 in Mem _ _ a0
               return ∀ (m : Machine addr val) (r : a0) (m' : Machine addr val),
                        interp (S f) p0 m = (Some r, m') → Eval p0 m r m'
         with
         | Ret x ⇒ fun m r m' Hrun ⇒ _
         | ReadVal ad k ⇒ fun m r m' Hrun ⇒ _
         | WriteVal ad v p' ⇒ fun m r m' Hrun ⇒ _
         | ReadAddr ad k ⇒ fun m r m' Hrun ⇒ _
         | WriteAddr ad q p' ⇒ fun m r m' Hrun ⇒ _
         | ReadInstr ad k ⇒ fun m r m' Hrun ⇒ _
         | WriteInstr ad i p' ⇒ fun m r m' Hrun ⇒ _
         | StackAlloc k ⇒ fun m r m' Hrun ⇒ _
         | Alloc n k ⇒ fun m r m' Hrun ⇒ _
         | Offset base off k ⇒ fun m r m' Hrun ⇒ _
         | GetSp k ⇒ fun m r m' Hrun ⇒ _
         | SetSp sp p' ⇒ fun m r m' Hrun ⇒ _
         | While c body p' ⇒ fun m r m' Hrun ⇒ _
         end); cbn [interp] in Hrun.
      + inversion Hrun; subst. apply EvRet.
      +
        destruct (store_lookup addr_eqb ad (mach_store m)) as [[v|q|i]|] eqn:El;
          try discriminate.
        eapply EvReadVal; [exact El | apply (IH _ _ _ _ _ Hrun)].
      +
        destruct (store_member addr_eqb ad (mach_store m)) eqn:Em;
          try discriminate.
        eapply EvWriteVal; [exact Em | apply (IH _ _ _ _ _ Hrun)].
      +
        destruct (store_lookup addr_eqb ad (mach_store m)) as [[v|q|i]|] eqn:El;
          try discriminate.
        eapply EvReadAddr; [exact El | apply (IH _ _ _ _ _ Hrun)].
      +
        destruct (store_member addr_eqb ad (mach_store m)) eqn:Em;
          try discriminate.
        eapply EvWriteAddr; [exact Em | apply (IH _ _ _ _ _ Hrun)].
      +
        destruct (store_lookup addr_eqb ad (mach_store m)) as [[v|q|i]|] eqn:El;
          try discriminate.
        eapply EvReadInstr; [exact El | apply (IH _ _ _ _ _ Hrun)].
      +
        destruct (store_member addr_eqb ad (mach_store m)) eqn:Em;
          try discriminate.
        eapply EvWriteInstr; [exact Em | apply (IH _ _ _ _ _ Hrun)].
      +
        destruct (addr_leb (mach_sp m) (mach_hp m)) eqn:Es; try discriminate.
        eapply EvStackAlloc; [exact Es | apply (IH _ _ _ _ _ Hrun)].
      +
        destruct (addr_geb (addr_sub (mach_hp m) (addr_of_nat (n - 1)))
                           (mach_sp m)) eqn:Eg;
          try discriminate.
        eapply EvAlloc; [reflexivity | reflexivity | exact Eg
                        | apply (IH _ _ _ _ _ Hrun)].
      + apply EvOffset, (IH _ _ _ _ _ Hrun).
      + apply EvGetSp, (IH _ _ _ _ _ Hrun).
      + apply EvSetSp, (IH _ _ _ _ _ Hrun).
      +
        destruct (interp f c m) as [rc mach1] eqn:Ec.
        destruct rc as [[[|]|]|]; try discriminate.
        × destruct (interp f body mach1) as [rb mach2] eqn:Eb.
          destruct rb as [[u|]|]; try discriminate.
          destruct u.
          eapply EvWhileTrue;
            [apply (IH _ _ _ _ _ Ec) | apply (IH _ _ _ _ _ Eb)
            | apply (IH _ _ _ _ _ Hrun)].
        × eapply EvWhileFalse;
            [apply (IH _ _ _ _ _ Ec) | apply (IH _ _ _ _ _ Hrun)].
  Qed.

  Lemma interp_adequate :
    ∀ (a : Type) (p : Mem addr val a)
           (m : Machine addr val) (r : a) (m' : Machine addr val),
      Eval p m r m' →
      ∃ fuel, interp fuel p m = (Some r, m').
  Proof.
    intros a p m r m' Hev.
    induction Hev as
      [ a x mach
      | a ad k v mach r m' Hl _ (f & Hf)
      | a ad v p mach r m' Hm _ (f & Hf)
      | a ad k q mach r m' Hl _ (f & Hf)
      | a ad q p mach r m' Hm _ (f & Hf)
      | a ad k i mach r m' Hl _ (f & Hf)
      | a ad i p mach r m' Hm _ (f & Hf)
      | a k mach r m' Hs _ (f & Hf)
      | a n k mach base st r m' Hbase Hst Hg _ (f & Hf)
      | a base off k mach r m' _ (f & Hf)
      | a k mach r m' _ (f & Hf)
      | a sp p mach r m' _ (f & Hf)
      | a c body k mach mach1 mach2 r m' _ (f1 & H1) _ (f2 & H2) _ (f3 & H3)
      | a c body k mach mach1 r m' _ (f1 & H1) _ (f2 & H2) ].
    - ∃ 1. reflexivity.
    - ∃ (S f). cbn [interp]. rewrite Hl. exact Hf.
    - ∃ (S f). cbn [interp]. rewrite Hm. exact Hf.
    - ∃ (S f). cbn [interp]. rewrite Hl. exact Hf.
    - ∃ (S f). cbn [interp]. rewrite Hm. exact Hf.
    - ∃ (S f). cbn [interp]. rewrite Hl. exact Hf.
    - ∃ (S f). cbn [interp]. rewrite Hm. exact Hf.
    - ∃ (S f). cbn [interp]. rewrite Hs. exact Hf.
    - ∃ (S f). cbn [interp]. subst base st. rewrite Hg. exact Hf.
    - ∃ (S f). cbn [interp]. exact Hf.
    - ∃ (S f). cbn [interp]. exact Hf.
    - ∃ (S f). cbn [interp]. exact Hf.
    -
      ∃ (S (Nat.max f1 (Nat.max f2 f3))). cbn [interp].
      rewrite (interp_mono f1 _ _ _ _ _ _ (Nat.le_max_l _ _) H1). cbn.
      rewrite (interp_mono f2 _ _ _ _ _ _
                 (Nat.le_trans _ _ _ (Nat.le_max_l _ _) (Nat.le_max_r _ _)) H2).
      cbn.
      exact (interp_mono f3 _ _ _ _ _ _
               (Nat.le_trans _ _ _ (Nat.le_max_r _ _) (Nat.le_max_r _ _)) H3).
    -
      ∃ (S (Nat.max f1 f2)). cbn [interp].
      rewrite (interp_mono f1 _ _ _ _ _ _ (Nat.le_max_l _ _) H1). cbn.
      exact (interp_mono f2 _ _ _ _ _ _ (Nat.le_max_r _ _) H2).
  Qed.

  Theorem eval_interp :
    ∀ (a : Type) (p : Mem addr val a)
           (m : Machine addr val) (r : a) (m' : Machine addr val),
      Eval p m r m' ↔ ∃ fuel, interp fuel p m = (Some r, m').
  Proof.
    intros a p m r m'. split.
    - apply interp_adequate.
    - intros (fuel & Hrun). exact (interp_sound fuel _ _ _ _ _ Hrun).
  Qed.

  Theorem eval_deterministic :
    ∀ (a : Type) (p : Mem addr val a) (m : Machine addr val)
           (r1 r2 : a) (m1 m2 : Machine addr val),
      Eval p m r1 m1 → Eval p m r2 m2 → r1 = r2 ∧ m1 = m2.
  Proof.
    intros a p m r1 r2 m1 m2 H1 H2.
    apply interp_adequate in H1 as (f1 & H1).
    apply interp_adequate in H2 as (f2 & H2).
    exact (interp_deterministic _ f1 f2 _ _ _ _ _ _ H1 H2).
  Qed.

  Lemma eval_ret :
    ∀ (a : Type) (x : a) (m : Machine addr val)
           (r : a) (m' : Machine addr val),
      Eval (Ret x) m r m' ↔ r = x ∧ m' = m.
  Proof.
    intros a x m r m'. split.
    - intros Hev. apply interp_adequate in Hev as (fuel & Hrun).
      destruct fuel; cbn [interp] in Hrun; inversion Hrun; auto.
    - intros [-> ->]. apply EvRet.
  Qed.


  Lemma interp_bind :
    ∀ (f1 : nat) (a b : Type) (p : Mem addr val a)
           (k : a → Mem addr val b) (m : Machine addr val)
           (r1 : a) (m1 : Machine addr val) (f2 : nat)
           (r : b) (m2 : Machine addr val),
      interp f1 p m = (Some r1, m1) →
      interp f2 (k r1) m1 = (Some r, m2) →
      interp (f1 + f2) (bind_mem p k) m = (Some r, m2).
  Proof.
    induction f1 as [|f1 IH]; intros a b p.
    - intros k m r1 m1 f2 r m2 Hp Hk. cbn [interp] in Hp. discriminate.
    -
      refine
        (match p as p0 in Mem _ _ a0
               return ∀ (k : a0 → Mem addr val b) (m : Machine addr val)
                             (r1 : a0) (m1 : Machine addr val) (f2 : nat)
                             (r : b) (m2 : Machine addr val),
                        interp (S f1) p0 m = (Some r1, m1) →
                        interp f2 (k r1) m1 = (Some r, m2) →
                        interp (S f1 + f2) (bind_mem p0 k) m = (Some r, m2)
         with
         | Ret x ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         | ReadVal ad kk ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         | WriteVal ad v p' ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         | ReadAddr ad kk ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         | WriteAddr ad q p' ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         | ReadInstr ad kk ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         | WriteInstr ad i p' ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         | StackAlloc kk ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         | Alloc n kk ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         | Offset base off kk ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         | GetSp kk ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         | SetSp sp p' ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         | While c body p' ⇒ fun k m r1 m1 f2 r m2 Hp Hk ⇒ _
         end); cbn [interp] in Hp; cbn [bind_mem Nat.add]; cbn [interp].
      +
        inversion Hp; subst.
        exact (interp_mono f2 _ _ _ _ _ (S f1 + f2)
                 (Nat.le_add_l f2 (S f1)) Hk).
      +
        destruct (store_lookup addr_eqb ad (mach_store m)) as [[v|q|i]|];
          try discriminate.
        exact (IH _ _ _ _ _ _ _ _ _ _ Hp Hk).
      +
        destruct (store_member addr_eqb ad (mach_store m));
          try discriminate.
        exact (IH _ _ _ _ _ _ _ _ _ _ Hp Hk).
      +
        destruct (store_lookup addr_eqb ad (mach_store m)) as [[v|q|i]|];
          try discriminate.
        exact (IH _ _ _ _ _ _ _ _ _ _ Hp Hk).
      +
        destruct (store_member addr_eqb ad (mach_store m));
          try discriminate.
        exact (IH _ _ _ _ _ _ _ _ _ _ Hp Hk).
      +
        destruct (store_lookup addr_eqb ad (mach_store m)) as [[v|q|i]|];
          try discriminate.
        exact (IH _ _ _ _ _ _ _ _ _ _ Hp Hk).
      +
        destruct (store_member addr_eqb ad (mach_store m));
          try discriminate.
        exact (IH _ _ _ _ _ _ _ _ _ _ Hp Hk).
      +
        destruct (addr_leb (mach_sp m) (mach_hp m)); try discriminate.
        exact (IH _ _ _ _ _ _ _ _ _ _ Hp Hk).
      +
        destruct (addr_geb (addr_sub (mach_hp m) (addr_of_nat (n - 1)))
                           (mach_sp m));
          try discriminate.
        exact (IH _ _ _ _ _ _ _ _ _ _ Hp Hk).
      + exact (IH _ _ _ _ _ _ _ _ _ _ Hp Hk).
      + exact (IH _ _ _ _ _ _ _ _ _ _ Hp Hk).
      + exact (IH _ _ _ _ _ _ _ _ _ _ Hp Hk).
      +
        destruct (interp f1 c m) as [rc mach1'] eqn:Ec.
        destruct rc as [[[|]|]|]; try discriminate.
        × destruct (interp f1 body mach1') as [rb mach2'] eqn:Eb.
          destruct rb as [[u|]|]; try discriminate.
          rewrite (interp_mono f1 _ _ _ _ _ _ (Nat.le_add_r _ _) Ec). cbn.
          rewrite (interp_mono f1 _ _ _ _ _ _ (Nat.le_add_r _ _) Eb). cbn.
          exact (IH _ _ (While c body p') _ _ _ _ _ _ _ Hp Hk).
        × rewrite (interp_mono f1 _ _ _ _ _ _ (Nat.le_add_r _ _) Ec). cbn.
          exact (IH _ _ p' _ _ _ _ _ _ _ Hp Hk).
  Qed.

  Lemma interp_bind_inv :
    ∀ (fuel : nat) (a b : Type) (p : Mem addr val a)
           (k : a → Mem addr val b) (m : Machine addr val)
           (r : b) (m2 : Machine addr val),
      interp fuel (bind_mem p k) m = (Some r, m2) →
      ∃ (r1 : a) (m1 : Machine addr val) (f1 f2 : nat),
        interp f1 p m = (Some r1, m1) ∧
        interp f2 (k r1) m1 = (Some r, m2).
  Proof.

    - intros k m r m2 H. cbn [interp] in H. discriminate.

                        ∃ (r1 : a0) (m1 : Machine addr val) (f1 f2 : nat),

                          interp f2 (k r1) m1 = (Some r, m2)

         | ReadVal ad kk ⇒ fun k m r m2 H ⇒ _
         | WriteVal ad v p' ⇒ fun k m r m2 H ⇒ _
         | ReadAddr ad kk ⇒ fun k m r m2 H ⇒ _
         | WriteAddr ad q p' ⇒ fun k m r m2 H ⇒ _
         | ReadInstr ad kk ⇒ fun k m r m2 H ⇒ _
         | WriteInstr ad i p' ⇒ fun k m r m2 H ⇒ _
         | StackAlloc kk ⇒ fun k m r m2 H ⇒ _
         | Alloc n kk ⇒ fun k m r m2 H ⇒ _
         | Offset base off kk ⇒ fun k m r m2 H ⇒ _
         | GetSp kk ⇒ fun k m r m2 H ⇒ _
         | SetSp sp p' ⇒ fun k m r m2 H ⇒ _
         | While c body p' ⇒ fun k m r m2 H ⇒ _
         end); cbn [bind_mem] in H.
      +
        ∃ x, m, 1, (S f). split; [reflexivity | exact H].

        cbn [interp]. rewrite Em. exact H1.

  Qed.

  Theorem bind_eval :
    ∀ (a b : Type) (p : Mem addr val a) (k : a → Mem addr val b)
           (m : Machine addr val) (r : b) (m2 : Machine addr val),
      Eval (bind_mem p k) m r m2 ↔
      ∃ (r1 : a) (m1 : Machine addr val),
        Eval p m r1 m1 ∧ Eval (k r1) m1 r m2.
  Proof.
    intros a b p k m r m2. split.
    - intros Hev. apply interp_adequate in Hev as (fuel & Hrun).
      destruct (interp_bind_inv fuel _ _ p k _ _ _ Hrun)
        as (r1 & m1 & f1 & f2 & H1 & H2).
      ∃ r1, m1.
      split; [exact (interp_sound f1 _ _ _ _ _ H1)
             | exact (interp_sound f2 _ _ _ _ _ H2)].
    - intros (r1 & m1 & H1 & H2).
      apply interp_adequate in H1 as (f1 & H1).
      apply interp_adequate in H2 as (f2 & H2).
      exact (interp_sound (f1 + f2) _ _ _ _ _
               (interp_bind f1 _ _ p k _ _ _ f2 _ _ H1 H2)).
  Qed.

End Semantics.
