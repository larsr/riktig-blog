Notation "x <- p ;; q" := (bind p (fun x ⇒ q))
  (at level 61, p at next level, right associativity).


Fixpoint list_to_string (l : list ascii) : string :=
  match l with
  | [] ⇒ EmptyString
  | c :: rest ⇒ String c (list_to_string rest)
  end.


Definition is_alpha (c : ascii) : bool :=
  let n := Ascii.nat_of_ascii c in
  orb (andb (Nat.leb 65 n) (Nat.leb n 90))
      (andb (Nat.leb 97 n) (Nat.leb n 122)).

Definition is_digit (c : ascii) : bool :=
  let n := Ascii.nat_of_ascii c in
  andb (Nat.leb 48 n) (Nat.leb n 57).

Definition is_hexdig (c : ascii) : bool :=
  let n := Ascii.nat_of_ascii c in
  orb (is_digit c)
      (orb (andb (Nat.leb 65 n) (Nat.leb n 70))
           (andb (Nat.leb 97 n) (Nat.leb n 102))).

Definition is_bit (c : ascii) : bool :=
  let n := Ascii.nat_of_ascii c in
  orb (Nat.eqb n 48) (Nat.eqb n 49).

Definition is_wsp (c : ascii) : bool :=
  let n := Ascii.nat_of_ascii c in
  orb (Nat.eqb n 32) (Nat.eqb n 9).

Definition is_vchar (c : ascii) : bool :=
  let n := Ascii.nat_of_ascii c in
  andb (Nat.leb 33 n) (Nat.leb n 126).

Definition is_cr (c : ascii) : bool := Nat.eqb (Ascii.nat_of_ascii c) 13.
Definition is_lf (c : ascii) : bool := Nat.eqb (Ascii.nat_of_ascii c) 10.


Definition p_alpha : Parser string ascii := satisfy stringSeq is_alpha.
Definition p_digit : Parser string ascii := satisfy stringSeq is_digit.
Definition p_hexdig : Parser string ascii := satisfy stringSeq is_hexdig.
Definition p_bit : Parser string ascii := satisfy stringSeq is_bit.
Definition p_wsp : Parser string ascii := satisfy stringSeq is_wsp.
Definition p_vchar : Parser string ascii := satisfy stringSeq is_vchar.


Definition p_crlf : Parser string unit :=
  
  (_ <- char "013"%char ;; _ <- char "010"%char ;; ret tt)
  <|> (_ <- char "010"%char ;; ret tt)
  <|> (_ <- char "013"%char ;; ret tt).


Definition digit_val (c : ascii) : nat :=
  Ascii.nat_of_ascii c - 48.

Definition hexdig_val (c : ascii) : nat :=
  let n := Ascii.nat_of_ascii c in
  if Nat.leb n 57 then n - 48
  else if Nat.leb n 70 then n - 55
  else n - 87.
Definition bit_val (c : ascii) : nat :=
  Ascii.nat_of_ascii c - 48.

Fixpoint digits_to_nat (base : nat) (ds : list nat) : nat :=
  match ds with
  | [] ⇒ 0
  | d :: rest ⇒ d + base × digits_to_nat base rest
  end.

Definition digits_value (base : nat) (ds : list nat) : nat :=
  digits_to_nat base (rev ds).

Definition p_dec_number (input : string) : Parser string nat :=
  ds <- many1 (map digit_val p_digit) input ;;
  ret (digits_value 10 ds).

Definition p_hex_number (input : string) : Parser string nat :=
  ds <- many1 (map hexdig_val p_hexdig) input ;;
  ret (digits_value 16 ds).

Definition p_bin_number (input : string) : Parser string nat :=
  ds <- many1 (map bit_val p_bit) input ;;
  ret (digits_value 2 ds).


Definition p_num_val_body
  (base : NumBase) (dig_p : string → Parser string nat) (prefix : ascii) (input : string)
  : Parser string NumSpec :=
  _ <- char prefix ;;
  first <- dig_p input ;;
  
  (
    _ <- char "-"%char ;;
    last <- dig_p input ;;
    ret (NumRange base first last)
  )
  <|>
  (
    rest <- many (_ <- char "."%char ;; dig_p input) input ;;
    ret (NumConcat base (first :: rest))
  ).

Definition p_num_val (input : string) : Parser string NumSpec :=
  _ <- char "%"%char ;;
  (p_num_val_body BinBase p_bin_number "b"%char input)
  <|> (p_num_val_body DecBase p_dec_number "d"%char input)
  <|> (p_num_val_body HexBase p_hex_number "x"%char input).


Definition dquote : ascii := Ascii.ascii_of_nat 34.

Definition is_char_val_char (c : ascii) : bool :=
  let n := Ascii.nat_of_ascii c in
  orb (andb (Nat.leb 32 n) (Nat.leb n 33))
      (andb (Nat.leb 35 n) (Nat.leb n 126)).

Definition p_char_val (input : string) : Parser string string :=
  _ <- char dquote ;;
  cs <- many (satisfy stringSeq is_char_val_char) input ;;
  _ <- char dquote ;;
  ret (list_to_string cs).


Definition is_prose_char (c : ascii) : bool :=
  let n := Ascii.nat_of_ascii c in
  orb (andb (Nat.leb 32 n) (Nat.leb n 61))
      (andb (Nat.leb 63 n) (Nat.leb n 126)).

Definition p_prose_val (input : string) : Parser string string :=
  _ <- char "<"%char ;;
  cs <- many (satisfy stringSeq is_prose_char) input ;;
  _ <- char ">"%char ;;
  ret (list_to_string cs).


Definition is_rulename_cont (c : ascii) : bool :=
  orb (is_alpha c) (orb (is_digit c) (Nat.eqb (Ascii.nat_of_ascii c) 45)).

Definition p_rulename (input : string) : Parser string string :=
  first <- p_alpha ;;
  rest <- many (satisfy stringSeq is_rulename_cont) input ;;
  ret (list_to_string (first :: rest)).


Definition p_nat (input : string) : Parser string nat := p_dec_number input.

Definition p_repeat (input : string) : Parser string RepeatSpec :=
  
  (lo <- optional (p_nat input) ;;
   _ <- char "*"%char ;;
   hi <- optional (p_nat input) ;;
   ret (mkRepeat lo hi))
  <|>
  
  (n <- p_nat input ;;
   ret (mkRepeat (Some n) (Some n))).


Definition p_comment (input : string) : Parser string unit :=
  _ <- char ";"%char ;;
  _ <- many (satisfy stringSeq (fun c ⇒ orb (is_wsp c) (is_vchar c))) input ;;
  _ <- p_crlf ;;
  ret tt.

Definition p_c_nl (input : string) : Parser string unit :=
  p_comment input <|> p_crlf.

Definition p_c_wsp (input : string) : Parser string unit :=
  (_ <- p_wsp ;; ret tt)
  <|> (_ <- p_c_nl input ;; _ <- p_wsp ;; ret tt).

Definition p_owsp (input : string) : Parser string unit :=
  skipMany (p_c_wsp input) input.


Fixpoint p_alternation (input : string) : Parser string ABNFAlternation :=
  match input with
  | EmptyString ⇒ fail
  | String _ input' ⇒
    first <- p_concatenation input' ;;
    rest <- many (
      _ <- p_owsp input' ;;
      _ <- char "/"%char ;;
      _ <- p_owsp input' ;;
      p_concatenation input'
    ) input' ;;
    ret (Alt (first :: rest))
  end

with p_concatenation (input : string) : Parser string ABNFConcatenation :=
  match input with
  | EmptyString ⇒ fail
  | String _ input' ⇒
    first <- p_repetition input' ;;
    rest <- many (
      _ <- many1 (p_c_wsp input') input' ;;
      p_repetition input'
    ) input' ;;
    ret (Concat (first :: rest))
  end

with p_repetition (input : string) : Parser string ABNFRepetition :=
  match input with
  | EmptyString ⇒ fail
  | String _ input' ⇒
    rep <- optional (p_repeat input') ;;
    elem <- p_element input' ;;
    ret (Rep rep elem)
  end

with p_element (input : string) : Parser string ABNFElement :=
  match input with
  | EmptyString ⇒ fail
  | String _ input' ⇒
    (map ElemRuleName (p_rulename input'))
    <|> (p_group input')
    <|> (p_option input')
    <|> (map ElemCharVal (p_char_val input'))
    <|> (map ElemNumVal (p_num_val input'))
    <|> (map ElemProseVal (p_prose_val input'))
  end

with p_group (input : string) : Parser string ABNFElement :=
  match input with
  | EmptyString ⇒ fail
  | String _ input' ⇒
    _ <- char "("%char ;;
    _ <- p_owsp input' ;;
    body <- p_alternation input' ;;
    _ <- p_owsp input' ;;
    _ <- char ")"%char ;;
    ret (ElemGroup body)
  end

with p_option (input : string) : Parser string ABNFElement :=
  match input with
  | EmptyString ⇒ fail
  | String _ input' ⇒
    _ <- char "["%char ;;
    _ <- p_owsp input' ;;
    body <- p_alternation input' ;;
    _ <- p_owsp input' ;;
    _ <- char "]"%char ;;
    ret (ElemOption body)
  end.


Definition p_defined_as (input : string) : Parser string DefinedAs :=
  _ <- p_owsp input ;;
  da <- ((_ <- char "="%char ;; _ <- char "/"%char ;; ret DefIncremental)
         <|> (_ <- char "="%char ;; ret DefEqual)) ;;
  _ <- p_owsp input ;;
  ret da.


Definition p_rule (input : string) : Parser string ABNFRule :=
  name <- p_rulename input ;;
  da <- p_defined_as input ;;
  body <- p_alternation input ;;
  _ <- p_owsp input ;;
  _ <- (p_c_nl input <|> (_ <- eos stringSeq ;; ret tt)) ;;
  ret (mkRule name da body).


Definition p_rulelist (input : string) : Parser string ABNFRulelist :=
  many1 (
    (p_rule input)
    <|>
    (_ <- p_owsp input ;; _ <- p_c_nl input ;; ret (mkRule EmptyString DefEqual (Alt [])))
  ) input.


Definition parse_abnf (input : string) : option ABNFRulelist :=
  match p_rulelist input input with
  | Some (rules, _) ⇒ Some (List.filter (fun r ⇒ negb (String.eqb (rule_name r) "")) rules)
  | None ⇒ None
  end.

  run (p_num_val "%x41") "%x41" = Some (NumConcat HexBase [65]).

  run (p_char_val (String dquote (String "a"%char (String "b"%char (String "c"%char (String dquote EmptyString))))))
      (String dquote (String "a"%char (String "b"%char (String "c"%char (String dquote EmptyString)))))
  = Some "abc".
Proof. reflexivity. Qed.

  run (p_rulename "my-rule") "my-rule" = Some "my-rule".
Proof. reflexivity. Qed.

Definition quote_str (s : string) : string :=
  String dquote (s ++ String dquote EmptyString).

Definition test_alt_input := (quote_str "a" ++ " / " ++ quote_str "b")%string.

Example test_simple_alt :
  run (p_alternation test_alt_input) test_alt_input
  = Some (Alt [Concat [Rep None (ElemCharVal "a")];
               Concat [Rep None (ElemCharVal "b")]]).
Proof. reflexivity. Qed.

Example test_repetition :
  run (p_repetition "*DIGIT") "*DIGIT"
  = Some (Rep (Some (mkRepeat None None)) (ElemRuleName "DIGIT")).
Proof. reflexivity. Qed.

Example test_group :
  run (p_element "(foo / bar)") "(foo / bar)"
  = Some (ElemGroup (Alt [Concat [Rep None (ElemRuleName "foo")];
                          Concat [Rep None (ElemRuleName "bar")]])).
Proof. reflexivity. Qed.

  run (p_rule test_input) test_input
